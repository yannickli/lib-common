204bool _z_step_is_skipped(int unused, ...)
205{
206 const char *s;
207 va_list ap;
208 bool hit = false;
209 lstr_t wip = LSTR("wip");
210
211 IN_TEST;
212
213 va_start(ap, unused);
214 while ((s = va_arg(ap, const char *))) {
215 lstr_t t = LSTR(s);
216
217 if (qh_find(lstr, &_G.tag_or, &t) >= 0
218 && qh_find(lstr, &_G.tag_or, &wip) < 0)
219 {
220 return false;
221 }
222 }
223 va_end(ap);
224
225 va_start(ap, unused);
226 while (!hit && (s = va_arg(ap, const char *))) {
227 lstr_t t = LSTR(s);
228
229 if (qh_find(lstr, &_G.tag_skip, &t) >= 0) {
230 _z_step_skip("skipping tests flagged '%*pM'", LSTR_FMT_ARG(t));
231 hit = true;
232 break;
233 }
234 }
235 va_end(ap);
236
237 return hit;
238}

313static int uint64_nxdigits(uint64_t lh, uint64_t rh)
314{
315 /* computes max number of significant octets */
316 int nocts = DIV_ROUND_UP(bsr64(lh | rh | 1) + 1, 8);
317
318 /* returns 2 for 1-2 octets, 4 for 3-4, 8 for 5-8, and 16 else */
319 return 2 * (bsr32(nocts) + 1);
320}

322static void sb_add_z_val(sb_t *buf, z_val_t v, int ndigits, int nxdigits)
323{
324 switch (v.type) {
325 case Z_VAL_TYPE_I:
326 if (ndigits && nxdigits) {
327 sb_addf(buf, "%*jd / %#0*jx", ndigits, v.i, 2 + nxdigits, v.i);
328 } else {
329 sb_addf(buf, "%jd / %jx", v.i, v.i);

333 case Z_VAL_TYPE_U:
334 if (ndigits && nxdigits) {
335 sb_addf(buf, "%*ju / %#0*jx", ndigits, v.u, 2 + nxdigits, v.u);

337 sb_addf(buf, "%ju / %jx", v.u, v.u);

342 /* TODO Improve format. */
343 sb_addf(buf, "%f", v.d);

346 case Z_VAL_TYPE_B:
347 sb_addf(buf, "%s", v.b ? "true" : "false");

349
350 case Z_VAL_TYPE_C:
351 sb_addf(buf, "'%c' (%x)", v.c, v.c);
352 break;
353 }

384 case Z_VAL_TYPE_C:

419/** Skip at least n bytes but avoid stopping in the middle of an UTF-8
420 * character.
421 *
422 * \return The number of skipped bytes.
423 */
424static size_t ps_skip_utf8_aligned(pstream_t *ps, size_t n,
425 pstream_t *nullable skipped)
426{
427 const byte *start = ps->b;
428
429 while (!ps_done(ps) && (size_t)(ps->b - start) < n) {
430 if (ps_getuc(ps) < 0) {
431 ps_skip(ps, 1);
432 }
433 }
434 n = ps->b - start;
435
436 if (skipped) {
437 *skipped = ps_init(start, n);
438 }
439
440 return n;
441}

443static void sb_adduc_escaped(sb_t *sb, int c)
444{
445 int replacement = -1;
446
447 switch (c) {
448#define CASE(_to_escape, _replacement) \
449 case _to_escape: \
450 replacement = _replacement; \
451 break;
452
453 CASE('\a', 'a');
454 CASE('\b', 'b');
455 CASE('\t', 't');
456 CASE('\n', 'n');
457 CASE('\v', 'v');
458 CASE('\f', 'f');
459 CASE('\r', 'r');
460 CASE('\\', '\\');
461#undef CASE
462 }
463
464 if (replacement >= 0) {
465 sb_addc(sb, '\\');
466 sb_addc(sb, replacement);
467 } else {
468 sb_adduc(sb, c);
469 }
470}

528static int z_ps_get_char(pstream_t *ps, bool full_hex, bool *printable)
529{
530 int c;
531
532 if (full_hex) {
533 /* The character *might* be printable but we won't try. */
534 *printable = false;
535 } else {
536 c = ps_getuc(ps);
537 *printable = c >= 0;
538 }
539 if (!*printable) {
540 c = ps_getc(ps);
541 }
542
543 return c;

572static void z_sb_put_string(sb_t *sb, lstr_t s, int first_diff_pos,
573 bool use_hex, bool can_clip)
574{
575 const int skip_threshold = use_hex ? 20 : 60;
576 const int skip_context = use_hex ? 10 : 20;
577
578 /* Maximum number of bytes displayed. */
579 const int clip_threshold = use_hex ? 30 : 80;
580
581 size_t clipped = 0;
582
583 /* Skip the beginning of the string if the first diff is too far away from
584 * the start. */
585 if (first_diff_pos > skip_threshold) {
586 pstream_t ps = ps_initlstr(&s);
587 size_t skipped;
588
589 skipped = first_diff_pos - skip_context;
590 skipped = ps_skip_utf8_aligned(&ps, skipped, NULL);
591 first_diff_pos -= skipped;
592 sb_addf(sb, "-(%zd skipped bytes)- ", skipped);
593 s = LSTR_PS_V(&ps);
594 }
595
596 /* Clip the string if it's too long. */
597 if (s.len > first_diff_pos + clip_threshold) {
598 pstream_t ps = ps_initlstr(&s);
599 pstream_t kept;
600
601 ps_skip_utf8_aligned(&ps, first_diff_pos + clip_threshold, &kept);
602 s = LSTR_PS_V(&kept);
603 clipped = ps_len(&ps);
604 }
605
606 z_sb_add_lstr(sb, s, first_diff_pos, use_hex);
607
608 if (clipped) {
609 sb_addf(sb, " -(%zd skipped bytes)-", clipped);
610 }
611}

656static int z_diff(lstr_t left, lstr_t right, sb_t *diff)
657{
658 t_scope;
659 const char *lfile;
660 const char *rfile;
661 const char *sed_color = "";
662 const char *diff_file;
663 char *dtemp;
664 SB_1k(cmd);
665 int ret;
666
667 /* The diff command takes files as input so we have to write files.
668 * Put all of them in a single temporary directory.
669 * We do not mind letting the temporary directory exist after this call as
670 * the files it contains may be useful for debugging. */
671 dtemp = t_fmt("%*pMdiff-XXXXXX", LSTR_FMT_ARG(z_tmpdir_g));
672 RETHROW_PN(mkdtemp(dtemp));
673
674 /* Write the two files. */
675 lfile = t_fmt("%s/left", dtemp);
676 rfile = t_fmt("%s/right", dtemp);
677 RETHROW(xwrite_file(lfile, left.s, left.len));
678 RETHROW(xwrite_file(rfile, right.s, right.len));
679
680 if (_G.do_fancy_output) {
681 sed_color = " | sed '"
682 /* Removals: lines that start with '-'. */
683 "s/^-/\\x1b[" TERM_COLOR_PURPLE "m-/;"
684
685 /* Additions: lines that start with '+'. */
686 "s/^+/\\x1b[" TERM_COLOR_BRIGHTER(TERM_COLOR_CYAN) "m+/;"
687
688 /* Line/column number information. */
689 "s/^@/\\x1b[" TERM_COLOR_YELLOW "m@/;"
690
691 /* Reset color at EOL. */
692 "s/$/\\x1b[0m/"
693
694 "'";
695 }
696
697 /* Run diff with a system command and write it in a diff file. */
698 diff_file = t_fmt("%s/diff", dtemp);
699 sb_setf(&cmd, "/bin/bash -c "
700 "\"set -o pipefail; diff -urNw %s %s%s >%s\"",
701 lfile, rfile, sed_color, diff_file);
702 ret = system(cmd.data);
703 THROW_ERR_IF(WEXITSTATUS(ret) != 0 && WEXITSTATUS(ret) != 1);
704
705 /* Load the diff from the diff file. */
706 RETHROW(sb_read_file(diff, diff_file));
707
708 return 0;
709}

1003static bool _z_group_run(void)
1004{
1005 pid_t pid;
1006 int st;
1007
1008 _G.grp.n_total_digits = snprintf(NULL, 0, "%d", _G.grp.n_total);
1009 do_plan();
1010
1011 fflush(stdout);
1012 if (!_G.do_fork) {
1013 return true;
1014 }
1015
1016 pid = thr_job_fork();
1017 if (pid == 0) {
1018 signal(SIGTERM, SIG_DFL);
1019 signal(SIGQUIT, SIG_DFL);
1020 signal(SIGINT, SIG_DFL);
1021 return true;
1022 }
1023 if (pid < 0) {
1024 do_bail_out("cannot fork(): %m");
1025 return false;
1026 }
1027 waitpid(pid, &st, 0);
1028 if (WIFEXITED(st)) {
1029 if (WEXITSTATUS(st))
1030 _G.res = -1;
1031 return false;
1032 }
1033 if (WIFSIGNALED(st)) {
1034 do_bail_out("exit with signal %d %s", WTERMSIG(st),
1035 strsignal(WTERMSIG(st)));
1036 } else {
1037 do_bail_out("should not happen");
1038 }
1039 _G.res = -1;
1040 return false;
1041}

1077
1078 case Z_CHILD_DIE:
