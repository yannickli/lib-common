36/** Farch archives data structures.
37 *
38 * XXX: You should never use these strutures directly: they are only exposed
39 * for farchc and the files it generates.
40 *
41 * Farch is used to store compressed file contents as C obfuscated symbols.
42 *
43 * File contents are obfuscated by splitting them in small chunks which are
44 * xored with a randomly selected string (the string is a symbol which, in
45 * principle, already exists in the binary). Each chunk is represented as a
46 * farch_data_t structure.
47 *
48 * A farch archive is an array of farch_entry_t. Each of those represents a
49 * single file with its obfuscated filename, its contents as described above,
50 * and the size of the contents after and before compression. The obfuscation
51 * method for the filename is the same than for the contents.
52 */
53typedef lstr_t farch_data_t;
54
55typedef struct farch_entry_t {
56 lstr_t name; /* obfuscated filename, its length is the obfuscation key */
57 const lstr_t * nonnull chunks; /* obfuscated content chunks */
58 int compressed_size;
59 int size;
60 int nb_chunks;
61} farch_entry_t;

68/** Get the filename of this entry.
69 *
70 * This function can be used to iterate on names entries: entries are stored
71 * in an array terminating by an empty entry (name.len == 0).
72 *
73 * for (const farch_entry_t *entry = entries,
74 * char __buf[FARCH_MAX_FILENAME];
75 * name = farch_get_filename(entry, __buf);
76 * entry++)
77 *
78 * \param[in] file a farch entry.
79 * \param[in|out] name a buffer to write the name -- should contains
80 * FARCH_MAX_FILENAME bytes.
81 * \return NULL if there is no entry or if an error occurs, name otherwise.
82 */
83char * nullable farch_get_filename(const farch_entry_t * nonnull file,
