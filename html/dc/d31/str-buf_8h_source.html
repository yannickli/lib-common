28/* sb_t is a wrapper type for a reallocatable byte array. Its internal
29 * representation is accessible to client code but care must be exercised to
30 * preserve strbuf invariants and avoid dangling pointers when strbuf data is
31 * reallocated by strbuf functions.
32 *
33 * Here is a description of the strbuf fields:
34 *
35 * - <data> is a pointer to the active portion of the byte array. As <data>
36 * may be reallocated by strbuf operations, it is error prone to copy its
37 * value to a local variable for content parsing or other such
38 * manipulations. As <data> may point to non allocated storage, static or
39 * dynamic or automatic, returning its value is guaranteed to cause
40 * potential problems, current of future. If some strbuf data was skipped,
41 * <data> may no longer point to the beginning of the original or allocated
42 * array. <data> cannot be NULL. It is initialized as the address of a
43 * global 1 byte array, or a buffer with automatic storage.
44 *
45 * - <len> is the length in bytes of the strbuf contents. A strbuf invariant
46 * specifies that data[len] == '\0'. This invariant must be kept at all
47 * times, and implies that data storage extend at least one byte beyond
48 * <len>. <len> is always positive or zero. <len> is initialized to zero for
49 * an empty strbuf.
50 *
51 * - <size> is the number of bytes available for strbuf contents starting at
52 * <data>. The strbuf invariant implies that size > len.
53 *
54 * - <skip> is a count of byte skipped from the beginning of the original data
55 * buffer. It is used for efficient pruning of initial bytes.
56 *
57 * - <allocated> is a 1 bit boolean to indicate if strbuf data was allocated
58 * with p_new and must be freed. Actual allocated buffer starts at .data -
59 * .skip.
60 *
61 * strbuf invariants:
62 * - sb.data != NULL
63 * - sb.len >= 0
64 * - sb.size > sb.len
65 * - sb.data - sb.skip points to an array of at least sb.size + sb.skip bytes.
66 * - sb.data[sb.len] == '\0'
67 * - sb.data - sb.skip is a pointer handled by mp_new/mp_delete for the pool
68 * sb.mp.
69 *
70 * │←─ sb->size ──────────────────────→│
71 * │←─ sb->skip ─→│←─ sb->len ─→│ │←─ sb_avail() ─→│
72 * ┌──────────────┬─────────────┬────┬────────────────┐
73 * │ skip area │ data │'\0'│ available room │
74 * └──────────────┴─────────────┴────┴────────────────┘
75 * ↑ ↑
76 * sb->data sb_end()
77 */
78

80 char * nonnull data;
81 int len, size, skip;
82 mem_pool_t * nullable mp;
83#ifdef __cplusplus
84 inline sb_t();
85 inline sb_t(void * nonnull buf, int len, int size,
86 mem_pool_t * nullable mp);
87 inline ~sb_t();
88
89 private:
90 DISALLOW_COPY_AND_ASSIGN(sb_t);
91#endif
92} sb_t;

251static inline void sb_optimize(sb_t * nonnull sb, size_t extra)
252{
253 size_t size = sb->size + sb->skip;
254 size_t len = sb->len + 1;
255
256 if (unlikely(size > BUFSIZ && (len + extra) * 8 < size))
257 __sb_optimize(sb, len + extra);

260static inline char * nonnull sb_grow(sb_t * nonnull sb, int extra)
261{
262 if (sb->len + extra >= sb->size) {
263 __sb_grow(sb, extra);
264 } else {
265 sb_optimize(sb, extra);
266 }
267 return sb_end(sb);
268}
269
270static inline char * nonnull sb_growlen(sb_t * nonnull sb, int extra)
271{
272 if (sb->len + extra >= sb->size)
273 __sb_grow(sb, extra);
274 __sb_fixlen(sb, sb->len + extra);
275
276 /* workaround for a false positive of -Warray-bounds in gcc 8
277 * https://gcc.gnu.org/bugzilla/show_bug.cgi?id=89689
278 */
279 if (extra && sb->data == __sb_slop) {
280 __builtin_unreachable();
281 }
282
283 return sb_end(sb) - extra;
284}

331/* Prepare the string buffer for deletion of length "rm_len" followed by an
332 * insertion of length "insert_len" at position "pos". */

334__sb_splice(sb_t * nonnull sb, int pos, int rm_len, int insert_len);

336/** Deletes and inserts data at a given position in a string buffer.
337 *
338 * \param[in, out] sb The string buffer to modify.
339 * \param[in] pos Position used for deletion and insertion.
340 * \param[in] rm_len Number of bytes to remove.
341 * \param[in] data Data to insert.
342 * \param[in] dlen Length of the data to insert.
343 */
344static inline char * nonnull
345sb_splice(sb_t * nonnull sb, int pos, int rm_len, const void * nullable data,
346 int dlen)
347{
348 char *res;
349
350 assert (pos >= 0 && rm_len >= 0 && dlen >= 0);
351 assert ((unsigned)pos <= (unsigned)sb->len);
352 assert ((unsigned)pos + (unsigned)rm_len <= (unsigned)sb->len);
353
354#ifndef __cplusplus
355 if (__builtin_constant_p(dlen)) {
356 if (dlen == 0 || (__builtin_constant_p(rm_len) && rm_len >= dlen)) {
357 p_move2(sb->data, pos + dlen, pos + rm_len,
358 sb->len - pos - rm_len);
359 __sb_fixlen(sb, sb->len + dlen - rm_len);
360 return sb->data + pos;
361 }
362 }
363 if (__builtin_constant_p(rm_len) && rm_len == 0 && pos == sb->len) {
364 res = sb_growlen(sb, dlen);
365 } else {
366 res = __sb_splice(sb, pos, rm_len, dlen);
367 }
368#else
369 res = __sb_splice(sb, pos, rm_len, dlen);
370#endif
371 return data ? (char *)memcpy(res, data, dlen) : res;
372}

401static inline void sb_skip(sb_t * nonnull sb, int len)
402{
403 assert (len >= 0 && len <= sb->len);
404 if ((sb->len -= len)) {
405 sb->data += len;
406 sb->skip += len;
407 sb->size -= len;
408 } else {
409 sb_reset(sb);
410 }
411}
