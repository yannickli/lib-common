151typedef enum yaml_parse_flags_t {
152 /** Save presentation data when parsing.
153 *
154 * The presentation data can then be retrieved using
155 * t_yaml_data_get_presentation.
156 */
157 YAML_PARSE_GEN_PRES_DATA = 1 << 0,
158
159 /** Forbid use of variables. */
160 YAML_PARSE_FORBID_VARIABLES = 1 << 1,
161
162 /** Allow unset variables in the parsed AST.
163 *
164 * YAML documents can have variables, that can be set by other including
165 * YAML documents. Usually, when parsing a yaml file for consumption,
166 * we need to AST to be complete, and thus to have all variables be
167 * properly set.
168 *
169 * However, when manipulating the YAML files themselves, keeping the
170 * variables unbound is required. This flag activates this behavior,
171 * and do not reject documents with unbound variables.
172 *
173 * \warning Do not use this flag if the parsed YAML data is to be
174 * interpreted.
175 */
176 YAML_PARSE_ALLOW_UNBOUND_VARIABLES = 1 << 2,
177} yaml_parse_flags_t;

277typedef enum yaml_pack_flags_t {
278 /** Do not recreate subfiles when packing.
279 *
280 * When repacking, the behavior is different whether we are packing into
281 * a file or not:
282 * * When packing into a file, included subfiles are recreated, and
283 * "!include" datas are properly written.
284 * * Otherwise, the included data is inlined in the written output, and
285 * the "!include" datas are lost.
286 *
287 * Sometimes however, we want to keep the "!include" datas, but avoid
288 * recreating all the subfiles. This is the case for example when
289 * reformating a document: we want to print out the document reformatted,
290 * with its proper includes, but avoid recreating the subfiles.
291 *
292 * This flag activates this behavior: it writes the "!include" data, but
293 * do not attempt to recreate subfiles.
294 */
295 YAML_PACK_NO_SUBFILES = 1 << 0,
296
297 /** Allow unbound variables when repacking.
298 *
299 * This is supposed to be used when a document has been parsed with the
300 * YAML_PARSE_ALLOW_UNBOUND_VARIABLES flag. It ensures that if variables
301 * are not found when repacking, there are still properly repacked as
302 * unbound variables.
303 */
304 YAML_PACK_ALLOW_UNBOUND_VARIABLES = 1 << 1,
305} yaml_pack_flags_t;
