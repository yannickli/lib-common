37#if __BYTE_ORDER == __BIG_ENDIAN
38# define A(x) ((x) >> 24)
39# define A1(x) ((x) >> 56)
40# define B(x) (((x) >> 16) & 0xFF)
41# define C(x) (((x) >> 8) & 0xFF)
42# define D(x) ((x) & 0xFF)
43
44# define S8(x) ((x) << 8)
45# define S32(x) ((x) << 32)
46#else
47# define A(x) ((x) & 0xFF)
48# define A1(x) A(x)
49# define B(x) (((x) >> 8) & 0xFF)
50# define C(x) (((x) >> 16) & 0xFF)
51# define D(x) ((x) >> 24)
52
53# define S8(x) ((x) >> 8)
54# define S32(x) ((x) >> 32)
55#endif
