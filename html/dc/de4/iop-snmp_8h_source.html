37/** Register a revision into a qv of revisions.
38 *
39 * \param[out] _vec The qv_t(revi) of revisions to register the
40 * revision into.
41 * \param[in] _timestamp Timestamp of the new revision.
42 * \param[in] _description Description of the new revision.
43 */
44#define mib_register_revision(_vec, _timestamp, _description) \
45 ({ \
46 mib_revision_t _rev = { \
47 .timestamp = LSTR(_timestamp), \
48 .description = LSTR(_description), \
49 }; \
50 qv_append(_vec, _rev); \
51 })

53/** Generate a MIB into a sb_t.
54 *
55 * \param[out] sb The string buffer in which the MIB will be generated.
56 * \param[in] pkgs List of the different iop packages that will be
57 * added to the MIB.
58 * \param[in] revisions List of the different revisions the MIB has had,
59 * \note: the order of the different elements of the
60 * qv must follow the chronological order, from the
