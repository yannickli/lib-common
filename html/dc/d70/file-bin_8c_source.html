24/* File header */
25#define CURRENT_VERSION 1

27/* Should always be 16 characters long */
28#define SIG_0100 "IS_binary/v01.0\0"
29#define SIG SIG_0100

31typedef struct file_bin_header_t {
32 char version[16];
33 le32_t slot_size;
34} file_bin_header_t;
35
36#define HEADER_VERSION_SIZE 16
37#define HEADER_SIZE(file) \
38 (file->version == 0 ? 0 : ssizeof(file_bin_header_t))
39
40/* Slot header */
41typedef le32_t slot_hdr_t;
42
43#define SLOT_HDR_SIZE(file) \
44 ((file)->version == 0 ? 0 : ssizeof(slot_hdr_t))
45
46/* Record header */
47typedef le32_t rc_hdr_t;
48#define RC_HDR_SIZE ssizeof(rc_hdr_t)

59static uint32_t file_bin_remaining_space_in_slot(const file_bin_t *file)
60{
61 return file->slot_size - (file->cur % file->slot_size);
62}
63
64static off_t file_bin_get_prev_slot(const file_bin_t *file, off_t pos)
65{
66 return pos < file->slot_size ? HEADER_SIZE(file)
67 : pos - (pos % file->slot_size);
68}

70static bool is_at_slot_start(const file_bin_t *f)
71{
72 return f->cur % f->slot_size == 0 || f->cur == HEADER_SIZE(f);
73}

97static off_t file_bin_get_entry_end_off(const file_bin_t *f, uint32_t d_len)
98{
99 uint32_t len = d_len;
100 off_t res;
101 uint32_t nb_slots;
102 uint32_t remaining = file_bin_remaining_space_in_slot(f);
103
104 if (is_at_slot_start(f)) {
105 len += SLOT_HDR_SIZE(f);
106 }
107 res = f->cur + len;
108 if (len <= remaining) {
109 return res;
110 }
111 /* compute the number of extra slots needed to store the entry */
112 nb_slots = DIV_ROUND_UP(len - remaining,
113 f->slot_size - SLOT_HDR_SIZE(f));
114 res += nb_slots * SLOT_HDR_SIZE(f);
115
116 return res;
117}
118
119static off_t file_bin_get_next_entry_off(const file_bin_t *f, uint32_t d_len)
120{
121 off_t res = file_bin_get_entry_end_off(f, d_len);
122 uint32_t remaining = 0;
123
124 if (res % f->slot_size) {
125 remaining = f->slot_size - (res % f->slot_size);
126 }
127 /* else it means res is exactly at the end of a slot so remaining is 0 */
128
129 if (remaining < RC_HDR_SIZE) {
130 /* if there is not enough space in the slot to put the record header,
131 * go to the beginning of the next slot
132 */
133 res += remaining + SLOT_HDR_SIZE(f);
134 }
135 return res;
136}

141static int file_bin_parse_header(lstr_t path, void *data, size_t len,
142 uint16_t *version, uint32_t *slot_size)
143{
144 file_bin_header_t *header = data;
145
146 if (len < sizeof(file_bin_header_t)) {
147 logger_error(&_G.logger,
148 "not enough data in '%*pM' to parse header: %ju < %ju",
149 LSTR_FMT_ARG(path), len, sizeof(file_bin_header_t));
150 return -1;
151 }
152
153 if (strequal(header->version, SIG_0100)) {
154 *version = 1;
155 *slot_size = le_to_cpu32p(&header->slot_size);
156 } else {
157 /* Unknown header. File is probably in version 0 */
158 *version = 0;
159 *slot_size = FILE_BIN_DEFAULT_SLOT_SIZE;
160 }
161
162 logger_trace(&_G.logger, 3, "parsed file header for '%*pM': version = %u,"
163 " slot size = %u", LSTR_FMT_ARG(path), *version, *slot_size);
164
165 return 0;
166}

168int file_bin_refresh(file_bin_t *file)
169{
170 struct stat st;
171 void *new_map = NULL;
172 bool parse_header = false;
173
174 assert (file->read_mode);
175
176 if (fstat(fileno(file->f), &st) < 0) {
177 return logger_error(&_G.logger, "cannot stat file '%*pM': %m",
178 LSTR_FMT_ARG(file->path));
179 }
180
181 if (file->length == st.st_size) {
182 return 0;
183 }
184
185 assert (file->map || file->length == 0);
186#ifdef __linux__
187 if (file->map) {
188 new_map = mremap(file->map, file->length, st.st_size, MREMAP_MAYMOVE);
189 }
190#else
191 if (file->map) {
192 munmap(file->map, file->length);
193 file->map = NULL;
194 new_map = NULL;
195 }
196#endif
197
198 if (!file->map) {
199 parse_header = true;
200 new_map = mmap(NULL, st.st_size, PROT_READ,
201 MAP_SHARED, fileno(file->f), 0);
202 }
203
204 if (new_map == MAP_FAILED) {
205 logger_error(&_G.logger, "cannot %smap file '%*pM': %m",
206 file->map ? "re" : "", LSTR_FMT_ARG(file->path));
207 return -1;
208 }
209
210 if (parse_header) {
211 RETHROW(file_bin_parse_header(file->path, new_map, st.st_size,
212 &file->version, &file->slot_size));
213 }
214
215 file->map = new_map;
216 file->length = st.st_size;
217
218 return 0;
219}

233static int file_bin_get_cpu32(file_bin_t *file, uint32_t *res)
234{
235 pstream_t ps = ps_init(file->map + file->cur, sizeof(le32_t));
236 le32_t le32;
237
238 THROW_ERR_IF(!file_bin_has(file, sizeof(le32_t)));
239
240 if (ps_get_le32(&ps, &le32) < 0) {
241 return logger_error(&_G.logger, "cannot read le32 at offset '%ju' "
242 "for file '%*pM'", file->cur,
243 LSTR_FMT_ARG(file->path));
244 }
245
246 *res = le32;
247 file->cur += sizeof(le32_t);
248
249 return 0;
250}

259static int _file_bin_get_next_record(file_bin_t *file, lstr_t *rec)
260{
261 off_t prev_off = file->cur;
262 off_t rec_end_off;
263 uint32_t sz;
264 uint32_t check_slot_hdr;
265 bool is_spanning = false;
266
267 if (file_bin_is_finished(file)) {
268 return -1;
269 }
270
271 if (file->version > 0) {
272 file->cur = MAX(file->cur, HEADER_SIZE(file));
273 }
274
275 sb_reset(&file->record_buf);
276
277 sz = file_bin_remaining_space_in_slot(file);
278 if (sz < RC_HDR_SIZE) {
279 if (_file_bin_skip(file, sz) < 0) {
280 return -1;
281 }
282 }
283
284 if (file->version > 0) {
285 while (is_at_slot_start(file)) {
286 if (file_bin_get_cpu32(file, &sz) < 0
287 || _file_bin_skip(file, sz) < 0)
288 {
289 goto error;
290 }
291 }
292 }
293
294 if (file_bin_is_finished(file)) {
295 return -1;
296 }
297 if (file_bin_get_cpu32(file, &sz) < 0) {
298 goto error;
299 }
300
301 rec_end_off = file_bin_get_entry_end_off(file, sz);
302 if (rec_end_off > file->length) {
303 slot_hdr_t tmp_size;
304
305 /* There is not enough data in the file to read the record. This could
306 * happen for two reasons:
307 * - the record header is corrupted and the length is non-sense;
308 * in that case, we want to skip this corrupted slot.
309 * - the record is being written and we do not have enough data yet;
310 * in that case, we want to stay here.
311 *
312 * In order to guess in which case we are, read the next slot header
313 * (if available) in which there is also the information about the
314 * offset of the end of the record. If the two offsets mismatch, we
315 * are in the first case.
316 */
317 if (_file_bin_skip(file, file_bin_remaining_space_in_slot(file)) < 0
318 || file_bin_get_cpu32(file, &tmp_size) < 0
319 || rec_end_off == file->cur + tmp_size)
320 {
321 file->cur = prev_off;
322 return -1;
323 }
324 logger_error(&_G.logger, "corrupted record length in file '%*pM' at "
325 "pos %jd", LSTR_FMT_ARG(file->path), prev_off);
326 file->cur -= sizeof(slot_hdr_t);
327 *rec = LSTR_NULL_V;
328 return 0;
329 }
330
331 if (is_at_slot_start(file)
332 && _file_bin_skip(file, SLOT_HDR_SIZE(file)) < 0)
333 {
334 goto error;
335 }
336
337 if (sz == 0) {
338 if (file->version == 0) {
339 /* In V0, the end of the slots are filled with 0s. Go to error so
340 * that we jump to next slot. */
341 goto error;
342 } else {
343 *rec = LSTR_EMPTY_V;
344 return 0;
345 }
346 }
347
348 check_slot_hdr = file_bin_get_next_entry_off(file, sz);
349
350 while (!file_bin_is_finished(file)) {
351 uint32_t remaining = file_bin_remaining_space_in_slot(file);
352 slot_hdr_t tmp_size;
353
354 remaining = MIN(remaining, file->length - file->cur);
355
356 if (sz <= remaining) {
357 /* The last part of this record is in this slot. */
358 lstr_t res = LSTR_DATA(file->map + file->cur, sz);
359
360 file->cur += sz;
361
362 if (!is_spanning) {
363 *rec = res;
364 } else {
365 sb_add_lstr(&file->record_buf, res);
366 *rec = LSTR_SB_V(&file->record_buf);
367 }
368 return 0;
369 }
370
371 assert (file->version > 0);
372
373 /* Record spans on multiple slots. */
374 if (!is_spanning) {
375 is_spanning = true;
376 sb_grow(&file->record_buf, sz);
377 }
378
379 sb_add(&file->record_buf, file->map + file->cur, remaining);
380 sz -= remaining;
381 file->cur += remaining;
382
383 if (!is_at_slot_start(file)) {
384 logger_error(&_G.logger, "corrupted file '%*pM', a slot start "
385 "was expected at pos %jd",
386 LSTR_FMT_ARG(file->path), file->cur);
387 assert (false);
388 goto error;
389 }
390
391 /* Consuming slot header */
392 if (file_bin_get_cpu32(file, &tmp_size) < 0) {
393 goto error;
394 }
395
396 if (tmp_size != check_slot_hdr - file->cur) {
397 logger_error(&_G.logger, "buggy slot header in file '%*pM', "
398 "expected %jd, got %u, jumping to next slot",
399 LSTR_FMT_ARG(file->path), check_slot_hdr - file->cur,
400 tmp_size);
401 goto error;
402 }
403 }
404
405 assert (false);
406
407 error:
408 /* An error occured, try to jump to the next slot. */
409 if (_file_bin_skip(file, file_bin_remaining_space_in_slot(file)) < 0) {
410 /* There is not enough data in the file, rollback position. */
411 file->cur = prev_off;
412 return -1;
413 }
414 *rec = LSTR_NULL_V;
415 return 0;
416}

418lstr_t file_bin_get_next_record(file_bin_t *file)
419{
420 int res;
421 lstr_t rec = LSTR_NULL_V;
422
423 assert (file->read_mode);
424
425 do {
426 res = _file_bin_get_next_record(file, &rec);
427 } while (res >= 0 && !rec.s);
428
429 return rec;
430}

432int t_file_bin_get_last_records(file_bin_t *file, int count, qv_t(lstr) *out)
433{
434 off_t save_cur = file->cur;
435 off_t slot_off = file->length;
436 off_t prev_slot;
437 qv_t(lstr) tmp;
438
439 assert (file->read_mode);
440
441 t_qv_init(&tmp, count);
442
443 do {
444 prev_slot = slot_off;
445 file->cur = slot_off = file_bin_get_prev_slot(file, prev_slot - 1);
446
447 while (file->cur <= prev_slot - RC_HDR_SIZE
448 && !file_bin_is_finished(file))
449 {
450 lstr_t res = file_bin_get_next_record(file);
451
452 if (!res.s) {
453 break;
454 }
455 qv_append(&tmp, t_lstr_dup(res));
456 }
457
458 qv_grow(out, tmp.len);
459 for (int pos = tmp.len; pos-- > 0 && count > 0; count--) {
460 qv_append(out, tmp.tab[pos]);
461 }
462
463 qv_clear(&tmp);
464 } while (count > 0 && slot_off > HEADER_SIZE(file));
465
466 file->cur = save_cur;
467
468 return 0;
469}

471file_bin_t *file_bin_open(lstr_t path)
472{
473 file_bin_t *res;
474 struct stat st;
475 void *mapping = NULL;
476 uint16_t version = 0;
477 uint32_t slot_size = 0;
478 lstr_t r_path = lstr_dup(path);
479 FILE *file = fopen(r_path.s, "r");
480
481 if (!file) {
482 logger_error(&_G.logger, "cannot open file '%*pM: %m",
483 LSTR_FMT_ARG(path));
484 goto error;
485 }
486
487 if (fstat(fileno(file), &st) < 0) {
488 logger_error(&_G.logger, "cannot get stat on file '%*pM': %m",
489 LSTR_FMT_ARG(path));
490 goto error;
491 }
492
493 if (st.st_size < 0) {
494 logger_error(&_G.logger, "invalid size of binary file '%*pM'",
495 LSTR_FMT_ARG(path));
496 goto error;
497 }
498
499 if (st.st_size > 0) {
500 mapping = mmap(NULL, st.st_size, PROT_READ,
501 MAP_SHARED, fileno(file), 0);
502 if (mapping == MAP_FAILED) {
503 logger_error(&_G.logger, "cannot map file '%*pM': %m",
504 LSTR_FMT_ARG(path));
505 goto error;
506 }
507
508 if (file_bin_parse_header(path, mapping, st.st_size,
509 &version, &slot_size) < 0)
510 {
511 goto error;
512 }
513 }
514
515 res = file_bin_new();
516 res->read_mode = true;
517 res->f = file;
518 res->path = r_path;
519 res->length = st.st_size;
520 res->map = mapping;
521 res->version = version;
522 res->slot_size = slot_size;
523 res->cur = HEADER_SIZE(res);
524
525 return res;
526
527 error:
528 p_fclose(&file);
529 lstr_wipe(&r_path);
530 return NULL;
531}

575static int file_bin_pad(file_bin_t *file, off_t new_pos)
576{
577 off_t real_cur = ftell(file->f);
578
579 if (real_cur < 0) {
580 return logger_error(&_G.logger, "cannot use ftell on file '%*pM': %m",
581 LSTR_FMT_ARG(file->path));
582 }
583
584 THROW_ERR_UNLESS(expect(real_cur <= new_pos));
585
586 if (real_cur == new_pos) {
587 return 0;
588 }
589
590 RETHROW(file_bin_truncate(file, new_pos));
591
592 real_cur = RETHROW(file_bin_tell(file));
593
594 THROW_ERR_UNLESS(expect(real_cur == new_pos));
595
596 return 0;
597}

600static int _write_file_bin(file_bin_t *file, const void *data, uint32_t len)
601{
602 uint32_t res;
603
604 RETHROW(file_bin_pad(file, file->cur));
605

607
608 if (res < len) {

610 return logger_error(&_G.logger, "cannot write in file '%*pM': %m",

612 }

630static int file_bin_write_slot_header(file_bin_t *file, off_t next_entry)
631{
632 off_t to_write = next_entry - (file->cur + SLOT_HDR_SIZE(file));
633 slot_hdr_t towrite_le32 = cpu_to_le32((uint32_t)to_write);

641/* Write data in file and write slot headers when necessary */
642static int file_bin_write_data(file_bin_t *f, const void *data, uint32_t len,
643 off_t next_entry, bool r_start)
644{
645 while (len > 0) {
646 uint32_t w_size;
647
648 if (is_at_slot_start(f)) {
649 if (r_start) {
650 file_bin_write_slot_header(f, f->cur + SLOT_HDR_SIZE(f));
651 } else {
652 file_bin_write_slot_header(f, next_entry);
653 }
654 }
655
656 w_size = MIN(file_bin_remaining_space_in_slot(f), len);
657
658 len -= w_size;
659 RETHROW(_write_file_bin(f, data, w_size));
660
661 data = (byte *)data + w_size;
662 r_start = false;
663 }
664
665 return 0;
666}

668int file_bin_put_record(file_bin_t *file, const void *data, uint32_t len)
669{
670 rc_hdr_t rec_len_le32 = cpu_to_le32(len);
671 uint32_t total_size = len + RC_HDR_SIZE;
672 off_t next_entry;
673 uint32_t remaining;
674
675 if (file->cur == 0 && file->version > 0) {
676 /* File beginning */
677 RETHROW(file_bin_write_header(file));
678 }
679
680 remaining = file_bin_remaining_space_in_slot(file);
681
682 if (remaining < RC_HDR_SIZE
683 || (file->version == 0 && remaining < total_size))
684 {
685 file->cur += remaining;
686 }
687 next_entry = file_bin_get_next_entry_off(file, total_size);
688
689 RETHROW(file_bin_write_data(file, &rec_len_le32, RC_HDR_SIZE,
690 next_entry, true));
691 RETHROW(file_bin_write_data(file, data, len, next_entry, false));
692
693 return 0;
694}

696file_bin_t *file_bin_create(lstr_t path, uint32_t slot_size, bool trunc)
697{
698 file_bin_t *res;
699 FILE *file;
700 lstr_t r_path = lstr_dup(path);
701 uint32_t min_slot_size;
702
703 file = fopen(r_path.s, trunc ? "w" : "a+");
704 if (!file) {
705 logger_error(&_G.logger, "cannot open file '%*pM': %m",
706 LSTR_FMT_ARG(path));
707 lstr_wipe(&r_path);
708 return NULL;
709 }
710
711#define GOTO_ERROR_IF_FAIL(expr) if (unlikely((expr) < 0)) { goto error; }
712
713 slot_size = slot_size > 0 ? slot_size : FILE_BIN_DEFAULT_SLOT_SIZE;
714
715 res = file_bin_new();
716 res->f = file;
717 res->path = r_path;
718
719 GOTO_ERROR_IF_FAIL(file_bin_seek(res, 0, SEEK_END));
720 GOTO_ERROR_IF_FAIL((res->cur = file_bin_tell(res)));
721
722 res->version = CURRENT_VERSION;
723 res->slot_size = slot_size;
724
725 if (res->cur > 0) {
726 /* Appending an already existing file */
727 byte buf[20];
728
729 if (res->cur < countof(buf)) {
730 res->slot_size = FILE_BIN_DEFAULT_SLOT_SIZE;
731 res->version = 0;
732 return res;
733 }
734
735 rewind(file);
736 if (fread(buf, 1, countof(buf), file) < countof(buf)) {
737 logger_error(&_G.logger, "cannot read binary file header "
738 "for file '%*pM': %m", LSTR_FMT_ARG(r_path));
739 goto error;
740 }
741
742 GOTO_ERROR_IF_FAIL(file_bin_seek(res, 0, SEEK_END));
743 GOTO_ERROR_IF_FAIL(file_bin_parse_header(r_path, buf, countof(buf),
744 &res->version,
745 &res->slot_size));
746 return res;
747 }
748
749 min_slot_size = HEADER_SIZE(res) + SLOT_HDR_SIZE(res) + RC_HDR_SIZE;
750
751 if (unlikely(slot_size < min_slot_size)) {
752 logger_error(&_G.logger, "slot size should be higher than %u, got "
753 " %u for file '%*pM'", min_slot_size, slot_size,
754 LSTR_FMT_ARG(r_path));
755 goto error;
756 }
757
758 return res;
759
760#undef GOTO_ERROR_IF_FAIL
761
762 error:
763 IGNORE(file_bin_close(&res));
764 return NULL;
765}

788 }
