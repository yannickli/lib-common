266static bool job_run(thr_job_t * nonnull job, thr_syn_t *syn)
267{
268#ifdef __has_thr_acc
269 unsigned long start = hardclock();
270
271 self_g->acc.jobs_run++;
272#endif
273
274 if ((uintptr_t)job & 3) {
275 block_t blk = (block_t)((uintptr_t)job & ~(uintptr_t)3);
276
277 blk();
278 if ((uintptr_t)job & 1)
279 Block_release(blk);
280 } else {
281 (*job->run)(job, syn);
282 }
283
284#ifdef __has_thr_acc
285 self_g->acc.time += (hardclock() - start);
286#endif
287 if (syn)
288 thr_syn__job_done(syn);
289 return true;
290}

292void thr_syn_schedule(thr_syn_t *syn, thr_job_t *job)
293{
294 unsigned bot, top;
295 struct deque_entry *e;
296
297 if (syn)
298 thr_syn__job_prepare(syn);
299
300 /* Read the current limits of the queue. Since 'bot' cannot be changed by
301 * another thread and 'top' can only be moved in the direction of
302 * consuming threads, 'bot' is effectively the position where the job
303 * should be inserted and 'bot - top' gives an upper bound to the number of
304 * job in the queue (and is the actual number if no other thread try to
305 * steal a job to that one concurrently to the insertion).
306 */
307 bot = atomic_load(&self_g->bot);
308 top = atomic_load(&self_g->top);
309
310 /* Looks like there may be too many jobs in the queue of that thread, run
311 * the new one immediately.
312 */
313 if (unlikely((int)(bot - top) >= THR_JOB_MAX)) {
314#ifdef __has_thr_acc
315 self_g->acc.jobs_local++;
316#endif
317 job_run(job, syn);
318 return;
319 }
320
321 e = &self_g->q[bot % THR_JOB_MAX];
322
323 /* Looks like we are inserting the job in an empty queue and that the
324 * current object is still used by another thread. Run it locally.
325 */
326 if (atomic_load_explicit(&e->job, memory_order_acquire) != NULL) {
327#ifdef __has_thr_acc
328 self_g->acc.jobs_local++;
329#endif
330 job_run(job, syn);
331 return;
332 }
333
334 /* Add the job in the queue and update bottom. Since other threads can
335 * only consume jobs from the queue (increment top), we can safely add the
336 * new job at q[bot] and then increment bot
337 */
338 e->syn = syn;
339 atomic_store_explicit(&e->job, job, memory_order_release);
340 atomic_store(&self_g->bot, bot + 1);
341
342#ifdef __has_thr_acc
343 self_g->acc.jobs_queued++;
344#endif
345
346
347 thr_ec_signal(&_G.ec);
348}

380/** Run the 'bottom' job of the queue of the current thread.
381 *
382 * @return true if a job has been run, false if the queue is empty.
383 */
384static bool thr_job_dequeue(void)
385{
386 unsigned top, bot;
387
388 /* Read the bottom job and update the mark to mark that job as consumed.
389 * The remaining of the function will ensure we were effectively the first
390 * to reclaim the ownership of that job.
391 */
392 bot = atomic_load_explicit(&self_g->bot, memory_order_relaxed) - 1;
393 atomic_store_explicit(&self_g->bot, bot, memory_order_seq_cst);
394 atomic_thread_fence(memory_order_acq_rel);
395
396 /* Read the top line. The memory barrier ensure that the read is effectively
397 * done after updating bot and thus that other thread may have seen the
398 * update of bot before touching top.
399 */
400 top = atomic_load_explicit(&self_g->top, memory_order_relaxed);
401
402 /* There are remaining jobs after moving the bottom line. Top has been
403 * read after bot, and since bot is refetched before each individual
404 * trysteal, we're sure 'top' cannot have reached 'bot', and thus
405 * we are the owner of the job we fetched, run it.
406 */
407 if (likely((int)(bot - top) > 0)) {
408 return thr_run_deque_entry(&self_g->q[bot % THR_JOB_MAX]);
409 }
410
411 /* 'bot' and 'top' are equal, that mean that either we're consuming the
412 * last job of the queue or another thread did it. To know if the job
413 * is for us, check that top hasn't been moved. If it has been moved, the
414 * CAS will fail and the job is not for us, if it hasn't been moved, then
415 * move it to ensure nobody else will do it, then, move back 'bot' and
416 * 'top' to the same point since the queue is empty.
417 */
418 if (likely(bot == top)) {
419 if (likely(thr_consume_top(self_g, top, 1))) {
420 atomic_store_explicit(&self_g->bot, top + 1, memory_order_relaxed);
421 return thr_run_deque_entry(&self_g->q[bot % THR_JOB_MAX]);
422 } else {
423#ifdef __has_thr_acc
424 self_g->acc.jobs_failed_dequeues++;
425#endif
426 }
427 }
428
429 /* 'top' has been moved to the previous value of bot, this means the job
430 * has already been run by another thread. Since top > bot, we're
431 * sure nobody else will try to read another job and that
432 * top == bot + 1 == previous value of bot. Thus, we have to restore 'bot'
433 * to its previous value since we didn't get ownership of the job.
434 */
435 atomic_store_explicit(&self_g->bot, bot + 1, memory_order_relaxed);
436 return false;
437}

439/** Try to steal a job from the queue of another thread.
440 *
441 * @param ti The thread info structure of another thread.
442 * @return 1 if a job has been stolen, 0 if the thread's queue is empty,
443 * -1 if the attempt failed (a retry may be needed)
444 */
445static int thr_job_try_steal(thr_info_t *ti, int depth)
446{
447 unsigned top, bot;
448
449 /* Read the limits of the queue of the thread and fetch the top 'job' of
450 * the queue.
451 */
452 top = atomic_load_explicit(&ti->top, memory_order_relaxed);
453 bot = atomic_load_explicit(&ti->bot, memory_order_relaxed);
454
455 /* If the queue does not seem to be empty, then we know we own the job if
456 * and only if we can CAS top. This works because a concurrent
457 * dequeue/steal will always CAS top when consuming the last job of the
458 * queue. The tricky part is the concurrency between dequeue and steal
459 * solved by the fact we'll always move 'bot' and 'top' in dequeue when
460 * emptying the queue.
461 */
462 if ((int)(bot - top) > 0) {
463 if (likely(thr_consume_top(ti, top, 1))) {
464#ifdef __has_thr_acc
465 self_g->acc.jobs_stealed += 1;
466 self_g->acc.jobs_steals++;
467
468#endif
469 return thr_run_deque_entry(&ti->q[top % THR_JOB_MAX]);
470 } else {
471#ifdef __has_thr_acc
472 self_g->acc.jobs_failed_steals++;
473#endif
474 return -1;
475 }
476 }
477 return 0;
478}

584static void thr_queue_drain(thr_queue_t *q)
585{
586 bool wipe = false;
587 mpsc_it_t it;
588 ssize_t id;
589
590 if (unlikely(mpsc_queue_looks_empty(&q->q))) {
591 /* The queue should not be marked empty *and* scheduled. the only
592 * queue that should go through this slowpath is the main queue
593 * since it's not really scheduled like the other ones
594 */
595 assert (q == thr_queue_main_g);
596 return;
597 }
598
599 id = thr_id();
600 atomic_store(&q->running_on, id);
601 mpsc_queue_drain_start(&it, &q->q);
602 do {
603 mpsc_node_t *m = mpsc_queue_drain_fast(&it, &thr_queue_run_node,
604 (data_t){ .ptr = NULL });
605 thr_qnode_t *n = thr_qnode_of(m);
606
607 job_run(n->job, n->syn);
608 wipe = (n->job == &q->destroy);
609 } while (!mpsc_queue_drain_end(&it, &thr_qnode_destroy));
610 atomic_compare_exchange_strong(&q->running_on, &id,
611 THR_QUEUE_NOT_RUNNING);
612
613 if (wipe) {
614 thr_queue_delete(&q);
615 }
616}

680 thr_qnode_t *n;
681 thr_syn_t syn;
682
683 if (thr_is_on_queue(q)) {
684 /* We're already on the queue, so don't pass the queue again */
685 job_run(job, NULL);
686 return;
687 }
688
689 thr_syn_init(&syn);
690 thr_syn__job_prepare(&syn);
691 n = thr_qnode_create();
692 n->job = job;
693 n->syn = &syn;
694 /*
695 * if mpsc_queue_push returns 1, then we're alone !
696 */
697 if (mpsc_queue_push(&q->q, &n->qnode)) {
698 if (q == thr_queue_main_g && thr_id() != 0) {
699 thr_wakeup_thr0();
700 } else {
701 thr_queue_drain(q);
702 thr_syn_wipe(&syn);
703 return;
704 }
705 }
706 thr_syn_wait(&syn);
707 thr_syn_wipe(&syn);
708}

757static void thr_info_cleanup(void *unused)
758{
759 mpsc_node_t *m;
760

762 do {

764 thr_queue_drain(thr_queue_main_g);

766 } while (thr_job_dequeue());
767 }

769 thr_queue_wipe(thr_queue_main_g);

775
776 self_g->ncache.qnode.next = n->qnode.next;

778 }

780 atomic_store(&self_g->alive, false);

783static int thr_sleep_until(thr_evc_t *ec, int timeout, bool (^cond)(void))
784{
785 uint64_t key = thr_ec_get(ec);
786 struct timeval start_tv;
787
788 lp_gettv(&start_tv);
789
790 while (!cond()) {
791 thr_ec_timedwait(ec, key, timeout / 10);
792
793 if (!cond()) {
794 struct timeval end_tv;
795
796 lp_gettv(&end_tv);
797 if (timeval_diffmsec(&end_tv, &start_tv) >= timeout) {
798 return -1;
799 }
800 }
801 key = thr_ec_get(ec);
802 }
803 return 0;
804}

806static void *thr_job_main(void *arg)
807{
808 sigset_t set;
809 bool (^cond)(void) = ^bool (void) {
810 /* Exit immediately if we are active or we are stopping */
811 return atomic_load(&_G.target_threads_count) > (size_t)self_g->id
812 || unlikely(atomic_load(&_G.stopping));
813 };
814
815 sigemptyset(&set);
816 sigaddset(&set, SIGABRT);
817 sigaddset(&set, SIGILL);
818 sigaddset(&set, SIGFPE);
819 sigaddset(&set, SIGSEGV);
820 sigaddset(&set, SIGBUS);
821#if defined(__linux__)
822 sigaddset(&set, SIGSTKFLT);
823#endif
824 pthread_sigmask(SIG_UNBLOCK, &set, NULL);
825
826 thr_info_init(arg);
827 pthread_cleanup_push(thr_info_cleanup, NULL);
828
829 /* Ensure there is no open frame on the t_stack. */
830 assert (mem_stack_pool_is_at_top(&t_pool_g));
831
832 /* explanation of thread states :
833 * thread_id < target_thread_count : active threads
834 * thread_id == target_thread_count : this thread is kept on disposal
835 * thread_id > target_thread_count : these threads were created but are
836 * not used anymore, keep them for 5 seconds just in case.
837 */
838 while (likely(!atomic_load(&_G.stopping))) {
839 int res;
840 uint64_t key = 0;
841
842 /* quarantine and obsolete threads loop */
843 if (!cond()) {
844
845 /* Eventually reset the thread-local t_pool. */
846 mem_stack_pool_try_reset(&t_pool_g);
847
848 /* We may have been woken up by the addition of a new job in the
849 * queue of a thread, so we must make sure another thread will
850 * take over so there is no deadlock.
851 */
852 thr_ec_signal(&_G.ec);
853 if (thr_sleep_until(&_G.threads_count_evc, 5000, cond) < 0) {
854 /* obsolete threads will be shut down after 5s */
855 if (atomic_load(&_G.target_threads_count) < (size_t)self_g->id) {
856 self_g->dequeue_all = false;
857 break;
858 }
859 /* quarantine thread will sleep again */
860 continue;
861 }
862 }
863
864 /* Eventually reset the thread-local t_pool. */
865 mem_stack_pool_try_reset(&t_pool_g);
866
867 while (likely(thr_job_dequeue())) {
868 continue;
869 }
870 if (thr_job_steal() > 0) {
871 continue;
872 }
873
874 do {
875 sched_yield();
876 key = thr_ec_get(&_G.ec);
877#ifdef __has_thr_acc
878 self_g->acc.ec_gets++;
879#endif
880 } while ((res = thr_job_steal()) < 0);
881 if (res == 0 && !atomic_load(&_G.stopping)) {
882#ifdef __has_thr_acc
883 unsigned long start = hardclock();
884
885 self_g->acc.ec_waits++;
886#endif
887 thr_ec_wait(&_G.ec, key);
888
889#ifdef __has_thr_acc
890 self_g->acc.ec_wait_time += (hardclock() - start);
891#endif
892 }
893 }
894 pthread_cleanup_pop(1);
895 return NULL;
896}

901static void thr_fork_threads(void)
902{
903 pthread_attr_t attr;
904 uint64_t target_threads_gen = atomic_load(&_G.threads_count_gen);
905
906 pthread_attr_init(&attr);
907 pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_DETACHED);
908
909 while (target_threads_gen != atomic_load(&_G.threads_gen)
910 && !atomic_load(&_G.stopping))
911 {
912 sigset_t fillset;
913 sigset_t old;
914 pthread_t tid;
915 atomic_thr_info_t *last = &_G.threads;
916 uint64_t target_count = atomic_load(&_G.target_threads_count) + 1;
917 uint64_t expected_key;
918 uint64_t key;
919
920 sigfillset(&fillset);
921 pthread_sigmask(SIG_SETMASK, &fillset, &old);
922
923 expected_key = thr_ec_get(&_G.start_bar_main);
924
925 while (atomic_load(last)) {
926 thr_info_t *info = atomic_load(last);
927
928 if ((uint64_t)info->id < target_count && !atomic_load(&info->alive)) {
929 assert (info->id > 0);
930
931 if (thr_create(&tid, &attr, thr_job_main, info)) {
932 e_fatal("unable to create new thread: %m");
933 }
934 expected_key++;
935 }
936 last = &info->next;
937 }
938
939 while (atomic_load(&_G.threads_count) < target_count) {
940 thr_info_t *info = p_new(thr_info_t, 1);
941
942 info->id = atomic_fetch_add(&_G.threads_count, 1);
943 atomic_store(last, info);
944 last = &info->next;
945
946 if (info->id == 0) {
947 thr_info_init(info);
948 } else {
949 if (thr_create(&tid, &attr, thr_job_main, info)) {
950 e_fatal("unable to create new thread: %m");
951 }
952 }
953 expected_key++;
954 }
955
956 while ((key = thr_ec_get(&_G.start_bar_main)) != expected_key) {
957 thr_ec_wait(&_G.start_bar_main, key);
958 }
959
960 thr_ec_broadcast(&_G.start_bar_thr);
961 pthread_sigmask(SIG_SETMASK, &old, NULL);
962
963 atomic_store(&_G.threads_gen, target_threads_gen);
964 }
965
966 pthread_attr_destroy(&attr);
967}

1133void thr_syn_wait_until(thr_syn_t *syn, bool (^cond)(void))
1134{
1135#ifdef __has_thr_acc
1136 unsigned long start;
1137#endif
1138
1139 cond = ^bool (void) {
1140 int pending = atomic_load(&syn->pending);
1141
1142 if (cond) {
1143 if (!cond()) {
1144 /* XXX ensure pending has been fetched before executing cond,
1145 * otherwise this would be racy.
1146 *
1147 * BTW, if this assert fails, the condition probably does not
1148 * really depend on the syn to be verified.
1149 */
1150 assert (pending);
1151 return false;
1152 }
1153 return true;
1154 }
1155 return pending == 0;
1156 };
1157
1158 thr_syn__retain(syn);
1159
1160 if (self_g->id == 0) {
1161 while (!cond()) {
1162 int ret;
1163 uint64_t key;
1164 thr_evc_t *ec;
1165
1166 if (!mpsc_queue_looks_empty(&thr_queue_main_g->q)) {
1167 thr_queue_drain(thr_queue_main_g);
1168 continue;
1169 }
1170
1171 atomic_store(&thr0_cur_ec_g, &syn->ec);
1172 key = thr_ec_get(&syn->ec);
1173#ifdef __has_thr_acc
1174 self_g->acc.ec_gets++;
1175#endif
1176 if (cond()) {
1177 goto cas;
1178 }
1179 if (!mpsc_queue_looks_empty(&thr_queue_main_g->q)) {
1180 goto cas;
1181 }
1182
1183#ifdef __has_thr_acc
1184 self_g->acc.ec_waits++;
1185 start = hardclock();
1186#endif
1187 /* allow quarantine thread to become active even if we wait
1188 * less than 100ms */
1189 atomic_fetch_add(&_G.target_threads_count, 1);
1190 thr_ec_broadcast(&_G.threads_count_evc);
1191 ret = thr_sleep_until(&syn->ec, 100, ^bool (void) {
1192 return cond() || !mpsc_queue_looks_empty(&thr_queue_main_g->q);
1193 });
1194 /* Decrease thread count. If we waited more than 100ms,
1195 * thr_enter_blocking_syscall() will increment it again. */
1196 atomic_fetch_sub(&_G.target_threads_count, 1);
1197 if (ret < 0) {
1198 thr_enter_blocking_syscall();
1199 thr_ec_wait(&syn->ec, key);
1200 thr_exit_blocking_syscall();
1201 }
1202#ifdef __has_thr_acc
1203 self_g->acc.ec_wait_time += (hardclock() - start);
1204#endif
1205 cas:
1206 ec = &syn->ec;
1207 while (unlikely(!atomic_compare_exchange_strong(&thr0_cur_ec_g,
1208 &ec, NULL)))
1209 {
1210 sched_yield();
1211 ec = &syn->ec;
1212 }
1213 }
1214 } else {
1215 uint64_t key = thr_ec_get(&syn->ec);
1216 int ret;
1217
1218#ifdef __has_thr_acc
1219 self_g->acc.ec_waits++;
1220 start = hardclock();
1221#endif
1222 atomic_fetch_add(&_G.target_threads_count, 1);
1223 thr_ec_broadcast(&_G.threads_count_evc);
1224 ret = thr_sleep_until(&syn->ec, 100, cond);
1225 atomic_fetch_sub(&_G.target_threads_count, 1);
1226 if (ret < 0) {
1227 thr_enter_blocking_syscall();
1228 do {
1229 thr_ec_timedwait(&syn->ec, key, 100);
1230 key = thr_ec_get(&syn->ec);
1231 } while (!cond());
1232 thr_exit_blocking_syscall();
1233 }
1234#ifdef __has_thr_acc
1235 self_g->acc.ec_wait_time += (hardclock() - start);
1236#endif
1237 }
1238
1239 thr_syn__release(syn);
1240}
