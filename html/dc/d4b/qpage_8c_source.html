23/*
24 * QPage allocator is inspired by qtlsf (look at qtlsf.[hc] for explanations).
25 *
26 * The mechanism used to find allocation slots in O(1) is directly inherited
27 * from tlsf. Unlike tlsf, as the page linking cannot be embedded inside the
28 * pages (since we care about page alignment a lot), allocation returns an
29 * extra cookie, called the segment.
30 *
31 * Caller should try to remember that segment as it makes deallocation
32 * an O(1) operation too, whereas it's not a O(1) if the proper segment
33 * is not provided (it's even quite slow).
34 *
35 * Unlike tlsf though, the page allocator provides mmap/unmap-like APIs: you
36 * can allocate many pages at once, and then dealloc some of the pages
37 * independently. This allows for example mapping all the pages that will be
38 * required to build a QDB column, and allows those pages to be unmapped
39 * individually (think of qdb_bitmaps who are doing that often).
40 *
41 * Also, unlike tlsf, we try to release memory under some circumstances. We
42 * don't release the addressing space, but we do release physical memory to
43 * the system. We also remember page dirtyness to some extent, making
44 * allocating clean pages quite efficient.
45 *
46 * All these features have a cost, and the qpage allocator is likely to be
47 * slower than qtlsf, but OTOH you allocate QPAGE_SIZE bytes at a time,
48 * which will likely not done a lot of times per second ;) I don't expect it to
49 * be a lot more than twice as slow as qtlsf, which makes it in the 100ns
50 * range per allocation (for raw pages).
51 */

114static ALWAYS_INLINE uint32_t mapping_class_upper(uint32_t npages)
115{
116 uint32_t level, mask;
117
118 if (npages <= CLASS_SMALL)
119 return npages - 1;
120
121 mask = (1U << (bsr32(npages) - CLASSES_SHIFT)) - 1;
122 npages += mask;
123 level = bsr32(npages) - CLASSES_SHIFT;
124 return (npages >> level) + (level << CLASSES_SHIFT) - 1;
125}
126
127static ALWAYS_INLINE uint32_t mapping_class(uint32_t npages)
128{
129 uint32_t level;
130
131 if (npages <= CLASS_SMALL)
132 return npages - 1;
133
134 level = bsr32(npages) - CLASSES_SHIFT;
135 return (npages >> level) + (level << CLASSES_SHIFT) - 1;
136}

138static ALWAYS_INLINE page_desc_t *find_suitable_block(uint32_t *class)
139{
140 unsigned vec = *class / bitsizeof(size_t);
141 size_t mask = BITMASK_GE(size_t, *class);
142
143 do {
144 size_t tmp = _G.bits[vec] & mask;
145 if (tmp) {
146 *class = vec * bitsizeof(size_t) + bsfsz(tmp);
147 return _G.blks[*class];
148 }
149 mask = (size_t)-1;
150 } while (++vec < BITS_LEN);
151
152 return NULL;
153}

182#if QPAGE_DEBUG
183static void qpages_check(page_run_t *run)
184{
185 uint32_t j = 0;
186
187 spin_lock(&_G.lock);
188 for (j = 0; j < run->npages + 1; j++) {
189 uint32_t flags = run->pages[j].flags;
190
191 if ((flags & BLK_FREE) && (flags & BLK_PREV_FREE))
192 e_panic("wrong flags for %p:%d", run, j);
193 }
194
195 for (j = 0; j < run->npages; ) {
196 page_desc_t *blk = run->pages + j;
197 uint32_t bsz = blk_size(blk);
198 page_desc_t *next = blk_next(blk, bsz);
199
200 if (blk->flags & BLK_FREE) {
201 if (next->flags & BLK_FREE) {
202 e_panic("two consecutive free blocks: %d, %d",
203 j, blk_no(next));
204 }
205 if (!(next->flags & BLK_PREV_FREE)) {
206 e_panic("missed that previous block is free [%d from %d]",
207 j, blk_no(next));
208 }
209 if (blk_get_prev(next) != blk) {
210 e_panic("previous free blk offset is wrong");
211 }
212 } else {
213 if (next->flags & BLK_PREV_FREE) {
214 e_panic("next block believe we're free %d, %d",
215 j, blk_no(next));
216 }
217 }
218 j += bsz;
219 }
220 spin_unlock(&_G.lock);
221}
222#else
223# define qpages_check(run) do { } while (0)
224#endif

226static inline void blk_insert(page_desc_t *blk, size_t npages)
227{
228 uint32_t class = mapping_class(npages);
229
230 blk->flags = npages | BLK_PREV_USED | BLK_FREE;
231 if ((blk->free_next = _G.blks[class])) {
232 blk->free_next->free_prev_next = &blk->free_next;
233 } else {
234 SET_BIT(_G.bits, class);
235 }
236 *(blk->free_prev_next = &_G.blks[class]) = blk;
237
238 blk[npages].flags |= BLK_PREV_FREE;
239 blk[npages].blk_prev = npages;
240}
241
242static inline uint32_t blk_remove(page_desc_t *blk)
243{
244 uint32_t npages = blk_size(blk);
245 uint32_t class = mapping_class(npages);
246
247 assert (blk->flags & BLK_FREE);
248 *blk->free_prev_next = blk->free_next;
249 if (blk->free_next) {
250 blk->free_next->free_prev_next = blk->free_prev_next;
251 } else
252 if (blk->free_prev_next == &_G.blks[class]) {
253 RST_BIT(_G.bits, class);
254 }
255 return npages;
256}

266 while (npages-- > 0)
267 (blk++)->dirty = true;

270static inline void
271blk_cleanse(page_run_t *run, uint32_t blkno, uint32_t npages)
272{
273 uint32_t end = npages + blkno;
274
275 for (uint32_t i = blkno; i < end; i++) {
276 if (run->pages[i].dirty) {
277 run->pages[i].dirty = false;
278 p_clear(run->mem_pages + i, 1);
279 }
280 }

283static inline void
284blk_setup_backptrs(page_desc_t *blk, uint32_t flags, uint32_t npages)
285{
286 blk->flags = flags | BLK_USED | npages;
287 for (uint32_t i = 1; i < npages; i++) {
288 blk[i].flags = BLK_PREV_USED | BLK_USED | BLK_PGINRUN | i;
289 }
290}

292static inline uint32_t blk_cut(page_desc_t *blk)
293{
294 uint32_t offs = blk->flags & ~BLK_STATE;
295 page_desc_t *tmp;
296 uint32_t tsz, bsz;
297
298 tmp = blk - offs;
299 tsz = blk_size(tmp);
300 assert (tsz >= offs);
301 tmp->flags = (tmp->flags & BLK_STATE) | offs;
302
303 blk_setup_backptrs(blk, BLK_PREV_USED, bsz = tsz - offs);
304 return bsz;
305}

307static NEVER_INLINE int create_arena(size_t npages)
308{
309 size_t pgsize = getpagesize();
310 size_t size, offset;
311 page_desc_t *blk, *end;
312 page_run_t *run;
313 qpage_t *pgs;
314
315 if (npages < QPAGE_ALLOC_MIN) {
316 npages = QPAGE_ALLOC_MIN;
317 } else {
318 if (npages & (npages - 1))
319 npages = 1U << (bsr32(npages) + 1);
320 }
321 size = npages * QPAGE_SIZE;
322 if (QPAGE_SIZE > pgsize)
323 size += QPAGE_SIZE;
324 pgs = mmap(0, size, PROT_READ | PROT_WRITE, MAP_PRIVATE | MAP_ANON, -1, 0);
325 if (pgs == MAP_FAILED)
326 return -1;
327 mem_tool_disallow_memory(pgs, size);
328
329 run = calloc(1, sizeof(page_run_t) + (npages + 1) * sizeof(page_desc_t));
330 if (run == NULL) {
331 mem_tool_allow_memory(pgs, size, true);
332 munmap(pgs, size);
333 return -1;
334 }
335 if (QPAGE_SIZE > pgsize) {
336 offset = (uintptr_t)pgs & QPAGE_MASK;
337 if (offset) {
338 npages--;
339 mem_tool_allow_memory(pgs, QPAGE_SIZE - offset, true);
340 munmap(pgs, QPAGE_SIZE - offset);
341 pgs = (qpage_t *)((uintptr_t)pgs + QPAGE_SIZE - offset);
342 mem_tool_allow_memory(pgs + npages, offset, true);
343 munmap(pgs + npages, offset);
344 } else {
345 mem_tool_allow_memory(pgs + npages, QPAGE_SIZE, true);
346 munmap(pgs + npages, QPAGE_SIZE);
347 }
348 }
349 run->mem_pages = pgs;
350 run->npages = npages;
351 run->segment = _G.segs.len;
352 qv_append(&_G.segs, run->pages);
353 for (uint32_t i = 0; i <= npages; i++) {
354 run->pages[i].blkno = i;
355 }
356
357 blk = run->pages + 0;
358 end = run->pages + npages;
359
360 blk->flags = BLK_PREV_USED | BLK_FREE | npages;
361 end->flags = BLK_PREV_FREE | BLK_USED;
362 blk_insert(blk, npages);
363 return 0;
364}

370static void
371free_n(page_run_t *run, page_desc_t *blk, size_t npages, uint32_t seg)
372{
373 uint32_t blkno = blk_no(blk);
374 page_desc_t *tmp;
375 size_t bsz;
376
377 spin_lock(&_G.lock);
378 if (blk->flags & BLK_PGINRUN) {
379 bsz = blk_cut(blk);
380 } else {
381 bsz = blk_size(blk);
382 }
383
384 assert (bsz >= npages);
385 if (bsz > npages) {
386 blk_cut(blk + npages);
387 bsz = npages;
388 } else {
389 tmp = blk_next(blk, bsz);
390 if (tmp->flags & BLK_FREE) {
391 bsz += blk_remove(tmp);
392 }
393 }
394 if (blk->flags & BLK_PREV_FREE) {
395 blk = blk_get_prev(blk);
396 bsz += blk_remove(blk);
397 }
398 blk_insert(blk, bsz);
399
400 if (bsz == run->npages) {
401#ifdef __linux__
402 madvise(run->mem_pages, bsz * QPAGE_SIZE, MADV_DONTNEED);
403#else
404 mmap(run->mem_pages, bsz * QPAGE_SIZE, PROT_READ | PROT_WRITE,
405 MAP_PRIVATE | MAP_ANON | MAP_FIXED, -1, 0);
406#endif
407 blk_set_clean(run->pages, bsz);
408 mem_tool_disallow_memory(run->mem_pages, bsz * QPAGE_SIZE);
409 } else {
410 /** Divide virtually the array of run->pages into chunk of size of
411 * QDB_MADVISE_THRESHOLD.
412 *
413 * (¹) blocks to free
414 * ↓↓ ↓ ↓
415 * ┌───────┬───────┬───────┬───────┬───────┐
416 * │ . dd│fffffff│fffff │ . │ │
417 * └───────┴───────┴───────┴───────┴───────┘
418 * ↑ ↑
419 * previous next
420 * free block free block
421 *
422 * When freeing last used blocks in a chunk, previous free block
423 * (blk) and next free block (blk + bsz) will point outside these
424 * chunk and we can call madvise.
425 * If it is not the last used blocks in the chunk, we still need to
426 * call blk_set_dirty for these blocks (see (¹)).
427 */
428 size_t chunk_begin;
429 size_t chunk_end;
430
431 /* find the first free chunk */
432 chunk_begin = ROUND(blkno, QDB_MADVISE_THRESHOLD);
433 if (chunk_begin <= blk_no(blk)) {
434 chunk_begin += QDB_MADVISE_THRESHOLD;
435 }
436 /* find the next non-free chunk */
437 chunk_end = ROUND_UP(blkno + npages, QDB_MADVISE_THRESHOLD);
438 if (chunk_end > blk_no(blk) + bsz) {
439 chunk_end -= QDB_MADVISE_THRESHOLD;
440 }
441
442 if (chunk_begin < chunk_end) {
443 const size_t chunk_sz = chunk_end - chunk_begin;
444
445#ifdef __linux__
446 madvise(run->mem_pages + chunk_begin, chunk_sz * QPAGE_SIZE,
447 MADV_DONTNEED);
448#else
449 mmap(run->mem_pages + chunk_begin, chunk_sz * QPAGE_SIZE,
450 PROT_READ | PROT_WRITE, MAP_PRIVATE | MAP_ANON | MAP_FIXED,
451 -1, 0);
452#endif
453 blk_set_clean(run->pages + chunk_begin, chunk_sz);
454 mem_tool_disallow_memory(run->mem_pages + chunk_begin,
455 chunk_sz * QPAGE_SIZE);
456 if (blkno < chunk_begin) {
457 blk_set_dirty(run->pages + blkno, (chunk_begin - blkno));
458 mem_tool_disallow_memory(run->mem_pages + blkno,
459 (chunk_begin - blkno) * QPAGE_SIZE);
460 }
461 if (blkno + npages > chunk_end) {
462 blk_set_dirty(run->pages + chunk_end,
463 (blkno + npages - chunk_end));
464 mem_tool_disallow_memory(run->mem_pages + chunk_end,
465 (blkno + npages - chunk_end) * QPAGE_SIZE);
466 }
467 } else {
468 blk_set_dirty(run->pages + blkno, npages);
469 mem_tool_disallow_memory(run->mem_pages + blkno,
470 npages * QPAGE_SIZE);
471 }
472 }
473
474 spin_unlock(&_G.lock);
475 qpages_check(run);
476}

478static page_desc_t *
479qpage_alloc_align_impl(size_t npages, size_t shift, bool zero, page_run_t **runp)
480{
481 page_desc_t *blk, *next, *split;
482 uint32_t class, blkno, size, offs;
483 uint32_t smask = (1U << shift) - 1;
484 page_run_t *run;
485
486 if (unlikely(npages == 0 || npages + smask > QPAGE_COUNT_MAX))
487 return NULL;
488
489 spin_lock(&_G.lock);
490 class = mapping_class_upper(npages + smask);
491 blk = find_suitable_block(&class);
492 if (unlikely(!blk)) {
493 if (create_arena(npages + smask) < 0) {
494 spin_unlock(&_G.lock);
495 return NULL;
496 }
497 class = mapping_class_upper(npages + smask);
498 blk = find_suitable_block(&class);
499 }
500
501 if ((_G.blks[class] = blk->free_next)) {
502 blk->free_next->free_prev_next = &_G.blks[class];
503 } else {
504 RST_BIT(_G.bits, class);
505 }
506
507 size = blk_size(blk);
508 next = blk_next(blk, size);
509 blkno = blk_no(blk);
510 run = run_of(blk, blkno);
511 offs = (((uintptr_t)run->mem_pages >> QPAGE_SHIFT) + blkno) & smask;
512
513 if (offs) {
514 offs = smask + 1 - offs;
515 blk_insert(blk, offs);
516 blk += offs;
517 blkno += offs;
518 size -= offs;
519 blk_setup_backptrs(blk, BLK_PREV_FREE, npages);
520 } else {
521 blk_setup_backptrs(blk, BLK_PREV_USED, npages);
522 }
523 if (size > npages) {
524 split = blk_next(blk, npages);
525 blk_insert(split, size - npages);
526 } else {
527 assert (size == npages);
528 next->flags &= ~BLK_PREV_FREE;
529 }
530
531 mem_tool_allow_memory(run->mem_pages + blk_no(blk), npages * QPAGE_SIZE,
532 zero);
533 if (zero) {
534 blk_cleanse(run, blkno, npages);
535 }
536 spin_unlock(&_G.lock);
537 *runp = run;
538 qpages_check(run);
539 return blk;
540}

564static inline int32_t qpage_find_seg(qpage_t *qp)
565{
566 spin_lock(&_G.lock);
567 for (int i = 0; i < _G.segs.len; i++) {
568 page_desc_t *blk = _G.segs.tab[i];
569 page_run_t *run = run_of(blk, blk_no(blk));
570
571 if ((size_t)(qp - run->mem_pages) < run->npages) {
572 spin_unlock(&_G.lock);
573 return i;
574 }
575 }
576 e_panic("invalid pointer or segment");
577}

579static void *remap(void *ptr, size_t old_n, uint32_t old_seg,
580 uint32_t new_n, uint32_t *new_seg,
581 bool may_move, bool zero)
582{
583 qpage_t *qp = ptr;
584 page_desc_t *blk, *tmp, *next;
585 page_run_t *run;
586 uint32_t bsz;
587
588 if (unlikely(old_seg > (uint32_t)_G.segs.len)) {
589 old_seg = qpage_find_seg(qp);
590 }
591 tmp = _G.segs.tab[old_seg];
592 run = run_of(tmp, blk_no(tmp));
593 blk = run->pages + (qp - run->mem_pages);
594 assert (blk + old_n <= run->pages + run->npages);
595 assert (!(blk->flags & BLK_FREE));
596 bsz = blk_size(blk);
597 assert (old_n <= bsz);
598
599 if (new_n <= bsz) {
600 if (new_n < bsz)
601 free_n(run, blk + new_n, bsz - new_n, old_seg);
602 if (new_seg)
603 *new_seg = old_seg;
604 return ptr;
605 }
606
607 spin_lock(&_G.lock);
608 next = blk_next(blk, bsz);
609 if ((next->flags & BLK_FREE) && new_n <= bsz + blk_size(next)) {
610 bsz += blk_remove(next);
611 if (bsz > new_n) {
612 tmp = blk_next(blk, new_n);
613 blk_insert(tmp, bsz - new_n);
614 } else {
615 next->flags &= ~BLK_PREV_FREE;
616 }
617 blk_setup_backptrs(blk, (blk->flags & BLK_PREV_FREE), new_n);
618 spin_unlock(&_G.lock);
619 mem_tool_allow_memory(qp + old_n, (new_n - old_n) * QPAGE_SIZE, zero);
620
621 if (zero) {
622 p_clear(qp + old_n, bsz - old_n);
623 blk_cleanse(run, blk_no(blk) + bsz, new_n - bsz);
624 }
625 if (new_seg)
626 *new_seg = old_seg;
627 qpages_check(run);
628 return ptr;
629 }
630 spin_unlock(&_G.lock);
631
632 if (may_move) {
633 page_run_t *new_run;
634 page_desc_t *new_blk;
635 uint32_t blkno;
636 qpage_t *res;
637
638 new_blk = qpage_alloc_align_impl(new_n, 0, false, &new_run);
639 if (!new_blk)
640 return NULL;
641
642 blkno = blk_no(new_blk);
643 res = new_run->mem_pages + blkno;
644 memcpy(res, ptr, old_n * QPAGE_SIZE);
645 free_n(run, blk, bsz, old_seg);
646
647 if (zero) {
648 p_clear(res + old_n, bsz - old_n);
649 blk_cleanse(new_run, blkno + bsz, new_n - bsz);
650 }
651 if (new_seg)
652 *new_seg = new_run->segment;
653 return res;
654 }
655
656 return NULL;
657}

671void qpage_free_n(void *ptr, size_t npages, uint32_t seg)
672{
673 qpage_t *qp = ptr;
674 page_desc_t *blk, *tmp;
675 page_run_t *run;
676
677 if (!ptr)
678 return;
679
680 if (unlikely(seg > (uint32_t)_G.segs.len)) {
681 seg = qpage_find_seg(qp);
682 }
683 tmp = _G.segs.tab[seg];
684 run = run_of(tmp, blk_no(tmp));
685 blk = run->pages + (qp - run->mem_pages);
686 assert (blk + npages <= run->pages + run->npages);
687 assert (!(blk->flags & BLK_FREE));
688 free_n(run, blk, npages, seg);
689}
