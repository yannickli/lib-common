43/* Note about GCD algorithms:
44 *
45 * Stein's algorithm is significantly better than Euclid's one for lower
46 * values (the switch is located around 1M on a 2009 quad core). For greater
47 * values, Euclid's algorithm seems to take advantage of the low-level
48 * optimization of modulo operator.
49 *
50 * We suppose that most basic usages of GCD will be alright with Stein's
51 * algorithm.
52 *
53 */
54
55uint32_t gcd_euclid(uint32_t a, uint32_t b)
56{
57 if (a < b) {
58 SWAP(uint32_t, a, b);
59 }
60
61 while (b) {
62 a = a % b;
63 SWAP(uint32_t, a, b);
64 }
65
66 return a;
67}
68
69uint32_t gcd_stein(uint32_t a, uint32_t b)
70{
71 uint8_t za;
72 uint8_t zb;
73
74 if (!a)
75 return b;
76 if (!b)
77 return a;
78
79 za = bsf32(a);
80 a >>= za;
81 zb = bsf32(b);
82 b >>= zb;
83
84 while (a != b) {
85 if (a > b) {
86 a -= b;
87 a >>= bsf32(a);
88 } else {
89 b -= a;
90 b >>= bsf32(b);
91 }
92 }
93
94 return a << MIN(za, zb);
95}
96
97uint32_t gcd(uint32_t a, uint32_t b)
98{
99 return gcd_stein(a, b);
100}

104uint32_t get_multiples_nb_in_range(uint32_t n, uint32_t min, uint32_t max)
105{
106 if (!expect(max >= min && n != 0)) {
107 return 0;
108 }
109 return 1 + (max / n) - DIV_ROUND_UP(min, n);
110}
