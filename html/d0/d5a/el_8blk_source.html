273static ev_t *el_create(ev_type_t type, void *cb, data_t priv, bool ref)
274{
275 ev_t *res;
276
277 if (unlikely(dlist_is_empty(&_G.evs_free))) {
278 if (unlikely(_G.evs_alloc_next >= _G.evs_alloc_end)) {
279 int bucket_len = 1 << (_G.buckets.len + EV_ALLOC_FACTOR);
280 ev_t *bucket = p_new(ev_t, bucket_len);
281
282 qv_append(&_G.buckets, bucket);
283 if (unlikely(_G.buckets.len > EVS_NB_MAX_BUCKETS)) {
284 e_panic(E_PREFIX("insane amount of events"));
285 }
286 _G.evs_alloc_next = bucket;
287 _G.evs_alloc_end = bucket + bucket_len;
288 }
289 res = _G.evs_alloc_next++;
290 } else {
291 res = container_of(_G.evs_free.next, ev_t, ev_list);
292 dlist_remove(&res->ev_list);
293 }
294
295 *res = (ev_t){
296 .type = type,
297 .cb.cb = cb,
298 };
299 res->priv = priv;
300 dlist_init(&res->ev_list);
301
302 logger_trace(&_G.logger, 2, "creating event %p (%s)",
303 res, ev_type_to_str(res->type));
304 assert (MODULE_IS_LOADED(el) || MODULE_IS_INITIALIZING(el));
305
306 _G.used++;
307 return ref ? el_ref(res) : res;
308}

328static data_t el_destroy(ev_t **evp)
329{
330 ev_t *ev = *evp;
331
332 logger_trace(&_G.logger, 2, "destroying event %p (%s)",
333 ev, ev_type_to_str(ev->type));
334 assert (MODULE_IS_LOADED(el) || MODULE_IS_SHUTTING_DOWN(el));
335 assert (!MODULE_IS_LOADED(thr) || thr_is_on_queue(thr_queue_main_g));
336
337 if (EV_FLAG_HAS(ev, IS_BLK)) {
338 block_t wipe = ev->wipe;
339
340 if (wipe) {
341 wipe();
342 Block_release(wipe);
343 }
344 Block_release(ev->cb.cb);
345 EV_FLAG_RST(ev, IS_BLK);
346 ev->priv = (data_t)NULL;
347 }
348 __el_unref(ev);
349 dlist_move(&_G.evs_gc, &ev->ev_list);
350 ev->generation = 0;
351 ev->type = EV_UNUSED;
352 ev->flags = 0;
353 *evp = NULL;
354 _G.used--;
355 return ev->priv;
356}

396{

451static void el_idle_process(uint64_t now)
452{
453 static uint64_t last_run = UINT64_MAX;
454

588static bool signal_is_terminating(int signo)

590 return signo == SIGTERM || signo == SIGINT || signo == SIGQUIT

594static void el_sighandler(int signum, siginfo_t *siginfo, void *ctx)
595{
596 _G.gotsigs |= (1 << signum);
597
598 if (signal_is_terminating(signum)) {
599 _G.terminating = true;
600 }
601
602 /* Refer to 'man 2 sigaction' for the meaning of the codes. */
603 logger_trace(&_G.logger, 1,
604 "received signal %d from PID %d, UID %d (code %s)",
605 signum, siginfo->si_pid, siginfo->si_uid,
606 si_code_to_str(signum, siginfo->si_code));
607}

609static void el_signal_process(void)
610{
611 uint8_t generation;
612 uint32_t gotsigs = _G.gotsigs;
613 struct timeval now;
614
615 if (!gotsigs)
616 return;
617
618 _G.gotsigs &= ~gotsigs;
619 lp_gettv(&now);
620
621 generation = ev_cache_list(&_G.sigs);
622 tab_for_each_entry(ev, &_G.cache) {
623 int signo;
624
625 if (ev->generation != generation) {
626 continue;
627 }
628
629 CHECK_EV_TYPE(ev, EV_SIGNAL);
630
631 signo = ev->signo;
632 if (gotsigs & (1 << signo)) {
633 if (signal_is_terminating(signo)) {
634 module_on_term(signo);
635 }
636 if (EV_FLAG_HAS(ev, IS_BLK)) {
637 ev->cb.signal_blk(ev, signo);
638 } else {
639 (*ev->cb.signal)(ev, signo, ev->priv);
640 }
641 _G.has_run = true;
642 }
643 }
644}

690static void el_sigchld_hook(ev_t *ev, int signo, data_t priv)
691{
692 t_scope;
693 qv_t(ev_sigchld_assoc) ev_assocs;
694
695 t_qv_init(&ev_assocs, qm_len(ev_assoc, &_G.childs));
696
697 /* XXX the callback called by el_child_fire or the module method
698 * at_fork_on_child_terminated may change the global qm `_G.childs`, so we
699 * cannot call them while iterating on the qm and removing elements from
700 * it.
701 * We first need to list the dead child processes, and then call
702 * el_child_fire() or the module method at_fork_on_child_terminated in
703 * another loop.
704 */
705 qm_for_each_pos(ev_assoc, pos, &_G.childs) {
706 int pid = _G.childs.keys[pos];
707 int status = 0;
708 int res;
709
710 res = waitpid(pid, &status, WNOHANG);
711 if (res == pid || (res == -1 && errno == ECHILD)) {
712 ev_sigchld_assoc_t ev_assoc;
713
714 /* The child process is dead. */
715 ev_assoc.pid = pid;
716 ev_assoc.child = _G.childs.values[pos];
717 if (likely(ev_assoc.child)) {
718 ev_assoc.child->child.status = status;
719 }
720 qv_append(&ev_assocs, ev_assoc);
721 qm_del_at(ev_assoc, &_G.childs, pos);
722 }
723 }
724
725 t_seal();
726
727 tab_for_each_ptr(ev_assoc, &ev_assocs) {
728 if (likely(ev_assoc->child)) {
729 el_child_fire(&ev_assoc->child);
730 } else {
731 MODULE_METHOD_RUN_INT(at_fork_on_child_terminated, ev_assoc->pid);
732 }

768ev_t *el_child_register_d(pid_t pid, el_child_f *cb, data_t priv)
769{
770 uint32_t pos;
771 sigset_t set;
772 sigset_t prev;
773 int status;
774
775 ev_t *ev = el_create(EV_CHILD, cb, priv, true);
776
777 assert (pid > 0);
778 ev->child.pid = pid;
779 ev->child.has_exited = false;
780
781 pos = qm_put(ev_assoc, &_G.childs, pid, ev, 0);
782 /* It is likely that the child process comes from ifork(). */
783 if (likely(pos & QHASH_COLLISION)) {
784 pos ^= QHASH_COLLISION;
785 ASSERT("pid is already watched", !_G.childs.values[pos]);
786 _G.childs.values[pos] = ev;
787 }
788
789 sigemptyset(&set);
790 sigaddset(&set, SIGCHLD);
791
792 pthread_sigmask(SIG_BLOCK, &set, &prev);
793
794 if (waitpid(pid, &status, WNOHANG) > 0) {
795 /* The process is already dead */
796 if (qm_del_key(ev_assoc, &_G.childs, ev->child.pid) < 0) {
797 ASSERT("event not found", false);
798 }
799 ev->child.status = status;
800 ev->child.has_exited = true;
801
802 pthread_sigmask(SIG_SETMASK, &prev, NULL);
803 return ev_add(&_G.before, ev);
804 }
805
806 el_sigchld_register();
807 pthread_sigmask(SIG_SETMASK, &prev, NULL);
808 return ev;
809}

847pid_t el_spawn_child(const char *file, const char *argv_in[],
848 const char *envp[], block_t child, el_child_b blk,
849 block_t wipe)
850{
851 pid_t pid;
852 qv_t(cstr) argv_final;
853
854 qv_init(&argv_final);
855 qv_append(&argv_final, file);
856
857 {
858 logger_debug_scope(&_G.tracing_logger);
859 const char **ptr = &argv_in[0];
860
861 logger_cont("running command %s", file);
862 while (*ptr) {
863 logger_cont(" %s", *ptr);
864 qv_append(&argv_final, *ptr);
865
866 ptr++;
867 }
868 }
869
870 qv_append(&argv_final, NULL);
871
872 if ((pid = ifork()) == 0) {
873 if (child) {
874 child();
875 }
876 if (envp) {
877 /* XXX we cannot use execvpe which appeared in glibc 2.11 since
878 * centos 5 only has gblic 2.5 */
879 environ = (char **)envp;
880 }
881 execvp(file, (char **)argv_final.tab);
882 logger_fatal(&_G.logger, "unable to execute `%s`: %m", file);
883 } else
884 if (pid < 0) {
885 logger_fatal(&_G.logger,
886 "unable to fork `%s` in the background: %m", file);
887 }
888 qv_wipe(&argv_final);
889 el_child_register_blk(pid, blk, wipe);
890 return pid;
891}

938static void spawn_child_timeout(el_t el, data_t data)
939{
940 spawn_child_capture_t *ctx = data.ptr;
941
942 if (killpg(ctx->pid, ctx->next_sig) < 0) {
943 if (errno == ESRCH) {
944 /* No process can be found in the process group. This can happen
945 * if the timeout is very low and the system is under stress,
946 * because there is a race with the creation of the process group
947 * in the child. In that case, no need to change the signal. */
948 return;
949 }
950 }
951 if (ctx->next_sig == SIGINT) {
952 ctx->next_sig = SIGTERM;
953 } else {
954 ctx->next_sig = SIGKILL;
955 }
956}

958pid_t el_spawn_child_capture(const char *file,
959 const char *argv[],
960 const char *envp[],
961 int timeout,
962 block_t child,
963 el_child_output_b blk,
964 block_t wipe)
965{
966 spawn_child_capture_t *ctx = spawn_child_capture_new();
967 int pfd[2];
968 int *pfd_ptr = pfd;
969
970 if (pipe(pfd) < 0) {
971 logger_fatal(&_G.logger,
972 "unable to execute `%s`: cannot prepare out fds", file);
973 }
974
975 ctx->blk = Block_copy(blk);
976 if (child) {
977 ctx->child = Block_copy(child);
978 }
979 if (wipe) {
980 ctx->wipe = Block_copy(wipe);
981 }
982
983 ctx->pid = el_spawn_child(file, argv, envp, ^{
984 qv_t(u32) fds_to_keep;
985
986 setpgid(0, 0); /* Create a process group id. */
987 p_close(&pfd_ptr[0]);
988
989 qv_inita(&fds_to_keep, 1);
990 qv_append(&fds_to_keep, pfd_ptr[1]);
991 close_fds(STDIN_FILENO, &fds_to_keep);
992
993 dup2(pfd_ptr[1], STDOUT_FILENO);
994 dup2(pfd_ptr[1], STDERR_FILENO);
995 p_close(&pfd_ptr[1]);
996
997 devnull_dup(STDIN_FILENO);
998
999 if (ctx->child) {
1000 ctx->child();
1001 }
1002 }, ^(el_t el, pid_t _pid, int status) {
1003 spawn_child_capture_on_output(ctx->capture,
1004 el_fd_get_fd(ctx->capture),
1005 POLLIN, (data_t){ .ptr = ctx });
1006 ctx->blk(el, _pid, status, LSTR_SB_V(&ctx->out));
1007 }, ^{
1008 spawn_child_capture_t *_ctx = ctx;
1009
1010 if (ctx->wipe) {
1011 ctx->wipe();
1012 }
1013 spawn_child_capture_delete(&_ctx);
1014 });
1015
1016 p_close(&pfd[1]);
1017 fd_set_features(pfd[0], O_NONBLOCK);
1018
1019 if (timeout > 0) {
1020 ctx->next_sig = SIGINT;
1021 ctx->timer = el_timer_register(timeout, timeout, EL_TIMER_LOWRES,
1022 &spawn_child_timeout, ctx);
1023 }
1024 ctx->capture = el_fd_register(pfd[0], true, POLLIN,
1025 &spawn_child_capture_on_output, ctx);
1026
1027 return ctx->pid;
1028}

1031/* {{{ timer events */
1032
1033/*
1034 * timers are extremely efficient, and use a binary-min-heap as a base
1035 * structure.
1036 *
1037 * Binary min-heaps are balanced binary trees stored in an array.
1038 * The invariant is that the "label" (for us the millisecond the timer has to
1039 * start at) of each parent node, is smaller than any of its children ones.
1040 *
1041 * Hence the next to be run timer is the root of the heap.
1042 *
1043 * Adding/Updating/... a timer is pseudo linear O(log(n)) in the number of
1044 * timers.
1045 */
1046
1047static data_t el_timer_unregister(ev_t **evp)
1048{
1049 if (unlikely(!*evp))
1050 return (data_t)NULL;
1051
1052 qhp_remove(timer, &_G.timers, (*evp)->timer.heappos);
1053
1054 return el_destroy(evp);
1055}

1057static int el_timer_next_expiration(int timeout, uint64_t clk)
1058{
1059 if (!qhp_is_empty(timer, &_G.timers)) {
1060 uint64_t nxt = TIMER_TOLERATED_EXPIRY(qhp_first(timer, &_G.timers));
1061
1062 if (nxt < (uint64_t)timeout + clk) {
1063 return MAX(0, (int)(nxt - clk));
1064 }
1065 }
1066 return timeout;
1067}

1069static void el_timer_process(uint64_t until)
1070{
1071 struct timeval tv;
1072
1073 lp_gettv(&tv);
1074 while (!qhp_is_empty(timer, &_G.timers)) {
1075 ev_t *ev = qhp_first(timer, &_G.timers);
1076
1077 ASSERT("should be a timer", ev->type == EV_TIMER);
1078 if (ev->timer.expiry > until) {
1079 return;
1080 }
1081
1082 logger_trace(&_G.logger, 3, "trigger timer %p", ev);
1083
1084 EV_FLAG_RST(ev, TIMER_UPDATED);
1085 if (EV_FLAG_HAS(ev, IS_BLK)) {
1086 ev->cb.cb_blk(ev);
1087 } else {
1088 (*ev->cb.cb)(ev, ev->priv);
1089 }
1090 _G.has_run = true;
1091
1092 /* ev has been unregistered in (*cb) */
1093 if (ev->type == EV_UNUSED) {
1094 continue;
1095 }
1096
1097 if (ev->timer.repeat > 0) {
1098 ev->timer.expiry += ev->timer.repeat;
1099 if (!EV_FLAG_HAS(ev, TIMER_NOMISS) && ev->timer.expiry < until) {
1100 uint64_t delta = until - ev->timer.expiry;
1101
1102 ev->timer.expiry += ROUND_UP(delta, (uint64_t)ev->timer.repeat);
1103 }
1104 __qhp_down(timer, &_G.timers, ev->timer.heappos);
1105 } else
1106 if (!EV_FLAG_HAS(ev, TIMER_UPDATED)) {
1107 el_timer_unregister(&ev);
1108 }
1109 }
1110}

1112static uint64_t get_clock(void)
1113{
1114 struct timespec ts;
1115 int res;
1116
1117#if defined(CLOCK_MONOTONIC) /* POSIX */
1118 res = clock_gettime(CLOCK_MONOTONIC, &ts);
1119#else
1120# error you need to find out how to get a monotonic clock for your system
1121#endif
1122 assert (res == 0);
1123 return 1000ull * ts.tv_sec + ts.tv_nsec / 1000000;
1124}

1126static bool el_timer_has_pending_events(void)
1127{
1128 ev_t *ev;
1129 uint64_t now = 0;
1130
1131 if (!qhp_is_empty(timer, &_G.timers) || _G.worker_running) {
1132 now = get_clock();
1133 }

1149static void el_timer_compute_tolerance(ev_t *ev)
1150{
1151 int64_t repeat = ev->timer.repeat;
1152
1153 if (repeat < 0) {
1154 repeat = -repeat;
1155 }
1156
1157 if (EV_FLAG_HAS(ev, TIMER_LOWRES) && repeat >= 500) {
1158 /* Timer is explicitly flagged as "low resolution" and has a
1159 * repetition of more than 500ms (and thus does not need high
1160 * precision), so allow a shift of up to 10% of the timer duration.
1161 * That shift is actually capped at EL_LOOP_TIMEOUT since this is the
1162 * duration of an event loop in case of inactivity.
1163 */
1164 ev->timer.tolerance = MIN(repeat / 10, EL_LOOP_TIMEOUT);
1165 } else {
1166 /* Otherwise, make it high-precision */
1167 ev->timer.tolerance = 0;
1168 }
1169}

1171ev_t *el_timer_register_d(int64_t next, int64_t repeat,
1172 ev_timer_flags_t flags, el_cb_f *cb, data_t priv)
1173{
1174 ev_t *ev = el_create(EV_TIMER, cb, priv, true);
1175
1176 if (flags & EL_TIMER_NOMISS) {
1177 EV_FLAG_SET(ev, TIMER_NOMISS);
1178 }
1179 if (flags & EL_TIMER_LOWRES) {
1180 EV_FLAG_SET(ev, TIMER_LOWRES);
1181 }
1182 if (repeat > 0) {
1183 ev->timer.repeat = repeat;
1184 } else {
1185 ev->timer.repeat = -next;
1186 }
1187 el_timer_compute_tolerance(ev);
1188 ev->timer.expiry = (uint64_t)next + get_clock();

1190
1191 if (logger_is_traced(&_G.logger, 2)) {
1192 logger_trace_scope(&_G.logger, 2);

1194
1195 logger_cont("register %stimer on event %p ",
1196 one_shot ? "one shot " : "", ev);
1197 logger_cont("(next: %jdms, ", next);
1198 if (!one_shot) {
1199 logger_cont("expiry: %ju.%03ju, ",
1200 ev->timer.expiry / 1000,
1201 ev->timer.expiry % 1000);
1202 }
1203 logger_cont("tolerance: %ums, ", ev->timer.tolerance);
1204 logger_cont("cb: %p)", (void *)ev->cb.cb);
1205 }

1416 if (ev->type == EV_WAKE) {
1417 ev = ev->wake.read_fd;
1418 }

1422 EV_FLAG_RST(ev, FD_FIRED);
1423 dlist_remove(&ev->ev_list);
1424 el_fd_fire(ev, POLLIN);
1425 timeout = 0;
1426 has_fired = true;
1427 }
1428 pfd[i] = (struct pollfd){
1429 .fd = ev->fd.fd,
1430 .events = ev->events_wanted
1431 };
1432 }
1433
1434 res = poll(pfd, count, timeout);
1435 if (flags & EV_FDLOOP_HANDLE_TIMERS) {
1436 if (!qhp_is_empty(timer, &_G.timers)) {
1437 el_timer_process(get_clock());
1438 }
1439 }
1440 if (flags & EV_FDLOOP_HANDLE_SIGNALS) {
1441 el_signal_process();
1442 }
1443 if (res > 0) {
1444 for (int i = 0; i < count; i++) {
1445 ev_t *ev = evs[i];
1446
1447 if (ev->type == EV_WAKE) {
1448 ev = ev->wake.read_fd;
1449 }
1450 if (pfd[i].revents) {
1451 el_fd_fire(ev, pfd[i].revents);
1452 has_fired = true;
1453 }
1454 }
1455 }
1456 return has_fired ? 1 : 0;
1457}

1600el_t el_wake_register_d(el_cb_f *cb, data_t priv)
1601{
1602 el_t el = el_create(EV_WAKE, cb, priv, true);
1603
1604 /* Use eventfd if available, since it's much more efficient than
1605 * the pipe alternative. (Benches show that eventfd wake up the main
1606 * thread in 7 micro-seconds while a pipe wake up the main thread in
1607 * 11 micro-seconds)
1608 */
1609#ifdef __linux__
1610 el->wake.write_fd = eventfd(0, O_NONBLOCK | O_CLOEXEC);
1611#else
1612 el->wake.write_fd = -1;
1613#endif
1614
1615 if (el->wake.write_fd >= 0) {
1616 el->wake.read_fd = el_fd_register(el->wake.write_fd, true, POLLIN,
1617 &el_wake_on_event, el);
1618 el->wake.use_eventfd = true;
1619 } else {
1620 int fd[2] = { -1, -1 };
1621
1622 if (pipe(fd) < 0) {
1623 el_destroy(&el);
1624 return NULL;
1625 }
1626
1627 fd_set_features(fd[0], O_NONBLOCK | O_CLOEXEC);
1628 fd_set_features(fd[1], O_NONBLOCK | O_CLOEXEC);
1629 el->wake.write_fd = fd[1];
1630 el->wake.read_fd = el_fd_register(fd[0], true, POLLIN,
1631 &el_wake_on_event, el);
1632 }
1633 el_unref(el->wake.read_fd);
1634 return el;
1635}

1758void el_loop_timeout(int timeout)
1759{
1760 uint64_t clk = get_clock();
1761
1762 _G.loop_depth++;
1763 el_timer_process(clk);
1764 if (unlikely(_G.unloop)) {
1765 _G.loop_depth--;
1766 return;
1767 }
1768 el_before_process();
1769 el_idle_process(clk);
1770 if (!dlist_is_empty(&_G.proxy_ready) || !dlist_is_empty(&_G.idle)
1771 || !dlist_is_empty(&_G.fired))
1772 {
1773 timeout = 0;
1774 }
1775 if (_G.worker && timeout && !el_has_pending_events()) {
1776 uint64_t start, end;
1777 int diff;
1778
1779 start = get_clock();
1780 timeout = el_timer_next_expiration(timeout, start);
1781 _G.worker_running = true;
1782 _G.worker_end = start + timeout;
1783 (*_G.worker)(timeout);
1784 _G.worker_running = false;;
1785 end = get_clock();
1786
1787 diff = end - start;
1788 if (diff > timeout + 100) {
1789 e_warning("worker is too long: %dms while expected %dms",
1790 diff, timeout);
1791 } else
1792 if (diff > MAX(timeout * 2, timeout + 10)) {
1793 e_trace(0, "worker is too long: %dms while expecting %dms",
1794 diff, timeout);
1795 }
1796 timeout = MAX(0, timeout - diff);
1797 clk = end;
1798 }
1799
1800 el_fd_process_fired();
1801 el_loop_fds(el_timer_next_expiration(timeout, clk));
1802 el_loop_proxies();
1803 el_signal_process();
1804 if (_G.loop_depth <= 1) {
1805 /* To be reentrant we can't reuse unregistered el_t until we came back
1806 * to the main loop */
1807 assert (_G.loop_depth == 1);
1808 dlist_splice(&_G.evs_free, &_G.evs_gc);
1809 }
1810 _G.loop_depth--;
1811
1812 /* Eventually reset the thread-local t_pool. */
1813 mem_stack_pool_try_reset(&t_pool_g);
1814}

1923static int el_get_state(sb_t *sb, bool only_blocking)
1924{
1925 /* XXX: do not use the t_stack here, since it could be called from
1926 * el_shutdown, where the t_stack can be destroyed. */
1927 int res;
1928 qv_t(table_hdr) hdr;
1929 qv_t(table_data) *rows = qv_new(table_data);
1930 table_hdr_t hdr_data[] = { {
1931 .title = LSTR_IMMED("EV POINTER"),
1932 }, {
1933 .title = LSTR_IMMED("TYPE"),
1934 }, {
1935 .title = LSTR_IMMED("CALLBACK"),
1936 }, {
1937 .title = LSTR_IMMED("DATA"),
1938 }, {
1939 .title = LSTR_IMMED("DETAILS"),
1940 .omit_if_empty = true,
1941 }
1942 };
1943 uint32_t hdr_size = countof(hdr_data);
1944
1945 qv_init_static(&hdr, hdr_data, hdr_size);
1946
1947 el_for_each(^(ev_t *ev) {
1948 qv_t(lstr) *tab;
1949 lstr_t details = LSTR_NULL;
1950
1951 if (only_blocking && !EV_FLAG_HAS(ev, REFS)) {
1952 return;
1953 }
1954
1955 tab = qv_growlen(rows, 1);
1956 qv_init(tab);
1957 qv_grow(tab, hdr_size);
1958
1959 qv_append(tab, lstr_fmt("%p", ev));
1960 qv_append(tab, LSTR(ev_type_to_str(ev->type)));
1961 if (ev->flags & EV_FLAG_IS_BLK) {
1962 qv_append(tab, LSTR("block"));
1963 } else {
1964 qv_append(tab, lstr_fmt("%p", (void *)ev->cb.cb));
1965 }
1966 qv_append(tab, lstr_fmt("%p", ev->priv.ptr));
1967
1968 switch (ev->type) {
1969 case EV_SIGNAL:
1970 details = lstr_fmt("signal: %d", ev->signo);
1971 break;
1972 case EV_CHILD:
1973 details = lstr_fmt("pid: %d", ev->child.pid);
1974 break;
1975 case EV_FD:
1976 details = lstr_fmt("fd: %d%s, events: %04x",
1977 ev->fd.fd,
1978 ev->fd.owned ? " (owned)" : "",
1979 ev->events_wanted);
1980 break;
1981 case EV_TIMER:
1982 if (ev->timer.repeat < 0) {
1983 details = lstr_fmt("one shot, expiry: %ju.%03ju, "
1984 "tolerance: %ums",
1985 ev->timer.expiry / 1000,
1986 ev->timer.expiry % 1000,
1987 ev->timer.tolerance);
1988 } else {
1989 details = lstr_fmt("repeat: %jdms, next: %ju.%03ju, "
1990 "tolerance: %ums",
1991 ev->timer.repeat,
1992 ev->timer.expiry / 1000,
1993 ev->timer.expiry % 1000,
1994 ev->timer.tolerance);
1995 }
1996 break;
1997 case EV_PROXY:
1998 details = lstr_fmt("events wanted:%04x, "
1999 "events available: %04x",
2000 ev->events_wanted, ev->events_avail);
2001 break;
2002 default:
2003 break;
2004 }
2005 qv_append(tab, details);
2006 });
2007
2008 sb_add_table(sb, &hdr, rows);
2009 sb_shrink(sb, 1);
2010 res = rows->len;
2011
2012 tab_for_each_ptr(tab, rows) {
2013 qv_deep_wipe(tab, lstr_wipe);
2014 }
2015 qv_delete(&rows);
2016
2017 return res;
2018}

2082static int el_shutdown(void)
2083{
2084 el_unregister(&_G.el_on_pwr);
2085 el_unregister(&_G.el_sigchld_hook);
2086
2087 /* Wipe all containers in order to remove traces in valgrind, however
2088 * ensure they remain valid in case some other destructor perform el
2089 * registrations/unregistrations.
2090 */
2091 if (_G.timers.len == 0) {
2092 qhp_wipe(timer, &_G.timers);
2093 qhp_init(timer, &_G.timers);
2094 }
2095 if (qm_len(ev_assoc, &_G.childs) == 0) {
2096 qm_wipe(ev_assoc, &_G.childs);
2097 qm_init(ev_assoc, &_G.childs);
2098 }
2099 if (qm_len(ev, &_G.fd_act) == 0) {
2100 qm_wipe(ev, &_G.fd_act);
2101 qm_init(ev, &_G.fd_act);
2102 }
2103 el_fs_watch_shutdown();
2104
2105 qv_wipe(&_G.cache);
2106 qv_init(&_G.cache);
2107
2108 /* Automatically destroy SIGNAL events (we don't want write boring code to
2109 * make it ourselves). */
2110 el_for_each(^(ev_t *ev) {
2111 if (!EV_FLAG_HAS(ev, REFS) && ev->type == EV_SIGNAL) {
2112 el_unregister(&ev);
2113 }
2114 });
2115
2116 /* Check for leaked events. */
2117 if (_G.used) {
2118 if (logger_is_traced(&_G.logger, 1)) {
2119 SB_1k(buf);
2120 int nb_used = el_get_state(&buf, false);
2121
2122 logger_trace(&_G.logger, 1, "%d events are leaked:\n%*pM",
2123 nb_used, SB_FMT_ARG(&buf));
