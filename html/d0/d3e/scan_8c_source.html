58static ALWAYS_INLINE uint32_t sum_epi64(__v2di xmm)
59{
60 union xmm x = { .q = xmm };
61
62 x.sf = _mm_movehl_ps(x.sf, x.sf); /* x.q[0] <- x.q[1] */
63 x.q += xmm; /* x.q[0] += xmm[0] */
64 return x.vd[0];
65}
66
67static ALWAYS_INLINE uint32_t sum_epi32(__v4si xmm)
68{
69 union xmm x = { .d = xmm };
70
71 x.sf = _mm_movehl_ps(x.sf, x.sf); /* x.d[0..1] <- x.q[2..3] */
72 x.d += xmm; /* x.d[0..1] += xmm[0..1] */
73 return x.vd[0] + x.vd[1];
74}

76static ALWAYS_INLINE uint32_t sum_epi16(__v8hi xmm)
77{
78 union xmm x1 = { .w = xmm }, x2 = { .w = xmm };
79
80 x1.d &= (__v4si){ 0x0000ffff, 0x0000ffff, 0x0000ffff, 0x0000ffff, };
81 x2.i = _mm_srli_epi32(x2.i, 16);
82 return sum_epi32(x1.d + x2.d);
83}

85static ALWAYS_INLINE bool is_128bits_zero(const void *v)
86{
87 __m128i t = _mm_cmpeq_epi32(*(const __m128i *)v, (__m128i){ 0, });
88 return _mm_movemask_epi8(t) == 0xffff;

91bool is_memory_zero(const void *_data, size_t n)
92{
93 const uint8_t *data = _data;
94
95 assert (n % 64 == 0);
96 assert ((uintptr_t)data % 16 == 0);
97 for (size_t i = 0; i < n; i += 64) {
98 if (!is_128bits_zero(data + i + 0))
99 return false;
100 if (!is_128bits_zero(data + i + 16))
101 return false;
102 if (!is_128bits_zero(data + i + 32))
103 return false;
104 if (!is_128bits_zero(data + i + 48))
105 return false;
106 }
107 return true;
108}

110ssize_t scan_non_zero16(const uint16_t u16[], size_t pos, size_t len)
111{
112 if (len - pos >= 8) {
113#define T(x, offs) \
114 ({ __m128i c = _mm_cmpeq_epi16(x, (__m128i){ 0, }); \
115 int m = _mm_movemask_epi8(c); \
116 if (m != 0xffff) \
117 return offs + __builtin_ctz(~m) / 2; \
118 })
119 for (; pos + 32 <= len; pos += 32) {
120 T(_mm_loadu_si128((__m128i *)(u16 + pos + 0)), pos + 0);
121 T(_mm_loadu_si128((__m128i *)(u16 + pos + 8)), pos + 8);
122 T(_mm_loadu_si128((__m128i *)(u16 + pos + 16)), pos + 16);
123 T(_mm_loadu_si128((__m128i *)(u16 + pos + 24)), pos + 24);
124 }
125 for (; pos + 8 <= len; pos += 8) {
126 T(_mm_loadu_si128((__m128i *)(u16 + pos)), pos);
127 }
128#undef T
129 }
130
131 for (; pos < len; pos++) {
132 if (u16[pos]) {
133 return pos;
134 }
135 }
136 return -1;
137}

139ssize_t scan_non_zero32(const uint32_t u32[], size_t pos, size_t len)
140{
141 if (len - pos >= 4) {
142#define T(x, offs) \
143 ({ __m128i c = _mm_cmpeq_epi32(x, (__m128i){ 0, }); \
144 int m = _mm_movemask_epi8(c); \
145 if (m != 0xffff) \
146 return offs + __builtin_ctz(~m) / 4; \
147 })
148 for (; pos + 16 <= len; pos += 16) {
149 T(_mm_loadu_si128((__m128i *)(u32 + pos + 0)), pos + 0);
150 T(_mm_loadu_si128((__m128i *)(u32 + pos + 4)), pos + 4);
151 T(_mm_loadu_si128((__m128i *)(u32 + pos + 8)), pos + 8);
152 T(_mm_loadu_si128((__m128i *)(u32 + pos + 12)), pos + 12);
153 }
154 for (; pos + 4 <= len; pos += 4) {
155 T(_mm_loadu_si128((__m128i *)(u32 + pos)), pos);
156 }
157#undef T
158 }
159
160 if (pos + 0 == len) return -1;
161 if (u32[pos + 0]) return pos + 0;
162 if (pos + 1 == len) return -1;
163 if (u32[pos + 1]) return pos + 1;
164 if (pos + 2 == len) return -1;
165 if (u32[pos + 2]) return pos + 2;
166 return -1;

169size_t count_non_zero8(const uint8_t u8[], size_t n)
170{
171 const __v16qi zero = { 0, };
172 __v2di acc = { 0, };
173
174 assert (n % 64 == 0);
175 assert ((uintptr_t)u8 % 16 == 0);
176 for (uint32_t i = 0; i < n; ) {
177 __v16qi acc0 = zero, acc1 = zero, acc2 = zero, acc3 = zero;
178
179 /* avoid overflows in acc0 + acc1 + acc2 + acc3, 63 * 4 < 256 */
180 for (uint32_t j = 0; j < 63 && i < n; j++, i += 64) {
181 acc0 -= __builtin_ia32_pcmpeqb128(*(__v16qi *)(u8 + i + 0), zero);
182 acc1 -= __builtin_ia32_pcmpeqb128(*(__v16qi *)(u8 + i + 16), zero);
183 acc2 -= __builtin_ia32_pcmpeqb128(*(__v16qi *)(u8 + i + 32), zero);
184 acc3 -= __builtin_ia32_pcmpeqb128(*(__v16qi *)(u8 + i + 48), zero);
185 }
186 /* psadbw(a0..a15, 0) -> (__v2di){ a0 + … + a7, a8 + … + a15 } */
187 acc += __builtin_ia32_psadbw128(acc0 + acc1 + acc2 + acc3, zero);
188 }
189 return n - sum_epi64(acc);
190}

193{

232static size_t count_non_zero64_naive(const uint64_t u64[], size_t n)
233{
234 register size_t acc0 = 0, acc1 = 0, acc2 = 0, acc3 = 0;
235
236 for (size_t i = 0; i < n; i += 4) {
237 acc0 += !u64[i + 0];
238 acc1 += !u64[i + 1];
239 acc2 += !u64[i + 2];
240 acc3 += !u64[i + 3];
241 }
242 return n - (acc0 + acc1 + acc2 + acc3);
243}

296size_t count_non_zero128(const void *_data, size_t n)
297{
298 const struct {
299 uint64_t h;
300 uint64_t l;

307 acc1 += is_128bits_zero(data + i + 1);
308 acc2 += is_128bits_zero(data + i + 2);
309 acc3 += is_128bits_zero(data + i + 3);
310 }
