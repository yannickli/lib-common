24/** \defgroup qkv__ll__wah Word Aligned Hybrid bitmaps.
25 * \ingroup qkv__ll
26 * \brief Word Aligned Hybrid bitmaps.
27 *
28 * \{
29 *
30 * WAH are a form of compressed bitmap highly compact in case the bitmap
31 * contains a lot of successive bits at the same value (1 or 0). In that case,
32 * the sequence is compressed as a single integer containing the number of
33 * words (32bits) of the sequence.
34 *
35 * \section format WAH format
36 *
37 * A WAH is a sequence of chunks each one composed of two parts:
38 * - a header of two words with:
39 * - a bit value (0 or 1)
40 * - the number M of successive words with that bit value (called a run)
41 * - the number N of following words that were not compressed
42 * - N uncompressed words.
43 *
44 * As a consequence, each chunk encodes M + N words in 2 + N words. This means
45 * the maximum overhead of the WAH is when M is zero, in which case, the
46 * overhead is 2 words (8 bytes). As a consequence, it is always at least as
47 * efficient as an uncompressed stored in term of memory usage.
48 *
49 * It's Hybrid because it contains both compressed and uncompressed data, and
50 * aligned since everything is done at the work level: the header only
51 * references a integral number of words of 32 bits.
52 *
53 * In memory, the chunks are stored in wah_word_t vectors called buckets.
54 * Each bucket contains \ref bit_wah_g.bits_in_bucket bits of the bitmap,
55 * except the last one that can be partially filled.
56 * This is done like that in order to avoid having too big vectors in memory.
57 *
58 * \section usage Use cases
59 *
60 * A WAH does not support efficient random accesses (reading / writing at a
61 * specific bit position) because the chunks encode a variable amount of words
62 * in a variable amount of memory. However, it efficiently support both
63 * sequential reading / writing.
64 *
65 * Bitwise operations are also supported but, with the exception of the
66 * negation operator, they are not in place (they always require either a
67 * brand new bitmap or a copy of one of the operands). Those operations are
68 * efficient since they can deal with long runs with a single word read.
69 */

207/*
208 * invariants for an enumerator not a WAH_ENUM_END:
209 * - current_word is non 0 and its last bit is set
210 * - if remain_bits <= WAH_BIT_IN_WORD we're consuming "current_word" else
211 * we're streaming ones.
212 * - remain_bits is the number of yet to stream bits including the current
213 * one.
214 */
215typedef struct wah_bit_enum_t {
216 wah_word_enum_t word_en;
217 uint64_t key;
218 uint64_t remain_bits;
219 uint32_t current_word;
220} wah_bit_enum_t;

222bool wah_bit_enum_scan_word(wah_bit_enum_t *en) __leaf;
223
224static ALWAYS_INLINE void wah_bit_enum_scan(wah_bit_enum_t *en)
225{
226 if (en->current_word == 0 && !wah_bit_enum_scan_word(en))
227 return;
228
229 assert (en->current_word);
230 if (en->remain_bits <= WAH_BIT_IN_WORD) {
231 uint32_t bit = bsf32(en->current_word);
232
233 assert (bit < en->remain_bits);
234 en->key += bit;
235 en->current_word >>= bit;
236 en->remain_bits -= bit;
237 }
238}
239
240static ALWAYS_INLINE void wah_bit_enum_next(wah_bit_enum_t *en)
241{
242 en->key++;
243 if (en->remain_bits <= WAH_BIT_IN_WORD)
244 en->current_word >>= 1;
245 en->remain_bits--;
246 wah_bit_enum_scan(en);
247}
