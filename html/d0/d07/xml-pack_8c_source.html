27static void
28xpack_value(sb_t *sb, const iop_struct_t *desc, const iop_field_t *f,
29 const void *v, unsigned flags)
30{
31 static lstr_t const types[] = {
32 [IOP_T_I8] = LSTR_IMMED(" xsi:type=\"xsd:byte\">"),
33 [IOP_T_U8] = LSTR_IMMED(" xsi:type=\"xsd:unsignedByte\">"),
34 [IOP_T_I16] = LSTR_IMMED(" xsi:type=\"xsd:short\">"),
35 [IOP_T_U16] = LSTR_IMMED(" xsi:type=\"xsd:unsignedShort\">"),
36 [IOP_T_I32] = LSTR_IMMED(" xsi:type=\"xsd:int\">"),
37 [IOP_T_ENUM] = LSTR_IMMED(" xsi:type=\"xsd:int\">"),
38 [IOP_T_U32] = LSTR_IMMED(" xsi:type=\"xsd:unsignedInt\">"),
39 [IOP_T_I64] = LSTR_IMMED(" xsi:type=\"xsd:long\">"),
40 [IOP_T_U64] = LSTR_IMMED(" xsi:type=\"xsd:unsignedLong\">"),
41 [IOP_T_BOOL] = LSTR_IMMED(" xsi:type=\"xsd:boolean\">"),
42 [IOP_T_DOUBLE] = LSTR_IMMED(" xsi:type=\"xsd:double\">"),
43 [IOP_T_STRING] = LSTR_IMMED(" xsi:type=\"xsd:string\">"),
44 [IOP_T_DATA] = LSTR_IMMED(" xsi:type=\"xsd:base64Binary\">"),
45 [IOP_T_XML] = LSTR_IMMED(">"),
46 [IOP_T_UNION] = LSTR_IMMED(">"),
47 [IOP_T_STRUCT] = LSTR_IMMED(">"),
48 [IOP_T_VOID] = LSTR_IMMED(" xsi:nil=\"true\">"),
49 };
50 const lstr_t *s;
51 const iop_field_attrs_t *attrs;
52 bool is_class = iop_field_is_class(f);
53 bool is_ref = iop_field_is_reference(f);
54
55 sb_grow(sb, 64 + f->name.len * 2);
56 sb_addc(sb, '<');
57 sb_add(sb, f->name.s, f->name.len);
58
59 if ((is_class || is_ref) && f->repeat != IOP_R_OPTIONAL) {
60 /* Non-optional reference fields have to be dereferenced
61 * (dereferencing of optional fields was already done by
62 * caller).
63 */
64 v = *(void **)v;
65 }
66 if (is_class) {
67 const iop_struct_t *real_desc = *(const iop_struct_t **)v;
68
69 /* If this assert fails, you are exporting private classes through
70 * a public interface... this is BAD!
71 */
72 assert (!real_desc->class_attrs->is_private
73 || !(flags & IOP_XPACK_SKIP_PRIVATE));
74
75 /* The "n" namespace is used here because it's the one used in
76 * ichttp_serialize_soap. */
77 sb_addf(sb, " xsi:type=\"n:%*pM\">",
78 LSTR_FMT_ARG(real_desc->fullname));
79 } else
80 if (((flags & IOP_XPACK_VERBOSE)
81 && !((flags & IOP_XPACK_LITERAL_ENUMS) && f->type == IOP_T_ENUM))
82 || f->type == IOP_T_VOID)
83 {
84 sb_add(sb, types[f->type].s, types[f->type].len);
85 } else {
86 sb_addc(sb, '>');
87 }
88
89 switch (f->type) {
90 double d;
91
92 case IOP_T_I8: sb_addf(sb, "%i", *( int8_t *)v); break;
93 case IOP_T_U8: sb_addf(sb, "%u", *( uint8_t *)v); break;
94 case IOP_T_I16: sb_addf(sb, "%i", *( int16_t *)v); break;
95 case IOP_T_U16: sb_addf(sb, "%u", *(uint16_t *)v); break;
96 case IOP_T_I32: sb_addf(sb, "%i", *( int32_t *)v); break;
97 case IOP_T_ENUM:
98 if (!(flags & IOP_XPACK_LITERAL_ENUMS)) {
99 sb_addf(sb, "%i", *( int32_t *)v);
100 } else {
101 lstr_t str = iop_enum_to_str_desc(f->u1.en_desc, *(int32_t *)v);
102 if (!str.s) {
103 sb_addf(sb, "%i", *( int32_t *)v);
104 } else {
105 sb_add_lstr(sb, str);
106 }
107 }
108 break;
109 case IOP_T_U32: sb_addf(sb, "%u", *(uint32_t *)v); break;
110 case IOP_T_I64: sb_addf(sb, "%jd", *( int64_t *)v); break;
111 case IOP_T_U64: sb_addf(sb, "%ju", *(uint64_t *)v); break;
112 case IOP_T_DOUBLE:
113 d = *(double *)v;
114 if (isinf(d)) {
115 sb_adds(sb, d < 0 ? "-INF" : "INF");
116 } else {
117 sb_addf(sb, "%.17g", d);
118 }
119 break;
120 case IOP_T_BOOL:
121 if (*(bool *)v) {
122 sb_addc(sb, '1');
123 } else {
124 sb_addc(sb, '0');
125 }
126 break;
127 case IOP_T_STRING:
128 s = v;
129 if (s->len) {
130 attrs = iop_field_get_attrs(desc, f);
131 if (attrs && TST_BIT(&attrs->flags, IOP_FIELD_CDATA)) {
132 sb_adds(sb, "<![CDATA[");
133 sb_add(sb, s->s, s->len);
134 sb_adds(sb, "]]>");
135 } else {
136 sb_add_lstr_xmlescape(sb, *s);
137 }
138 }
139 break;
140 case IOP_T_DATA:
141 s = v;
142 if (s->len)
143 sb_add_lstr_b64(sb, *s, -1);
144 break;
145 case IOP_T_XML:
146 s = v;
147 sb_add(sb, s->s, s->len);
148 break;
149 case IOP_T_UNION:
150 xpack_union(sb, f->u1.st_desc, v, flags);
151 break;
152 case IOP_T_VOID:
153 break;
154 case IOP_T_STRUCT:
155 if (is_class) {
156 xpack_class(sb, f->u1.st_desc, v, flags);
157 } else {
158 xpack_struct(sb, f->u1.st_desc, v, flags);
159 }
160 break;
161 }
162 sb_adds(sb, "</");
163 sb_add(sb, f->name.s, f->name.len);
164 sb_addc(sb, '>');
165}

167static void
168xpack_struct(sb_t *sb, const iop_struct_t *desc, const void *v,
169 unsigned flags)
170{
171 for (int i = 0; i < desc->fields_len; i++) {
172 const iop_field_t *f = desc->fields + i;
173 const void *ptr = (char *)v + f->data_offs;
174 int len = 1;
175
176 if (flags & IOP_XPACK_SKIP_PRIVATE) {
177 const iop_field_attrs_t *attrs = iop_field_get_attrs(desc, f);
178 if (attrs && TST_BIT(&attrs->flags, IOP_FIELD_PRIVATE))
179 continue;
180 }
181
182 if (f->repeat == IOP_R_OPTIONAL) {
183 if (!iop_opt_field_isset(f->type, ptr)) {
184 continue;
185 }
186 if ((1 << f->type) & IOP_STRUCTS_OK)
187 ptr = *(void **)ptr;
188 }
189 if (f->repeat == IOP_R_REPEATED) {
190 const lstr_t *data = ptr;
191 len = data->len;
192 ptr = data->data;
193 }
194
195 while (len-- > 0) {
196 if (!(f->type == IOP_T_VOID && f->repeat == IOP_R_REQUIRED)) {
197 xpack_value(sb, desc, f, ptr, flags);
198 }
199 ptr = (const char *)ptr + f->size;
200 }
201 }
202}

204static void xpack_class(sb_t *sb, const iop_struct_t *desc, const void *v,
205 unsigned flags)
206{
207 qv_t(iop_struct) parents;
208 const iop_struct_t *real_desc = *(const iop_struct_t **)v;
209
210 e_assert(panic, !real_desc->class_attrs->is_abstract,
211 "packing of abstract class '%*pM' is forbidden",
212 LSTR_FMT_ARG(real_desc->fullname));
213
214 /* We want to write the fields in the order "master -> children", and
215 * not "children -> master", so first build a qvector of the parents.
216 */
217 qv_inita(&parents, 8);
218 do {
219 qv_append(&parents, real_desc);
220 } while ((real_desc = real_desc->class_attrs->parent));
221
222 /* Write fields of different levels */
223 for (int pos = parents.len; pos-- > 0; ) {
224 xpack_struct(sb, parents.tab[pos], v, flags);
225 }
226 qv_wipe(&parents);
227}

229static void
230xpack_union(sb_t *sb, const iop_struct_t *desc, const void *v,

232{

238void iop_xpack_flags(sb_t *sb, const iop_struct_t *desc, const void *v,
239 unsigned flags)
240{
241 if (desc->is_union) {
242 xpack_union(sb, desc, v, flags);
243 } else
244 if (iop_struct_is_class(desc)) {
245 xpack_class(sb, desc, v, flags);
246 } else {
247 xpack_struct(sb, desc, v, flags);
248 }
249}
