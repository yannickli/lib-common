28/** XXX The backtrace() function calls an init() function which uses malloc()
29 * and leads to deadlock in the signals handler. So we always call backtrace()
30 * once outside of the signals hander. It's an horrible hack but it works.

33{
34 void *arr[256];

39static bool should_dump_maps(void)
40{
41 /* By default, memory maps are not dumped in .debug files, because it
42 * makes the files big and this is rarely used.
43 * It can be enabled with the IS_DEBUG_FILES_DUMP_MAPS environment
44 * variable. */
45 static int dump_maps = -1;
46
47 if (unlikely(dump_maps < 0)) {
48 const char *env = getenv("IS_DEBUG_FILES_DUMP_MAPS");
49
50 dump_maps = env && *env && atoi(env) > 0;
51 }
52
53 return dump_maps;
54}

58void ps_dump_backtrace(int signum, const char *prog, int fd, bool full)
59{
60 char buf[256];
61 void *arr[256];
62 int bt, n;
63
64 if (signum >= 0) {
65 n = snprintf(buf, sizeof(buf), "---> %s[%d] %s at %jd\n\n",
66 prog, getpid(), strsignal(signum), time(NULL));
67 } else {
68 n = snprintf(buf, sizeof(buf),
69 "---> %s[%d] expect violation at %jd\n\n",
70 prog, getpid(), time(NULL));
71 }
72 if (xwrite(fd, buf, n) < 0) {
73 return;
74 }
75
76 if (debug_stack_has_frames()) {
77 XWRITE("WARNING: additional user context available at the end of the "
78 "file\n\n");
79 }
80
81 bt = backtrace(arr, countof(arr));
82 backtrace_symbols_fd(arr, bt, fd);
83 fsync(fd);
84
85 if (full && should_dump_maps()) {
86 int maps_fd = open("/proc/self/smaps", O_RDONLY);
87
88 if (maps_fd != -1) {
89 XWRITE("\n--- Memory maps:\n\n");
90 for (;;) {
91 n = read(maps_fd, buf, sizeof(buf));
92 if (n < 0) {
93 if (ERR_RW_RETRIABLE(errno)) {
94 continue;
95 }
96 break;
97 }
98 if (n == 0) {
99 break;
100 }
101 if (xwrite(fd, buf, n) < 0) {
102 break;
103 }
104 }
105 close(maps_fd);
106 }
107 } else {
108 XWRITE("\n");
109 }
110}

140void ps_write_backtrace(int signum, bool allow_fork)
141{
142 const char *debug_dir = getenv("IS_DEBUG_FILES_DIR");
143 char path[PATH_MAX];
144 int fd;
145 int saved_errno = errno;
146
147 if (!debug_dir || !*debug_dir) {
148 debug_dir = "/tmp";
149 }
150
151 snprintf(path, sizeof(path), "%s/%s.%d.%ld.debug",
152 debug_dir, program_invocation_short_name, (uint32_t)time(NULL),
153 (long)getpid());
154 fd = open(path, O_EXCL | O_CREAT | O_WRONLY | O_TRUNC, 0600);
155
156 if (fd >= 0) {
157 int main_versions_printed = 0;
158 char buf[256];
159
160 for (int i = 0; i < core_versions_nb_g; i++) {
161 const core_version_t *version = &core_versions_g[i];
162
163 if (version->is_main_version) {
164 ps_panic_sighandler_print_version(fd, version);
165 main_versions_printed++;
166 }
167 }
168 if (main_versions_printed > 0) {
169 XWRITE("\n");
170 }
171 for (int i = 0; i < core_versions_nb_g; i++) {
172 const core_version_t *version = &core_versions_g[i];
173
174 if (!version->is_main_version) {
175 ps_panic_sighandler_print_version(fd, version);
176 }
177 }
178 XWRITE("\n");
179
180 snprintf(buf, sizeof(buf), "\n--- errno: %s (%d)\n",
181 strerror(saved_errno), saved_errno);
182 XWRITE(buf);
183
184 errno = saved_errno;
185 ps_dump_backtrace(signum, program_invocation_short_name, fd, true);
186 p_close(&fd);
187
188 if (allow_fork) {
189 ps_print_file(path,
190 "echo '\n--- File descriptors (using ls):\n'");
191 ps_print_file(path, "ls -al /proc/self/fd");
192
193 ps_print_file(path,
194 "echo '\n--- File descriptors (using lsof):\n'");
195 ps_print_file(path, "lsof -p %d", getpid());
196 }
197 }
198#ifndef NDEBUG
199 errno = saved_errno;
200 ps_dump_backtrace(signum, program_invocation_short_name,
201 STDERR_FILENO, false);
202#endif
203 errno = saved_errno;
204
205 _debug_stack_print(path);
206 errno = saved_errno;
207}
