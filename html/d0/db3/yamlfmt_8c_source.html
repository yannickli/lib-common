65static qh_t(lstr) *
66t_get_repacked_yaml_pres_subfiles(const char *main_file,
67 const yaml__document_presentation__t *pres)
68{
69 qv_t(lstr) yaml_subfiles;
70 qh_t(lstr) *res = t_qh_new(lstr, 16);
71
72 t_qv_init(&yaml_subfiles, 16);
73 t_yaml_pres_get_flat_subfiles(pres, &yaml_subfiles);
74
75 res = t_qh_new(lstr, yaml_subfiles.len);
76
77 if (yaml_subfiles.len) {
78 char main_file_dir[PATH_MAX];
79
80 path_dirname(main_file_dir, sizeof(main_file_dir), main_file);
81
82 tab_for_each_entry(subfile, &yaml_subfiles) {
83 char subfile_path[PATH_MAX];
84 lstr_t subfile_lstr;
85
86 path_extend(subfile_path, main_file_dir, "%*pM",
87 LSTR_FMT_ARG(subfile));
88 path_simplify(subfile_path);
89
90 subfile_lstr = t_lstr_dups(subfile_path, -1);
91 qh_add(lstr, res, &subfile_lstr);
92 }
93 }
94
95 return res;
96}

98static int
99t_parse_yaml(yaml_parse_t *env, const iop_dso_t * nullable dso,
100 const iop_struct_t * nullable st, yaml_data_t * nonnull data,
101 sb_t * nonnull err)
102{
103 RETHROW(t_yaml_parse(env, data, err));
104
105 if (dso && !st) {
106 if (!data->tag.s) {
107 sb_setf(err, "document should start with a tag equals to the "
108 "fullname of the IOP type serialized");
109 return -1;
110 }
111 st = RETHROW_PN(get_iop_type(dso, data->tag, err));
112 }
113
114 if (st) {
115 void *out = NULL;
116
117 RETHROW(t_iop_yunpack_ptr_yaml_data(data, st, &out, 0, err));
118 }
119
120 return 0;
121}

123static int
124pack_yaml(yaml_data_t * nonnull data,
125 const yaml__document_presentation__t * nullable pres,
126 sb_t * nonnull err)
127{
128 t_scope;
129 yaml_pack_env_t *pack_env;
130 int res = 0;
131 unsigned flags = YAML_PACK_ALLOW_UNBOUND_VARIABLES;
132
133 pack_env = t_yaml_pack_env_new();
134 if (opts_g.raw_mode) {
135 /* use an empty document presentation to prevent any presentation
136 * data stored in the AST from being used. */
137 yaml__document_presentation__t empty_pres;
138
139 iop_init(yaml__document_presentation, &empty_pres);
140 t_yaml_pack_env_set_presentation(pack_env, &empty_pres);
141 } else
142 if (pres) {
143 t_yaml_pack_env_set_presentation(pack_env, pres);
144 }
145
146 if (opts_g.output_path) {
147 yaml_pack_env_set_flags(pack_env, flags);
148 res = t_yaml_pack_file(pack_env, opts_g.output_path, data, err);
149 } else {
150 flags |= YAML_PACK_NO_SUBFILES;
151 yaml_pack_env_set_flags(pack_env, flags);
152 res = t_yaml_pack(pack_env, data, yaml_pack_write_stdout, NULL, err);
153 printf("\n");
154 }
155
156 return res;
157}

159static int
160repack_yaml(const char * nullable filename, const iop_dso_t * nullable dso,
161 const iop_struct_t * nullable st, sb_t * nonnull err)
162{
163 t_scope;
164 yaml_parse_t *env;
165 yaml_data_t data;
166 lstr_t file = LSTR_NULL_V;
167 int res = 0;
168 unsigned flags = YAML_PARSE_GEN_PRES_DATA;
169
170 if (!dso && !st) {
171 /* no IOP validation, so variables can be unbound */
172 flags |= YAML_PARSE_ALLOW_UNBOUND_VARIABLES;
173 }
174
175 env = t_yaml_parse_new(flags);
176 if (filename) {
177 if (t_yaml_parse_attach_file(env, filename, NULL, err) < 0) {
178 res = -1;
179 goto end;
180 }
181 } else {
182 if (lstr_init_from_fd(&file, 0, PROT_READ, MAP_SHARED) < 0) {
183 sb_setf(err, "cannot read from stdin: %m");
184 res = -1;
185 goto end;
186 }
187 yaml_parse_attach_ps(env, ps_initlstr(&file));
188 }
189
190 if (t_parse_yaml(env, dso, st, &data, err) < 0) {
191 res = -1;
192 goto end;
193 }
194
195 res = pack_yaml(&data, NULL, err);
196
197 end:
198 lstr_wipe(&file);
199 yaml_parse_delete(&env);
200 return res;
201}

203static int
204repack_json(const char * nullable filename, const iop_struct_t * nonnull st,
205 sb_t * nonnull err)
206{
207 t_scope;
208 lstr_t file = LSTR_NULL_V;
209 void *value = NULL;
210 yaml__document_presentation__t *pres;
211 qv_t(iop_json_subfile) subfiles;
212 iop_json_subfile__array_t subfiles_array;
213 yaml_data_t data;
214 int res = 0;
215
216 t_qv_init(&subfiles, 0);
217
218 /* Unpack json */
219 if (filename) {
220 RETHROW(t_iop_junpack_ptr_file(filename, st, &value, 0, &subfiles,
221 err));
222 } else {
223 pstream_t ps;
224
225 if (lstr_init_from_fd(&file, 0, PROT_READ, MAP_SHARED) < 0) {
226 sb_setf(err, "cannot read from stdin: %m");
227 res = -1;
228 goto end;
229 }
230 ps = ps_initlstr(&file);
231 if (t_iop_junpack_ptr_ps(&ps, st, &value, 0, err) < 0) {
232 res = -1;
233 goto end;
234 }
235 }
236
237 /* Pack yaml */
238 subfiles_array = IOP_TYPED_ARRAY_TAB(iop_json_subfile, &subfiles);
239 pres = t_build_yaml_pres_from_json_subfiles(&subfiles_array, st, value);
240
241 t_iop_to_yaml_data(st, value, &data);
242 res = pack_yaml(&data, pres, err);
243 if (res < 0) {
244 goto end;
245 }
246
247 /* Delete input files? */
248 if (filename && opts_g.output_path && opts_g.delete_input_files) {
249 char json_dir[PATH_MAX];
250 qh_t(lstr) *yaml_subfiles;
251
252 path_dirname(json_dir, sizeof(json_dir), filename);
253 yaml_subfiles = t_get_repacked_yaml_pres_subfiles(opts_g.output_path,
254 pres);
255
256 /* Delete the main json file */
257 delete_input_file(filename);
258
259 /* Delete json subfiles, but only if they are not also yaml
260 * subfiles */
261 tab_for_each_ptr(subfile, &subfiles_array) {
262 char subfile_path[PATH_MAX];
263 lstr_t subfile_lstr;
264
265 snprintf(subfile_path, sizeof(subfile_path), "%s/%*pM",
266 json_dir, LSTR_FMT_ARG(subfile->file_path));
267 path_simplify(subfile_path);
268
269 subfile_lstr = LSTR(subfile_path);
270 if (qh_find(lstr, yaml_subfiles, &subfile_lstr) < 0) {
271 delete_input_file(subfile_path);
272 }
273 }
274 }
275
276 end:
277 lstr_wipe(&file);
278 return res;
279}

281static int
282parse_and_repack(const char * nullable filename,
283 const iop_dso_t * nullable dso, sb_t * nonnull err)
284{
285 const iop_struct_t *st = NULL;
286
287 if (dso && opts_g.type_name) {
288 st = RETHROW_PN(get_iop_type(dso, LSTR(opts_g.type_name), err));
289 }
290
291 if (opts_g.json_input) {
292 return repack_json(filename, st, err);
293 } else {
294 return repack_yaml(filename, dso, st, err);
295 }
296}

346int main(int argc, char **argv)
347{
348 iop_dso_t *dso = NULL;
349 const char *filename = NULL;
350 const char *arg0;
351 popt_t options[] = {
352 OPT_FLAG('h', "help", &opts_g.help, "show help"),
353 OPT_STR('d', "dso", &opts_g.dso_path, "Path to IOP dso file"),
354 OPT_FLAG('j', "json", &opts_g.json_input, "Unpack the input as JSON"),
355 OPT_STR('t', "type", &opts_g.type_name, "Name of the IOP type"),
356 OPT_STR('o', "output", &opts_g.output_path,
357 "Path to the output file"),
358 OPT_FLAG('r', "raw", &opts_g.raw_mode,
359 "Format without any presentation details."),
360 OPT_FLAG(0, "delete", &opts_g.delete_input_files,
361 "Delete input files (if in JSON format only)."),
362 OPT_END(),
363 };
364 SB_1k(err);
365 int ret = EXIT_SUCCESS;
366
367 arg0 = NEXTARG(argc, argv);
368 argc = parseopt(argc, argv, options, 0);
369 if (argc < 0 || opts_g.help) {
370 makeusage(!opts_g.help, arg0, "[<file>]", description, options);
371 }
372
373 if (argc >= 1) {
374 filename = NEXTARG(argc, argv);
375 }
376
377 if (opts_g.json_input && (!opts_g.dso_path || !opts_g.type_name)) {
378 fprintf(stderr, "both `-d` and `-t`` are required with JSON input\n");
379 return EXIT_FAILURE;
380 }
381
382 if (opts_g.dso_path) {
383 dso = iop_dso_open(opts_g.dso_path, LM_ID_BASE, &err);
384 if (!dso) {
385 fprintf(stderr, "cannot open dso `%s`: %pL\n", opts_g.dso_path,
386 &err);
387 return EXIT_FAILURE;
388 }
389 }
390
391 if (parse_and_repack(filename, dso, &err) < 0) {
392 fprintf(stderr, "%.*s\n", err.len, err.data);
393 ret = EXIT_FAILURE;
394 }
395
396 iop_dso_close(&dso);
397 return ret;
398}
