26/** Convert IOP-XML to an IOP C structure.
27 *
28 * This function unpacks an IOP structure encoded in XML format. You have to
29 * provide yourself a xml_reader_t setup on the XML data to unpack.
30 *
31 * iop_xunpack() assumes that the provided xml_reader_t is given with the
32 * following flags:
33 * XML_PARSE_NOENT | XML_PARSE_NOBLANKS | XML_PARSE_NONET
34 * | XML_PARSE_NOCDATA
35 *
36 * Concerning the unpacking flags, the XML unpacker supports the following
37 * ones: IOP_UNPACK_IGNORE_UNKNOWN.
38 *
39 * This function cannot be used to unpack a class; use `iop_xunpack_ptr_flags`
40 * instead.
41 *
42 * \param[in] xp The xml_reader_t setup on the XML data (see xmlr.h).
43 * \param[in] mp Memory pool to use for memory allocations.
44 * \param[in] st The IOP structure description.
45 * \param[in] out Pointer on the IOP structure to write.
46 * \param[in] flags Bitfield of flags to use (see iop_unpack_flags in iop.h).
47 */
48__must_check__
49int iop_xunpack_flags(void * nonnull xp, mem_pool_t * nonnull mp,
50 const iop_struct_t * nonnull st, void * nonnull out,
51 int flags);
52
53/** Convert IOP-XML to an IOP C structure.
54 *
55 * This function acts exactly as `iop_xunpack_flags` but allocates (or
56 * reallocates) the destination structure.
57 *
58 * This function MUST be used to unpack a class instead of
59 * `iop_xunpack_flags`, because the size of a class is not known before
60 * unpacking it (this could be a child).
61 *
62 */
63__must_check__
64int iop_xunpack_ptr_flags(void * nonnull xp, mem_pool_t * nonnull mp,
65 const iop_struct_t * nonnull st,
66 void * nullable * nonnull out, int flags);

94/** Convert IOP-XML to an IOP C structure with XML parts support.

96 * This function just works as iop_xunpack_flags() but supports XML parts

104 * \param[in] mp Memory pool to use for memory allocations.
105 * \param[in] st The IOP structure description.
106 * \param[in] out Pointer on the IOP structure to write.
107 * \param[in] flags Bitfield of flags to use (see iop_unpack_flags in iop.h).
108 * \param[in] parts Hashtable to retrieve XML parts.
