58/** Load a DSO from a file, and register its packages.
59 *
60 * The DSO is opened with dlmopen(3) with the following flags:
61 * - RTLD_LAZY | RTLD_GLOBAL | RTLD_DEEPBIND when lmid is LM_ID_BASE. This is
62 * equivalent of calling dlopen(3) with the same flags.
63 * - RTLD_LAZY | RTLD_DEEPBIND when lmid is LM_ID_NEWLM or an already
64 * existing namespace.
65 *
66 * Due to a bug in glibc < 2.24, we are not able to call dlmopen(3) with
67 * RTLD_GLOBAL. This means that the DSO creating a new namespace must contain
68 * all the symbols needed by the other DSOs that will use that namespace.
69 * See http://man7.org/linux/man-pages/man3/dlopen.3.html#BUGS
70 *
71 * \param[in] path path to the DSO.
72 * \param[in] lmid lmid argument passed to dlmopen(3).
73 * \param[out] err error description in case of error.
74 */
75iop_dso_t * nullable iop_dso_open(const char * nonnull path,
76 Lmid_t lmid, sb_t * nonnull err);
77
78/** Load a DSO from an already opened DSO handle, and register its packages.
79 *
80 * On success, the DSO will own the handle afterwards.
81 * On error, the handle is not closed.
82 *
83 * \param[in] handle handle to the opened DSO.
84 * \param[in] path path to the DSO.
85 * \param[in] lmid lmid argument used to dlmopen(3) the DSO.
86 * \param[out] err error description in case of error.
87 */
88iop_dso_t * nullable iop_dso_load_handle(void * nonnull handle,
89 const char * nonnull path,
90 Lmid_t lmid, sb_t * nonnull err);
