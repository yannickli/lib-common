75static void
76get_configurations_recursive(logger_t *logger, lstr_t prefix,
77 qv_t(logger_conf) *res)
78{
79 core__logger_configuration__t conf;
80
81 /* called first as it can force the update of several parameters
82 * including the full name (calling __logger_refresh) */
83 iop_init(core__logger_configuration, &conf);
84
85 /* Don't use logger_get_level since it takes the update_lock */
86 __logger_do_refresh(logger);
87 conf.level = MAX(logger->level, LOG_CRIT);
88
89 /* check if the first element in the full name is the prefix */
90 if (lstr_startswith(logger->full_name, prefix)) {
91 conf.full_name = lstr_dupc(logger->full_name);
92 conf.force_all = logger->level_flags & LOG_FORCED;
93 conf.is_silent = logger->level_flags & LOG_SILENT;
94
95 qv_append(res, conf);
96
97 /* all children will have the same prefix. No need to check it
98 * anymore, we set it to null */
99 prefix = LSTR_NULL_V;
100 }
101
102 dlist_for_each_entry(logger_t, child, &logger->children, siblings) {
103 get_configurations_recursive(child, prefix, res);
104 }
105}

107void logger_get_all_configurations(lstr_t prefix, qv_t(logger_conf) *confs)
108{
109 log_spin_lock();
110 get_configurations_recursive(logger_get_root(), prefix, confs);
111 log_spin_unlock();
112}
