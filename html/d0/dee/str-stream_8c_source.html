22int ps_copyv(pstream_t *ps, struct iovec *iov, size_t *iov_len, int *flags)
23{
24 int orig_len = ps_len(ps);
25 size_t i;
26
27 for (i = 0; !ps_done(ps) && i < *iov_len; i++) {
28 if (iov[i].iov_len > ps_len(ps))
29 iov[i].iov_len = ps_len(ps);
30 memcpy(iov[i].iov_base, ps->b, iov[i].iov_len);
31 ps_skip(ps, iov[i].iov_len);
32 }
33 *iov_len = i;
34
35 if (flags) {
36 if (ps_done(ps)) {
37 *flags &= ~MSG_TRUNC;
38 return orig_len;
39 }
40 if (*flags & MSG_TRUNC)
41 return orig_len;
42 *flags |= MSG_TRUNC;
43 }
44 return orig_len - ps_len(ps);
45}

47static int ps_get_csv_quoted_field(mem_pool_t *mp, pstream_t *ps, int quote,
48 qv_t(lstr) *fields)
49{
50 SB_8k(sb);
51
52 __ps_skip(ps, 1);
53
54 for (;;) {
55 pstream_t part;
56
57 PS_CHECK(ps_get_ps_chr_and_skip(ps, quote, &part));
58 if (!ps_done(ps) && *ps->s == quote) {
59 __ps_skip(ps, 1);
60 sb_add(&sb, part.s, ps_len(&part) + 1);
61 } else
62 if (sb.len == 0) {
63 qv_append(fields, LSTR_PS_V(&part));
64 return 0;
65 } else {
66 sb_add(&sb, part.s, ps_len(&part));
67
68 if (mp) {
69 qv_append(fields, mp_lstr_dups(mp, sb.data, sb.len));
70 } else {
71 lstr_t dst = LSTR_NULL;
72
73 lstr_transfer_sb(&dst, &sb, false);
74 qv_append(fields, dst);
75 }
76 return 0;
77 }
78 }
79
80 return 0;
81}

83int ps_get_csv_line(mem_pool_t *mp, pstream_t *ps, int sep, int quote,
84 qv_t(lstr) *fields, pstream_t *out_line)
85{
86 ctype_desc_t cdesc;
87 char cdesc_tok[] = { '\r', '\n', sep, '\0' };
88 pstream_t out = *ps;
89
90 ctype_desc_build(&cdesc, cdesc_tok);
91
92 if (!out_line) {
93 out_line = &out;
94 }
95
96 if (ps_done(ps)) {
97 *out_line = ps_init(NULL, 0);
98 return 0;
99 }
100
101 for (;;) {
102 if (ps_done(ps)) {
103 qv_append(fields, LSTR_NULL_V);
104 *out_line = ps_initptr(out.s, ps->s);
105 return 0;
106 } else
107 if (*ps->s == quote) {
108 PS_CHECK(ps_get_csv_quoted_field(mp, ps, quote, fields));
109 } else {
110 pstream_t field = ps_get_cspan(ps, &cdesc);
111
112 if (!ps_len(&field)) {
113 qv_append(fields, LSTR_NULL_V);
114 } else {
115 qv_append(fields, LSTR_PS_V(&field));
116 }
117 }
118
119 switch (ps_getc(ps)) {
120 case '\r':
121 *out_line = ps_initptr(out.s, ps->s - 1);
122 return ps_skipc(ps, '\n');
123
124 case '\n':
125 *out_line = ps_initptr(out.s, ps->s - 1);
126 return 0;
127
128 case EOF:
129 *out_line = ps_initptr(out.s, ps->s);
130 return 0;
131
132 default:
133 PS_WANT(ps->s[-1] == sep);
134 break;
135 }
136 }
137
138 return 0;
139}

141void ps_split(pstream_t ps, const ctype_desc_t *sep, unsigned flags,
142 qv_t(lstr) *res)
143{
144 if (flags & PS_SPLIT_SKIP_EMPTY) {
145 ps_skip_span(&ps, sep);
146 }
147 while (!ps_done(&ps)) {
148 pstream_t n = ps_get_cspan(&ps, sep);
149
150 qv_append(res, LSTR_PS_V(&n));
151 if (flags & PS_SPLIT_SKIP_EMPTY) {
152 ps_skip_span(&ps, sep);
153 } else {
154 ps_skip(&ps, 1);
155 }
156 }
157}

159void ps_split_escaped(mem_pool_t *nullable mp, pstream_t ps,
160 const ctype_desc_t *nonnull sep, const char escape,
161 unsigned flags, qv_t(lstr) *res)
162{
163 SB_1k(sb);
164 pstream_t tmp;
165 ctype_desc_t sep_esc;
166 ctype_desc_t esc;
167
168 if (escape) {
169 ctype_desc_build2(&esc, &escape, 1);
170 ctype_desc_combine(&sep_esc, sep, &esc);
171 } else {
172 sep_esc = *sep;
173 }
174
175 if (flags & PS_SPLIT_SKIP_EMPTY) {
176 ps_skip_span(&ps, sep);
177 }
178
179 while (!ps_done(&ps)) {
180 tmp = ps_get_cspan(&ps, &sep_esc);
181 sb_add(&sb, tmp.s, tmp.s_end - tmp.s);
182
183 if (ctype_desc_contains(sep, *ps.s)) {
184 if (mp) {
185 qv_append(res, mp_lstr_dup(mp, LSTR_SB_V(&sb)));
186 } else {
187 lstr_t dst = LSTR_NULL;
188
189 lstr_transfer_sb(&dst, &sb, false);
190 qv_append(res, dst);
191 }
192 sb_reset(&sb);
193
194 if (flags & PS_SPLIT_SKIP_EMPTY) {
195 ps_skip_span(&ps, sep);
196 } else {
197 ps_skip(&ps, 1);
198 }
199 } else
200 if (escape && escape == *ps.s) {
201 /* Check if next character is a separator
202 * An escape character is removed if followed by an other escape
203 * character or a separator.
204 */
205 if (ps_has(&ps, 2)
206 && ctype_desc_contains(&sep_esc, ps.s[1]))
207 {
208 /* add next character, skip escape character */
209 sb_add(&sb, ps.s + 1, 1);
210 ps_skip(&ps, 2);
211 } else {
212 /* add escape character only */
213 sb_add(&sb, ps.s, 1);
214 ps_skip(&ps, 1);
215 }
216 }
217 }
218
219 if (!(flags & PS_SPLIT_SKIP_EMPTY) /* last character is a separator */
220 || (sb.len != 0)) /* last character is an escape */
221 {
222 qv_append(res, t_lstr_dup(LSTR_SB_V(&sb)));
223 }
224}
