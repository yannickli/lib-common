123#define do_loc_(fmt, level, t, loc, ...) \
124 do { \
125 const typeof(loc) *__loc = &(loc); \
126 \
127 if (__loc->file) { \
128 logger_log(&iopc_g.logger, level, "%s:%d:%d: %s: "fmt, \
129 get_print_path(__loc->file), \
130 __loc->lmin, __loc->cmin, (t), ##__VA_ARGS__); \
131 } else { \
132 logger_log(&iopc_g.logger, level, "%s: "fmt, (t), \
133 ##__VA_ARGS__); \
134 } \
135 } while (0)
136
137#define do_loc(fmt, level, t, loc, ...) \
138 do { \
139 do_loc_(fmt, level, t, loc, ##__VA_ARGS__); \
140 for (int i_ = 0; i_ < iopc_g.loc_stack.len; i_++) { \
141 iopc_loc_t loc_ = iopc_g.loc_stack.tab[i_]; \
142 do_loc_("%s", level, " from", loc_, loc_.comment); \
143 } \
144 } while (0)
145
146#define t_push_loc(loc, ...) \
147 do { \
148 qv_append(&iopc_g.loc_stack, loc); \
149 tab_last(&iopc_g.loc_stack)->comment = t_fmt(__VA_ARGS__); \
150 } while (0)
151
152#define pop_loc() qv_shrink(&iopc_g.loc_stack, 1)
153#define clear_loc() qv_clear(&iopc_g.loc_stack)
