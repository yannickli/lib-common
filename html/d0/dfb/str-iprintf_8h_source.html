74static inline char * nonnull ivasprintf(const char * nonnull fmt, va_list ap)
75
76{
77 char buf[BUFSIZ], *s;
78 int len;
79 va_list ap2;
80
81 va_copy(ap2, ap);
82 len = ivsnprintf(buf, ssizeof(buf), fmt, ap2);
83 va_end(ap2);
84
85 if (len < ssizeof(buf))
86 return (char *)p_dupz(buf, len);
87
88 ivsnprintf(s = p_new(char, len + 1), len + 1, fmt, ap);
89 return s;
90}

107/** Formatter function type.
108 *
109 * A formatter is called when, for a formatter f, the format %*pf is
110 * encountered in the format string. The formatter is then called with the
111 * parameters passed as argument of that particular format as well as the
112 * modifier.
113 *
114 * A formatter can either write in a stream or in a buffer, in both cases, it
115 * must return the size of the formatted object if it had enough place to
116 * write in it. The function must check if the output is supposed to be a
117 * stream or a buffer based on the presence of a stream.
118 *
119 * The buffer may not be large enough to write the received data, and the
120 * formatter is not responsible for the appending of a terminating zero.
121 *
122 * \param[in] modifier The modifier that triggered the call
123 * \param[in] val The pointer received as parameter.
124 * \param[in] val_len The length received as parameter.
125 * \param[out] stream If the formatter is expected to write on a stream,
126 * this is the destination stream.
127 * \param[out] buf If the formatter is expected to write in a buffer,
128 * this is the destination buffer.
129 * \param[out] buf_len The size of the buffer.
130 *
131 * \return The size required to format the \p val (not including the trailing
132 * zero) or -1 in case of error.
133 */
134typedef ssize_t
135(formatter_f)(int modifier, const void * null_unspecified val, size_t val_len,
136 FILE * nullable stream, char * nullable buf, size_t buf_len);

141 * character).
142 */

147/** Pointer Formatter function type.
148 *
149 * A pointer formatter is called when, for a formatter f, the format %pf is
150 * encountered in the format string. The formatter is then called with the
151 * parameters passed as argument of that particular format as well as the
152 * modifier.
153 *
154 * see \p formatter_f description for more details.
155 *
156 * \param[in] modifier The modifier that triggered the call
157 * \param[in] val The pointer received as parameter.
158 * \param[out] stream If the formatter is expected to write on a stream,
159 * this is the destination stream.
160 * \param[out] buf If the formatter is expected to write in a buffer,
161 * this is the destination buffer.
162 * \param[out] buf_len The size of the buffer.
163 *
164 * \return The size required to format the \p val (not including the trailing
165 * zero) or -1 in case of error.
