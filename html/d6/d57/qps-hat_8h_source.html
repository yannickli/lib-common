26/** \defgroup qkv__ll__hat QPS HAT-Trie
27 * \ingroup qkv__ll
28 * \brief QPS HAT-Trie
29 *
30 * \{
31 *
32 * QHAT-Trie is a a HAT-Trie implementation built on the top of \ref qps. It
33 * provides an ordered mapping between fixed-length keys (32bits) and
34 * fixed-length values (<= 128 bits).
35 *
36 * \section nullables Nullable vs Non-nullable tries
37 *
38 * Hat tries can be either nullable or non-nullable. On non-nullable tries,
39 * the initial value for all the entries is 0 and can never be NULL,
40 * while it is NULL on nullable trie. That means that on a non-nullable trie
41 * you cannot both storing 0 and know the exact list of keys you explicitly
42 * set.
43 *
44 * For both nullable and non-nullable tries, 0 is a special value and if you
45 * want to store a 0 in the trie, you must use the function \ref qhat_set0 (or
46 * its alias \ref qhat_set0_path).
47 *
48 * Here are a more in-depth description of the API depending on the type of
49 * the trie:
50 *
51 * \subsection nonnullable Non-nullable tries
52 *
53 * * \ref qhat_get: return indifferently NULL or a pointer to 0 if the value
54 * associated with the key is 0.
55 * * \ref qhat_set: allocate a slot where to store the value.
56 * * \ref qhat_set0 and \ref qhat_remove are synonyms and deallocate the slot
57 * associated with the given key.
58 *
59 * \subsection nullable Nullable tries
60 * * \ref qhat_get: returns NULL if the key is marked as set, else return a
61 * pointer to the value.
62 * * \ref qhat_set: mark the key as set and allocate a slot where to store its
63 * value.
64 * * \ref qhat_set0: mark the key as set and deallocate the slot associated
65 * with the key.
66 * * \ref qhat_remove: mark the key as unset and deallocate the slot
67 * associated with the key.
68 *
69 *
70 * \section internal In depth: storing 0.
71 *
72 * This section explains why you must not use \ref qhat_set and \ref
73 * qhat_set_path in order to store 0s in the trie. Note that this section use
74 * a non-nullable trie for the sake of simplicity, but you should consider
75 * this information to also apply to nullable tries.
76 *
77 * 0s are the default value of flat nodes. This means that whenever a compact
78 * node is flatten, all the holes between the key defined in the compact node
79 * are filled with 0s. Thus, if you allocated slots in the compact node to
80 * store some 0s, you 0s become undistinguishable from those filling the holes
81 * between the keys.
82 *
83 * In this situation, calling \ref qhat_get on any key that falls in the new
84 * flat node will return a non-null pointer (on non-nullable tries, qhat_get
85 * returns NULL when there's either no leaf associated with the key or if it
86 * falls on a missing key of a compact node).
87 *
88 * Now, suppose you remove one key from that flat node. This will trigger the
89 * optimizer that will check that there's enough non-zeros values in the flat
90 * node to remain flat and thus it will be unflattened. The unflattener just
91 * scan the flat node for keys with non-zero values. So what will happen is
92 * that you'll have a new compact node that will not contain the key you
93 * explicitly set to 0 using \ref qhat_set (or \ref qhat_set0) and didn't
94 * explicitly removed. That means that using \ref qhat_get() on those key will
95 * return NULL.
96 *
97 * So, what happened: you wanted to store 0s while this is the default value
98 * and finally you get NULL, so you lose one information: "did I explicitly
99 * set the value or not". That's why you have to use a *nullable* trie if you
100 * want to have a distinction between stored 0 and not-set values.
101 *
102 * What also happened is that you allocated a useless slot in the compact node
103 * to store the default value. This is a waste of memory since this didn't add
104 * any bit of information. That's why you must use \ref qhat_set0 (or \ref
105 * qhat_set0_path) when you want to store a 0 in the trie and by no mean use
106 * \ref qhat_set (or \ref qhat_set_path).
107 */

266/** Return 'false' if modifications in the QHAT made the path out of sync.
267 *
268 * In that case, the QHAT library will refresh it before use.
269 * This function is exposed for debugging and testing purpose.

271static inline bool qhat_path_is_sync(const qhat_path_t *path)

273 return path->gen.s.struct_gen == path->hat->gen.s.struct_gen;

276/** Increment the write access counter.
277 *
278 * Should be called after any write access in the QHAT, even if it had no
279 * effect. This allows us to detect when an unsafe access should have been
280 * safe: the problem is detected even when the QHAT struct was unchanged or
281 * when the access had no effect in the QHAT.
282 */
283static ALWAYS_INLINE void qhat_path_touch(qhat_path_t *path)
284{
285#ifndef NDEBUG
286 path->hat->gen.s.write_access_gen++;
287 path->gen = path->hat->gen;
288#endif /* NDEBUG */
289}

291/** Remove the value described by a path.
292 *
293 * \param[in,out] path Already resolved path to the key in the trie.
294 * \param[out] ptr If specified, filled with the erased value (or with 0 if
295 * the key was not present in the trie)
296 * \return true if the key was present before removal.
297 */
298static ALWAYS_INLINE
299bool qhat_remove_path(qhat_path_t *path, void *ptr)
300{
301 bool removed;
302
303 removed = (*path->hat->desc->removef)(path, ptr);
304 qhat_path_touch(path);
305 return removed;
306}

321/** Get a read-write pointer to the value described by a path.
322 *
323 * This function takes an already resolved path to a key in a hat, allocates a
324 * slot if necessary and returns a pointer to this slot. It also updates the
325 * specified path according to the changes in the structure of the trie
326 * implied by the allocation of the slot.
327 *
328 * \warning The path returned by this function is invalidated as soon as the
329 * structure of the trie changes. Thus don't cache this path after a set or a
330 * remove.
331 *
332 * \warning You must not set the returned slot to 0, if you wish to store 0s
333 * in the trie, use \ref qhat_set0 or \ref qhat_set0_path.
334 *
335 * \param[in,out] path Already resolved path to the key in the trie.
336 * \return A pointer to the slot associated to the key in the trie.
337 */
338static ALWAYS_INLINE
339void *qhat_set_path(qhat_path_t *path)
340{
341 void *value_ptr;
342
343 value_ptr = (*path->hat->desc->setf)(path);
344 qhat_path_touch(path);
345 return value_ptr;
346}

415/** Check if an entry is NULL.
416 */
417static ALWAYS_INLINE
418bool qhat_is_null(qhat_t *hat, uint32_t key)
419{
420 /* Check if the at is nullable without having to deref hat->root for
421 * non-nullable tries.
422 */
423 if (!hat->bitmap.root) {
424 return false;
425 }
426
427 qps_hptr_deref(hat->qps, &hat->root_cache);
428 return !qps_bitmap_get(&hat->bitmap, key);
429}

658/** Qhat enumerator.
659 *
660 * Allows to scan keys and values of a QPS hat (either nullable or not).
661 * The qhat can only be scanned forward: in increasing key order.
662 *
663 * The enumerator is meant to be created with one of the following functions:
664 * - #qhat_get_enumerator
665 * - #qhat_get_enumerator_at
666 *
667 * Public functions:
668 * - #qhat_enumerator_next
669 * - #qhat_enumerator_go_to
670 * - #qhat_enumerator_get_path
671 * - #qhat_enumerator_get_value_safe
