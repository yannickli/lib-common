24/** \brief Decode a ber-encoded length to a 32 bits unsigned int.
25 * \param[inout]ps Input byte stream.
26 * \param[out]_len Output length.
27 */
28int ber_decode_len32(pstream_t *ps, uint32_t *_len)
29{
30 int c = ps_getc(ps);
31 uint32_t len;
32
33 if (unlikely(c == EOF))
34 return -1;
35
36 if (!(c & 0x80)) {
37 /* Case 1 . Local variable first contains length. */
38 *_len = c;
39 return 0;
40 }
41
42 /* Case 2 . */
43 /* Local variable first contains byte length of the computed value */
44 c &= 0x7f;
45 if (!ps_has(ps, c) || c <= 0 || c > 4) {
46 if (c == 0) {
47 e_trace(3, "(Indefinite length)");
48 return 1;
49 }
50
51 e_trace(1, "invalid length encoding");
52
53 return -1;
54 }
55
56 len = 0;
57 while (c--) {
58 len = (len << 8) | __ps_getc(ps);
59 }
60 *_len = len;
61 return 0;
62}
