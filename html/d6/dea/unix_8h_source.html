104/** Lock a directory
105 * \param dfd directory file descriptor
106 * \param dlock directory lock
107 * \retval 0 on success
108 * \retval -1 on error
109 * \see unlockdir
110 *
111 * Try to create a .lock file (with u+rw,g+r,o+r permissions) into the given
112 * directory and lock it.
113 *
114 * An error is returned if the directory is already locked (.lock exists and
115 * is locked) or if the directory cannot be written. In this case, errno is
116 * set appropriately.
117 *
118 * After a sucessful call to lockdir(), the directory file descriptor (dfd) is
119 * duplicated for internal usage. The file descriptors of the dir_lock_t
120 * should not be used by the application.
121 *
122 * Use unlockdir() to unlock a directory locked with unlock().
123 */
124int lockdir(int dfd, dir_lock_t * nonnull dlock);
125
126/** Unlock a directory
127 * \param dlock directory lock
128 * \see lockdir
129 *
130 * Unlock the .lock file and delete it. unlockdir() be called on a file
131 * descriptor returned by lockdir().
132 *
133 * To be safe, this function resets the file descriptors to -1.
134 */

458/* XXX: This function MUST be inlined in check_strace() to avoid appearing
459 * in the stack.
460 */
461static ALWAYS_INLINE int _psinfo_get_tracer_pid(pid_t pid)
462{
463 t_scope;
464 sb_t buf;
465 pstream_t ps;
466 pid_t tpid;
467
468 t_sb_init(&buf, (2 << 10));
469
470 if (pid <= 0) {
471 RETHROW(sb_read_file(&buf, "/proc/self/status"));
472 } else {
473 char path[PATH_MAX];
474
475 snprintf(path, sizeof(path), "/proc/%d/status", pid);
476 RETHROW(sb_read_file(&buf, path));
477 }
478
479 ps = ps_initsb(&buf);
480
481 while (!ps_done(&ps)) {
482 if (ps_skipstr(&ps, "TracerPid:") >= 0) {
483 tpid = ps_geti(&ps);
484 return tpid > 0 ? tpid : 0;
485 }
486 RETHROW(ps_skip_afterchr(&ps, '\n'));
487 }
488
489 return -1;
490}
