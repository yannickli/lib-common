46static inline int str_digit_value(int x)
47{
48 if (__builtin_constant_p(x)) {
49 switch (x) {
50 case '0' ... '9': return x - '0';
51 case 'a' ... 'z': return 10 + x - 'a';
52 case 'A' ... 'Z': return 10 + x - 'A';
53 default: return -1;
54 }
55 } else {
56 return __str_digit_value[x + 128];
57 }
58}
59static inline int hexdigit(int x)
60{
61 if (__builtin_constant_p(x)) {
62 switch (x) {
63 case '0' ... '9': return x - '0';
64 case 'a' ... 'f': return 10 + x - 'a';
65 case 'A' ... 'F': return 10 + x - 'A';
66 default: return -1;
67 }
68 } else {
69 int i = __str_digit_value[x + 128];
70 return i < 16 ? i : -1;
71 }
72}
73static inline int hexdecode(const char * nonnull str)
74{
75 int h = hexdigit(str[0]);
76
77 return (h < 0) ? h : ((h << 4) | hexdigit(str[1]));
78}

104static inline uint8_t __pstrputuc(char * nonnull dst, int32_t c)
105{
106 uint8_t len;
107
108 if (c < 0x80) {
109 *dst = c;
110 return 1;
111 }
112 if (__builtin_constant_p(c)) {
113 if (c >= 0 && c < 0x200000) {
114 len = 1 + (c >= 0x80) + (c >= 0x800) + (c >= 0x10000);
115 } else {
116 len = 0;
117 }
118 } else {
119 /* XXX: 31 ^ clz(c) is actually bsr in x86 assembly */
120 len = __utf8_clz_to_charlen[31 ^ __builtin_clz(c | 1)];
121 }
122 switch (__builtin_expect(len, 2)) {
123 default: dst[3] = (c | 0x80) & 0xbf; c >>= 6;
124 /* FALLTHROUGH */
125 case 3: dst[2] = (c | 0x80) & 0xbf; c >>= 6;
126 /* FALLTHROUGH */
127 case 2: dst[1] = (c | 0x80) & 0xbf; c >>= 6;
128 dst[0] = (c | __utf8_mark[len]);
129 /* FALLTHROUGH */
130 case 0: break;
131 }
132 return len;
133}

135/* XXX 0 means invalid utf8 char */
136static inline uint8_t utf8_charlen(const char * nonnull s, int len)
137{
138 uint8_t charlen = __utf8_char_len[(unsigned char)*s >> 3];
139
140 if (len >= 0 && unlikely(len < charlen)) {
141 return 0;
142 }
143
144 switch (charlen) {
145 default: if (unlikely((*++s & 0xc0) != 0x80)) return 0;
146 /* FALLTHROUGH */
147 case 3: if (unlikely((*++s & 0xc0) != 0x80)) return 0;
148 /* FALLTHROUGH */
149 case 2: if (unlikely((*++s & 0xc0) != 0x80)) return 0;
150 /* FALLTHROUGH */
151 case 1: return charlen;
152 case 0: return charlen;
153 }
154}

156/** Get the number of UTF8 characters contained in a string.
157 *
158 * \return -1 in case of invalid UTF8.
159 */
160static inline ssize_t utf8_strnlen(const char * nonnull s, size_t len)
161{
162 const char *end = s + len;
163
164 len = 0;
165 while (s < end) {
166 uint8_t charlen = utf8_charlen(s, end - s);
167
168 if (unlikely(charlen == 0)) {
169 return -1;
170 }
171 len++;
172 s += charlen;
173 }
174
175 return len;
176}

187static inline int utf8_getc_slow(const char * nonnull s,
188 const char * nullable * nullable out)
189{
190 uint32_t ret = 0;
191 uint8_t len = utf8_charlen(s, -1) - 1;
192
193 switch (len) {
194 default: return -1;
195 case 3: ret += (unsigned char)*s++; ret <<= 6;
196 /* FALLTHROUGH */
197 case 2: ret += (unsigned char)*s++; ret <<= 6;
198 /* FALLTHROUGH */
199 case 1: ret += (unsigned char)*s++; ret <<= 6;
200 /* FALLTHROUGH */
201 case 0: ret += (unsigned char)*s++;
202 /* FALLTHROUGH */
203 }
204
205 if (out) {
206 *out = s;
207 }
208
209 return ret - __utf8_offs[len];
210}

212static ALWAYS_INLINE int utf8_getc(const char * nonnull s,
213 const char * nullable * nullable out)
214{
215 if ((unsigned char)*s < 0x80) {
216 if (out) {
217 *out = s + 1;
218 }
219 return (unsigned char)*s;
220 } else {
221 return utf8_getc_slow(s, out);
222 }
223}

225static ALWAYS_INLINE int utf8_ngetc(const char * nonnull s, int len,
226 const char * nullable * nullable out)
227{
228 if (len && (unsigned char)*s < 0x80) {
229 if (out) {
230 *out = s + 1;
231 }
232 return (unsigned char)*s;
233 }
234 if (unlikely(len < __utf8_char_len[(unsigned char)*s >> 3])) {
235 return -1;
236 }
237 return utf8_getc_slow(s, out);
238}

240static ALWAYS_INLINE int utf8_ngetc_at(const char * nonnull s, int len,
241 int * nonnull offp)
242{
243 int off = *offp;
244 const char *out = NULL;
245 int res;
246
247 if (off < 0 || off >= len) {
248 return -1;
249 }
250 s += off;
251 len -= off;
252 res = RETHROW(utf8_ngetc(s, len, &out));
253
254 *offp += (out - s);
255 return res;
256}
