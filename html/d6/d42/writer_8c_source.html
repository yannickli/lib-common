98const char *t_asn1_oid_print(lstr_t oid)
99{
100 static const char digits[10] = "0123456789";
101
102 char *str = t_new_raw(char, oid.len * 4);
103 char *w = str;
104
105 for (int i = 0;; i++) {
106 uint8_t u = (uint8_t)oid.s[i];
107 uint8_t d = u / 10;
108 uint8_t c = d / 10;
109
110 u -= 10 * d;
111 d -= 10 * c;
112
113 if (c)
114 *w++ = digits[c];
115
116 if (c || d)
117 *w++ = digits[d];
118
119 *w++ = digits[u];
120
121 if (i >= oid.len)
122 break;
123
124 *w++ = '.';
125 }
126
127 *w = '\0';
128
129 return str;
130}

143asn1_bit_string_t
144t_asn1_bstring_from_bf64(uint64_t bit_field, int min_bit_len)
145{
146 int len, bit_len;
147 be64_t be;
148
149 assert (min_bit_len <= 64);
150 bit_len = bit_field ? bsr64(bit_field) + 1 : 0;
151 bit_len = MAX(bit_len, min_bit_len);
152 len = DIV_ROUND_UP(bit_len, 8);
153
154 be = cpu_to_be64(bit_reverse64(bit_field));
155
156 return (asn1_bit_string_t){
157 .bit_len = bit_len,
158 .data = t_dup((const byte *)&be, len),
159 };
160}

194const void *asn1_opt_field(const void *field, enum obj_type type)
195{
196 switch (type) {
197 case ASN1_OBJ_TYPE(bool):
198 return ((const opt_bool_t *)field)->has_field ? field : NULL;
199 case ASN1_OBJ_TYPE(int8_t): case ASN1_OBJ_TYPE(uint8_t):
200 return ((const opt_i8_t *)field)->has_field ? field : NULL;
201 case ASN1_OBJ_TYPE(int16_t): case ASN1_OBJ_TYPE(uint16_t):
202 return ((const opt_i16_t *)field)->has_field ? field : NULL;
203 case ASN1_OBJ_TYPE(int32_t): case ASN1_OBJ_TYPE(uint32_t):
204 case ASN1_OBJ_TYPE(enum):
205 return ((const opt_i32_t *)field)->has_field ? field : NULL;
206 case ASN1_OBJ_TYPE(int64_t): case ASN1_OBJ_TYPE(uint64_t):
207 return ((const opt_i64_t *)field)->has_field ? field : NULL;
208 case ASN1_OBJ_TYPE(NULL): /* Should not happen. */
209 return NULL;
210 case ASN1_OBJ_TYPE(OPT_NULL):
211 return *(const bool *)field ? field : NULL;
212 case ASN1_OBJ_TYPE(lstr_t):
213 case ASN1_OBJ_TYPE(OPEN_TYPE): case ASN1_OBJ_TYPE(asn1_bit_string_t):
214 return ((const lstr_t *)field)->data ? field : NULL;
215 case ASN1_OBJ_TYPE(asn1_ext_t):
216 return ((const asn1_ext_t *)field)->data ? field : NULL;
217 case ASN1_OBJ_TYPE(OPAQUE): case ASN1_OBJ_TYPE(SEQUENCE):
218 case ASN1_OBJ_TYPE(CHOICE): case ASN1_OBJ_TYPE(UNTAGGED_CHOICE):
219 return field;
220 case ASN1_OBJ_TYPE(SKIP):
221 return NULL;
222 }
223 return NULL;
224}

339static int asn1_pack_value_size(const void *dt, const asn1_field_t *spec,
340 qv_t(i32) *stack, int32_t *len)
341{
342 int32_t data_size;
343
344 switch (spec->type) {
345 case ASN1_OBJ_TYPE(bool): case ASN1_OBJ_TYPE(int8_t):
346 qv_append(stack, data_size = 1);
347 break;
348 case ASN1_OBJ_TYPE(uint8_t):
349 data_size = asn1_int32_size(*(const uint8_t *)dt);
350 qv_append(stack, data_size);
351 break;
352 case ASN1_OBJ_TYPE(int16_t):
353 data_size = asn1_int32_size(*(const int16_t *)dt);
354 qv_append(stack, data_size);
355 break;
356 case ASN1_OBJ_TYPE(uint16_t):
357 data_size = asn1_int32_size(*(const uint16_t *)dt);
358 qv_append(stack, data_size);
359 break;
360 case ASN1_OBJ_TYPE(int32_t):
361 data_size = asn1_int32_size(*(const int32_t *)dt);
362 qv_append(stack, data_size);
363 break;
364 case ASN1_OBJ_TYPE(enum):
365 data_size = asn1_int32_size(*(const int *)dt);
366 qv_append(stack, data_size);
367 break;
368 case ASN1_OBJ_TYPE(uint32_t):
369 data_size = asn1_uint32_size(*(const uint32_t *)dt);
370 qv_append(stack, data_size);
371 break;
372 case ASN1_OBJ_TYPE(int64_t):
373 data_size = asn1_int64_size(*(const int64_t *)dt);
374 qv_append(stack, data_size);
375 break;
376 case ASN1_OBJ_TYPE(uint64_t):
377 data_size = asn1_uint64_size(*(const uint64_t *)dt);
378 qv_append(stack, data_size);
379 break;
380 case ASN1_OBJ_TYPE(NULL):
381 case ASN1_OBJ_TYPE(OPT_NULL):
382 data_size = 0;
383 qv_append(stack, 0);
384 break;
385 case ASN1_OBJ_TYPE(lstr_t):
386 case ASN1_OBJ_TYPE(OPEN_TYPE):
387 /* IF ASSERT: user maybe forgot to declare field as optional */
388 if (!((lstr_t *)dt)->data) {
389 e_trace(0, "%s", spec->name);
390 }
391 assert (((lstr_t *)dt)->data);
392 data_size = ((lstr_t *)dt)->len;
393 qv_append(stack, data_size);
394 break;
395 case ASN1_OBJ_TYPE(asn1_bit_string_t):
396 /* IF ASSERT: user maybe forgot to declare field as optional */
397 assert (((asn1_bit_string_t *)dt)->data);
398 data_size = asn1_bit_string_size((asn1_bit_string_t *)dt);
399 qv_append(stack, data_size);
400 break;
401 case ASN1_OBJ_TYPE(SEQUENCE): case ASN1_OBJ_TYPE(CHOICE):
402 case ASN1_OBJ_TYPE(UNTAGGED_CHOICE):
403 { /* In this case, length is known after but must be written before
404 * any contained field length, so we must keep a space in the
405 * stack. */
406 size_t len_pos = stack->len;
407 qv_append(stack, 0);
408 RETHROW(data_size = asn1_pack_size_rec(dt, spec->u.comp, stack));
409 stack->tab[len_pos] = data_size;
410 }
411 break;
412 case ASN1_OBJ_TYPE(asn1_ext_t):
413 {
414 size_t len_pos = stack->len;
415 qv_append(stack, 0);
416 RETHROW(data_size = asn1_pack_size_rec(
417 ((const asn1_ext_t *)dt)->data,
418 ((const asn1_ext_t *)dt)->desc,
419 stack));
420 stack->tab[len_pos] = data_size;
421 }
422 break;
423 case ASN1_OBJ_TYPE(OPAQUE):
424 RETHROW(data_size = (*spec->u.opaque.pack_size)(dt));
425 qv_append(stack, data_size);
426 break;
427 case ASN1_OBJ_TYPE(SKIP):
428 default:
429 e_panic("should not happen");
430 return -1;
431 }
432
433
434 if (unlikely(!asn1_field_is_tagged(spec))) {
435 *len += data_size;
436 } else {
437 *len += data_size + asn1_length_size(data_size) + spec->tag_len;
438 }
439
440 return *len;
441}

443static int asn1_pack_field_size(const void *st, const asn1_field_t *spec,
444 qv_t(i32) *stack, int32_t *len)
445{
446 if (unlikely(spec->type == ASN1_OBJ_TYPE(SKIP))) {
447 return 0;
448 }
449
450 switch (spec->mode) {
451 case ASN1_OBJ_MODE(MANDATORY):
452 /* IF ASSERT: user maybe forgot to declare field as optional */
453 assert (GET_DATA_P(st, spec, void));
454 RETHROW(asn1_pack_value_size(GET_DATA_P(st, spec, void),
455 spec, stack, len));
456 break;
457 case ASN1_OBJ_MODE(OPTIONAL):
458 {
459 const void *field = asn1_opt_field(GET_DATA_P(st, spec, uint8_t),
460 spec->type);
461
462 if (field) {
463 RETHROW(asn1_pack_value_size(field, spec, stack, len));
464 }
465 break;
466 }
467 case ASN1_OBJ_MODE(SEQ_OF):
468 {
469 const uint8_t *tab = (const uint8_t *)GET_VECTOR_DATA(st, spec);
470 int vec_len = GET_VECTOR_LEN(st, spec);
471
472 if (spec->pointed) {
473 for (int j = 0; j < vec_len; j++) {
474 RETHROW(asn1_pack_value_size(((const void **)tab)[j],
475 spec, stack, len));
476 }
477 } else {
478 for (int j = 0; j < vec_len; j++) {
479 RETHROW(asn1_pack_value_size(tab + j * spec->size,
480 spec, stack, len));
481 }
482 }
483 }
484 break;
485 }
486
487 return 0;
488}

584 * \brief Serialize a single given field following specs.
585 */
586static uint8_t *asn1_pack_value(uint8_t *dst, const void *dt,
587 const asn1_field_t *spec, int32_t depth,
588 qv_t(i32) *stack)
589{
590 int32_t data_size = stack->tab[stack->len++];
591
592 if (likely(asn1_field_is_tagged(spec))) {
593 dst = asn1_pack_tag(dst, spec->tag, spec->tag_len);
594 dst = asn1_pack_len(dst, data_size);
595 }
596
597 switch (spec->type) {
598 case ASN1_OBJ_TYPE(bool):
599 dst = asn1_pack_bool(dst, *(bool *)dt);
600 e_trace(4, "value: %s", *(bool *)dt ? "true" : "false");
601 break;
602 case ASN1_OBJ_TYPE(int8_t):
603 *dst++ = *(int8_t *)dt;
604 e_trace(4, "value: %d", *(int8_t *)dt);
605 break;
606 case ASN1_OBJ_TYPE(uint8_t):
607 dst = asn1_pack_int32(dst, *(uint8_t *)dt);
608 e_trace(4, "value: %u", *(uint8_t *)dt);
609 break;
610 case ASN1_OBJ_TYPE(int16_t):
611 dst = asn1_pack_int32(dst, *(int16_t *)dt);
612 e_trace(4, "value: %d", *(int16_t *)dt);
613 break;
614 case ASN1_OBJ_TYPE(uint16_t):
615 dst = asn1_pack_int32(dst, *(uint16_t *)dt);
616 e_trace(4, "value: %u", *(uint16_t *)dt);
617 break;
618 case ASN1_OBJ_TYPE(int32_t):
619 dst = asn1_pack_int32(dst, *(int32_t *)dt);
620 e_trace(4, "value: %d", *(int32_t *)dt);
621 break;
622 case ASN1_OBJ_TYPE(enum):
623 dst = asn1_pack_int32(dst, *(int32_t *)dt);
624 e_trace(4, "value: %d", *(int32_t *)dt);
625 break;
626 case ASN1_OBJ_TYPE(uint32_t):
627 dst = asn1_pack_uint32(dst, *(uint32_t *)dt);
628 e_trace(4, "value: %u", *(uint32_t *)dt);
629 break;
630 case ASN1_OBJ_TYPE(int64_t):
631 dst = asn1_pack_int64(dst, *(int64_t *)dt);
632 e_trace(4, "value: %jd", *(int64_t *)dt);
633 break;
634 case ASN1_OBJ_TYPE(uint64_t):
635 dst = asn1_pack_uint64(dst, *(uint64_t *)dt);
636 e_trace(4, "value: %ju", *(uint64_t *)dt);
637 break;
638 case ASN1_OBJ_TYPE(NULL):
639 case ASN1_OBJ_TYPE(OPT_NULL):
640 break;
641 case ASN1_OBJ_TYPE(lstr_t):
642 case ASN1_OBJ_TYPE(OPEN_TYPE):
643 dst = asn1_pack_data(dst, (const lstr_t *)dt);
644 e_trace_hex(4, "value:", ((const lstr_t *)dt)->data,
645 ((const lstr_t *)dt)->len);
646 break;
647 case ASN1_OBJ_TYPE(asn1_bit_string_t):
648 dst = asn1_pack_bit_string(dst, (const asn1_bit_string_t *)dt);
649 /* TODO print */
650 break;
651 case ASN1_OBJ_TYPE(SEQUENCE): case ASN1_OBJ_TYPE(CHOICE):
652 case ASN1_OBJ_TYPE(UNTAGGED_CHOICE):
653 dst = asn1_pack_rec(dst, dt, spec->u.comp, depth + 1, stack);
654 break;
655 case ASN1_OBJ_TYPE(asn1_ext_t):
656 dst = asn1_pack_rec(dst, ((const asn1_ext_t *)dt)->data,
657 ((const asn1_ext_t *)dt)->desc, depth + 1,
658 stack);
659 break;
660 case ASN1_OBJ_TYPE(OPAQUE):
661 dst = (*spec->u.opaque.pack)(dst, dt);
662 break;
663 case ASN1_OBJ_TYPE(SKIP):
664 break;
665 }
666
667 return dst;
668}

670static uint8_t *asn1_pack_field(uint8_t *dst, const void *st,
671 const asn1_field_t *spec, int32_t depth,
672 qv_t(i32) *stack)
673{
674 if (unlikely(spec->type == ASN1_OBJ_TYPE(SKIP))) {
675 return dst;
676 }
677
678 switch (spec->mode) {
679 case ASN1_OBJ_MODE(MANDATORY):
680 dst = asn1_pack_value(dst, GET_DATA_P(st, spec, void),
681 spec, depth, stack);
682 break;
683 case ASN1_OBJ_MODE(OPTIONAL):
684 {
685 const void *field = asn1_opt_field(GET_DATA_P(st, spec, void),
686 spec->type);
687
688 if (field) {
689 dst = asn1_pack_value(dst, field, spec, depth, stack);
690 }
691 }
692 break;
693 case ASN1_OBJ_MODE(SEQ_OF):
694 {
695 const uint8_t *tab = (const uint8_t *)GET_VECTOR_DATA(st,
696 spec);
697 int vec_len = GET_VECTOR_LEN(st, spec);
698
699 if (spec->pointed) {
700 for (int j = 0; j < vec_len; j++) {
701 dst = asn1_pack_value(dst, ((const void **)tab)[j],
702 spec, depth, stack);
703 }
704 } else {
705 for (int j = 0; j < vec_len; j++) {
706 dst = asn1_pack_value(dst, tab + j * spec->size, spec,
707 depth, stack);
708 }
709 }
710 }
711 break;
712 }
713
714 return dst;
715}

718 const asn1_desc_t *desc, int32_t depth,

732 const asn1_desc_t *desc, int32_t depth,

734{ /* Could be way shorter but far more reader friendly this way */

871void *asn1_opt_field_w(void *field, enum obj_type type, bool has_field)
872{
873 switch (type) {
874 case ASN1_OBJ_TYPE(bool):
875 if (!(((opt_bool_t *)field)->has_field = has_field)) {
876 ((opt_bool_t *)field)->v = false;
877 }
878 return (uint8_t *)field + offsetof(opt_bool_t, v);
879 case ASN1_OBJ_TYPE(int8_t): case ASN1_OBJ_TYPE(uint8_t):
880 if (!(((opt_i8_t *)field)->has_field = has_field)) {
881 ((opt_i8_t *)field)->v = 0;
882 }
883 return (uint8_t *)field + offsetof(opt_i8_t, v);
884 case ASN1_OBJ_TYPE(int16_t): case ASN1_OBJ_TYPE(uint16_t):
885 if (!(((opt_i16_t *)field)->has_field = has_field)) {
886 ((opt_i16_t *)field)->v = 0;
887 }
888 return (uint16_t *)field + offsetof(opt_i16_t, v);
889 case ASN1_OBJ_TYPE(int32_t): case ASN1_OBJ_TYPE(uint32_t):
890 case ASN1_OBJ_TYPE(enum):
891 if (!(((opt_i32_t *)field)->has_field = has_field)) {
892 ((opt_i32_t *)field)->v = 0;
893 }
894 return (uint32_t *)field + offsetof(opt_i32_t, v);
895 case ASN1_OBJ_TYPE(int64_t): case ASN1_OBJ_TYPE(uint64_t):
896 if (!(((opt_i64_t *)field)->has_field = has_field)) {
897 ((opt_i64_t *)field)->v = 0;
898 }
899 return (uint64_t *)field + offsetof(opt_i64_t, v);
900 case ASN1_OBJ_TYPE(NULL): /* Should not happen. */
901 return NULL;
902 case ASN1_OBJ_TYPE(OPT_NULL):
903 *(bool *)field = has_field;
904 return field;
905 case ASN1_OBJ_TYPE(lstr_t):
906 case ASN1_OBJ_TYPE(OPEN_TYPE): case ASN1_OBJ_TYPE(asn1_bit_string_t):
907 if (!has_field) {
908 *(lstr_t *)field = LSTR_NULL_V;
909 }
910 return field;
911 case ASN1_OBJ_TYPE(asn1_ext_t):
912 if (!has_field) {
913 ((asn1_ext_t *)field)->has_value = false;
914 ((asn1_ext_t *)field)->value = ps_initptr(NULL, NULL);
915 }
916 return field;
917 case ASN1_OBJ_TYPE(OPAQUE): case ASN1_OBJ_TYPE(SEQUENCE):
918 case ASN1_OBJ_TYPE(CHOICE): case ASN1_OBJ_TYPE(UNTAGGED_CHOICE):
919 if (!has_field) {
920 *((void **)field) = NULL;
921 }
922 return *(void **)field;
923 case ASN1_OBJ_TYPE(SKIP):
924 return NULL;
925 default:
926 e_panic("unexpected type");
927 }
928}

957static int asn1_skip_ber_tag(pstream_t *ps)
958{
959 if (!ps_has(ps, 1)) {
960 e_trace(1, "error: no tag to read, stream end");
961 return -1;
962 }
963
964 /** Skip ASN1 BER tag according to ITU-T X.690 8.1.2. */
965 if (unlikely((ps->b[0] & 0x1f) == 0x1f)) {
966 /* Tag is encoded on more bytes. */
967 do {
968 __ps_skip(ps, 1);
969
970 if (unlikely(!ps_has(ps, 1))) {
971 e_trace(1, "error: missing byte in extended tag");
972 return -1;
973 }
974 /* While bit 8 is on there are more bytes to read */
975 } while (ps->b[0] >> 7);
976 }
977 __ps_skip(ps, 1);
978
979 return 0;
980}

982int asn1_get_ber_field(pstream_t *ps, bool indef_father, pstream_t *sub_ps)
983{
984 uint32_t data_size;
985 int n_eoc = indef_father ? 1: 0;
986
987 if (sub_ps) {
988 sub_ps->b = ps->b;
989 }
990
991 do {
992 if (!ps_has(ps, 2)) {
993 e_trace(1, "error: stream end");
994 return -1;
995 }
996
997 if (ps->b[0]) {
998 RETHROW(asn1_skip_ber_tag(ps));
999
1000 if (RETHROW(ber_decode_len32(ps, &data_size))) {
1001 n_eoc++;
1002 } else {
1003 if (unlikely(ps_skip(ps, data_size) < 0)) {
1004 e_trace(1, "error: not enough bytes");
1005 return -1;
1006 }
1007 }
1008 } else {
1009 if (unlikely(ps->b[1])) {
1010 /* See: ITU Fascicle VIII.4 - Rec. X.209 - 23.5.4 */
1011 e_trace(1, "invalid EOC");
1012 return -1;
1013 }
1014
1015 if (likely(n_eoc > 1 || !indef_father)) {
1016 __ps_skip(ps, 2);
1017 }
1018
1019 n_eoc--;
1020 }
1021 } while (n_eoc);
1022
1023 if (sub_ps) {
1024 sub_ps->b_end = ps->b;
1025 }
1026
1027 return 0;
1028}

1039static int asn1_unpack_value(pstream_t *ps, const asn1_field_t *spec,
1040 mem_pool_t *mem_pool, int depth, void *dt,
1041 bool copy)
1042{
1043 uint32_t data_size;
1044 pstream_t field_ps;
1045 bool indef_len;
1046
1047 switch (spec->type) {
1048 case ASN1_OBJ_TYPE(SKIP):
1049 return asn1_get_ber_field(ps, false, NULL);
1050 case ASN1_OBJ_TYPE(OPEN_TYPE):
1051 RETHROW(asn1_get_ber_field(ps, false, &field_ps));
1052 data_size = ps_len(&field_ps);
1053 indef_len = false;
1054 break;
1055 default:
1056 RETHROW(ps_skip(ps, 1));
1057
1058 if (!RETHROW(ber_decode_len32(ps, &data_size))) {
1059 if (ps_get_ps(ps, data_size, &field_ps) < 0) {
1060 e_trace(1, "p-stream does not have enough bytes "
1061 "(got %zd needed %u)", ps_len(ps), data_size);
1062 return -1;
1063 }
1064
1065 indef_len = false;
1066 } else {
1067 field_ps = *ps;
1068
1069 if (spec->type < ASN1_OBJ_TYPE(asn1_ext_t)) {
1070 e_trace(1, "error: unexpected indefinite length");
1071 return -1;
1072 }
1073
1074 indef_len = true;
1075 }
1076 break;
1077 }
1078
1079 switch (spec->type) {
1080 case ASN1_OBJ_TYPE(bool):
1081 *(bool *)dt = (ps_getc(&field_ps) ? true : false);
1082 e_trace(4, "value: %s", *(bool *)dt ? "true" : "false");
1083 break;
1084 case ASN1_OBJ_TYPE(int8_t):
1085 *(int8_t *)dt = ps_getc(&field_ps);
1086 e_trace(4, "value: %d", *(int8_t *)dt);
1087 break;
1088 case ASN1_OBJ_TYPE(uint8_t):
1089 if (unlikely(ps_len(&field_ps) == 2)) {
1090 if (*field_ps.b == 0x00) {
1091 __ps_skip(&field_ps, 1);
1092 } else {
1093 e_trace(1, "wrong uint8 size");
1094 return -1;
1095 }
1096 }
1097 *(uint8_t *)dt = ps_getc(&field_ps);
1098 e_trace(4, "value: %u", *(uint8_t *)dt);
1099 break;
1100 case ASN1_OBJ_TYPE(int16_t):
1101 RETHROW(ber_decode_int16(&field_ps, (int16_t *)dt));
1102 e_trace(4, "value: %d", *(int16_t *)dt);
1103 break;
1104 case ASN1_OBJ_TYPE(uint16_t):
1105 RETHROW(ber_decode_uint16(&field_ps, (uint16_t *)dt));
1106 e_trace(4, "value: %u", *(uint16_t *)dt);
1107 break;
1108 case ASN1_OBJ_TYPE(int32_t): case ASN1_OBJ_TYPE(enum):
1109 RETHROW(ber_decode_int32(&field_ps, (int32_t *)dt));
1110 e_trace(4, "value: %d", *(int32_t *)dt);
1111 break;
1112 case ASN1_OBJ_TYPE(uint32_t):
1113 RETHROW(ber_decode_uint32(&field_ps, (uint32_t *)dt));
1114 e_trace(4, "value: %u", *(uint32_t *)dt);
1115 break;
1116 case ASN1_OBJ_TYPE(int64_t):
1117 RETHROW(ber_decode_int64(&field_ps, (int64_t *)dt));
1118 e_trace(4, "value: %jd", *(int64_t *)dt);
1119 break;
1120 case ASN1_OBJ_TYPE(uint64_t):
1121 RETHROW(ber_decode_uint64(&field_ps, (uint64_t *)dt));
1122 e_trace(4, "value: %ju", *(uint64_t *)dt);
1123 break;
1124 case ASN1_OBJ_TYPE(NULL):
1125 case ASN1_OBJ_TYPE(OPT_NULL):
1126 break;
1127 case ASN1_OBJ_TYPE(lstr_t):
1128 case ASN1_OBJ_TYPE(OPEN_TYPE):
1129 *(lstr_t *)dt = LSTR_PS_V(&field_ps);
1130 if (copy) {
1131 mp_lstr_persists(mem_pool, (lstr_t *)dt);
1132 }
1133 e_trace_hex(4, "value:", field_ps.s, (int)ps_len(&field_ps));
1134 break;
1135 case ASN1_OBJ_TYPE(asn1_bit_string_t):
1136 if (copy) {
1137 ((asn1_bit_string_t *)dt)->data = mp_dup(mem_pool, field_ps.b + 1,
1138 ps_len(&field_ps) - 1);
1139 } else {
1140 ((asn1_bit_string_t *)dt)->data = field_ps.b + 1;
1141 }
1142 ((asn1_bit_string_t *)dt)->bit_len = 8 * (ps_len(&field_ps) - 1)
1143 - field_ps.b[0];
1144 /* TODO print */
1145 break;
1146 case ASN1_OBJ_TYPE(SEQUENCE): case ASN1_OBJ_TYPE(CHOICE):
1147 case ASN1_OBJ_TYPE(UNTAGGED_CHOICE):
1148 RETHROW(asn1_unpack_rec(&field_ps, spec->u.comp, mem_pool,
1149 depth + 1, dt, copy, indef_len));
1150 break;
1151 case ASN1_OBJ_TYPE(asn1_ext_t):
1152 ((asn1_ext_t *)dt)->data = NULL;
1153 ((asn1_ext_t *)dt)->desc = NULL;
1154 ((asn1_ext_t *)dt)->has_value = true;
1155
1156 if (indef_len) {
1157 RETHROW(asn1_get_ber_field(&field_ps, true,
1158 &((asn1_ext_t *)dt)->value));
1159 } else {
1160 ((asn1_ext_t *)dt)->value = field_ps;
1161 }
1162 break;
1163 case ASN1_OBJ_TYPE(OPAQUE):
1164 RETHROW((*spec->u.opaque.unpack)(&field_ps, mem_pool, dt));
1165 break;
1166 case ASN1_OBJ_TYPE(SKIP):
1167 e_panic("should not happen");
1168 break;
1169 }
1170
1171 if (indef_len) {
1172 *ps = field_ps;
1173
1174 /* Skip every trailing fields up to EOC. */
1175 RETHROW(asn1_get_ber_field(ps, true, NULL));
1176
1177 /* Skip EOC. */
1178 __ps_skip(ps, 2);
1179 }
1180
1181 return 0;
1182}

1232static int asn1_unpack_field(pstream_t *ps, const asn1_field_t *spec,
1233 mem_pool_t *mem_pool, int depth, void *st,
1234 bool copy, bool indef_len)
1235{
1236 switch (spec->mode) {
1237 case ASN1_OBJ_MODE(MANDATORY):
1238 if (!asn1_field_is_tagged(spec)
1239 || (ps_has(ps, 1) && *ps->b == spec->tag))
1240 {
1241 void *value = asn1_alloc_if_pointed(spec, mem_pool, st);
1242
1243 RETHROW(asn1_unpack_value(ps, spec, mem_pool, depth, value,
1244 copy));
1245 } else {
1246 if (ps_has(ps, 1)) {
1247 e_trace(0, "mandatory value -- %s -- not found (got tag %x)",
1248 spec->name, *ps->b);
1249 } else {
1250 e_trace(0, "mandatory value -- %s -- not found (input end)",
1251 spec->name);
1252 }
1253 return -1;
1254 }
1255 break;
1256
1257 case ASN1_OBJ_MODE(OPTIONAL):
1258 if (!ps_done(ps) && *ps->b
1259 && (!asn1_field_is_tagged(spec) || *ps->b == spec->tag))
1260 {
1261 void *value;
1262
1263 asn1_alloc_if_pointed(spec, mem_pool, st);
1264 value = asn1_opt_field_w(GET_PTR(st, spec, void), spec->type,
1265 true);
1266 RETHROW(asn1_unpack_value(ps, spec, mem_pool, depth, value,
1267 copy));
1268 } else {
1269 asn1_opt_field_w(GET_PTR(st, spec, void), spec->type, false);
1270 }
1271 break;
1272 case ASN1_OBJ_MODE(SEQ_OF):
1273 {
1274 int count;
1275
1276 RETHROW(count = asn1_sequenceof_len(*ps, spec->tag));
1277
1278 if (unlikely(!count)) {
1279 p_clear(GET_PTR(st, spec, asn1_void_vector_t), 1);
1280 break;
