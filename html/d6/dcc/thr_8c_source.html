28/*
29 * This code gets pulled automatically as soon as pthread_create() is used
30 * because we make our symbol strong and visible, and it then overrides the
31 * libc one.
32 *
33 * When we build a shared library, we export our own pthread_create and it
34 * also overrides the libc one, meaning that the library can use all intersec
35 * APIs and be pthread compatible.
36 */
37
38static struct {
39 pthread_once_t key_once;
40 pthread_key_t key;
41} core_thread_g = {
42#define _G core_thread_g
43 .key_once = PTHREAD_ONCE_INIT,
44};

46void thr_detach(void)
47{
48 pthread_setspecific(_G.key, NULL);
49 dlist_for_each(it, &thr_hooks_g.exit_cbs) {
50 (container_of(it, struct thr_ctor, link)->cb)();
51 }
52}

81static void *thr_hooks_wrapper(void *data)
82{
83 void *(*fn)(void *) = ((void **)data)[0];
84 void *arg = ((void **)data)[1];
85 void *ret;
86
87 p_delete(&data);
88 thr_attach();
89 ret = fn(arg);
90 thr_detach();
91 return ret;
92}

94int thr_create(pthread_t *restrict thread,
95 const pthread_attr_t *restrict attr,
96 void *(*fn)(void *), void *restrict arg)
97{
98 static typeof(pthread_create) *real_pthread_create;
99 void **pair = p_new(void *, 2);
100 int res;
101
102#if !defined(__has_asan) && !defined(__has_tsan)
103 if (unlikely(!real_pthread_create))
104 real_pthread_create = dlsym(RTLD_NEXT, "pthread_create");
105#else
106 if (unlikely(!real_pthread_create))
107 real_pthread_create = dlsym(RTLD_DEFAULT, "pthread_create");
108#endif
109 pair[0] = fn;
110 pair[1] = arg;
111 res = (*real_pthread_create)(thread, attr, &thr_hooks_wrapper, pair);
112 if (res != 0) {
113 errno = res;
114 p_delete(&pair);
115 }
116 return res;
117}
