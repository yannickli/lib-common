22void ob_check_invariants(outbuf_t *ob)
23{
24 int len = ob->length, sb_len = ob->sb.len;
25
26 htlist_for_each(it, &ob->chunks_list) {
27 outbuf_chunk_t *obc = htlist_entry(it, outbuf_chunk_t, chunks_link);
28
29 sb_len -= obc->sb_leading;
30 len -= obc->sb_leading;
31 len -= (obc->length - obc->offset);
32 }
33
34 assert (len == ob->sb_trailing);
35 assert (sb_len == ob->sb_trailing);
36}

50static void ob_merge_(outbuf_t *dst, outbuf_t *src, bool wipe)
51{
52 outbuf_chunk_t *obc;
53
54 sb_addsb(&dst->sb, &src->sb);
55
56 if (!htlist_is_empty(&src->chunks_list)) {
57 obc = htlist_first_entry(&src->chunks_list, outbuf_chunk_t,
58 chunks_link);
59 obc->sb_leading += dst->sb_trailing;
60 dst->sb_trailing = src->sb_trailing;
61 htlist_splice_tail(&dst->chunks_list, &src->chunks_list);
62 } else {
63 dst->sb_trailing += src->sb_trailing;
64 }
65 dst->length += src->length;
66
67 if (wipe) {
68 sb_wipe(&src->sb);
69 } else {
70 src->length = 0;
71 src->sb_trailing = 0;
72 htlist_init(&src->chunks_list);
73 sb_reset(&src->sb);
74 }
75}

122int ob_add_file(outbuf_t *ob, const char *file, int size)
123{
124 int fd = RETHROW(open(file, O_RDONLY));
125
126 if (size < 0) {
127 struct stat st;
128
129 if (fstat(fd, &st)) {
130 PROTECT_ERRNO(close(fd));
131 return -1;
132 }
133 size = st.st_size;
134 }
135
136 if (size <= OUTBUF_CHUNK_MIN_SIZE) {
137 if (ob_xread(ob, fd, size)) {
138 PROTECT_ERRNO(close(fd));
139 return -1;
140 }
141 close(fd);
142 } else {
143 void *map = mmap(NULL, size, PROT_READ, MAP_SHARED, fd, 0);
144
145 PROTECT_ERRNO(close(fd));
146 if (map == MAP_FAILED)
147 return -1;
148 madvise(map, size, MADV_SEQUENTIAL);
149 ob_add_memmap(ob, map, size);
150 }
151 return 0;
152}

154static int ob_consume(outbuf_t *ob, int len)
155{
156 ob->length -= len;
157
158 while (!htlist_is_empty(&ob->chunks_list)) {
159 outbuf_chunk_t *obc;
160
161 obc = htlist_first_entry(&ob->chunks_list, outbuf_chunk_t,
162 chunks_link);
163 if (len < obc->sb_leading) {
164 sb_skip(&ob->sb, len);
165 obc->sb_leading -= len;
166 return 0;
167 }
168 if (obc->sb_leading) {
169 sb_skip(&ob->sb, obc->sb_leading);
170 len -= obc->sb_leading;
171 obc->sb_leading = 0;
172 }
173 if (obc->offset + len < obc->length) {
174 obc->offset += len;
175 return 0;
176 }
177 len -= (obc->length - obc->offset);
178
179 htlist_pop(&ob->chunks_list);
180 ob_chunk_delete(&obc);
181 }
182
183 assert (len <= ob->sb_trailing);
184 sb_skip(&ob->sb, len);
185 ob->sb_trailing -= len;
186 return 0;
187}

189int ob_write_with(outbuf_t *ob, int fd,
190 ssize_t (*writerv)(int, const struct iovec *, int, void *),
191 void *priv)
192{
193#define PREPARE_AT_LEAST (64U << 10)
194 struct iovec iov[IOV_MAX];
195 size_t iovcnt = 0, sb_pos = 0, iov_size = 0;
196
197 if (!ob->length)
198 return 0;
199
200 htlist_for_each(it, &ob->chunks_list) {
201 outbuf_chunk_t *obc = htlist_entry(it, outbuf_chunk_t, chunks_link);
202 size_t len;
203
204 len = obc->sb_leading;
205 if (len) {
206 iov[iovcnt++] = MAKE_IOVEC(ob->sb.data + sb_pos, len);
207 sb_pos += len;
208 iov_size += len;
209 }
210
211 len = obc->length - obc->offset;
212 iov[iovcnt++] = MAKE_IOVEC(obc->u.b + obc->offset, len);
213 iov_size += len;
214 if (iov_size > PREPARE_AT_LEAST || iovcnt + 2 >= sizeof(iov))
215 goto doit;
216 }
217
218 if (ob->sb_trailing) {
219 iov[iovcnt++] = MAKE_IOVEC(ob->sb.data + sb_pos, ob->sb_trailing);
220 assert ((size_t)ob->sb.len == sb_pos + ob->sb_trailing);
221 }
222
223 doit:
224 return ob_consume(ob, RETHROW(writerv ?
225 (*writerv)(fd, iov, iovcnt, priv) :
226 writev(fd, iov, iovcnt)));
227}
