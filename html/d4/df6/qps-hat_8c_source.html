126static uint32_t qhat_compact_lookup(const qhat_compacthdr_t *header,
127 uint32_t from, uint32_t key)
128{
129 uint32_t count = header->count - from;
130
131 if (count == 0 || key > header->keys[header->count - 1]) {
132 return header->count;
133 } else
134 if (count < 32) {
135 for (uint32_t i = from; i < header->count; i++) {
136 if (header->keys[i] >= key) {
137 return i;
138 }
139 }
140 return header->count;
141 }
142 return from + bisect32(key, header->keys + from, count, NULL);
143}

145static uint32_t qhat_depth_shift(const qhat_t *hat, uint32_t depth)
146{
147 /* depth 0: shift (20 + leaf_bits)
148 * depth 1: shift (10 + leaf_bits)
149 * depth 2: shift leaf_bits
150 * depth 3: shift 0
151 */
152 if (depth != QHAT_DEPTH_MAX) {
153 return (2 - depth) * QHAT_SHIFT + hat->desc->leaf_index_bits;
154 } else {
155 return 0;
156 }
157}
158
159static uint32_t qhat_depth_prefix(const qhat_t *hat, uint32_t key,
160 uint32_t depth)
161{
162 uint32_t shift = qhat_depth_shift(hat, depth);
163 if (shift == bitsizeof(uint32_t)) {
164 return 0;
165 }
166 return key & ~((1U << shift) - 1);
167}
168
169static uint32_t qhat_lshift(const qhat_t *hat, uint32_t key, uint32_t depth)
170{
171 uint32_t shift = qhat_depth_shift(hat, depth);
172 if (shift == bitsizeof(uint32_t)) {
173 return 0;
174 }
175 return key << shift;
176}

178static uint32_t qhat_get_key_bits(const qhat_t *hat, uint32_t key,
179 uint32_t depth)
180{
181 if (depth == QHAT_DEPTH_MAX) {
182 return key & hat->desc->leaf_index_mask;
183 } else {
184 uint32_t shift = qhat_depth_shift(hat, depth);
185 if (shift == bitsizeof(uint32_t)) {
186 return 0;
187 } else {
188 return (key >> shift) & QHAT_MASK;
189 }
190 }
191}

243static __must_check__ int
244qhat_flat_check_consistency(qhat_t *hat, uint32_t from, uint32_t to,
245 qhat_node_memory_t memory,
246 bool *is_suboptimal)
247{
248 bool non_null = false;
249
250 for (uint32_t i = 0; i < hat->desc->leaves_per_flat; i++) {
251#define CASE(Size, Compact, Flat) \
252 if (!IS_ZERO(Size, Flat[i])) { \
253 non_null = true; \
254 break; \
255 }
256 QHAT_VALUE_LEN_SWITCH(hat, memory, CASE);
257#undef CASE
258 }
259 SUBOPTIMAL(non_null);
260
261 return 0;
262}

264static __must_check__ int
265qhat_compact_check_consistency(qhat_t *hat, uint32_t from, uint32_t to,
266 qhat_node_memory_t memory,
267 int flags,
268 bool *nullable is_suboptimal)
269{
270 int64_t prev_key = -1;
271
272 if (flags & QHAT_CHECK_CONTENT) {
273 SUBOPTIMAL(memory.compact->count > 0);
274 } else if (memory.compact->count == 0) {
275 return 0;
276 }
277
278 CRITICAL(memory.compact->count <= hat->desc->leaves_per_compact,
279 "compact overflow: %u > %u",
280 memory.compact->count, hat->desc->leaves_per_compact);
281
282 for (uint32_t i = 0; i < memory.compact->count; i++) {
283 uint32_t k = memory.compact->keys[i];
284
285 CRITICAL(k > prev_key, "bad key order: [%d]%jx >= [%d]%x",
286 i - 1, prev_key, i, k);
287 CRITICAL(from <= k && k <= to, "key [%d]%u out of range [%x, %x]",
288 i, k, from, to);
289
290 if (flags & QHAT_CHECK_CONTENT) {
291#define CASE(Size, Compact, Flat) SUBOPTIMAL(!IS_ZERO(Size, Compact->values[i]));
292 QHAT_VALUE_LEN_SWITCH(hat, memory, CASE);
293#undef CASE
294 }
295
296 prev_key = k;
297 }
298
299 return 0;
300}

324static __must_check__ int
325qhat_node_check_child(qhat_t *hat, uint32_t key, uint32_t from,
326 uint32_t to, uint32_t depth, qhat_node_t node,
327 int flags, bool *nullable is_suboptimal)
328{
329 qhat_node_memory_t memory;
330 uint32_t key_from;
331 uint32_t key_to;
332 qhat_node_check_ctx_t debug_ctx = {
333 .key_from = 0,
334 .key_to = 0,
335 .depth = depth,
336 .is_leaf = node.leaf,
337 .is_compact = node.compact,
338 };
339 size_t page_sz;
340
341 debug_stack_scope(DATA_PTR(&debug_ctx), &qhat_node_check_ctx_print);
342
343 if (!node.value) {
344 return 0;
345 }
346
347 key_from = key | qhat_lshift(hat, from, depth);
348 key_to = key | qhat_lshift(hat, to - 1, depth);
349 key_to += qhat_lshift(hat, 1, depth) - 1;
350
351 debug_ctx.key_from = key_from;
352 debug_ctx.key_to = key_to;
353
354 CRITICAL(qps_pg_is_in_range(hat->qps, node.page),
355 "bad page number %x", node.page);
356 memory.raw = qps_pg_deref(hat->qps, node.page);
357
358 page_sz = qps_pg_sizeof(hat->qps, node.page);
359 if (node.leaf && node.compact) {
360 size_t exp_size;
361
362#define CASE(Size, Compact, Flat) \
363 exp_size = sizeof(*(Compact)) / QHAT_SIZE;
364
365 QHAT_VALUE_LEN_SWITCH(hat, memory, CASE);
366#undef CASE
367
368 CRITICAL(page_sz == exp_size, "bad page size for compact %zu != %zu",
369 page_sz, exp_size);
370 CRITICAL(memory.compact->parent_left == from,
371 "%u != %u", memory.compact->parent_left, from);
372 CRITICAL(memory.compact->parent_right == to,
373 "%u != %u", memory.compact->parent_right, to);
374
375 RETHROW(qhat_compact_check_consistency(hat, key_from, key_to,
376 memory, flags, is_suboptimal));
377 } else if (node.leaf) {
378 size_t exp_size;
379
380 exp_size = hat->desc->value_len *
381 hat->desc->leaves_per_flat / QHAT_SIZE;
382 CRITICAL(page_sz == exp_size, "bad page size for flat %zu != %zu",
383 page_sz, exp_size);
384 CRITICAL(to == from + 1, "to=%x, from=%x", to, from);
385 if (flags & QHAT_CHECK_CONTENT && is_suboptimal) {
386 RETHROW(qhat_flat_check_consistency(hat, key_from, key_to, memory,
387 is_suboptimal));
388 }
389 } else {
390 CRITICAL(page_sz == 1, "bad page size for node, %zu != 1", page_sz);
391 CRITICAL(to == from + 1, "to=%x, from=%x", to, from);
392 CRITICAL(depth < QHAT_DEPTH_MAX, "depth=%u", depth);
393 RETHROW(qhat_node_check_consistency(hat, key_from, depth + 1,
394 memory, QHAT_COUNT, flags,
395 is_suboptimal));
396 }
397
398 return 0;
399}

402static __must_check__ int
403qhat_node_check_consistency(qhat_t *hat, uint32_t key, uint32_t depth,
404 qhat_node_memory_t memory, int c,
405 int flags, bool *nullable is_suboptimal)
406{
407 bool non_null = false;
408 qhat_node_t previous = QHAT_NULL_NODE;
409 int from = 0;
410 int res = 0;
411
412 for (int i = 0; i <= c; i++) {
413 qhat_node_t current;
414
415 if (i < c) {
416 current = memory.nodes[i];
417 if (current.value) {
418 non_null = true;
419 }
420 if (current.value == previous.value) {
421 continue;
422 }
423 } else {
424 current = QHAT_NULL_NODE;
425 }
426 if (qhat_node_check_child(hat, key, from, i, depth, previous, flags,
427 is_suboptimal) < 0)
428 {
429 if ((flags & QHAT_CHECK_REPAIR_NODES)) {
430 logger_notice(&hat->qps->logger,
431 "removing broken node [%x -> %x] depth=%u",
432 from, i, depth);
433 for (int pos = from; pos < i; pos++) {
434 /* Not sure we can have several nodes to patch. I would
435 * expect range of identical nodes to be all null. Should
436 * be checked.
437 */
438 memory.nodes[pos] = QHAT_NULL_NODE;
439 }
440 hat->gen.s.struct_gen++;
441 /* FIXME We should probably optimize the QHAT again because it
442 * may be suboptimal now. */
443 /* FIXME If the QHAT has stats then they are probably outdated
444 * now. */
445 } else {
446 res = -1;
447 if (!(flags & QHAT_CHECK_FULL_SCAN)) {
448 return res;
449 }
450 }
451 }
452
453 previous = current;
454 from = i;
455 }
456
457 if ((flags & QHAT_CHECK_CONTENT) && c == QHAT_COUNT) {
458 SUBOPTIMAL(non_null);
459 }
460
461 return res;
462}

708static void qhat_create_leaf(qhat_path_t *path)
709{
710 qhat_node_memory_t memory;
711 qhat_node_memory_t parent;
712 qhat_node_t node;
713 uint32_t idx;
714 uint32_t max;
715 assert (likely(PATH_NODE(path).value == 0));
716 node = qhat_alloc_leaf(path->hat, true);
717 PATH_NODE(path) = node;
718 assert (likely(PATH_NODE(path).value != 0));
719
720 memory = qhat_node_w_deref(path);
721 parent = qhat_parent_w_deref(path, &max);
722
723 idx = PATH_IN_PARENT_IDX(path);
724 memory.compact->count = 0;
725 memory.compact->parent_left = 0;
726 memory.compact->parent_right = max;
727 for (uint32_t i = idx; i > 0; i--) {
728 if (parent.nodes[i - 1].value == 0) {
729 parent.nodes[i - 1] = node;
730 } else {
731 memory.compact->parent_left = i;
732 break;
733 }
734 }
735 for (uint32_t i = idx; i < max; i++) {
736 if (parent.nodes[i].value == 0) {
737 parent.nodes[i] = node;
738 } else {
739 memory.compact->parent_right = i;
740 break;
741 }
742 }

745static NEVER_INLINE
746void qhat_split_leaf(qhat_path_t *path)
747{
748 qhat_node_t node;
749
750 assert (path->depth < QHAT_DEPTH_MAX);
751 assert (PATH_NODE(path).value != 0);
752 assert (PATH_NODE(path).leaf);
753
754 if (qhat_node_is_pure(path)) {
755 /* The leaf is referenced by a single pointer from the parent
756 * thus, in order to split this leaf, we must introduce new
757 * pointer. So, we add a intermediate node between the parent and
758 * the leaf with all its pointer going to the leaf.
759 *
760 * current state:
761 * parent[i] ---------> leaf
762 *
763 * action:
764 * parent[i] ----> dispatch 0 ---------> leaf
765 * 1 -------/
766 * ... ----/
767 * QHAT_COUNT -'
768 */
769 qhat_node_t new_node = qhat_alloc_node(path->hat);
770 qhat_node_memory_t memory;
771 assert (path->depth < QHAT_DEPTH_MAX - 1);
772 node = PATH_NODE(path);
773 qhat_update_parent_pure(path, new_node);
774 PATH_NODE(path) = new_node;
775
776 memory = qhat_node_w_deref(path);
777 for (uint32_t i = 0; i < QHAT_COUNT; i++) {
778 memory.nodes[i] = node;
779 }
780 path->depth++;
781 PATH_NODE(path) = node;
782 if (node.leaf && node.compact) {
783 memory = qhat_node_w_deref(path);
784 memory.compact->parent_left = 0;
785 memory.compact->parent_right = QHAT_COUNT;
786 }
787
788 e_named_trace(2, "trie/insert/split", "add intermediate node %u "
789 "above node %u (depth: %d)",
790 new_node.page, node.page, path->depth);
791 PATH_STRUCTURE_CHANGED("trie/insert/split", path);
792 }
793
794 assert (path->depth < QHAT_DEPTH_MAX);
795 assert (PATH_NODE(path).value != 0);
796 assert (PATH_NODE(path).leaf);
797
798 {
799 qhat_node_memory_t memory = qhat_node_w_deref(path);
800 qhat_compacthdr_t *compact = memory.compact;
801 uint32_t count = compact->count;
802 uint32_t split;
803 uint32_t prefix = 0;
804 uint32_t sep;
805 split = memory.compact->keys[count / 2];
806 split = qhat_get_key_bits(path->hat, split, path->depth);
807 if (split == compact->parent_left) {
808 split++;
809 }
810
811 if (path->depth > 0) {
812 prefix = qhat_depth_prefix(path->hat, path->key,
813 path->depth - 1);
814 }
815 prefix |= qhat_lshift(path->hat, split, path->depth);
816 e_named_trace(4, "trie/insert/split", "key %x, splitting at prefix %x"
817 " (depth %d, split %x)", path->key, prefix, path->depth,
818 split);
819 sep = qhat_compact_lookup(memory.compact, 0, prefix);
820
821 if (sep == 0 || sep == count) {
822 uint32_t max = 0;
823 qhat_node_memory_t parent_memory = qhat_parent_w_deref(path, &max);
824 uint32_t prev_parent_start = compact->parent_left;
825 uint32_t prev_parent_end = compact->parent_right;
826
827 split = memory.compact->keys[count - 1];
828 split = qhat_get_key_bits(path->hat, split, path->depth);
829 if (split + 1 != compact->parent_right) {
830 for (uint32_t i = split + 1; i < compact->parent_right; i++) {
831 parent_memory.u32[i] = 0;
832 }
833 compact->parent_right = split + 1;
834 }
835 split = memory.compact->keys[0];
836 split = qhat_get_key_bits(path->hat, split, path->depth);
837 if (split != compact->parent_left) {
838 for (uint32_t i = compact->parent_left; i < split; i++) {
839 parent_memory.u32[i] = 0;
840 }
841 compact->parent_left = split;
842 }
843 e_named_trace(3, "trie/insert/split",
844 "split at value %u generates a single block "
845 " changing parent pointers: [%u->%u] -> [%u->%u]",
846 split, prev_parent_start, prev_parent_end -1,
847 compact->parent_left, compact->parent_right - 1);
848 assert (compact->parent_left != prev_parent_start ||
849 compact->parent_right != prev_parent_end);
850 assert (compact->parent_left < compact->parent_right);
851 assert (compact->parent_left >= prev_parent_start);
852 assert (compact->parent_right <= prev_parent_end);
853
854 if (path->depth == QHAT_DEPTH_MAX - 1 &&
855 count > path->hat->desc->split_compact_threshold &&
856 compact->parent_left == compact->parent_right + 1)
857 {
858 /* FIXME This code was not reachable before fixing a
859 * PATH_MAX/QHAT_DEPTH_MAX typo. As a consequence, enabling
860 * the following line should be done with caution and maybe it
861 * should just be removed.
862 *
863 * Anyway, missing the flattening there is very unlikely to
864 * cause troubles, because the 'split_compact_threshold' is
865 * set to 75% of the compact capacity, so we can wait for a
866 * subsequent insertion to flatten that leaf. */
867#if 0
868 (*path->hat->desc->flattenf)(path);
869#endif
870 }
871 } else {
872 uint32_t max = 0;
873 qhat_node_t new_node;
874 qhat_node_memory_t new_memory;
875 qhat_node_memory_t parent_memory = qhat_parent_w_deref(path, &max);
876
877 e_named_trace(3, "trie/insert/split", "split [%u-%u] at %u "
878 "(%d elements, depth %d)",
879 compact->parent_left, compact->parent_right - 1,
880 split, sep, path->depth);
881
882 if (count - sep > path->hat->desc->split_compact_threshold
883 && path->depth == QHAT_DEPTH_MAX - 1
884 && split + 1 == compact->parent_right) {
885 /* Create a new flat leaf */
886 new_node = qhat_alloc_leaf(path->hat, false);
887 parent_memory.nodes[split] = new_node;
888 PATH_NODE(path) = new_node;
889
890 compact->parent_right--;
891 new_memory = qhat_node_w_deref(path);
892
893#define CASE(Size, Compact, Flat) \
894 Compact->count = sep; \
895 for (uint32_t i = sep; i < count; i++) { \
896 uint32_t key = Compact->keys[i]; \
897 key &= path->hat->desc->leaf_index_mask; \
898 new_memory.u##Size[key] = Compact->values[i]; \
899 }
900
901 QHAT_VALUE_LEN_SWITCH(path->hat, memory, CASE);
902#undef CASE
903 MOVED_TO_NEW_FLAT(path, count - sep);
904 } else
905 if (sep > path->hat->desc->split_compact_threshold
906 && path->depth == QHAT_DEPTH_MAX - 1
907 && (int)split == compact->parent_left + 1) {
908 /* Replace the current leaf by a new flat leaf */
909 new_node = qhat_alloc_leaf(path->hat, false);
910 parent_memory.nodes[compact->parent_left] = new_node;
911 PATH_NODE(path) = new_node;
912
913 compact->parent_left++;
914 new_memory = qhat_node_w_deref(path);
915
916#define CASE(Size, Compact, Flat) \
917 for (uint32_t i = 0; i < (uint32_t)sep ; i++) { \
918 uint32_t key = Compact->keys[i]; \
919 key &= path->hat->desc->leaf_index_mask; \
920 new_memory.u##Size[key] = Compact->values[i]; \
921 } \
922 Compact->count = count - sep; \
923 p_move(Compact->keys, Compact->keys + sep, count - sep); \
924 p_move(Compact->values, Compact->values + sep, count - sep);
925
926 QHAT_VALUE_LEN_SWITCH(path->hat, memory, CASE);
927#undef CASE
928 MOVED_TO_NEW_FLAT(path, sep);
929 } else {
930 /* Create a new compact leaf */
931 new_node = qhat_alloc_leaf(path->hat, true);
932 for (uint32_t i = split; i < compact->parent_right; i++) {
933 parent_memory.nodes[i] = new_node;
934 }
935 PATH_NODE(path) = new_node;
936 new_memory = qhat_node_w_deref(path);
937
938#define CASE(Size, Compact, Flat) \
939 qhat_compact##Size##_t *new_compact = new_memory.compact##Size;\
940 Compact->count = sep; \
941 new_compact->count = count - sep; \
942 new_compact->parent_left = split; \
943 new_compact->parent_right = compact->parent_right; \
944 Compact->parent_right = split; \
945 p_copy(new_compact->values, Compact->values + sep, count - sep);\
946 p_copy(new_compact->keys, Compact->keys + sep, count - sep);
947
948 QHAT_VALUE_LEN_SWITCH(path->hat, memory, CASE);
949#undef CASE
950 }
951 }
952 }

956static void qhat_optimize_parent(qhat_path_t *path)
957{
958 uint32_t max;
959 qhat_node_memory_t memory = qhat_parent_w_deref(path, &max);
960 uint32_t idx = PATH_IN_PARENT_IDX(path);
961 uint32_t count = 0;
962 bool changed = false;
963 qhat_node_memory_t child;
964
965 /* Node has been removed, try to find a leaf before or after
966 * this one.
967 */
968 if (PATH_NODE(path).value == 0) {
969 uint32_t before_count = 0;
970 uint32_t before_idx = idx;
971 uint32_t after_count = 0;
972 uint32_t after_idx = idx;
973 for (uint32_t i = idx; i > 0; i--) {
974 if (memory.nodes[i - 1].leaf) {
975 PATH_NODE(path) = memory.nodes[i - 1];
976 before_count = qhat_node_count(path);
977 before_idx = i - 1;
978 break;
979 } else if (memory.nodes[i].value) {
980 break;
981 }
982 }
983 for (uint32_t i = idx + 1; i < max; i++) {
984 if (memory.nodes[i].leaf) {
985 PATH_NODE(path) = memory.nodes[i];
986 after_count = qhat_node_count(path);
987 after_idx = i - 1;
988 break;
989 } else if (memory.nodes[i].value) {
990 break;
991 }
992 }
993 if (before_count == 0 && after_count == 0) {
994 return;
995 } else
996 if (after_count == 0 || before_count <= after_count) {
997 idx = before_idx;
998 count = before_count;
999 } else {
1000 idx = after_idx;
1001 count = after_count;
1002 }
1003 PATH_NODE(path) = memory.nodes[idx];
1004 } else
1005 if (!PATH_NODE(path).leaf) {
1006 return;
1007 } else {
1008 count = qhat_node_count(path);
1009 }
1010
1011 if (!PATH_NODE(path).compact) {
1012 if (count < path->hat->desc->leaves_per_flat / 2
1013 && count < (uint32_t)(2 * path->hat->desc->leaves_per_compact) / 3) {
1014 (*path->hat->desc->unflattenf)(path);
1015 } else {
1016 return;
1017 }
1018 }
1019 child = qhat_node_w_deref(path);
1020
1021 for (uint32_t i = idx; i > 0; i--) {
1022 if (memory.nodes[i - 1].value == 0) {
1023 memory.nodes[i - 1] = PATH_NODE(path);
1024 child.compact->parent_left = i - 1;
1025 changed = true;
1026 } else {
1027 break;
1028 }
1029 }
1030 for (uint32_t i = idx + 1; i < max; i++) {
1031 if (memory.nodes[i].value == 0) {
1032 memory.nodes[i] = PATH_NODE(path);
1033 child.compact->parent_right = i + 1;
1034 changed = true;
1035 } else {
1036 break;
1037 }
1038 }
1039 if (changed) {
1040 PATH_STRUCTURE_CHANGED("trie/optimize", path);
1041 }
1042
1043 if (changed && memory.nodes[0].value == PATH_NODE(path).value
1044 && memory.nodes[max - 1].value == PATH_NODE(path).value
1045 && path->depth > 0) {
1046 qhat_node_t leaf = PATH_NODE(path);
1047 path->depth--;
1048 qhat_update_parent_pure(path, leaf);
1049 e_named_trace(2, "trie/optimize", "removing dispatch node %u",
1050 PATH_NODE(path).page);
1051 qhat_unmap_node(path->hat, PATH_NODE(path));
1052 child.compact->parent_left = PATH_IN_PARENT_IDX(path);
1053 child.compact->parent_right = child.compact->parent_left + 1;
1054 PATH_NODE(path) = leaf;
1055 PATH_STRUCTURE_CHANGED("trie/optimize", path);
1056 qhat_optimize_parent(path);
1057 }
1058}

1060static void qhat_merge_nodes(qhat_path_t *path, qhat_node_t second)
1061{
1062 qhat_path_t second_path = *path;
1063 qhat_node_memory_t first_memory;
1064 qhat_node_const_memory_t second_memory;
1065
1066 assert (PATH_NODE(path).leaf);
1067 assert (PATH_NODE(path).compact);
1068 assert (second.leaf);
1069 assert (second.compact);
1070
1071 PATH_NODE(&second_path) = second;
1072 first_memory = qhat_node_w_deref(path);
1073 second_memory = qhat_node_deref(&second_path);
1074
1075#define CASE(Size, Compact, Flat) \
1076 qhat_compact##Size##_t *first_compact = first_memory.compact##Size; \
1077 p_copy(first_compact->keys + first_compact->count, \
1078 Compact->keys, Compact->count); \
1079 p_copy(first_compact->values + first_compact->count, \
1080 Compact->values, Compact->count); \
1081 first_compact->count += Compact->count;
1082
1083 QHAT_VALUE_LEN_SWITCH(path->hat, second_memory, CASE);
1084#undef CASE
1085
1086 e_named_trace(2, "trie/optimize/merge", "merged leaf %u in %u",
1087 second.page, PATH_NODE(path).page);
1088 qhat_unmap_node(path->hat, second);
1089}

1091static NEVER_INLINE
1092void qhat_optimize(qhat_path_t *path)
1093{
1094 /* Kill useless branches */
1095 while ((!PATH_NODE(path).leaf || PATH_NODE(path).compact)
1096 && qhat_node_is_empty(path))
1097 {
1098 qhat_node_t node = PATH_NODE(path);
1099 qhat_update_parent(path, QHAT_NULL_NODE);
1100 e_named_trace(2, "trie/optimize", "removing empty bucket %u",
1101 node.page);
1102 qhat_unmap_node(path->hat, node);
1103 PATH_STRUCTURE_CHANGED("trie/optimize", path);
1104 PATH_NODE(path) = QHAT_NULL_NODE;
1105 if (path->depth > 0) {
1106 path->depth--;
1107 } else {
1108 break;
1109 }
1110 }
1111
1112 qhat_optimize_parent(path);
1113 if (!PATH_NODE(path).leaf || !PATH_NODE(path).compact
1114 || qhat_node_count(path) >= path->hat->desc->leaves_per_flat / 2) {
1115 return;
1116 }
1117
1118 {
1119 const uint32_t limit = path->hat->desc->split_compact_threshold;
1120 uint32_t max;
1121 qhat_node_const_memory_t node_memory = qhat_node_deref(path);
1122 qhat_node_memory_t memory = qhat_parent_w_deref(path, &max);
1123 uint32_t from_idx, to_idx;
1124 uint32_t count = node_memory.compact->count;
1125 qhat_node_t node, previous_node;
1126 qhat_path_t new_path = *path;;
1127
1128 from_idx = node_memory.compact->parent_left;
1129 to_idx = node_memory.compact->parent_right;
1130
1131 node = PATH_NODE(path);
1132 while (from_idx > 0) {
1133 uint32_t current_count;
1134 qhat_node_t current_node = memory.nodes[from_idx - 1];
1135
1136 assert (current_node.value != node.value);
1137 if (!current_node.leaf || !current_node.compact) {
1138 break;
1139 }
1140 PATH_NODE(&new_path) = current_node;
1141 node_memory = qhat_node_deref(&new_path);
1142 current_count = node_memory.compact->count;
1143
1144 if (current_count + count > limit) {
1145 break;
1146 }
1147 from_idx = node_memory.compact->parent_left;
1148 count += current_count;
1149 node = current_node;
1150 }
1151
1152 node = PATH_NODE(path);
1153 while (to_idx < max) {
1154 uint32_t current_count;
1155 qhat_node_t current_node = memory.nodes[to_idx];
1156
1157 assert (current_node.value != node.value);
1158 if (!current_node.leaf || !current_node.compact) {
1159 break;
1160 }
1161 PATH_NODE(&new_path) = current_node;
1162 node_memory = qhat_node_deref(&new_path);
1163 current_count = node_memory.compact->count;
1164
1165 if (current_count + count > limit) {
1166 break;
1167 }
1168 to_idx = node_memory.compact->parent_right;
1169 count += current_count;
1170 node = current_node;
1171 }
1172
1173 if (memory.nodes[from_idx].value == memory.nodes[to_idx - 1].value) {
1174 return;
1175 }
1176 e_named_trace(3, "trie/optimize/merge", "merging siblings of %u "
1177 "from parent id %u to parent id %u (depth %u, max %u)",
1178 PATH_NODE(path).page, from_idx, to_idx, path->depth, max);
1179
1180 PATH_NODE(path) = previous_node = memory.nodes[from_idx];
1181 for (uint32_t i = from_idx + 1; i < to_idx; i++) {
1182 qhat_node_t current_node = memory.nodes[i];
1183 if (current_node.value != previous_node.value
1184 && current_node.value != PATH_NODE(path).value) {
1185 qhat_merge_nodes(path, current_node);
1186 }
1187 previous_node = current_node;
1188 memory.nodes[i] = PATH_NODE(path);
1189 }
1190 assert (PATH_NODE(path).compact);
1191 memory = qhat_node_w_deref(path);
1192 memory.compact->parent_left = from_idx;
1193 memory.compact->parent_right = to_idx;
1194 PATH_STRUCTURE_CHANGED("trie/optimize/merge", path);
1195 }
1196}

1274void qhat_clear(qhat_t *hat)
1275{
1276 qhat_node_const_memory_t root;
1277 bool do_stats = hat->do_stats;
1278
1279 qps_hptr_w_deref(hat->qps, &hat->root_cache);
1280
1281 /* Disable statistics, this will avoid useless updates of the counters
1282 * since we already know what their final values should be (0, 0, 0, 0...)
1283 */
1284 hat->do_stats = false;
1285 hat->root->do_stats = false;
1286
1287 root.nodes = hat->root->nodes;
1288 e_named_trace(3, "trie/clear", "wipe start root");
1289 qhat_wipe_dispatch_node(hat, root, hat->desc->root_node_count);
1290 e_named_trace(3, "trie/clear", "wipe done root");
1291 p_clear(hat->root->nodes, QHAT_ROOTS);
1292 hat->gen.s.struct_gen++;
1293
1294 if (hat->root->is_nullable) {
1295 qps_bitmap_clear(&hat->bitmap);
1296 }
1297
1298 /* Reenable stats if they are computed, since we cleared the hat, it will
1299 * only reset the counters.
1300 */
1301 if (do_stats) {
1302 qhat_compute_counts(hat, true);
1303 }

1306void qhat_destroy(qhat_t *hat)
1307{
1308 if (hat != NULL) {
1309 qhat_node_const_memory_t root;
1310
1311 qps_hptr_deref(hat->qps, &hat->root_cache);
1312
1313 /* Disable stats during destruction of the trie. There's no need to
1314 * update hat->root->do_stats since its not read during and thus avoid
1315 * a useless w_deref.
1316 */
1317 hat->do_stats = false;
1318 root.nodes = hat->root->nodes;

1490#define QHAT_TREE_EN_KEY_REMOVED 1

1492/* Fixup 'pos' and 'count' if the compact is modified. */
1493static int
1494qhat_tree_enumerator_fixup_compact_pos(qhat_tree_enumerator_t *en)
1495{
1496 assert(en->compact);
1497
1498 en->count = en->memory.compact->count;
1499
1500 if (qhat_tree_enumerator_compact_pos_is_valid(en)) {
1501 /* Nothing to do.
1502 * The compact *might* have been modified but 'pos' is still right. */
1503 return 0;
1504 }
1505
1506 /* The compact has been modified. Update the position. */
1507 en->pos = qhat_compact_lookup(en->memory.compact, 0, en->key);
1508
1509 if (!qhat_tree_enumerator_compact_pos_is_valid(en)) {
1510 /* The key has been removed from the compact.
1511 * We're already at the next key. */
1512 return QHAT_TREE_EN_KEY_REMOVED;

1515 return 0;

1520{

1523 }

1534 en->key = key;

1537 } else {

1542 QHAT_TREE_EN_KEY_REMOVED)

1551/* Find the key associated to the current position of the enumerator.
1552 * Update the attribute 'key' or end the enumerator. */
1553static void qhat_tree_enumerator_find_entry(qhat_tree_enumerator_t *en)
1554{
1555 qhat_t *hat = en->path.hat;
1556 uint32_t new_key = en->path.key;
1557 uint32_t next = 1;
1558 uint32_t shift;
1559
1560 if (en->compact) {
1561 if (en->pos < en->count) {
1562 /* We're still in the current compact. We're done. */
1563 en->key = en->memory.compact->keys[en->pos];
1564 return;
1565 }
1566 next = en->memory.compact->parent_right;
1567 next -= qhat_get_key_bits(hat, new_key, en->path.depth);
1568 } else {
1569 if (!en->is_nullable) {
1570 /* Go to the first non-null flat element. */
1571#define CASE(Size, Compact, Flat) \
1572 while (en->pos < en->count && IS_ZERO(Size, Flat[en->pos])) { \
1573 en->pos++; \
1574 }
1575 QHAT_VALUE_LEN_SWITCH(hat, en->memory, CASE);
1576#undef CASE
1577 }
1578
1579 if (en->pos < en->count) {
1580 /* We're still in the current flat. We're done. */
1581 en->key = en->path.key | en->pos;
1582 return;
1583 }
1584 }
1585
1586 /* We're after the current compact/flat.
1587 * We've got to go to the next leaf. */
1588
1589 shift = qhat_depth_shift(hat, en->path.depth);
1590 if (shift == 32) {
1591 /* There is no next leaf. The enumerator is done. */
1592 en->end = true;
1593 return;
1594 }
1595 new_key += next << shift;
1596 qhat_tree_enumerator_dispatch_up(en, en->path.key, new_key);
1597}

1599static void qhat_tree_enumerator_find_entry_from(qhat_tree_enumerator_t *en,
1600 uint32_t key)
1601{
1602 if (en->compact) {
1603 en->pos = qhat_compact_lookup(en->memory.compact, en->pos, key);
1604 } else {
1605 en->pos = key % en->count;
1606 }
1607
1608 qhat_tree_enumerator_find_entry(en);
1609}

1612static void qhat_tree_enumerator_find_up_down(qhat_tree_enumerator_t *en,
1613 uint32_t key)
1614{
1615 qhat_tree_enumerator_find_root(en, key);
1616}
1617
1618static void qhat_tree_enumerator_find_down_up(qhat_tree_enumerator_t *en,
1619 uint32_t key)
1620{
1621 qhat_t *hat = en->path.hat;
1622 uint32_t last_key = en->path.key;
1623 const uint32_t diff = key ^ last_key;
1624 uint32_t shift;
1625
1626 assert (key >= en->path.key);
1627 if (key == en->path.key) {
1628 return;
1629 }
1630
1631 shift = qhat_depth_shift(hat, en->path.depth);
1632 if (shift == 32) {
1633 /* The current leaf is a compact (ie. not a flat) because flats
1634 * appears only at maximum depth and shift 32 means depth == 0. */
1635 assert(en->compact);
1636
1637 if (en->memory.compact->keys[en->memory.compact->count - 1] < key) {
1638 en->end = true;
1639 } else {
1640 qhat_tree_enumerator_find_entry_from(en, key);
1641 }
1642 return;
1643 }
1644 if (en->compact) {
1645 uint32_t next = en->memory.compact->parent_right;
1646 next -= qhat_get_key_bits(hat, en->path.key, en->path.depth);
1647 last_key += next << shift;
1648 } else {
1649 last_key += 1 << shift;
1650 }
1651
1652 if (key < last_key) {
1653 qhat_tree_enumerator_find_entry_from(en, key);
1654 } else
1655 if (qhat_get_key_bits(hat, diff, 0)) {
1656 qhat_tree_enumerator_find_root(en, key);
1657 } else
1658 if (en->path.depth >= 1 && qhat_get_key_bits(hat, diff, 1)) {
1659 en->path.depth = 0;
1660 qhat_tree_enumerator_find_node(en, key);
1661 } else
1662 if (en->path.depth >= 2 && qhat_get_key_bits(hat, diff, 2)) {
1663 en->path.depth = 1;
1664 qhat_tree_enumerator_find_node(en, key);
1665 } else {
1666 qhat_tree_enumerator_find_entry_from(en, key);
1667 }
1668}

1670/* Similar to 'qhat_enumerator_next()' but only apply to the trie. */
1671uint32_t qhat_tree_enumerator_next(qhat_tree_enumerator_t *en, bool safe)
1672{
1673 en->key_was_removed = false;
1674 if (safe) {
1675 if (unlikely(!qhat_path_is_sync(&en->path))) {
1676 en->key++;
1677 qhat_tree_enumerator_refresh_path(en);
1678 return en->key;
1679 }
1680 qhat_path_sync_write_access(&en->path);
1681 } else {
1682 /* The caller should probably have used the safe version. */
1683 assert(qhat_path_is_fully_sync(&en->path));
1684 }
1685
1686 if (safe && en->compact &&
1687 qhat_tree_enumerator_fixup_compact_pos(en) ==
1688 QHAT_TREE_EN_KEY_REMOVED)
1689 {
1690 /* The key was removed from the compact, we're already positioned on
1691 * the next entry. Nothing to do. */
1692 } else {
1693 en->pos++;
1694 }
1695
1696 qhat_tree_enumerator_find_entry(en);
1697
1698 return en->key;
1699}

1701/* Similar to 'qhat_enumerator_go_to()' but only apply to the trie. */
1702void qhat_tree_enumerator_go_to(qhat_tree_enumerator_t *en, uint32_t key,
1703 bool safe)
1704{
1705 /* The tree enumerator should only go forward. */
1706 assert(key >= en->key);
1707
1708 if (en->end) {
1709 return;
1710 }
1711 if (key == en->key) {
1712 if (!safe) {
1713 /* The key is already the current one and the qhat is not supposed
1714 * to have changed. Nothing to do. */
1715 return;
1716 }
1717 } else if (key == en->key + 1) {
1718 qhat_tree_enumerator_next(en, safe);
1719 return;
1720 }
1721 en->key_was_removed = false;
1722 if (safe) {
1723 if (unlikely(!qhat_path_is_sync(&en->path))) {
1724 qhat_tree_enumerator_find_up_down(en, key);
1725 return;
1726 }
1727
1728 if (en->compact) {
1729 /* Refresh the attributes 'pos' and 'count' so that
1730 * 'qhat_tree_enumerator_find_down_up()' can work properly. */
1731 qhat_tree_enumerator_fixup_compact_pos(en);
1732 }
1733 qhat_path_sync_write_access(&en->path);
1734 } else {
1735 /* The caller should probably have used the safe version. */
1736 assert(qhat_path_is_fully_sync(&en->path));
1737 }
1738
1739 qhat_tree_enumerator_find_down_up(en, key);
1740}

1744 bool safe)
1745{

1747
1748 if (en->bitmap.end) {
1749 en->end = true;
1750 return;
1751 }
1752 en->key = en->bitmap.key.key;
1753 if (value) {

1755 * is kept untouched as long as we don't need the associated value:
1756 * the bitmap is sufficient if we only want to iterate on the keys. */
1757
1758 if (!en->trie.end && en->trie.key < en->key) {
1759 /* Make the tree enumerator catchup with the bitmap enumerator so
1760 * that we can get or update the associated value. */
1761 qhat_tree_enumerator_go_to(&en->trie, en->key, safe);
1762 }
1763 }

1864 * match the enumerator key because the entries with null values
1865 * exist in the presence bitmap but not in the tree.

1876/* After call:
1877 * - memory set to the current leaf
1878 * - compact updated
1879 * - pos set to the position of the key
1880 * - key updated
1881 * - value_pos updated
1882 */
1883static void qhat_tree_enumerator_enter_leaf(qhat_tree_enumerator_t *en,
1884 uint32_t key)
1885{
1886 en->memory = qhat_node_deref(&en->path);
1887
1888 if (PATH_NODE(&en->path).compact) {
1889 en->compact = true;
1890 en->count = en->memory.compact->count;
1891#define CASE(Size, Compact, Flat) en->value_tab = Compact->values;
1892 QHAT_VALUE_LEN_SWITCH(en->path.hat, en->memory, CASE);
1893#undef CASE
1894 } else {
1895 en->compact = false;
1896 en->count = en->path.hat->desc->leaves_per_flat;
1897#define CASE(Size, Compact, Flat) en->value_tab = Flat;
1898 QHAT_VALUE_LEN_SWITCH(en->path.hat, en->memory, CASE);
1899#undef CASE

1901 en->pos = 0;
1902 qhat_tree_enumerator_find_entry_from(en, key);

1905/* Guaranteed to either enter a leaf or end the enumerator. */
1906void qhat_tree_enumerator_find_root(qhat_tree_enumerator_t *en, uint32_t key)
1907{
1908 qhat_t *hat = en->path.hat;
1909 uint32_t root = qhat_get_key_bits(hat, key, 0);
1910 ssize_t i;
1911
1912 /* We're going to refresh the whole path so the structure generation can
1913 * be updated. */
1914 en->path.gen = hat->gen;
1915
1916 en->path.depth = 0;
1917 en->path.key = 0;
1918
1919 i = scan_non_zero32(&hat->root->nodes[0].value, root,
1920 hat->desc->root_node_count);
1921 if (i >= 0) {
1922 PATH_NODE(&en->path) = hat->root->nodes[i];
1923 en->path.key = qhat_lshift(hat, i, 0);
1924
1925 if (root != i) {
1926 key = en->path.key;
1927 }
1928
1929 if (PATH_NODE(&en->path).leaf) {
1930 qhat_tree_enumerator_enter_leaf(en, key);
1931 } else {
1932 qhat_tree_enumerator_find_node(en, key);
1933 }
1934 } else {
1935 en->end = true;
1936 }
1937}

1939/* Guaranteed to either enter a leaf or end the enumerator. */
1940void qhat_tree_enumerator_dispatch_up(qhat_tree_enumerator_t *en, uint32_t key,
1941 uint32_t new_key)
1942{
1943 qhat_t *hat = en->path.hat;
