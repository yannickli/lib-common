271static inline int win1252_to_gsm7(uint8_t u8, int unknown, gsm_conv_plan_t plan)
272{
273 int c = __win1252_to_gsm7[u8];
274
275 if (unlikely(c < 0))
276 return unknown;
277 if (plan == GSM_DEFAULT_PLAN && c > 0xff)
278 return unknown;
279 return c;
280}
281
282int unicode_to_gsm7(int c, int unknown, gsm_conv_plan_t plan)
283{
284 assert (plan != GSM_CIMD_PLAN);
285
286 if ((unsigned)c <= 0xff) {
287 return win1252_to_gsm7(c, unknown, plan);
288 }
289
290 if (plan == GSM_EXTENSION_PLAN) {
291 switch (c) {
292 case 0x20AC: return 0x1B65; /* EURO */
293 default: break;
294 }
295 }
296
297 switch (c) {
298 case 0x0394: return 0x10; /* GREEK CAPITAL LETTER DELTA */
299 case 0x03A6: return 0x12; /* GREEK CAPITAL LETTER PHI */
300 case 0x0393: return 0x13; /* GREEK CAPITAL LETTER GAMMA */
301 case 0x039B: return 0x14; /* GREEK CAPITAL LETTER LAMDA */
302 case 0x03A9: return 0x15; /* GREEK CAPITAL LETTER OMEGA */
303 case 0x03A0: return 0x16; /* GREEK CAPITAL LETTER PI */
304 case 0x03A8: return 0x17; /* GREEK CAPITAL LETTER PSI */
305 case 0x03A3: return 0x18; /* GREEK CAPITAL LETTER SIGMA */
306 case 0x0398: return 0x19; /* GREEK CAPITAL LETTER THETA */
307 case 0x039E: return 0x1A; /* GREEK CAPITAL LETTER XI */

320{
321 uint8_t hash;
322 struct cimd_esc_table const *esc;
323
324 if (p + 2 > end)
325 return -1;
326
327 if (p[0] == 'X') {
328 int c;
329
330 if (p[1] != 'X')
331 return -1;
332 p += 2;
333 if (p + 1 > end)
334 return -1;
335
336 switch (*p++) {
337 case 'e': c = 0x20AC; break; /* EURO symbol */
338 case '(': c = '{'; break;
339 case ')': c = '}'; break;
340 case '\n': c = '\n'; break; /* should be '\f' ? */
341 case '<': c = '['; break;
342 case '>': c = ']'; break;
343 case '=': c = '~'; break;
344 case '/': c = '\\'; break;
345 case '_':

357 }
358 /* FALLTHROUGH */
359 default:

367 esc = &cimd_esc_map[hash];
368 if (esc->c1 == p[0] && esc->c2 == p[1] && esc->unicode) {
369 *out = p + 2;

515/* Decode a hex encoded (IRA) char array into UTF-8 at end of sb */
516int sb_conv_from_gsm_hex(sb_t *sb, const void *data, int slen)
517{
518 const char *p = data, *end = p + slen;
519 sb_t orig = *sb;
520
521 if (slen & 1)
522 return -1;
523
524 sb_grow(sb, slen / 2 + 4);
525
526 while (p < end) {
527 int c = hexdecode(p);
528
529 p += 2;
530 if (c < 0 || c > 0x7f)
531 goto error;
532
533 if (c == 0x1b) {
534 if (p == end)
535 goto error;
536 c = hexdecode(p);
537 p += 2;
538 if (c < 0 || c > 0x7f)
539 goto error;
540 c |= 0x80;
541 }
542 c = gsm7_to_unicode(c, '.');
543 sb_adduc(sb, c);
544 }
545 return 0;
546
547 error:
548 return __sb_rewind_adds(sb, &orig);
549}

551int sb_conv_from_gsm_plan(sb_t *sb, const void *data, int slen, int plan)
552{
553 const byte *p = data, *end = p + slen;
554 int nb_invalid = 0;
555
556 sb_grow(sb, slen + 4);
557
558 while (p < end) {
559 int c = *p++;
560
561 if (plan == GSM_CIMD_PLAN) {
562 if (c == '_') {
563 c = cimd_special_to_unicode(p, end, &p);
564 if (unlikely(c < 0))
565 goto invalid;
566 } else {
567 c = cimd_to_unicode(c, '.');
568 }
569 goto next;
570 }
571
572 if (c & 0x80)
573 goto invalid;
574
575 if (c == 0x1b) {
576 if (unlikely(plan != GSM_EXTENSION_PLAN))
577 goto invalid;
578 if (unlikely(p == end))
579 goto invalid;
580 c = *p++;
581 if (c & 0x80)
582 goto invalid;
583 c |= 0x80;
584 }
585 c = gsm7_to_unicode(c, '.');
586 goto next;
587
588 invalid:
589 nb_invalid++;
590 c = '.';
591
592 next:
593 sb_adduc(sb, c);
594 }
595 return nb_invalid;
596}

677 * write up to 8 septets in the 56 least significant bits of "pack"
678 * XXX: the tragic truth about septets is that writing 8 septets means we pass
679 * `7` as %len value, the same value we use for 7 septets, except that

701/*
702 * gsm_start points to the first octet of out->data holding 7-bits packed GSM
703 * data. sb_conv_to_gsm7 is not able to be restartable, it assumes the
704 * octets between position %gsm_start and sb->len are 8bit aligned (UDH) and
705 * pads with 0 bits up to the next septet boundary.
706 *
707 * If we mean to have a restartable API, we need to know the current amount of
708 * septets written since %gsm_start since we have no way to know if when we
709 * have for example gsm->len - gsm_start == 7 if 7 or 8 septets are written.
710 *
711 * For us here, we know that if gsm->len - gsm_start is 7, 7 _octets_ of UDH
712 * are written and no padding is needed to be aligned on the next septet
713 * boundary.
714 *
715 * %max_len is the maximum length of the output in octets. If the size written
716 * is bigger than the size given, it returns an error.
717 * If -1 is given, no check is done.
718 */
719int sb_conv_to_gsm7(sb_t *out, int gsm_start, const char *utf8, int unknown,
720 gsm_conv_plan_t plan, int max_len)
721{
722 uint64_t pack = 0;
723 int septet = (out->len - gsm_start) % 7;
724 int current_len = 0;
725
726 if (septet) {
727 pack = get_gsm7_pack(out->data + out->len - septet, septet);
728 sb_shrink(out, septet);
729 septet++;
730 }
731
732 for (;;) {
733 int c = utf8_getc(utf8, &utf8);
734
735 if (c <= 0) {
736 int len = 8 * (out->len - gsm_start) / 7;
737
738 /* XXX: septet is always < 8 means septet is the number of bytes
739 * to write as explain for put_gsm_pack
740 */
741 current_len += septet;
742 if (max_len >= 0 && current_len > max_len) {
743 return -1;
744 }
745 put_gsm_pack(out, pack, septet);
746 return c < 0 ? c : len + septet;
747 }
748
749 c = RETHROW(unicode_to_gsm7(c, unknown, plan));
750 if (c > 0xff) {
751 pack |= ((uint64_t)(c >> 8) << (7 * septet));
752 if (++septet == 8) {
753 current_len += 7;
754 if (max_len >= 0 && current_len > max_len) {
755 return -1;
756 }
757 put_gsm_pack(out, pack, 7);
758 septet = 0;
759 pack = 0;
760 }
761 }
762
763 pack |= ((uint64_t)(c & 0x7f) << (7 * septet));
764 if (++septet == 8) {
765 current_len += 7;
766 if (max_len >= 0 && current_len > max_len) {
767 return -1;
768 }
769 put_gsm_pack(out, pack, 7);
770 septet = 0;
771 pack = 0;
772 }
773 }
774}

776static int decode_gsm7_pack(sb_t *out, uint64_t pack, int nbchars, int c)
777{
778 for (int i = 0; i < nbchars; i++) {
779 c |= pack & 0x7f;
780 pack >>= 7;
781 if (c == 0x1b) {
782 c = 0x80;
783 } else {
784 c = gsm7_to_unicode(c, '.');
785 sb_adduc(out, c);
786 c = 0;
787 }
788 }
789 return c;
790}

792/*
793 * %gsmlen is the number of septets in %_src.
794 *
795 * The caller has to ensure %udhlen (in octets) and %gsmlen and the actual
796 * size of the %_src buffer make sense:
797 * IOW that %udhlen is smaller or equal to the size of %_src,
798 * which in turn must be equal to (7 * %gsmlen + 7) / 8.
799 */
800int sb_conv_from_gsm7(sb_t *out, const void *_src, int gsmlen, int udhlen)
801{
802 const char *src = _src;
803 int c = 0;
804
805 sb_grow(out, (gsmlen - udhlen) * 2);
806
807 src += udhlen - (udhlen % 7);
808 gsmlen -= 8 * (udhlen / 7);
809 if (udhlen % 7) {
810 /* udh overlaps up to the next (hence +1) septet boundary included */
811 int overlap = (udhlen % 7) + 1;
812 uint64_t pack;
813
814 if (gsmlen >= 8) {
815 pack = get_gsm7_pack(src, 7) >> (7 * overlap);
816 c = decode_gsm7_pack(out, pack, 8 - overlap, c);
817 } else {
818 pack = get_gsm7_pack(src, gsmlen) >> (7 * overlap);
819 c = decode_gsm7_pack(out, pack, gsmlen - overlap, c);
820 goto done;
821 }
822 src += 7;
823 gsmlen -= 8;
824 }
825
826 for (; gsmlen >= 8; src += 7, gsmlen -= 8) {
827 c = decode_gsm7_pack(out, get_gsm7_pack(src, 7), 8, c);
828 }
829 if (gsmlen) {
830 c = decode_gsm7_pack(out, get_gsm7_pack(src, gsmlen), gsmlen, c);
831 }
832 done:
833 return c ? -1 : 0;
834}
