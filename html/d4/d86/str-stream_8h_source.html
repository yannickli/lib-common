28/*
29 * pstream_t's are basically a pointer and a length
30 * and are meant to simplify parsing a lot
31 *
32 * Those structures are not meant to be allocated ever.
33 */
34
35/*
36 * When you write a pstream function to parse a type_t, here is how you should
37 * proceed:
38 *
39 * Case 1:
40 * ~~~~~~
41 * When the size of the data is _easy_ to know before hand (we're reading a
42 * fixed amount of bytes for example) :
43 *
44 * - depending on the fact that `type_t` is scalar or complex, provide:
45 * type_t __ps_get_type(pstream_t *ps, ...);
46 * void __ps_get_type(pstream_t *ps, type_t *res, ...);
47 *
48 * This function shall assume that there is space to read the `type_t`.
49 *
50 * - provide then a safe ps_get_type function looking like this:
51 * int ps_get_type(pstream_t *ps, type_t *res, ...) {
52 * if (unlikely(!ps_has(ps, EASY_LEN_TO_COMPUTE)))
53 * return -1;
54 * res = __ps_get_type(ps, ...);
55 * return 0;
56 * }
57 *
58 * Case 2:
59 * ~~~~~~
60 * When the size of the data is hard to know, or too complex, you may only
61 * provide the safe interface. For example, ps_gets, we have to do a memchr
62 * to know the size, and it almost gives the result, you don't want to have
63 * both a __ps_gets and a ps_gets it's absurd.
64 *
65 *
66 * In any case, if the task to perform is complex, do not inline the function,
67 * and providing the unsafe version becomes useless too. The goal of the
68 * unsafe function is to be able to inline the sizes checks.
69 *
70 */
71
72typedef struct pstream_t {
73 union {
74 const void * nullable p;
75 const char * nullable s;
76 const byte * nullable b;
77 };
78 union {
79 const void * nullable p_end;
80 const char * nullable s_end;
81 const byte * nullable b_end;
82 };
83} pstream_t;

111static inline pstream_t ps_initptr(const void * nullable s,

113{
114 return (pstream_t){ { s }, { p } };

133static inline size_t ps_len(const pstream_t * nonnull ps)
134{
135 return ps->s_end - ps->s;

142static inline bool ps_done(const pstream_t * nonnull ps)
143{
144 return ps->p >= ps->p_end;
145}
146static inline bool ps_has(const pstream_t * nonnull ps, size_t len)
147{
148 return (ps_done(ps) && len == 0) || (size_t)(ps->s_end - ps->s) >= len;
149}
150static inline bool ps_contains(const pstream_t * nonnull ps,
151 const void * nonnull p)
152{
153 return p >= ps->p && p <= ps->p_end;
154}

156static inline bool ps_is_equal(pstream_t ps1, pstream_t ps2)
157{
158 return ps_len(&ps1) == ps_len(&ps2) && !memcmp(ps1.b, ps2.b, ps_len(&ps1));
159}
160static inline int ps_cmp(pstream_t ps1, pstream_t ps2)
161{
162 size_t len = MIN(ps_len(&ps1), ps_len(&ps2));
163 return memcmp(ps1.b, ps2.b, len) ?: CMP(ps_len(&ps1), ps_len(&ps2));
164}

207static inline bool ps_memcaseequal(const pstream_t * nonnull ps,
208 const char * nonnull s, size_t len)
209{
210 if (ps_len(ps) != len) {
211 return false;
212 }
213 for (size_t i = 0; i < len; i++) {
214 if (tolower(ps->b[i]) != tolower((unsigned char)s[i])) {
215 return false;
216 }
217 }
218 return true;
219}
220
221static inline bool ps_strcaseequal(const pstream_t * nonnull ps,
222 const char * nonnull s)
223{
224 return ps_memcaseequal(ps, s, strlen(s));
225}

645static inline int ps_hex16(pstream_t * nonnull ps, int len,
646 uint16_t * nonnull res)
647{
648 const byte *b = ps->b;
649
650 PS_WANT(len <= 4);
651 PS_WANT(ps_has(ps, len));
652
653 *res = 0;
654 while (b < ps->b + len) {
655 PS_WANT(ctype_desc_contains(&ctype_ishexdigit, *b));
656 *res <<= 4;
657 *res |= hexdigit(*b);
658 b++;
659 }
660 return __ps_skip(ps, len);
661}

672static inline const char * nullable ps_gets(pstream_t * nonnull ps,
673 int * nullable len)
674{
675 const char *end = (const char *)memchr(ps->s, '\0', ps_len(ps));
676 const char *res = ps->s;
677
678 if (unlikely(!end))
679 return NULL;
680 if (len)
681 *len = end - ps->s;
682 __ps_skip(ps, end + 1 - ps->s);
683 return res;
684}

815/** Read a CSV line from the pstream.
816 *
817 * The CSV is parsed according to RFC 4180 with the following relaxed rules:
818 * - TEXTDATA is set to all that is neither a separator nor \r or \n. Which
819 * mean a field can contain a quotation mark.
820 * - \n alone is allowed as a line break.
821 *
822 * The result of the parsing is put in a vector of strings. Copies of data are
823 * avoided as much as possible, which mean that unquoted field will be direct
824 * references to the content of the pstream and thus will have the same
825 * lifetime. Fields that must be copied (those containing escaped quotation
826 * marks) are allocated in the provided memory pool (which may be NULL if you
827 * want allocations on the heap).
828 *
829 * In case of error, the pstream is placed where it encountered the error.
830 * In case of success it is placed at the beginning of the next line.
831 *
832 * \param[in] mp The pool on which data that must be copied is allocated (NULL
833 * to allocate data on the heap).
834 * \param[in,out] ps The stream from which data is read.
835 * \param[in] sep The separator (usually comma or semi-colon)
836 * \param[in] quote The quoting character (usually double quote), can be -1 if
837 * quoting is disallowed.
838 * \param[out] fields A vector that will be filled with the fields.
839 * \param[out] out_line A pstream which will contain the content of the read
840 * CSV line (without the final line break).
841 * Can be NULL if you are not interrested in this peace
842 * of information.
843 * \return -1 if the content of the pstream does not starts with a valid CSV
844 * record.
845 */
846int ps_get_csv_line(mem_pool_t * nullable mp, pstream_t * nonnull ps, int sep,
847 int quote, union qv_lstr_t * nonnull fields,
848 pstream_t * nullable out_line);
