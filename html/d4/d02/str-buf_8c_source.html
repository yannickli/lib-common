26char *sb_detach(sb_t *sb, int *len)
27{
28 char *s;
29
30 if (len) {
31 *len = sb->len;
32 }
33 if (mp_ipool(sb->mp) == &mem_pool_libc && sb->data != __sb_slop) {
34 if (sb->skip) {
35 memmove(sb->data - sb->skip, sb->data, sb->len + 1);
36 }
37 s = sb->data - sb->skip;
38 sb_init(sb);
39 } else {
40 s = p_dupz(sb->data, sb->len);
41 sb_reset(sb);
42 }
43 return s;
44}

51static void __sb_reset(sb_t *sb, int threshold)
52{
53 mem_pool_t *mp = mp_ipool(sb->mp);
54 char *ptr = sb->data - sb->skip;
55
56 if (!(mp->mem_pool & MEM_BY_FRAME) && sb->skip + sb->size > threshold) {
57 if (ptr != __sb_slop) {
58 mp_delete(mp, &ptr);
59 }
60 sb_init_full(sb, (char *)__sb_slop, 0, 1, sb->mp);
61 } else {
62 sb_reset_keep_mem(sb);
63 }
64}

76/*
77 * this function is meant to rewind any change on a sb in a function doing
78 * repetitive appends that may fail.
79 *
80 * It cannot rewind a sb where anything has been skipped between the store and
81 * the rewind. It assumes only appends have been performed.
82 *
83 */
84int __sb_rewind_adds(sb_t *sb, const sb_t *orig)
85{
86 if (orig->mp != sb->mp) {
87 sb_t tmp = *sb;
88 int save_errno = errno;
89
90 if (orig->skip) {
91 sb_init_full(sb, orig->data - orig->skip, orig->len,
92 orig->size + orig->skip, orig->mp);
93 memcpy(sb->data, tmp.data, orig->len);
94 } else {
95 *sb = *orig;
96 __sb_fixlen(sb, orig->len);
97 }
98 mp_ifree(tmp.mp, tmp.data - tmp.skip);
99 errno = save_errno;
100 } else {
101 __sb_fixlen(sb, orig->len);
102 }
103 return -1;
104}

106static void sb_destroy_skip(sb_t *sb)
107{
108 if (sb->data == __sb_slop || sb->skip == 0) {
109 return;
110 }
111
112 memmove(sb->data - sb->skip, sb->data, sb->len + 1);
113 sb->data -= sb->skip;
114 sb->size += sb->skip;
115 sb->skip = 0;
116}

118void __sb_optimize(sb_t *sb, size_t len)
119{
120 mem_pool_t *mp = mp_ipool(sb->mp);
121 size_t sz = p_alloc_nr(len + 1);
122 char *buf;
123
124 if (len == 0) {
125 sb_reset(sb);
126 return;
127 }
128 if ((mp->mem_pool & MEM_BY_FRAME)) {
129 return;
130 }
131 buf = mp_new_raw(mp, char, sz);
132 p_copy(buf, sb->data, sb->len + 1);
133 mp_ifree(mp, sb->data - sb->skip);
134 sb_init_full(sb, buf, sb->len, sz, mp);
135}

137void __sb_grow(sb_t *sb, int extra)
138{
139 mem_pool_t *mp = mp_ipool(sb->mp);
140 int newlen = sb->len + extra;
141 int newsz;
142
143 if (unlikely(newlen < 0)) {
144 e_panic("trying to allocate insane amount of memory");
145 }
146
147 /* if data fits and skip is worth it, shift it left */
148 /* most of our pool have expensive reallocs wrt a typical memcpy,
149 * and optimize the last realloc so we don't want to alloc and free
150 */
151 if (newlen < sb->skip + sb->size
152 && (sb->skip > sb->size / 4 || !(mp->mem_pool & MEM_EFFICIENT_REALLOC)))
153 {
154 sb_destroy_skip(sb);
155 return;
156 }
157
158 newsz = p_alloc_nr(sb->size + sb->skip);
159 if (newsz < newlen + 1) {
160 newsz = newlen + 1;
161 }
162 /* TODO: avoid memmove + memcpy in case of a fallback */
163 sb_destroy_skip(sb);
164 if (sb->data == __sb_slop) {
165 sb->data = mp_new_raw(sb->mp, char, newsz);
166 sb->data[0] = '\0';
167 } else {
168 sb->data = mp_irealloc_fallback(&sb->mp, sb->data, sb->len + 1, newsz,
169 1, MEM_RAW);
170 }
171 sb->size = newsz;
172}

174char *__sb_splice(sb_t *sb, int pos, int rm_len, int insert_len)
175{
176 assert (pos >= 0 && rm_len >= 0 && insert_len >= 0);
177 assert (pos <= sb->len && pos + rm_len <= sb->len);
178
179 if (rm_len >= insert_len) {
180 /* More data to suppress than to insert, move the tail of the buffer
181 * to the left. */
182 p_move2(sb->data, pos + insert_len, pos + rm_len,
183 sb->len - pos - rm_len);
184 __sb_fixlen(sb, sb->len + insert_len - rm_len);
185 } else
186 if (rm_len + sb->skip >= insert_len) {
187 /* The skip area is at least as large as the data to insert
188 * (substracted from the data to remove), move the head of the buffer
189 * to the left, in the skip area. */
190 sb->skip -= insert_len - rm_len;
191 sb->data -= insert_len - rm_len;
192 sb->size += insert_len - rm_len;
193 sb->len += insert_len - rm_len;
194 p_move2(sb->data, 0, insert_len - rm_len, pos);
195 } else {
196 /* Default case: move the tail of the buffer to the right to leave
197 * some room for the data to insert. */
198 sb_grow(sb, insert_len - rm_len);
199 p_move2(sb->data, pos + insert_len, pos + rm_len,
200 sb->len - pos - rm_len);
201 __sb_fixlen(sb, sb->len + insert_len - rm_len);
202 }
203 sb_optimize(sb, 0);
204 return sb->data + pos;
205}

222int sb_addvf(sb_t *sb, const char *fmt, va_list ap)
223{
224 va_list ap2;
225 int len;
226
227 len = sb_avail(sb);
228 if (len != 0) {
229 len = len + 1;
230 }
231
232 va_copy(ap2, ap);
233 len = ivsnprintf(sb_end(sb), len, fmt, ap2);
234 va_end(ap2);
235
236 if (len <= sb_avail(sb)) {
237 __sb_fixlen(sb, sb->len + len);
238 } else {
239 ivsnprintf(sb_growlen(sb, len), len + 1, fmt, ap);
240 }
241 return len;
242}

256int sb_prependvf(sb_t *sb, const char *fmt, va_list ap)
257{
258 char buf[BUFSIZ];
259 va_list ap2;
260 int len;
261
262 va_copy(ap2, ap);
263 len = ivsnprintf(buf, BUFSIZ, fmt, ap2);
264 va_end(ap2);
265
266 if (len < BUFSIZ) {
267 sb_splice(sb, 0, 0, buf, len);
268 } else {
269 char *tbuf = p_new(char, len + 1);
270
271 ivsnprintf(tbuf, len + 1, fmt, ap);
272 sb_splice(sb, 0, 0, tbuf, len);
273 p_delete(&tbuf);
274 }
275 return len;
276}

290static void sb_add_ps_int_fmt(sb_t *out, pstream_t ps, int thousand_sep)
291{
292 if (thousand_sep < 0) {
293 sb_add(out, ps.p, ps_len(&ps));

295 }
296
297 while (!ps_done(&ps)) {

302 if (!ps_done(&ps)) {
303 sb_addc(out, thousand_sep);
304 }

325void sb_add_double_fmt(sb_t *sb, double val, uint8_t nb_max_decimals,
326 int dec_sep, int thousand_sep)
327{
328 char buf[BUFSIZ];
329 pstream_t ps, integer_part = ps_init(NULL, 0);
330
331 if (isnan(val) || isinf(val)) {
332 sb_addf(sb, "%f", val);
333 return;
334 }
335
336 ps = ps_init(buf, snprintf(buf, sizeof(buf), "%.*f",
337 MAX(1, nb_max_decimals), val));
338
339 /* Sign */
340 if (ps_skipc(&ps, '-') == 0) {
341 sb_addc(sb, '-');
342 }
343
344 /* Integer part */
345 ps_get_ps_chr_and_skip(&ps, '.', &integer_part);
346 sb_add_ps_int_fmt(sb, integer_part, thousand_sep);
347
348 /* Decimal part */
349 if (nb_max_decimals > 0) {
350 pstream_t decimal_part = ps;
351
352 while (ps_shrinkc(&decimal_part, '0') == 0);
353
354 if (ps_len(&decimal_part)) {
355 sb_addc(sb, dec_sep);
356 sb_add(sb, ps.s, ps_len(&ps));
357 }
358 }
359}

397void _sb_add_duration_ms(sb_t *sb, uint64_t ms, bool print_ms)
398{
399 uint8_t nb_prints = 0;
400 static const uint32_t units[] = {
401 24 * 60 * 60 * 1000, /* day */
402 60 * 60 * 1000, /* hour */
403 60 * 1000, /* minute */
404 1000, /* second */
405 1, /* millisecond */
406 };
407
408 if (!ms) {
409 sb_adds(sb, "0s");
410 return;
411 }
412
413 for (int j = 0; j < countof(units) - 1; j++) {
414 if (ms >= units[j]) {
415 /* only units j and j + 1 will be printed, round the remainder to
416 * the nearest value in unit j + 1 */
417 ms = ROUND(ms + units[j + 1] / 2, units[j + 1]);
418 break;
419 }
420 }
421
422#define SB_ADD_DUR(_unit, i) \
423 do { \
424 uint32_t nb_ms_unit = units[i]; \
425 \
426 if ((ms) >= nb_ms_unit || nb_prints == 1) { \
427 if (nb_prints) { \
428 sb_addc(sb, ' '); \
429 } \
430 sb_addf(sb, "%lu" _unit, ms / (nb_ms_unit)); \
431 ms %= (nb_ms_unit); \
432 nb_prints++; \
433 } \
434 } while (0)
435
436 SB_ADD_DUR( "d", 0);
437 SB_ADD_DUR( "h", 1);
438 SB_ADD_DUR( "m", 2);
439 SB_ADD_DUR( "s", 3);
440 if (print_ms) {
441 SB_ADD_DUR("ms", 4);
442 }
443
444#undef SB_ADD_DUR
445}

459int sb_getline(sb_t *sb, FILE *f)
460{
461 int64_t start = ftell(f);
462 sb_t orig = *sb;
463
464 do {
465 int64_t end;
466 char *buf = sb_grow(sb, BUFSIZ);
467
468 if (!fgets(buf, sb_avail(sb) + 1, f)) {
469 if (ferror(f))
470 return __sb_rewind_adds(sb, &orig);
471 break;
472 }
473
474 end = ftell(f);
475 if (start != -1 && end != -1) {
476 sb->len += (end - start);
477 } else {
478 sb->len += strlen(buf);
479 }
480 start = end;
481 } while (sb->data[sb->len - 1] != '\n');
482
483 if (likely(sb->len - orig.len > 0)
484 && unlikely(sb->data[sb->len - 1] != '\n'))
485 {
486 sb_addc(sb, '\n');
487 }
488
489 return sb->len - orig.len;
490}

492/* OG: returns the number of elements actually appended to the sb,
493 * -1 if error
494 */
495int sb_fread(sb_t *sb, int size, int nmemb, FILE *f)
496{
497 sb_t orig = *sb;
498 int res = size * nmemb;
499 char *buf = sb_grow(sb, size * nmemb);
500
501 if (unlikely(((long long)size * (long long)nmemb) != res))
502 e_panic("trying to allocate insane amount of memory");
503
504 res = fread(buf, size, nmemb, f);
505 if (res < 0)
506 return __sb_rewind_adds(sb, &orig);
507
508 __sb_fixlen(sb, sb->len + res * size);
509 return res;
510}

512/* Return the number of bytes appended to the sb, negative value
513 * indicates an error.
514 * OG: this function insists on reading a complete file. If the file
515 * cannot be read completely, no data is kept in the sb and an error
516 * is returned.
517 */
518int sb_read_fd(sb_t *sb, int fd)
519{
520 sb_t orig = *sb;
521 struct stat st;
522 char *buf;
523 int res;
524
525 if (fstat(fd, &st) < 0 || st.st_size <= 0) {
526 for (;;) {
527 res = sb_read(sb, fd, 0);
528
529 if (res < 0) {
530 return __sb_rewind_adds(sb, &orig);
531 }
532 if (res == 0) {
533 return sb->len - orig.len;
534 }
535 }
536 }
537
538 if (st.st_size > INT_MAX) {
539 errno = ENOMEM;
540 return -1;
541 }
542
543 res = st.st_size;
544 buf = sb_growlen(sb, res);
545 if (xread(fd, buf, res) < 0) {
546 return __sb_rewind_adds(sb, &orig);
547 }
548
549 return res;
550}
