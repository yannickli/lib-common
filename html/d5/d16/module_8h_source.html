67/** Declare a new method.
68 *
69 * This macro should be called in headers files to export a method.
70 *
71 * \param[in] Type prototype of the method. One of \ref module_method_type_t.
72 * \param[in] name name of the method.
73 */
74#define MODULE_METHOD_DECLARE(Type, Order, name) \
75 extern const module_method_t name##_method; \
76 extern const module_method_t * const nonnull name##_method_ptr
77
78/** Define a new method.
79 *
80 * This macro should be called in an implementation file to define a method.
81 * It can be prefixed by static in order to create a local definition. If the
82 * method is public, it must be declared using \ref MODULE_METHOD_DECLARE.
83 *
84 * \param[in] Type prototype of the method.
85 * \param[in] name name of the method.
86 */
87#define MODULE_METHOD(Type, Order, name) \
88 const module_method_t name##_method = { \
89 .type = METHOD_##Type, \
90 .order = MODULE_##Order, \
91 }; \
92 const module_method_t * const name##_method_ptr = &name##_method
93
94/** Run a void method.
95 *
96 * This will run the method whose name has been provided.
97 *
98 * This macro is only suitable for void methods.
99 *
100 * \see module_run_method
101 */
102#define MODULE_METHOD_RUN_VOID(method) do { \
103 assert (method##_method.type == METHOD_VOID); \
104 module_run_method(&method##_method, (data_t)NULL); \
105 } while (0)
106
107/** Run a pointer method.
108 *
109 * This will run the method whose name has been provided and passing the given
110 * pointer as argument.
111 *
112 * This macro is only suitable for methods that take a pointer as argument.
113 *
114 * \see module_run_method
115 */
116#define MODULE_METHOD_RUN_PTR(method, arg) do { \
117 assert (method##_method.type == METHOD_PTR); \
118 module_run_method(&method##_method, (data_t){ .ptr = arg }); \
119 } while (0)
120
121/** Run an integer method.
122 *
123 * This will run the method whose name has been provided and passing the given
124 * integer as argument.
125 *
126 * This macro is only suitable for methods that take an integer as argument.
127 *
128 * \see module_run_method
129 */
130#define MODULE_METHOD_RUN_INT(method, arg) do { \
131 assert (method##_method.type == METHOD_INT); \
132 module_run_method(&method##_method, (data_t){ .u32 = arg }); \
133 } while (0)
134
135/** Run a method.
136 *
137 * This will run the method whose name has been provided with the given \ref
138 * data_t object.
139 *
140 * This macro is only suitable for generic methods.
141 *
142 * \see module_run_method
143 */
144#define MODULE_METHOD_RUN(method, data) do { \
145 assert (method##_method.type == METHOD_GENERIC); \
146 module_run_method(&method##_method, data); \
147 } while (0)
148
149/** Run a method.
150 *
151 * Traverse the list of started modules and run the given method in all
152 * modules that implement it according to the given
153 * \ref module_method_t::order. If the method is non-void, it will receive
154 * the provided \p arg as argument.
155 *
156 * \param[in] method The method to call.
157 * \param[in] arg Argument passed to the methods.
158 */
159__attr_nonnull__((1))
160void module_run_method(const module_method_t * nonnull method, data_t arg);

165/** Pointer to the module of the given name.

169/** Declare a module.
170 *
171 * This macro declares a module variable.

173#define MODULE_DECLARE(name) \

175
176/** Add declarations of a module.
177 *
178 * Do not use this, use \ref MODULE_BEGIN.
179 */
180#define _MODULE_ADD_DECLS(name) \
181 module_t *name##_get_module(void) \
182 { \
183 __attr_section("intersec", "module") \
184 static module_t *mod; \
185 \
186 if (!mod) { \
187 mod = module_register(LSTR(#name)); \
188 } \
189 return mod; \
190 }
191
192/** Begin the definition of a module.
193 *
194 * This begin a section of code that can contain the description of a module.
195 * The section can currently contain the following descriptions:
196 * - \ref MODULE_DEPENDS_ON to add a dependency.
197 * - \ref MODULE_IMPLEMENTS to add a method.
198 *
199 * The macro can also take a variadic list of dependencies as strings, but
200 * this usage is discouraged since it does not add compile-time dependencies
201 * between compilation units. Using \ref MODULE_DEPENDS_ON should be the
202 * prefered solution to add a new dependence.
203 *
204 * The section must be closed by calling \ref MODULE_END().
205 */
206#define MODULE_BEGIN(name) \
207 _MODULE_ADD_DECLS(name); \
208 \
209 static __attribute__((constructor)) \
210 void __##name##_module_register(void) { \
211 __unused__ \
212 module_t *__mod = module_implement(MODULE(name), \
213 &name##_initialize, \
214 &name##_shutdown, \
215 MODULE(log)); \
216
217/** Macro to end the definition of a module.
218 *
219 * \see MODULE_BEGIN
220 */
221#define MODULE_END() }
222
223/** Add a dependence on another module.
224 *
225 * This macro can only be used in a MODULE_BEGIN/MODULE_END block of code. It
226 * declares a dependence from the current module on \p dep.
227 */
228#define MODULE_DEPENDS_ON(dep) \
229 module_add_dep(__mod, MODULE(dep))
230
231/** Add a dependence to another module.
232 *
233 * As \ref MODULE_DEPENDS_ON this macro can only be used in a
234 * MODULE_BEGIN/MODULE_END block.
235 * It declares a dependence from the current module to \p need.
236 */
237#define MODULE_NEEDED_BY(need) \
238 module_add_dep(MODULE(need), __mod)

242/** Declare the implementation of the method \p hook.
243 *
244 * This macro can only be used in a MODULE_BEGIN/MODULE_END block of code. It
245 * declares that the current module implements the method \p hook with the
246 * implementation being \p cb.
247 *
248 * This macro can only be used if the method has a void prototype.

250#define MODULE_IMPLEMENTS_VOID(hook, cb) \

253/** Declare the implementation of the method \p hook.
254 *
255 * \see MODULE_IMPLEMENTS_VOID
256 *
257 * This macro can only be used if the method takes a pointer as argument.
258 */
259#define MODULE_IMPLEMENTS_PTR(Type, hook, cb) do { \
260 void (*__hook_cb)(Type *ptr) = (cb); \
261 module_implement_method_ptr(__mod, &hook##_method, \
262 (void (*)(void *))__hook_cb); \
263 } while (0)
264
265/** Declare the implementation of the method \p hook.
266 *
267 * \see MODULE_IMPLEMENTS_VOID
268 *
269 * This macro can only be used if the method takes an integer as argument.
270 */
271#define MODULE_IMPLEMENTS_INT(hook, cb) \
272 module_implement_method_int(__mod, &hook##_method, (cb))
273
274/** Declare the implementation of the method \p hook.
275 *
276 * \see MODULE_IMPLEMENTS_VOID
277 *
278 * This macro can only be used if the method takes a \ref data_t as argument.
279 */
280#define MODULE_IMPLEMENTS(hook, cb) \
281 module_implement_method_generic(__mod, &hook##_method, (cb))

356/** Provide an argument for module constructor.
357 *
358 * Use:
359 * MODULE_PROVIDE(module, (void *)arg)
360 *
361 * Your initialization function (alias constructor function) should
362 * return a negative number if it needs an argument and this argument
363 * is not provided (by default NULL)
364 * Exe:
365 * int module_initialize(void *arg) {
366 * if (arg == NULL)
367 * return -1; (or assert)
368 * ...
369 * }
370 */
371#define MODULE_PROVIDE(name, argument) \
372 module_provide(MODULE(name), argument)
373
374/** Macro for requiring a module.
375 *
376 * Use:
377 * MODULE_REQUIRE(module1);
378 *
379 * module1
380 * / \
381 * module2 module3
382 *
383 * Behavior:
384 * + If every thing works
385 * - module1, module2, module3 initialized
386 * - module1 will have state MANU_REQ
387 * - module2, module3 will have state AUTO_REQ
388 * - MODULE_REQUIRE will return
389 * + If module3 fail to initialize
390 * - module_require will throw a logger_fatal
391 */
392#define MODULE_REQUIRE(name) module_require(MODULE(name), NULL)
393
394/** Macro for releasing a module.
395 *
396 * Use:
397 * MODULE_RELEASE(module);
398 *
399 * Behavior:
400 * + If you try to RELEASE a module that has not been manually initialized
401 * the program will assert
402 * In other words only RELEASE modules that has been require with REQUIRE
403 * + For returns value see module_release and module_shutdown
404 */
405#define MODULE_RELEASE(name) module_release(MODULE(name))
