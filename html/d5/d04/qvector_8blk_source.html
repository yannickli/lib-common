22static void __qvector_reset(qvector_t *vec, size_t v_size, size_t threshold)
23{
24 mem_pool_t *mp = mp_ipool(vec->mp);
25 size_t sz = vec->size;
26
27 if (!(mp->mem_pool & MEM_BY_FRAME) && sz * v_size >= threshold) {
28 mp_delete(mp, &vec->tab);
29 __qvector_init(vec, NULL, 0, 0, vec->mp);
30 } else {
31 vec->len = 0;
32 }
33}

45static void qvector_get_grow_info(qvector_t *vec, size_t v_size, int extra,
46 int *newsz, bool *needs_alloc)
47{
48 int newlen = vec->len + extra;
49
50 if (unlikely(newlen < 0)) {
51 e_panic("trying to allocate insane amount of memory");
52 }
53
54 if (newlen < vec->size) {
55 *needs_alloc = false;
56 return;
57 }
58
59 *newsz = p_alloc_nr(vec->size);
60 if (*newsz < newlen) {
61 *newsz = newlen;
62 }
63
64 *needs_alloc = true;
65}

67uint64_t __qvector_grow_get_new_alloc_size(qvector_t *nonnull vec,
68 size_t v_size, int extra)
69{

81void __qvector_grow(qvector_t *vec, size_t v_size, size_t v_align, int extra)
82{
83 int newsz;
84 bool needs_alloc;
85
86 qvector_get_grow_info(vec, v_size, extra, &newsz, &needs_alloc);
87 if (!needs_alloc) {
88 return;
89 }
90
91 vec->tab = mp_irealloc_fallback(&vec->mp, vec->tab, vec->len * v_size,
92 newsz * v_size, v_align, MEM_RAW);
93 vec->size = newsz;
94}

96void __qvector_optimize(qvector_t *vec, size_t v_size, size_t v_align,
97 size_t size)
98{
99 mem_pool_t *mp = mp_ipool(vec->mp);
100 char *buf;
101
102 if (size == 0) {
103 qvector_reset(vec, v_size);
104 return;
105 }
106 if ((mp->mem_pool & MEM_BY_FRAME)) {
107 return;
108 }
109 buf = mpa_new_raw(mp, char, size * v_size, v_align);
110 memcpy(buf, vec->tab, vec->len * v_size);
111 mp_delete(mp, &vec->tab);
112 __qvector_init(vec, buf, vec->len, size, vec->mp);
113}

115void *__qvector_splice(qvector_t *vec, size_t v_size, size_t v_align,
116 int pos, int rm_len, int inserted_len)
117{
118 assert (pos >= 0 && rm_len >= 0 && inserted_len >= 0);
119 assert (pos <= vec->len && pos + rm_len <= vec->len);
120
121 if (rm_len >= inserted_len) {
122 memmove(vec->tab + v_size * (pos + inserted_len),
123 vec->tab + v_size * (pos + rm_len),
124 v_size * (vec->len - pos - rm_len));
125 qvector_grow(vec, v_size, v_align, 0);
126 } else {
127 qvector_grow(vec, v_size, v_align, inserted_len - rm_len);
128 memmove(vec->tab + v_size * (pos + inserted_len),
129 vec->tab + v_size * (pos + rm_len),
130 v_size * (vec->len - pos - rm_len));
131 }
132 vec->len += inserted_len - rm_len;
133 return vec->tab + pos * v_size;
134}

152void __qvector_shuffle(qvector_t * nonnull vec, size_t v_size)
153{
154 /* XXX: index 0 is excluded from the loop, which is intended in
155 * Fisher-Yates shuffle algorithm. */
156 for (int i = vec->len - 1; i > 0; i--) {
157 int j = rand() % (i + 1);
158 uint8_t tmp[v_size];
159
160#define nth(_i) ((void *)((uint8_t *)(vec->tab) + (_i) * v_size))
161 memcpy(tmp, nth(j), v_size);
162 memcpy(nth(j), nth(i), v_size);
163 memcpy(nth(i), tmp, v_size);
164#undef nth
165 }
166}

168void __qvector_diff(const qvector_t *vec1, const qvector_t *vec2,
169 qvector_t *add, qvector_t *del, qvector_t *inter,
170 size_t v_size, size_t v_align, qvector_cmp_b cmp)
171{
172 int pos1 = 0, pos2 = 0;
173 const void *tab1 = vec1->tab;
174 const void *tab2 = vec2->tab;
175 const void *elem1, *elem2;
176
177#define nth(base, i) ((void *)((uint8_t *)(base) + (i) * v_size))
178#define append_tab(tab, src, len) \
179 if ((tab)) \
180 memcpy(qvector_growlen((tab), v_size, v_align, (len)), \
181 src, (len) * v_size);
182#define append_add(src, len) append_tab(add, src, len)
183#define append_del(src, len) append_tab(del, src, len)
184#define append_inter(src, len) append_tab(inter, src, len)
185
186 if (add) {
187 qvector_grow(add, v_size, v_align, vec2->len + add->len);
188 }
189 if (del) {
190 qvector_grow(del, v_size, v_align, vec1->len + del->len);
191 }
192
193 elem1 = nth(tab1, 0);
194 elem2 = nth(tab2, 0);
195
196 while (pos1 < vec1->len && pos2 < vec2->len) {
197 int s = cmp(elem1, elem2);
198
199 if (s == 0) {
200 append_inter(elem1, 1);
201 elem1 = nth(tab1, ++pos1);
202 elem2 = nth(tab2, ++pos2);
203 } else
204 if (s < 0) {
205 append_del(elem1, 1);
206 elem1 = nth(tab1, ++pos1);
207 } else {
208 append_add(elem2, 1);
209 elem2 = nth(tab2, ++pos2);
210 }
211 }
212 if (pos1 != vec1->len) {
213 append_del(elem1, vec1->len - pos1);
214 } else {
215 append_add(elem2, vec2->len - pos2);
216 }
217
218#undef append_add
219#undef append_del
220#undef append_tab
221#undef nth
222}

230int __qvector_find(const qvector_t *vec, size_t v_size, const void *elt,
231 bool sorted, qvector_cmp_b cmp)
232{
233 if (sorted) {
234 bool found = false;
235 int pos = __qvector_bisect(vec, v_size, elt, &found, cmp);
236
237 return found ? pos : -1;
238 } else {
239 for (int i = 0; i < vec->len; i++) {
240 if (cmp(elt, vec->tab + v_size * i) == 0) {
241 return i;
242 }
243 }
244 return -1;
245 }
246}

260void __qvector_deep_extend(qvector_t *vec_dst, const qvector_t *vec_src,
261 size_t v_size, size_t v_align, qvector_cpy_b cpy_f)
262{
263 size_t original_len = vec_dst->len;
264
265 qvector_growlen(vec_dst, v_size, v_align, vec_src->len);
266 tab_for_each_pos(pos, vec_src) {
267 /* XXX: qvector_t.tab is of type (uint8_t *) so we need to do the
268 * pointer arithmetic ourselves because the true type of the
269 * array could be different.
270 */
271 cpy_f(vec_dst->tab + (pos + original_len) * v_size,
272 vec_src->tab + pos * v_size);
273 }
