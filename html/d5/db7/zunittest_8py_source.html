21"""
22z
23
24z is a wrapper around unnitest (python 2.7+) or unittest2
25
26To write a "group" of test, write a class deriving from unittest.TestCase
27and decorate it with z.ZGroup this way:
28
29>>> import z
30>>>
31>>> @z.ZGroup
32>>> class MyTest(z.TestCase):
33>>> pass
34
35Any function in MyTest whose name starts with 'test' will be treated as a
36test.
37
38Tests functions can be decorated with any of the unittest usual decorator but
39for expectedFailure where @z.ZTodo(reason) must be used instead.
40
41As a conveniency, the z.ZFlags() decorator exists to put flags on a test, it's
42a conveniency wrapper around unittest.skip()
43
44Doctests can also be run:
45
46>>> import my_python_module
47>>>
48>>> @z.ZGroup
49>>> class MyTest(z.DocTestModule):
50>>> module = my_python_module
51
52At the end of your test, just call z.main() (and not unittest.main()).
53if Z_HARNESS is set, then z.main() will run in "Z" mode, else it fallbacks to
54unittest.main() allowing all the python unittest module features (such as
55running tests as specified on the command line).
56
57Note that 'z' extends the unittest module, so you just need to import z and
58use z.* as you would use unittest instead
59"""

111def ZGroup(cls): # pylint: disable=invalid-name
112 """
113 ZGroup
114
115 This decorator generates a 'load_tests' function for the module
116 where the class lives, for use with unittest.main()
117 """
118 __import__(cls.__module__)

160class IopDocTestRunner(doctest.DocTestRunner):
161 """ Custom Doc test runner
162
163 Used to wrap tests in a "print_iop" function
164 """
165 def run(self, test, *args, **kwargs):
166 for example in test.examples:
167 source = example.source.strip()
168 # Wrap the code with a print call, unless a for loop is detected
169 if not (source.startswith('for ') or source.startswith('with')):
170 example.source = 'print({0})\n'.format(source)
171
172 return super().run(test, *args, **kwargs)

215_Z_MODES = set(os.getenv('Z_MODE', '').split())
216_FLAGS = set(os.getenv('Z_TAG_SKIP', '').split())
217_TAG_OR = os.getenv('Z_TAG_OR', '')

220@public
221def ZFlags(*flags): # pylint: disable=invalid-name
222 """
223 ZFlags
224
225 This decorator is a wrapper around unittest.skip() to implement the
226 Z_TAG_SKIP required interface.
227 """
228 def wrap(func):
229 func_flags = _ALL_FLAGS.setdefault(func, [])
230 func_flags.extend(flags)
231 fl = set(func_flags) & _FLAGS
232 if any((f in _TAG_OR for f in func_flags)) and "wip" not in fl:
233 func.__unittest_skip__ = False
234 return func
235 if fl:
236 return unittest.skip("skipping tests flagged with %s" %
237 (" ".join(fl)))(func)
238 return func
239 return wrap
