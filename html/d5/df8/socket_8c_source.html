41int bindx(int sock, const sockunion_t *addrs, int cnt,
42 int type, int proto, int flags)
43{
44 int to_close = -1;
45
46 if (proto != IPPROTO_SCTP && cnt != 1) {
47 errno = EINVAL;
48 return -1;
49 }
50
51 if (sock < 0) {
52 to_close = sock = RETHROW(socket(addrs->family, type, proto));
53 }
54
55 if (type != SOCK_DGRAM && sock_reuseaddr(sock) < 0) {
56 goto error;
57 }
58
59 if (fd_set_features(sock, flags))
60 goto error;
61
62 if (proto != IPPROTO_SCTP || cnt == 1) {
63 if (addrs->family == AF_UNIX) {
64 unlink(addrs->sunix.sun_path);
65 }
66
67 if (bind(sock, &addrs->sa, sockunion_len(addrs)) < 0)
68 goto error;
69 } else {
70 socklen_t sz = 0;
71 while (cnt-- > 0) {
72 sockunion_t *su = (sockunion_t *)((char *)addrs + sz);
73 socklen_t len = sockunion_len(su);
74 if (len == (socklen_t)-1) {
75 errno = EINVAL;
76 goto error;
77 }
78 sz += len;
79 }
80 if (setsockopt(sock, SOL_SCTP, SCTP_SOCKOPT_BINDX_ADD, addrs, sz))
81 goto error;
82 }
83
84 return sock;
85
86 error:
87 p_close(&to_close);
88 return -1;
89}

114int connectx_as(int sock, const sockunion_t *addrs, int cnt,
115 const sockunion_t * nullable src, int type, int proto,
116 int flags, int timeout)
117{
118 int to_close = -1;
119
120 if (proto != IPPROTO_SCTP && cnt != 1) {
121 errno = EINVAL;
122 return -1;
123 }
124
125 if (sock < 0) {
126 to_close = sock = RETHROW(socket(addrs->family, type, proto));
127 }
128 if (src && bind(sock, &src->sa, sockunion_len(src)) < 0) {
129 goto error;
130 }
131
132 if (fd_set_features(sock, flags)) {
133 goto error;
134 }
135
136 if (timeout) {
137 struct timeval tv = { .tv_sec = timeout };
138
139 assert (!(flags & O_NONBLOCK));
140 /* Set a socket timeout to avoid blocking indefinitely. */
141 setsockopt(sock, SOL_SOCKET, SO_RCVTIMEO, &tv, sizeof(tv));
142 }
143
144 if (proto != IPPROTO_SCTP || cnt == 1) {
145 if (connect(sock, &addrs->sa, sockunion_len(addrs)) < 0
146 && !ERR_CONNECT_RETRIABLE(errno))
147 {
148 goto error;
149 }
150 } else {
151 socklen_t sz = 0;
152 while (cnt-- > 0) {
153 sockunion_t *su = (sockunion_t *)((char *)addrs + sz);
154 socklen_t len = sockunion_len(su);
155 if (len == (socklen_t)-1) {
156 errno = EINVAL;
157 goto error;
158 }
159 sz += len;
160 }
161 if (setsockopt(sock, SOL_SCTP, SCTP_SOCKOPT_CONNECTX, addrs, sz))
162 goto error;
163 }
164
165 return sock;
166
167 error:
168 PROTECT_ERRNO(p_close(&to_close));
169 return -1;
170}
