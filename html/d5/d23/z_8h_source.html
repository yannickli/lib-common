26/** \defgroup Z Intersec Unit Testing Framework.
27 * \brief Intersec Unit Testing Framework.
28 *
29 * \{
30 *
31 * \section Z_C Pure C framework.
32 *
33 * Tests are groupped into #Z_GROUP. Those are run in a separate process.
34 * In a #Z_GROUP, tests are defined in a \code Z_TEST { } Z_TEST_END; \endcode
35 * block.
36 *
37 * Beware, do not use anything but:
38 * - variable declarations;
39 * - #z_todo_start, #z_todo_end, #z_skip_start, #z_skip_end
40 * between #Z_TEST blocks, because a #Z_GROUP is called several times, with
41 * possible non linear (#Z_TEST are mostly groups of gotos) execution.
42 *
43 * If you need setup/teardowns, please use the blocks interface.
44 *
45 *
46 * \section Z_BLK C + Blocks framework.
47 *
48 * This part of the framework is more convenient for metageneration of tests.
49 *
50 * Here the #Z_GROUP is a \a z_blkgrp, which has the following members:
51 *
52 * - \a before that is run at the beginning of a group execution. If it fails
53 * (return non zero) then the group is aborted and nothing else is done.
54 * - \a after that is run after all tests are run. It is always run (but for
55 * the case where \a before failed).
56 * - \a setup and \a teardown. Those are run before and after each test.
57 * \a setup cannot directly fail, but can use the Z_ASSERT* macros or
58 * #Z_SKIP. Its failure will hence be seen as a test failure, so #Z_SKIP is
59 * mostly prefered as a failure to setup() isn't a test failure, rather a
60 * reason to skip it.
61 * - \a test is an array of \a z_blktst containing a name and the \a run field
62 * that is the test itself.
63 *
64 * #z_register_blkgroup is used to register a group into Z. Note that this
65 * will fully duplicate the z_blkgrp, so if it was allocated, #z_blkgrp_wipe
66 * can be of help.
67 *
68 * \section Z_ENV Environment.
69 *
70 * Tests are run in a separate process, on a group basis. If a group crashes,
71 * some tests may be skipped, but other groups will run.
72 *
73 * A temporary directory #z_tmpdir_g is created, and emptied before each test
74 * run, so that tests don't need to clean-up files.
75 *
76 * A temporary directory #z_grpdir_g is created, and emptied before each group
77 * run, but are kept for the whole group run.
78 *
79 * #z_cmddir_g is the path to the directory containing the command being run.
80 *
81 * Tests can use chdir() (usually to do Z_ASSERT_N(chdir(z_tmpdir_g.s))),
82 * because the current working directory is reset before each test run.
83 */
84/** \} */

164/* Detect the type for '_v' and fill a 'z_val_t' structure. */
165#define _Z_VAL_BUILD(_v) \
166 ({ \
167 z_val_t _z_val_build_res; \
168 typeof(_v) _z_val_build_v = (_v); \
169 \
170 p_clear(&_z_val_build_res, 1); \
171 if (_Z_VAL_TEST(int8_t) || _Z_VAL_TEST(int16_t) \
172 || _Z_VAL_TEST(int32_t) || _Z_VAL_TEST(int64_t) \
173 || _Z_VAL_TEST(int) || _Z_VAL_TEST(long) || _Z_VAL_TEST(long long)) \
174 { \
175 _z_val_build_res.type = Z_VAL_TYPE_I; \
176 _z_val_build_res.i = _z_val_build_v; \
177 } else \
178 if (_Z_VAL_TEST(uint8_t) || _Z_VAL_TEST(uint16_t) \
179 || _Z_VAL_TEST(uint32_t) || _Z_VAL_TEST(uint64_t) \
180 || _Z_VAL_TEST(unsigned) || _Z_VAL_TEST(unsigned long) \
181 || _Z_VAL_TEST(unsigned long long)) \
182 { \
183 _z_val_build_res.type = Z_VAL_TYPE_U; \
184 _z_val_build_res.u = _z_val_build_v; \
185 } else \
186 if (_Z_VAL_TEST(float) || _Z_VAL_TEST(double)) { \
187 _z_val_build_res.type = Z_VAL_TYPE_D; \
188 _z_val_build_res.d = _z_val_build_v; \
189 } else \
190 if (_Z_VAL_TEST(bool)) { \
191 _z_val_build_res.type = Z_VAL_TYPE_B; \
192 _z_val_build_res.b = _z_val_build_v; \
193 } else \
194 if (_Z_VAL_TEST(char)) { \
195 _z_val_build_res.type = Z_VAL_TYPE_C; \
196 _z_val_build_res.c = _z_val_build_v; \
197 } else { \
198 __error__(#_v " has an unsupported type (for now)"); \
199 } \
200 _z_val_build_res; \
