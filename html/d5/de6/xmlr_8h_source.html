97static inline int xmlr_node_is(xml_reader_t xr, const char *s, size_t len)
98{
99 lstr_t name = LSTR_NULL_V;
100
101 return !RETHROW(xmlr_node_is_closing(xr))
102 && RETHROW(xmlr_node_get_local_name(xr, &name)) >= 0
103 && lstr_equal(name, LSTR_INIT_V(s, len));
104}
