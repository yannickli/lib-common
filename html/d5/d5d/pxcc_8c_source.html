233static pxcc_typedef_kind_t
234get_typedef_kind(CXType type, CXType underlying_type)
235{
236 CXType canonical_type;
237 CXString type_spelling;
238 lstr_t type_spelling_str;
239 CXString canonical_spelling;
240 lstr_t canonical_spelling_str;
241 CXString underlying_spelling;
242 lstr_t underlying_spelling_str;
243 pstream_t ps;
244 pxcc_typedef_kind_t kind;
245
246 if (underlying_type.kind != CXType_Record
247 && underlying_type.kind != CXType_Enum)
248 {
249 return PXCC_TYPEDEF_DIFFERENT;
250 }
251
252 type_spelling = clang_getTypeSpelling(type);
253 type_spelling_str = get_unconst_type_spelling(type_spelling);
254
255 canonical_type = clang_getCanonicalType(type);
256 canonical_spelling = clang_getTypeSpelling(canonical_type);
257 canonical_spelling_str = get_unconst_type_spelling(canonical_spelling);
258
259 /* If type and canonical have the same spelling, it means that the
260 * original struct, union or enum is unnamed.
261 * Example: typedef struct { ... } foo_t;
262 */
263 if (lstr_equal(type_spelling_str, canonical_spelling_str)) {
264 kind = PXCC_TYPEDEF_UNNAMED;
265 goto end;
266 }
267
268 underlying_spelling = clang_getTypeSpelling(underlying_type);
269 underlying_spelling_str = get_unconst_type_spelling(underlying_spelling);
270
271 /* If type and underlying have the same spelling minus the data type
272 * keyword, the typedef is transparent.
273 * Example: typedef struct plop_t { ... } plop_t;
274 */
275 ps = ps_initlstr(&underlying_spelling_str);
276 if (expect(ps_skip_afterchr(&ps, ' ') >= 0)
277 && ps_is_equal(ps, ps_initlstr(&type_spelling_str)))
278 {
279 kind = PXCC_TYPEDEF_TRANSPARENT;
280 } else {
281 kind = PXCC_TYPEDEF_DIFFERENT;
282 }
283
284 clang_disposeString(underlying_spelling);
285
286 end:
287 clang_disposeString(canonical_spelling);
288 clang_disposeString(type_spelling);
289 return kind;
290}

414static lstr_t t_concat_type_stack(const qv_t(cstr) *type_stack,
415 const char *separator, const char *prefix,
416 const char *suffix)
417{
418 t_SB_1k(sb);
419
420 sb_adds(&sb, prefix);
421 tab_for_each_pos(pos, type_stack) {
422 if (pos > 0) {
423 sb_adds(&sb, separator);
424 }
425 sb_adds(&sb, type_stack->tab[pos]);
426 }
427 sb_adds(&sb, suffix);
428 return LSTR_SB_V(&sb);

446static CXChildVisitResult
447t_visit_register_type_fields(CXCursor cursor, CXCursor parent,
448 CXClientData data)
449{
450 qv_t(cstr) *type_stack = data;
451 CXCursorKind kind = clang_getCursorKind(cursor);
452
453 switch (kind) {
454 case CXCursor_FieldDecl:
455 if (t_register_type_cursor(cursor, type_stack) < 0) {
456 return CXChildVisit_Break;
457 }
458 break;
459
460 case CXCursor_UnionDecl:
461 case CXCursor_StructDecl:
462 return CXChildVisit_Recurse;
463
464 case CXCursor_EnumDecl:
465 case CXCursor_EnumConstantDecl:
466 break;
467
468 default: {
469 lstr_t path = t_concat_type_stack(type_stack, "::", "", "");
470
471 fprintf(stderr, "unsupported field type of kind %d for %pL\n",
472 kind, &path);
473 assert (false);
474 return CXChildVisit_Break;
475 }
476 }
477
478 return CXChildVisit_Continue;
479}

493static int t_register_typedef_type(CXType type, qv_t(cstr) *type_stack)
494{
495 pxcc_record_name_t *record_name;
496 CXCursor cursor;
497 CXType underlying_type;
498 pxcc_typedef_kind_t typedef_kind;
499
500 record_name = get_or_add_new_record_name_type(type);
501 switch (record_name->status) {
502 case PXCC_RECORD_NAME_NEW:
503 /* It is new, process it as usual. */
504 record_name->status = PXCC_RECORD_NAME_VISITED;
505 break;
506
507 case PXCC_RECORD_NAME_VISITED:
508 /* Recursive call for this record name.
509 * We need to continue until it is resolved by forwarding canonical
510 * type.
511 */
512 break;
513
514 case PXCC_RECORD_NAME_FORWARDED:
515 /* Should not happen. */
516 assert(false);
517 break;
518
519 case PXCC_RECORD_NAME_COMPLETED:
520 /* The record was already registered, nothing to do here. */
521 return 0;
522 }
523
524 cursor = clang_getTypeDeclaration(type);
525 underlying_type = get_underlying_type(cursor);
526 typedef_kind = get_typedef_kind(type, underlying_type);
527
528 switch (typedef_kind) {
529 case PXCC_TYPEDEF_UNNAMED:
530 if (clang_visitChildren(cursor, t_visit_register_type_fields,
531 type_stack))
532 {
533 return -1;
534 }
535 add_new_typedef_record(record_name, cursor, typedef_kind);
536 break;
537
538 default: {
539 switch (underlying_type.kind) {
540 case CXType_FunctionNoProto:
541 case CXType_FunctionProto:
542 /* In case of typedef to function, we need to do first add the
543 * record and then register the type. */
544 add_new_typedef_record(record_name, cursor, typedef_kind);
545 RETHROW(t_register_type(underlying_type, type_stack));
546 break;
547
548 default:
549 RETHROW(t_register_type(underlying_type, type_stack));
550 add_new_typedef_record(record_name, cursor, typedef_kind);
551 break;
552 }
553 } break;
554 }
555
556 return 0;
557}

559static int t_register_function_type(CXType type, qv_t(cstr) *type_stack)
560{
561 CXType res_type = clang_getResultType(type);
562 int n = clang_getNumArgTypes(type);
563 int res;
564
565 qv_append(type_stack, "res");
566 res = t_register_type(res_type, type_stack);
567 qv_remove_last(type_stack);
568 RETHROW(res);
569
570 for (int i = 0; i < n; i++) {
571 qv_append(type_stack, t_fmt("arg%d", i));
572 res = t_register_type(clang_getArgType(type, i), type_stack);
573 qv_remove_last(type_stack);
574 RETHROW(res);
575 }
576 return 0;
577}

592static int t_register_record_enum_type(CXType type, qv_t(cstr) *type_stack)
593{
594 const ctype_desc_t *non_anonymous_ctype;
595 CXString spelling;
596 lstr_t name;
597 CXCursor cursor;
598 bool is_anonymous;
599 pxcc_record_name_t *record_name;
600 lstr_t canonical_name;
601 qv_t(cstr) local_type_stack;
602
603 spelling = clang_getTypeSpelling(type);
604 name = get_unconst_type_spelling(spelling);
605 cursor = clang_getTypeDeclaration(type);
606
607 non_anonymous_ctype = get_non_anonymous_ctype();
608 is_anonymous = !lstr_match_ctype(name, non_anonymous_ctype);
609 if (is_anonymous) {
610 /* When the record is anonymous, we need to create a custom type in
611 * Cython */
612 const char *prefix = get_cursor_kind_prefix(cursor);
613
614 RETHROW_PN(prefix);
615 name = t_concat_type_stack(type_stack, "__", prefix, "_t");
616 }
617
618 record_name = get_or_add_new_record_name(name);
619 clang_disposeString(spelling);
620
621 switch (record_name->status) {
622 case PXCC_RECORD_NAME_NEW:
623 /* It is new, process it as usual. */
624 record_name->status = PXCC_RECORD_NAME_VISITED;
625 break;
626
627 case PXCC_RECORD_NAME_VISITED:
628 /* Recursive call for this record name.
629 * Create a forward record, and wait for complete record later on.
630 */
631 add_new_record(record_name, cursor, PXCC_RECORD_FORWARD);
632 record_name->status = PXCC_RECORD_NAME_FORWARDED;
633 return 0;
634
635 case PXCC_RECORD_NAME_FORWARDED:
636 case PXCC_RECORD_NAME_COMPLETED:
637 /* The record was already forwarded or registered,
638 * nothing to do here. */
639 return 0;
640 }
641
642 canonical_name = get_canonical_record_enum_type_name(record_name->name);
643 if (is_anonymous) {
644 /* When the record is anonymous, we want to register the name of the
645 * custom type corresponding to the cursor */
646 qm_add(cxcursor_name, &_G.anonymous_types, &cursor, canonical_name);
647 } else {
648 t_qv_init(&local_type_stack, 1);
649 qv_append(&local_type_stack, canonical_name.s);
650 type_stack = &local_type_stack;
651 }
652
653 if (clang_visitChildren(cursor, t_visit_register_type_fields, type_stack))
654 {
655 return -1;
656 }
657
658 add_new_record(record_name, cursor, PXCC_RECORD_CANONICAL_TYPE);
659
660 return 0;
661}

663static int t_register_type(CXType type, qv_t(cstr) *type_stack)
664{
665 for (;;) {
666 switch (type.kind) {
667 case CXType_Void ... CXType_Complex:
668 return 0;
669
670 case CXType_Pointer:
671 type = clang_getPointeeType(type);
672 break;
673
674 case CXType_Typedef:
675 return t_register_typedef_type(type, type_stack);
676
677 case CXType_FunctionNoProto:
678 case CXType_FunctionProto:
679 return t_register_function_type(type, type_stack);
680
681 case CXType_Unexposed:
682 type = resolve_unexposed_type(type);
683 break;
684
685 case CXType_Record:
686 case CXType_Enum:
687 return t_register_record_enum_type(type, type_stack);
688
689 case CXType_ConstantArray:
690 case CXType_IncompleteArray:
691 case CXType_VariableArray:
692 case CXType_DependentSizedArray:
693 type = clang_getArrayElementType(type);
694 break;
695
696#ifdef PXCC_HAS_ELABORATED_TYPE
697 case CXType_Elaborated:
698 type = clang_Type_getNamedType(type);
699 break;
700#endif
701
702 default: {
703 CXString type_spelling = clang_getTypeSpelling(type);
704 CXString kind_spelling = clang_getTypeKindSpelling(type.kind);
705 lstr_t path = t_concat_type_stack(type_stack, "::", "", "");
706
707 fprintf(stderr, "unsupported type of kind %s (%s - %d) for %pL\n",
708 clang_getCString(type_spelling),
709 clang_getCString(kind_spelling), type.kind, &path);
710 clang_disposeString(type_spelling);
711 clang_disposeString(kind_spelling);
712 assert (false);
713 return -1;
714 }
715 }
716 }
717
718 return 0;
719}

721static CXChildVisitResult
722type_decl_visitor(CXCursor cursor, CXCursor parent, CXClientData data)
723{
724 t_scope;
725 CXCursorKind kind = clang_getCursorKind(cursor);
726 qv_t(cstr) type_stack;

752static int register_symbol(CXCursor cursor)
753{
754 t_scope;
755 pxcc_record_name_t *record_name;
756 qv_t(cstr) type_stack;
757
758 cursor = clang_getCursorReferenced(cursor);
759 record_name = get_or_add_new_record_name_symbol(cursor);
760
761 switch (record_name->status) {
762 case PXCC_RECORD_NAME_NEW:
763 /* It is new, process it as usual. */
764 record_name->status = PXCC_RECORD_NAME_VISITED;
765 break;
766
767 case PXCC_RECORD_NAME_VISITED:
768 case PXCC_RECORD_NAME_FORWARDED:
769 /* Should not happen, recursive call should not be present for
770 * symbols. */
771 assert(false);
772 break;

774 case PXCC_RECORD_NAME_COMPLETED:
775 /* The record was already forwarded or registered,
776 * nothing to do here. */
777 return 0;
778 }
779
780 t_qv_init(&type_stack, 0);
781 RETHROW(t_register_type_cursor(cursor, &type_stack));
782
783 add_new_record(record_name, cursor, PXCC_RECORD_SYMBOL);

1054static void t_print_field_type(CXType type, pxcc_print_field_t *ctx)
1055{
1056 bool loop_prev_is_ptr = false;
1057 bool is_elaborated_type_const = false;
1058
1059 for (bool end_loop = false; !end_loop;) {
1060 bool prev_is_ptr = loop_prev_is_ptr;
1061
1062 loop_prev_is_ptr = false;
1063 switch (type.kind) {
1064 case CXType_Void:
1065 case CXType_Char_U ... CXType_Complex:
1066 case CXType_Typedef:
1067 t_print_canonical_type(type, ctx);
1068 end_loop = true;
1069 break;
1070
1071 case CXType_Bool:
1072 t_print_bool_type(type, ctx);
1073 end_loop = true;
1074 break;
1075
1076 case CXType_Pointer:
1077 type = t_print_pointer_type(type, ctx);
1078 loop_prev_is_ptr = true;
1079 break;
1080
1081 case CXType_Unexposed:
1082 t_print_parentheses_prev_is_ptr(ctx, prev_is_ptr);
1083 type = t_print_unexposed_type(type, ctx);
1084 break;
1085
1086 case CXType_ConstantArray:
1087 case CXType_IncompleteArray:
1088 case CXType_VariableArray:
1089 case CXType_DependentSizedArray:
1090 type = t_print_array_type(type, ctx);
1091 break;
1092
1093 case CXType_FunctionNoProto:
1094 case CXType_FunctionProto:
1095 t_print_parentheses_prev_is_ptr(ctx, prev_is_ptr);
1096 type = t_print_function_type(type, ctx);
1097 break;
1098
1099 case CXType_Record:
1100 case CXType_Enum:
1101 t_print_record_enum_field(type, ctx);
1102 end_loop = true;
1103 break;
1104
1105#ifdef PXCC_HAS_ELABORATED_TYPE
1106 case CXType_Elaborated:
1107#ifdef PXCC_IS_ELABORATED_TYPE_RESOLVE_UNCONST
1108 if (clang_isConstQualifiedType(type)) {
1109 is_elaborated_type_const = true;
1110 }
1111#endif /* PXCC_IS_ELABORATED_TYPE_RESOLVE_UNCONST */
1112 type = clang_Type_getNamedType(type);
1113 break;
1114#endif /* PXCC_HAS_ELABORATED_TYPE */
1115
1116 default:
1117 fprintf(stderr, "unsupported type of kind %d\n", type.kind);
1118 assert (false);
1119 end_loop = true;
1120 break;
1121 }
1122 }
1123
1124 if (is_elaborated_type_const) {
1125 t_print_field_add_before(ctx, "const ");
1126 }

1475static int call_cmd(const char *args[], sb_t *err)
1476{
1477 const char *prg_name = args[0];
1478 int fde[2];
1479 pid_t pid;
1480 int status;
1481
1482 if (pipe(fde) < 0) {
1483 fprintf(stderr, "unable to pipe(): %m");
1484 return -1;
1485 }
1486
1487 pid = ifork();
1488 if (pid < 0) {
1489 fprintf(stderr, "unable to fork(): %m");
1490 return -1;
1491 }
1492
1493 if (pid == 0) {
1494 close(fde[0]);
1495 dup2(fde[1], STDERR_FILENO);
1496 close(fde[1]);
1497
1498 setpgid(0, 0);
1499 execvp(args[0], (char * const *)args);
1500 e_fatal("unable to run %s: %m", prg_name);
1501 }
1502
1503 close(fde[1]);
1504 read_call_fd(prg_name, fde[0], err);
1505 close(fde[0]);
1506
1507 for (;;) {
1508 if (waitpid(pid, &status, 0) < 0) {
1509 e_fatal("waitpid: %m");
1510 }
1511 if (WIFEXITED(status)) {
1512 return WEXITSTATUS(status) ? -1 : 0;
1513 }
1514 if (WIFSIGNALED(status)) {
1515 fprintf(stderr, "%s killed with signal %s",
1516 prg_name, strsignal(WTERMSIG(status)));
1517 return -1;
1518 }
1519 }
1520
1521 return 0;
1522}
