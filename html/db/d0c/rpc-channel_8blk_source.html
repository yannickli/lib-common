96static void ic_read_tracing(void)
97{
98#ifndef NDEBUG
99 /* IC_TRACE environment variable contains a list of interface or
100 * interface.rpc to trace.
101 */
102 const char *trace = getenv("IC_TRACE");
103 pstream_t ps;
104
105 if (!trace) {
106 return;
107 }
108
109 ps = ps_initstr(trace);
110 ps_ltrim(&ps);
111
112 while (!ps_done(&ps)) {
113 pstream_t n;
114
115 ps_ltrim(&ps);
116 if (ps_get_ps_chr(&ps, ' ', &n) < 0) {
117 n = ps;
118 ps = ps_init(NULL, 0);
119 }
120
121 qv_append(&_G.traced_names, LSTR_PS_V(&n));
122 }
123#endif
124}

144/* This function can return false positives, because several RPCs can have the
145 * same cmd... */
146static _Bool ic_cmd_is_traced(int cmd)
147{
148 static bool first_call = true;
149
150 if (unlikely(first_call)) {
151 /* XXX: The filling of _G.traced_cmds is done lazily because when
152 * ic_initialize is called, the IOP packages were likely not
153 * registered yet. */
154 iop_for_each_registered_pkgs(^void (const iop_pkg_t *pkg) {
155 for (const iop_mod_t *const *m = pkg->mods; *m; m++) {
156 const iop_mod_t *mod = *m;
157
158 for (int ia = 0; ia < mod->ifaces_len; ia++) {
159 const iop_iface_alias_t *iface_alias = mod->ifaces + ia;
160 const iop_iface_t *iface = iface_alias->iface;
161
162 for (int i = 0; i < iface->funs_len; i++) {
163 const iop_rpc_t *rpc = &iface->funs[i];
164
165 if (__ic_rpc_is_traced(iface, rpc)) {
166 uint32_t rcmd = iface_alias->tag << 16 | rpc->tag;
167
168 qh_add(u32, &_G.traced_cmds, rcmd);
169 }
170 }
171 }
172 }
173 });
174 first_call = false;
175 }
176
177 return qh_find(u32, &_G.traced_cmds, cmd & 0x7fff7fff) >= 0;
178}
179#endif

195static void ic_get_state(sb_t *buf)
196{
197 t_scope;
198 qv_t(table_hdr) hdr;
199 qv_t(table_data) rows;
200 table_hdr_t hdr_data[] = { {
201 .title = LSTR_IMMED("ICHANNEL"),
202 }, {
203 .title = LSTR_IMMED("NAME"),
204 }, {
205 .title = LSTR_IMMED("ELH"),
206 }, {
207 .title = LSTR_IMMED("CALLBACK"),
208 }, {
209 .title = LSTR_IMMED("KIND"),
210 }, {
211 .title = LSTR_IMMED("ADDRESS"),
212 }, {
213 .title = LSTR_IMMED("FLAGS"),
214 }, {
215 .title = LSTR_IMMED("PRIORITY"),
216 }, {
217 .title = LSTR_IMMED("NEXT SLOT"),
218 }, {
219 .title = LSTR_IMMED("PENDING QUERIES"),
220 }, {
221 .title = LSTR_IMMED("NB IOV"),
222 }, {
223 .title = LSTR_IMMED("IOV TOTAL LEN"),
224 }
225 };
226 uint32_t hdr_size = countof(hdr_data);
227
228 qv_init_static(&hdr, hdr_data, hdr_size);
229 t_qv_init(&rows, 200);
230
231 qm_for_each_pos(ic, pos, &_G.ics) {
232 t_SB(flags, 64);
233 ichannel_t *ic = _G.ics.values[pos];
234 qv_t(lstr) *tab = qv_growlen(&rows, 1);
235 lstr_t addr;
236
237 t_qv_init(tab, hdr_size);
238
239 qv_append(tab, t_lstr_fmt("%d / %p", ic->id, ic));
240 qv_append(tab, ic->name);
241 qv_append(tab, t_lstr_fmt("%p", ic->elh));
242 qv_append(tab, t_lstr_fmt("%p", ic->on_event));
243
244 if (ic->is_local) {
245 qv_append(tab, LSTR("local"));
246 } else
247 if (ic->is_spawned) {
248 qv_append(tab, LSTR("server"));
249 } else {
250 qv_append(tab, LSTR("client"));
251 }
252
253 if (ic->is_local) {
254 addr = LSTR("-");
255 } else
256 if (ic->is_spawned) {
257 addr = ic_get_client_addr(ic);
258 addr = t_lstr_fmt("peer: %*pM", LSTR_FMT_ARG(addr));
259 } else {
260 addr = t_addr_fmt_lstr(&ic->su);
261 }
262 if (ic->remote_addr.s) {
263 addr = t_lstr_fmt("%*pM (%*pM)", LSTR_FMT_ARG(addr),
264 LSTR_FMT_ARG(ic->remote_addr));
265 }
266 qv_append(tab, addr);
267
268#define ADD_FLAG(_field, _str) \
269 do { \
270 if (ic->_field) { \
271 if (flags.len) { \
272 sb_adds(&flags, ", "); \
273 } \
274 sb_adds(&flags, _str); \
275 } \
276 } while (0)
277
278 ADD_FLAG(is_trusted, "trusted");
279 ADD_FLAG(is_closing, "closing");
280 ADD_FLAG(no_autodel, "no autodel");
281 ADD_FLAG(is_seqpacket, "seqpacket");
282 ADD_FLAG(is_unix, "unix");
283 ADD_FLAG(auto_reconn, "auto reconn");
284 ADD_FLAG(do_el_unref, "do el_unref");
285 ADD_FLAG(cancel_guard, "cancel guard");
286 ADD_FLAG(queuable, "queuable");
287 ADD_FLAG(is_wiped, "wiped");
288 assert (!ic->is_wiped);
289
290 qv_append(tab, LSTR_SB_V(&flags));
291#undef ADD_FLAG
292
293 qv_append(tab, ev_priority_to_str(ic->priority));
294 qv_append(tab, t_lstr_fmt("%u", ic->nextslot));
295 qv_append(tab, t_lstr_fmt("%d", ic_queue_len(ic)));
296 qv_append(tab, t_lstr_fmt("%d", ic->iov.len));
297 qv_append(tab, t_lstr_fmt("%d", ic->iov_total_len));
298 }
299
300 sb_add_table(buf, &hdr, &rows);
301 sb_shrink(buf, 1);
302}

315/** Check the peer certificate is the Intersec's one.
316 *
317 * IC peers should all have the same certificate. This function is called
318 * whenever a peer sends its certificate to check if its certificate is the
319 * same than ours.
320 *
321 * OpenSSL server:
322 * - with SSL_VERIFY_PEER, will ask the client to give a certificate.
323 * - with SSL_VERIFY_FAIL_IF_NO_PEER_CERT, will reject the client if it does
324 * not provide a certificate.
325 *
326 * These two options, together with this callback, should ensure that the
327 * client and the server authenticate each other. However, because we're a bit
328 * paranoïd, we double check that they actually have received the expected
329 * certificate (see the call to SSL_get_peer_certificate in ic_tls_handshake).
330 */
331static int ic_verify_callback(int preverify_ok, X509_STORE_CTX *store_ctx)
332{
333 X509 *cert;
334
335 if (preverify_ok) {
336 /* OpenSSL considers the server OK. (It happens this function is
337 * called two times successively, the second time with preverify_ok
338 * being 1. In both case the comparison below seems ok. */
339 return preverify_ok;
340 }
341
342 cert = X509_STORE_CTX_get_current_cert(store_ctx);
343 return (cert && X509_cmp(cert, _G.certificate) == 0) ? 1 : preverify_ok;
344}

366static int ic_initialize(void *arg)
367{
368 t_scope;
369 lstr_t tls_key = t_farch_get_data(ic_fc_g, "ic-key.pem");
370 lstr_t tls_cert = t_farch_get_data(ic_fc_g, "ic-cert.pem");
371 SB_1k(errbuf);
372
373 _G.certificate = ssl_ctx_convert_certificate_lstr(tls_cert);
374 if (!_G.certificate) {
375 logger_fatal(&_G.logger, "cannot load IC TLS certificate");
376 }
377
378 /* Configure a default TLS context with the IC TLS key. */
379 _G.ssl_ctx = ssl_ctx_new_tls(TLS_method(), tls_key, tls_cert,
380 SSL_VERIFY_PEER | SSL_VERIFY_FAIL_IF_NO_PEER_CERT,
381 &ic_verify_callback, &errbuf);
382 if (!_G.ssl_ctx) {
383 logger_fatal(&_G.logger, "cannot initialize IC TLS context: %*pM",
384 SB_FMT_ARG(&errbuf));
385 }
386
387 /* Other module initialization. */
388 qm_init(ic, &_G.ics);
389 qm_init(ic_hook_ctx, &_G.hook_ctxs);
390 qv_init(&_G.traced_names);
391 qh_init(u32, &_G.traced_cmds);
392 ic_read_tracing();
393 return 0;
394}

396static int ic_shutdown(void)
397{
398 if (qm_len(ic, &_G.ics)) {
399 if (logger_is_traced(&_G.logger, 1)) {
400 SB_1k(buf);
401
402 ic_get_state(&buf);
403 logger_trace(&_G.logger, 1, "%d ichannels are leaked:\n%*pM",
404 qm_len(ic, &_G.ics), SB_FMT_ARG(&buf));
405 } else {
406 logger_trace(&_G.logger, 0, "%d ichannels are leaked",
407 qm_len(ic, &_G.ics));
408 }
409 }
410
411 if (_G.ic_hook_ctx) {
412 ic_hook_ctx_t *ctx = _G.ic_hook_ctx;
413
414 ic_hook_ctx_delete(&ctx);
415 }
416
417 qv_deep_wipe(&_G.traced_names, lstr_wipe);
418 qh_wipe(u32, &_G.traced_cmds);
419 qm_deep_wipe(ic_hook_ctx, &_G.hook_ctxs, IGNORE, ic_hook_ctx_delete);
420 qm_wipe(ic, &_G.ics);
421 SSL_CTX_free(_G.ssl_ctx);
422 X509_free(_G.certificate);
423 _G.certificate = NULL;
424 return 0;
425}

539 msg->cmd = -cmd;

589int
590ic_query_do_pre_hook(ichannel_t *ic, uint64_t slot, const ic_cb_entry_t *e,
591 ic__hdr__t *hdr, bool *hdr_modified)
592{
593 if (e->pre_hook) {
594 _G.post_hook = e->post_hook;
595 _G.rpc = e->rpc;
596 _G.post_args = e->post_hook_args;
597 if (_G.ic_hook_ctx) {
598 assert (_G.ic_hook_ctx->slot != slot);
599 /* XXX Make sure "_G.ic_hook_ctx" is null or we won't be
600 * able to detect errors in pre_hook.
601 */
602 ic_hook_ctx_save_current();
603 }
604 (*e->pre_hook)(ic, slot, hdr, e->pre_hook_args, hdr_modified);
605 /* XXX: if we reply to the query during pre_hook then
606 * _G.ic_hook_ctx will be NULL, and we mustn't call the implementation
607 * of the RPC
608 * TODO: it would be better to rely on a pre_hook return code than on
609 * _G.ic_hook_ctx
610 */
611 if (!_G.ic_hook_ctx) {
612 return -1;
613 }
614 }
615
616 return 0;
617}

646static void ic_proxify(ichannel_t *pxy_ic, ic_msg_t *msg, int cmd,
647 const void *data, int dlen,
648 const ic_msg_t *unpacked_msg)
649{
650 uint64_t slot = get_unaligned_le64(msg->priv);
651 ichannel_t *ic;
652
653 cmd = -cmd;
654
655 if (unlikely(ic_slot_is_http(slot))) {
656 __ichttp_proxify(slot, cmd, data, dlen);
657 return;
658 }
659 ic = ic_get_from_slot(slot);
660 if (unlikely(!ic)) {
661 return;
662 }
663 if (likely(ic_can_reply(ic, slot))) {
664 ic_msg_t *tmp = ic_msg_new_fd(pxy_ic ? ic_get_fd(pxy_ic) : -1, 0);
665
666 ic_query_do_post_hook(ic, cmd, slot, NULL, NULL);
667 ic_msg_init_for_reply(ic, tmp, slot, cmd);
668 if (unpacked_msg && (likely(cmd == IC_MSG_OK) || cmd == IC_MSG_EXN)) {
669 const iop_struct_t *st;
670
671 assert (msg->rpc);
672 st = likely(cmd == IC_MSG_OK) ? msg->rpc->result : msg->rpc->exn;
673 tmp->force_dup = msg->force_dup;
674 __ic_msg_build(tmp, st, unpacked_msg->data, !ic_is_local(ic));
675 } else {
676 void *buf;
677
678 assert (!unpacked_msg || dlen == 0 || cmd == IC_MSG_INVALID);
679
680 if (ic_is_local(ic)) {
681 tmp->force_pack = !unpacked_msg;

685 if (data) {
686 memcpy(buf, data, dlen);
687 } else {
688 assert (dlen == 0);
689 }
690 }
691 ic_queue_for_reply(ic, tmp);
692 }

708void __ic_forward_reply_to(ichannel_t *pxy_ic, uint64_t slot, int cmd,
709 const void *res, const void *exn)
710{
711 ichannel_t *ic;
712
713 if (!pxy_ic) {
714 logger_panic(&_G.logger,
715 "are you trying to forward a webservices answer?");
716 }
717
718 if (unlikely(ic_slot_is_http(slot))) {
719 __ichttp_forward_reply(pxy_ic, slot, cmd, res, exn);
720 return;
721 }
722
723 if (cmd != IC_MSG_OK && cmd != IC_MSG_EXN) {
724 assert (!res);
725 ic_reply_err2(NULL, slot, cmd, exn);
726 return;
727 }
728
729 if (unlikely(!(ic = ic_get_from_slot(slot)))) {
730 return;
731 }
732 if (likely(ic_can_reply(ic, slot))) {
733 ic_msg_t *tmp = ic_msg_new_fd(ic_get_fd(pxy_ic), 0);
734 const sb_t *buf = &pxy_ic->rbuf;
735 const void *data = buf->data + IC_MSG_HDR_LEN;
736 int dlen = get_unaligned_le32(buf->data + IC_MSG_DLEN_OFFSET);
737
738 ic_query_do_post_hook(ic, cmd, slot, NULL, NULL);
739 ic_msg_init_for_reply(ic, tmp, slot, cmd);
740 if (ic_is_local(pxy_ic) && !get_unaligned_le32(buf->data)) {
741 const ic_msg_t *msg_org = data;
742 const iop_struct_t *st;
743
744 assert (dlen == sizeof(ic_msg_t) && msg_org->rpc);
745
746 if (likely(cmd == IC_MSG_OK)) {
747 st = msg_org->rpc->result;
748 data = res;
749 } else {
750 st = msg_org->rpc->exn;
751 data = exn;
752 }
753 tmp->force_dup = msg_org->force_dup;
754 __ic_msg_build(tmp, st, data, !ic_is_local(ic));
755 } else {
756 if (ic_is_local(ic)) {
757 tmp->force_pack = true;
758 }
759
760 memcpy(__ic_get_buf(tmp, dlen), data, dlen);
761 }
762 ic_queue_for_reply(ic, tmp);
763 }

784static void ic_cancel_all(ichannel_t *ic)
785{
786 ic_msg_t *msg;
787 qm_t(ic_msg) h;
788
789#ifndef NDEBUG
790 ic->cancel_guard = true;
791#endif
792
793 while (!htlist_is_empty(&ic->iov_list)) {
794 msg = htlist_pop_entry(&ic->iov_list, ic_msg_t, msg_link);
795 if (msg->cmd <= 0 || msg->slot == 0) {
796 ic_msg_delete(&msg);
797 }
798 }
799 while (!htlist_is_empty(&ic->msg_list)) {
800 msg = ic_pop_msg(ic);
801 if (msg->cmd <= 0 || msg->slot == 0) {
802 ic_msg_delete(&msg);
803 }
804 }
805
806 while (!htlist_is_empty(&ic->local_async_queries)) {
807 msg = htlist_pop_entry(&ic->local_async_queries, ic_msg_t, msg_link);
808 __ic_msg_reply_err(ic, msg, IC_MSG_ABORT);
809 if (msg->cmd <= 0 || msg->slot == 0) {
810 ic_msg_delete(&msg);
811 }
812 }
813 while (!htlist_is_empty(&ic->local_async_replies)) {
814 msg = htlist_pop_entry(&ic->local_async_replies, ic_msg_t, msg_link);
815 ic_msg_delete(&msg);
816 }
817
818 ic->last_normal_prio_msg = NULL;
819
820 h = ic->queries;
821 qm_init(ic_msg, &ic->queries);
822 qm_for_each_pos(ic_msg, pos, &h) {
823 qm_del_at(ic_msg, &h, pos);
824
825 msg = h.values[pos];
826 __ic_msg_reply_err(ic, msg, IC_MSG_ABORT);
827 ic_msg_delete(&msg);
828 }
829 qm_wipe(ic_msg, &h);
830 qv_deep_clear(&ic->fds, p_close);
831
832 /* if that crashes, one of the IC_MSG_ABORT callback reenqueues directly in
833 * that ichannel_t which is forbidden, fix the code
834 */
835 assert (qm_len(ic_msg, &ic->queries) == 0);
836
837#ifndef NDEBUG
838 ic->cancel_guard = false;
839#endif
840}

916static int ic_write(ichannel_t *ic, int fd)
917{
918#define IC_MAX_FD 32
919 char buf[CMSG_SPACE(sizeof(int[IC_MAX_FD]))]
920 __attribute__((aligned(alignof(struct cmsghdr))));
921 bool timer_restarted = false;
922
923 /* TODO Reestablish
924 assert (ic->is_connected);
925 */
926
927 do {
928 struct cmsghdr *hdr = (struct cmsghdr *)buf;
929 int *fdv = (int *)CMSG_DATA(hdr);
930 int fdc = 0;
931 struct msghdr msgh = { .msg_name = NULL };
932 ssize_t res;
933 ic_msg_t *msg;
934 ic_msg_t *last_fd_msg = NULL;
935
936 while (!htlist_is_empty(&ic->msg_list) && ic->iov_total_len < IC_PKT_MAX) {
937 msg = ic_pop_msg(ic);
938
939 if (msg->canceled) {
940 ic_msg_take_and_delete(msg);
941 continue;
942 }
943
944 if (unlikely(fdc == IC_MAX_FD && msg->fd >= 0)) {
945 htlist_add(&ic->msg_list, &msg->msg_link);
946 break;
947 }
948
949 ic_msg_trace(msg, "putting %d bytes in out vector", msg->dlen);
950 htlist_add_tail(&ic->iov_list, &msg->msg_link);
951
952 if (ic->is_seqpacket && ic->iov_total_len + msg->dlen > IC_PKT_MAX)
953 {
954 size_t to_send = IC_PKT_MAX - ic->iov_total_len;
955 size_t remain = msg->dlen;
956
957 while (remain) {
958 qv_append(&ic->iov,
959 MAKE_IOVEC((char *)msg->data + msg->dlen - remain,
960 to_send));
961 remain -= to_send;
962 to_send = MIN(IC_PKT_MAX, remain);
963 }
964 } else {
965 qv_append(&ic->iov, MAKE_IOVEC(msg->data, msg->dlen));
966 }
967 ic->iov_total_len += msg->dlen;
968
969 if (msg->fd >= 0) {
970 fdv[fdc++] = msg->fd;
971 last_fd_msg = msg;
972 }
973 }
974
975 if (!ic->iov_total_len) {
976 break;
977 }
978
979 if (fdc > 0) {
980 assert (ic->is_unix);
981 hdr->cmsg_len = (char *)(fdv + fdc) - buf;
982 hdr->cmsg_level = SOL_SOCKET;
983 hdr->cmsg_type = SCM_RIGHTS;
984 msgh.msg_control = hdr;
985 msgh.msg_controllen = hdr->cmsg_len;
986 } else {
987 msgh.msg_control = NULL;
988 msgh.msg_controllen = 0;
989 }
990 msgh.msg_iov = ic->iov.tab;
991 msgh.msg_iovlen = ic->iov.len;
992 if (ic->is_seqpacket && ic->iov_total_len > IC_PKT_MAX) {
993 size_t total = ic->iov_total_len;
994
995 while (total > IC_PKT_MAX) {
996 msgh.msg_iovlen--;
997 total -= msgh.msg_iov[msgh.msg_iovlen].iov_len;
998 }
999 }
1000 msgh.msg_iovlen = MIN(msgh.msg_iovlen, IOV_MAX);
1001
1002 /* We can use ssl_writev because the msg control header is only used
1003 * with unix sockets (fdc > 0). */
1004 res = ic->ssl ?
1005 ssl_writev(fd, msgh.msg_iov, msgh.msg_iovlen, ic->ssl) :
1006 sendmsg(fd, &msgh, 0);
1007
1008 if (res < 0) {
1009 return ERR_RW_RETRIABLE(errno) ? 0 : -1;
1010 }
1011 if (ic->timer && !timer_restarted) {
1012 el_timer_restart(ic->timer, 0);
1013 timer_restarted = true;
1014 }
1015 iovec_vector_kill_first(&ic->iov, res);
1016 ic->iov_total_len -= res;
1017
1018 while (!htlist_is_empty(&ic->iov_list)) {
1019 msg = htlist_first_entry(&ic->iov_list, ic_msg_t, msg_link);
1020
1021 if (msg->fd >= 0) {
1022 close(msg->fd);
1023 msg->fd = -1;
1024 ic_msg_trace(msg, "fd sent");
1025
1026 if (msg == last_fd_msg) {
1027 last_fd_msg = NULL;
1028 }
1029 }
1030
1031 if (ic->iov.len != 0) {
1032 const struct iovec *iov = &ic->iov.tab[0];
1033 const char *msg_end = (char *)msg->data + msg->dlen;
1034 const char *iov_end = (char *)iov->iov_base + iov->iov_len;
1035
1036 if (iov_end >= (const char *)msg->data && iov_end <= msg_end)
1037 {
1038 /* The message is not fully sent */
1039 break;
1040 }
1041 }
1042
1043 htlist_pop(&ic->iov_list);
1044 ic_msg_trace(msg, "written on socket");
1045 if (msg->cmd <= 0 || msg->slot == 0) {
1046 ic_msg_delete(&msg);
1047 }
1048 }
1049
1050 if (last_fd_msg) {
1051 htlist_for_each_entry(msg, &ic->iov_list, msg_link) {
1052 if (msg->fd >= 0) {
1053 close(msg->fd);
1054 msg->fd = -1;
1055 ic_msg_trace(msg, "fd sent");
1056
1057 if (msg == last_fd_msg) {
1058 break;
1059 }
1060 }
1061 }
1062 }
1063
1064 assert (htlist_is_empty(&ic->iov_list) || ic->iov_total_len);
1065 } while (ic->iov_total_len || !htlist_is_empty(&ic->msg_list));
1066
1067 if (ic->elh) {
1068 el_fd_set_mask(ic->elh, POLLIN);
1069 }
1070 return 1;
1071}

1073static void ic_parse_cmsg(ichannel_t *ic, struct msghdr *msgh)
1074{
1075 struct cmsghdr *cmsg;
1076
1077 for (cmsg = CMSG_FIRSTHDR(msgh); cmsg; cmsg = CMSG_NXTHDR(msgh, cmsg)) {
1078 if (cmsg->cmsg_level != SOL_SOCKET) {
1079 continue;
1080 }
1081 switch (cmsg->cmsg_type) {
1082 case SCM_RIGHTS: {
1083 int fd_count = (cmsg->cmsg_len - sizeof(*cmsg)) / sizeof(int);
1084 int *fds = acast(int, CMSG_DATA(cmsg));
1085
1086 p_copy(qv_growlen(&ic->fds, fd_count), fds, fd_count);
1087 } break;
1088 }
1089 }

1096static ALWAYS_INLINE int
1097ic_read_process_answer(ichannel_t *ic, int cmd, uint32_t slot,
1098 const void *data, int dlen,
1099 const ic_msg_t *unpacked_msg)
1100{
1101 ic_msg_t *tmp = ic_query_take(ic, slot);
1102 const iop_struct_t *st;
1103
1104 if (unlikely(!tmp)) {
1105 errno = 0;
1106 return -1;
1107 }
1108
1109 ic_msg_trace(tmp, "processing answer");
1110
1111 if (tmp->canceled) {
1112 goto wipe;
1113 }
1114
1115 if (ic_is_local(ic)) {
1116 sb_reset(&ic->rbuf);
1117 sb_addnc(&ic->rbuf, IC_MSG_HDR_LEN, 0);
1118 put_unaligned_le32(ic->rbuf.data + IC_MSG_CMD_OFFSET, cmd);
1119 if (data) {
1120 put_unaligned_le32(ic->rbuf.data, slot);
1121 put_unaligned_le32(ic->rbuf.data + IC_MSG_DLEN_OFFSET, dlen);
1122 sb_add(&ic->rbuf, data, dlen);
1123 } else
1124 if (unpacked_msg) {
1125 put_unaligned_le32(ic->rbuf.data + IC_MSG_DLEN_OFFSET,
1126 sizeof(ic_msg_t));
1127 sb_add(&ic->rbuf, unpacked_msg, sizeof(ic_msg_t));
1128 ((ic_msg_t *)(ic->rbuf.data + IC_MSG_HDR_LEN))->rpc = tmp->rpc;
1129 }
1130 }
1131
1132 tmp->raw_res = ps_init(data, dlen);
1133 if (tmp->cb == IC_PROXY_MAGIC_CB) {
1134 ic_proxify(ic, tmp, cmd, data, dlen, unpacked_msg);
1135 goto wipe;
1136 }
1137
1138 cmd = -cmd;
1139 if (likely(cmd == IC_MSG_OK)) {
1140 st = tmp->rpc->result;
1141 } else
1142 if (cmd == IC_MSG_EXN) {
1143 st = tmp->rpc->exn;
1144 } else {
1145 assert (dlen == 0 || cmd == IC_MSG_INVALID);
1146 if (dlen > 0) {
1147 lstr_t err = LSTR_INIT_V(data, dlen - 1);
1148 iop_set_err2(&err);
1149 ic_msg_trace(tmp, "%s %*pM", ic_status_to_string(cmd),
1150 LSTR_FMT_ARG(err));
1151 (*tmp->cb)(ic, tmp, cmd, NULL, &err);
1152 } else {
1153 iop_clear_err();
1154 ic_msg_trace(tmp, "%s", ic_status_to_string(cmd));
1155 (*tmp->cb)(ic, tmp, cmd, NULL, NULL);
1156 }
1157 goto wipe;
1158 }
1159 if (!tmp->raw) {
1160 t_scope;
1161 void *value = NULL;
1162 pstream_t ps = tmp->raw_res;
1163 unsigned unpack_flags = ic->is_public ? IOP_UNPACK_FORBID_PRIVATE : 0;
1164
1165 if (unlikely(t_get_value_of_st(st, unpacked_msg, ps,
1166 unpack_flags, &value) < 0))
1167 {
1168 if (tmp->trace) {
1169 const char *err = iop_get_err();
1170
1171 if (err) {
1172 ic_msg_trace(tmp, "rpc(%04x:%04x):%*pM: %s",
1173 (tmp->cmd >> 16) & 0x7fff, tmp->cmd & 0x7fff,
1174 LSTR_FMT_ARG(tmp->rpc->name), err);
1175 } else {
1176 ic_msg_trace(tmp, "rpc(%04x:%04x):%*pM: answer with "
1177 "invalid encoding",
1178 (tmp->cmd >> 16) & 0x7fff, tmp->cmd & 0x7fff,
1179 LSTR_FMT_ARG(tmp->rpc->name));
1180 }
1181 }
1182 t_seal();
1183 (*tmp->cb)(ic, tmp, IC_MSG_INVALID, NULL, NULL);
1184 } else
1185 if (cmd == IC_MSG_OK) {
1186 t_seal();
1187 ic_msg_trace(tmp, "OK");
1188 (*tmp->cb)(ic, tmp, cmd, value, NULL);
1189 } else {
1190 t_seal();
1191 ic_msg_trace(tmp, "%s", ic_status_to_string(cmd));
1192 (*tmp->cb)(ic, tmp, cmd, NULL, value);
1193 }
1194 } else {
1195 (*tmp->cb)(ic, tmp, cmd, NULL, NULL);
1196 }
1197
1198 wipe:
1199 ic_msg_delete(&tmp);
1200 if (ic_is_local(ic)) {
1201 sb_reset(&ic->rbuf);
1202 }
1203 return 0;
1204}

1286static int
1287t_get_hdr_value_of_query(ichannel_t *ic, int cmd,
1288 uint32_t slot, uint32_t flags,
1289 const void *data, int dlen,
1290 const ic_msg_t *unpacked_msg, const iop_struct_t *st,
1291 int *packed_hdr_len, ic__hdr__t **hdr, void **value)
1292{
1293 pstream_t ps = ps_init(data, dlen);
1294 unsigned unpack_flags = ic->is_public ? IOP_UNPACK_IGNORE_UNKNOWN : 0;
1295
1296#define QUERY_FMT "query %04x:%04x, type %s: "
1297#define QUERY_FMT_ARG \
1298 (cmd >> 16) & 0x7fff, cmd & 0x7fff, (st ? st->fullname.s : "<nil>")
1299
1300 if (ic_is_local(ic) && !unpacked_msg) {
1301 /* IOP payload comes from a volatile ic_msg_t that will be destroy
1302 * with ic_reply(), thus we need to duplicate strings */
1303 unpack_flags |= IOP_UNPACK_COPY_STRINGS;
1304 }
1305
1306 if (unlikely(flags & IC_MSG_HAS_HDR)) {
1307 if (unlikely(t_get_hdr_of_query(unpacked_msg, &ps,
1308 unpack_flags, hdr) < 0))
1309 {
1310 logger_warning(&_G.logger, QUERY_FMT "invalid header encoding",
1311 QUERY_FMT_ARG);
1312 return -1;
1313 }
1314 /* XXX on simple header we write the payload size of the iop query */
1315 if (unlikely(IOP_UNION_IS(ic__hdr, *hdr, simple))) {
1316 ic__simple_hdr__t *shdr = &(*hdr)->simple;
1317
1318 if (shdr->payload < 0) {
1319 shdr->payload = dlen;
1320 }
1321
1322 if (!shdr->host.len) {
1323 shdr->host = ic_get_client_addr(ic);
1324 } else {
1325 sockunion_t out;
1326 pstream_t host;
1327 in_port_t port;
1328
1329 if (addr_parse(ps_initlstr(&shdr->host), &host, &port, 0) < 0
1330 || addr_info(&out, AF_UNSPEC, host, port) < 0)
1331 {
1332 shdr->host = LSTR_NULL_V;
1333 }
1334 }
1335 }
1336 if (unlikely(ic_slot_is_traced(slot, flags)
1337 && logger_is_traced(&_G.tracing_logger, 2)))
1338 {
1339 SB_1k(sb);
1340
1341 sb_addf(&sb, "[slot:%x]; unpacked header: ",
1342 (uint32_t)(slot & IC_MSG_SLOT_MASK));
1343 iop_sb_jpack(&sb, &ic__hdr__s, *hdr, IOP_JPACK_SHORTEN_DATA);
1344 logger_trace(&_G.tracing_logger, 2, "%*pM", SB_FMT_ARG(&sb));
1345 }
1346 }
1347
1348 if (packed_hdr_len) {
1349 *packed_hdr_len = dlen - ps_len(&ps);
1350 }
1351
1352 if (value) {
1353 if (unlikely(t_get_value_of_st(st, unpacked_msg, ps,
1354 unpack_flags, value) < 0))
1355 {
1356 const char *err = iop_get_err();
1357
1358 if (err) {
1359 /* Constraints violation */
1360 logger_trace(&_G.logger, 0, QUERY_FMT "%s",
1361 QUERY_FMT_ARG, err);
1362 } else {
1363 logger_warning(&_G.logger, QUERY_FMT "invalid encoding",
1364 QUERY_FMT_ARG);
1365 }
1366 return -1;
1367 }
1368 if (unlikely(ic_slot_is_traced(slot, flags)
1369 && logger_is_traced(&_G.tracing_logger, 1)))
1370 {
1371 SB_1k(sb);
1372
1373 sb_addf(&sb, "[slot:%x]; unpacked %*pM: ",
1374 (uint32_t)(slot & IC_MSG_SLOT_MASK),
1375 LSTR_FMT_ARG(st->fullname));
1376 iop_sb_jpack(&sb, st, *value, IOP_JPACK_SHORTEN_DATA);
1377 logger_trace(&_G.tracing_logger, 1, "%*pM", SB_FMT_ARG(&sb));
1378 }
1379 }
1380
1381 return 0;
1382#undef QUERY_FMT
1383#undef QUERY_FMT_ARG
1384}

1386static void ic_report_unimplemented_rpc(int cmd)
1387{
1388 uint32_t icmd = cmd >> 16 & 0x7fff;
1389 uint32_t rcmd = cmd & 0x7fff;
1390 SB_1k(sb);
1391 __block sb_t *psb = &sb;
1392
1393 iop_for_each_registered_pkgs(^(const iop_pkg_t *pkg) {
1394 for (const iop_mod_t *const *m = pkg->mods; *m; m++) {
1395 const iop_mod_t *mod = *m;
1396
1397 for (int ia = 0; ia < mod->ifaces_len; ia++) {
1398 const iop_iface_alias_t *iface_alias = mod->ifaces + ia;
1399 const iop_iface_t *iface;
1400
1401 if (iface_alias->tag != icmd) {
1402 continue;
1403 }
1404 iface = iface_alias->iface;
1405 for (int i = 0; i < iface->funs_len; i++) {
1406 const iop_rpc_t *rpc = &iface->funs[i];
1407
1408 if (rpc->tag != rcmd) {
1409 continue;
1410 }
1411 if (psb->len) {
1412 sb_adds(psb, ", ");
1413 } else {
1414 sb_adds(psb, " so potentially ");
1415 }
1416 sb_add_lstr(psb, mod->fullname);
1417 sb_adds(psb, "::");
1418 sb_add_lstr(psb, iface_alias->name);
1419 sb_adds(psb, "::");
1420 sb_add_lstr(psb, rpc->name);
1421 }
1422 }
1423 }
1424 });
1425 logger_trace(&_G.logger, 0, "received query for unimplemented RPC "
1426 "(%04x:%04x%*pM)", icmd, rcmd, SB_FMT_ARG(&sb));
1427}

1467/* Returns an error if the ic must be closed with ic_mark_disconnected. */
1468static ALWAYS_INLINE __must_check__ int
1469ic_read_process_query(ichannel_t *ic, int cmd, uint32_t slot,
1470 uint32_t flags, const void *data, int dlen,
1471 const ic_msg_t * nullable unpacked_msg)
1472{
1473 t_scope;
1474 const ic_cb_entry_t *e;
1475 const iop_struct_t *st;
1476 ic__hdr__t *hdr = NULL;
1477 int hlen = -1;
1478 ichannel_t *pxy;
1479 ic__hdr__t *pxy_hdr = NULL;
1480 bool take_pxy_hdr = false;
1481 bool hdr_modified = false;
1482 uint64_t query_slot;
1483 ic_msg_t *tmp;
1484 int pos;
1485 void *value = NULL;
1486
1487 pos = likely(ic->impl) ? qm_find_safe(ic_cbs, ic->impl, cmd) : -1;
1488 if (unlikely(pos < 0)) {
1489 if (logger_is_traced(&_G.logger, 0)) {
1490 ic_report_unimplemented_rpc(cmd);
1491 }
1492 if (slot) {
1493 ic_reply_err(ic, MAKE64(ic->id, slot), IC_MSG_UNIMPLEMENTED);
1494 }
1495 return 0;
1496 }
1497 e = ic->impl->values + pos;
1498 st = e->rpc ? e->rpc->args : NULL;
1499
1500#ifndef NDEBUG
1501 /* Flag as traced incoming queries that are not traced but should be in
1502 * our process. */
1503 if (!ic_slot_is_traced(slot, flags) && unlikely(ic_cmd_is_traced(cmd))) {
1504 flags |= IC_MSG_IS_TRACED;
1505 }
1506#endif
1507 query_slot = MAKE64(ic->id, slot | flags);
1508 ic_slot_trace(slot, flags, "received traced query");
1509
1510 /* get details from the data of the msg. Each type of callback has
1511 * different details to retrieve. */
1512 if (t_get_details_of_query(ic, cmd, slot ,flags, data, dlen, unpacked_msg,
1513 st, e, &hlen, &hdr, &value) < 0)
1514 {
1515 goto invalid_iop;
1516 }
1517
1518 if (e->pre_hook) {
1519 t_seal();
1520 ic->cmd = cmd;
1521 if (ic_query_do_pre_hook(ic, query_slot, e, hdr, &hdr_modified) < 0) {
1522 ic->cmd = 0;
1523 return 0;
1524 }
1525 ic->cmd = 0;
1526 t_unseal();
1527 }
1528
1529 switch (e->cb_type) {
1530 case IC_CB_NORMAL:
1531 case IC_CB_NORMAL_BLK:
1532 case IC_CB_WS_SHARED: {
1533 bool is_async = e->rpc->async;
1534
1535 t_seal();
1536 ic->desc = e->rpc;
1537 ic->cmd = cmd;
1538 assert (value);
1539 if (e->cb_type == IC_CB_NORMAL_BLK) {
1540 (e->u.blk.cb)(ic, query_slot, value, hdr);
1541 } else {
1542 (*e->u.cb.cb)(ic, query_slot, value, hdr);
1543 }
1544 if (is_async) {
1545 ic_query_do_post_hook(ic, cmd, query_slot, NULL, NULL);
1546 }
1547 ic->desc = NULL;
1548 ic->cmd = 0;
1549 return 0;
1550 }
1551
1552 case IC_CB_PROXY_P:
1553 pxy = e->u.proxy_p.ic_p;
1554 pxy_hdr = e->u.proxy_p.hdr_p;
1555 break;
1556 case IC_CB_PROXY_PP:
1557 pxy = *e->u.proxy_pp.ic_pp;
1558 if (e->u.proxy_pp.hdr_pp) {
1559 pxy_hdr = *e->u.proxy_pp.hdr_pp;
1560 }
1561 break;
1562 case IC_CB_DYNAMIC_PROXY:
1563 {
1564 ic_dynproxy_t dynproxy;
1565
1566 /* XXX dynproxy are allowed to return memory allocated on the
1567 * t_pool() and thus mustn't be wrapped in a local t_scope */
1568 dynproxy = (*e->u.dynproxy.get_ic)(hdr, e->u.dynproxy.priv);
1569 pxy = dynproxy.ic;
1570 pxy_hdr = dynproxy.hdr;
1571 /* check for header replacement forcing */
1572 if (pxy_hdr && hdr) {
1573 take_pxy_hdr = true;
1574 }
1575 }
1576 break;
1577 default:
1578 logger_panic(&_G.logger, "should not happen");
1579 break;
1580 }
1581
1582 if (unlikely(!pxy || !ic_is_ready(pxy))) {
1583 if (slot) {
1584 ic_reply_err(ic, query_slot, IC_MSG_PROXY_ERROR);
1585 }
1586 return 0;
1587 }
1588
1589 /* do post hook for proxied async RPCs */
1590 if (e->pre_hook && !slot) {
1591 t_seal();
1592 ic->cmd = cmd;
1593 ic_query_do_post_hook(ic, cmd, query_slot, NULL, NULL);
1594 ic->cmd = 0;
1595 t_unseal();
1596 }
1597
1598 /* if the hdr has been modified, we have to force a repacking, to
1599 * ensure that the new hdr is properly saved in the proxied msg */
1600 if (hdr_modified && expect(hdr)) {
1601 /* if take_pxy_hdr is set, we have to use pxy_hdr instead of hdr, but
1602 * the pxy_hdr has been built using the modified hdr, so it can be
1603 * kept. */
1604 pxy_hdr = take_pxy_hdr ? pxy_hdr : hdr;
1605 take_pxy_hdr = true;
1606 }
1607
1608 take_pxy_hdr |= !(flags & IC_MSG_HAS_HDR) && pxy_hdr;
1609 tmp = ic_msg_proxy_new(ic_get_fd(ic), slot, NULL);
1610 if (slot) {
1611 tmp->cb = IC_PROXY_MAGIC_CB;
1612 } else {
1613 tmp->async = true;
1614 }
1615 tmp->cmd = cmd;
1616 put_unaligned_le64(tmp->priv, query_slot);
1617
1618 /* XXX We do not support header replacement with static proxyfication.
1619 */
1620
1621 if (unpacked_msg) {
1622 tmp->hdr = take_pxy_hdr ? pxy_hdr : unpacked_msg->hdr;
1623 if (ic_is_local(pxy)) {
1624 tmp->force_dup = unpacked_msg->force_dup;
1625 tmp->rpc = unpacked_msg->rpc;
1626 }
1627 __ic_msg_build(tmp, st, unpacked_msg->data, !ic_is_local(pxy));
1628 } else {
1629 if (ic_is_local(pxy)) {
1630 tmp->force_pack = true;
1631 }
1632
1633 if (take_pxy_hdr) {
1634 qv_t(i32) szs;
1635 uint8_t *buf;
1636
1637 /* skip the original hdr inside the data */
1638 if (hlen < 0) {
1639 assert (!(flags & IC_MSG_HAS_HDR));
1640 hlen = 0;
1641 }
1642 data = (const char *)data + hlen;
1643 dlen -= hlen;
1644
1645 /* Pack the given header in proxyfied request */
1646 qv_inita(&szs, 128);
1647 hlen = iop_bpack_size(&ic__hdr__s, pxy_hdr, &szs);
1648 buf = __ic_get_buf(tmp, hlen + dlen);
1649 iop_bpack(buf, &ic__hdr__s, pxy_hdr, szs.tab);
1650 qv_wipe(&szs);
1651 memcpy(buf + hlen, data, dlen);
1652 } else {
1653 memcpy(__ic_get_buf(tmp, dlen), data, dlen);
1654 }
1655 }
1656
1657 if (take_pxy_hdr) {
1658 flags |= IC_MSG_HAS_HDR;
1659 }
1660 __ic_query_flags(pxy, tmp, flags);
1661 return 0;
1662
1663 invalid_iop:
1664 if (slot) {
1665 lstr_t err_str = iop_get_err_lstr();
1666 ic_reply_err2(ic, query_slot, IC_MSG_INVALID, &err_str);
1667 }
1668
1669 /* Close connection unless we just had a constraint violation */
1670 THROW_ERR_UNLESS(iop_get_err());
1671 return 0;
1672}

1675 int flags)
1676{
1677 if (!ic->is_unix && (flags & IC_MSG_HAS_FD)) {
1678 ic_slot_trace(slot, flags, "invalid flags HAS_FD on stream ic %p",
1679 ic);
1680 assert (!ic->is_trusted);
1681 return -1;

1687 flags, ic);
1688 assert (!ic->is_trusted);
1689 return -1;
1690 }

1694static int ic_check_msg_hdr(ichannel_t *ic)
1695{
1696 const byte *hdr;
1697 uint32_t slot;
1698 uint32_t flags;
1699 int cmd;
1700 int dlen;
1701
1702 if (ic->hdr_checked) {
1703 return 0;
1704 }
1705
1706 hdr = (const byte *)ic->rbuf.data;
1707 slot = get_unaligned_le32(hdr);
1708 flags = slot & ~IC_MSG_SLOT_MASK;
1709 slot &= IC_MSG_SLOT_MASK;
1710 cmd = get_unaligned_le32(hdr + IC_MSG_CMD_OFFSET);
1711 dlen = get_unaligned_le32(hdr + IC_MSG_DLEN_OFFSET);
1712
1713 if (dlen < 0 || (uint32_t)dlen > MEM_ALLOC_MAX) {
1714 /* length is invalid */
1715 goto reject;
1716 }
1717
1718 RETHROW(ic_check_msg_hdr_flags(ic, slot, flags));
1719
1720 if (cmd == IC_MSG_STREAM_CONTROL) {
1721 switch (slot) {
1722 case IC_SC_BYE:
1723 case IC_SC_NOP:
1724 if (ic->peer_version <= IC_VERSION && dlen != 0) {
1725 goto reject;
1726 }
1727 break;
1728 case IC_SC_VERSION:
1729 if (ic->peer_version <= 0) {
1730 goto reject;
1731 }
1732 /* In case the remote peer shares a protocol version lower or
1733 * equal to ours, we check that the message payload is in the
1734 * min/max range. Invalid in-between lengths will be actually
1735 * caught during the parsing depending on the flags found in the
1736 * message.
1737 */
1738 if (ic->peer_version <= IC_VERSION &&
1739 (dlen < IC_MSG_VERSION_DLEN_MIN ||
1740 dlen > IC_MSG_VERSION_DLEN_MAX))
1741 {
1742 goto reject;
1743 }
1744 /* In case the remote peer has a greater version than ours, we can
1745 * only check that we have enough bytes (dlen-min) to parse the
1746 * message flags to handle forward compatibility. Indeed this peer
1747 * could use new flags and thus use a greater payload than we
1748 * thought possible.
1749 */
1750 if (dlen < IC_MSG_VERSION_DLEN_MIN) {
1751 goto reject;
1752 }
1753 break;
1754 default:
1755 if (ic->peer_version <= IC_VERSION) {
1756 goto reject;
1757 }
1758 }
1759 } else
1760 if (cmd <= 0) {
1761 switch (-cmd) {
1762 case IC_MSG_OK:
1763 case IC_MSG_EXN:
1764 case IC_MSG_INVALID:
1765 break;
1766
1767 case IC_MSG_RETRY:
1768 case IC_MSG_ABORT:
1769 case IC_MSG_UNIMPLEMENTED:
1770 case IC_MSG_SERVER_ERROR:
1771 case IC_MSG_PROXY_ERROR:
1772 /* no data are expected with those reply codes */
1773 if (ic->peer_version <= IC_VERSION && dlen != 0) {
1774 goto reject;
1775 }
1776 break;
1777
1778 default:
1779 /* unkown reply code */
1780 if (ic->peer_version <= IC_VERSION) {
1781 goto reject;
1782 }
1783 }
1784 if (dlen < 10 << 20) {
1785 /* don't lose time validating small messages, the correct checks
1786 * will be performed when treating the message. */
1787 goto accept;
1788 }
1789 /* reject messages that reply to unkown slot. */
1790 RETHROW(qm_find_safe(ic_msg, &ic->queries, slot));
1791 } else {
1792 if (!ic->impl) {
1793 /* no implementation, so nothing to reply */
1794 goto reject;
1795 }
1796 if (dlen < 10 << 20) {
1797 /* don't lose time validating small messages, the correct checks
1798 * will be performed when treating the message. */
1799 goto accept;
1800 }
1801 if (!ic->is_trusted) {
1802 /* unimplemented command */
1803 RETHROW(qm_find_safe(ic_cbs, ic->impl, cmd));
1804 }
1805 }
1806 accept:
1807 ic->hdr_checked = true;
1808 return 0;
1809
1810 reject:
1811 assert (!ic->is_trusted);
1812 return -1;
1813}

1815/* Return -1 on error, 0 on EAGAIN, the number of bytes read otherwise. */
1816static ssize_t _ic_read(ichannel_t *ic, short events, int sock, int to_read)
1817{
1818 ssize_t res;
1819 sb_t *buf = &ic->rbuf;
1820 char cmsgbuf[BUFSIZ];
1821 struct iovec iov;
1822 struct msghdr msgh = {
1823 .msg_iov = &iov,
1824 .msg_iovlen = 1,
1825 .msg_control = cmsgbuf,
1826 .msg_controllen = sizeof(cmsgbuf),
1827 };
1828
1829 if (!ic->is_unix) {
1830 res = ic->ssl ?
1831 ssl_sb_read(buf, ic->ssl, to_read) :
1832 sb_read(buf, sock, to_read);
1833 } else {
1834 iov = (struct iovec){
1835 .iov_base = sb_grow(buf, to_read),
1836 .iov_len = to_read,
1837 };
1838 res = recvmsg(sock, &msgh, 0);
1839 }
1840
1841 if (res < 0) {
1842 /* XXX: Workaround linux < 2.6.24 bug: it returns -EAGAIN *
1843 * instead of 0 in the hangup case for SOCK_SEQPACKETS */
1844 if ((events & POLLHUP) || !ERR_RW_RETRIABLE(errno)) {
1845 return -1;
1846 }
1847 return 0;
1848 }
1849 if (res == 0) {
1850 /* Graceful exit, errno was unchanged: give him a meaningful value. */
1851 errno = 0;
1852 return -1;
1853 }
1854
1855 if (ic->wa_soft > 0) {
1856 if (!ic->wa_soft_timer) {
1857 ic->on_event(ic, IC_EVT_ACT);
1858 ic->wa_soft_timer = el_timer_register(ic->wa_soft, 0, 0,
1859 ic_watch_act_soft, ic);
1860 el_unref(ic->wa_soft_timer);
1861 } else {
1862 el_timer_restart(ic->wa_soft_timer, 0);
1863 }
1864 }
1865
1866 if (ic->is_unix) {
1867 __sb_fixlen(buf, buf->len + res);
1868 ic_parse_cmsg(ic, &msgh);
1869
1870 /* XXX fd_overflow is set when the kernel decided that it cannot
1871 * transfer some ancillary data. In our case, this means we didn't
1872 * properly receive the expected file descriptor, probably because
1873 * there is no file descriptor available anymore on our side of
1874 * the channel.
1875 */
1876 ic->fd_overflow |= !!(msgh.msg_flags & MSG_CTRUNC);
1877 }
1878
1879 return res;
1880}

1882static int ic_read(ichannel_t *ic, short events, int sock)
1883{
1884 sb_t *buf = &ic->rbuf;
1885 ssize_t seqpkt_at_least = IC_PKT_MAX;
1886 int to_read = IC_PKT_MAX;
1887 bool starves = false;
1888 int write_errno = 0;
1889 bool try_write = true;
1890 ssize_t res;
1891
1892 if (likely(events & POLLIN)) {
1893 if (buf->len >= IC_MSG_HDR_LEN) {
1894 errno = 0;
1895 RETHROW(ic_check_msg_hdr(ic));
1896 to_read = get_unaligned_le32(buf->data + IC_MSG_DLEN_OFFSET);
1897 to_read += IC_MSG_HDR_LEN;
1898 to_read -= buf->len;
1899 }
1900
1901 again:
1902 to_read = MAX(to_read, IC_PKT_MAX);
1903 res = _ic_read(ic, events, sock, to_read);
1904 if (res <= 0) {
1905 return res;
1906 }
1907 if (!ic->is_unix) {
1908 starves = (sb_avail(buf) == 0);
1909 }
1910 if (ic->is_seqpacket) {
1911 seqpkt_at_least -= res;
1912 }
1913 }
1914
1915 while (buf->len >= IC_MSG_HDR_LEN) {
1916 void *data = buf->data + IC_MSG_HDR_LEN;
1917 int slot, dlen, cmd;
1918 int flags;
1919
1920 slot = get_unaligned_le32(buf->data);
1921 flags = slot & ~IC_MSG_SLOT_MASK;
1922 slot &= IC_MSG_SLOT_MASK;
1923 cmd = get_unaligned_le32(buf->data + IC_MSG_CMD_OFFSET);
1924 dlen = get_unaligned_le32(buf->data + IC_MSG_DLEN_OFFSET);
1925
1926 if (dlen < 0 || (uint32_t)dlen > MEM_ALLOC_MAX) {
1927 return -1;
1928 }
1929 if (IC_MSG_HDR_LEN + dlen > buf->len) {
1930 errno = 0;
1931 /* Early checks to drop malformed messages before trying
1932 * to read again. */
1933 RETHROW(ic_check_msg_hdr(ic));
1934 to_read = IC_MSG_HDR_LEN + dlen - buf->len;
1935 break;
1936 }
1937
1938 starves = true;
1939 errno = 0;
1940 RETHROW(ic_check_msg_hdr_flags(ic, slot, flags));
1941 if (unlikely(flags & IC_MSG_HAS_FD)) {
1942 if (ic->fds.len < 1) {
1943 assert (ic->fd_overflow);
1944 return -1;
1945 } else {
1946 ic->current_fd = ic->fds.tab[0];
1947 qv_pop(&ic->fds);
1948 }
1949 if (unlikely(ic->current_fd < 0)) {
1950 logger_warning(&_G.logger, "file descriptor table overflow");
1951 }
1952 flags &= ~IC_MSG_HAS_FD;
1953 }
1954
1955 if (unlikely(cmd == IC_MSG_STREAM_CONTROL)) {
1956 ic->is_closing |= slot == IC_SC_BYE;
1957 } else
1958 if (cmd <= 0) {
1959 RETHROW(ic_read_process_answer(ic, cmd, slot, data, dlen, NULL));
1960 } else {
1961 /* deal with queries */
1962 ic_update_pending(ic, slot);
1963
1964 if (unlikely(ic->is_closing)) {
1965 if (slot) {
1966 ic_reply_err(ic, MAKE64(ic->id, slot | flags),
1967 IC_MSG_RETRY);
1968 }
1969 } else {
1970 if (ic_read_process_query(ic, cmd, slot, flags, data, dlen,
1971 NULL) < 0)
1972 {
1973 errno = 0;
1974 return -1;
1975 }
1976 }
1977 }
1978
1979 if (unlikely(ic->current_fd >= 0)) {
1980 close(ic->current_fd);
1981 ic->current_fd = -1;
1982 }
1983
1984 if (unlikely(!ic->elh)) {
1985 assert (buf->len == 0);
1986 /* a problem occured and the ichannel has been closed */
1987 errno = 0;
1988 return -1;
1989 }
1990 sb_skip(buf, IC_MSG_HDR_LEN + dlen);

1992

2001 * a potential pending IC_BYE on the channel.
2002 */
2003 write_errno = errno ?: EINVAL;
2004 }
2005 }
2006 }
2007 }
2008
2009 if (buf->len < IC_MSG_HDR_LEN) {
2010 to_read = IC_MSG_HDR_LEN;
2011 }
2012
2013 if (ic->is_seqpacket && seqpkt_at_least > 0) {
2014 goto again;
2015 }
2016 if (!ic->is_seqpacket && !starves) {
2017 goto again;
2018 }
2019 if (write_errno) {
2020 errno = write_errno;
2021 return -1;
2022 }
2023 return 0;

2187static void ic_queue(ichannel_t *ic, ic_msg_t *msg, uint32_t flags)
2188{
2189 char *buffer = msg->data;
2190
2191 assert (!ic->is_wiped);
2192 assert (msg->dlen >= IC_MSG_HDR_LEN && msg->data);
2193
2194 flags |= msg->slot;
2195 ic_msg_update_flags(msg, &flags);
2196 put_unaligned_le32(buffer, flags);
2197 put_unaligned_le32(buffer + IC_MSG_CMD_OFFSET, msg->cmd);
2198 put_unaligned_le32(buffer + IC_MSG_DLEN_OFFSET,
2199 msg->dlen - IC_MSG_HDR_LEN);
2200 if (htlist_is_empty(&ic->msg_list) && ic->elh) {
2201 el_fd_set_mask(ic->elh, POLLINOUT);

2203
2204 ic_msg_trace(msg, "queueing message on ic %p with cmd %x", ic, msg->cmd);
2205
2206 if (htlist_is_empty(&ic->msg_list)
2207 || msg->priority <= htlist_last_entry(&ic->msg_list, ic_msg_t,
2208 msg_link)->priority)
2209 {
2210 htlist_add_tail(&ic->msg_list, &msg->msg_link);
2211 } else
2212 if (msg->priority > htlist_first_entry(&ic->msg_list, ic_msg_t,
2213 msg_link)->priority) {
2214 htlist_add(&ic->msg_list, &msg->msg_link);
2215 } else
2216 if (msg->priority == EV_PRIORITY_NORMAL && ic->last_normal_prio_msg) {
2217 htlist_add_after(&ic->msg_list, ic->last_normal_prio_msg,
2218 &msg->msg_link);
2219 } else {
2220 ic_msg_t *curr;
2221 ic_msg_t *prev = NULL;
2222
2223 htlist_for_each_entry(curr, &ic->msg_list, msg_link) {
2224 if (curr->priority != EV_PRIORITY_HIGH) {
2225 break;
2226 }
2227 prev = curr;
2228 }
2229 htlist_add_after(&ic->msg_list, &prev->msg_link,
2230 &msg->msg_link);
2231 }
2232
2233 if (msg->priority == EV_PRIORITY_NORMAL) {
2234 ic->last_normal_prio_msg = &msg->msg_link;
2235 }
2236}
