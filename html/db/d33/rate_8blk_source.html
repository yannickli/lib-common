22static void net_rctl_cron_cb(el_t ev, data_t priv)
23{
24 net_rctl_t *rctl = priv.ptr;
25 int slot;
26
27 if ((slot = ++rctl->slot) == countof(rctl->rates))
28 slot = rctl->slot = 0;
29 rctl->remains = MIN(rctl->remains + rctl->rates[slot], rctl->slice_max);
30 if (rctl->remains) {
31 if (rctl->is_blk) {
32 block_t blk = rctl->blk;
33
34 blk();
35 } else {
36 rctl->on_ready(rctl);
37 }
38 }
39}

41static void net_rctl_init_(net_rctl_t *rctl, int rate)
42{
43 unsigned sum = 0;
44
45 p_clear(rctl, 1);
46 rctl->rate = rate;
47 rctl->slice_max = DIV_ROUND_UP(rctl->rate * 3, countof(rctl->rates));
48 for (int i = 0; i < countof(rctl->rates); i++) {
49 sum += (rctl->rates[i] = (i + 1) * rctl->rate / countof(rctl->rates) - sum);
50 }
51}

53void net_rctl_init(net_rctl_t *rctl, int rate, void (*cb)(net_rctl_t *))
54{
