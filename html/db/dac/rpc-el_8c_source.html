133/** El thread loop. */
134static void *el_loop_thread_fun(void *arg)
135{
136 static struct sigaction sa_old_term;
137 static struct sigaction sa_old_quit;
138 static bool sig_handlers_saved;
139 el_t el_sigterm;
140 el_t el_sigquit;
141
142 pthread_mutex_lock(&_G.el_mutex);
143
144 /* XXX: sa_old_* only written once, no rewritten at fork,
145 * sa_handler at NULL cannot be used to know if the action has been
146 * rewritten because it stands for default action for the signal (SIG_DFL)
147 */
148 if (!sig_handlers_saved) {
149 sigaction(SIGTERM, NULL, &sa_old_term);
150 sigaction(SIGQUIT, NULL, &sa_old_quit);
151 sig_handlers_saved = true;
152 }
153
154 el_sigterm = el_signal_register(SIGTERM, &el_loop_thread_on_term, NULL);
155 el_sigquit = el_signal_register(SIGQUIT, &el_loop_thread_on_term, NULL);
156
157 _G.el_thr_status = EL_THR_STARTED;
158 pthread_cond_broadcast(&_G.el_thr_start_cond);
159
160 assert (atomic_load(&_G.el_mutex_wait_lock_cnt) > 0);
161 el_loop_wait_after_lock_cond();
162
163 for (;;) {
164 int wait_lock_cnt = atomic_load(&_G.el_mutex_wait_lock_cnt);
165
166 /* If we request a mutex lock, we need to process the events without
167 * waiting. */
168 assert (wait_lock_cnt >= 0);
169 el_loop_timeout(wait_lock_cnt > 0 ? 0 : 100);
170
171 ic_el_server_el_process();
172 ic_el_client_el_process();
173
174 if (_G.el_thr_signo) {
175 break;
176 }
177
178 if (el_has_pending_events()
179 || (wait_lock_cnt = atomic_load(&_G.el_mutex_wait_lock_cnt)) == 0)
180 {
181 pthread_mutex_unlock(&_G.el_mutex);
182 sched_yield();
183 pthread_mutex_lock(&_G.el_mutex);
184 } else {
185 assert (wait_lock_cnt > 0);
186 el_loop_wait_after_lock_cond();
187 }
188 }
189
190 el_unregister(&el_sigterm);
191 el_unregister(&el_sigquit);
192
193 sigaction(SIGTERM, &sa_old_term, NULL);
194 sigaction(SIGQUIT, &sa_old_quit, NULL);
195
196 if (_G.el_thr_signo != SIGINT) {
197 pthread_kill(_G.main_thread, _G.el_thr_signo);
198 pthread_detach(pthread_self());
199 }
200
201 pthread_mutex_unlock(&_G.el_mutex);
202
203 return NULL;
204}

206/** Increment wait lock counter. */
207static int ic_el_inc_el_mutex_wait_lock_cnt(void)
208{
209 int old_wait_lock_cnt;
210
211 old_wait_lock_cnt = atomic_fetch_add(&_G.el_mutex_wait_lock_cnt, 1);
212 assert (old_wait_lock_cnt >= 0);

216/** Decrement wait lock counter.
217 *
218 * It will signal _G.el_mutex_after_lock_cond if needed.
219 */
220static void ic_el_dec_el_mutex_wait_lock_cnt(void)
221{
222 int old_wait_lock_cnt;
223
224 old_wait_lock_cnt = atomic_fetch_sub(&_G.el_mutex_wait_lock_cnt, 1);
225 assert (old_wait_lock_cnt >= 1);
226 if (old_wait_lock_cnt == 1) {
227 pthread_cond_signal(&_G.el_mutex_after_lock_cond);
228 }
229}

231/** Request el mutex lock from el loop.
232 *
233 * It will start the el thread loop if needed.
234 * el_mutex will be locked after calling this function.
235 */
236static void ic_el_mutex_lock(bool start_thr)
237{
238 int old_wait_lock_cnt;
239
240 old_wait_lock_cnt = ic_el_inc_el_mutex_wait_lock_cnt();
241 if (old_wait_lock_cnt == 0) {
242 el_wake_fire(_G.el_mutex_before_lock_el);
243 }
244
245 pthread_mutex_lock(&_G.el_mutex);
246
247 if (!start_thr
248 || likely(_G.el_thr_status == EL_THR_STARTED)
249 || unlikely(is_el_thr_stopped()))
250 {
251 ic_el_dec_el_mutex_wait_lock_cnt();
252 return;
253 }
254
255 if (_G.el_thr_status == EL_THR_NOT_STARTED) {
256 _G.el_thr_status = EL_THR_STARTING;
257
258 if (thr_create(&_G.el_thread, NULL, &el_loop_thread_fun, NULL) < 0) {
259 e_fatal("cannot launch el_loop thread");
260 }
261 }
262
263 do {
264 pthread_cond_wait(&_G.el_thr_start_cond, &_G.el_mutex);
265 } while (_G.el_thr_status != EL_THR_STARTED);
266
267 ic_el_dec_el_mutex_wait_lock_cnt();
268}

295/** Result of wait_thread_cond(). */
296typedef enum wait_thread_cond_res_t {
297 WAIT_THR_COND_OK = 0,
298 WAIT_THR_COND_TIMEOUT = -1,
299 WAIT_THR_COND_SIGINT = -2,
300} wait_thread_cond_res_t;
301
302/** Wait thread for specified condition.
303 *
304 * XXX: _G.el_mutex must be locked before calling this function.
305 * ic_el_dec_el_mutex_wait_lock_cnt() will be called if the condition has
306 * been fulfilled.
307 * So, you *MUST* call ic_el_inc_el_mutex_wait_lock_cnt() when signaling
308 * the condition.
309 * You also *MUST* use pthread_cond_broadcast() on _G.el_wait_thr_cond
310 * since multiple threads can wait to _G.el_wait_thr_cond
311 *
312 * Example:
313 * ctx->is_terminated = true;
314 * pthread_cond_broadcast(&_G.el_wait_thr_cond);
315 * ic_el_inc_el_mutex_wait_lock_cnt();
316 *
317 * \param[in] is_terminated A callback called to check if the condition has
318 * been fulfilled.
319 * \param[in] terminated_arg The argument passed to the callback.
320 * \param[in] timeout The timeout in seconds.
321 * <= 0 for unlimited timeout.
322 * \return The result as wait_thread_cond_res_t.
323 */
324static wait_thread_cond_res_t
325wait_thread_cond(bool (*is_terminated)(void *), void *terminated_arg,
326 double timeout)
327{
328 struct timeval begin_time;
329 int64_t timeout_msec = 0;
330 bool timeout_expired = false;
331 el_t timeout_el = NULL;
332 bool terminated;
333 wait_thread_cond_res_t res = WAIT_THR_COND_TIMEOUT;
334
335 if (timeout > 0.0) {
336 lp_gettv(&begin_time);
337 timeout_msec = (int64_t)(timeout * 1000.0);
338 timeout_el = el_timer_register(timeout_msec, 0, EL_TIMER_LOWRES,
339 &ic_el_wait_thr_timeout_cb,
340 &timeout_expired);
341 }
342
343 /* Save old sigint handler if we don't saved it already. */
344 assert (_G.el_wait_thr_sigint_cnt >= 0);
345 if (_G.el_wait_thr_sigint_cnt++ == 0) {
346 sigaction(SIGINT, NULL, &_G.el_wait_thr_py_sigint);
347 _G.el_wait_thr_sigint_el =
348 el_signal_register(SIGINT, &ic_el_wait_thr_cond_sigint, NULL);
349 _G.el_wait_thr_sigint_received = false;
350 }
351
352 while (!(terminated = is_terminated(terminated_arg)) &&
353 !timeout_expired && !_G.el_wait_thr_sigint_received &&
354 !is_el_thr_stopped() &&
355 !pthread_cond_wait(&_G.el_wait_thr_cond, &_G.el_mutex))
356 {
357 }
358
359 if (timeout_expired) {
360 struct timeval after_time;
361 int64_t diff_msec;
362
363 lp_gettv(&after_time);
364 diff_msec = timeval_diffmsec(&after_time, &begin_time);
365 if (unlikely((timeout_msec + 500) < diff_msec)) {
366 e_warning("thread starvation detected, expected timeout in "
367 "%jdms, took %jdms", timeout_msec, diff_msec);
368 }
369 } else {
370 el_unregister(&timeout_el);
371 }
372
373 if (terminated) {
374 ic_el_dec_el_mutex_wait_lock_cnt();
375 res = WAIT_THR_COND_OK;
376 }
377
378 if (_G.el_wait_thr_sigint_received || is_el_thr_stopped()) {
379 res = WAIT_THR_COND_SIGINT;
380 }
381
382 /* Restore old sigint handler if we are the last one to wait. */
383 assert (_G.el_wait_thr_sigint_cnt > 0);
384 if (--_G.el_wait_thr_sigint_cnt == 0) {
385 el_unregister(&_G.el_wait_thr_sigint_el);
386 sigaction(SIGINT, &_G.el_wait_thr_py_sigint, NULL);
387 }
388
389 return res;
390}

513static void ic_el_server_on_event(ichannel_t *ic, ic_event_t evt)
514{
515 ic_el_server_t *server = ic->priv;
516
517 if (unlikely(!server)) {
518 return;
519 }
520
521 switch (evt) {
522 case IC_EVT_CONNECTED:
523 if (server->cb_cfg.on_connect && !is_el_thr_stopped()) {
524 t_scope;
525 lstr_t server_uri = t_lstr_dup(server->uri);
526 lstr_t client_addr = t_lstr_dup(ic_get_client_addr(ic));
527 ic_el_server_t *server_dup = ic_el_server_retain(server);
528
529 ic_el_mutex_unlock();
530 (*server_dup->cb_cfg.on_connect)(server_dup, server_uri,
531 client_addr);
532 ic_el_mutex_lock(false);
533 ic_el_server_delete(&server_dup);
534 }
535 break;
536
537 case IC_EVT_DISCONNECTED:
538 if (server->cb_cfg.on_disconnect && !is_el_thr_stopped()) {
539 t_scope;
540 lstr_t server_uri = t_lstr_dup(server->uri);
541 lstr_t client_addr = t_lstr_dup(ic_get_client_addr(ic));
542 ic_el_server_t *server_dup = ic_el_server_retain(server);
543
544 ic_el_mutex_unlock();
545 (*server_dup->cb_cfg.on_disconnect)(server_dup, server_uri,
546 client_addr);
547 ic_el_mutex_lock(false);
548 ic_el_server_delete(&server_dup);
549 }
550 if (ic->peer) {
551 ((ic_el_peer_t *)ic->peer)->ic = NULL;
552 ic_el_peer_delete((ic_el_peer_t **)&ic->peer);
553 }
554 break;
555
556 default:
557 break;
558 }
559}

580static void ic_el_server_rpc_cb(ichannel_t *ic, uint64_t slot, void *arg,
581 const ic__hdr__t *hdr)
582{
583 t_scope;
584 ic_el_server_t *server = ic->priv;
585 ic_status_t status;
586 void *res = NULL;
587 const iop_struct_t *res_st = NULL;
588
589 if (!unlikely(server)) {
590 if (!ic_slot_is_async(slot)) {
591 ic_reply_err(ic, slot, IC_MSG_UNIMPLEMENTED);
592 }
593 return;
594 }
595
596 ic_el_server_retain(server);
597 ic_el_mutex_unlock();
598 status = (*server->cb_cfg.t_on_rpc)(server, ic, slot, arg, hdr, &res,
599 &res_st);
600 ic_el_mutex_lock(false);
601 ic_el_server_delete(&server);
602
603 switch (status) {
604 case IC_MSG_OK:
605 case IC_MSG_EXN:
606 if (!ic_slot_is_async(slot)) {
607 __ic_reply(ic, slot, status, -1, res_st, res);
608 }
609 break;
610
611 default:
612 if (!ic_slot_is_async(slot)) {
613 ic_reply_err(ic, slot, status);
614 }
615 break;
616 }
617}

667static ic_el_sync_res_t
668ic_el_server_wait_for_stop(ic_el_server_t *server, double timeout)
669{
670 ic_el_sync_res_t res = IC_EL_SYNC_OK;
671 wait_thread_cond_res_t wait_res;
672
673 server->wait_for_stop = true;
674
675 wait_res = wait_thread_cond(&ic_el_server_is_stopped, server, timeout);
676 switch (wait_res) {
677 case WAIT_THR_COND_OK:
678 case WAIT_THR_COND_TIMEOUT:
679 break;
680
681 case WAIT_THR_COND_SIGINT:
682 res = IC_EL_SYNC_SIGINT;
683 break;
684 }
685
686 server->wait_for_stop = false;
687 return res;
688}

862 dlist_splice(&disconnecting_clients, &_G.disconnecting_clients);

879/** Notify client connection or disconnection for async connections. */
880static void ic_el_client_notify_async_connections(ic_el_client_t *client)
881{
882 SB_1k(connect_err);
883 dlist_t async_ctxs;
884 sb_t *err_ptr = NULL;
885
886 /* Splice the client asynchronous contexts here to avoid race
887 * conditions when calling the callbacks with the mutex unlocked */
888 dlist_init(&async_ctxs);
889 dlist_splice(&async_ctxs, &client->async_connect_ctxs);
890
891 /* If empty, do nothing */
892 if (dlist_is_empty(&async_ctxs)) {
893 return;
894 }
895
896 /* If the client is not connected, the connection is in error */
897 if (!client->connected) {
898 ic_el_client_set_connection_error(client, &connect_err);
899 err_ptr = &connect_err;
900 }
901
902 /* Clear the aynchronous contexts first to avoid race conditions when
903 * calling the callbacks with the mutex unlocked */
904 dlist_for_each_entry(ic_el_client_async_connect_t, async_ctx,
905 &async_ctxs, link)
906 {
907 el_unregister(&async_ctx->timeout_el);
908 assert(async_ctx->client == client);
909 async_ctx->client = NULL;
910 }
911
912 /* First, delete the contexts and call all the callbacks that require the
913 * lock */
914 dlist_for_each_entry(ic_el_client_async_connect_t, async_ctx,
915 &async_ctxs, link)
916 {
917 if (async_ctx->is_locked_cb) {
918 ic_client_async_connect_f cb = async_ctx->cb;
919 void *cb_arg = async_ctx->cb_arg;
920
921 ic_el_client_async_connect_delete(&async_ctx);
922 (*cb)(err_ptr, cb_arg);
923 }
924 }
925
926 /* If there are no more contexts, no need to unlock/lock the mutex */
927 if (dlist_is_empty(&async_ctxs)) {
928 return;
929 }
930
931 /* Then, delete the contexts and call all the callbacks with the mutex
932 * unlocked */
933 ic_el_mutex_unlock();
934 dlist_for_each_entry(ic_el_client_async_connect_t, async_ctx,
935 &async_ctxs, link)
936 {
937 ic_client_async_connect_f cb = async_ctx->cb;
938 void *cb_arg = async_ctx->cb_arg;
939
940 assert(!async_ctx->is_locked_cb);
941 ic_el_client_async_connect_delete(&async_ctx);
942 (*cb)(err_ptr, cb_arg);
943 }
944 ic_el_mutex_lock(false);
945}

947/** Notify client connection or disconnection when in_connect. */
948static void ic_el_client_notify_in_connect(ic_el_client_t *client)
949{
950 /* Notify synchronous waiting connections */
951 if (client->wait_connect_sync > 0) {
952 pthread_cond_broadcast(&_G.el_wait_thr_cond);
953 ic_el_inc_el_mutex_wait_lock_cnt();
954 }
955
956 /* Notify asynchronous connections */
957 ic_el_client_notify_async_connections(client);
958}

960/** Called on event on the ic channel. */
961static void ic_el_client_on_event(ichannel_t *ic, ic_event_t evt)
962{
963 ic_el_client_t *client = container_of(ic, ic_el_client_t, ic);
964
965 switch (evt) {
966 case IC_EVT_CONNECTED: {
967 client->connected = true;
968 if (client->cb_cfg.on_connect) {
969 ic_el_mutex_unlock();
970 (*client->cb_cfg.on_connect)(client);
971 ic_el_mutex_lock(false);
972 }
973 } break;
974
975 case IC_EVT_DISCONNECTED: {
976 bool connected = client->connected;
977
978 client->connected = false;
979 if (client->cb_cfg.on_disconnect && !client->force_disconnect &&
980 !is_el_thr_stopped())
981 {
982 ic_el_mutex_unlock();
983 (*client->cb_cfg.on_disconnect)(client, connected);
984 ic_el_mutex_lock(false);
985 }
986 } break;
987
988 case IC_EVT_ACT:
989 case IC_EVT_NOACT:
990 return;
991 }
992
993 if (client->in_connect) {
994 if (!client->force_disconnect) {
995 ic_el_client_notify_in_connect(client);
996 }
997 client->in_connect = false;
998 }
999
1000 client->force_disconnect = false;
1001}

1085/** Synchronously connect the client and wait for connection.
1086 *
1087 * This function will return WAIT_THR_COND_TIMEOUT if the client cannot be
1088 * connected.
1089 */
1090static wait_thread_cond_res_t
1091ic_el_client_ic_sync_connect_wait(ic_el_client_t *client, int timeout)
1092{
1093 wait_thread_cond_res_t wait_res;
1094
1095 if (ic_el_client_start_connection(client) < 0) {
1096 return WAIT_THR_COND_TIMEOUT;
1097 }
1098
1099 assert(client->wait_connect_sync >= 0);
1100 client->wait_connect_sync++;
1101
1102 wait_res = wait_thread_cond(&ic_el_client_connect_is_terminated,
1103 client, timeout);
1104
1105 assert(client->wait_connect_sync > 0);
1106 client->wait_connect_sync--;
1107
1108 if (wait_res == WAIT_THR_COND_OK && !client->connected) {
1109 wait_res = WAIT_THR_COND_TIMEOUT;
1110 }
1111
1112 return wait_res;
1113}

1115/** Synchronously connect the client with the el mutex locked. */
1116static ic_el_sync_res_t
1117ic_el_client_sync_connect_locked(ic_el_client_t *client, double timeout,
1118 sb_t *err)
1119{
1120 wait_thread_cond_res_t wait_res;
1121
1122 if (client->connected) {
1123 return IC_EL_SYNC_OK;
1124 }
1125
1126 wait_res = ic_el_client_ic_sync_connect_wait(client, timeout);
1127 switch (wait_res) {
1128 case WAIT_THR_COND_OK:
1129 return IC_EL_SYNC_OK;
1130
1131 case WAIT_THR_COND_TIMEOUT:
1132 ic_el_client_set_connection_error(client, err);
1133 return IC_EL_SYNC_ERR;
1134
1135 case WAIT_THR_COND_SIGINT:
1136 return IC_EL_SYNC_SIGINT;
1137 }
1138
1139 assert (false);
1140 sb_setf(err, "unexpected connect status %d", wait_res);
1141 return IC_EL_SYNC_ERR;
1142}

1179/** Asynchronously connect the client with the el mutex locked. */
1180static ic_el_async_res_t ic_el_client_async_connect_locked(
1181 ic_el_client_t *client, double timeout, ic_client_async_connect_f cb,
1182 void *nullable cb_arg, bool is_locked_cb,
1183 ic_el_client_async_connect_t *nullable *nullable async_ctx_ptr, sb_t *err)
1184{
1185 ic_el_client_async_connect_t *async_ctx;
1186
1187 if (client->connected) {
1188 return IC_EL_ASYNC_OK;
1189 }
1190
1191 if (ic_el_client_start_connection(client) < 0) {
1192 ic_el_client_set_connection_error(client, err);
1193 return IC_EL_ASYNC_ERR;
1194 }
1195
1196 async_ctx = ic_el_client_async_connect_new();
1197 async_ctx->client = client;

1199 async_ctx->cb_arg = cb_arg;

1203 if (timeout > 0.0) {
1204 int64_t timeout_msec = (int64_t)(timeout * 1000.0);
1205
1206 async_ctx->timeout_el = el_timer_register(
1207 timeout_msec, 0, EL_TIMER_LOWRES,
1208 &ic_el_client_async_connect_timeout_cb, async_ctx);
1209 }
1210
1211 if (async_ctx_ptr) {
1212 *async_ctx_ptr = async_ctx;
1213 }
1214
1215 return IC_EL_ASYNC_PENDING;
1216}

1315 ic_status_t status, void *res,

1317{

1322

1324 goto end;
1325 }
1326

1328 goto end;
1329 }

1341 NULL);
1342 break;

1345 break;
1346 }

1349 pthread_cond_broadcast(&_G.el_wait_thr_cond);
1350 ic_el_inc_el_mutex_wait_lock_cnt();
1351
1352 end:

1367 int32_t cmd, const ic__hdr__t *hdr, double timeout,
1368 const void *arg, ic_status_t *status, void **res,
1369 sb_t *err)
1370{
1371 ic_el_sync_res_t call_res = IC_EL_SYNC_OK;
1372 ic_el_client_sync_query_ctx_t *query_ctx = NULL;
1373 ic_msg_t *msg;
1374 wait_thread_cond_res_t wait_res;
1375
1376 ic_el_mutex_lock(true);
