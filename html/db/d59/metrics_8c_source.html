35static uint32_t qv_lstr_hash(const qhash_t *qh, const qv_t(cstr) *qv)
36{
37 uint32_t res = 0;
38
39 tab_for_each_entry(str, qv) {
40 res ^= qhash_str_hash(qh, str);
41 }
42
43 return res;
44}

46static bool
47qv_lstr_equal(const qhash_t *qh,
48 const qv_t(cstr) *qv1,
49 const qv_t(cstr) *qv2)
50{
51 if (qv1->len != qv2->len) {
52 return false;
53 }
54 for (int i = 0; i < qv1->len; i++) {
55 if (!strequal(qv1->tab[i], qv2->tab[i])) {
56 return false;
57 }
58 }
59 return true;
60}

75static void prom_metric_wipe(prom_metric_t *self)
76{
77 /* Remove self from parent's children. */
78 if (self->parent) {
79 qm_del_key(prom_metric, self->parent->children_by_labels,
80 &self->label_values);
81 }
82
83 /* Wipe */
84 lstr_wipe(&self->name);
85 lstr_wipe(&self->documentation);
86
87#define cstr_wipe(str) p_delete((char **)str)
88 qv_deep_wipe(&self->label_names, cstr_wipe);
89 qv_deep_wipe(&self->label_values, cstr_wipe);
90#undef cstr_wipe
91
92 dlist_remove(&self->children_list);
93 dlist_remove(&self->siblings_list);
94
95 qm_deep_delete(prom_metric, &self->children_by_labels, IGNORE,
96 obj_delete);
97}

99int prom_metric_check_name(lstr_t name)
100{
101 pstream_t ps = ps_initlstr(&name);
102 ctype_desc_t desc;
103 int c;
104
105 THROW_ERR_IF(ps_done(&ps));
106
107 /* First character must be a letter or _ or : */
108 c = __ps_getc(&ps);
109 THROW_ERR_IF(!isalpha(c) && c != '_' && c != ':');
110
111 /* Other characters must be alphanumeric or _ or : */
112 ctype_desc_build(&desc, "_:");
113 ctype_desc_combine(&desc, &desc, &ctype_isalnum);
114 ps_skip_span(&ps, &desc);
115 THROW_ERR_IF(!ps_done(&ps));
116
117 return 0;
118}
119
120int prom_metric_check_label_name(const char *name)
121{
122 pstream_t ps = ps_initstr(name);
123 ctype_desc_t desc;
124 int c;
125
126 THROW_ERR_IF(ps_done(&ps));
127
128 /* Label names beginning with __ are reserved for internal use */
129 THROW_ERR_IF(ps_startswithstr(&ps, "__"));
130
131 /* First character must be a letter or _ */
132 c = __ps_getc(&ps);
133 THROW_ERR_IF(!isalpha(c) && c != '_');
134
135 /* Other characters must be alphanumeric or _ */
136 ctype_desc_build(&desc, "_");
137 ctype_desc_combine(&desc, &desc, &ctype_isalnum);
138 ps_skip_span(&ps, &desc);
139 THROW_ERR_IF(!ps_done(&ps));
140
141 return 0;
142}

146 const char *fmt, ...)
147{
148 SB_1k(err);

150
151 sb_addf(&err, "invalid call to %s() method of metric `%*pM`: ",
152 func, LSTR_FMT_ARG(self->name));

156 va_end(args);
157

161static void prom_metric_register(prom_metric_t *self)
162{
163 /* Consistency checks */
164 if (self->parent || self->label_values.len) {
165 prom_metric_panic(self, "do_register",
166 "only parent metrics can be registered");
167 }
168 if (!dlist_is_empty(&self->siblings_list)) {
169 prom_metric_panic(self, "do_register",
170 "metric is already registered");
171 }
172 if (!dlist_is_empty(&self->children_list) || self->children_by_labels) {
173 prom_metric_panic(self, "do_register", "metric already has children");
174 }
175 if (prom_metric_check_name(self->name) < 0) {
176 prom_metric_panic(self, "do_register", "invalid metric name `%*pM`",
177 LSTR_FMT_ARG(self->name));
178 }
179 tab_for_each_entry(label_name, &self->label_names) {
180 if (prom_metric_check_label_name(label_name)) {
181 prom_metric_panic(self, "do_register",
182 "invalid label name `%s`",
183 label_name);
184 }
185 }
186 if (!self->documentation.len) {
187 prom_metric_panic(self, "do_register", "metric has no description");
188 }
189
190 /* Register */
191 dlist_add_tail(&prom_collector_g, &self->siblings_list);
192 if (self->label_names.len) {
193 self->children_by_labels = qm_new(prom_metric, 0);
194 }
195}

197static void prom_metric_unregister(prom_metric_t *self)
198{
199 /* Consistency checks */
200 if (self->parent || self->label_values.len) {
201 prom_metric_panic(self, "unregister",
202 "only parent metrics can be unregistered");
203 }
204
205 /* Unregister */
206 dlist_remove(&self->siblings_list);
207}

209static void prom_metric_check_labels(prom_metric_t *self, const char *func,
210 const qv_t(cstr) *label_values)
211{
212 if (!self->label_names.len) {
213 prom_metric_panic(self, func, "no label names defined in metric");
214 }
215 if (label_values->len != self->label_names.len) {
216 prom_metric_panic(self, func, "incorrect labels count (%d != %d)",
217 label_values->len, self->label_names.len);
218 }
219 if (!self->children_by_labels) {
220 prom_metric_panic(self, func, "no children_by_labels defined");
221 }
222}

224static prom_metric_t *
225(prom_metric_labels)(prom_metric_t *self, const qv_t(cstr) *label_values)
226{
227 uint32_t pos;
228 prom_metric_t *child;
229
230 /* Consistency checks */
231 prom_metric_check_labels(self, "labels", label_values);
232
233 spin_lock(&self->lock);
234
235 /* Lookup */
236 pos = qm_reserve(prom_metric, self->children_by_labels, label_values, 0);
237 if (pos & QHASH_COLLISION) {
238 child = self->children_by_labels->values[pos ^ QHASH_COLLISION];
239 spin_unlock(&self->lock);
240 return child;
241 }
242
243 /* Create child */
244 child = obj_new_of_class(prom_metric, self->v.as_obj_cls);
245 child->parent = self;
246 dlist_add_tail(&self->children_list, &child->siblings_list);
247
248 qv_grow(&child->label_values, label_values->len);
249 tab_for_each_entry(label, label_values) {
250 qv_append(&child->label_values, p_strdup(label));
251 }
252
253 self->children_by_labels->keys[pos] = &child->label_values;
254 self->children_by_labels->values[pos] = child;
255
256 spin_unlock(&self->lock);
257
258 return child;
259}

261static void
262(prom_metric_remove)(prom_metric_t *self, const qv_t(cstr) *label_values)
263{
264 int pos;
265
266 /* Consistency checks */
267 prom_metric_check_labels(self, "labels", label_values);
268
269 /* Remove child */
270 pos = qm_del_key(prom_metric, self->children_by_labels, label_values);
271 if (pos >= 0) {
272 obj_delete(&self->children_by_labels->values[pos]);
273 }
274}

303prom_metric_t *prom_metric_new(const object_class_t *cls,
304 const char *name, const char *documentation,
305 const char **labels, int nb_labels)
306{
307 prom_metric_t *res;
308
309 /* Create metric */
310 res = obj_new_of_class(prom_metric, cls);
311 res->name = lstr_dups(name, -1);
312 res->documentation = lstr_dups(documentation, -1);
313
314 qv_growlen(&res->label_names, nb_labels);
315 for (int i = 0; i < nb_labels; i++) {
316 res->label_names.tab[i] = p_strdup(labels[i]);
317 }
318
319 /* Register it */
320 obj_vcall(res, do_register);
321
322 return res;
323}

441static void (prom_histogram_set_buckets)(prom_histogram_t *self,
442 const qv_t(double) *upper_bounds)
443{
444 double prev_bound = -INFINITY;
445
446 /* Consistency checks */
447 if (self->parent) {
448 prom_metric_panic(obj_vcast(prom_metric, self), "set_buckets",
449 "buckets can only be set on parent histogram");
450 }
451 if (self->nb_buckets || self->bucket_upper_bounds) {
452 prom_metric_panic(obj_vcast(prom_metric, self), "set_buckets",
453 "buckets are already set");
454 }
455 if (!upper_bounds->len) {
456 prom_metric_panic(obj_vcast(prom_metric, self), "set_buckets",
457 "upper_bounds table is empty");
458 }
459 tab_for_each_entry(bound, upper_bounds) {
460 if (!isfinite(bound)) {
461 prom_metric_panic(obj_vcast(prom_metric, self), "set_buckets",
462 "upper bounds must be finite");
463 }
464 if (bound <= prev_bound) {
465 prom_metric_panic(obj_vcast(prom_metric, self), "set_buckets",
466 "upper_bounds must be sorted");
467 }
468 prev_bound = bound;
469 }
470
471 /* Create buckets */
472 self->nb_buckets = upper_bounds->len;
473 self->bucket_upper_bounds = p_new_raw(double, upper_bounds->len);
474 p_copy(self->bucket_upper_bounds, upper_bounds->tab, upper_bounds->len);
475 if (is_metric_observable(obj_ccast(prom_metric, self))) {
476 self->bucket_counts = p_new(double, upper_bounds->len);
477 }
478}

480void prom_histogram_set_linear_buckets(prom_histogram_t *self,
481 double start, double width, int count)
482{
483 t_scope;
484 qv_t(double) upper_bounds;
485
486 /* Consistency checks */
487 if (!isfinite(start) || !isfinite(width)) {
488 prom_metric_panic(obj_vcast(prom_metric, self), "set_linear_buckets",
489 "start and width must be finite numbers");
490 }
491 if (width <= 0 || count <= 0) {
492 prom_metric_panic(obj_vcast(prom_metric, self), "set_linear_buckets",
493 "width and count must be strictly positive");
494 }
495
496 /* Build upper bounds vector */
497 t_qv_init(&upper_bounds, count);
498 for (int i = 0; i < count; i++) {
499 qv_append(&upper_bounds, start);
500 start += width;
501 }
502
503 /* Set the buckets in the histogram */
504 obj_vcall(self, set_buckets, &upper_bounds);
505}

507void prom_histogram_set_exponential_buckets(prom_histogram_t *self,
508 double start, double factor,
509 int count)
510{
511 t_scope;
512 qv_t(double) upper_bounds;
513
514 /* Consistency checks */
515 if (!isfinite(start) || !isfinite(factor)) {
516 prom_metric_panic(obj_vcast(prom_metric, self),
517 "set_exponential_buckets",
518 "start and factor must be finite numbers");
519 }
520 if (start <= 0 || count <= 0) {
521 prom_metric_panic(obj_vcast(prom_metric, self),
522 "set_exponential_buckets",
523 "start and count must be strictly positive");
524 }
525 if (factor <= 1) {
526 prom_metric_panic(obj_vcast(prom_metric, self),
527 "set_exponential_buckets",
528 "factor must be strictly greater than 1");
529 }
530
531 /* Build upper bounds vector */
532 t_qv_init(&upper_bounds, count);
533 for (int i = 0; i < count; i++) {
534 qv_append(&upper_bounds, start);
535 start *= factor;
536 }
537
538 /* Set the buckets in the histogram */
539 obj_vcall(self, set_buckets, &upper_bounds);
540}

542static prom_histogram_t *
543(prom_histogram_labels)(prom_histogram_t *self,
544 const qv_t(cstr) *label_values)
545{
546 prom_metric_t *child_metric;
547 prom_histogram_t *child;
548
549 /* Get the child metric */
550 child_metric = super_call(prom_histogram, self, labels, label_values);
551 child = obj_vcast(prom_histogram, child_metric);
552
553 /* Create the buckets counts if the child was just created */
554 if (!child->bucket_counts) {
555 child->nb_buckets = self->nb_buckets;
556 child->bucket_counts = p_new(double, self->nb_buckets);
557 }
558
559 return child;

562static void prom_histogram_observe(prom_histogram_t *self, double value)
563{
564 prom_histogram_t *parent = self->parent ?: self;
565
566 if (!is_metric_observable(obj_ccast(prom_metric, self))) {
567 prom_metric_panic(obj_vcast(prom_metric, self), "observe",
568 "histogram is not observable");
569 }
570 if (!self->nb_buckets) {
571 prom_metric_panic(obj_vcast(prom_metric, self), "observe",
572 "histogram buckets were not initialized");
573 }
574
575 spin_lock(&self->lock);
576
577 self->count++;
578 self->sum += value;
579
580 for (int i = self->nb_buckets; i-- > 0;) {
581 if (value > parent->bucket_upper_bounds[i]) {
582 break;
583 }
584 self->bucket_counts[i]++;
585 }
586
587 spin_unlock(&self->lock);
588}

625static void bridge_simple_value(prom_simple_value_metric_t *metric, sb_t *out)
626{
627 lstr_t name = metric->parent ? metric->parent->name : metric->name;
628
629 sb_add_lstr(out, name);
630
631 if (metric->label_values.len) {
632 sb_addc(out, '{');
633 for (int i = 0; i < metric->label_values.len; i++) {
634 const char *label_name = metric->parent->label_names.tab[i];
635 const char *label_value = metric->label_values.tab[i];
636
637 if (i > 0) {
638 sb_addc(out, ',');
639 }
640 sb_addf(out, "%s=\"", label_name);
641 sb_adds_slashes(out, label_value, "\\\n", "\\n");
642 sb_addc(out, '"');
643 }
644 sb_addc(out, '}');
645 }
646
647 sb_addf(out, " %g\n", metric->value);
648}

650static void bridge_histogram(prom_histogram_t *metric, sb_t *out)
651{
652 SB_1k(label_names);
653 prom_histogram_t *parent = metric->parent ?: metric;
654
655 /* Build a buffer containing label names */
656 for (int i = 0; i < metric->label_values.len; i++) {
657 const char *label_name = metric->parent->label_names.tab[i];
658 const char *label_value = metric->label_values.tab[i];
659
660 if (i > 0) {
661 sb_addc(&label_names, ',');
662 }
663 sb_addf(&label_names, "%s=\"", label_name);
664 sb_adds_slashes(&label_names, label_value, "\\\n", "\\n");
665 sb_addc(&label_names, '"');
666 }
667
668 /* Add the line for each bucket */
669 for (int i = 0; i < metric->nb_buckets; i++) {
670 sb_add_lstr(out, parent->name);
671 sb_addf(out, "_bucket{%*pM", SB_FMT_ARG(&label_names));
672 if (metric->label_values.len) {
673 sb_addc(out, ',');
674 }
675 sb_addf(out, "le=\"%g\"} %g\n",
676 parent->bucket_upper_bounds[i],
677 metric->bucket_counts[i]);
678 }
679
680 /* Add the line for the "+Inf" bucket */
681 sb_add_lstr(out, parent->name);
682 sb_addf(out, "_bucket{%*pM", SB_FMT_ARG(&label_names));
683 if (metric->label_values.len) {
684 sb_addc(out, ',');
685 }
686 sb_addf(out, "le=\"+Inf\"} %g\n", metric->count);
687
688 /* Add the sum line */
689 sb_add_lstr(out, parent->name);
690 sb_adds(out, "_sum");
691 if (metric->label_values.len) {
692 sb_addf(out, "{%*pM}", SB_FMT_ARG(&label_names));
693 }
694 sb_addf(out, " %g\n", metric->sum);
695
696 /* Add the count line */
697 sb_add_lstr(out, parent->name);
698 sb_adds(out, "_count");
699 if (metric->label_values.len) {
700 sb_addf(out, "{%*pM}", SB_FMT_ARG(&label_names));
701 }
702 sb_addf(out, " %g\n", metric->count);
703
704 sb_addc(out, '\n');
705}

719static void prom_collector_bridge_metric(prom_metric_t *metric, sb_t *out)
720{
721 lstr_t metric_type = LSTR_NULL_V;
722
723 /* Skip metrics without samples */
724 if (metric->label_names.len
725 && !qm_len(prom_metric, metric->children_by_labels))
726 {
727 return;
728 }
729
730 /* Ensure there is an empty line between each metric */
731 if (out->len >= 2
732 && (out->data[out->len - 1] != '\n'
733 || out->data[out->len - 2] != '\n'))
734 {
735 sb_adds(out, "\n");
736 }
737
738 /* Add HELP and TYPE */
739 sb_addf(out, "# HELP %*pM ", LSTR_FMT_ARG(metric->name));
740 sb_add_slashes(out,
741 metric->documentation.s, metric->documentation.len,
742 "\\\n", "\\n");
743 sb_addc(out, '\n');
744
745 /* Add TYPE */
746 if (obj_is_a(metric, prom_counter)) {
747 metric_type = LSTR("counter");
748 } else
749 if (obj_is_a(metric, prom_gauge)) {
750 metric_type = LSTR("gauge");
751 } else
752 if (obj_is_a(metric, prom_histogram)) {
753 metric_type = LSTR("histogram");
754 } else {
755 assert (false);
756 }
757 sb_addf(out, "# TYPE %*pM %*pM\n",
758 LSTR_FMT_ARG(metric->name),
759 LSTR_FMT_ARG(metric_type));
760
761 /* Add values */
762 if (is_metric_observable(metric)) {
763 bridge_sample(metric, out);
764 } else {
765 dlist_for_each_entry(prom_metric_t, child, &metric->children_list,
766 siblings_list)
767 {
768 spin_lock(&child->lock);
769 bridge_sample(child, out);
770 spin_unlock(&child->lock);
771 }
772 }
773}

775void prom_collector_bridge(const dlist_t *collector, sb_t *out)
776{
777 dlist_for_each_entry(prom_metric_t, metric, collector, siblings_list) {
778 spin_lock(&metric->lock);
779 prom_collector_bridge_metric(metric, out);
780 spin_unlock(&metric->lock);
781 }
782
783 if (out->len && out->data[out->len - 1] == '\n') {
784 sb_shrink(out, 1);
785 }
786}
