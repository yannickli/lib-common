27static iopc_path_t *parse_path(lstr_t name, bool is_type, sb_t *err)
28{
29 pstream_t ps = ps_initlstr(&name);
30 iopc_path_t *path = iopc_path_new();
31
32 while (!ps_done(&ps)) {
33 pstream_t bit;
34
35 if (ps_get_ps_chr_and_skip(&ps, '.', &bit) < 0) {
36 bit = ps;
37 __ps_skip_upto(&ps, ps.s_end);
38 } else
39 if (ps_done(&ps)) {
40 sb_sets(err, "trailing dot in package name");
41 goto error;
42 }
43
44 if (ps_done(&bit)) {
45 sb_setf(err, "empty package or sub-package name");
46 goto error;
47 }
48 qv_append(&path->bits, p_dupz(bit.s, ps_len(&bit)));
49 }
50
51 return path;
52
53 error:
54 iopc_path_delete(&path);
55 return NULL;
56}

58int iop_type_to_iop(iop_type_t type, iop__type__t *out)
59{
60 switch (type) {
61 case IOP_T_I8:
62 case IOP_T_I16:
63 case IOP_T_I32:
64 case IOP_T_I64:
65 case IOP_T_U8:
66 case IOP_T_U16:
67 case IOP_T_U32:
68 case IOP_T_U64:
69 *out = IOP_UNION_VA(iop__type, i,
70 .is_signed = iop_int_type_is_signed(type),
71 .size = iopsq_int_type_to_int_size(type));
72 break;
73
74 case IOP_T_BOOL:
75 *out = IOP_UNION_VOID(iop__type, b);
76 break;
77
78 case IOP_T_DOUBLE:
79 *out = IOP_UNION_VOID(iop__type, d);
80 break;
81
82 case IOP_T_STRING:
83 *out = IOP_UNION(iop__type, s, STRING_TYPE_STRING);
84 break;
85
86 case IOP_T_DATA:
87 *out = IOP_UNION(iop__type, s, STRING_TYPE_BYTES);
88 break;
89
90 case IOP_T_XML:
91 *out = IOP_UNION(iop__type, s, STRING_TYPE_XML);
92 break;
93
94 case IOP_T_VOID:
95 *out = IOP_UNION_VOID(iop__type, v);
96 break;
97
98 case IOP_T_ENUM:
99 case IOP_T_UNION:
100 case IOP_T_STRUCT:
101 return -1;
102 }
103
104 return 0;
105}

138/** Fill an iopsq type from an iop_full_type_t. */
139static int iopsq_fill_type(const iop_full_type_t *ftype, iop__type__t *type)
140{
141 lstr_t typename;
142 const iop_obj_t *obj;
143
144 if (iop_type_to_iop(ftype->type, type) >= 0) {
145 return 0;
146 }
147
148 if (ftype->type == IOP_T_ENUM) {
149 typename = ftype->en->fullname;
150 } else {
151 assert (!iop_type_is_scalar(ftype->type));
152 typename = ftype->st->fullname;
153 }
154
155 if ((obj = iop_get_obj(typename))) {
156 switch (obj->type) {
157 case IOP_OBJ_TYPE_PKG:
158 break;
159
160 case IOP_OBJ_TYPE_ENUM:
161 if (ftype->type == IOP_T_ENUM && obj->desc.en == ftype->en) {
162 /* The enumeration is registered in the environment so it can
163 * be refered to with a type name. */
164 *type = IOP_UNION(iop__type, type_name, typename);
165 return 0;
166 }
167 break;
168
169 case IOP_OBJ_TYPE_ST:
170 if (ftype->type != IOP_T_ENUM && obj->desc.st == ftype->st) {
171 /* The struct/union/class is registered in the environment so
172 * it can be refered to with a type name. */
173 *type = IOP_UNION(iop__type, type_name, typename);
174 return 0;
175 }
176 break;
177 }
178 }
179
180 return -1;
181}

183void iopsq_type_table_fill_type(iopsq_type_table_t *table,
184 const iop_full_type_t *ftype,
185 iop__type__t *type)
186{
187 if (iopsq_fill_type(ftype, type) < 0) {
188 uint32_t pos;
189
190 /* The type is unknown and has probably been built by the user.
191 * Register it in the table. */
192 pos = qm_put(iopsq_type_id, &table->map, ftype, table->types.len, 0);
193 if (pos & QHASH_COLLISION) {
194 pos &= ~QHASH_COLLISION;
195 } else {
196 qv_append(&table->types, *ftype);
197 }
198
199 *type = IOP_UNION(iop__type, type_id, table->map.values[pos]);
200 }
201}

213static iop_type_t iop_type_from_iop(const iop__type__t *iop_type)
214{
215 IOP_UNION_SWITCH(iop_type) {
216 IOP_UNION_CASE_P(iop__type, iop_type, i, i) {
217 switch (i->size) {
218#define CASE(_sz) \
219 case INT_SIZE_S##_sz: \
220 return i->is_signed ? IOP_T_I##_sz : IOP_T_U##_sz
221
222 CASE(8);
223 CASE(16);
224 CASE(32);
225 CASE(64);
226
227#undef CASE
228 }
229 }
230 IOP_UNION_CASE_V(iop__type, iop_type, b) {
231 return IOP_T_BOOL;
232 }
233 IOP_UNION_CASE_V(iop__type, iop_type, d) {
234 return IOP_T_DOUBLE;
235 }
236 IOP_UNION_CASE(iop__type, iop_type, s, s) {
237 switch (s) {
238 case STRING_TYPE_STRING:
239 return IOP_T_STRING;
240
241 case STRING_TYPE_BYTES:
242 return IOP_T_DATA;
243
244 case STRING_TYPE_XML:
245 return IOP_T_XML;
246 }
247 }
248 IOP_UNION_CASE_V(iop__type, iop_type, v) {
249 return IOP_T_VOID;
250 }
251 IOP_UNION_CASE_V(iop__type, iop_type, type_name) {
252 /* This case should be handled at higher level. */
253 e_panic("should not happen");
254 }
255 IOP_UNION_CASE_V(iop__type, iop_type, array) {
256 /* This case should be handled at higher level. */
257 e_panic("should not happen");
258 }
259 IOP_UNION_CASE_V(iop__type, iop_type, type_id) {
260 /* This case should be handled at higher level. */
261 e_panic("should not happen");
262 }
263 }
264
265 return 0;
266}

268static int
269iopc_field_set_typename(iopc_field_t *nonnull f, lstr_t typename,
270 sb_t *nonnull err)
271{
272 f->kind = iop_get_type(typename);
273
274 if (f->kind == IOP_T_STRUCT) {
275 if (lstr_contains(typename, LSTR("."))) {
276 /* TODO Could parse and check that the type name looks like a
277 * proper type name. */
278 const iop_obj_t *obj;
279
280 obj = iop_get_obj(typename);
281 if (obj) {
282 switch (obj->type) {
283 case IOP_OBJ_TYPE_PKG:
284 /* Not expected to happen if we properly check the name.
285 */
286 sb_sets(err, "is a package name");
287 return -1;
288
289 case IOP_OBJ_TYPE_ST:
290 f->external_st = obj->desc.st;
291 f->kind = obj->desc.st->is_union ? IOP_T_UNION
292 : IOP_T_STRUCT;
293 break;
294
295 case IOP_OBJ_TYPE_ENUM:
296 f->external_en = obj->desc.en;
297 f->kind = IOP_T_ENUM;
298 break;
299 }
300
301 f->has_external_type = true;
302 }
303 } else {
304 if (iopc_check_type_name(typename, err) < 0) {
305 sb_prepends(err, "invalid type name: ");
306 return -1;
307 }
308 }
309 }
310 f->type_name = p_dupz(typename.s, typename.len);
311 return 0;
312}

314static int
315iopc_field_set_type(iopc_field_t *nonnull f,
316 const iop__type__t *nonnull type,
317 const iopsq_type_table_t *nullable type_table,
318 sb_t *nonnull err)
319{
320 struct iopsq__type__t * const * array_type;
321
322 if ((array_type = IOP_UNION_GET(iop__type, type, array))) {
323 type = *array_type;
324
325 if (IOP_UNION_IS(iop__type, type, array)) {
326 sb_setf(err, "multi-dimension arrays are not supported");
327 return -1;
328 }
329
330 f->repeat = IOP_R_REPEATED;
331 }
332
333 IOP_UNION_SWITCH(type) {
334 IOP_UNION_CASE(iop__type, type, type_name, typename) {
335 if (iopc_field_set_typename(f, typename, err) < 0) {
336 return -1;
337 }
338 }
339
340 IOP_UNION_CASE(iop__type, type, type_id, type_id) {
341 const iop_full_type_t *ftype;
342
343 if (!type_table) {
344 sb_sets(err, "got type ID but no type table");
345 return -1;
346 }
347
348 ftype = iopsq_type_table_get_type(type_table, type_id);
349 f->kind = ftype->type;
350 if (ftype->type == IOP_T_ENUM) {
351 f->external_en = ftype->en;
352 f->has_external_type = true;
353 } else
354 if (!iop_type_is_scalar(ftype->type)) {
355 f->external_st = ftype->st;
356 f->has_external_type = true;
357 }
358 }
359
360 IOP_UNION_DEFAULT() {
361 f->kind = iop_type_from_iop(type);
362 }
363 }
364
365 RETHROW(iopc_check_field_type(f, err));
366
367 return 0;
368}

370static void
371iopc_field_set_defval(iopc_field_t *f, const iop__value__t *defval)
372{
373 IOP_UNION_SWITCH(defval) {
374 IOP_UNION_CASE(iop__value, defval, i, i) {
375 f->defval.u64 = i;
376 f->defval_is_signed = (i < 0);
377 f->defval_type = IOPC_DEFVAL_INTEGER;
378 }
379 IOP_UNION_CASE(iop__value, defval, u, u) {
380 f->defval.u64 = u;
381 f->defval_type = IOPC_DEFVAL_INTEGER;
382 }
383 IOP_UNION_CASE(iop__value, defval, d, d) {
384 f->defval.d = d;
385 f->defval_type = IOPC_DEFVAL_DOUBLE;
386 }
387 IOP_UNION_CASE(iop__value, defval, s, s) {
388 f->defval.ptr = p_dupz(s.s, s.len);
389 f->defval_type = IOPC_DEFVAL_STRING;
390 }
391 IOP_UNION_CASE(iop__value, defval, b, b) {
392 f->defval.u64 = b;
393 f->defval_type = IOPC_DEFVAL_INTEGER;
394 }
395 }

413static iopc_field_t *
414iopc_field_load(const iop__field__t *nonnull field_desc,
415 const qv_t(iopc_field) *fields,
416 const iopsq_type_table_t *nullable type_table,
417 sb_t *nonnull err)
418{
419 iopc_field_t *f = NULL;
420
421 if (iopc_check_field_name(field_desc->name, err) < 0) {
422 goto error;
423 }
424
425 f = iopc_field_new();
426 f->name = p_dupz(field_desc->name.s, field_desc->name.len);
427 f->field_pos = fields->len;
428
429 if (OPT_ISSET(field_desc->tag)) {
430 f->tag = OPT_VAL(field_desc->tag);
431 } else {
432 f->tag = fields->len ? (*tab_last(fields))->tag + 1 : 1;
433 }
434 if (iopc_check_tag_value(f->tag, err) < 0) {
435 goto error;
436 }
437 tab_for_each_entry(other_field, fields) {
438 if (strequal(other_field->name, f->name)) {
439 sb_sets(err, "name already used by another field");
440 goto error;
441 }
442 if (other_field->tag == f->tag) {
443 sb_setf(err, "tag `%d' is already used by field `%s'", f->tag,
444 other_field->name);
445 goto error;
446 }
447 }
448
449 if (iopc_field_set_type(f, &field_desc->type, type_table, err) < 0) {
450 goto error;
451 }
452 if (f->repeat == IOP_R_REPEATED) {
453 if (field_desc->optional) {
454 sb_setf(err, "repeated field cannot be optional "
455 "or have a default value");
456 goto error;
457 }
458 } else {
459 iopc_field_set_opt_info(f, field_desc->optional);
460 }
461
462 if (field_desc->is_reference) {
463 if (f->repeat == IOP_R_OPTIONAL) {
464 sb_setf(err, "optional references are not supported");
465 goto error;
466 }
467 if (f->repeat == IOP_R_REPEATED) {
468 sb_setf(err, "arrays of references are not supported");
469 goto error;
470 }
471 f->is_ref = true;
472 }
473
474 return f;
475
476 error:
477 sb_prependf(err, "field `%pL': ", &field_desc->name);
478 iopc_field_delete(&f);
479 return NULL;
480}

504static iopc_struct_t *
505iopc_struct_load(const iop__structure__t *nonnull st_desc,
506 const iopsq_type_table_t *nullable type_table,
507 sb_t *nonnull err)
508{
509 iopc_struct_t *st;
510 iop__field__array_t fields = IOP_ARRAY_EMPTY;
511
512 st = iopc_struct_new();
513 st->name = p_dupz(st_desc->name.s, st_desc->name.len);
514 iop_structure_get_type_and_fields(st_desc, &st->type, &fields);
515
516 qv_grow(&st->fields, fields.len);
517 tab_for_each_ptr(field_desc, &fields) {
518 iopc_field_t *f;
519
520 if (!(f = iopc_field_load(field_desc, &st->fields, type_table, err)))
521 {
522 iopc_struct_delete(&st);
523 return NULL;
524 }
525
526 qv_append(&st->fields, f);
527 }
528
529 return st;
530}

535static iopc_enum_t *iopc_enum_load(const iop__enum__t *en_desc, sb_t *err)
536{
537 t_scope;
538 iopc_enum_t *en;
539 int next_val = 0;
540 qh_t(lstr) keys;
541 qh_t(u32) values;
542
543 t_qh_init(lstr, &keys, en_desc->values.len);
544 t_qh_init(u32, &values, en_desc->values.len);
545 tab_for_each_ptr(enum_val, &en_desc->values) {
546 int32_t val = OPT_DEFVAL(enum_val->val, next_val);
547
548 if (qh_add(u32, &values, val) < 0) {
549 sb_setf(err, "key `%pL': the value `%d' is already used",
550 &enum_val->name, val);
551 return NULL;
552 }
553 if (qh_add(lstr, &keys, &enum_val->name) < 0) {
554 sb_setf(err, "the key `%pL' is duplicated", &enum_val->name);
555 return NULL;
556 }
557 next_val = val + 1;
558 }
559
560 en = iopc_enum_new();
561 en->name = p_dupz(en_desc->name.s, en_desc->name.len);
562 next_val = 0;
563 tab_for_each_ptr(enum_val, &en_desc->values) {
564 iopc_enum_field_t *field = iopc_enum_field_new();
565
566 field->name = p_dupz(enum_val->name.s, enum_val->name.len);
567 field->value = OPT_DEFVAL(enum_val->val, next_val);
568 next_val = field->value + 1;
569
570 qv_append(&en->values, field);
571 }
572
573 return en;
574}

596static iopc_pkg_t *
597iopc_pkg_load_from_iop(const iop__package__t *nonnull pkg_desc,
598 const iopsq_type_table_t *nullable type_table,
599 sb_t *nonnull err)
600{
601 t_scope;
602 iopc_pkg_t *pkg = iopc_pkg_new();
603 qh_t(lstr) things;
604
605 pkg->file = p_strdup("<none>");
606 pkg->name = parse_path(pkg_desc->name, false, err);
607 if (!pkg->name) {
608 sb_prepends(err, "invalid name: ");
609 goto error;
610 }
611 /* XXX Nothing to do for attribute "base" (related to package file path).
612 */
613
614 t_qh_init(lstr, &things, pkg_desc->elems.len);
615 tab_for_each_entry(elem, &pkg_desc->elems) {
616 if (iopc_check_type_name(elem->name, err) < 0) {
617 sb_prependf(err, "invalid %s name: ", pkg_elem_type_to_str(elem));
618 goto error;
619 }
620
621 if (qh_add(lstr, &things, &elem->name) < 0) {
622 sb_setf(err, "already got a thing named `%pL'", &elem->name);
623 goto error;
624 }
625
626 IOP_OBJ_SWITCH(iop__package_elem, elem) {
627 IOP_OBJ_CASE(iop__structure, elem, st_desc) {
628 iopc_struct_t *st;
629
630 if (!(st = iopc_struct_load(st_desc, type_table, err))) {
631 sb_prependf(err, "cannot load `%pL': ", &elem->name);
632 goto error;
633 }
634
635 qv_append(&pkg->structs, st);
636 }
637
638 IOP_OBJ_CASE(iop__enum, elem, en_desc) {
639 iopc_enum_t *en;
640
641 if (!(en = iopc_enum_load(en_desc, err))) {
642 sb_prependf(err, "cannot load enum `%pL': ", &elem->name);
643 goto error;
644 }
645
646 qv_append(&pkg->enums, en);
647 }
648
649 /* TODO Classes */
650 /* TODO Typedefs */
651 /* TODO Interfaces */
652 /* TODO Modules */
653 /* TODO SNMP stuff */
654
655 IOP_OBJ_DEFAULT(iop__package_elem) {
656 sb_setf(err,
657 "package elements of type `%pL' are not supported yet",
658 &elem->__vptr->fullname);
659 goto error;
660 }
661 }
662 }
663
664 return pkg;
665
666 error:
667 iopc_pkg_delete(&pkg);
668 return NULL;
669}

675iop_pkg_t *mp_iopsq_build_pkg(mem_pool_t *nonnull mp,
676 const iop__package__t *nonnull pkg_desc,
677 const iopsq_type_table_t *nullable type_table,
678 sb_t *nonnull err)
679{
680 iop_pkg_t *pkg = NULL;
681 iopc_pkg_t *iopc_pkg;
682
683 if (!expect(mp->mem_pool & MEM_BY_FRAME)) {
684 sb_sets(err, "incompatible memory pool type");
685 return NULL;
686 }
687
688 if (!(iopc_pkg = iopc_pkg_load_from_iop(pkg_desc, type_table, err))) {
689 sb_prependf(err, "invalid package `%pL': ", &pkg_desc->name);
690 return NULL;
691 }
692
693 log_start_buffering_filter(false, LOG_ERR);
694 if (iopc_resolve(iopc_pkg) < 0 || iopc_resolve_second_pass(iopc_pkg) < 0)
695 {
696 const qv_t(log_buffer) *logs = log_stop_buffering();
697
698 sb_sets(err, "failed to resolve the package");
699 tab_for_each_ptr(log, logs) {
700 sb_addf(err, ": %pL", &log->msg);
701 }
702 goto end;
703 }
704 IGNORE(log_stop_buffering());
705
706 pkg = mp_iopc_pkg_to_desc(mp, iopc_pkg, err);
707 if (!pkg) {
708 sb_prependf(err, "failed to generate package `%s': ",
709 iopc_path_dot(iopc_pkg->name));
710 goto end;
711 }
712
713 end:
714 iopc_pkg_delete(&iopc_pkg);
715 return pkg;
716}

718iop_pkg_t *
719mp_iopsq_build_mono_element_pkg(mem_pool_t *nonnull mp,
720 const iop__package_elem__t *nonnull elem,
721 const iopsq_type_table_t *nullable type_table,
722 sb_t *nonnull err)
