41static int do_call(char * const argv[], sb_t *err)
42{
43 pid_t pid;
44
45 pid = ifork();
46 if (pid < 0) {
47 sb_setf(err, "unable to fork(): %m");
48 return -1;
49 }
50
51 if (pid == 0) {
52 setpgid(0, 0);
53 execvp(argv[0], argv);
54 logger_fatal(&_G.logger, "unable to run %s: %m", argv[0]);
55 }
56
57 for (;;) {
58 int status;
59
60 if (waitpid(pid, &status, 0) < 0) {
61 sb_setf(err, "waitpid: %m");
62 }
63 if (WIFEXITED(status)) {
64 int exit_status = WEXITSTATUS(status);
65
66 if (exit_status) {
67 sb_setf(err, "%s returned %d", argv[0], exit_status);
68 return -1;
69 }
70 return 0;
71 }
72 if (WIFSIGNALED(status)) {
73 sb_setf(err, "%s killed with signal %s", argv[0],
74 strsignal(WTERMSIG(status)));
75 return -1;
76 }
77 }
78
79 return 0;
80}

82static int do_compile(const qv_t(str) *in, const char *out, sb_t *err)
83{
84 t_scope;
85 qv_t(cstr) args;
86
87 t_qv_init(&args, 20);
88
89 qv_append(&args, "cc");
90
91 qv_append(&args, "-std=gnu99");
92 qv_append(&args, "-shared");
93 qv_append(&args, "-fPIC");
94
95 qv_append(&args, "-Wall");
96 qv_append(&args, "-Werror");
97 qv_append(&args, "-Wextra");
98 qv_append(&args, "-Wno-unused-parameter");
99
100#ifdef NDEBUG
101 qv_append(&args, "-s"); /* strip DSO */
102 qv_append(&args, "-O3");
103#else
104 qv_append(&args, "-O0");
105 /* XXX valgrind does not support loading dso built with -g3, it fails with
106 * "Warning: DWARF2 reader: Badly formed extended line op encountered"
107 */
108 if (mem_tool_is_running(MEM_TOOL_VALGRIND)) {
109 qv_append(&args, "-g");
110 } else {
111 qv_append(&args, "-g3");
112 }
113#endif
114 qv_append(&args, "-fno-strict-aliasing");
115
116 qv_append(&args, "-o");
117 qv_append(&args, out); /* DSO output */
118 tab_for_each_entry(s, in) {
119 qv_append(&args, s);
120 }
121 qv_append(&args, NULL);
122
123 return do_call((char * const *)args.tab, err);
124}

126static int
127iopc_build(const char *pfxdir, bool display_pfx, const qm_t(iopc_env) *env,
128 const char *iopfile, const char *iopdata, const char *outdir,
129 bool is_main_pkg, lstr_t * nullable pkgname,
130 lstr_t * nullable pkgpath)
131{
132 t_scope;
133 SB_1k(sb);
134 iopc_pkg_t *pkg;
135 lstr_t farch;
136
137 farch = t_farch_get_data(iopc_farch, "../../src/iop/compat.h");
138 sb_add_lstr(&sb, farch);
139 farch = t_farch_get_data(iopc_farch, "../../src/iop/internals.h");
140 sb_add_lstr(&sb, farch);
141
142 iopc_g.prefix_dir = pfxdir;
143 iopc_g.display_prefix = display_pfx;
144
145 iopc_do_c_g.resolve_includes = false;
146 iopc_do_c_g.no_const = true;
147 iopc_do_c_g.iop_compat_header = sb.data;
148
149 iopc_parser_typer_initialize();
150
151 pkg = iopc_parse_file(NULL, env, iopfile, iopdata, is_main_pkg);
152 if (!pkg) {
153 goto error;
154 }
155
156 if (iopc_resolve(pkg) < 0) {
157 goto error;
158 }
159
160 if (iopc_resolve_second_pass(pkg) < 0) {
161 goto error;
162 }
163
164 iopc_types_fold(pkg);
165
166 if (iopc_do_c(pkg, outdir) < 0) {
167 goto error;
168 }
169
170 if (is_main_pkg && iopc_do_json(pkg, outdir) < 0) {
171 goto error;
172 }
173
174 if (pkgname) {
175 *pkgname = lstr_dups(iopc_path_dot(pkg->name), -1);
176 }
177 if (pkgpath) {
178 *pkgpath = lstr_dups(iopc_path_slash(pkg->name), -1);
179 }
180
181 iopc_parser_typer_shutdown();
182 return 0;
183
184 error:
185 iopc_parser_typer_shutdown();
186 return -1;
187}

195int iopc_dso_build(const char *pfxdir, bool display_pfx,
196 const char *iopfile, const qm_t(iopc_env) *env,
197 const char *outdir, sb_t *err)
198{
199 SB_1k(sb);
200 SB_1k(local_err);
201 const qv_t(log_buffer) *log_buffer;
202 lstr_t pkgname = LSTR_NULL_V, pkgpath = LSTR_NULL_V;
203 char so_path[PATH_MAX], path[PATH_MAX], tmppath[PATH_MAX];
204 char json_path[PATH_MAX];
205 qv_t(str) sources;
206 int ret = 0;
207 const char *filepart = path_filepart(iopfile);
208
209 qv_init(&sources);
210
211 path_extend(so_path, outdir, "%s.so", filepart);
212
213 path_extend(tmppath, outdir, "%s.%d.XXXXXX", filepart, getpid());
214 if (!mkdtemp(tmppath)) {
215 sb_setf(err, "failed to create temporary directory %s: %m",
216 tmppath);
217 return -1;
218 }
219
220 /* We'll get the error produced by iopc_build by using the log buffers. */
221 log_start_buffering_filter(false, LOG_ERR);
222
223 if (iopc_build(pfxdir, display_pfx, env, iopfile, NULL, tmppath, true,
224 &pkgname, &pkgpath) < 0)
225 {
226 goto iopc_build_error;
227 }
228
229 /* move json to outdir */
230 path_extend(json_path, outdir, "%*pM.json", LSTR_FMT_ARG(pkgpath));
231 path_extend(path, tmppath, "%*pM.json", LSTR_FMT_ARG(pkgpath));
232 if (rename(path, json_path) < 0) {
233 sb_setf(err, "failed to create json file `%s`: %m", json_path);
234 return -1;
235 }
236
237 qv_append(&sources, asprintf("-I%s", tmppath));
238
239 path_extend(path, tmppath, "%*pM.c", LSTR_FMT_ARG(pkgpath));
240 qv_append(&sources, p_strdup(path));
241
242 sb_addf(&sb, "#include \"%*pM.h\"\n", LSTR_FMT_ARG(pkgpath));
243 sb_adds(&sb, "IOP_EXPORT_PACKAGES_COMMON;\n");
244 sb_adds(&sb, "IOP_USE_EXTERNAL_PACKAGES;\n");
245 sb_addf(&sb, "IOP_EXPORT_PACKAGES(&%*pM__pkg);\n", LSTR_FMT_ARG(pkgname));
246
247 path_extend(path, tmppath, "%*pM-iop-plugin.c", LSTR_FMT_ARG(pkgname));
248 sb_write_file(&sb, path);
249 qv_append(&sources, p_strdup(path));
250
251 qm_for_each_pos(iopc_env, pos, env) {
252 const char *depfile = env->keys[pos];
253 const char *depdata = env->values[pos];
254
255 if (iopc_build(pfxdir, display_pfx, env, depfile, depdata, tmppath,
256 false, NULL, NULL) < 0)
257 {
258 goto iopc_build_error;
259 }
260 }
261 log_stop_buffering();
262
263 if (do_compile(&sources, so_path, &local_err) < 0) {
264 sb_setf(err, "failed to build `%s`: %*pM",
265 so_path, SB_FMT_ARG(&local_err));
266 ret = -1;
267 goto end;
268 }
269 logger_trace(&_G.logger, 1, "iop plugin %s successfully built from %s",
270 so_path, iopfile);
271
272 end:
273 qv_deep_wipe(&sources, p_delete);
274 lstr_wipe(&pkgname);
275 lstr_wipe(&pkgpath);
276 rmdir_r(tmppath, false);
277 return ret;
278
279 iopc_build_error:
280 log_buffer = log_stop_buffering();
281 if (expect(log_buffer->len)) {
282 sb_reset(err);
283 tab_for_each_pos_rev(pos, log_buffer) {
284 if (!err->len) {
285 sb_add_lstr(err, log_buffer->tab[pos].msg);
286 } else {
287 sb_addf(err, ": %*pM", LSTR_FMT_ARG(log_buffer->tab[pos].msg));
288 }
289 }
290 } else {
291 sb_sets(err, "unknown iopc parser error");
292 }
293 ret = -1;
294 goto end;
295}
