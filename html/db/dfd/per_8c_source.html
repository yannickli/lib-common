99 bool copy, lstr_t *res)
100{
101 THROW_ERR_IF(!bs_has_bytes(bs, olen));
102 if (bs_is_aligned(bs)) {

104

106 *res = LSTR_PS_V(&ps);
107 if (copy) {
108 t_lstr_persists(res);
109 }
110 } else {

112
113 buf = t_new_raw(uint8_t, olen);

116 }
117
118 *res = mp_lstr_init(t_pool(), buf, olen);
119 }
120
121 return 0;
122}

127static bool is_bstring_aligned(const asn1_cnt_info_t *constraints,
128 size_t len)
129{
130 /* No need to realign for an empty bit string. */
131 THROW_FALSE_IF(!len);
132
133 if (constraints->max <= 16 && constraints->min == constraints->max) {
134 /* Only fixed-sized bit string with size <= 16 may be not aligned. */
135 if (len != constraints->min) {
136 /* The length is not within the root. */
137 assert(constraints->extended);
138
139 return true;
140 }
141 return false;
142 }
143
144 return true;
145}

152/* Fully constrained integer - d_max < 65536 */
153void aper_write_u16_m(bb_t *bb, uint16_t u16, uint16_t blen, uint16_t d_max)
154{
155 bb_push_mark(bb);
156
157 if (!blen) {
158 goto end;
159 }
160
161 if (blen == 8 && d_max == 255) {
162 /* "The one-octet case". */
163 aper_write_aligned_u8(bb, u16);
164 goto end;
165 }
166
167 if (blen <= 8) {
168 /* "The bit-field case". */
169 bb_be_add_bits(bb, u16, blen);
170 goto end;
171 }
172
173 assert (blen <= 16);
174
175 /* "The two-octet case". */
176 aper_write_aligned_u16(bb, u16);
177 /* FALLTHROUGH */
178
179 end:
180 e_trace_be_bb_tail(5, bb, "constrained number (n = %u)", u16);
181 bb_pop_mark(bb);
182}

184#define PER_FRAG_64K (64 << 10)
185#define PER_FRAG_16K (16 << 10)

187/* Unconstrained length */
188static ALWAYS_INLINE void
189aper_write_ulen(bb_t *bb, size_t l, bool *nullable need_fragmentation)
190{
191 /* See aper_write_len(). */
192 assert (!need_fragmentation || !*need_fragmentation);
193
194 bb_push_mark(bb);
195
196 bb_align(bb);
197
198 e_trace_be_bb_tail(5, bb, "align");
199 bb_reset_mark(bb);
200
201 if (l <= 127) {
202 aper_write_aligned_u8(bb, l);
203
204 e_trace_be_bb_tail(5, bb, "unconstrained length (l = %zd)", l);
205 bb_pop_mark(bb);
206
207 return;
208 }
209
210 if (l < PER_FRAG_16K) {
211 uint16_t u16 = l | (1 << 15);
212
213 aper_write_aligned_u16(bb, u16);
214
215 e_trace_be_bb_tail(5, bb, "unconstrained length (l = %zd)", l);
216 bb_pop_mark(bb);
217
218 return;
219 }
220
221 bb_pop_mark(bb);
222
223 /* The length should be check in advance. */
224 assert(need_fragmentation);
225 *need_fragmentation = true;
226}

228static ALWAYS_INLINE void aper_write_2c_number(bb_t *bb, int64_t v,
229 bool is_signed)
230{
231 uint8_t olen;
232
233 /* XXX Handle the special case of unsigned 64-bits integers
234 * in [ INT64_MAX + 1, UINT64_MAX ]. */
235 if (unlikely(!is_signed && TST_BIT(&v, 63))) {
236 olen = 8;
237 aper_write_ulen(bb, 9, NULL);
238 bb_align(bb);
239 bb_add0s(bb, 8);
240 } else {
241 olen = i64_olen(v);
242 aper_write_ulen(bb, olen, NULL);
243 }
244 aper_write_aligned_int(bb, v, olen);
245}

247/* XXX semi-constrained or constrained numbers */
248static ALWAYS_INLINE void
249aper_write_number(bb_t *bb, uint64_t v, const asn1_int_info_t *info)
250{
251 uint8_t olen;
252
253 if (info && info->has_min && info->has_max) {
254 if (info->max_blen <= 16) {
255 aper_write_u16_m(bb, v, info->max_blen, info->d_max);
256 return;
257 }
258
259 olen = u64_olen(v);
260 aper_write_u16_m(bb, olen - 1, info->max_olen_blen, info->d_max);
261 } else {
262 olen = u64_olen(v);
263 aper_write_ulen(bb, olen, NULL);
264 }
265
266 aper_write_aligned_int(bb, v, olen);
267}

286void aper_write_len(bb_t *bb, size_t l, size_t l_min, size_t l_max,
287 bool *nullable need_fragmentation)
288{
289 /* If set, it is caller's responsibility to pass a boolean initialized
290 * with the value 'false'.
291 */
292 assert (!need_fragmentation || !*need_fragmentation);
293
294 if (l_max != SIZE_MAX) {
295 uint32_t d_max = l_max - l_min;
296 uint32_t d = l - l_min;
297
298 assert (l <= l_max);
299
300 if (d_max < (1 << 16)) {
301 /* TODO pre-process u16_blen(d_max) */
302 aper_write_u16_m(bb, d, u16_blen(d_max), d_max);
303 return;
304 }
305
306 /* FIXME It doesn't look like this case is properly encoded
307 * ("indefinite length case" cf. [1] ITU-T X.691 - §11.5.7)
308 * It looks like we should encode it as a non-negative-binary-integer
309 * in a bit-field (cf. [1] §11.5.7.4), for which the encoding is
310 * described in [1] §11.3.
311 */
312 }
313
314 aper_write_ulen(bb, l, need_fragmentation);
315}

381/* Check constraints, write extension bit (if needed) and prepare encoding
382 * context. */
383static int
384aper_encode_len_extension_bit(bb_t *bb, size_t l, const asn1_cnt_info_t *info,
385 aper_len_encoding_ctx_t *ctx)
386{
387 aper_len_encoding_ctx_init(ctx);
388 ctx->len = l;
389 ctx->remains = l;
390 ctx->done = false;
391
392 if (info) {
393 if (l < info->min || l > info->max) {
394 if (info->extended) {
395 ctx->extension_present = true;
396
397 if (l < info->ext_min || l > info->ext_max) {
398 aper_trace_constraint_violation(info, l);
399 return e_error("extended constraint not respected");
400 }
401
402 /* Extension present */
403 bb_be_add_bit(bb, true);
404 } else {
405 aper_trace_constraint_violation(info, l);
406 return e_error("root constraint not respected");
407 }
408 } else {
409 if (info->extended) {
410 /* Extension not present */
411 bb_be_add_bit(bb, false);
412 }
413
414 ctx->min_root_len = info->min;
415 ctx->max_root_len = info->max;
416 }
417 } else {
418 ctx->max_root_len = SIZE_MAX;
419 }
420
421 return 0;
422}

424/** Encode the length of a repeated element (octet string, bit string,
425 * sequence of, set of, etc...).
426 *
427 * To call before encoding the data. This function also handles data
428 * fragmentation. After the call the number of elements to encode is set in
429 * \p ctx->to_encode and the caller can tell if it was the last bit of data to
430 * encode by checking \p ctx->done.
431 *
432 * Details about the fragmentation:
433 *
434 * The principle and encoding rules for fragmentation is given in the
435 * ITU-T specification X.691, especially in §11.9.3.8.1.
436 *
437 * General case:
438 *
439 * 1. The items are written per fragment of 64k items max.,
440 * 2. Then we write a penultimate fragment of 16k, 32k or 48k items
441 * (if there a less than 16k items left, directly go to next step),
442 * 3. Then we write the remainder.
443
444 * ┌───┬───────┬───────────┬───┬───────┬───────────┬──┬────────┬─────────┐
445 * │ 11 000100 │ 64K items │ 11 000001 │ 16K items │ 0 0000011 │ 3 items │
446 * └───┴───────┴───────────┴───┴───────┴───────────┴──┴────────┴─────────┘
447 * fragment value fragment value unconstrained value
448 * length length length
449 * (16k blocks) (16k blocks) (remainder)
450
451 * Special case when the number of elements is a multiple of 16k.
452 * We encode an empty remainder:
453
454 * ┌───┬───────┬───────────┬───┬───────┬───────────┬──┬────────┐
455 * │ 11 000100 │ 64K items │ 11 000011 │ 48K items │ 0 0000000 │
456 * └───┴───────┴───────────┴───┴───────┴───────────┴──┴────────┘
457 * fragment value fragment value unconstrained
458 * length length length == 0
459 * (16k blocks) (16k blocks) (empty remainder)
460 */
461static void aper_encode_len(bb_t *bb, aper_len_encoding_ctx_t *ctx)
462{
463 if (!ctx->use_fragmentation) {
464 if (ctx->extension_present) {
465 aper_write_ulen(bb, ctx->len, &ctx->use_fragmentation);
466 } else {
467 aper_write_len(bb, ctx->len, ctx->min_root_len,
468 ctx->max_root_len, &ctx->use_fragmentation);
469 }
470 if (!ctx->use_fragmentation) {
471 ctx->done = true;
472 ctx->to_encode = ctx->len;
473 }
474 }
475 if (ctx->use_fragmentation) {
476 if (ctx->remains < PER_FRAG_16K) {
477 aper_write_ulen(bb, ctx->remains, NULL);
478 ctx->to_encode = ctx->remains;
479 ctx->done = true;
480 } else {
481 int nb_16k_blocks;
482 int to_encode;
483
484 to_encode = MIN(ctx->remains, PER_FRAG_64K);
485 nb_16k_blocks = to_encode / PER_FRAG_16K;
486 to_encode = nb_16k_blocks * PER_FRAG_16K;
487 ctx->to_encode = to_encode;
488
489 bb_align(bb);
490 bb_be_add_byte(bb, 0xc0 | nb_16k_blocks);
491 }
492 }
493 ctx->remains -= ctx->to_encode;
494}

531int aper_encode_number(bb_t *bb, int64_t n, const asn1_int_info_t *info,
532 bool is_signed)
533{
534 if (aper_check_int_root_constraints(n, info, is_signed) < 0) {
535 if (info->extended) {
536 if (aper_check_int_ext_constraints(n, info, is_signed) < 0) {
537 return e_error("extended constraint not respected");
538 }
539
540 /* Extension present */
541 bb_be_add_bit(bb, true);
542
543 /* XXX Extension constraints are not PER-visible */
544 aper_write_number(bb, n, NULL);
545
546 return 0;
547 } else {
548 e_error("root constraint not respected");
549 return -1;
550 }
551 } else {
552 if (info->extended) {
553 /* Extension not present */
554 bb_be_add_bit(bb, false);
555 }
556 }
557
558 if (info->has_min) {
559 aper_write_number(bb, n - info->min.i, info);
560 } else { /* Only 2's-complement case */
561 aper_write_2c_number(bb, n, is_signed);
562 }
563
564 return 0;
565}

567int aper_encode_enum(bb_t *bb, int32_t val, const asn1_enum_info_t *e)
568{
569 bool extended_val = false;
570 int pos = asn1_enum_find_val(e, val, &extended_val);
571
572 bb_push_mark(bb);
573
574 if (pos < 0) {
575 e_error("undeclared enumerated value: %d", val);
576 return -1;
577 }
578
579 if (extended_val) {
580 bb_be_add_bit(bb, true);
581 aper_write_nsnnwn(bb, pos);
582
583 return 0;
584 }
585
586 if (e->extended) {
587 bb_be_add_bit(bb, false);
588 }
589
590 aper_write_number(bb, pos, &e->constraints);
591
592 e_trace_be_bb_tail(5, bb, "enum value (value = %d)", val);
593 bb_pop_mark(bb);
594
595 return 0;
596}

601int aper_encode_octet_string(bb_t *bb, lstr_t os, const asn1_cnt_info_t *info)
602{
603 aper_len_encoding_ctx_t ctx;
604 pstream_t ps;
605 bool align_before_data = true;
606
607 if (aper_encode_len_extension_bit(bb, os.len, info, &ctx) < 0) {
608 return -1;
609 }
610
611 if (info && info->max <= 2 && info->min == info->max
612 && os.len == (int)info->max)
613 {
614 /* Short form: the string isn't realigned. */
615 align_before_data = false;
616 }
617
618 ps = ps_initlstr(&os);
619 do {
620 aper_encode_len(bb, &ctx);
621 if (align_before_data) {
622 bb_align(bb);
623 }
624 bb_be_add_bytes(bb, ps.b, ctx.to_encode);
625 __ps_skip(&ps, ctx.to_encode);
626 } while (!ctx.done);
627
628
629 return 0;
630}

632int aper_encode_bstring(bb_t *bb, const bit_stream_t *bits,
633 const asn1_cnt_info_t *info)
634{
635 bit_stream_t bs = *bits;
636 size_t len = bs_len(&bs);
637 aper_len_encoding_ctx_t ctx;
638
639 if (aper_encode_len_extension_bit(bb, len, info, &ctx) < 0) {
640 return -1;
641 }
642 do {
643 bit_stream_t to_write;
644
645 aper_encode_len(bb, &ctx);
646 if (is_bstring_aligned(info, len)) {
647 bb_align(bb);
648 }
649 if (!expect(bs_get_bs(&bs, ctx.to_encode, &to_write) >= 0)) {
650 return e_error("bit string: unexpected length error");
651 }
652 bb_be_add_bs(bb, &to_write);
653 } while (!ctx.done);
654
655 return 0;
656}

679static int
680aper_encode_value(bb_t *bb, const void *v, const asn1_field_t *field)
681{
682 switch (field->type) {
683 case ASN1_OBJ_TYPE(bool):
684 aper_encode_bool(bb, *(const bool *)v);
685 break;
686#define ASN1_ENCODE_INT_CASE(type_t, is_signed) \
687 case ASN1_OBJ_TYPE(type_t): \
688 return aper_encode_number(bb, *(type_t *)v, &field->int_info, \
689 (is_signed));
690 ASN1_ENCODE_INT_CASE(int8_t, true);
691 ASN1_ENCODE_INT_CASE(uint8_t, false);
692 ASN1_ENCODE_INT_CASE(int16_t, true);
693 ASN1_ENCODE_INT_CASE(uint16_t, false);
694 ASN1_ENCODE_INT_CASE(int32_t, true);
695 ASN1_ENCODE_INT_CASE(uint32_t, false);
696 ASN1_ENCODE_INT_CASE(int64_t, true);
697 ASN1_ENCODE_INT_CASE(uint64_t, false);
698#undef ASN1_ENCODE_INT_CASE
699 case ASN1_OBJ_TYPE(enum):
700 return aper_encode_enum(bb, *(int32_t *)v, field->enum_info);
701 case ASN1_OBJ_TYPE(NULL):
702 case ASN1_OBJ_TYPE(OPT_NULL):
703 break;
704 case ASN1_OBJ_TYPE(lstr_t):
705 return aper_encode_octet_string(bb, *(const lstr_t *)v,
706 &field->str_info);
707 case ASN1_OBJ_TYPE(asn1_bit_string_t):
708 return aper_encode_bit_string(bb, (const asn1_bit_string_t *)v,
709 &field->str_info);
710 case ASN1_OBJ_TYPE(SEQUENCE): case ASN1_OBJ_TYPE(CHOICE):
711 case ASN1_OBJ_TYPE(UNTAGGED_CHOICE):
712 return aper_encode_constructed(bb, v, field->u.comp, field);
713 case ASN1_OBJ_TYPE(asn1_ext_t):
714 assert (0);
715 e_error("ext type not supported");
716 break;
717 case ASN1_OBJ_TYPE(OPAQUE):
718 assert (0);
719 e_error("opaque type not supported");
720 break;
721 case ASN1_OBJ_TYPE(SKIP):
722 e_error("skip not supported"); /* We cannot stand squirrels */
723 break;
724 case ASN1_OBJ_TYPE(OPEN_TYPE):
725 e_error("open type not supported");
726 break;
727 }
728
729 return 0;

732static int
733aper_encode_field(bb_t *bb, const void *v, const asn1_field_t *field)
734{
735 int res;
736
737 e_trace(5, "encoding value %s:%s", field->oc_t_name, field->name);
738
739 bb_push_mark(bb);
740
741 if (field->is_open_type || field->is_extension) {
742 lstr_t os;
743 bb_t buf;

746 aper_encode_value(&buf, v, field);

766static void field_bitmap_add_bit(bb_t *bitmap, const void *st,
767 const asn1_field_t *field, bool *present)
768{
769 const void *opt;
770 const void *val;
771 bool field_present;
772
773 assert (field->mode == ASN1_OBJ_MODE(OPTIONAL));
774 opt = GET_DATA_P(st, field, uint8_t);
775 val = asn1_opt_field(opt, field->type);
776 field_present = !!val;

816static int
817aper_encode_sequence(bb_t *bb, const void *st, const asn1_desc_t *desc)
818{
819 BB(ext_bb, desc->fields.len - desc->ext_pos);
820 const void *v;
821 bool extended_fields_reached = false;
822
823 if (desc->is_extended) {
824 uint16_t ext_fields_cnt;
825
826 ext_fields_cnt = fill_ext_bitmap(st, desc, &ext_bb);
827
828#ifndef NDEBUG
829 {
830 t_scope;
831 const char *bits = t_print_be_bb(&ext_bb, NULL);
832
833 e_trace(5, "extension bitmap = [ %s ]", bits);
834 }
835#endif
836
837 /* Put extension bit */
838 e_trace(5, "sequence is extended (extension bit = %d)",
839 !!ext_fields_cnt);
840 bb_be_add_bit(bb, !!ext_fields_cnt);
841 }
842
843 bb_push_mark(bb);
844
845 /* Encode optional fields bit-map */
846 fill_opt_bitmap(st, desc, bb);
847
848 e_trace_be_bb_tail(5, bb, "SEQUENCE OPTIONAL fields bit-map");
849 bb_pop_mark(bb);
850
851 for (int i = 0; i < desc->fields.len; i++) {
852 const asn1_field_t *field = &desc->fields.tab[i];
853
854 assert (field->mode != ASN1_OBJ_MODE(SEQ_OF));
855
856 if (field->mode == ASN1_OBJ_MODE(OPTIONAL)) {
857 const void *opt = GET_DATA_P(st, field, uint8_t);
858
859 v = asn1_opt_field(opt, field->type);
860
861 if (v == NULL) {
862 continue; /* XXX field not present */
863 }
864 } else {
865 v = GET_DATA_P(st, field, uint8_t);
866 }
867
868 if (!extended_fields_reached && field->is_extension) {
869 bit_stream_t ext_bs = bs_init_bb(&ext_bb);
870
871 bb_push_mark(bb);
872
873 /* First extension field reached, write presence bitmap for fields
874 * to come. */
875 extended_fields_reached = true;
876 aper_write_nsnnwn(bb, bs_len(&ext_bs) - 1);
877 e_trace_be_bb_tail(5, bb, "extension bitmap length (l=%zd)",
878 bs_len(&ext_bs));
879 bb_be_add_bs(bb, &ext_bs);
880
881 e_trace_be_bb_tail(5, bb, "extension bitmap");
882 bb_pop_mark(bb);
883 }
884
885 if (aper_encode_field(bb, v, field) < 0) {
886 e_error("failed to encode value %s:%s", field->oc_t_name,
887 field->name);
888 goto error;
889 }
890 }
891
892 bb_wipe(&ext_bb);
893 return 0;
894
895 error:
896 bb_wipe(&ext_bb);
897 return -1;
898}

900static int
901aper_encode_choice(bb_t *bb, const void *st, const asn1_desc_t *desc)
902{
903 const asn1_field_t *choice_field;
904 const asn1_field_t *enum_field;
905 int index;
906 const void *v;
907 bool extension_present = false;
908
909 assert (desc->fields.len > 1);
910
911 enum_field = &desc->fields.tab[0];
912
913 index = __asn1_get_int(st, enum_field);
914 if (index < 1) {
915 return e_error("wrong choice initialization");
916 }
917 e_trace(5, "index = %d", index);
918 choice_field = &desc->fields.tab[index];
919 assert (choice_field->mode == ASN1_OBJ_MODE(MANDATORY));
920
921 /* Put extension bit */
922 if (desc->is_extended) {
923 e_trace(5, "choice is extended");
924
925 if ((extension_present = index >= desc->ext_pos)) {
926 e_trace(5, "extension is present");
927 } else {
928 e_trace(5, "extension is not present");
929 }
930
931 bb_be_add_bit(bb, extension_present);
932 }
933
934 bb_push_mark(bb);
935
936 if (extension_present) {
937 aper_write_nsnnwn(bb, index - desc->ext_pos);
938 } else {
939 /* XXX Indexes start from 0 */
940 aper_write_number(bb, index - 1, &desc->choice_info);
941 }
942
943 e_trace_be_bb_tail(5, bb, "CHOICE index");
944 bb_pop_mark(bb);
945
946 v = GET_DATA_P(st, choice_field, uint8_t);
947 assert (v);
948
949 if (aper_encode_field(bb, v, choice_field) < 0) {
950 return e_error("failed to encode choice element %s:%s",
951 choice_field->oc_t_name, choice_field->name);
952 }
953
954 return 0;

975static int
976aper_encode_seq_of(bb_t *bb, const void *st, const asn1_field_t *field)
977{
978 const asn1_field_t *repeated_field;
979 const asn1_desc_t *desc = field->u.comp;
980 const asn1_void_vector_t *tab;
981 aper_len_encoding_ctx_t ctx;
982 int offset;
983
984 assert (desc->fields.len == 1);
985 repeated_field = &desc->fields.tab[0];
986
987 assert (repeated_field->mode == ASN1_OBJ_MODE(SEQ_OF));
988
989 tab = GET_CONST_PTR(st, asn1_void_vector_t, repeated_field->offset);
990
991 if (aper_encode_len_extension_bit(bb, tab->len, &field->seq_of_info,
992 &ctx) < 0)
993 {
994 return -1;
995 }
996
997 offset = 0;
998 do {
999 aper_encode_len(bb, &ctx);
1000
1001 /* Check for overflow. */
1002 assert(offset + ctx.to_encode <= tab->len);
1003
1004 RETHROW(aper_encode_seq_of_field(bb, repeated_field, tab->data,
1005 offset, offset + ctx.to_encode));
1006 offset += ctx.to_encode;
1007 } while (!ctx.done);
1008
1009 return 0;
1010}

1082int aper_read_u16_m(bit_stream_t *bs, size_t blen, uint16_t d_max,
1083 uint16_t *u16)
1084{
1085 uint64_t res;
1086 assert (blen); /* u16 is given by constraints */
1087
1088 if (blen == 8 && d_max == 255) {
1089 /* "The one-octet case". */
1090 *u16 = 0;
1091 if (aper_read_aligned_u8(bs, (uint8_t *)u16) < 0) {
1092 e_info("cannot read contrained integer: end of input "
1093 "(expected at least one aligned octet)");
1094 return -1;
1095 }
1096
1097 return 0;
1098 }
1099
1100 if (blen <= 8) {
1101 /* "The bit-field case". */
1102 if (bs_be_get_bits(bs, blen, &res) < 0) {
1103 e_info("not enough bits to read constrained integer "
1104 "(got %zd, need %zd)", bs_len(bs), blen);
1105 return -1;
1106 }
1107
1108 *u16 = res;
1109 return 0;
1110 }
1111
1112 /* "The two-octet case". */
1113 if (aper_read_aligned_u16(bs, u16) < 0) {
1114 e_info("cannot read constrained integer: end of input "
1115 "(expected at least two aligned octet left)");
1116 return -1;
1117 }
1118 return 0;
1119}

1121static ALWAYS_INLINE int
1122aper_read_ulen(bit_stream_t *bs, size_t *l, bool *nullable is_fragmented)
1123{
1124 uint64_t len = 0;
1125
1126 /* XXX Same remark as for "need_fragmentation" in aper_write_len(). */
1127 assert (!is_fragmented || !*is_fragmented);
1128
1129 if (bs_align(bs) < 0 || !bs_has(bs, 8)) {
1130 e_info("cannot read unconstrained length: end of input "
1131 "(expected at least one aligned octet left)");
1132 return -1;
1133 }
1134
1135 len = __bs_be_peek_bits(bs, 8);
1136 if (!(len & (1 << 7))) {
1137 __bs_skip(bs, 8);
1138 *l = len;
1139 return 0;
1140 }
1141
1142 if (len & (1 << 6)) {
1143 if (is_fragmented) {
1144 *is_fragmented = true;
1145 return 0;
1146 }
1147 e_info("cannot read unconstrained length: "
1148 "fragmented values are not supported");
1149 return -1;
1150 }
1151
1152 if (bs_be_get_bits(bs, 16, &len) < 0) {
1153 e_info("cannot read unconstrained length: end of input "
1154 "(expected at least a second octet left)");
1155 return -1;
1156 }
1157
1158 *l = len & 0x7fff;
1159 return 0;
1160}

1162static ALWAYS_INLINE int
1163aper_read_2c_number(bit_stream_t *bs, int64_t *v, bool is_signed)
1164{
1165 size_t olen;
1166
1167 if (aper_read_ulen(bs, &olen, NULL) < 0) {
1168 e_info("cannot read unconstrained whole number length");
1169 return -1;
1170 }
1171
1172 /* XXX Handle the special case of unsigned 64-bits integers
1173 * in [ INT64_MAX + 1, UINT64_MAX ]. */
1174 if (olen == 9 && !is_signed) {
1175 uint8_t o = 0;
1176 uint64_t u;
1177
1178 if (aper_read_aligned_u8(bs, &o) < 0) {
1179 goto not_enough_bytes;
1180 }
1181
1182 if (o) {
1183 goto overflow;
1184 }
1185
1186 if (aper_read_aligned_uint(bs, 8, &u) < 0) {
1187 goto not_enough_bytes;
1188 }
1189
1190 *v = u;
1191 return 0;
1192 }
1193
1194 if (olen > 8) {
1195 goto overflow;
1196 }
1197
1198 if (aper_read_aligned_int(bs, olen, v) < 0) {
1199 goto not_enough_bytes;
1200 }
1201
1202 if (!is_signed && *v < 0) {
1203 e_info("cannot write negative number to unsigned integer");
1204 return -1;
1205 }
1206
1207 return 0;
1208
1209 not_enough_bytes:
1210 e_info("not enough bytes to read unconstrained number "
1211 "(got %zd, need %zd)", bs_len(bs) / 8, olen);
1212 return -1;
1213
1214 overflow:
1215 e_info("the number is too big not to overflow");
1216 return -1;
1217}

1220aper_read_number(bit_stream_t *bs, const asn1_int_info_t *info, uint64_t *v)
1221{
1222 size_t olen;
1223
1224 if (info && info->has_min && info->has_max) {
1225 if (info->max_blen <= 16) {
1226 uint16_t u16 = 0;
1227
1228 if (info->max_blen == 0) {
1229 *v = 0;
1230 return 0;
1231 }
1232
1233 if (aper_read_u16_m(bs, info->max_blen, info->d_max, &u16) < 0) {
1234 e_info("cannot read constrained whole number");
1235 return -1;
1236 }
1237
1238 *v = u16;
1239
1240 return 0;
1241 } else {
1242 uint16_t u16 = 0;
1243

1351/* Contextual information about length decoding. */

1405static int
1406aper_decode_fragment_len(bit_stream_t *bs, aper_len_decoding_ctx_t *ctx)
1407{
1408 uint64_t len;
1409
1410 assert(ctx->more_fragments_to_read);
1411
1412 if (bs_align(bs) < 0 || !bs_has(bs, 8)) {
1413 e_info("cannot read fragment len: unexpected end of input");
1414 return -1;
1415 }
1416
1417 len = __bs_peek_bits(bs, 8);
1418 if ((len & 0xc0) == 0xc0) {
1419 /* Got a 16k, 32k, 48k or 64k block fragment. */
1420 if (ctx->to_decode != 0 && ctx->to_decode != PER_FRAG_64K) {
1421 /* Each block fragment except the last one should be a 64k block.
1422 * This fragment isn't the first (ctx->len != 0) and the previous
1423 * fragment wasn't a 64k block fragment, so the rule is broken. */
1424 e_info("unexpected >16k fragment block");
1425 return -1;
1426 }
1427
1428 len &= ~0xc0;
1429 if (!len) {
1430 e_info("unexpected empty fragment block");
1431 return -1;
1432 }
1433 if (len > 4) {
1434 e_info("unexpected >64k fragment block length");
1435 return -1;
1436 }
1437
1438 len *= PER_FRAG_16K;
1439 __bs_skip(bs, 8);
1440 } else {
1441 /* Remainder. */
1442 if (aper_read_ulen(bs, &len, NULL) < 0) {
1443 e_info("cannot read remainder length");
1444 return -1;
1445 }
1446
1447 ctx->more_fragments_to_read = false;
1448 }
1449
1450 ctx->cumulated_len += len;
1451
1452 /* Check the max length isn't exceeded before any further decoding. */
1453 RETHROW(aper_len_check_max(ctx, ctx->cumulated_len));
1454
1455 if (ctx->to_decode < PER_FRAG_16K) {
1456 /* Reached last fragment. The minimum length can be checked now. */
1457 RETHROW(aper_len_check_min(ctx, ctx->cumulated_len));
1458 }
1459 ctx->to_decode = len;
1460
1461 return 0;
1462}

1481/* Read extension bit (if any) and resolve min/max length. */
1482static int
1483aper_decode_len_extension_bit(bit_stream_t *bs, const asn1_cnt_info_t *info,
1484 aper_len_decoding_ctx_t *ctx)
1485{
1486 aper_len_decoding_ctx_init(ctx);
1487
1488 if (info) {
1489 if (info->extended) {
1490 if (bs_done(bs)) {
1491 e_info("cannot read extension bit: end of input");
1492 return -1;
1493 }
1494
1495 ctx->extension_present = __bs_be_get_bit(bs);
1496 }
1497
1498 if (ctx->extension_present) {
1499 ctx->min_len = info->ext_min;
1500 ctx->max_len = info->ext_max;
1501 } else {
1502 ctx->min_len = info->min;
1503 ctx->max_len = info->max;
1504 }
1505 } else {
1506 ctx->max_len = SIZE_MAX;

1508

1512/* Decode a length. The 'ctx' parameter should be initialized by
1513 * 'aper_decode_len_extension_bit()' first. */
1514static int aper_decode_len(bit_stream_t *bs, aper_len_decoding_ctx_t *ctx)
1515{
1516 bool is_fragmented = false;
1517 size_t l = 0;
1518
1519 if (ctx->more_fragments_to_read) {
1520 return aper_decode_fragment_len(bs, ctx);
1521 }
1522 if (ctx->extension_present) {
1523 if (aper_read_ulen(bs, &l, &is_fragmented) < 0) {
1524 e_info("cannot read extended length");
1525 return -1;
1526 }
1527 } else {
1528 if (aper_read_len(bs, ctx->min_len, ctx->max_len, &l,
1529 &is_fragmented) < 0)
1530 {
1531 e_info("cannot read constrained length");
1532 return -1;
1533 }
1534 }
1535 if (is_fragmented) {
1536 ctx->more_fragments_to_read = true;
1537 RETHROW(aper_decode_fragment_len(bs, ctx));
1538 } else {
1539 ctx->to_decode = l;
1540 ctx->cumulated_len = ctx->to_decode;
1541 RETHROW(aper_len_check_constraints(ctx, ctx->cumulated_len));
1542 }
1543
1544 return 0;
1545}

1547int aper_decode_number(bit_stream_t *nonnull bs,
1548 const asn1_int_info_t *nonnull info, bool is_signed,
1549 int64_t *nonnull n)
1550{
1551 int64_t res = 0;
1552
1553 if (info->extended) {
1554 bool extension_present;
1555
1556 if (bs_done(bs)) {
1557 e_info("cannot read extension bit: end of input");
1558 return -1;
1559 }
1560
1561 extension_present = __bs_be_get_bit(bs);
1562
1563 if (extension_present) {
1564 if (aper_read_2c_number(bs, n, is_signed) < 0) {
1565 e_info("cannot read extended unconstrained number");
1566 return -1;
1567 }
1568
1569 if (aper_check_int_ext_constraints(*n, info, is_signed) < 0) {
1570 e_info("extension constraint not respected");
1571 return -1;
1572 }
1573
1574 return 0;
1575 }
1576 }
1577
1578 if (info->has_min) {
1579 uint64_t d;
1580
1581 if (aper_read_number(bs, info, &d) < 0) {
1582 e_info("cannot read constrained or semi-constrained number");
1583 return -1;
1584 }
1585
1586 if (is_signed) {
1587 if (d > (uint64_t)INT64_MAX - info->min.i) {
1588 e_info("cannot decode: overflow of signed 64-bits integer");
1589 return -1;
1590 }
1591
1592 res = info->min.i + d;
1593 } else {
1594 if (d > (uint64_t)UINT64_MAX - info->min.u) {
1595 e_info("cannot decode: overflow of unsigned 64-bits integer");
1596 return -1;
1597 }
1598
1599 res = info->min.u + d;
1600 }
1601 } else {
1602 if (aper_read_2c_number(bs, &res, is_signed) < 0) {
1603 e_info("cannot read unconstrained number");
1604 return -1;
1605 }
1606 }
1607
1608 if (aper_check_int_root_constraints(res, info, is_signed) < 0) {
1609 e_info("root constraint not respected");
1610 return -1;
1611 }
1612
1613 *n = res;
1614
1615 return 0;
1616}

1618int aper_decode_enum(bit_stream_t *bs, const asn1_enum_info_t *e,
1619 int32_t *val)
1620{
1621 int64_t pos;
1622
1623 if (e->extended) {
1624 if (bs_done(bs)) {
1625 e_info("cannot read enumerated type: end of input");
1626 return -1;
1627 }
1628
1629 if (__bs_be_get_bit(bs)) {
1630 size_t nsnnwn;
1631

1662 return -1;

1665 *val = e->values.tab[pos];

1684
1685int t_aper_decode_octet_string(bit_stream_t *bs, const asn1_cnt_info_t *info,
1686 bool copy, lstr_t *os)
1687{
1688 aper_len_decoding_ctx_t len_ctx;
1689 qv_t(u8) buf __attribute__((cleanup(aper_buf_wipe))) = QV_INIT();
1690
1691 if (aper_decode_len_extension_bit(bs, info, &len_ctx) < 0) {
1692 e_info("cannot read extension bit");
1693 return -1;
1694 }
1695
1696 do {
1697 lstr_t data;
1698
1699 if (aper_decode_len(bs, &len_ctx) < 0) {
1700 e_info("cannot decode octet string length");
1701 return -1;
1702 }
1703 if (!buf.len && info && info->max <= 2 &&
1704 info->min == info->max && len_ctx.to_decode == info->max)
1705 {
1706 /* Special case: unaligned fixed-size octet string
1707 * (size 1 or 2). */
1708 } else {
1709 bs_align(bs);
1710 }
1711
1712 if (t_aper_get_unaligned_bytes(bs, len_ctx.to_decode, copy,
1713 &data) < 0)
1714 {
1715 e_info("cannot read octet string: not enough bits");
1716 return -1;
1717 }
1718 if (buf.len) {
1719 memcpy(qv_growlen(&buf, len_ctx.to_decode),
1720 data.data, len_ctx.to_decode);
1721 } else {
1722 qv_init_static(&buf, data.data, data.len);
1723 }
1724 } while (len_ctx.more_fragments_to_read);
1725
1726 if (likely(buf.mp == &mem_pool_libc)) {
1727 /* XXX There were more than one fragment so the buffer was reallocated
1728 * on LIBC. The content has to be transferred on the t_stack or it
1729 * will be lost. */
1730 copy = true;
1731 }
1732 *os = LSTR_DATA_V(buf.tab, buf.len);
1733 if (copy) {
1734 *os = t_lstr_dup(*os);
1735 }
1736
1737 e_trace_hex(6, "Decoded OCTET STRING", os->data, (int)os->len);
1738
1739 return 0;
1740}

1809static int
1810t_aper_decode_value(bit_stream_t *bs, const asn1_field_t *field,
1811 bool copy, void *v)
1812{
1813 switch (field->type) {
1814 case ASN1_OBJ_TYPE(bool):
1815 return aper_decode_bool(bs, (bool *)v);
1816 break;
1817
1818#define ASN1_DECODE_INT_CASE(type_t, type64_t, is_signed) \
1819 case ASN1_OBJ_TYPE(type_t): \
1820 { \
1821 int64_t i64; \
1822 \
1823 RETHROW(aper_decode_number(bs, &field->int_info, (is_signed), \
1824 &i64)); \
1825 e_trace(5, "decoded number value (n = %jd)", i64); \
1826 \
1827 if ((type64_t)i64 != (type_t)i64) { \
1828 e_info("overflow detected for field `%s` (" #type_t ")", \
1829 field->name); \
1830 return -1; \
1831 } \
1832 *(type_t *)v = i64; \
1833 } \
1834 return 0;
1835
1836 ASN1_DECODE_INT_CASE(int8_t, int64_t, true);
1837 ASN1_DECODE_INT_CASE(uint8_t, uint64_t, false);
1838 ASN1_DECODE_INT_CASE(int16_t, int64_t, true);
1839 ASN1_DECODE_INT_CASE(uint16_t, uint64_t, false);
1840 ASN1_DECODE_INT_CASE(int32_t, int64_t, true);
1841 ASN1_DECODE_INT_CASE(uint32_t, uint64_t, false);
1842 ASN1_DECODE_INT_CASE(int64_t, int64_t, true);
1843 ASN1_DECODE_INT_CASE(uint64_t, uint64_t, false);
1844
1845#undef ASN1_DECODE_INT_CASE
1846
1847 case ASN1_OBJ_TYPE(enum):
1848 RETHROW(aper_decode_enum(bs, field->enum_info, (int32_t *)v));
1849 e_trace(5, "decoded enum value (n = %u)", *(int32_t *)v);
1850 return 0;
1851 case ASN1_OBJ_TYPE(NULL):
1852 case ASN1_OBJ_TYPE(OPT_NULL):
1853 break;
1854 case ASN1_OBJ_TYPE(lstr_t):
1855 return t_aper_decode_data(bs, &field->str_info, copy,
1856 (lstr_t *)v);
1857 case ASN1_OBJ_TYPE(asn1_bit_string_t):
1858 return t_aper_decode_bit_string(bs, &field->str_info,
1859 (asn1_bit_string_t *)v);
1860 case ASN1_OBJ_TYPE(SEQUENCE): case ASN1_OBJ_TYPE(CHOICE):
1861 case ASN1_OBJ_TYPE(UNTAGGED_CHOICE):
1862 return t_aper_decode_constructed(bs, field->u.comp, field, copy, v);
1863 case ASN1_OBJ_TYPE(asn1_ext_t):
1864 assert (0);
1865 e_error("ext type not supported");
1866 break;
1867 case ASN1_OBJ_TYPE(OPAQUE):
1868 assert (0);
1869 e_error("opaque type not supported");
1870 break;
1871 case ASN1_OBJ_TYPE(SKIP):
1872 break;
1873 case ASN1_OBJ_TYPE(OPEN_TYPE):
1874 e_error("open type not supported");
1875 break;
1876 }
1877
1878 return 0;
1879}

1942static int
1943t_aper_decode_sequence(bit_stream_t *bs, const asn1_desc_t *desc,
1944 bool copy, void *st)
1945{
1946 bit_stream_t opt_bitmap;
1947 bit_stream_t ext_bitmap;
1948 bit_stream_t *fields_bitmap = &opt_bitmap;
1949 bool extension_present = false;
1950 bool extended_fields_reached = false;
1951
1952 if (desc->is_extended) {
1953 e_trace(5, "the sequence is extended");
1954
1955 if (bs_done(bs)) {
1956 e_info("cannot read extension bit: end of input");
1957 return -1;
1958 }
1959
1960 extension_present = __bs_be_get_bit(bs);
1961 e_trace(5, "extension present");
1962 }
1963
1964 if (!bs_has(bs, desc->opt_fields.len)) {
1965 e_info("cannot read optional fields bit-map: not enough bits");
1966 return -1;
1967 }
1968
1969 opt_bitmap = __bs_get_bs(bs, desc->opt_fields.len);
1970
1971 for (int i = 0; i < desc->fields.len; i++) {
1972 const asn1_field_t *field = &desc->fields.tab[i];
1973 void *v;
1974
1975 if (!extended_fields_reached && field->is_extension) {
1976 extended_fields_reached = true;
1977
1978 if (extension_present) {
1979 e_trace(5, "extended fields reached, read extension bitmap");
1980
1981 if (read_ext_bitmap(bs, &ext_bitmap) < 0) {
1982 e_info("cannot read extension bitmap");
1983 return -1;
1984 }
1985
1986 fields_bitmap = &ext_bitmap;
1987 }
1988 }
1989
1990 if (field->mode == ASN1_OBJ_MODE(OPTIONAL)) {
1991 if (bs_done(fields_bitmap)) {
1992 if (likely(extended_fields_reached)) {
1993 e_trace(5, "extended field `%s:%s` not present "
1994 "(out of bitmap range)", field->oc_t_name,
1995 field->name);
1996
1997 /* Extended field not present (out of extension bitmap
1998 * range). */
1999 asn1_opt_field_w(GET_PTR(st, field, void), field->type,
2000 false);
2001 continue;
2002 }
2003
2004 assert (0);
2005 return e_error("sequence is broken");
2006 }
2007
2008 if (!__bs_be_get_bit(fields_bitmap)) {
2009 e_trace(5, "field `%s:%s` not present", field->oc_t_name,
2010 field->name);
2011
2012 /* Extended field not present (bit=0 in extension bitmap). */
2013 asn1_opt_field_w(GET_PTR(st, field, void), field->type,
2014 false);
2015 continue;
2016 }
2017
2018 t_alloc_if_pointed(field, st);
2019 v = asn1_opt_field_w(GET_PTR(st, field, void), field->type, true);
2020 } else {
2021 assert (field->mode != ASN1_OBJ_MODE(SEQ_OF));
2022
2023 /* Should be checked in "asn1_reg_field()". */
2024 assert (!field->is_extension);
2025
2026 v = t_alloc_if_pointed(field, st);
2027 }
2028
2029 e_trace(5, "decoding SEQUENCE value %s:%s",
2030 field->oc_t_name, field->name);
2031
2032 if (t_aper_decode_field(bs, field, copy, v) < 0) {
2033 e_info("cannot read sequence field %s:%s",
2034 field->oc_t_name, field->name);
2035 return -1;
2036 }
2037 }
2038
2039 if (extension_present) {
2040 if (!extended_fields_reached) {
2041 e_trace(5, "skipping extension bitmap");
2042
2043 /* The sequence is registered as extended but no extended field is
2044 * registered. Skip the extended field bitmap. */
2045 if (read_ext_bitmap(bs, &ext_bitmap) < 0) {
2046 e_info("cannot read extension bitmap (for skipping)");
2047 return -1;
2048 }
2049 }
2050
2051 /* Skip all the unknown extended fields. */
2052 while (!bs_done(&ext_bitmap)) {
2053 lstr_t os;
2054
2055 if (!__bs_be_get_bit(&ext_bitmap)) {
2056 e_trace(5, "skipping unknown extension (absent)");
2057 continue;
2058 }
2059
2060 e_trace(5, "skipping unknown extension (present)");
2061 if (t_aper_decode_octet_string(bs, NULL, false, &os) < 0) {
2062 e_info("cannot skip unknown extension field");
2063 return -1;
2064 }
2065 e_trace(5, "skipped unknown extension with encoding size %d",
2066 os.len);
2067 }
2068 }
2069
2070 return 0;
2071}

2074t_aper_decode_choice(bit_stream_t *bs, const asn1_desc_t *desc, bool copy,

2078 const asn1_field_t *enum_field;

2080 size_t index;
2081 void *v;
2082 bool extension_present = false;
2083
2084 if (desc->is_extended) {

2086 e_info("cannot read extension bit: end of input");

2088 }

2091 if (extension_present) {
2092 e_trace(5, "extension present");
2093 } else {
2094 e_trace(5, "extension not present");
2095 }
2096 } else {
2097 e_trace(5, "choice is not extended");
2098 }

2102 e_info("cannot read choice extension index");

2142 e_info("cannot decode choice value");

2144 }

2147}
