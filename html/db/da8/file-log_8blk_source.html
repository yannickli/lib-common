33log_file_t *log_file_init(log_file_t *log_file, const char *nametpl, int flags)
34{
35 const char *ext = path_ext(nametpl);
36 int len = strlen(nametpl);
37
38 p_clear(log_file, 1);
39 log_file->flags = flags;
40 log_file->mode = 0644;
41
42 if (len + 8 + 1 + 6 + 4 >= ssizeof(log_file->prefix)) {
43 logger_panic(&logger_g, "path format too long");
44 }
45 if (ext) {
46 pstrcpy(log_file->ext, sizeof(log_file->ext), ext + 1);
47 len -= strlen(ext);
48 } else {
49 pstrcpy(log_file->ext, sizeof(log_file->ext), "log");
50 }
51 pstrcpymem(log_file->prefix, sizeof(log_file->prefix), nametpl, len);
52
53 qh_init(u64, &log_file->files_being_compressed);
54 log_file->refcnt = 1;
55
56 return log_file;
57}

98static const char *t_build_real_path(log_file_t *log_file, time_t date)
99{
100 struct tm tm;
101
102 if (log_file->flags & LOG_FILE_UTCSTAMP) {
103 gmtime_r(&date, &tm);
104 } else {
105 localtime_r(&date, &tm);
106 }
107 return t_fmt("%s_" LOG_FILE_DATE_FMT ".%s",
108 log_file->prefix, LOG_FILE_DATE_FMT_ARG(tm), log_file->ext);
109}

111#define GZIP_ERROR "background compression of log file `%*pM` failed: "
112
113static void
114log_file_bgcompress_check(el_t ev, pid_t pid, int st, data_t priv)
115{
116 bgcompr_ctx_t *ctx = priv.ptr;
117
118 if (!WIFEXITED(st)) {
119 logger_error(&logger_g, GZIP_ERROR "gzip killed with signal %d (%s)",
120 LSTR_FMT_ARG(ctx->path), WTERMSIG(st),
121 strsignal(WTERMSIG(st)));
122 } else
123 if (WEXITSTATUS(st) != 0) {
124 logger_error(&logger_g, GZIP_ERROR "gzip exited with status %d",
125 LSTR_FMT_ARG(ctx->path), WEXITSTATUS(st));
126 }
127
128 /* Unlocking the file once the compression done. */
129 qh_del_key(u64, &ctx->log_file->files_being_compressed, ctx->ts);
130 bgcompr_ctx_delete(&ctx);
131}
132
133static void log_file_bgcompress(log_file_t *log_file, const char *path)
134{
135 bgcompr_ctx_t *ctx;
136 pid_t pid;
137 time_t ts;
138
139 if (log_file_get_file_stamp(log_file, path, &ts) < 0) {
140 return;
141 }
142
143 if (qh_add(u64, &log_file->files_being_compressed, ts) < 0) {
144 /* File is already being compressed. */
145 return;
146 }
147
148 ctx = bgcompr_ctx_new();
149 ctx->log_file = log_file_retain(log_file);
150 ctx->path = lstr_dups(path, -1);
151 ctx->ts = ts;
152
153 pid = ifork();
154 if (pid < 0) {
155 logger_error(&logger_g, "unable to fork gzip in the background, %m");
156 bgcompr_ctx_delete(&ctx);
157 return;
158 }
159 if (pid > 0) {
160 el_child_register(pid, log_file_bgcompress_check, ctx);
161 } else {
162 setsid();
163 setpriority(PRIO_PROCESS, getpid(), NZERO / 4);
164 execlp("gzip", "gzip", "-9", path, NULL);
165 logger_error(&logger_g, "execl failed: %m");
166 _exit(0);
167 }
168}

189static void log_delete_file(log_file_t *log_file, const char *path,
190 off_t size)
191{
192 time_t ts;
193
194 log_file_get_file_stamp(log_file, path, &ts);
195 if (qh_find(u64, &log_file->files_being_compressed, ts) >= 0) {
196 /* The file is being compressed, we prevent its deletion for the
197 * current rotation. */
198 return;
199 }
200
201 if (unlink(path) < 0) {
202 logger_error(&logger_g, "cannot delete log file `%s`: %m", path);
203 } else {
204 assert (log_file->total_size >= (uint64_t)size);
205 log_file->total_size -= size;
206 log_file_call_cb(log_file, LOG_FILE_DELETE, path);
207 }
208}

210static void log_check_invariants(log_file_t *log_file)
211{
212 t_scope;
213 glob_t globbuf;
214 char buf[PATH_MAX];
215 struct stat *sts;
216 char **fv;
217 int fc;
218
219 snprintf(buf, sizeof(buf), "%s_????????_??????.%s{,.gz}",
220 log_file->prefix, log_file->ext);
221 if (glob(buf, GLOB_BRACE, NULL, &globbuf)) {
222 globfree(&globbuf);
223 return;
224 }
225
226 fv = globbuf.gl_pathv;
227 fc = globbuf.gl_pathc;
228 qsort(fv, fc, sizeof(fv[0]), qsort_strcmp);
229
230 log_file->total_size = 0;
231 sts = t_new_raw(struct stat, fc);
232 for (int i = 0; i < fc; i++) {
233 if (lstat(fv[i], &sts[i]) < 0 || !S_ISREG(sts[i].st_mode)) {
234 sts[i].st_size = 0;
235 } else {
236 log_file->total_size += sts[i].st_size;
237 }
238 }
239
240 if (log_file->max_files) {
241 for (; fc > log_file->max_files; fc--, fv++, sts++) {
242 if (log_file->flags & LOG_FILE_COMPRESS) {
243 if (strequal(path_extnul(fv[0]), ".gz")) {
244 log_delete_file(log_file, fv[0], sts[0].st_size);
245 } else {
246 break;
247 }
248 } else {
249 log_delete_file(log_file, fv[0], sts[0].st_size);
250 }
251 }
252 }
253
254 if (log_file->max_total_size) {
255 int64_t totalsize = (int64_t)log_file->max_total_size << 20;
256
257 for (int i = fc; i-- > 0; ) {
258 if ((log_file->flags & LOG_FILE_COMPRESS)
259 && !strequal(path_extnul(fv[i]), ".gz"))
260 {
261 /* XXX: uncompressed files must be compressed so skip it
262 * while accounting totalsize */
263 continue;
264 }
265 totalsize -= sts[i].st_size;
266 if (totalsize < 0) {
267 for (int j = 0; j <= i; j++) {
268 log_delete_file(log_file, fv[j], sts[j].st_size);
269 }
270 fv += i + 1;
271 fc -= i + 1;
272 /* the assignment to sts is unused, but kept to keep the
273 * invariant between fv, sts and fc */
274 sts += i + 1;
275 IGNORE(sts);
276 break;
277 }
278 }
279 }
280
281 if (log_file->flags & LOG_FILE_COMPRESS) {
282 for (int i = 0; i < fc - 1; i++) {
283 if (!strequal(path_extnul(fv[i]), ".gz")) {
284 log_file_bgcompress(log_file, fv[i]);
285 }
286 }
287 }
288 globfree(&globbuf);
289}

291static void log_file_open_new(log_file_t *log_file, time_t date)
292{
293 t_scope;
294 const char *real_path;
295 bool existed;
296
297 real_path = t_build_real_path(log_file, date);
298 existed = access(real_path, R_OK) == 0;
299
300 log_file->open_date = date;
301
302 if (log_file->is_file_bin) {
303 log_file->_bin_internal = file_bin_create(LSTR(real_path),
304 1 << 20, false);
305 if (log_file->_internal) {
306 log_file->total_size += log_file->_bin_internal->cur;
307 } else {
308 logger_error(&logger_g, "could not open log file `%s`: %m",
309 real_path);
310 }
311 } else {
312 log_file->_internal = file_open(real_path, FILE_WRONLY | FILE_CREATE,
313 log_file->mode);
314 if (!log_file->_internal
315 || file_seek(log_file->_internal, 0, SEEK_END) == (off_t)-1)
316 {
317 logger_error(&logger_g, "could not open log file `%s`: %m",
318 real_path);
319 }
320 }
321
322 if (!existed && log_file->_internal) {
323 log_file_call_cb(log_file, LOG_FILE_CREATE, NULL);
324 }
325
326 if (!(log_file->flags & LOG_FILE_NOSYMLINK)) {
327 char sym_path[PATH_MAX];
328
329 snprintf(sym_path, sizeof(sym_path), "%s%s.%s", log_file->prefix,
330 log_file->flags & LOG_FILE_USE_LAST ? "_last" : "",
331 log_file->ext);
332 unlink(sym_path);
333 if (symlink(real_path, sym_path)) {
334 logger_error(&logger_g, "could not symlink `%s` to `%s`: %m",
335 real_path, sym_path);
336 }
337 }
338 log_check_invariants(log_file);
339}

342{
343 char buf[PATH_MAX];
344 glob_t globbuf;
345
346 snprintf(buf, sizeof(buf), "%s_????????_??????.%s",
347 log_file->prefix, log_file->ext);
348 if (!glob(buf, 0, NULL, &globbuf) && globbuf.gl_pathc) {

465static int log_file_rotate_(log_file_t *file, time_t now)
466{
467 if (file->open_date == now) {
468 return 0;
469 }
470
471 if (file->is_file_bin) {
472 RETHROW(file_bin_close(&file->_bin_internal));
473 } else {
474 RETHROW(file_close(&file->_internal));
475 }
476
477 log_file_call_cb(file, LOG_FILE_ROTATE, NULL);
478
479 log_file_open_new(file, now);
480 if (!file->_internal) {
481 logger_trace(&logger_g, 1, "could not rotate");
482 return -1;
483 }
484
485 return 0;
486}

488static int log_check_rotate(log_file_t *lf)
489{
490 if (lf->disable_rotation) {
491 return 0;
492 }
493
494 if (lf->max_size > 0) {
495 off_t size;
496
497 if (lf->is_file_bin) {
498 size = lf->_bin_internal->cur;
499 } else {
500 size = file_tell(lf->_internal);
501 }
502
503 if (size >= lf->max_size) {
504 return log_file_rotate_(lf, time(NULL));
505 }
506 }
507
508 if (lf->rotate_delay > 0) {
509 time_t now = time(NULL);
510
511 if (lf->open_date + lf->rotate_delay <= now) {
512 return log_file_rotate_(lf, now);
513 }
514 }
515
516 return 0;
517}

633int log_file_get_file_stamp(const log_file_t *file, const char *path,
634 time_t *out)
635{
636 struct tm tm;
637 const byte *d;
638 int prefix_len = strlen(file->prefix);
639
640
641 if (strlen(path) < prefix_len + strlen("YYYYMMDD_hhmmss")) {
642 goto error;
643 }
644 d = (const byte *)path + prefix_len + 1;
645
646 p_clear(&tm, 1);
647 tm.tm_isdst = -1; /* We don't know current dst */
648 tm.tm_year = memtoip(d, 4, &d) - 1900;
649 tm.tm_mon = memtoip(d, 2, &d) - 1;
650 tm.tm_mday = memtoip(d, 2, &d);
651 if (*d++ != '_') {
652 goto error;
653 }
654 tm.tm_hour = memtoip(d, 2, &d);
655 tm.tm_min = memtoip(d, 2, &d);
656 tm.tm_sec = memtoip(d, 2, &d);
657 if (file->flags & LOG_FILE_UTCSTAMP) {
658 *out = timegm(&tm);
659 } else {
660 *out = mktime(&tm);
661 }
662
663 return 0;
664
665 error:
666 logger_error(&logger_g, "failed to get the timestamp of file `%s`", path);
667 *out = 0;
668 assert (false);
669
670 return -1;
671}
