35static lstr_t t_split_camelcase_word(lstr_t s)
36{
37 t_SB(buf, s.len + 1);
38
39 sb_addc(&buf, toupper(s.s[0]));
40 for (int i = 1; i < s.len; i++) {
41 if (i && isupper(s.s[i])) {
42 sb_addc(&buf, ' ');
43 }
44 sb_addc(&buf, s.s[i]);
45 }
46
47 return LSTR_SB_V(&buf);
48}

157static lstr_t t_struct_build_oid(qv_t(u16) oids,
158 const iop_struct_t *snmp_obj)
159{
160 t_SB(sb, 128);
161
162 assert (iop_struct_is_snmp_obj(snmp_obj) ||
163 iop_struct_is_snmp_tbl(snmp_obj));
164
165 do {
166 qv_append(&oids, snmp_obj->snmp_attrs->oid);
167 } while ((snmp_obj = snmp_obj->snmp_attrs->parent));
168
169 tab_for_each_pos_rev(pos, &oids) {
170 sb_addf(&sb, ".%d", oids.tab[pos]);
171 }
172 return LSTR_SB_V(&sb);
173}

254static void doc_put_rpc(sb_t *buf, int tag, lstr_t iface_name,
255 const iop_rpc_t *rpc,
256 const iop_iface_t *parent)
257{
258 t_scope;
259 const iop_struct_t *st = rpc->args;
260 lstr_t name = rpc->name;
261 lstr_t camelcase = t_split_camelcase_word(name);
262 lstr_t oid_str = t_notif_build_oid(st, parent);
263 lstr_t help = t_rpc_get_help(&parent->rpc_attrs[tag]);
264 lstr_t severity = doc_rpc_get_severity(parent, rpc);
265
266 sb_addf(buf,
267 "| ALM-%*pM-%u |\n"
268 "*%*pM* (%*pM) +\n"
269 "\n%*pM +\n"
270 "\n*Severity:* %*pM.\n"
271 "\n*Parameters*\n\n",
272 LSTR_FMT_ARG(iface_name), st->snmp_attrs->oid,
273 LSTR_FMT_ARG(camelcase), LSTR_FMT_ARG(oid_str),
274 LSTR_FMT_ARG(help),
275 LSTR_FMT_ARG(severity));
276
277 if (st->fields_len == 0) {
278 sb_adds(buf, "*No parameter*\n");
279 return;
280 }
281 /* Parameters */
282 for (int i = 0; i < st->fields_len; i++) {
283 const iop_field_t *field = &st->fields[i];
284 const iop_snmp_attrs_t *attr;
285 const iop_field_t *field_origin;
286
287 if (!iop_field_has_snmp_info(field)) {
288 continue;
289 }
290
291 if (!(attr = iop_get_snmp_attrs(&st->fields_attrs[i]))) {
292 logger_panic(&_G.logger,
293 "no snmp attribute found for field `%*pM`",
294 LSTR_FMT_ARG(st->fields[i].name));
295 }
296 field_origin = iop_get_field_match_oid(attr->parent, attr->oid);
297 doc_put_arg_field(buf, field_origin, attr->parent, attr->oid);
298
299 if (i == st->fields_len - 1) {
300 sb_adds(buf, ".\n");
301 } else {
302 sb_adds(buf, ";\n");
303 }
304 }
305}

307static void doc_put_alarms(sb_t *buf, const iop_pkg_t *pkg)
308{
309 for (const iop_iface_t *const *it = pkg->ifaces; *it; it++) {
310 t_scope;
311 const iop_iface_t *iface = *it;
312 lstr_t name_full_up = t_get_name_full_up(pkg->name);
313
314 if (!iop_iface_is_snmp_iface(iface)) {
315 continue;
316 }
317
318 if (iface->funs_len) {
319 doc_put_alarms_header(buf, name_full_up);
320 }
321 for (int i = 0; i < iface->funs_len; i++) {
322 doc_put_rpc(buf, i, name_full_up, &iface->funs[i], iface);
323 }
324 if (iface->funs_len) {
325 sb_adds(buf, "|===\n");
326 }
327 }
328}

365static void doc_put_field(sb_t *buf, int pos, const iop_struct_t *st)
366{
367 t_scope;
368 const iop_field_attrs_t field_attrs = st->fields_attrs[pos];
369 const iop_field_t *field = &st->fields[pos];
370 const iop_snmp_attrs_t *snmp_attrs = doc_field_get_snmp_attr(field_attrs);
371 lstr_t oid = t_field_build_oid(field, st);
372 lstr_t help;
373
374 help = t_field_get_help_without_dot(
375 iop_get_field_attr_match_oid(st, snmp_attrs->oid));
376
377 sb_addf(buf,
378 "|[[%*pM]]%*pM\n"
379 "|32436%*pM\n"
380 "|%s\n"
381 "|%*pM.\n\n",
382 LSTR_FMT_ARG(field->name), LSTR_FMT_ARG(field->name),
383 LSTR_FMT_ARG(oid),
384 iop_type_get_string_desc(field->type),
385 LSTR_FMT_ARG(help));
386
387 if (field->type == IOP_T_ENUM) {
388 const iop_enum_t *en = field->u1.en_desc;
389
390 sb_adds(buf, "Possible values:\n\n");
391 for (int i = 0; i < en->enum_len; i++) {
392 sb_addf(buf, "- %*pM (%d)\n",
393 LSTR_FMT_ARG(en->names[i]), en->values[i]);
394 }
395 sb_adds(buf, "\n\n");
396 }

399static void doc_put_fields(sb_t *buf, const iop_pkg_t *pkg)
400{
401 int compt = 0;
402
403 for (const iop_struct_t *const *it = pkg->structs; *it; it++) {
404 const iop_struct_t *desc = *it;
405
406 if (!iop_struct_is_snmp_st(desc)) {
407 continue;
408 }
409
410 if (iop_struct_is_snmp_tbl(desc)) {
411 doc_put_tbl(buf, desc);
412 }
413
414 if (desc->fields_len > 0) {
415 t_scope;
416 lstr_t short_name = t_get_short_name(desc->fullname);
417
418 if (compt == 0) {
419 doc_put_field_header(buf);
420 }
421 compt++;
422 sb_addf(buf, "4+^s|*%*pM*\n\n", LSTR_FMT_ARG(short_name));
423 }
424
425 /* deal with snmp fields */
426 for (int i = 0; i < desc->fields_len; i++) {
427 const iop_field_t field = desc->fields[i];
428
429 if (!iop_field_has_snmp_info(&field)) {
430 continue;
431 }
432 doc_put_field(buf, i, desc);
433 }
434 }
435 if (compt > 0) {
436 sb_adds(buf, "|===\n");
437 }
438}

474int iop_snmp_doc(int argc, char **argv, const qv_t(pkg) *pkgs)
475{
476 lstr_t path_notif = LSTR_NULL;
477 lstr_t path_object = LSTR_NULL;
478 SB_8k(notif_sb);
479 SB_8k(object_sb);
480
481 doc_parseopt(argc, argv, &path_notif, &path_object);
482 iop_write_snmp_doc(&notif_sb, &object_sb, pkgs);
483
484 if (sb_write_file(&notif_sb, path_notif.s) < 0) {
485 logger_error(&_G.logger,
486 "couldn't write SNMP notification doc file `%*pM`: %m",
487 LSTR_FMT_ARG(path_notif));
488 return -1;
489 }
490 if (sb_write_file(&object_sb, path_object.s) < 0) {
491 logger_error(&_G.logger,
492 "couldn't write SNMP object doc file `%*pM`: %m",
493 LSTR_FMT_ARG(path_object));
494 return -1;
495 }
496
497 return 0;
