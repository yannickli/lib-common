22bb_t *bb_init(bb_t *bb)
23{
24 p_clear(bb, 1);
25
26 bb->alignment = 8;
27
28#ifndef NDEBUG
29 qv_init(&bb->marks);
30#endif
31
32 return bb;
33}

35void bb_wipe(bb_t *bb)
36{
37#ifndef NDEBUG
38 qv_wipe(&bb->marks);
39#endif
40
41 mp_delete(bb->mp, &bb->data);
42}

44void bb_reset(bb_t *bb)
45{
46 mem_pool_t *mp = mp_ipool(bb->mp);
47 const size_t alignment = bb->alignment;
48
49 if (!(mp->mem_pool & MEM_BY_FRAME) && bb->size > (16 << 10)) {
50 bb_wipe(bb);
51 bb_init(bb);
52 bb->alignment = alignment;
53 } else {
54 bzero(bb->data, DIV_ROUND_UP(bb->len, 8));
55 bb->len = 0;
56 }
57}

59void bb_init_sb(bb_t *bb, sb_t *sb)
60{
61 if (sb->data == __sb_slop) {
62 bb_init(bb);
63 } else {
64 /* bb->size is a number of 64 bits words so the sb size must be bigger
65 * than the sb length rounded up to 8 bytes */
66 sb_grow(sb, ROUND_UP(sb->len, 8) - sb->len);
67 bb_init_full(bb, sb->data, sb->len * 8, sb->size / 8, 8, sb->mp);
68
69 /* We took ownership of the memory so ensure clear the sb */
70 sb_init(sb);
71 }
72}

75void bb_transfer_to_sb(bb_t *bb, sb_t *sb)
76{
77 sb_wipe(sb);
78 bb_grow(bb, 8);
79 sb_init_full(sb, bb->data, DIV_ROUND_UP(bb->len, 8), bb->size * 8,
80 bb->mp);
81 bb->data = NULL;
82 bb_wipe(bb);
83 bb_init(bb);
84}

86void __bb_grow(bb_t *bb, size_t extra)
87{
88 size_t newlen = DIV_ROUND_UP(bb->len + extra, 64);
89 size_t newsz;
90
91 newsz = p_alloc_nr(bb->size);
92 if (newsz < newlen) {
93 newsz = newlen;
94 }
95 assert (bb->alignment && bb->alignment % 8 == 0);
96 newsz = ROUND_UP(newsz, bb->alignment / 8);
97
98 bb->data = mp_irealloc_fallback(&bb->mp, bb->data, bb->size * 8,
99 newsz * 8, bb->alignment, 0);
100 bb->size = newsz;
101}

103void bb_add_bs(bb_t *bb, const bit_stream_t *b)
104{
105 bit_stream_t bs = *b;
106 pstream_t ps;
107
108 while (!bs_done(&bs) && !bs_is_aligned(&bs)) {
109 bb_add_bit(bb, __bs_get_bit(&bs));
110 }
111
112 ps = __bs_get_bytes(&bs, bs_len(&bs) / 8);
113 bb_add_bytes(bb, ps.b, ps_len(&ps));
114
115 while (!bs_done(&bs)) {
116 bb_add_bit(bb, __bs_get_bit(&bs));
117 }
118}

120void bb_be_add_bs(bb_t *bb, const bit_stream_t *b)
121{
122 bit_stream_t bs = *b;
123 pstream_t ps;
124
125 while (!bs_done(&bs) && !bs_is_aligned(&bs)) {
126 bb_be_add_bit(bb, __bs_be_get_bit(&bs));
127 }
128
129 ps = __bs_get_bytes(&bs, bs_len(&bs) / 8);
130 bb_be_add_bytes(bb, ps.b, ps_len(&ps));
131
132 while (!bs_done(&bs)) {
133 bb_be_add_bit(bb, __bs_be_get_bit(&bs));
134 }
135}

137void bb_shift_left(bb_t *bb, size_t shift)
138{
139 if (shift >= bb->len) {
140 bb_reset(bb);
141 } else
142 if (shift % 8 == 0) {
143 /* Shift is nicely aligned */
144 const size_t bshift = shift / 8;
145 const size_t blen = DIV_ROUND_UP(bb->len, 8);
146
147 p_move(bb->bytes, (bb->bytes + bshift), blen - bshift);
148 p_clear(bb->bytes + blen - bshift, bshift);
149
150 bb->len -= shift;
151 } else {
152 const size_t nwords = shift / 64, wshift = shift % 64;
153 const size_t rwshift = 64 - wshift;
154 size_t src = nwords, dst = 0;
155
156 /* We have to shift the bit-buffer word per word… Here we are allowed
157 * to overflow a little because we know that we are always aligned on
158 * 8 bytes. */
159 while (src < bb->word) {
160 /* Copy the bits from src which have to be shifted in dst */
161 bb->data[dst++] = ((bb->data[src] >> wshift)
162 | (bb->data[src + 1] << rwshift));
163 src++;
164 }
165
166 /* Handle last word */
167 bb->data[dst] = (bb->data[src] >> wshift);
168
169 /* Now we have to cleanup the old bits */
170 if (dst < bb->word) {
171 p_clear(&bb->data[dst + 1], bb->word - dst);
172 }
173
174 bb->len -= shift;
175 }
176}

178char *t_print_bits(uint8_t bits, uint8_t bstart, uint8_t blen)
179{
180 char *str = t_new(char, blen + 1);
181 char *w = str;
182
183 for (int i = bstart; i < blen; i++) {
184 *w++ = (bits & (1 << i)) ? '1' : '0';
185 }
186
187 *w = '\0';
188
189 return str;
190}

199int z_set_be_bb(bb_t *bb, const char *bits, sb_t *err)
200{
201 int c;
202 uint8_t u = 0;
203 int blen = 0;
204 const char *r = bits;
205
206 bb_reset(bb);
207 for (;;) {
208 c = *r++;
209
210 if (c == '0' || c == '1') {
211 u <<= 1;
212 blen++;
213
214 if (blen > 8) {
215 sb_sets(err, "invalid input");
216 return -1;
217 }
218
219 if (c == '1') {
220 u |= 1;
221 }
222 } else
223 if (c == '.' || !c) {
224 bb_add_bits(bb, u, blen);
225
226 if (!c) {
227 break;
228 }
229
230 u = 0;
231 blen = 0;
232 } else {
233 sb_setf(err, "unexpected character '%c'", c);
234 return -1;
235 }
236 }
237
238 {
239 t_scope;
240 const char *s = t_print_be_bb(bb, NULL);
241
242 if (!strequal(s, bits)) {
243 sb_setf(err, "input different when re-generated: got `%s`", s);
244 return -1;
245 }
246 }
247
248 return 0;
249}
