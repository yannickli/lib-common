22int
23iopc_set_path(const char *outdir, const iopc_pkg_t *pkg,
24 const char *ext, int max_len, char *path,
25 bool only_pkg)
26{
27 char dpath[max_len];
28 int res;
29
30 if (outdir) {
31 if (only_pkg) {
32 lstr_t sp = LSTR(iopc_path_slash(pkg->name));
33
34 /* pretty path force the .iop extension which may differs of the
35 * given extension. */
36 sp.len -= strlen(".iop");
37
38 res = snprintf(path, max_len, "%s/%*pM%s", outdir,
39 LSTR_FMT_ARG(sp), ext);
40 } else {
41 res = snprintf(path, max_len, "%s/%s%s", outdir,
42 iopc_path_basename(pkg->name), ext);
43 }
44 path_dirname(dpath, max_len, path);
45 if (mkdir_p(dpath, 0777) < 0) {
46 throw_error("cannot create directory `%s`: %m", dpath);
47 }
48 } else {
49 path_dirname(dpath, max_len, pkg->file);
50 res = snprintf(path, max_len, "%s/%s%s", dpath,
51 iopc_path_basename(pkg->name), ext);
52 }
53 return res;
54}

56int iopc_write_file(const sb_t *buf, const char *path)
57{
58 if (unlink(path) < 0 && errno != ENOENT) {
59 throw_error("unable to remove existing file `%s`: %m", path);
60 }
61 if (sb_write_file(buf, path) < 0 || chmod(path, 0444) < 0) {
62 PROTECT_ERRNO(unlink(path));
63 throw_error("unable to write file `%s`: %m", path);
64 }
65 return 0;
66}
