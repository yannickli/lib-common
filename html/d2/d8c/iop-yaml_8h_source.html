28/** Convert IOP-YAML to an IOP C structure using the t_pool().
29 *
30 * This function allow to unpack an IOP structure encoded in YAML format in
31 * one call. This is equivalent to:
32 *
33 * This function cannot be used to unpack a class; use `t_iop_yunpack_ptr_ps`
34 * instead.
35 *
36 * \param[in] ps The pstream_t to parse.
37 * \param[in] st The IOP structure description.
38 * \param[out] out Pointer on the IOP structure to write.
39 * \param[in] flags Bitfield of iop_unpack_flags_t elements. See
40 * documentation of iop_unpack_flags_t to find out which
41 * flags are handled.
42 * \param[out] pres If non NULL, it will be set to YAML presentation data
43 * of the parsed YAML. See yaml.h for details.
44 * \param[out] err If the unpacking fails, this pointer is set to a
45 * description of the error, allocated on the t_scope.
46 */
47__must_check__
48int t_iop_yunpack_ps(
49 pstream_t * nonnull ps, const iop_struct_t * nonnull st,
50 void * nonnull out, unsigned flags,
51 yaml__document_presentation__t * nonnull * nullable pres,
52 sb_t * nonnull out_err
53);

174/** Convert JSON subfiles to a YAML document presentation.
175 *
176 * This helper can be used to convert an IOP-JSON document into an IOP-YAML
177 * one. IOP-JSON only handles includes as "presentation data". This function
178 * can be used to convert those includes into a YAML document presentation,
179 * so that the repacked YAML data will keep the same includes and subfiles.
180 *
181 * If type IOP struct definition that is referenced by the IOP path is
182 * provided with \p st, the generated presentation will use proper includes:
183 * includeraw for strings, and include for the rest.
184 */

190);
