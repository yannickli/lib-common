258static lstr_t
259t_iop_help_to_string(const iop_help_t * nonnull help, bool ignore_brief)
260{
261 t_SB_1k(sb);
262
263 if (help->brief.s && !ignore_brief) {
264 sb_add_lstr(&sb, help->brief);
265 }
266 if (help->details.s) {
267 if (sb.len > 0) {
268 sb_adds(&sb, "\n\n");
269 }
270 sb_add_lstr(&sb, help->details);
271 }
272 if (help->warning.s) {
273 if (sb.len > 0) {
274 sb_adds(&sb, "\n\n");
275 }
276 sb_add_lstr(&sb, help->warning);
277 }
278
279 return sb.len > 0 ? LSTR_SB_V(&sb) : LSTR_NULL_V;

286static void
287get_enum_help(const iop_enum_t * nonnull en,
288 const iop_help_t * nullable * nonnull help,
289 bool * nonnull is_v2)
290{
291 *help = NULL;
292
293 if (!TST_BIT(&en->flags, IOP_ENUM_EXTENDED) || !en->en_attrs) {
294 return;
295 }
296
297 for (int k = 0; k < en->en_attrs->attrs_len; k++) {
298 const iop_enum_attr_t *attr = &en->en_attrs->attrs[k];
299
300 switch (attr->type) {
301 case IOP_ENUM_ATTR_HELP:
302 *help = attr->args[0].v.p;
303 *is_v2 = false;
304 return;
305 case IOP_ENUM_ATTR_HELP_V2:
306 *help = attr->args[0].v.p;
307 *is_v2 = true;
308 return;
309 default:
310 break;
311 }
312 }
313}

486static void
487t_class_field_to_schema_object(const iop_struct_t *class_desc,
488 qh_t(schemas) *existing_schemas,
489 qv_t(schema_object) *schemas,
490 schema_object_t *out)
491{
492 /* When using a class, it is not possible to point only to the class
493 * schema directly, as any of the child is a valid response as well.
494 * To handle this, the schema for the field must be a oneOf of all
495 * possible children (excluding abstract classes).
496 */
497 qv_t(iop_struct) children;
498 qv_t(iop_struct) *_children_p = &children;
499
500 out->type = TYPE_NONE;
501 out->discriminator = LSTR("_class");
502 t_qv_init(&out->one_of, 0);
503
504 /* TODO: this is extremely inefficient, we go through every existing
505 * class, and then look through all their inheritance chain to find the
506 * children of a class. This should be at the very least cached per class.
507 */
508
509 /* we need to sort the classes, as the order of the registered classes is
510 * not deterministic, breaking our tests. So a temp vector is used. */
511 t_qv_init(&children, 0);
512
513 /* add all children to the schemas, and build a list of them */
514 iop_for_each_registered_classes(^(const iop_struct_t *st) {
515 if (iop_class_is_a(st, class_desc) && !st->class_attrs->is_private &&
516 !st->class_attrs->is_abstract)
517 {
518 qv_t(iop_struct) *children_p = _children_p;
519
520 qv_append(children_p, st);
521 }
522 });
523
524 qv_sort(iop_struct)(&children,
525 ^int (const iop_struct_t * const *st1,
526 const iop_struct_t * const *st2)
527 {
528 return lstr_cmp((*st1)->fullname, (*st2)->fullname);
529 });
530
531 tab_for_each_entry(st, &children) {
532 schema_object_t *obj;
533
534 t_iop_struct_to_schema_object(st, existing_schemas, schemas);
535 obj = t_new(schema_object_t, 1);
536 obj->type = TYPE_REF;
537 obj->name = st->fullname;
538 qv_append(&out->one_of, obj);
539 }
540}

583static bool
584apply_raw_field_constraints(const iop_field_attrs_t *attrs,
585 const iop_field_t *field,
586 schema_object_t *schema)
587{
588 bool did_update = false;
589
590 for (int i = 0; i < attrs->attrs_len; i++) {
591 const iop_field_attr_t *attr = &attrs->attrs[i];
592
593 switch (attr->type) {
594 case IOP_FIELD_MIN:
595 did_update = true;
596 schema->minimum = t_attr_get_value(attr, field->type);
597 break;
598 case IOP_FIELD_MAX:
599 did_update = true;
600 schema->maximum = t_attr_get_value(attr, field->type);
601 break;
602 case IOP_FIELD_MIN_LENGTH:
603 did_update = true;
604 OPT_SET(schema->min_length, attr->args[0].v.i64);
605 break;
606 case IOP_FIELD_MAX_LENGTH:
607 did_update = true;
608 OPT_SET(schema->max_length, attr->args[0].v.i64);
609 break;
610 case IOP_FIELD_PATTERN:
611 did_update = true;
612 schema->pattern = attr->args[0].v.s;
613 break;
614 default:
615 break;
616 }
617 }
618
619 if (TST_BIT(&attrs->flags, IOP_FIELD_NON_ZERO)) {
620 /* only apply this constraint on unsigned integers, we have no
621 * way of expressing it otherwise */
622 if (schema->minimum
623 && (schema->minimum->type == SCALAR_INT64
624 || schema->minimum->type == SCALAR_UINT64)
625 && schema->minimum->i == 0)
626 {
627 did_update = true;
628 schema->minimum->i = 1;
629 }
630 }
631 if (TST_BIT(&attrs->flags, IOP_FIELD_NON_EMPTY)
632 && !OPT_ISSET(schema->min_length))
633 {
634 did_update = true;
635 OPT_SET(schema->min_length, 1);
636 }
637
638 return did_update;
639}

663t_get_iop_field_defval(const iop_field_t *desc)
664{
665 scalar_value_t *val = t_new(scalar_value_t, 1);
666
667 switch (desc->type) {
668 case IOP_T_I8: case IOP_T_I16: case IOP_T_I32: case IOP_T_I64:
669 val->i = (int64_t)desc->u1.defval_u64;
670 val->type = SCALAR_INT64;
671 break;
672
673 case IOP_T_U8: case IOP_T_U16: case IOP_T_U32: case IOP_T_U64:
674 val->u = desc->u1.defval_u64;
675 val->type = SCALAR_UINT64;
676 break;
677
678 case IOP_T_BOOL:
679 val->b = !!desc->u1.defval_u64;
680 val->type = SCALAR_BOOL;
681 break;
682
683 case IOP_T_DOUBLE:
684 val->d = desc->u1.defval_d;
685 val->type = SCALAR_DOUBLE;
686 break;
687
688 case IOP_T_DATA: case IOP_T_STRING: case IOP_T_XML:
689 val->s = LSTR_INIT_V(desc->u1.defval_data, desc->u0.defval_len);
690 val->type = SCALAR_STRING;
691 break;
692
693 case IOP_T_ENUM:
694 /* FIXME: use string repr, as we type ENUM as string in the
695 * schema */
696 val->i = desc->u0.defval_enum;
697 val->type = SCALAR_INT64;
698 break;
699
700 case IOP_T_VOID:
701 case IOP_T_UNION:
702 case IOP_T_STRUCT:
703 assert (false);
704 return NULL;
705 }
706
707 return val;
708}

710static schema_object_t *
711t_iop_field_to_schema_object(const iop_struct_t *st, const iop_field_t *desc,
712 qh_t(schemas) *existing_schemas,
713 qv_t(schema_object) *schemas)
714{
715 const iop_field_attrs_t *attrs = iop_field_get_attrs(st, desc);
716 schema_object_t *schema;
717 const iop_help_t *help;
718 bool is_v2 = false;
719 bool has_update = false;
720
721 if (attrs && TST_BIT(&attrs->flags, IOP_FIELD_PRIVATE)) {
722 return NULL;
723 }
724
725 schema = t_new(schema_object_t, 1);
726
727 switch (desc->type) {
728 case IOP_T_I8:
729 t_schema_set_integer(schema, TYPE_INT32, INT8_MIN, INT8_MAX);
730 break;
731 case IOP_T_U8:
732 t_schema_set_integer(schema, TYPE_UINT32, 0, UINT8_MAX);
733 break;
734 case IOP_T_I16:
735 t_schema_set_integer(schema, TYPE_INT32, INT16_MIN, INT16_MAX);
736 break;
737 case IOP_T_U16:
738 t_schema_set_integer(schema, TYPE_UINT32, 0, UINT16_MAX);
739 break;
740 case IOP_T_I32:
741 schema->type = TYPE_INT32;
742 break;
743 case IOP_T_U32:
744 schema->type = TYPE_UINT32;
745 schema->minimum = t_new(scalar_value_t, 1);
746 schema->minimum->u = 0;
747 schema->minimum->type = SCALAR_UINT64;
748 break;
749 case IOP_T_I64:
750 schema->type = TYPE_INT64;
751 break;
752 case IOP_T_U64:
753 schema->type = TYPE_UINT64;
754 schema->minimum = t_new(scalar_value_t, 1);
755 schema->minimum->u = 0;
756 schema->minimum->type = SCALAR_UINT64;
757 break;
758 case IOP_T_BOOL:
759 schema->type = TYPE_BOOL;
760 break;
761 case IOP_T_DOUBLE:
762 schema->type = TYPE_DOUBLE;
763 break;
764 case IOP_T_VOID:
765 schema->type = TYPE_BOOL;
766 schema->_nullable = true;
767 break;
768 case IOP_T_DATA:
769 schema->type = TYPE_BYTE;
770 break;
771 case IOP_T_STRING:
772 case IOP_T_XML:
773 schema->type = TYPE_STRING;
774 break;
775
776 case IOP_T_ENUM:
777 t_iop_enum_to_schema_object(desc->u1.en_desc, existing_schemas,
778 schemas);
779 schema->type = TYPE_REF;
780 schema->name = desc->u1.en_desc->fullname;
781 break;
782
783 case IOP_T_UNION:
784 case IOP_T_STRUCT:
785 if (iop_struct_is_class(desc->u1.st_desc)) {
786 /* a class field needs to be expressed as a oneOf of all children.
787 */
788 t_class_field_to_schema_object(desc->u1.st_desc, existing_schemas,
789 schemas, schema);
790 } else {
791 t_iop_struct_to_schema_object(desc->u1.st_desc, existing_schemas,
792 schemas);
793 schema->type = TYPE_REF;
794 schema->name = desc->u1.st_desc->fullname;
795 }
796 break;
797 }
798
799 if (attrs) {
800 has_update |= apply_raw_field_constraints(attrs, desc, schema);
801 /* FIXME: must transform the schema into an allOf if it is a REF
802 * and has_update is true */
803 }
804
805 switch (desc->repeat) {
806 case IOP_R_REPEATED: {
807 schema_object_t *array_schema;
808
809 array_schema = t_new(schema_object_t, 1);
810 array_schema->type = TYPE_ARRAY;
811 array_schema->items = schema;
812 schema = array_schema;
813
814 if (attrs) {
815 apply_array_field_constraints(attrs, desc, schema);
816 }
817 } break;
818 case IOP_R_DEFVAL:
819 has_update = true;
820 schema->defval = t_get_iop_field_defval(desc);
821 break;
822 default:
823 break;
824 }
825
826 get_field_help(attrs, &help, &is_v2);
827 if (help) {
828 has_update = true;
829 schema->description = t_iop_help_to_string(help, false);
830 if (is_v2) {
831 schema->example = help->example;
832 }
833 }
834
835 if (has_update && schema->type == TYPE_REF) {
836 /* We cannot specify any fields with the REF type... so
837 * a allOf must be used to split the ref from the details */
838 schema_object_t *ref;
839
840 t_qv_init(&schema->all_of, 1);
841 ref = t_new(schema_object_t, 1);
842 ref->type = TYPE_REF;
843 ref->name = schema->name;
844 qv_append(&schema->all_of, ref);
845 schema->type = TYPE_NONE;
846 schema->name = LSTR_NULL_V;
847 }
848
849 return schema;
850}

852static bool field_is_required(const iop_field_t *desc,
853 const schema_object_t *schema)
854{
855 switch (desc->repeat) {
856 case IOP_R_OPTIONAL:
857 case IOP_R_DEFVAL:
858 return false;
859 case IOP_R_REPEATED:
860 return OPT_DEFVAL(schema->min_items, 0) > 0;
861 case IOP_R_REQUIRED:
862 return desc->type != IOP_T_STRUCT
863 || !iop_struct_is_optional(desc->u1.st_desc, true);
864 }
865
866 assert (false);
867 return false;
868}

870static void
871t_iop_union_to_schema_object(const iop_struct_t *st,
872 qh_t(schemas) *existing_schemas,
873 qv_t(schema_object) *schemas)
874{
875 schema_object_t *obj;
876 qv_t(schema_object_p) one_of;
877
878 t_qv_init(&one_of, st->fields_len);
879
880 iop_struct_for_each_field(field_desc, field_st, st) {
881 schema_object_t *field_obj;
882 schema_object_t *field_schema;
883 schema_prop_t *prop;
884
885 field_schema = t_iop_field_to_schema_object(st, field_desc,
886 existing_schemas,
887 schemas);
888 if (!field_schema) {
889 continue;
890 }
891
892 /* create an anonymous schema for an object that only contains this
893 * field. The union is then a "oneOf" of all those objects */
894 field_obj = t_new(schema_object_t, 1);
895 field_obj->type = TYPE_OBJECT;
896 t_qv_init(&field_obj->required, 1);
897 qv_append(&field_obj->required, field_desc->name);
898 t_qv_init(&field_obj->properties, 1);
899
900 prop = qv_growlen0(&field_obj->properties, 1);
901 prop->field_name = field_desc->name;
902 prop->schema = field_schema;
903
904 qv_append(&one_of, field_obj);
905 }
906
907 obj = add_schema_object(st->fullname, TYPE_NONE, schemas);
908 obj->one_of = one_of;
909 add_struct_help(st, obj);
910}

912static void
913t_iop_struct_to_schema_object(const iop_struct_t *st,
914 qh_t(schemas) *existing_schemas,
915 qv_t(schema_object) *schemas)
916{
917 schema_object_t *obj;
918
919 if (qh_add(schemas, existing_schemas, &st->fullname) < 0) {
920 return;
921 }
922
923 if (st->is_union) {
924 t_iop_union_to_schema_object(st, existing_schemas, schemas);
925 return;
926 }
927
928 /* we should never add private classes */
929 assert (!iop_struct_is_class(st) || !st->class_attrs->is_private);
930
931 obj = t_new(schema_object_t, 1);
932 obj->type = TYPE_OBJECT;
933 add_struct_help(st, obj);
934 t_qv_init(&obj->required, st->fields_len);
935 t_qv_init(&obj->properties, st->fields_len + 1);
936
937 if (iop_struct_is_class(st) && !st->class_attrs->parent) {
938 schema_prop_t *class_prop = qv_growlen0(&obj->properties, 1);
939
940 /* Class without ancestors: add the _class field, and set it as the
941 * discriminator field */
942 class_prop->field_name = LSTR("_class");
943 class_prop->schema = t_new(schema_object_t, 1);
944 class_prop->schema->type = TYPE_STRING;
945 obj->discriminator = class_prop->field_name;
946 }
947
948 for (int i = 0; i < st->fields_len; i++) {
949 const iop_field_t *field_desc = &st->fields[i];
950 schema_object_t *field_schema;
951 schema_prop_t *prop;
952
953 field_schema = t_iop_field_to_schema_object(st, field_desc,
954 existing_schemas,
955 schemas);
956 if (!field_schema) {
957 continue;
958 }
959
960 prop = qv_growlen0(&obj->properties, 1);
961 prop->field_name = field_desc->name;
962 prop->schema = field_schema;
963
964 if (field_is_required(field_desc, field_schema)) {
965 qv_append(&obj->required, prop->field_name);
966 }
967 }
968
969 if (iop_struct_is_class(st) && st->class_attrs->parent) {
970 const iop_struct_t *parent_st = st->class_attrs->parent;
971 schema_object_t *all;
972 schema_object_t *parent;
973
974 t_iop_struct_to_schema_object(parent_st, existing_schemas, schemas);
975
976 /* the class is a allOf of first the parent, then the details of
977 * the current struct */
978 all = add_schema_object(st->fullname, TYPE_NONE, schemas);
979 t_qv_init(&all->all_of, 2);
980
981 parent = t_new(schema_object_t, 1);
982 parent->type = TYPE_REF;
983 parent->name = parent_st->fullname;
984 qv_append(&all->all_of, parent);
985 qv_append(&all->all_of, obj);
986 } else {
987 obj->name = st->fullname;
988 qv_append(schemas, *obj);
989 }
990}

1051static void
1052t_schema_object_to_yaml(const schema_object_t * nonnull obj,
1053 yaml_data_t * nonnull out)
1054{
1055 static lstr_t types[] = {
1056 [TYPE_NONE] = LSTR_NULL,
1057 [TYPE_REF] = LSTR_NULL,
1058 [TYPE_ARRAY] = LSTR_IMMED("array"),
1059 [TYPE_OBJECT] = LSTR_IMMED("object"),
1060 [TYPE_STRING] = LSTR_IMMED("string"),
1061 [TYPE_BYTE] = LSTR_IMMED("string"),
1062 [TYPE_BOOL] = LSTR_IMMED("boolean"),
1063 [TYPE_DOUBLE] = LSTR_IMMED("number"),
1064 [TYPE_INT32] = LSTR_IMMED("integer"),
1065 [TYPE_INT64] = LSTR_IMMED("integer"),
1066 [TYPE_UINT32] = LSTR_IMMED("integer"),
1067 [TYPE_UINT64] = LSTR_IMMED("integer"),
1068 };
1069 static lstr_t formats[] = {
1070 [TYPE_NONE] = LSTR_NULL,
1071 [TYPE_REF] = LSTR_NULL,
1072 [TYPE_ARRAY] = LSTR_NULL,
1073 [TYPE_OBJECT] = LSTR_NULL,
1074 [TYPE_STRING] = LSTR_NULL,
1075 [TYPE_BYTE] = LSTR_IMMED("byte"),
1076 [TYPE_BOOL] = LSTR_NULL,
1077 [TYPE_DOUBLE] = LSTR_IMMED("double"),
1078 [TYPE_INT32] = LSTR_IMMED("int32"),
1079 [TYPE_INT64] = LSTR_IMMED("int64"),
1080 [TYPE_UINT32] = LSTR_IMMED("uint32"),
1081 [TYPE_UINT64] = LSTR_IMMED("uint64"),
1082 };
1083 yaml_data_t data;
1084
1085 t_yaml_data_new_obj(out, 2);
1086
1087 if (obj->type == TYPE_REF) {
1088 yaml_data_set_string(&data, t_lstr_fmt("#/components/schemas/%pL",
1089 &obj->name));
1090 yaml_obj_add_field(out, LSTR("$ref"), data);
1091 return;
1092 }
1093
1094 if (types[obj->type].s) {
1095 yaml_data_set_string(&data, types[obj->type]);
1096 yaml_obj_add_field(out, LSTR("type"), data);
1097 }
1098
1099 if (formats[obj->type].s) {
1100 yaml_data_set_string(&data, formats[obj->type]);
1101 yaml_obj_add_field(out, LSTR("format"), data);
1102 }
1103
1104 if (obj->description.s) {
1105 yaml_data_set_string(&data, obj->description);
1106 yaml_obj_add_field(out, LSTR("description"), data);
1107 }
1108
1109 if (obj->items) {
1110 t_schema_object_to_yaml(obj->items, &data);
1111 yaml_obj_add_field(out, LSTR("items"), data);
1112 }
1113
1114 if (obj->required.len > 0) {
1115 yaml_obj_add_field(out, LSTR("required"),
1116 t_strings_to_yaml_seq(&obj->required));
1117 }
1118
1119 if (obj->properties.len > 0) {
1120 t_yaml_data_new_obj(&data, obj->properties.len);
1121 tab_for_each_ptr(prop, &obj->properties) {
1122 yaml_data_t elem;
1123
1124 t_schema_object_to_yaml(prop->schema, &elem);
1125 yaml_obj_add_field(&data, prop->field_name, elem);
1126 }
1127 yaml_obj_add_field(out, LSTR("properties"), data);
1128 }
1129
1130 if (obj->one_of.len > 0) {
1131 yaml_obj_add_field(out, LSTR("oneOf"),
1132 t_schemas_to_yaml_seq(&obj->one_of));
1133 }
1134
1135 if (obj->all_of.len > 0) {
1136 yaml_obj_add_field(out, LSTR("allOf"),
1137 t_schemas_to_yaml_seq(&obj->all_of));
1138 }
1139
1140 if (obj->discriminator.s) {
1141 yaml_data_t elem;
1142
1143 t_yaml_data_new_obj(&data, 1);
1144 yaml_data_set_string(&elem, obj->discriminator);
1145 yaml_obj_add_field(&data, LSTR("propertyName"), elem);
1146 yaml_obj_add_field(out, LSTR("discriminator"), data);
1147 }
1148
1149 if (obj->enum_values.len > 0) {
1150 yaml_obj_add_field(out, LSTR("enum"),
1151 t_strings_to_yaml_seq(&obj->enum_values));
1152 }
1153
1154 if (obj->defval) {
1155 number_value_to_yaml_data(obj->defval, &data);
1156 yaml_obj_add_field(out, LSTR("default"), data);
1157 }
1158
1159 if (obj->minimum) {
1160 number_value_to_yaml_data(obj->minimum, &data);
1161 yaml_obj_add_field(out, LSTR("minimum"), data);
1162 }
1163
1164 if (obj->maximum) {
1165 number_value_to_yaml_data(obj->maximum, &data);
1166 yaml_obj_add_field(out, LSTR("maximum"), data);
1167 }
1168
1169 if (obj->_nullable) {
1170 yaml_data_set_bool(&data, true);
1171 yaml_obj_add_field(out, LSTR("nullable"), data);
1172 }
1173
1174 if (OPT_ISSET(obj->min_length)) {
1175 yaml_data_set_int(&data, OPT_VAL(obj->min_length));
1176 yaml_obj_add_field(out, LSTR("minLength"), data);
1177 }
1178
1179 if (OPT_ISSET(obj->max_length)) {
1180 yaml_data_set_int(&data, OPT_VAL(obj->max_length));
1181 yaml_obj_add_field(out, LSTR("maxLength"), data);
1182 }
1183
1184 if (obj->pattern.s) {
1185 yaml_data_set_string(&data, obj->pattern);
1186 yaml_obj_add_field(out, LSTR("pattern"), data);
1187 }
1188
1189 if (OPT_ISSET(obj->min_items)) {
1190 yaml_data_set_int(&data, OPT_VAL(obj->min_items));
1191 yaml_obj_add_field(out, LSTR("minItems"), data);
1192 }
1193
1194 if (OPT_ISSET(obj->max_items)) {
1195 yaml_data_set_int(&data, OPT_VAL(obj->max_items));
1196 yaml_obj_add_field(out, LSTR("maxItems"), data);
1197 }
1198
1199 if (obj->example.s) {
1200 yaml_data_set_string(&data, obj->example);
1201 yaml_obj_add_field(out, LSTR("example"), data);
1202 }
1203
1204}

1567static void
1568t_iop_iface_alias_add(iop_openapi_t * nonnull oa,
1569 const iop_mod_t * nonnull mod,
1570 const iop_iface_alias_t * nonnull alias)
1571{
1572 tag_object_t *tag;
1573 const iop_help_t *help;
1574 t_SB_1k(sb);
1575 bool has_rpcs = false;
1576
1577 for (int j = 0; j < alias->iface->funs_len; j++) {
1578 const iop_rpc_t *rpc = &alias->iface->funs[j];
1579 const unsigned rpc_flags = rpc->flags;
1580
1581 /* ignore rpc, will be taken into account in the following HAS_ALIAS
1582 * loop */
1583 if (TST_BIT(&rpc_flags, IOP_RPC_IS_ALIAS)) {
1584 continue;
1585 }
1586
1587 has_rpcs |= t_iop_rpc_add(oa, rpc, alias);
1588
1589 /* loop on rpc aliases */
1590 if (TST_BIT(&rpc_flags, IOP_RPC_HAS_ALIAS)) {
1591 const iop_rpc_attrs_t *attrs;
1592
1593 attrs = iop_rpc_get_attrs(alias->iface, rpc);
1594 for (int k = 0; k < attrs->attrs_len; k++) {
1595 const iop_rpc_attr_t attr = attrs->attrs[k];
1596
1597 if (attr.type == IOP_RPC_ALIAS) {
1598 const iop_rpc_attr_arg_t arg = attr.args[0];
1599 const iop_rpc_t *falias = (const iop_rpc_t *)arg.v.p;
1600
1601 has_rpcs |= t_iop_rpc_add(oa, falias, alias);
1602 }
1603 }
1604 }
1605 }
1606
1607 if (!has_rpcs) {
1608 /* no RPCs have been added (probably because none was whitelisted).
1609 * do not add a unused tag in that case which could leak info about
1610 * an unexposed interface. */
1611 return;
1612 }
1613
1614 /* add a tag for the iface */
1615 tag = qv_growlen0(&oa->tags, 1);
1616 tag->name = alias->name;
1617
1618 /* Get description from both the iface alias in the module, and the iface
1619 * itself.
1620 */
1621 get_mod_iface_help(mod, alias, &help);
1622 if (help) {
1623 sb_add_lstr(&sb, t_iop_help_to_string(help, false));
1624 }
1625

1627 if (help) {

1629 sb_adds(&sb, "\n\n");
1630 }
1631 sb_add_lstr(&sb, t_iop_help_to_string(help, false));
1632 }
