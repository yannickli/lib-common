32static lstr_t t_farch_aggregate(const farch_entry_t *entry)
33{
34 char *contents = t_new_raw(char, entry->compressed_size);
35 char *tail = contents;
36 int compressed_size = 0;
37
38 for (int i = 0; i < entry->nb_chunks; i++) {
39 lstr_t chunk = entry->chunks[i];
40 lstr_t content_chunk = LSTR_INIT(tail, chunk.len);
41
42 compressed_size += chunk.len;
43 if (!expect(compressed_size <= entry->compressed_size)) {
44 return LSTR_NULL_V;
45 }
46 lstr_unobfuscate(chunk, chunk.len, content_chunk);
47 tail += chunk.len;
48 }
49
50 if (!expect(compressed_size == entry->compressed_size)) {
51 return LSTR_NULL_V;
52 }
53
54 return LSTR_INIT_V(contents, entry->compressed_size);
55}

57char *farch_get_filename(const farch_entry_t *entry, char *name)
58{
59 lstr_t out = LSTR_INIT(name, entry->name.len);
60
61 if (!entry->name.s) {
62 return NULL;
63 }
64 lstr_unobfuscate(entry->name, entry->nb_chunks, out);
65 name[entry->name.len] = '\0';
66 return name;
67}

69/** Get compressed farch entry.
70 *
71 * Get a farch entry by its name in a farch archive.
72 *
73 * \return a pointer on the (compressed) farch entry if found,
74 * NULL otherwise.
75 */
76static const farch_entry_t *farch_get_entry(const farch_entry_t files[],
77 const char *name)
78{
79 char real_name[FARCH_MAX_FILENAME];
80 int len = strlen(name);
81
82 for (; files->name.len > 0; files++) {
83 if (len == files->name.len
84 && strequal(farch_get_filename(files, real_name), name))
85 {
86 return files;
87 }
88 }
89 return NULL;
90}

92/** Unarchive and append a '\0' if needed. */
93lstr_t t_farch_unarchive(const farch_entry_t *entry)
94{
95 lstr_t res;
96 char real_name[FARCH_MAX_FILENAME];
97
98 res = LSTR_INIT_V(t_new_raw(char, entry->size + 1), entry->size + 1);
99
100 do {
101 t_scope;
102 lstr_t contents;
103
104 contents = t_farch_aggregate(entry); /* (and unobfuscate) */
105 if (!contents.s) {
106 goto error;
107 }
108 if (entry->compressed_size == entry->size) {
109 memcpy(res.v, contents.s, entry->size);
110 res.len = entry->size;
111 } else { /* uncompress */
112 res.len = qlzo1x_decompress_safe(res.v, entry->size,
113 ps_initlstr(&contents));
114 if (res.len != entry->size) {
115 goto error;
116 }
117 }
118 } while (0);
119
120 res.v[res.len] = '\0';
121
122 return res;
123
124 error:
125 lstr_unobfuscate(entry->name, entry->nb_chunks,
126 LSTR_INIT_V(real_name, entry->name.len));
127 real_name[entry->name.len] = '\0';
128 e_panic("cannot uncompress farch entry `%s`", real_name);
129}

131lstr_t farch_unarchive_persist(const farch_entry_t * nonnull entry)
132{
133 t_scope;
134 lstr_t content;
135 uint32_t pos;
136
137 assert (MODULE_IS_LOADED(farch));
138
139 pos = qm_reserve(persisted, &_G.persisted, entry, 0);
140 if (pos & QHASH_COLLISION) {
141 return _G.persisted.values[pos ^ QHASH_COLLISION];
142 }
143
144 content = t_farch_unarchive(entry);
145 lstr_persists(&content);
146 _G.persisted.values[pos] = content;
147
148 return content;
149}
