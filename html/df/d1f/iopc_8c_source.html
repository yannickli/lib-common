84static void parse_incpath(qv_t(cstr) *ipath, char *spec)
85{
86 if (!spec)
87 return;
88 for (;;) {
89 if (!*spec) {
90 qv_append(ipath, ".");
91 return;
92 }
93 if (*spec == ':') {
94 qv_append(ipath, ".");
95 } else {
96 qv_append(ipath, spec);
97 spec = strchr(spec, ':');
98 if (!spec)
99 break;
100 }
101 *spec++ = '\0';
102 }
103 for (int i = ipath->len - 1; i >= 0; --i) {
104 struct stat st;
105
106 if (stat(ipath->tab[i], &st) || !S_ISDIR(st.st_mode)) {
107 qv_remove(ipath, i);
108 }
109 }
110}

112static int parse_class_id_range(const char *class_id_range)
113{
114 pstream_t ps = ps_initstr(class_id_range);
115
116 if (!ps_len(&ps))
117 return 0;
118
119 /* Format: "min-max" */
120 iopc_g.class_id_min = ps_geti(&ps);
121 THROW_ERR_IF(iopc_g.class_id_min < 0 || iopc_g.class_id_min > 0xFFFF);
122
123 RETHROW(ps_skipc(&ps, '-'));
124
125 iopc_g.class_id_max = ps_geti(&ps);
126 THROW_ERR_IF(iopc_g.class_id_max < iopc_g.class_id_min
127 || iopc_g.class_id_max > 0xFFFF);
128
129 return 0;
130}

140static int build_doit_table(qv_t(doit) *doits)
141{
142 qv_t(lstr) langs;
143 ctype_desc_t sep;
144
145 if (!opts.lang) {
146 /* No language specified; this is authorized in case of --depends. */
147 if (!opts.depends) {
148 print_error("no language specified");
149 return -1;
150 }
151 return 0;
152 }
153
154 qv_inita(&langs, 2);
155 ctype_desc_build(&sep, ",");
156 ps_split(ps_initstr(opts.lang), &sep, 0, &langs);
157
158 tab_for_each_entry(lang, &langs) {
159 struct doit doit;
160
161 if (lstr_ascii_iequal(lang, LSTR("c"))) {
162 doit = (struct doit){
163 .cb = &iopc_do_c,
164 .outpath = opts.c_outpath
165 };
166 } else
167 if (lstr_ascii_iequal(lang, LSTR("json"))) {
168 doit = (struct doit){
169 .cb = &iopc_do_json,
170 .outpath = opts.json_outpath
171 };
172 } else
173 if (lstr_ascii_iequal(lang, LSTR("typescript"))) {
174 doit = (struct doit){
175 .cb = &iopc_do_typescript,
176 .outpath = opts.typescript_outpath
177 };
178 } else {
179 print_error("unsupported language `%*pM`", LSTR_FMT_ARG(lang));
180 goto error;
181 }
182
183 if (doit.outpath) {
184 if (mkdir_p(doit.outpath, 0777) < 0) {
185 print_error("cannot create output directory `%s`: %m",
186 doit.outpath);
187 goto error;
188 }
189 }
190 qv_append(doits, doit);
191 }
192
193 qv_wipe(&langs);
194 return 0;
195
196 error:
197 qv_wipe(&langs);
198 return -1;
199}

226int main(int argc, char **argv)
227{
228 const char *arg0 = NEXTARG(argc, argv);
229 qv_t(cstr) incpath;
230 qv_t(doit) doits;
231 iopc_pkg_t *pkg;
232 SB_8k(deps);
233
234 setlinebuf(stderr);
235 argc = parseopt(argc, argv, options, 0);
236 if (argc < 0 || opts.help) {
237 makeusage(!opts.help, arg0, "<iop file>", NULL, options);
238 }
239 if (opts.version) {
240 printf("%s\n", libcommon_git_revision);
241 return 0;
242 }
243
244 MODULE_REQUIRE(iopc);
245
246 qv_init(&incpath);
247 qv_init(&doits);
248
249 opts.c_outpath = opts.c_outpath ?: opts.outpath;
250 opts.json_outpath = opts.json_outpath ?: opts.outpath;
251
252 _G.prefix_dir = getcwd(NULL, 0);
253 _G.display_prefix = true;
254
255 log_set_handler(&iopc_log_handler);
256
257 if (opts.c_outpath && iopc_do_c_g.resolve_includes) {
258 print_error("outdir and --c-resolve-includes are incompatible");
259 goto error;
260 }
261
262 if (build_doit_table(&doits) < 0) {
263 goto error;
264 }
265
266 parse_incpath(&incpath, opts.incpath);
267
268 if (opts.class_id_range && parse_class_id_range(opts.class_id_range) < 0)
269 {
270 print_error("invalid class-id-range `%s`", opts.class_id_range);
271 goto error;
272 }
273
274 if (argc == 0) {
275 argc = 1;
276 argv = (char **)only_stdin;
277 }
278 for (int i = 0; i < argc; i++) {
279 iopc_parser_typer_initialize();
280
281 if (!(pkg = iopc_parse_file(&incpath, NULL, argv[i], NULL, true))) {
282 iopc_parser_typer_shutdown();
283 goto error;
284 }
285 if (iopc_resolve(pkg) < 0) {
286 iopc_parser_typer_shutdown();
287 goto error;
288 }
289 if (iopc_resolve_second_pass(pkg) < 0) {
290 iopc_parser_typer_shutdown();
291 goto error;
292 }
293
294 iopc_types_fold(pkg);
295
296 tab_for_each_ptr(doit, &doits) {
297 if ((*doit->cb)(pkg, doit->outpath) < 0) {
298 iopc_parser_typer_shutdown();
299 goto error;
300 }
301 }
302
303 if (opts.depends) {
304 sb_add_depends(pkg, &deps);
305 }
306
307 iopc_parser_typer_shutdown();
308 }
309
310 if (opts.depends) {
311 char dir[PATH_MAX];
312
313 path_dirname(dir, sizeof(dir), opts.depends);
314 if (mkdir_p(dir, 0755) < 0) {
315 print_error("cannot create directory `%s`: %m", dir);
316 goto error;
317 }
318 if (sb_write_file(&deps, opts.depends) < 0) {
319 print_error("cannot write file `%s`: %m", opts.depends);
320 goto error;
321 }
322 }
323
324 MODULE_RELEASE(iopc);
325 qv_wipe(&incpath);
326 qv_wipe(&doits);
327 p_delete((char **)&_G.prefix_dir);
328 return 0;
329
330 error:
331 MODULE_RELEASE(iopc);
332 qv_wipe(&incpath);
333 qv_wipe(&doits);
334 p_delete((char **)&_G.prefix_dir);
335 return -1;
336}
