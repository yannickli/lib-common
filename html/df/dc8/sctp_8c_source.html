23int sctp_enable_events(int sd, int flags)
24{
25 struct sctp_event_subscribe ev;
26
27 p_clear(&ev, 1);
28
29 if (flags & SCTP_DATA_IO_EV)
30 ev.sctp_data_io_event = 1;
31
32 if (flags & SCTP_ASSOCIATION_EV)
33 ev.sctp_association_event = 1;
34
35 if (flags & SCTP_ADDRESS_EV)
36 ev.sctp_address_event = 1;
37
38 if (flags & SCTP_SEND_FAILURE_EV)
39 ev.sctp_send_failure_event = 1;
40
41 if (flags & SCTP_PEER_ERROR_EV)
42 ev.sctp_peer_error_event = 1;
43
44 if (flags & SCTP_SHUTDOWN_EV)
45 ev.sctp_shutdown_event = 1;
46
47 if (flags & SCTP_PARTIAL_DELIVERY_EV)
48 ev.sctp_partial_delivery_event = 1;
49
50 if (flags & SCTP_ADAPTATION_LAYER_EV)
51 ev.sctp_adaptation_layer_event = 1;
52
53 return setsockopt(sd, IPPROTO_SCTP, SCTP_EVENTS, &ev, sizeof(ev));

56ssize_t sctp_sendv(int sd, const struct iovec *iov, int iovlen,
57 const struct sctp_sndrcvinfo *sinfo, int flags)
58{
59 char buf[CMSG_SPACE(sizeof(*sinfo))];
60 struct msghdr msg = {
61 .msg_iov = (struct iovec *)iov,
62 .msg_iovlen = iovlen,
63 .msg_control = buf,
64 .msg_controllen = sizeof(buf),
65 .msg_flags = 0,
66 };
67
68 struct cmsghdr *cmsg = CMSG_FIRSTHDR(&msg);
69
70 cmsg->cmsg_level = IPPROTO_SCTP;
71 cmsg->cmsg_type = SCTP_SNDRCV;
72 cmsg->cmsg_len = CMSG_LEN(sizeof(*sinfo));
73
74 msg.msg_controllen = cmsg->cmsg_len;
75 memcpy(CMSG_DATA(cmsg), sinfo, sizeof(*sinfo));
76 return sendmsg(sd, &msg, flags);
77}

85/* Implementation of sctp_sendmsg(3), we don't want to -lsctp */
86int sctp_sendmsg(int sd, const void *_msg, size_t len,
87 struct sockaddr *to, socklen_t tolen,
88 uint32_t ppid, uint32_t flags, uint16_t stream_no,
89 uint32_t ttl, uint32_t context)
90{
91 struct sctp_sndrcvinfo *sinfo;
92 char buf[CMSG_SPACE(sizeof(*sinfo))];
93 struct iovec iov = {
94 .iov_base = (void *)_msg,
95 .iov_len = len,
96 };
97 struct msghdr msg = {
98 .msg_name = to,
99 .msg_namelen = tolen,
100 .msg_iov = &iov,
101 .msg_iovlen = 1,
102 .msg_control = buf,
103 .msg_controllen = sizeof(buf),
104 .msg_flags = 0,
105 };
106
107 struct cmsghdr *cmsg = CMSG_FIRSTHDR(&msg);
108
109 cmsg->cmsg_level = IPPROTO_SCTP;
110 cmsg->cmsg_type = SCTP_SNDRCV;
111 cmsg->cmsg_len = CMSG_LEN(sizeof(*sinfo));
112
113 msg.msg_controllen = cmsg->cmsg_len;
114 sinfo = (void *)CMSG_DATA(cmsg);
115 memset(sinfo, 0, sizeof(*sinfo));
116 sinfo->sinfo_ppid = ppid;
117 sinfo->sinfo_flags = flags;
118 sinfo->sinfo_stream = stream_no;
119 sinfo->sinfo_timetolive = ttl;
120 sinfo->sinfo_context = context;
121
122 return sendmsg(sd, &msg, 0);
123}

125/* Implementation of sctp_recmsg(3), we don't want to -lsctp */
126int sctp_recvmsg(int sd, void *_msg, size_t len,
127 struct sockaddr *from, socklen_t *fromlen,
128 struct sctp_sndrcvinfo *sinfo, int *msg_flags)
129{
130 struct iovec iov = {
131 .iov_base = _msg,
132 .iov_len = len,
133 };
134 char buf[CMSG_SPACE(sizeof(*sinfo))];
135 struct msghdr msg = {
136 .msg_name = from,
137 .msg_namelen = fromlen ? *fromlen : 0,
138 .msg_iov = &iov,
139 .msg_iovlen = 1,
140 .msg_control = buf,
141 .msg_controllen = sizeof(buf),
142 };
143
144 int res = recvmsg(sd, &msg, 0);
145
146 RETHROW(res);
147 if (fromlen)
148 *fromlen = msg.msg_namelen;
149 if (sinfo) {
150 struct cmsghdr *cmsg = CMSG_FIRSTHDR(&msg);
151 while (cmsg) {
152 if ((cmsg->cmsg_level == IPPROTO_SCTP)
153 && (cmsg->cmsg_type == SCTP_SNDRCV))
154 {
155 memcpy(sinfo, CMSG_DATA(cmsg), sizeof(*sinfo));
156 break;
157 }
158 cmsg = CMSG_NXTHDR(&msg, cmsg);
159 }
160 }
161 if (msg_flags)
162 *msg_flags = msg.msg_flags;
163
164 return res;
165}

186int sctp_addr_len(const sockunion_t *addrs, int count)
187{
188 const byte *buf = (const byte *)addrs;
189
190 for (int i = 0; i < count; i++) {
191 int len = sockunion_len((const sockunion_t *)buf);
192 if (unlikely(len < 0)) {
193 errno = EINVAL;
194 return -1;
195 }
196 buf += len;
197 }
198
199 return buf - (const byte *)addrs;
200}

210int sctp_connectx_ng(int fd, struct sockaddr *addrs, int count,
211 sctp_assoc_t *nullable id)
212{
213 int res;
214 int size = sctp_addr_len((const sockunion_t *)addrs, count);
215 struct sctp_getaddrs_old opt_val = {
216 .assoc_id = 0,
217 .addr_num = size,
218 .addrs = addrs,
219 };
220 socklen_t opt_len = sizeof(opt_val);
221
222 RETHROW(size);
223
224 if (!id) {
225 /* No need to use CONNECTX3 */
226 return setsockopt(fd, SOL_SCTP, SCTP_SOCKOPT_CONNECTX, addrs, size);
227 }
228
229 res = getsockopt(fd, SOL_SCTP, SCTP_SOCKOPT_CONNECTX3, &opt_val,
230 &opt_len);
231 if (res == 0 || ERR_CONNECT_RETRIABLE(errno)) {
232 /* We should have a valid association ID */
233 *id = opt_val.assoc_id;
234 }
235
236 return res;
237}

240 struct sockaddr *addrs, int addr_size)
241{
242 struct sctp_getaddrs *ga = (void *)addrs;
243 int res;
244 socklen_t len = addr_size;
245
246 ga->assoc_id = id;
247 RETHROW(getsockopt(fd, SOL_SCTP, optnum, ga, &len));
248 res = ga->addr_num;
249 memmove(ga, ga + 1, len);
250 return res;
251}

253int sctp_close_assoc(int fd, int assoc_id)
254{
255 int res;
256
257 for (;;) {
258 struct sctp_sndrcvinfo sinfo = {
259 .sinfo_assoc_id = assoc_id,
260 .sinfo_flags = SCTP_EOF | SCTP_ABORT,
261 };
262 res = sctp_sendv(fd, NULL, 0, &sinfo, 0);
263 if (res < 0 && ERR_RW_RETRIABLE(errno))
264 continue;
265 return res;
266 }
267}

269void sctp_dump_notif(char *buf, int len)
270{
271 struct sctp_assoc_change *sac;
272 struct sctp_send_failed *ssf;
273 struct sctp_remote_error *sre;
274 union sctp_notification *snp;
275
276 if (len < ssizeof(snp->sn_header)) {
277 e_fatal("invalid NOTIF: len = %d != %zd", len, sizeof(snp->sn_header));
278 }
279
280 snp = (union sctp_notification *)buf;
281
282 switch (snp->sn_header.sn_type) {
283 case SCTP_ASSOC_CHANGE:
284 if (len < ssizeof(struct sctp_assoc_change)) {
285 e_fatal("invalid NOTIF assoc change: len = %d != %zd", len, sizeof(struct sctp_assoc_change));
286 }
287
288 sac = &snp->sn_assoc_change;
289 e_trace(0, "SCTP_ASSOC_CHANGE");
290 e_trace(0, "state=%u, error=%u, instreams=%u, outstreams=%u)",
291 sac->sac_state, sac->sac_error,
292 sac->sac_inbound_streams,
293 sac->sac_outbound_streams);
294 break;
295
296 case SCTP_SEND_FAILED:
297 if (len < ssizeof(struct sctp_send_failed)) {
298 e_fatal("invalid NOTIF send_failed: len = %d != %zd", len, sizeof(struct sctp_send_failed));
299 }
300 ssf = &snp->sn_send_failed;
301 e_trace(0, "SCTP_SEND_FAILED");
302 e_trace(0, "sendfailed: len=%u, err=%d",
303 ssf->ssf_length, ssf->ssf_error);
304 break;
305
306 case SCTP_REMOTE_ERROR:
307 if (len < ssizeof(struct sctp_remote_error)) {
308 e_fatal("invalid NOTIF remote_error: len = %d != %zd", len, sizeof(struct sctp_remote_error));
309 }
310 sre = &snp->sn_remote_error;
311 e_trace(0, "SCTP_REMOTE_ERROR");
312 e_trace(0, "remote_error: err=%u", ntohs(sre->sre_error));
313 break;
314
315 case SCTP_SHUTDOWN_EVENT:
316 if (len < ssizeof(struct sctp_send_failed)) {
317 e_fatal("invalid NOTIF send_failed: len = %d != %zd", len, sizeof(struct sctp_send_failed));
318 }
319 e_trace(0, "SCTP_SHUTDOWN_EVENT");
320 break;
321
322 default:
323 e_trace(0, "unknown type: %u", snp->sn_header.sn_type);
324 break;
325 }
326}
