87std::unique_ptr<ASTConsumer>
88RewriteBlocksAction::CreateASTConsumer(CompilerInstance &CI,
89 StringRef InFile) {
90 std::string suffix("rw." + llvm::sys::path::extension(InFile).str());
91
92 if (std::unique_ptr<raw_ostream> OS =
93 CI.createDefaultOutputFile(false, InFile, suffix))
94 return CreateBlocksRewriter(std::string(InFile), std::move(OS),
95 CI.getDiagnostics(), CI.getLangOpts(),
96 CI.getDiagnosticOpts().NoRewriteMacros);
97 return nullptr;

100static bool ExecuteCompilerInvocationRewriteBlock(CompilerInstance *Clang) {
101 // Honor -help.
102 if (Clang->getFrontendOpts().ShowHelp) {
103#if CLANG_VERSION_MAJOR >= 13
104 driver::getDriverOptTable().printHelp(
105#elif CLANG_VERSION_MAJOR >= 10
106 driver::getDriverOptTable().PrintHelp(
107#else
108 std::unique_ptr<OptTable> Opts = driver::createDriverOptTable();
109 Opts->PrintHelp(
110#endif /* CLANG_VERSION_MAJOR >= 13 || CLANG_VERSION_MAJOR >= 10 */
111 llvm::outs(),
112 "clang-rewrite-blocks [options] file.blk -o file.blk.c",
113 "LLVM 'Clang' Compiler Rewriter Block: "
114 "http://clang.llvm.org http://intersec.com",
115 /*Include=*/driver::options::CC1Option,
116 /*Exclude=*/0, /*ShowAllAliases=*/false);
117 return true;
118 }
119
120 // Honor -version.
121 //
122 // FIXME: Use a better -version message?
123 if (Clang->getFrontendOpts().ShowVersion) {
124 llvm::cl::PrintVersionMessage();
125 return true;
126 }
127
128 // Load any requested plugins.
129 for (unsigned i = 0,
130 e = Clang->getFrontendOpts().Plugins.size(); i != e; ++i) {
131 const std::string &Path = Clang->getFrontendOpts().Plugins[i];
132 std::string Error;
133 if (llvm::sys::DynamicLibrary::LoadLibraryPermanently(Path.c_str(), &Error))
134 Clang->getDiagnostics().Report(diag::err_fe_unable_to_load_plugin)
135 << Path << Error;
136 }
137
138 // Check if any of the loaded plugins replaces the main AST action
139 for (FrontendPluginRegistry::iterator it = FrontendPluginRegistry::begin(),
140 ie = FrontendPluginRegistry::end();
141 it != ie; ++it) {
142 std::unique_ptr<PluginASTAction> P(it->instantiate());
143 if (P->getActionType() == PluginASTAction::ReplaceAction) {
144 Clang->getFrontendOpts().ProgramAction = clang::frontend::PluginAction;
145#if CLANG_VERSION_MAJOR >= 11
146 Clang->getFrontendOpts().ActionName = it->getName().str();
147#else
148 Clang->getFrontendOpts().ActionName = it->getName();
149#endif /* CLANG_VERSION_MAJOR >= 11 */
150 break;
151 }
152 }
153
154 // If there were errors in processing arguments, don't do anything else.
155 if (Clang->getDiagnostics().hasErrorOccurred())
156 return false;
157 // Create and execute the frontend action.
158#if CLANG_VERSION_MAJOR >= 10
159 std::unique_ptr<FrontendAction> Act(std::make_unique<RewriteBlocksAction>());
160#else
161 std::unique_ptr<FrontendAction> Act(llvm::make_unique<RewriteBlocksAction>());
162#endif /* CLANG_VERSION_MAJOR >= 10 */
163 if (!Act)
164 return false;
165 bool Success = Clang->ExecuteAction(*Act);
166 if (Clang->getFrontendOpts().DisableFree)
167 llvm::BuryPointer(std::move(Act));
168 return Success;
169}

173int main(int argc, const char **argv) {
174 const char *Argv0 = argv[0];
175
176 if (argc <= 1) {
177 fprintf(stderr, "%s: error: no input files\n", Argv0);
178 exit(1);
179 }
180
181 void *MainAddr = (void*) (intptr_t) GetExecutablePath;
182
183 // Initialize targets first, so that --version shows registered targets.
184 llvm::InitializeAllTargets();
185 llvm::InitializeAllTargetMCs();
186 llvm::InitializeAllAsmPrinters();
187 llvm::InitializeAllAsmParsers();
188
189 // Buffer diagnostics from argument parsing so that we can output them using a
190 // well formed diagnostic object.
191 std::unique_ptr<CompilerInstance> Clang(new CompilerInstance());
192 IntrusiveRefCntPtr<DiagnosticOptions> DiagOpts = new DiagnosticOptions();
193 TextDiagnosticBuffer *DiagsBuffer = new TextDiagnosticBuffer;
194 TextDiagnosticPrinter *DiagClient =
195 new TextDiagnosticPrinter(llvm::errs(), &*DiagOpts);
196 IntrusiveRefCntPtr<DiagnosticIDs> DiagID(new DiagnosticIDs());
197 DiagnosticsEngine Diags(DiagID, &*DiagOpts, DiagClient);
198 bool HasTimeTrace;
199
200 ArrayRef<const char*> Args(argv + 1, argv + argc);
201
202#if CLANG_VERSION_MAJOR >= 10
203 bool Success = CompilerInvocation::CreateFromArgs(
204 Clang->getInvocation(), Args, Diags);
205#else
206 bool Success = CompilerInvocation::CreateFromArgs(
207 Clang->getInvocation(), Args.begin(), Args.end(), Diags);
208#endif /* CLANG_VERSION_MAJOR >= 10 */
209
210#if CLANG_VERSION_MAJOR >= 17
211 HasTimeTrace = !Clang->getFrontendOpts().TimeTracePath.empty();
212#else
213 HasTimeTrace = Clang->getFrontendOpts().TimeTrace;
214#endif /* CLANG_VERSION_MAJOR >= 17 */
215
216 if (HasTimeTrace) {
217#if CLANG_VERSION_MAJOR >= 10
218 llvm::timeTraceProfilerInitialize(
219 Clang->getFrontendOpts().TimeTraceGranularity, Argv0);
220#else
221 llvm::timeTraceProfilerInitialize();
222#endif /* CLANG_VERSION_MAJOR >= 10 */
223 }
224
225 // Infer the builtin include path if unspecified.
226 if (Clang->getHeaderSearchOpts().UseBuiltinIncludes &&
227 Clang->getHeaderSearchOpts().ResourceDir.empty())
228 Clang->getHeaderSearchOpts().ResourceDir =
229 CompilerInvocation::GetResourcesPath(Argv0, MainAddr);
230
231 // Create the actual diagnostics engine.
232 Clang->createDiagnostics();
233 if (!Clang->hasDiagnostics())
234 return 1;
235
236 // Set an error handler, so that any LLVM backend diagnostics go through our
237 // error handler.
238 llvm::install_fatal_error_handler(LLVMErrorHandler,
239 static_cast<void*>(&Clang->getDiagnostics()));
240
241 DiagsBuffer->FlushDiagnostics(Clang->getDiagnostics());
242 if (!Success)
243 return 1;
244
245 // Execute the frontend actions.
246 {
247 llvm::TimeTraceScope TimeScope("ExecuteCompiler", StringRef(""));
248 Success = ExecuteCompilerInvocationRewriteBlock(Clang.get());
249 }
250
251 // If any timers were active but haven't been destroyed yet, print their
252 // results now. This happens in -disable-free mode.
253 llvm::TimerGroup::printAll(llvm::errs());
254
255 if (llvm::timeTraceProfilerEnabled()) {
256#if CLANG_VERSION_MAJOR >= 17
257 SmallString<128> Path(Clang->getFrontendOpts().TimeTracePath);
258#else
259 SmallString<128> Path(Clang->getFrontendOpts().OutputFile);
260#endif /* CLANG_VERSION_MAJOR >= 17 */
261 llvm::sys::path::replace_extension(Path, "json");
262#if CLANG_VERSION_MAJOR >= 12
263 auto profilerOutput = Clang->createOutputFile(
264 Path.str(), /*Binary=*/false, /*RemoveFileOnSignal=*/false,
265 /*useTemporary=*/false);
266#else
267 auto profilerOutput =
268 Clang->createOutputFile(Path.str(),
269 /*Binary=*/false,
270 /*RemoveFileOnSignal=*/false, "",
271 /*Extension=*/"json",
272 /*useTemporary=*/false);
273#endif /* CLANG_VERSION_MAJOR >= 12 */
274
275 if (profilerOutput) {
276 llvm::timeTraceProfilerWrite(*profilerOutput);
277 // FIXME(ibiryukov): make profilerOutput flush in destructor instead.
278 profilerOutput->flush();
279 llvm::timeTraceProfilerCleanup();
280#if CLANG_VERSION_MAJOR >= 12
281 Clang->clearOutputFiles(false);
282#endif /* CLANG_VERSION_MAJOR >= 12 */
283
284 llvm::errs() << "Time trace json-file dumped to " << Path.str() << "\n";
285 llvm::errs()
286 << "Use chrome://tracing or Speedscope App "
287 "(https://www.speedscope.app) for flamegraph visualization\n";
288 }
289 }
290
291 // Our error handler depends on the Diagnostics object, which we're
292 // potentially about to delete. Uninstall the handler now so that any
293 // later errors use the default handling behavior instead.
294 llvm::remove_fatal_error_handler();
295
296 // When running with -disable-free, don't do any destruction or shutdown.
297 if (Clang->getFrontendOpts().DisableFree) {
298 llvm::BuryPointer(std::move(Clang));
299 return !Success;
300 }
301
302 return !Success;
303}
