29/** \brief representation of a string with length.
30 *
31 * This type provides a unified way to talk about immutable strings (unlike
32 * #sb_t that are mutable buffers).
33 *
34 * It remembers wether the string has been allocated through p_new, t_new, a
35 * pool or is static.
36 */
37typedef struct lstr_t {
38 union {
39 const char * nullable s;
40 char * nullable v;
41 void * nullable data;
42 };
43 int len;
44 unsigned mem_pool;
45} lstr_t;
46

376/** \brief return the left-trimmed lstr.
377 */
378static __must_check__ inline lstr_t lstr_ltrim(lstr_t s)
379{
380 while (s.len && isspace((unsigned char)s.s[0])) {
381 s.s++;
382 s.len--;
383 }
384
385 s.mem_pool = MEM_STATIC;
386 return s;
387}
388
389/** \brief return the right-trimmed lstr.
390 */
391static __must_check__ inline lstr_t lstr_rtrim(lstr_t s)
392{
393 while (s.len && isspace((unsigned char)s.s[s.len - 1])) {
394 s.len--;
395 }
396
397 return s;
398}
399
400/** \brief return the trimmed lstr.
401 */
402static __must_check__ inline lstr_t lstr_trim(lstr_t s)
403{
404 return lstr_rtrim(lstr_ltrim(s));
405}

427/** Returns "memcmp" ordering of \v s1 and \v s2. */
428static ALWAYS_INLINE int lstr_cmp(const lstr_t s1, const lstr_t s2)
429{
430 /* workaround for a warning of -Wstringop-overflow in gcc 8
431 * see https://gcc.gnu.org/bugzilla/show_bug.cgi?id=89699
432 */
433 size_t len = MIN((size_t)s1.len, (size_t)s2.len);
434
435 return memcmp(s1.s, s2.s, len) ?: CMP(s1.len, s2.len);
436}

462/** Returns whether \v s1 and \v s2 contents are equal. */
463static ALWAYS_INLINE bool lstr_equal(const lstr_t s1, const lstr_t s2)
464{

775static inline void lstr_xor(lstr_t in, lstr_t key, lstr_t out)
