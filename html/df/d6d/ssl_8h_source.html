22/**
23 * This file offers an interface which wrap the openssl AES-CBC & PKEY
24 * interfaces.
25 *
26 * \section AES-CBS
27 *
28 * The generic way to performs encrypt is to get an SSL context using one of
29 * the initializer, then update() the encrypted data as much as you want and
30 * finally call the finish() function to close the flow. Using the reset()
31 * operation you could repeat these operations several times. Example:
32 *
33 * // Get the SSL context
34 * init_aes256();
35 *
36 * // Encrypt some data
37 * update();
38 * update();
39 * update();
40 *
41 * // Prepare to encrypt some others data
42 * reset();
43 *
44 * // Encrypt the others data
45 * update();
46 * update();
47 *
48 * // Close the flow and wipe the SSL context
49 * finish();
50 * wipe();
51 *
52 * Decryption works in the same way.
53 *
54 * \section PKEY
55 *
56 * \subsection encryption
57 *
58 * The generic way to performs encryption is to init the SSL context with your
59 * public key (or private if you have access to it, although it's not required
60 * for encryption), call encrypt() on your data, and wipe the context.
61 *
62 * Example:
63 *
64 * ssl_ctx_t ctx;
65 * if (!ssl_ctx_init_pkey(&ctx, LSTR_NULL_V, pub_key, LSTR_NULL_V)) {
66 * return logger_error(… "key loading error: %s", ssl_get_error());
67 * }
68 * if (ssl_encrypt(&ctx, msg, &msg_encrypted) < 0) {
69 * ssl_ctx_wipe(&ctx);
70 * return logger_error(… "decrypt error: %s", ssl_get_error());
71 * }
72 * ssl_ctx_wipe(&ctx);
73 *
74 * \subsection decryption
75 *
76 * Decryption works in the same way, except that you need the private key and
77 * the passphrase if this key have been created with one.
78 *
79 * Example:
80 *
81 * ssl_ctx_t ctx;
82 * if (!ssl_ctx_init_pkey(&ctx, priv_key, LSTR_NULL_V, pass)) {
83 * return logger_error(… "key loading error: %s", ssl_get_error());
84 * }
85 * if (ssl_decrypt(&ctx, msg_encrypted, &msg_clear) < 0) {
86 * ssl_ctx_wipe(&ctx);
87 * return logger_error(… "decrypt error: %s", ssl_get_error());
88 * }
89 * ssl_ctx_wipe(&ctx);
90 *
91 */

118#ifndef RSA_OAEP_PADDING_SIZE
119/* RSA can just encrypt a message as large as the key, i.e. 256 bytes for a
120 * 2048 key. But the algorithm used some random padding to increase security.
121 * Thus, RSA encrypt message up to key_size - PADDING bytes.
122 *
123 * Sadly enough, if there is a RSA_PKCS1_PADDING_SIZE constant defined for the
124 * old PKCS1 method, I didn't found an equivalent way to get the maximum
125 * padding size used with OAEP -- which seems to be the current standard. The
126 * RSA_public_encrypt(3) man page says it's 41 bytes.
127 */
128# define RSA_OAEP_PADDING_SIZE 41
129#endif

313/**
314 * Encrypt an arbitrarily long lstr_t.
315 *
316 * Encrypt the data with a RSA key. If the data is too large to fit in a
317 * single RSA message, we use hybrid cryptography (RSA + AES). The format is
318 * described in the code.
319 *
320 * \param[in] ctx An ssl context initialized with an RSA key.
321 * \param[in] data The data to encrypt.
322 * \param[out] out The buffer where encrypted data are accumulated.
323 * \return 0 on success, -1 on error. You can call ssl_get_error() on error to
324 * have stringified informations.
325 */
326__must_check__ int
327ssl_encrypt_pkey_sb(ssl_ctx_t *ctx, lstr_t data, sb_t *out);
328
329/**
330 * Encrypt an arbitrarily long lstr_t.
331 *
332 * \param[in] ctx An ssl context initialized with an RSA key.
333 * \param[in] data The data to encrypt.
334 * \return the encrypted data as a t-stack allocated lstr_t (or LSTR_NULL_V on
335 * error). You can call ssl_get_error() on error to have stringified
336 * informations.
337 */
338static inline lstr_t t_ssl_encrypt_pkey_lstr(ssl_ctx_t *ctx, lstr_t data)
339{
340 SB_1k(out);
341
342 /* grow(data + AES key + AES IV + AES padding + RSA padding). */
343 sb_grow(&out, data.len + 32 + 16 + 16 + RSA_OAEP_PADDING_SIZE);
344 THROW_IF(ssl_encrypt_pkey_sb(ctx, data, &out) < 0, LSTR_NULL_V);
345 return t_lstr_dup(LSTR_SB_V(&out));
346}

394/**
395 * Decrypt an arbitrarily long lstr_t.
396 *
397 * \param[in] ctx An ssl context initialized with an RSA private key.
398 * \param[in] data The data to decrypt.
399 * \return the decrypted data as a t-stack allocated lstr_t (or LSTR_NULL_V on
400 * error). You can call ssl_get_error() on error to have stringified
401 * informations.
402 */
403static inline lstr_t t_ssl_decrypt_pkey_lstr(ssl_ctx_t *ctx, lstr_t data)
404{

406 t_SB(out, data.len + 1 - RSA_OAEP_PADDING_SIZE - min_aes_data);

408 THROW_IF(ssl_decrypt_pkey_sb(ctx, data, &out) < 0, LSTR_NULL_V);

524/** Wrapper to SSL_read that mimic read(2).
525 *
526 * \warning please carefully read what follows:
527 *
528 * Unlike read(), ssl_read() can actually read more data from the file
529 * descriptor than returned (the rest being bufferized internally by openssl,
530 * cf. SSL_pending()).
531 *
532 * This behavior can cause deadlocks in the following situation:
533 * - the event loop selects the file descriptor for reading and calls the
534 * on_event applicative callback.
535 * - this on_event callback calls ssl_read() with a given size; ssl_read()
536 * returns the given size, but for its internal purposes it actually reads
537 * all the data on the file descriptor.
538 * - the process goes back in event loop.
539 *
540 * In this situation, some data was not read by the applicative code, but the
541 * file descriptor won't ever be selected for reading by the event loop
542 * because there is no more data to read on the file descriptor -> the data
543 * bufferized by openssl is lost.
544 *
545 * \ref ssl_sb_read is safe to use in this context as it workarounds the
546 * problem.
547 *
548 * For more information, cf. this discussion on the subject:
549 * https://openssl-dev.openssl.narkive.com/yBiaSl0l/non-blocking-ssl-read-api-problem
550 *
551 * \param[in] ssl The ssl context for which data must be received.
552 * \param[in] buf The buffer into which data are received.
553 * \param[in] len The maximum number of bytes to receive into `buf`.
554 * \return the number of bytes sent.
555 */
556ssize_t ssl_read(SSL *ssl, void *buf, size_t len);
557
558/** Wrapper to SSL_write that mimic write(2).
559 *
560 * \warning: because of the implementation of the underlying function
561 * SSL_write(), ssl_write() behaves slightly differently than
562 * write() when it fails with a retriable error (ie. when
563 * res < 0 and ERR_RW_RETRIABLE(errno) is true). In that case:
564 * - the next calls MUST take the same arguments (same data and length -
565 * the pointer on the data can change), until it succeeds (even partially).
566 * - some data could actually have been written on the socket, and thus the
567 * remote peer can receive a part of the data even if ssl_write() fails.
568 *
569 * \param[in] ssl The ssl context for which data must be sent. It must be
570 * configured to allow partial write. (See
571 * SSL_MODE_ENABLE_PARTIAL_WRITE and
572 * SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER options).
573 * \param[in] buf The buffer to write.
574 * \param[in] len The number of bytes to send from `buf`.
575 *
576 * \return the number of bytes sent.
577 */
