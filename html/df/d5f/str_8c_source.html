66/** Replaces blank characters at end of string with '\0'.
67 *
68 * @return a pointer to the \0 at the end of str
69 */
70char *strrtrim(char *str)
71{
72 if (str) {
73 char *p = str + strlen(str);
74
75 while (p > str && isspace((unsigned char)p[-1])) {
76 *--p = '\0';
77 }
78 return p;
79 }
80
81 return NULL;
82}

84/** Tells whether str begins with p.
85 *
86 * @param pp if not null and str begins with p, pp is given the address of the
87 * first following character in str.
88 *
89 * @return 1 if a match is found, 0 otherwise.
90 */
91int strstart(const char *str, const char *p, const char **pp)
92{
93 if (!str)
94 return 0;
95
96 while (*p) {
97 if (*str++ != *p++)
98 return 0;
99 }
100 if (pp)
101 *pp = str;
102 return 1;
103}

105/** Tells whether str begins with p, case insensitive.
106 *
107 * @param pp if not null and str begins with p, pp is given the address of the
108 * first following character in str.
109 *
110 * @return 1 if a match is found, 0 otherwise.
111 */
112int stristart(const char *str, const char *p, const char **pp)
113{
114 if (!str)
115 return 0;
116
117 while (*p) {
118 if (tolower((unsigned char)*str++) != tolower((unsigned char)*p++))
119 return 0;
120 }
121 if (pp)
122 *pp = str;
123 return 1;
124}

127/** Find the first occurrence of the substring needle in str, case
128 * insensitively.
129 *
130 * @return a pointer to the beginning of the substring, or NULL if
131 * non was found.
132 * a final embedded '\0' char in needle can match the final '\0'.
133 */
134const char *stristrn(const char *str, const char *needle, size_t nlen)
135{
136 size_t i;
137 int c, nc;
138
139 if (!nlen)
140 return str;
141
142 nc = toupper((unsigned char)*needle);
143 for (;; str++) {
144 /* find first char of pattern in str */
145 c = toupper((unsigned char)*str);
146 if (c != nc) {
147 if (c == '\0')
148 return NULL;
149 } else {
150 /* compare the rest of needle */
151 for (i = 1;; i++) {
152 if (i == nlen)
153 return str;
154 if (c == '\0')
155 return NULL;
156 c = toupper((unsigned char)str[i]);
157 if (c != toupper((unsigned char)needle[i]))
158 break;
159 }
160 }
161 }
162}

165/* find a word in a list of words separated by sep.
166 */
167bool strfind(const char *keytable, const char *str, int sep)
168{
169 int c, len;
170 const char *p;
171
172 c = *str;
173 len = strlen(str);
174 /* need to special case the empty string */
175 if (len == 0) {
176 char empty[3];
177 empty[0] = empty[1] = sep;
178 empty[2] = '\0';
179 return strstr(keytable, empty) != NULL;
180 }
181
182 /* initial and trailing separators are optional */
183 /* they do not cause the empty string to match */
184 for (p = keytable;;) {
185 if (!memcmp(p, str, len) && (p[len] == sep || p[len] == '\0'))
186 return true;
187 for (;;) {
188 p = strchr(p + 1, c);
189 if (!p)
190 return false;
191 if (p[-1] == sep)
192 break;
193 }
194 }
195}

198/** Increment last counter in a buffer
199 *
200 * <code>buf</code> points to the start of the buffer.
201 * <code>len</code> can be negative, in this case <code>len=strlen(buf)</code>
202 *
203 * Examples :
204 * "000" => "001"
205 * "999" => "000"
206 * "foobar-0-01" => "foobar-0-02"
207 * "foobar-0-99" => "foobar-0-00"
208 *
209 * @return 1 if an overflow occurred, 0 otherwise
210 */
211int buffer_increment(char *buf, int len)
212{
213 int pos;
214
215 if (!buf) {
216 return 1;
217 }
218 if (len < 0) {
219 len = strlen(buf);
220 }
221 for (pos = len - 1; pos >= 0; pos--) {
222 switch (buf[pos]) {
223 case '0': case '1': case '2': case '3': case '4': case '5':
224 case '6': case '7': case '8':
225 buf[pos]++;
226 return 0;
227 case '9':
228 buf[pos] = '0';
229 break;
230 default:
231 return 1;
232 }
233 }
234 return 1;
235}

284/** Put random characters from alphabet in destination buffer, and
285 * terminate with a '\0' character.
286 * If no alphabet given, a 64 bytes alphabet will be used.
287 *
288 * @param dest the output buffer.
289 *
290 * @param size the size of the buffer
291 *
292 * @param alphabet a pointer to a lstr_t containing the alphabet we want
293 * to use. If LSTR_NULL a default b64 alphabet will be used
294 *
295 * @return the number of bytes (excluding '\0') written in the buffer
296 */
297size_t strrand(char dest[], size_t dest_size, lstr_t alphabet)
298{
299 char *p = dest;
300 char *last = p + dest_size - 1;
301
302 if (!alphabet.s && alphabet.len == 0) {
303 alphabet = LSTR("ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnop"
304 "qrstuvwxyz0123456789+/");
305 }
306
307 assert (alphabet.len > 0);
308 THROW_ERR_IF(alphabet.len <= 0);
309
310 for (; p < last; p++) {
311 *p = *(alphabet.s + rand_range(0, alphabet.len - 1));
312 }
313
314 *p = '\0';
315 return dest_size - 1;
316}

334int str_replace(const char search, const char replace, char *subject)
335{
336 int nb_replace = 0;
337 char *p = subject;
338
339 p = strchr(p, search);
340 while (p && *p != '\0') {
341 *p = replace;
342 /* Read replace char */
343 p++;
344 nb_replace++;
345 p = strchr(p, search);
346 }
347 return nb_replace;
348}
