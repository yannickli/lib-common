23 * \file crc64.c
24 * \brief CRC64 calculation

26 * Calculate the CRC64 using the slice-by-four algorithm. This is the same
27 * idea that is used in crc32_fast.c, but for CRC64 we use only four tables

40static ALWAYS_INLINE
41uint64_t naive_icrc64(uint64_t crc, const uint8_t *buf, ssize_t len)
42{
43 if (len) {
44 do {
45 crc = crc64table[0][*buf++ ^ A1(crc)] ^ S8(crc);
46 } while (--len);
47 }
48 return crc;
49}

51static uint64_t fast_icrc64(uint64_t crc, const uint8_t *buf, size_t size)
52{
53 size_t words;
54
55 if (unlikely((uintptr_t)buf & 3)) {
56 size -= 4 - ((uintptr_t)buf & 3);
57 do {
58 crc = crc64table[0][*buf++ ^ A1(crc)] ^ S8(crc);
59 } while ((uintptr_t)buf & 3);
60 }
61
62 words = size >> 2;
63 do {
64#if __BYTE_ORDER == __BIG_ENDIAN
65 const uint32_t tmp = (crc >> 32) ^ *(uint32_t *)(buf);
66#else
67 const uint32_t tmp = crc ^ *(uint32_t *)(buf);
68#endif
69 buf += 4;
70
71 crc = crc64table[3][A(tmp)]
72 ^ crc64table[2][B(tmp)]
73 ^ S32(crc)
74 ^ crc64table[1][C(tmp)]
75 ^ crc64table[0][D(tmp)];
76 } while (--words);
77
78 return naive_icrc64(crc, buf, size & (size_t)3);
79}

81__flatten
82uint64_t icrc64(uint64_t crc, const void *data, ssize_t len)
83{
84 crc = ~le_to_cpu64(crc);
85 if (len < 64)
86 return ~le_to_cpu64(naive_icrc64(crc, data, len));
87 return ~le_to_cpu64(fast_icrc64(crc, data, len));
88}
