19/*
20 * Intersec ASN.1 aligned PER encoder/decoder
21 *
22 * Based on previous ASN.1 library, it uses the same field registration
23 * macros. However, some features of the previous ASN.1 library are not
24 * supported.
25 *
26 * New features that were not included into previous library are:
27 * - Constraint support
28 * - Extension support
29 * - Explicit open type support
30 *
31 *
32 * - Supported native C types:
33 * - int8_t uint8_t int16_t uint16_t int32_t uint32_t int64_t
34 * - enum
35 * - bool
36 * - lstr_t
37 *
38 * - Supported ASN.1 types
39 * - INTEGER
40 * - Unconstrained
41 * - Constrained - ex. : (1 .. 42)
42 * - Extended - ex. : (1 .. 42, ...)
43 * (1 .. 42, ... , 0 .. 128)
44 * - BOOLEAN
45 * - OCTET STRING
46 * - Unconstrained
47 * - Constrained - ex. : (SIZE (3))
48 * (SIZE (0 .. 10))
49 * - Extended - ex. : (SIZE (1 .. 4, ...))
50 * - Length >= 16384 NOT supported yet (fragmented form)
51 * - Constraint FROM not supported
52 * - BIT STRING
53 * - Idem
54 * - ENUMERATED
55 * - Full support, extension included
56 * - Needs optimization
57 * - SEQUENCE
58 * - Full support when no extension is included
59 * - Extensions of type "ComponentType" supported (declared
60 * with "...") are supported (not the ones of type
61 * "ExtensionAdditionGroup", declared with "[[ <extensions ]]")
62 * - CHOICE
63 * - Warning : field order not checked yet (fields must use
64 * canonical order defined for tag values).
65 * See [3] 8.6 (p.12).
66 * - SET
67 * - PER encoding of a SET is the same as SEQUENCE encoding
68 * - Make sure you respect canonical ordering when registering
69 * a SET (see [3] 8.6 - p.12).
70 * - OPEN TYPE
71 * - When encountered type is predictible before beginning
72 * decoding, user shall only set the OPEN TYPE flag when
73 * registering the field. If the type is not predicable, then
74 * user shall declare an octet string and then decode its
75 * content later.
76 *
77 * References :
78 *
79 * [1] ITU-T X.691 (02/2002)
80 * data/dev/doc/asn1/per/X.691-0207.pdf
81 * [2] Olivier Dubuisson
82 * ASN.1 Communication entre systèmes hétérogènes
83 * Springer 1999
84 * French version : Paper book on Seb's desk
85 * English version : data/dev/doc/asn1/ASN1dubuisson.pdf
86 * [3] ITU-T X.680 (07/2002)
87 * data/dev/doc/asn1/X-680-0207.pdf
88 */

99static inline int asn1_enum_find_val(const asn1_enum_info_t *nonnull e,
100 int32_t val, bool *nonnull extended)
101{
102 struct {
103 const qv_t(i32) *values;
104 bool extended;
105 } vals_tabs[] = {
106 { &e->values, false },
107 { &e->ext_values, true },
108 };
109
110 carray_for_each_ptr(v, vals_tabs) {
111 tab_enumerate(pos, enum_val, v->values) {
112 if (enum_val == val) {
113 assert (e->extended || !v->extended);
114 *extended = v->extended;
115
116 return pos;
117 }
118 }
119 }
120
121 return -1;
122}

124static inline void asn1_enum_append(asn1_enum_info_t *e, int32_t val)
125{
126 qv_t(i32) *values;
127 const char *kind;
128
129 if (e->extended) {
130 values = &e->ext_values;
131 kind = "root";
132 } else {
133 values = &e->values;
134 kind = "extended";
135 }
136
137 if (values->len) {
138 int32_t last = *tab_last(values);
139
140 if (val < last) {
141 e_panic("enumeration %s value `%d` "
142 "should be registered before value `%d`", kind, val,
143 last);
144 }
145
146 if (val == last) {
147 e_panic("duplicated enumeration %s value `%d`", kind, val);
148 }
149 }
150
151 qv_append(values, val);
152}
