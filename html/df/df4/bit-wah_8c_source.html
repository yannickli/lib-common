39static void __wah_word_enum_start(wah_word_enum_t *en, qv_t(wah_word) *bucket)
40{
41 if (bucket->tab[en->pos].head.words > 0) {
42 en->state = WAH_ENUM_RUN;
43 en->remain_words = bucket->tab[en->pos].head.words;
44 en->current = 0 - bucket->tab[en->pos].head.bit;
45 } else
46 if (bucket->tab[en->pos + 1].count > 0) {
47 en->state = WAH_ENUM_LITERAL;
48 en->remain_words = bucket->tab[en->pos + 1].count;
49 en->current = bucket->tab[en->pos + 2].literal;
50 en->pos = bucket->tab[en->pos + 1].count + 2;
51 assert (en->pos <= bucket->len);
52 assert ((int)en->remain_words <= en->pos);
53 } else {
54 en->state = WAH_ENUM_PENDING;
55 en->remain_words = 1;
56 en->current = en->map->_pending;
57 }
58 en->current ^= en->reverse;
59}

61wah_word_enum_t wah_word_enum_start(const wah_t *map, bool reverse)
62{
63 wah_word_enum_t en = {
64 .map = map,
65 .state = WAH_ENUM_END,
66 .reverse = (uint32_t)0 - reverse,
67 };
68 qv_t(wah_word) *bucket = wah_word_enum_get_cur_bucket(&en);
69
70 if (map->len == 0) {
71 en.state = WAH_ENUM_END;
72 en.current = en.reverse;
73 return en;
74 }
75 __wah_word_enum_start(&en, bucket);
76 return en;
77}

79bool wah_word_enum_next(wah_word_enum_t *en)
80{
81 qv_t(wah_word) *bucket = wah_word_enum_get_cur_bucket(en);
82
83 if (en->remain_words != 1) {
84 en->remain_words--;
85 if (en->state == WAH_ENUM_LITERAL) {
86 en->current = bucket->tab[en->pos - en->remain_words].literal;
87 en->current ^= en->reverse;
88 }
89 return true;
90 }
91
92 switch (__builtin_expect(en->state, WAH_ENUM_RUN)) {
93 case WAH_ENUM_END:
94 return false;
95
96 case WAH_ENUM_PENDING:
97 en->state = WAH_ENUM_END;
98 en->current = en->reverse;
99 return false;
100
101 default: /* WAH_ENUM_RUN */
102 assert (en->state == WAH_ENUM_RUN);
103 en->pos++;
104 en->remain_words = bucket->tab[en->pos++].count;
105 en->pos += en->remain_words;
106 assert (en->pos <= bucket->len);
107 assert ((int)en->remain_words <= en->pos);
108 en->state = WAH_ENUM_LITERAL;
109 if (en->remain_words != 0) {
110 en->current = bucket->tab[en->pos - en->remain_words].literal;
111 en->current ^= en->reverse;
112 return true;
113 }
114
115 /* Transition to literal, so don't break here */
116 /* FALLTHROUGH */
117
118 case WAH_ENUM_LITERAL:
119 if (en->pos == bucket->len) {
120 if (en->bucket < en->map->_buckets.len - 1) {
121 en->bucket++;
122 bucket = wah_word_enum_get_cur_bucket(en);
123 en->pos = 0;
124 } else
125 if ((en->map->len % WAH_BIT_IN_WORD)) {
126 en->state = WAH_ENUM_PENDING;
127 en->remain_words = 1;
128 en->current = en->map->_pending;
129 en->current ^= en->reverse;
130 return true;
131 } else {
132 en->state = WAH_ENUM_END;
133 en->current = en->reverse;
134 return false;
135 }
136 }
137 __wah_word_enum_start(en, bucket);
138 return true;
139 }
140}

142static bool wah_word_enum_skip(wah_word_enum_t *en, uint32_t skip)
143{
144 uint32_t skippable = 0;
145
146 while (skip != 0) {
147 switch (__builtin_expect(en->state, WAH_ENUM_RUN)) {
148 case WAH_ENUM_END:
149 return false;
150
151 case WAH_ENUM_PENDING:
152 return wah_word_enum_next(en);
153
154 default:
155 skippable = MIN(skip, en->remain_words);
156 skip -= skippable;
157
158 /* XXX: Use next to skip the last word because:
159 * - if we reach the end of a run, this will automatically select
160 * the next run
161 * - if we end within a run of literal, this will properly update
162 * 'en->current' with the next literal word
163 */
164 en->remain_words -= skippable - 1;
165 wah_word_enum_next(en);
166 break;
167 }
168 }
169 return true;
170}

172uint32_t wah_word_enum_skip0(wah_word_enum_t *en)
173{
174 uint32_t skipped = 0;
175
176 while (en->current == 0) {
177 switch (__builtin_expect(en->state, WAH_ENUM_RUN)) {
178 case WAH_ENUM_END:
179 return skipped;
180
181 case WAH_ENUM_PENDING:
182 skipped++;
183 wah_word_enum_next(en);
184 return skipped;
185
186 case WAH_ENUM_RUN:
187 skipped += en->remain_words;
188 en->remain_words = 1;
189 wah_word_enum_next(en);
190 break;
191
192 case WAH_ENUM_LITERAL:
193 skipped++;
194 wah_word_enum_next(en);
195 break;
196 }
197 }
198 return skipped;
199}

204bool wah_bit_enum_scan_word(wah_bit_enum_t *en)
205{
206 /* realign to a word boundary */
207 assert (en->current_word == 0);
208 en->key += en->remain_bits;
209 assert (en->word_en.state <= WAH_ENUM_PENDING ||
210 (en->key % WAH_BIT_IN_WORD) == 0);
211
212 while (wah_word_enum_next(&en->word_en)) {
213 en->current_word = en->word_en.current;
214 if (en->word_en.state == WAH_ENUM_RUN) {
215 if (en->current_word) {
216 en->remain_bits = en->word_en.remain_words * WAH_BIT_IN_WORD;
217 en->word_en.remain_words = 1;
218 return true;
219 }
220 en->key += en->word_en.remain_words * WAH_BIT_IN_WORD;
221 en->word_en.remain_words = 1;
222 } else {
223 if (unlikely(en->word_en.state == WAH_ENUM_PENDING)) {
224 en->remain_bits = en->word_en.map->len % WAH_BIT_IN_WORD;
225 en->current_word &= BITMASK_LT(uint32_t, en->remain_bits);
226 } else {
227 en->remain_bits = WAH_BIT_IN_WORD;
228 }
229 if (likely(en->current_word)) {
230 return true;
231 }
232 en->key += WAH_BIT_IN_WORD;
233 }
234 }
235 return false;
236}

238wah_bit_enum_t wah_bit_enum_start(const wah_t *wah, bool reverse)
239{
240 wah_bit_enum_t en;
241
242 p_clear(&en, 1);
243 en.word_en = wah_word_enum_start(wah, reverse);
244 if (en.word_en.state != WAH_ENUM_END) {
245 en.current_word = en.word_en.current;
246 en.remain_bits = WAH_BIT_IN_WORD;
247 if (en.word_en.state == WAH_ENUM_PENDING) {
248 en.remain_bits = en.word_en.map->len % WAH_BIT_IN_WORD;
249 en.current_word &= BITMASK_LT(uint32_t, en.remain_bits);
250 }
251 wah_bit_enum_scan(&en);
252 }
253 return en;
254}

256void wah_bit_enum_skip1s(wah_bit_enum_t *en, uint64_t to_skip)
257{
258 if (to_skip == 0) {
259 return;
260 }
261
262 while (to_skip) {
263 uint64_t bits;
264
265 switch (en->word_en.state) {
266 case WAH_ENUM_PENDING:
267 case WAH_ENUM_LITERAL:
268 bits = bitcount32(en->current_word);
269
270 if (bits > to_skip) {
271 goto end;
272 }
273
274 to_skip -= bits;
275 en->current_word = 0;
276 break;
277
278 case WAH_ENUM_RUN:
279 bits = MIN(to_skip, en->remain_bits);
280 en->key += bits;
281 en->remain_bits -= bits;
282 to_skip -= bits;
283 if (en->remain_bits < WAH_BIT_IN_WORD) {
284 en->current_word = BITMASK_LT(uint32_t, en->remain_bits);
285 }
286 if (en->current_word) {
287 return;
288 }
289 break;
290
291 case WAH_ENUM_END:
292 return;
293 }
294
295 if (!wah_bit_enum_scan_word(en)) {
296 return;
297 }
298 }
299
300 end:
301 wah_bit_enum_scan(en);
302 while (to_skip > 0 && en->word_en.state != WAH_ENUM_END) {
303 wah_bit_enum_next(en);
304 to_skip--;
305 }
306}

325{

350void wah_reset_map(wah_t *map)
351{
352 map->len = 0;
353 map->active = 0;
354 map->previous_run_pos = -1;
355 map->last_run_pos = 0;
356 map->_pending = 0;
357
358 if (!map->_buckets.len) {
359 wah_create_bucket(map, 0);
360 }
361
362 qv_clear(&map->_buckets.tab[0]);
363 qv_growlen0(&map->_buckets.tab[0], 2);
364
365 for (int pos = 1; pos < map->_buckets.len; pos++) {
366 qv_wipe(&map->_buckets.tab[pos]);
367 }
368 qv_clip(&map->_buckets, 1);
369}

371/* WAH copy requires an initialized wah as target
372 */
373void wah_copy(wah_t *map, const wah_t *src)
374{
375 qv_t(wah_word_vec) buckets = map->_buckets;
376
377 p_copy(map, src, 1);
378 map->_buckets = buckets;
379
380 /* Wipe buckets which are not needed anymore. */
381 for (int pos = map->_buckets.len; pos-- > src->_buckets.len; ) {
382 qv_wipe(&map->_buckets.tab[pos]);
383 qv_remove(&map->_buckets, pos);
384 }
385
386 /* Create new buckets. */
387 for (int i = map->_buckets.len; i < src->_buckets.len; i++) {
388 wah_create_bucket(map, 0);
389 }
390
391 /* Copy buckets. */
392 tab_enumerate_ptr(pos, src_bucket, &src->_buckets) {
393 qv_t(wah_word) *dst_bucket = &map->_buckets.tab[pos];
394
395 qv_splice(dst_bucket, 0, dst_bucket->len,
396 src_bucket->tab, src_bucket->len);
397 }
398}

471static
472void wah_check_normalized(const wah_t *map)
473{
474#ifdef WAH_CHECK_NORMALIZED
475 uint32_t prev_word = 0xcafebabe;
476
477 tab_for_each_ptr(bucket, &map->_buckets) {
478 int pos = 0;
479
480 while (pos < bucket->len) {
481 wah_header_t *head = &bucket->tab[pos++].head;
482 uint32_t count = bucket->tab[pos++].count;
483
484 assert (head->words >= 2 || pos == bucket->len || pos == 2);
485 if (prev_word == UINT32_MAX || prev_word == 0) {
486 assert (prev_word != head->bit ? UINT32_MAX : 0);
487 prev_word = head->bit ? UINT32_MAX : 0;
488 }
489
490 for (uint32_t i = 0; i < count; i++) {
491 if (prev_word == UINT32_MAX || prev_word == 0) {
492 assert (prev_word != bucket->tab[pos].literal);
493 }
494 prev_word = bucket->tab[pos++].literal;
495 }
496 }
497 }
498#endif
499}

501static ALWAYS_INLINE
502void wah_check_invariant(const wah_t *map)
503{
504 qv_t(wah_word) *last_bucket = tab_last(&map->_buckets);
505
506 assert (map->last_run_pos >= 0);
507 assert (map->previous_run_pos >= -1);
508 tab_for_each_ptr(bucket, &map->_buckets) {
509 assert (bucket->len >= 2);
510 }
511 assert ((int)*wah_last_run_count(map) + map->last_run_pos + 2
512 == last_bucket->len);
513 assert (map->len >= map->active);
514 assert (map->len >= _G.bits_in_bucket * (map->_buckets.len - 1));
515 assert (map->len <= _G.bits_in_bucket * map->_buckets.len
516 + WAH_BIT_IN_WORD);
517 wah_check_normalized(map);
518}

521static inline
522void wah_flatten_last_run(wah_t *map)
523{
524 qv_t(wah_word) *bucket = tab_last(&map->_buckets);
525 wah_header_t *head;
526
527 head = wah_last_run_header(map);
528 if (likely(head->words != 1)) {
529 return;
530 }
531 assert (*wah_last_run_count(map) == 0);
532 assert (bucket->len == map->last_run_pos + 2);
533
534 if (map->last_run_pos > 0) {
535 bucket->len -= 2;
536 bucket->tab[map->previous_run_pos + 1].count++;
537 map->last_run_pos = map->previous_run_pos;
538 map->previous_run_pos = -1;
539 } else {
540 head->words = 0;
541 bucket->tab[1].count = 1;
542 }
543
544 wah_append_literal(map, head->bit ? UINT32_MAX : 0);
545 wah_check_invariant(map);
546}

548static void __wah_push_pending(wah_t *map, uint64_t words)
549{
550 const bool is_trivial = map->_pending == UINT32_MAX || map->_pending == 0;
551
552 if (!is_trivial) {
553 wah_flatten_last_run(map);
554 *wah_last_run_count(map) += words;
555 while (words > 0) {
556 wah_append_literal(map, map->_pending);
557 words--;
558 }
559 } else {
560 qv_t(wah_word) *bucket = tab_last(&map->_buckets);
561 wah_header_t *head = wah_last_run_header(map);
562
563 if (*wah_last_run_count(map) == 0
564 && (!head->bit == !map->_pending || head->words == 0))
565 {
566 uint64_t to_add;
567
568 to_add = MIN(words, WAH_MAX_WORDS_IN_RUN - head->words);
569
570 /* Merge with previous */
571 head->words += to_add;
572 head->bit = !!map->_pending;
573 words -= to_add;
574 }
575 if (head->words < 2) {
576 wah_flatten_last_run(map);
577 }
578
579 while (words) {
580 /* Create a new run */
581 wah_header_t new_head;
582 uint64_t to_add = MIN(words, WAH_MAX_WORDS_IN_RUN);
583
584 new_head.bit = !!map->_pending;
585 new_head.words = to_add;
586 words -= to_add;
587 map->previous_run_pos = map->last_run_pos;
588 map->last_run_pos = bucket->len;
589 wah_append_header(map, new_head);
590 }
591 }
592 map->_pending = 0;
593}

595static void wah_push_pending(wah_t *map, uint64_t words, uint64_t active)
596{
597 uint32_t pending = map->_pending;
598
599 assert (words > 0);
600 assert (map->len % WAH_BIT_IN_WORD == 0);
601
602 while (words) {
603 uint64_t bucket_len = map->len % _G.bits_in_bucket;
604 uint64_t to_add;
605
606 if (map->len && map->len == map->_buckets.len * _G.bits_in_bucket) {
607 assert (bucket_len == 0);
608 __wah_create_bucket(map);
609 }
610
611 to_add = MIN(words,
612 (_G.bits_in_bucket - bucket_len) / WAH_BIT_IN_WORD);
613 map->len += to_add * WAH_BIT_IN_WORD;
614 map->_pending = pending;
615 __wah_push_pending(map, to_add);
616 words -= to_add;
617 }
618
619 map->active += active;
620}

622void wah_add0s(wah_t *map, uint64_t count)
623{
624 uint64_t remain = map->len % WAH_BIT_IN_WORD;
625
626 if (map->len > map->_buckets.len * _G.bits_in_bucket) {
627 __wah_create_bucket(map);
628 }
629
630 wah_check_invariant(map);
631 if (remain + count < WAH_BIT_IN_WORD) {
632 map->len += count;
633 wah_check_invariant(map);
634 return;
635 }
636 if (remain > 0) {
637 count -= WAH_BIT_IN_WORD - remain;
638 map->len += WAH_BIT_IN_WORD - remain;
639 __wah_push_pending(map, 1);
640 }
641 if (count >= WAH_BIT_IN_WORD) {
642 uint64_t words = count / WAH_BIT_IN_WORD;
643
644 wah_push_pending(map, words, 0);
645 count -= words * WAH_BIT_IN_WORD;
646 }
647 map->len += count;
648 wah_check_invariant(map);
649}

660void wah_add1s(wah_t *map, uint64_t count)
661{
662 uint64_t remain = map->len % WAH_BIT_IN_WORD;
663
664 if (map->len > map->_buckets.len * _G.bits_in_bucket) {
665 __wah_create_bucket(map);
666 }
667
668 wah_check_invariant(map);
669 if (remain + count < WAH_BIT_IN_WORD) {
670 map->_pending |= BITMASK_LT(uint32_t, count) << remain;
671 map->len += count;
672 map->active += count;
673 wah_check_invariant(map);
674 return;
675 }
676 if (remain > 0) {
677 map->_pending |= BITMASK_GE(uint32_t, remain);
678 map->len += WAH_BIT_IN_WORD - remain;
679 map->active += WAH_BIT_IN_WORD - remain;
680 count -= WAH_BIT_IN_WORD - remain;
681 __wah_push_pending(map, 1);
682 }
683 if (count >= WAH_BIT_IN_WORD) {
684 uint64_t words = count / WAH_BIT_IN_WORD;
685
686 map->_pending = UINT32_MAX;
687 wah_push_pending(map, words, words * WAH_BIT_IN_WORD);
688 count -= words * WAH_BIT_IN_WORD;
689 }
690 map->_pending = BITMASK_LT(uint32_t, count);
691 map->len += count;
692 map->active += count;
693 wah_check_invariant(map);
694}

713static
714const void *wah_read_word(const uint8_t *src, uint64_t count,
715 uint64_t *res, int *bits)
716{
717 uint64_t mask;
718
719 if (count >= 64) {
720 *res = get_unaligned_le64(src);
721 *bits = 64;
722 return src + 8;
723 }
724
725 *res = 0;
726 *bits = 0;
727 mask = BITMASK_LT(uint64_t, count);
728#define get_unaligned_le8(src) (*src)
729#define READ_SIZE(Size) \
730 if (count > (Size - 8)) { \
731 const uint64_t to_read = MIN(count, Size); \
732 *res |= ((uint64_t)get_unaligned_le##Size(src)) << *bits; \
733 *bits += to_read; \
734 src += Size / 8; \
735 if (to_read == count) { \
736 *res &= mask; \
737 return src; \
738 } \
739 count -= to_read; \
740 }
741 READ_SIZE(32);
742 READ_SIZE(24);
743 READ_SIZE(16);
744 READ_SIZE(8);
745#undef READ_SIZE
746 *res &= mask;
747 return src;
748}

750static void wah_add_bits(wah_t *map, uint64_t word, int bits)
751{
752 bool on_0 = true;
753
754 while (bits > 0) {
755 if (word == 0) {
756 if (on_0) {
757 wah_add0s(map, bits);
758 } else {
759 wah_add1s(map, bits);
760 }
761 return;
762 } else {
763 int first = bsf64(word);
764 if (first > bits) {
765 first = bits;
766 }
767 if (first != 0) {
768 if (on_0) {
769 wah_add0s(map, first);
770 } else {
771 wah_add1s(map, first);
772 }
773 bits -= first;
774 word >>= first;
775 }
776 word = ~word;
777 on_0 = !on_0;
778 }
779 }
780}

782static
783const void *wah_add_unaligned(wah_t *map, const uint8_t *src, uint64_t count)
784{
785 while (count >= 64) {
786 ssize_t run_length = 64;
787 bool bit = false;
788 uint64_t word = get_unaligned_le64(src);
789
790 switch (word) {
791 case 0:
792 bit = false;
793 break;
794
795 case UINT64_MAX:
796 bit = true;
797 break;
798
799 default:
800 wah_add_bits(map, word, 64);
801 goto end;
802 }
803
804 run_length = bsf(src, 0, count, bit);
805 if (run_length < 0) {
806 run_length = count;
807 }
808 run_length = ROUND_2EXP(run_length, 8);
809 if (bit) {
810 wah_add1s(map, run_length);
811 } else {
812 wah_add0s(map, run_length);
813 }
814
815 end:
816 src += run_length / 8;
817 count -= run_length;
818 }
819
820 while (count > 0) {
821 uint64_t word;
822 int bits = 0;
823
824 src = wah_read_word(src, count, &word, &bits);
825 count -= bits;
826
827 wah_add_bits(map, word, bits);
828 }
829 wah_check_invariant(map);
830 return src;
831}

833static void wah_add_literal(wah_t *map, const uint8_t *src, uint64_t count)
834{
835 qv_t(wah_word) *bucket = tab_last(&map->_buckets);
836
837 wah_flatten_last_run(map);
838 map->active += membitcount(src, count);
839
840 while (count) {
841 uint64_t bucket_len = map->len % _G.bits_in_bucket;
842 uint64_t to_add;
843
844 if (map->len && map->len == map->_buckets.len * _G.bits_in_bucket) {
845 assert (bucket_len == 0);
846 bucket = __wah_create_bucket(map);
847 }
848
849 to_add = MIN(count / 4,
850 (_G.bits_in_bucket - bucket_len) / WAH_BIT_IN_WORD);
851
852 *wah_last_run_count(map) += to_add;
853 qv_extend(bucket, (wah_word_t *)src, to_add);
854
855 count -= to_add * 4;
856 src += to_add * 4;
857 map->len += to_add * WAH_BIT_IN_WORD;
858 }

861static
862void wah_add_aligned(wah_t *map, const uint8_t *src, uint64_t count)
863{
864 uint64_t exp_len = map->len + count;
865
866 while (count >= 32) {
867 ssize_t run_length = 32;
868 bool bit = false;
869
870 switch (get_unaligned_le32(src)) {
871 case 0:
872 bit = false;
873 break;
874
875 case UINT32_MAX:
876 bit = true;
877 break;
878
879 default:
880 wah_add_literal(map, src, 4);
881 goto end;
882 }
883
884 run_length = bsf(src, 0, ROUND_2EXP(count, 32), bit);
885 if (run_length < 0) {
886 run_length = count;
887 }
888 run_length = ROUND_2EXP(run_length, 32);
889 if (bit) {
890 wah_add1s(map, run_length);
891 } else {
892 wah_add0s(map, run_length);
893 }
894
895 end:
896 src += run_length / 8;
897 count -= run_length;
898 }
899 wah_check_invariant(map);
900 map->_pending = 0;
901
902 if (count > 0) {
903 uint64_t word = 0;
904 int bits;
905
906 wah_read_word(src, count, &word, &bits);
907 assert ((size_t)bits == count);
908 map->_pending = word;
909 map->len += bits;
910 map->active += bitcount32(map->_pending);
911 wah_check_invariant(map);
912 }
913 assert (map->len == exp_len);
914}

917void wah_add(wah_t *map, const void *data, uint64_t count)
918{
919 uint32_t remain = WAH_BIT_IN_WORD - (map->len % WAH_BIT_IN_WORD);
920
921 wah_check_invariant(map);
922 if (remain != WAH_BIT_IN_WORD) {
923 if (remain >= count || (remain % 8) != 0) {
924 wah_add_unaligned(map, data, count);
925 wah_check_invariant(map);
926 return;
927 } else {
928 data = wah_add_unaligned(map, data, remain);
929 count -= remain;
930 }
931 }
932 assert (map->len % WAH_BIT_IN_WORD == 0);
933 wah_add_aligned(map, data, count);
934 wah_check_invariant(map);
935}

957{
958 uint64_t count = MIN(run->remain_words, data->remain_words);
959
960 assert (count > 0);
961 wah_word_enum_skip(run, count);
962
963 if (unlikely(!data->current || data->current == UINT32_MAX)) {
964 map->_pending = data->current;
965 wah_push_pending(map, 1, bitcount32(map->_pending));
966 wah_word_enum_next(data);
967 count--;
968 }
969 if (likely(count > 0)) {
970 const wah_word_t *words;
971 qv_t(wah_word) *bucket = wah_word_enum_get_cur_bucket(data);
972
973 words = &bucket->tab[data->pos - data->remain_words];
974 wah_word_enum_skip(data, count);
975
976 wah_flatten_last_run(map);
977 if (map->len && map->len == map->_buckets.len * _G.bits_in_bucket) {
978 __wah_create_bucket(map);
979 }
980
981 *wah_last_run_count(map) += count;
982 bucket = tab_last(&map->_buckets);
983 qv_extend(bucket, words, count);
984 if (data->reverse) {
985 for (int i = bucket->len - count; i < bucket->len; i++) {
986 bucket->tab[i].literal = ~bucket->tab[i].literal;
987 }
988 }
989 map->len += count * WAH_BIT_IN_WORD;
990 map->active += membitcount(&bucket->tab[bucket->len - count],
991 count * sizeof(wah_word_t));
992 }
993}

1000static
1001void wah_and_(wah_t *map, const wah_t *other, bool map_not, bool other_not)
1002{
1003 t_scope;
1004 const wah_t *src = t_wah_dup(map);
1005 wah_word_enum_t src_en = wah_word_enum_start(src, map_not);
1006 wah_word_enum_t other_en = wah_word_enum_start(other, other_not);
1007
1008 wah_check_invariant(map);
1009 wah_reset_map(map);
1010 while (src_en.state != WAH_ENUM_END || other_en.state != WAH_ENUM_END) {
1011 if (src_en.state == WAH_ENUM_END) {
1012 src_en.remain_words = REMAIN_WORDS(other, map);
1013 } else
1014 if (other_en.state == WAH_ENUM_END) {
1015 other_en.remain_words = REMAIN_WORDS(src, map);
1016 }
1017
1018 switch (src_en.state | (other_en.state << 2)) {
1019 case WAH_ENUM_END | (WAH_ENUM_PENDING << 2):
1020 case WAH_ENUM_PENDING | (WAH_ENUM_END << 2):
1021 case WAH_ENUM_PENDING | (WAH_ENUM_PENDING << 2):
1022 map->len = MAX(other->len, src->len);
1023 map->_pending = src_en.current & other_en.current;
1024 map->active += bitcount32(map->_pending);
1025 wah_word_enum_next(&src_en);
1026 wah_word_enum_next(&other_en);
1027 break;
1028
1029 case WAH_ENUM_RUN | (WAH_ENUM_LITERAL << 2):
1030 case WAH_ENUM_END | (WAH_ENUM_LITERAL << 2):
1031 if (src_en.current) {
1032 PUSH_COPY(src_en, other_en);
1033 } else {
1034 PUSH_0RUN(src_en.remain_words);
1035 }
1036 break;
1037
1038 case WAH_ENUM_LITERAL | (WAH_ENUM_RUN << 2):
1039 case WAH_ENUM_LITERAL | (WAH_ENUM_END << 2):
1040 if (other_en.current) {
1041 PUSH_COPY(other_en, src_en);
1042 } else {
1043 PUSH_0RUN(other_en.remain_words);
1044 }
1045 break;
1046
1047 case WAH_ENUM_RUN | (WAH_ENUM_RUN << 2):
1048 case WAH_ENUM_END | (WAH_ENUM_RUN << 2):
1049 case WAH_ENUM_RUN | (WAH_ENUM_END << 2):
1050 if (!other_en.current || !src_en.current) {
1051 uint64_t run = 0;
1052
1053 if (!other_en.current) {
1054 run = other_en.remain_words;
1055 }
1056 if (!src_en.current) {
1057 run = MAX(run, src_en.remain_words);
1058 }
1059 PUSH_0RUN(run);
1060 } else {
1061 PUSH_1RUN(MIN(other_en.remain_words, src_en.remain_words));
1062 }
1063 break;
1064
1065 default:
1066 map->_pending = src_en.current & other_en.current;
1067 wah_push_pending(map, 1, bitcount32(map->_pending));
1068 wah_word_enum_next(&src_en);
1069 wah_word_enum_next(&other_en);
1070 break;
1071 }
1072 }
1073 wah_check_invariant(map);
1074
1075 assert (map->len == MAX(src->len, other->len));
1076 {
1077 uint64_t src_active = src->active;
1078 uint64_t other_active = other->active;
1079
1080 if (map_not) {
1081 src_active = MAX(other->len, src->len) - src->active;
1082 }
1083 if (other_not) {
1084 other_active = MAX(other->len, src->len) - other->active;
1085 }
1086 assert (map->active <= MIN(src_active, other_active));
1087 }
1088}

1107static void wah_add_en(wah_t *dest, wah_word_enum_t *en, uint64_t words)
1108{
1109 uint64_t exp_len = (words * WAH_BIT_IN_WORD) + dest->len;
1110
1111 while (en->state != WAH_ENUM_END && words > 0) {
1112 uint32_t to_read = MIN(words, en->remain_words);
1113
1114 switch (en->state) {
1115 case WAH_ENUM_LITERAL: {
1116 qv_t(wah_word) *bucket = wah_word_enum_get_cur_bucket(en);
1117
1118 wah_add_aligned(dest,
1119 (const uint8_t *)&bucket->tab[en->pos - en->remain_words],
1120 to_read * WAH_BIT_IN_WORD);
1121 } break;
1122
1123 case WAH_ENUM_PENDING:
1124 wah_add_aligned(dest, (const uint8_t *)&en->current, WAH_BIT_IN_WORD);
1125 break;
1126
1127 case WAH_ENUM_RUN:
1128 if (en->current) {
1129 wah_add1s(dest, to_read * WAH_BIT_IN_WORD);
1130 } else {
1131 wah_add0s(dest, to_read * WAH_BIT_IN_WORD);
1132 }
1133 break;
1134
1135 case WAH_ENUM_END:
1136 break;
1137 }
1138 words -= to_read;
1139 wah_word_enum_skip(en, to_read);
1140 }
1141
1142 if (words > 0) {
1143 wah_add0s(dest, words * WAH_BIT_IN_WORD);
1144 }
1145 assert (exp_len == dest->len);
1146}

1157 case WAH_ENUM_RUN:

1159 return 0xff00000000UL | a->remain_words;

1162 }

1164 case WAH_ENUM_LITERAL:
1165 case WAH_ENUM_PENDING:

1167
1168 case WAH_ENUM_END:

1170 }

1175wah_t *wah_multi_or(const wah_t *src[], int len, wah_t * restrict dest)
1176{
1177 t_scope;
1178 qv_t(wah_word_enum) enums;
1179 uint32_t buffer[1024];
1180 byte buffer_flags[countof(buffer)];
1181 uint64_t exp_len = 0;
1182 uint64_t min_act = 0;
1183 uint64_t max_act = 0;
1184
1185 if (!dest) {
1186 dest = wah_pool_acquire();
1187 } else {
1188 wah_reset_map(dest);
1189 }
1190
1191 t_qv_init(&enums, len);
1192 for (int i = 0; i < len; i++) {
1193 wah_word_enum_t *en = t_new_raw(wah_word_enum_t, 1);
1194
1195 exp_len = MAX(exp_len, src[i]->len);
1196 min_act = MAX(min_act, src[i]->active);
1197 max_act += src[i]->active;
1198 wah_check_invariant(src[i]);
1199 *en = wah_word_enum_start(src[i], false);
1200 if (en->state != WAH_ENUM_END) {
1201 qv_append(&enums, en);
1202 }
1203 }
1204 max_act = MIN(exp_len, max_act);
1205
1206 if (enums.len == 1) {
1207 wah_copy(dest, enums.tab[0]->map);
1208 return dest;
1209 }
1210
1211#define CONSUME_ALL(amount, skip_first) do { \
1212 const uint64_t __amount = (amount); \
1213 const bool __skip_first = (skip_first); \
1214 \
1215 tab_for_each_pos_safe(pos, &enums) { \
1216 wah_word_enum_t *en = enums.tab[pos]; \
1217 \
1218 if (!__skip_first || en != first) { \
1219 wah_word_enum_skip(en, __amount); \
1220 } \
1221 if (en->state == WAH_ENUM_END) { \
1222 if (pos != enums.len - 1) { \
1223 enums.tab[pos] = enums.tab[enums.len - 1]; \
1224 } \
1225 enums.len--; \
1226 } \
1227 } \
1228 } while (0)
1229
1230 while (enums.len) {
1231 uint32_t bits = 0;
1232 uint32_t buf_pos = 0;
1233 uint32_t end_pos = 0;
1234 uint64_t first_weight = 0;
1235 uint64_t second_weight = 0;
1236 wah_word_enum_t *first = NULL;
1237 wah_word_enum_t *second = NULL;
1238
1239 tab_for_each_entry(e, &enums) {
1240 uint64_t w = wah_word_enum_weight(e);
1241
1242 if (w > first_weight || !first) {
1243 second = first;
1244 second_weight = first_weight;
1245 first = e;
1246 first_weight = w;
1247 } else
1248 if (w > second_weight || !second) {
1249 second = e;
1250 second_weight = w;
1251 }
1252 }
1253 assert (first);
1254 assert (second || enums.len == 1);
1255
1256 if (second && second->state == WAH_ENUM_RUN && !second->current) {
1257 wah_add_en(dest, first, second->remain_words);
1258 CONSUME_ALL(second->remain_words, true);
1259 continue;
1260 } else
1261 if (enums.len == 1 && first->state != WAH_ENUM_PENDING) {
1262 uint64_t to_consume = (first->map->len - dest->len) / 32;
1263 wah_add_en(dest, first, to_consume);
1264 if (first->state == WAH_ENUM_END) {
1265 enums.len--;
1266 }
1267 continue;
1268 } else
1269 if (first->state == WAH_ENUM_RUN) {
1270 if (first->current) {
1271 wah_add1s(dest, (uint64_t)first->remain_words * 32);
1272 } else {
1273 wah_add0s(dest, (uint64_t)first->remain_words * 32);
1274 }
1275 CONSUME_ALL(first->remain_words, false);
1276 continue;
1277 }
1278
1279 p_clear(&buffer_flags, 1);
1280 tab_for_each_pos_safe(pos, &enums) {
1281 uint32_t remain = countof(buffer);
1282 uint32_t en_bits = 0;
1283 wah_word_enum_t *en = enums.tab[pos];
1284
1285 buf_pos = 0;
1286 while (en->state != WAH_ENUM_END && remain > 0) {
1287 uint32_t to_consume = MIN(remain, en->remain_words);
1288
1289 switch (en->state) {
1290 case WAH_ENUM_LITERAL: {
1291 qv_t(wah_word) *bucket = wah_word_enum_get_cur_bucket(en);
1292 const uint32_t *data = (const uint32_t *)bucket->tab;
1293
1294 data = &data[en->pos - en->remain_words];
1295 for (uint32_t i = 0; i < to_consume; i++) {
1296 if (buffer_flags[buf_pos + i] != FLAG_RUN_1) {
1297 if (buffer_flags[buf_pos + i] == FLAG_RUN_0) {
1298 buffer[buf_pos + i] = data[i];
1299 buffer_flags[buf_pos + i] = FLAG_LITTERAL;
1300 } else {
1301 buffer[buf_pos + i] |= data[i];
1302 }
1303 if (buffer[buf_pos + i] == 0xffffffff) {
1304 buffer_flags[buf_pos + i] = FLAG_RUN_1;
1305 }
1306 }
1307 }
1308 en_bits += to_consume * 32;
1309 } break;
1310
1311 case WAH_ENUM_RUN:
1312 if (en->current) {
1313 memset(&buffer_flags[buf_pos], 0xff, to_consume);
1314 }
1315 en_bits += to_consume * 32;
1316 break;
1317
1318 case WAH_ENUM_PENDING:
1319 if (buffer_flags[buf_pos] != FLAG_RUN_1) {
1320 if (buffer_flags[buf_pos] == FLAG_RUN_0) {
1321 buffer[buf_pos] = en->current;
1322 buffer_flags[buf_pos] = FLAG_LITTERAL;
1323 } else {
1324 buffer[buf_pos] |= en->current;
1325 if (buffer[buf_pos] == 0xffffffff) {
1326 buffer_flags[buf_pos] = FLAG_RUN_1;
1327 }
1328 }
1329 }
1330 en_bits += en->map->len % 32;
1331 break;
1332
1333 case WAH_ENUM_END:
1334 e_panic("this should not happen");
1335 }
1336 wah_word_enum_skip(en, to_consume);
1337 buf_pos += to_consume;
1338 remain -= to_consume;
1339 }
1340 bits = MAX(bits, en_bits);
1341 if (en->state == WAH_ENUM_END) {
1342 if (pos != enums.len - 1) {
1343 enums.tab[pos] = enums.tab[enums.len - 1];
1344 }
1345 enums.len--;
1346 }
1347 }
1348 assert (!enums.len || (bits % 32) == 0);
1349
1350 buf_pos = 0;
1351 end_pos = DIV_ROUND_UP(bits, 32);
1352 while (buf_pos < end_pos) {
1353 byte val = buffer_flags[buf_pos];
1354 uint32_t end = buf_pos + 1;
1355
1356 while (end < end_pos) {
1357 if (buffer_flags[end] != val) {
1358 break;
1359 }
1360 end++;
1361 }
1362
1363 switch (val) {
1364 case FLAG_RUN_1:
1365 wah_add1s(dest, 32 * (end - buf_pos));
1366 break;
1367
1368 case FLAG_RUN_0:
1369 wah_add0s(dest, 32 * (end - buf_pos));
1370 break;
1371
1372 case FLAG_LITTERAL:
1373 if (32 * (end - buf_pos) > bits) {
1374 wah_add_aligned(dest, (const uint8_t *)&buffer[buf_pos],
1375 bits);
1376 } else {
1377 wah_add_literal(dest, (const uint8_t *)&buffer[buf_pos],
1378 4 * (end - buf_pos));
1379 }
1380 break;
1381 }
1382
1383 bits -= 32 * (end - buf_pos);
1384 buf_pos = end;
1385 }
1386 }
1387
1388 wah_check_invariant(dest);
1389 assert (dest->len == exp_len);
1390 assert (dest->active >= min_act);
1391 assert (dest->active <= max_act);
1392 return dest;
1393}

1409 wah_check_invariant(map);
1410
1411 tab_for_each_ptr(bucket, &map->_buckets) {
1412 uint32_t pos = 0;
1413
1414 while (pos < (uint32_t)bucket->len) {
1415 wah_header_t *head = &bucket->tab[pos++].head;
1416 uint32_t count = bucket->tab[pos++].count;
1417
1418 head->bit = !head->bit;
1419 for (uint32_t i = 0; i < count; i++) {
1420 bucket->tab[pos].literal = ~bucket->tab[pos].literal;
1421 pos++;
1422 }
1423 }
1424 }
1425
1426 if ((map->len % WAH_BIT_IN_WORD) != 0) {
1427 map->_pending = ~map->_pending & BITMASK_LT(uint32_t, map->len);
1428 }
1429 map->active = map->len - map->active;
1430 wah_check_invariant(map);
1431}
1432
1433bool wah_get(const wah_t *map, uint64_t pos)
1434{
1435 qv_t(wah_word) *bucket;
1436 uint64_t count;
1437 uint32_t remain = map->len % WAH_BIT_IN_WORD;
1438 int i = 0;
1439
1440 if (pos >= map->len) {
1441 return false;
1442 }
1443 if (pos >= map->len - remain) {
1444 pos %= WAH_BIT_IN_WORD;
1445 return map->_pending & (1 << pos);
1446 }
1447
1448 bucket = &map->_buckets.tab[pos / _G.bits_in_bucket];
1449 pos %= _G.bits_in_bucket;
1450
1451 while (i < bucket->len) {
1452 wah_header_t head = bucket->tab[i++].head;
1453 uint32_t words = bucket->tab[i++].count;
1454
1455 count = head.words * WAH_BIT_IN_WORD;
1456 if (pos < count) {
1457 return !!head.bit;
1458 }
1459 pos -= count;
1460
1461 count = words * WAH_BIT_IN_WORD;
1462 if (pos < count) {
1463 i += pos / WAH_BIT_IN_WORD;
1464 pos %= WAH_BIT_IN_WORD;
1465 return !!(bucket->tab[i].literal & (1 << pos));
1466 }
1467 pos -= count;
1468 i += words;
1469 }
1470 e_panic("this should not happen");
1471}

1475
1476typedef struct from_data_ctx_t {
1477 wah_t *map;
1478
1479 pstream_t data;
1480 wah_word_t *tab;
1481 uint64_t pos;
1482
1483 qv_t(wah_word) *bucket;
1484 uint64_t bucket_len; /* (in represented bits) */
1485} from_data_ctx_t;
1486
1487static void
1488from_data_split_chunk(from_data_ctx_t *ctx, wah_header_t head, uint64_t words)
1489{
1490 /* Create a bucket if necessary. */
1491 if (!ctx->bucket) {
1492 ctx->bucket = wah_create_bucket(ctx->map, 0);
1493 }
1494
1495 /* In any case, we copy all the previous chunks. */
1496 qv_extend(ctx->bucket, ctx->tab, ctx->pos - 2);
1497
1498 /* Deal with the run. */
1499 if (ctx->bucket_len + head.words * WAH_BIT_IN_WORD > _G.bits_in_bucket) {
1500 /* Chunk's run is too big and has to be split. */
1501 while (head.words) {
1502 wah_header_t to_add = head;
1503 uint64_t avail_words;
1504
1505 ctx->map->previous_run_pos = ctx->map->last_run_pos;
1506 ctx->map->last_run_pos = ctx->bucket->len;
1507
1508 avail_words = (_G.bits_in_bucket - ctx->bucket_len)
1509 / WAH_BIT_IN_WORD;
1510 to_add.words = MIN(head.words, avail_words);
1511 qv_append(ctx->bucket, (wah_word_t){ .head = to_add });
1512
1513 ctx->bucket_len += to_add.words * WAH_BIT_IN_WORD;
1514 head.words -= to_add.words;
1515
1516 if (head.words) {
1517 /* Close this chunk, and create a new bucket. */
1518 qv_append(ctx->bucket, (wah_word_t){ .count = 0 });
1519 ctx->bucket = wah_create_bucket(ctx->map, 0);
1520 ctx->map->previous_run_pos = -1;
1521 ctx->map->last_run_pos = -1;
1522 ctx->bucket_len = 0;
1523 }
1524 }
1525 } else {
1526 /* The run fits, copy it. */
1527 qv_append(ctx->bucket, (wah_word_t){ .head = head});
1528 }
1529
1530 /* We now have to deal with the uncompressed words. */
1531 for (;;) {
1532 if (ctx->bucket_len + words * WAH_BIT_IN_WORD > _G.bits_in_bucket) {
1533 /* Split them. */
1534 uint64_t count;
1535
1536 count = (_G.bits_in_bucket - ctx->bucket_len) / WAH_BIT_IN_WORD;
1537 qv_append(ctx->bucket, (wah_word_t){ .count = count });
1538 qv_extend(ctx->bucket, &ctx->tab[ctx->pos], count);
1539
1540 ctx->bucket = wah_create_bucket(ctx->map, 0);
1541 ctx->map->previous_run_pos = -1;
1542 ctx->map->last_run_pos = 0;
1543 ctx->bucket_len = 0;
1544 head = (wah_header_t){ .words = 0 };
1545 qv_append(ctx->bucket, (wah_word_t){ .head = head });
1546
1547 ctx->pos += count;
1548 words -= count;
1549 continue;
1550 }
1551
1552 /* We can safely copy the rest of the uncompressed words. */
1553 qv_append(ctx->bucket, (wah_word_t){ .count = words });
1554 qv_extend(ctx->bucket, &ctx->tab[ctx->pos], words);
1555 ctx->bucket_len += words * WAH_BIT_IN_WORD;
1556 ctx->pos += words;
1557 break;
1558 }
1559}
1560
1561wah_t *mp_wah_init_from_data(mem_pool_t *mp, wah_t *map, pstream_t data)
1562{
1563 from_data_ctx_t ctx;
1564
1565 p_clear(map, 1);
1566 mp_qv_init(mp, &map->_buckets, 0);
1567
1568 THROW_NULL_IF(ps_len(&data) % sizeof(wah_word_t));
1569 THROW_NULL_IF(ps_len(&data) < 2 * sizeof(wah_word_t));
1570
1571 map->previous_run_pos = -1;
1572 map->last_run_pos = -1;
1573
1574 p_clear(&ctx, 1);
1575 ctx.map = map;
1576 ctx.data = data;
1577
1578 while (!ps_done(&ctx.data)) {
1579 uint64_t size = ps_len(&ctx.data) / sizeof(wah_word_t);
1580
1581 ctx.tab = (wah_word_t *)ctx.data.p;
1582 ctx.pos = 0;
1583
1584 while (ctx.pos < size - 1) {
1585 wah_header_t head = ctx.tab[ctx.pos++].head;
1586 uint64_t words = ctx.tab[ctx.pos++].count;
1587 uint64_t chunk_len = WAH_BIT_IN_WORD * (head.words + words);
1588
1589 THROW_NULL_IF(words > size || ctx.pos > size - words);
1590
1591 if (head.bit) {
1592 map->active += WAH_BIT_IN_WORD * head.words;
1593 }
1594 if (words) {
1595 map->active += membitcount(&ctx.tab[ctx.pos],
1596 words * sizeof(wah_word_t));
1597 }
1598 map->len += chunk_len;
1599
1600 if (unlikely(ctx.bucket_len + chunk_len > _G.bits_in_bucket)) {
1601 /* This wah do not respect the max length of the buckets. We
1602 * have to split this chunk and create a new bucket. */
1603 from_data_split_chunk(&ctx, head, words);
1604 } else {
1605 ctx.bucket_len += chunk_len;
1606 if (ctx.bucket) {
1607 /* We have an opened bucket, add this chunk. */
1608 map->previous_run_pos = map->last_run_pos;
1609 map->last_run_pos = ctx.bucket->len;
1610 qv_extend(ctx.bucket, &ctx.tab[ctx.pos - 2], words + 2);
1611 } else {
1612 /* No opened bucket, the chunk will be added after. */
1613 map->previous_run_pos = map->last_run_pos;
1614 map->last_run_pos = ctx.pos - 2;
1615 }
1616 ctx.pos += words;
1617 }
1618
1619 if (ctx.bucket_len >= _G.bits_in_bucket) {
1620 /* The current bucket is full, close it. */
1621 assert (ctx.bucket_len == _G.bits_in_bucket);
1622 if (!ctx.bucket) {
1623 ctx.bucket = qv_growlen(&map->_buckets, 1);
1624 qv_init_static(ctx.bucket, ctx.tab, ctx.pos);
1625 }
1626 ctx.bucket = NULL;
1627 ctx.bucket_len = 0;
1628 map->previous_run_pos = -1;
1629 map->last_run_pos = -1;
1630 goto next;
1631 }
1632
1633 if (ctx.bucket) {
1634 goto next;
1635 }
1636 }
1637
1638 THROW_NULL_IF(ctx.pos != size);
1639 assert (!ctx.bucket);
1640 ctx.bucket = qv_growlen(&map->_buckets, 1);
1641 qv_init_static(ctx.bucket, ctx.tab, ctx.pos);
1642
1643 next:
1644 ps_skip(&ctx.data, ctx.pos * sizeof(wah_word_t));
1645 }
1646
1647 wah_check_invariant(map);
1648 return map;
1649}
