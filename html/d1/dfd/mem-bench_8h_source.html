66/** Initialize mem_bench object.
67 *
68 * The mem_bench object is initialized to dump into a file
69 * each \p period iterations. The filename is
70 * "./mem.[\p name].[pid].[address of \p sp]".
71 * "[\p type].[address of \p sp] is also used for logger name.
72 *
73 * \param[in] type Name of mem_bench object.
74 * \param[in] period Logging period. 0 means no logging.
75 */
76mem_bench_t *mem_bench_init(mem_bench_t *sp, lstr_t type, uint32_t period);
77
78/** Wipes a mem_bench object.
79 *
80 * Note : the mem_bench object is still usable after call to this method,
81 * but any operation on it will be no-op.
82 */
83void mem_bench_wipe(mem_bench_t *sp);
84
85__unused__
86static inline mem_bench_t *mem_bench_new(lstr_t type, uint32_t period)
87{
88 return mem_bench_init(p_new_raw(mem_bench_t, 1), type, period);
89}
90GENERIC_DELETE(mem_bench_t, mem_bench)
91
92/** Declare a mem_bench object to be leaked.
93 *
94 * If you cannot ensure the mem_bench object will be
95 * wiped before the log module termination
96 * (if it must be destroyed by a thr_hook for example),
97 * it must be registered with this function.
98 * It will do a partial wipe, allowing to work properly.
99 *
100 * The mem_bench object must still be mem_bench_wipe'd manually
101 * to finish the cleanup.
102 *
103 * Wipe time is unspecified, and the allocated memory
104 * will not be reclaimed.
105 */
106void mem_bench_leak(mem_bench_t *sp);

108/** Update state of the mem-bench object.

118/** Print stats in human-readable form.
