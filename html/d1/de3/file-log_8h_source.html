47enum log_file_event {
48 LOG_FILE_CREATE, /* just after a new file creation */
49 LOG_FILE_CLOSE, /* called after file_close is called */
50 LOG_FILE_DELETE, /* called after a log file is deleted. */
51 LOG_FILE_ROTATE, /* at rotation, after the log file is closed,

57/* Log file event callback. fpath is built using the prefix passed during

160static inline off_t log_file_tell(log_file_t *log_file)
161{
162 if (!log_file->_internal) {
163 errno = EBADF;
164 return -1;
165 }
166 if (log_file->is_file_bin) {
167 return log_file->_bin_internal->cur;
168 } else {
169 return file_tell(log_file->_internal);
170 }
171}

175/** Write logs in the current file in a transaction-like mode.
176 *
177 * This function should be used for logs we don't want to split on two (or
178 * more) files because of the rotation mechanism.
179 *
180 * This function executes the block given as an argument with file rotations

183 *
184 * XXX Warning: we speak about transaction here because of the rollback
185 * capability of the feature. For example, this function does *not* unlock the
186 * possibility to write logs from different threads using the same log_file_t.
187 *
188 * \param[in] file The log_file_t.
189 * \param[in] log_b The callback block.
190 *
191 * \return 0 on success, a negative value on failure.
