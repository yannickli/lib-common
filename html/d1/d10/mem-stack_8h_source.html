27/*
28 * Stacked memory allocator
29 * ~~~~~~~~~~~~~~~~~~~~~~~~
30 *
31 * This allocator mostly has the same properties as the GNU Obstack have.
32 *
33 * This mostly works like alloca() wrt your stack, except that it's a chain of
34 * malloc()ed blocks. It also works like alloca in the sense that it aligns
35 * allocated memory bits to the lowest required alignment possible (IOW
36 * allocating blocks of sizes < 2, < 4, < 8, < 16 or >= 16 yield blocks aligned
37 * to a 1, 2, 4, 8, and 16 boundary respectively.
38 *
39 * Additionnally to that, you have mem_stack_{push,pop} APIs that push/pop new
40 * stack frames (ring a bell ?). Anything that has been allocated since the
41 * last push is freed when you call pop.
42 *
43 * push/pop return void* cookies that you can match if you want to be sure
44 * you're not screwing yourself with non matching push/pops. Matching push/pop
45 * should return the same cookie value.
46 *
47 * Note that a pristine stacked pool has an implicit push() done, with a
48 * matching cookie of `NULL`. Note that this last pop rewinds the stack pool
49 * into its pristine state in the sense that it's ready to allocate memory and
50 * no further push() is needed (an implicit push is done here too).
51 *
52 * Reallocing the last allocated data is efficient in the sense that it tries
53 * to keep the data at the same place.
54 *
55 *
56 * Physical memory is reclaimed based on different heuristics. pop() is not
57 * enough to reclaim the whole pool memory, only deleting the pool does that.
58 * The pool somehow tries to keep the largest chunks of data allocated around,
59 * and to discard the ones that are definitely too small to do anything useful
60 * with them. Also, new blocks allocated are always bigger than the last
61 * biggest block that was allocated (or of the same size).
62 *
63 *
64 * A word on how it works, there is a list of chained blocks, with the huge
65 * kludge that the pool looks like a block enough to be one (it allows to have
66 * no single point and that the list of blocks is never empty even when there
67 * is no physical block of memory allocated). The pool also contains the base
68 * frame so that the base frame doesn't prevent any block from being collected
69 * at any moment.
70 *
71 * [pool]==[blk 1]==[blk 2]==...==[blk n]
72 * \\ //
73 * \=============================/
74 *
75 * In addition to the based frame pointer, The pool contains a stack of
76 * chained frames. The frames are allocated into the stacked allocator,
77 * except for the base one. Frames points to the first octet in the block
78 * "rope" that is free. IOW it looks like this:
79 *
80 * [ fake block 0 == pool ] [ octets of block 1 ] [ block 2 ] .... [ block n ]
81 * base_ frame1_ frame2_ |
82 * \____________________/ \_/ \_______________________/
83 *
84 * consecutive frames may or may not be in the same physical block.
85 * the bottom of a frame may or may not be in the same physical block where it
86 * lives.
87 */

164 * sealing a stack frame ensures that people wanting to allocate in that stack

200static ALWAYS_INLINE
201const void * nonnull mem_stack_pool_pop(mem_stack_pool_t * nonnull sp)
202{
203 mem_stack_frame_t *frame = sp->stack;
204
205#ifndef NDEBUG
206 /* bypass mem_pool if demanded */
207 if (!mem_pool_is_enabled()) {
208 return mem_stack_pool_pop_libc(sp);
209 }
210#endif
211
212 sp->stack = mem_stack_pool_prev(frame);
213#ifdef MEM_BENCH
214 mem_stack_pool_bench_pop(sp, frame);
215#endif
216 assert (sp->stack);
217 mem_stack_pool_protect(sp, frame);
218
219 if (mem_stack_pool_is_at_top(sp)) {
220 mem_stack_pool_try_reset(sp);
221 }
222 return frame;
223}

320#ifndef __cplusplus
321/*
322 * t_scope protects all the code after its use up to the end of the block
323 * scope with an implicit mem_stack_pool_push(), mem_stack_pool_pop() pair.
324 *
325 * It works using the same principle as C++ RAII, see the C++ TScope class
326 * below, but for C.
327 *
328 * As a rule, it's better to use `t_scope;` just after the scope opening so
329 * that it gards the full block properly, e.g.:
330 *
331 * {
332 * t_scope;
333 * char *buf = t_new_raw(char, BUFSIZ); // safe
334 *
335 * // ...
336 * }
337 */
338static ALWAYS_INLINE
339void t_scope_cleanup(const void * nonnull * nonnull unused)
340{
341#ifndef NDEBUG
342 if (unlikely(*unused != mem_stack_pool_pop(&t_pool_g)))
343 e_panic("unbalanced t_stack");
344#else
345 mem_stack_pool_pop(&t_pool_g);
346#endif
347}
348#define t_scope \
349 const void *PFX_LINE(t_scope_) \
350 __attribute__((unused,cleanup(t_scope_cleanup))) \
351 = mem_stack_pool_push(&t_pool_g)
352#else
353/*
354 * RAII scoped mem_stask_pool_push/mem_stack_pool_pop
355 */
356class TScope {
357 public:
358 inline TScope() { mem_stack_pool_push(&t_pool_g); };
359 inline ~TScope() { mem_stack_pool_pop(&t_pool_g); };
360 private:
361 DISALLOW_COPY_AND_ASSIGN(TScope);
362 void* null_unspecified operator new(size_t);
363 void operator delete(void * null_unspecified , size_t);
364};
365#define t_scope \
366 TScope PFX_LINE(t_scope_because_cpp_sucks_donkeys_)
367#endif
