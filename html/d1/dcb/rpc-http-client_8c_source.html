115static int http_iop_get_ready_remote(http_iop_channel_t *channel,
116 bool do_connection,
117 http_iop_channel_remote_t **remote_ptr,
118 httpc_t **httpc_ptr)
119{
120 tab_for_each_entry(remote, &channel->remotes) {
121 httpc_t *httpc;
122
123 if (!do_connection && !httpc_pool_has_ready(&remote->pool)) {
124 /* We don't want to recreate the connections and the remote
125 * doesn't have a ready connection.
126 */
127 continue;
128 }
129
130 httpc = httpc_pool_get(&remote->pool);
131 if (httpc) {
132 *remote_ptr = remote;
133 *httpc_ptr = httpc;
134 return 0;
135 }
136 }
137 return -1;
138}

140static void http_iop_restart_messages(http_iop_channel_t *channel)
141{
142 htlist_t temp;
143
144 htlist_move(&temp, &channel->queries_waiting_conn);
145 while (!htlist_is_empty(&temp)) {
146 http_iop_channel_remote_t *remote;
147 httpc_t *httpc;
148 http_iop_msg_t *msg;
149
150 if (http_iop_get_ready_remote(channel, false, &remote, &httpc) < 0) {
151 htlist_splice(&channel->queries_waiting_conn, &temp);
152 return;
153 }
154
155 msg = htlist_pop_entry(&temp, http_iop_msg_t, link);
156 http_iop_start_msg(channel, remote, httpc, msg, msg->args);
157 }
158}

178static bool http_iop_channel_has_remote_connections(
179 const http_iop_channel_t *channel,
180 const http_iop_channel_remote_t *disconnecting_remote)
181{
182 tab_for_each_entry(remote, &channel->remotes) {
183 int len = remote->pool.len;
184
185 if (remote == disconnecting_remote) {
186 /* One connection is being disconnected. */
187 len--;
188 }
189
190 if (len > 0) {
191 return true;
192 }
193 }
194
195 return false;
196}

198static void http_iop_on_connect_error(const httpc_t *httpc, int errnum)
199{
200 http_iop_channel_remote_t *remote;
201 http_iop_channel_t *channel;
202
203 remote = container_of(httpc->pool, http_iop_channel_remote_t, pool);
204 channel = remote->channel;
205
206 logger_trace(&_G.logger, 1, "connection on remote `%*pM` error: %s",
207 LSTR_FMT_ARG(remote->pool.host), strerror(errnum));
208
209 http_iop_restart_messages(channel);
210
211 if (channel->on_connection_error_cb) {
212 (*channel->on_connection_error_cb)(remote, errnum);
213 }
214
215 if (!http_iop_channel_has_remote_connections(channel, remote)) {
216 /* Cancel remaining messages when there are no available remote
217 * connections anymore. */
218 while (!htlist_is_empty(&channel->queries_waiting_conn)) {
219 http_iop_msg_t *msg;
220
221 msg = htlist_first_entry(&channel->queries_waiting_conn,
222 http_iop_msg_t, link);
223 msg->cb(msg, IC_MSG_CANCELED, (opt_http_code_t)OPT_NONE, NULL,
224 NULL);
225 htlist_pop_entry(&channel->queries_waiting_conn,
226 http_iop_msg_t, link);
227 http_iop_msg_delete(&msg);
228 }
229 }
230}

235static void http_iop_timeout_queries(el_t el, data_t data)
236{
237 http_iop_channel_t *channel = data.ptr;
238 struct timeval now;
239
240 channel->queries_conn_timeout_el = NULL;
241 lp_gettv(&now);
242
243 while (!htlist_is_empty(&channel->queries_waiting_conn)) {
244 http_iop_msg_t *msg;
245 struct timeval msg_expiry;
246 struct timeval time_left;
247
248 msg = htlist_first_entry(&channel->queries_waiting_conn,
249 http_iop_msg_t, link);
250 msg_expiry = timeval_addmsec(msg->query_time,
251 channel->connection_timeout_msec);
252 if (is_expired(&msg_expiry, &now, &time_left)) {
253 logger_trace(&_G.logger, 1, "canceling query `%*pM`: connection "
254 "timeout reached", LSTR_FMT_ARG(msg->rpc->name));
255 msg->cb(msg, IC_MSG_TIMEDOUT, (opt_http_code_t)OPT_NONE, NULL,
256 NULL);
257 htlist_pop_entry(&channel->queries_waiting_conn, http_iop_msg_t,
258 link);
259 http_iop_msg_delete(&msg);
260 } else {
261 int64_t next_check = timeval_to_msec(time_left);
262
263 http_iop_register_timeout_check(channel, next_check);
264 break;
265 }
266 }
267}

272 channel->queries_conn_timeout_el = el_unref(

282static ic_status_t ic_status_from_httpc_status(httpc_status_t status)
283{
284 switch (status) {
285 case HTTPC_STATUS_OK:
286 return IC_MSG_OK;
287
288 case HTTPC_STATUS_EXP100CONT:
289 /* This does not seem possible to have this code here. This is only
290 * set if the httpc_query has set expect100cont (which is not the
291 * case in this module.
292 */
293 assert(false);
294 /* FALLTHROUGH */
295
296 case HTTPC_STATUS_INVALID:
297 return IC_MSG_SERVER_ERROR;
298
299 case HTTPC_STATUS_ABORT:
300 return IC_MSG_ABORT;
301
302 case HTTPC_STATUS_TOOLARGE:
303 return IC_MSG_CANCELED;
304
305 case HTTPC_STATUS_TIMEOUT:
306 return IC_MSG_TIMEDOUT;
307 }
308
309 return IC_MSG_INVALID;
310}

314static void
315http_iop_on_query_done(httpc_query_t *http_query, httpc_status_t httpc_status)
316{
317 t_scope;
318 http_iop_msg_t *msg;
319 opt_http_code_t http_code = OPT_NONE;
320 void *exn = NULL;
321 void *res = NULL;
322 ic_status_t ic_status;
323 SB_1k(err);
324
325 msg = container_of(http_query, http_iop_msg_t, query);
326
327 logger_trace(&_G.logger, 1, "query `%*pM` finished (%d)",
328 LSTR_FMT_ARG(msg->rpc->name), httpc_status);
329 if (http_query->qinfo) {
330 logger_trace(&_G.logger, 2, "query `%*pM` HTTP code: %d",
331 LSTR_FMT_ARG(msg->rpc->name), http_query->qinfo->code);
332 logger_trace(&_G.logger, 3, "payload: `%*pM`",
333 SB_FMT_ARG(&http_query->payload));
334 OPT_SET(http_code, http_query->qinfo->code);
335 }
336
337 ic_status = ic_status_from_httpc_status(httpc_status);
338 if (http_query->payload.len) {
339 pstream_t ps = ps_initsb(&http_query->payload);
340 const iop_struct_t *st = msg->rpc->result;
341 void **dest = &res;
342 bool content_type_json = false;
343 lstr_t content_type = LSTR_NULL_V;
344 const http_qhdr_t *ctype;
345
346 if (!http_code_is_successful(http_query->qinfo->code)) {
347 st = msg->rpc->exn;
348 dest = &exn;
349 ic_status = IC_MSG_EXN;
350 }
351 ctype = http_qhdr_find(http_query->qinfo->hdrs,
352 http_query->qinfo->hdrs_len,
353 HTTP_WKHDR_CONTENT_TYPE);
354 if (ctype) {
355 pstream_t v = ctype->val;
356
357 /* TODO: should be factorized with is_ctype_json. */
358 ps_skipspaces(&v);
359 content_type_json = ps_startswithstr(&ctype->val,
360 CONTENT_TYPE_JSON);
361 content_type = LSTR_PS_V(&ctype->val);
362 }
363
364 if (content_type_json) {
365 if (t_iop_junpack_ptr_ps(&ps, st, dest, 0, &err) < 0) {
366 logger_error(&_G.logger, "cannot unpack result of query "
367 "`%*pM`: %*pM", LSTR_FMT_ARG(msg->rpc->name),
368 SB_FMT_ARG(&err));
369 ic_status = IC_MSG_INVALID;
370 *dest = NULL;
371 }
372 } else {
373 ic_status = IC_MSG_INVALID;
374 logger_error(&_G.logger, "invalid or missing content-type "
375 "received from server for query `%*pM`: `%*pM` "
376 "(code %d)",
377 LSTR_FMT_ARG(msg->rpc->name),
378 LSTR_FMT_ARG(content_type),
379 http_query->qinfo->code);
380 }
381 } else if (httpc_status == HTTPC_STATUS_OK && http_query->qinfo &&
382 http_code_is_successful(http_query->qinfo->code))
383 {
384 ic_status = IC_MSG_INVALID;
385 logger_error(&_G.logger, "invalid reply from server, empty payload "
386 "for a succesful query");
387 }
388
389 msg->cb(msg, ic_status, http_code, res, exn);
390
391 http_iop_msg_delete(&msg);
392}

394static void http_iop_start_msg(http_iop_channel_t *channel,
395 http_iop_channel_remote_t *remote,
396 httpc_t *httpc, http_iop_msg_t *msg,
397 void *args)
398{
399 SB_1k(sb);
400 SB_1k(query_data);
401 outbuf_t *ob;
402
403 if (htlist_is_empty(&channel->queries_waiting_conn)) {
404 el_unregister(&channel->queries_conn_timeout_el);
405 }
406 if (channel->response_max_size) {
407 httpc_bufferize(&msg->query, channel->response_max_size);
408 }
409 msg->query.on_done = http_iop_on_query_done;
410 httpc_query_attach(&msg->query, httpc);
411 if (channel->encode_url) {
412 sb_add_urlencode(&sb, remote->base_path.s, remote->base_path.len);
413 sb_addc(&sb, '/');
414 sb_add_urlencode(&sb, msg->iface_alias->name.s,
415 msg->iface_alias->name.len);
416 sb_addc(&sb, '/');
417 sb_add_urlencode(&sb, msg->rpc->name.s, msg->rpc->name.len);
418 } else {
419 sb_addf(&sb, "/%*pM/%*pM/%*pM", LSTR_FMT_ARG(remote->base_path),
420 LSTR_FMT_ARG(msg->iface_alias->name),
421 LSTR_FMT_ARG(msg->rpc->name));
422 }
423
424 httpc_query_start_flags(&msg->query, HTTP_METHOD_POST,
425 remote->pool.host, LSTR_SB_V(&sb), false);
426
427 if (channel->user.len && channel->password.len) {
428 httpc_query_hdrs_add_auth(&msg->query, channel->user,
429 channel->password);
430 }
431
432 ob = httpc_get_ob(&msg->query);
433 ob_adds(ob, "Content-Type: application/json\r\n");
434 httpc_query_hdrs_done(&msg->query, -1, false);
435 iop_sb_jpack(&query_data, msg->rpc->args, args, 0);
436 ob_addsb(ob, &query_data);
437
438 logger_trace(&_G.logger, 1, "%*pM/%*pM: `%*pM`",
439 LSTR_FMT_ARG(msg->iface_alias->name),
440 LSTR_FMT_ARG(msg->rpc->name), SB_FMT_ARG(&query_data));
441 httpc_query_done(&msg->query);
442}

447http_iop_channel_t *
448http_iop_channel_create(const http_iop_channel_cfg_t *cfg, sb_t *err)
449{
450 http_iop_channel_t *res = http_iop_channel_new();
451
452 res->connection_timeout_msec = OPT_DEFVAL(cfg->connection_timeout_msec,
453 10 * 1000);
454 res->response_max_size = OPT_DEFVAL(cfg->response_max_size, 1 << 20);
455 res->encode_url = OPT_DEFVAL(cfg->encode_url, true);
456 res->name = lstr_dup(cfg->name);
457 res->user = lstr_dup(cfg->user);
458 res->password = lstr_dup(cfg->password);
459 res->on_connection_error_cb = cfg->on_connection_error_cb;
460 res->on_ready_cb = cfg->on_ready_cb;
461 res->priv = cfg->priv;
462
463 if (!cfg->urls.len) {
464 sb_setf(err, "there must be at least one URL");
465 goto error;
466 }
467
468 tab_for_each_entry(url, &cfg->urls) {
469 http_iop_channel_remote_t *remote;
470
471 remote = http_iop_channel_remote_new();
472 if (parse_http_url(url.s, true, &remote->url) < 0) {
473 sb_setf(err, "cannot parse URL `%*pM`", LSTR_FMT_ARG(url));
474 http_iop_channel_remote_delete(&remote);
475 goto error;
476 }
477
478 remote->pool.name = lstr_dupc(res->name);
479 remote->pool.host = lstr_fmt("%s:%d", remote->url.host,
480 remote->url.port);
481 remote->pool.resolve_on_connect = true;
482
483 remote->base_path = LSTR(remote->url.path_without_args);
484 if (httpc_cfg_from_iop(remote->pool.cfg, cfg->iop_cfg) < 0) {
485 sb_sets(err, "cannot create channel from IOP configuration");
486 lstr_wipe(&remote->pool.host);
487 http_iop_channel_remote_delete(&remote);
488 goto error;
489 }
490
491 remote->pool.max_len = OPT_DEFVAL(cfg->max_connections, 1);
492 remote->pool.on_ready = http_iop_on_connection_ready;
493 remote->pool.on_connect_error = http_iop_on_connect_error;
494 remote->channel = res;
495
496 qv_append(&res->remotes, remote);
497 }
498
499 return res;
500
501error:
502 http_iop_channel_delete(&res);
503 return NULL;
504}

513void http_iop_query_(http_iop_channel_t *channel, http_iop_msg_t *msg,
514 void *args)
515{
516 http_iop_channel_remote_t *remote;
517 httpc_t *httpc;
518
519#ifndef NDEBUG
520 /* If that crashes, one of the IC_MSG_ABORT callback on wipe reenqueues
521 * directly in that channel which is forbidden, fix the code.
522 */
523 assert(!channel->wipe_guard);
524#endif /* NDEBUG */
525
526 if (timeval_is_eq0(msg->query_time)) {
527 lp_gettv(&msg->query_time);
528 }
529 if (http_iop_get_ready_remote(channel, true, &remote, &httpc) < 0) {
530 logger_trace(&_G.logger, 1,
531 "no connection ready, query `%*pM` will wait for "
532 "connection", LSTR_FMT_ARG(msg->rpc->name));
533 if (!msg->args) {
534 msg->args = mp_iop_dup_desc_flags_sz(NULL, msg->rpc->args, args,
535 0, NULL);
536 }
537 htlist_add_tail(&channel->queries_waiting_conn, &msg->link);
538 if (!channel->queries_conn_timeout_el) {
539 http_iop_register_timeout_check(
540 channel, channel->connection_timeout_msec);
541 }
542 return;
543 }
544
545 http_iop_start_msg(channel, remote, httpc, msg, args);
546}
