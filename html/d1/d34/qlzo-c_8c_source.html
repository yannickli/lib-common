22#define D_BITS 14
23#define D_MASK ((1u << D_BITS) - 1)
24#define D_HIGH ((D_MASK >> 1) + 1)

26static ALWAYS_INLINE uint32_t HASH3(const uint8_t *p)
27{
28 const uint32_t s1 = 5;
29 const uint32_t s2 = 5;
30 const uint32_t s3 = 6;
31
32 uint32_t p0 = p[0], p1 = p[1], p2 = p[2], p3 = p[3];
33 uint32_t h;
34
35 h = (p3 << (s1 + s2 + s3))
36 ^ (p2 << (s1 + s2))
37 ^ (p1 << (s1))
38 ^ (p0);
39
40 h = (h << 5) + h;
41 return (h >> 5) & D_MASK;
42}
43
44static ALWAYS_INLINE uint32_t HASH3_SECONDARY(uint32_t h)
45{
46 return (h & (D_MASK & 0x7ff)) ^ (D_HIGH | 0x1f);
47}

49static ALWAYS_INLINE uint8_t *
50lzo_put_varlen(uint8_t *out, unsigned sz, uint32_t mask, uint32_t marker)
51{
52 if (sz <= mask) {
53 *out++ = marker | sz;
54 return out;
55 }
56 *out++ = marker;
57 sz -= mask;
58 while (sz > 255) {
59 sz -= 255;
60 *out++ = 0;
61 }
62
63 *out++ = sz;
64 return out;
65}

67static ALWAYS_INLINE uint8_t *
68lzo_put_m1(uint8_t *out, const uint8_t *in, uint32_t sz)
69{
70 if (sz <= 3) {
71 out[-2] |= sz;
72 out[0] = in[0];
73 if (sz > 1) {
74 out[1] = in[1];
75 if (sz > 2)
76 out[2] = in[2];
77 }
78 return out + sz;
79 }
80
81 out = lzo_put_varlen(out, sz - 3, 15, 0);
82 return mempcpy(out, in, sz);
83}

85static ALWAYS_INLINE uint8_t *compress(uint8_t *out, pstream_t *in, void *buf)
86{
87 const uint8_t * const orig_in = in->b;
88 const uint8_t * const ip_end = in->b_end - LZO_M2_MAX_LEN - 5;
89 unsigned * const dict = buf;
90
91 const uint8_t *ii = in->b;
92 uint32_t dindex;
93
94 __ps_skip(in, 4);
95 dindex = HASH3(in->b);
96 goto literal;
97
98 while (likely(in->b < ip_end)) {
99 uint32_t word;
100 uint32_t m_off, m_len, m_len_max;
101 const uint8_t *m_pos;
102
103 word = get_unaligned_cpu32(in->b);
104 dindex = HASH3(in->b);
105 m_pos = orig_in + dict[dindex];
106
107 if ((size_t)(m_pos + LZO_M4_MAX_OFFSET - in->b) >= LZO_M4_MAX_OFFSET)
108 goto literal;
109
110 if (get_unaligned_cpu32(m_pos) == word)
111 goto match;
112
113 dindex = HASH3_SECONDARY(dindex);
114 m_pos = orig_in + dict[dindex];
115
116 if ((size_t)(m_pos + LZO_M4_MAX_OFFSET - in->b) >= LZO_M4_MAX_OFFSET
117 || get_unaligned_cpu32(m_pos) != word)
118 {
119literal:
120 dict[dindex] = in->b++ - orig_in;
121 continue;
122 }
123
124match:
125 dict[dindex] = in->b - orig_in;
126 if (in->b != ii)
127 out = lzo_put_m1(out, ii, in->b - ii);
128
129 m_len_max = ps_len(in);
130 for (m_len = 4; m_len + 2 <= m_len_max; m_len += 2) {
131 if (get_unaligned_cpu16(m_pos + m_len) !=
132 get_unaligned_cpu16(in->b + m_len))
133 {
134 break;
135 }
136 }
137 m_len += m_len < m_len_max && m_pos[m_len] == in->b[m_len];
138
139 m_off = in->b - m_pos;
140 if (m_len <= LZO_M2_MAX_LEN) {
141 if (m_off <= LZO_M2_MAX_OFFSET) {
142 m_off -= 1;
143 *out++ = ((m_len - 1) << 5) | ((m_off & 7) << 2);
144 *out++ = (m_off >> 3);
145 goto m2_offset_already_done;
146 } else if (m_off <= LZO_M3_MAX_OFFSET) {
147 m_off -= 1;
148 *out++ = LZO_M3_MARKER | (m_len - 2);
149 } else {
150 m_off -= LZO_M3_MAX_OFFSET;
151 *out++ = LZO_M4_MARKER | ((m_off >> 11) & 8) | (m_len - 2);
152 }
153 } else {
154 if (m_off <= LZO_M3_MAX_OFFSET) {
155 m_off -= 1;
156 out = lzo_put_varlen(out, m_len - 2, 31, LZO_M3_MARKER);
157 } else {
158 m_off -= LZO_M3_MAX_OFFSET;
159 out = lzo_put_varlen(out, m_len - 2, 7,
160 LZO_M4_MARKER | ((m_off >> 11) & 8));
161 }
162 }
163 out = put_unaligned_le16(out, (m_off << 2));
164
165m2_offset_already_done:
166 __ps_skip(in, m_len);
167 ii = in->b;
168 }
169
170 in->b = ii;
171 return out;
172}

174size_t qlzo1x_compress(void *orig_out, size_t outlen, pstream_t in, void *buf)
175{
176 uint8_t *out = orig_out;
177 size_t t;
178
179 /* XXX: initializing the dictionnary is absolutely not useful,
180 * algorithm continues to work properly with random data in.
181 *
182 * but having valgrind complain sucks during debugging.
183 */
184 if (mem_tool_is_running(MEM_TOOL_VALGRIND))
185 memset(buf, 0, LZO_BUF_MEM_SIZE);
186
187 if (likely(ps_has(&in, LZO_M2_MAX_LEN + 5)))
188 out = compress(out, &in, buf);
189 t = ps_len(&in);
190 if (t > 0) {
191 if (out == orig_out && t <= 238) {
192 *out++ = (17 + t);
193 out = mempcpy(out, in.b, t);
194 } else {
195 out = lzo_put_m1(out, in.b, t);
196 }
197 }
198
199 out[0] = LZO_M4_MARKER | 1;
200 out[1] = 0;
201 out[2] = 0;
202 return out + 3 - (uint8_t *)orig_out;
203}
