52typedef enum module_state_t {

122static void
123set_require_type(module_t *module, module_t *required_by)
124{
125 if (!required_by) {
126 module->state = MANU_REQ;
127 module->manu_req_count++;
128 } else {
129 qv_append(&module->required_by, required_by);
130 if (module->state != MANU_REQ)
131 module->state = AUTO_REQ;
132 }

135module_t *module_register(lstr_t name)
136{
137 module_t *module;
138 uint32_t pos;
139
140 pos = qm_reserve(module, &_G.modules, &name, 0);
141 if (!expect(!(pos & QHASH_COLLISION))) {
142 logger_error(&_G.logger, "%*pM has already been registered",
143 LSTR_FMT_ARG(name));
144 return _G.modules.values[pos & ~QHASH_COLLISION];
145 }
146
147 module = module_new();
148 module->state = REGISTERED;
149 module->name = lstr_dup(name);
150
151 _G.modules.keys[pos] = &module->name;
152 _G.modules.values[pos] = module;
153
154 return module;
155}

180static int modules_topo_visit(qh_t(module) *temporary_mark,
181 qh_t(module) *permanent_mark,
182 qv_t(module) *ordered_modules,
183 module_t *m, sb_t *err)
184{
185 if (qh_find(module, permanent_mark, m) >= 0) {
186 return 0;
187 }
188 if (qh_add(module, temporary_mark, m) < 0) {
189 sb_addf(err, "-> %*pM", LSTR_FMT_ARG(m->name));
190 return -1;
191 }
192 tab_for_each_entry(dep, &m->dependent_of) {
193 if (modules_topo_visit(temporary_mark, permanent_mark,
194 ordered_modules, dep, err) < 0)
195 {
196 sb_prependf(err, "-> %*pM", LSTR_FMT_ARG(m->name));
197 return -1;
198 }
199 }
200 qh_add(module, permanent_mark, m);
201 qv_append(ordered_modules, m);
202 return 0;
203}
204
205static int
206modules_topo_sort_rev(qm_t(module) *modules, qv_t(module) *sorted, sb_t *err)
207{
208 int ret = 0;
209 qh_t(module) temporary_mark;
210 qh_t(module) permanent_mark;
211
212 qh_init(module, &temporary_mark);
213 qh_init(module, &permanent_mark);
214
215 qm_for_each_pos(module, pos, &_G.modules) {
216 module_t *m = _G.modules.values[pos];
217
218 if (qh_find(module, &temporary_mark, m) < 0
219 && qh_find(module, &permanent_mark, m) < 0)
220 {
221 if (modules_topo_visit(&temporary_mark, &permanent_mark, sorted,
222 m, err) < 0)
223 {
224 sb_prepends(err, "module dependency error: ");
225 ret = -1;
226 goto end;
227 }
228 }
229 }
230
231 end:
232 qh_wipe(module, &temporary_mark);
233 qh_wipe(module, &permanent_mark);
234 return ret;
235}

237void module_require(module_t *module, module_t *required_by)
238{
239 if (module->state == INITIALIZING) {
240 logger_fatal(&_G.logger,
241 "`%*pM` has been recursively required %s%*pM",
242 LSTR_FMT_ARG(module->name), required_by ? "by " : "",
243 LSTR_FMT_ARG(required_by ? required_by->name
244 : LSTR_NULL_V));
245 }
246 if (module->state == SHUTTING) {
247 logger_fatal(&_G.logger,
248 "`%*pM` has been required %s%*pM while shutting down",
249 LSTR_FMT_ARG(module->name), required_by ? "by " : "",
250 LSTR_FMT_ARG(required_by ? required_by->name
251 : LSTR_NULL_V));
252 }
253
254 _G.in_initialization++;
255
256 if (!module_is_loaded(module)) {
257 logger_trace(&_G.logger, 1, "`%*pM` has been required %s%*pM",
258 LSTR_FMT_ARG(module->name), required_by ? "by " : "",
259 LSTR_FMT_ARG(required_by ? required_by->name
260 : LSTR_NULL_V));
261 }
262
263 if (module->state == AUTO_REQ || module->state == MANU_REQ) {
264 set_require_type(module, required_by);
265 _G.in_initialization--;
266 return;
267 }
268
269 module->state = INITIALIZING;
270 logger_trace(&_G.logger, 1, "requiring `%*pM` dependencies",
271 LSTR_FMT_ARG(module->name));
272
273 _G.methods_dirty = true;
274
275 tab_for_each_entry(dep, &module->dependent_of) {
276 module_require(dep, module);
277 }
278
279 logger_trace(&_G.logger, 1, "calling `%*pM` constructor",
280 LSTR_FMT_ARG(module->name));
281
282 if ((*module->constructor)(module->constructor_argument) < 0) {
283 logger_fatal(&_G.logger, "unable to initialize %*pM",
284 LSTR_FMT_ARG(module->name));
285 }
286
287 set_require_type(module, required_by);
288 _G.methods_dirty = true;
289 _G.in_initialization--;
290}

306__attr_nonnull__((1))
307static int module_shutdown(module_t *module);
308
309static int notify_shutdown(module_t *module, module_t *dependence)
310{
311 logger_trace(&_G.logger, 2, "module '%*pM' notify shutdown to '%*pM'"
312 ": %d pending dependencies", LSTR_FMT_ARG(dependence->name),
313 LSTR_FMT_ARG(module->name), module->required_by.len);
314
315 tab_enumerate(pos, dep, &module->required_by) {
316 if (dep == dependence) {
317 qv_remove(&module->required_by, pos);
318 break;
319 }
320 }
321 if (module->required_by.len == 0 && module->state != MANU_REQ) {
322 return module_shutdown(module);
323 }
324
325 return 0;
326}

328/** \brief Shutdown a module
329 *
330 * Two steps : - Shutdown the module.
331 * - Notify dependent modules that it has been shutdown
332 * if the dependent modules don't have any other parent
333 * modules and they have been automatically initialize
334 * they will shutdown
335 *
336 * If the module is not able to shutdown (destructor returns a negative
337 * number), module state change to FAIL_SHUT but we considered as shutdown
338 * and notify dependent modules.
339 *
340 *
341 * @param mod The module to shutdown
342 */
343__attr_nonnull__((1))
344static int module_shutdown(module_t *module)
345{
346 int shut_dependent = 1;
347 int shut_self;
348
349 assert (module->state == MANU_REQ || module->state == AUTO_REQ);
350
351 /* shutdown must be symmetric to require */
352 module->manu_req_count = 0;
353
354 module->state = SHUTTING;
355 logger_trace(&_G.logger, 1, "shutting down `%*pM`",
356 LSTR_FMT_ARG(module->name));
357
358 if ((shut_self = (*module->destructor)()) < 0) {
359 logger_warning(&_G.logger, "unable to shutdown %*pM",
360 LSTR_FMT_ARG(module->name));
361 module->state = FAIL_SHUT;
362 }
363
364 _G.methods_dirty = true;
365
366 tab_for_each_entry(dep, &module->dependent_of) {
367 int shut;
368
369 shut = notify_shutdown(dep, module);
370 if (shut < 0) {
371 shut_dependent = shut;
372 }
373 }
374
375 if (shut_self >= 0) {
376 module->state = REGISTERED;
377 }
378
379 RETHROW(shut_dependent);
380 RETHROW(shut_self);
381 return 0;
382}

384void module_release(module_t *module)
385{
386 if (module->manu_req_count == 0) {
387 /* You are trying to manually release a module that have been spawn
388 * automatically (AUTO_REQ)
389 */
390 logger_panic(&_G.logger, "unauthorized release for module '%*pM'",
391 LSTR_FMT_ARG(module->name));
392 return;
393 }
394
395 if (module->state == MANU_REQ && module->manu_req_count > 1) {
396 module->manu_req_count--;
397 return;
398 }
399
400 if (module->state == MANU_REQ && module->manu_req_count == 1) {
401 if (module->required_by.len > 0) {
402 module->manu_req_count = 0;
403 module->state = AUTO_REQ;
404 return;
405 }
406 }
407
408 module_shutdown(module);
409}

432static void module_hard_shutdown(void)
433{
434 /* Shutdown manually required modules that were not released. */
435 qm_for_each_pos(module, pos, &_G.modules) {
436 module_t *module = _G.modules.values[pos];
437
438 if (module->state == MANU_REQ) {
439 bool warned = false;
440
441 while (module->manu_req_count && !(module->state & FAIL_SHUT)) {
442 if (!warned) {
443 logger_trace(&_G.logger, 1,
444 "%*pM was not released, forcing release",
445 LSTR_FMT_ARG(module->name));
446 warned = true;
447 }
448 module_release(module);
449 }
450 }
451 }
452
453 /* All modules should be shutdown now. */
454 qm_for_each_pos(module, pos, &_G.modules) {
455 module_t *module = _G.modules.values[pos];
456
457 assert (module->state == REGISTERED || module->state & FAIL_SHUT);
458 }
459}

504void module_run_method(const module_method_t *method, data_t arg)
505{
506 module_method_impl_t *m;
507
508 if (unlikely(_G.methods_dirty)) {
509 module_method_register_all_cb();
510 }
511
512 m = qm_get_def(methods_impl, &_G.methods, method, NULL);
513 if (!m) {
514 /* method not implemented */
515 return;
516 }
517
518 switch (method->type) {
519 case METHOD_VOID:
520 tab_for_each_entry(cb, &m->callbacks) {
521 ((void (*)(void))cb)();
522 }
523 break;
524
525 case METHOD_INT:
526 tab_for_each_entry(cb, &m->callbacks) {
527 ((void (*)(int))cb)(arg.u32);
528 }
529 break;
530
531 case METHOD_PTR:
532 case METHOD_GENERIC:
533 tab_for_each_entry(cb, &m->callbacks) {
534 ((void (*)(data_t))cb)(arg);
535 }
536 break;
537 }
538}

549static void module_method_register_cb(module_method_impl_t *method,
550 qv_t(module) *modules)
551{
552
553 if (method->params->order == MODULE_DEPS_BEFORE) {
554 tab_for_each_entry(m, modules) {
555 if (module_is_loaded(m)) {
556 module_add_method(m, method);
557 }
558 }
559 } else
560 if (method->params->order == MODULE_DEPS_AFTER) {
561 tab_for_each_pos_rev(pos, modules) {
562 module_t *m = modules->tab[pos];
563
564 if (module_is_loaded(m)) {
565 module_add_method(m, method);
566 }
567 }
568 }
569}

571static void module_method_register_all_cb(void)
572{
573 /* XXX: Do not use t_scope here. This function can be called by
574 * pthread_fork hooks and the t_pool_g is not necessarily
575 * initialized here. */
576 qv_t(module) sorted_modules;
577 SB_1k(err);
578
579 qv_init(&sorted_modules);
580 qv_grow(&sorted_modules, qm_len(module, &_G.modules));
581 if (modules_topo_sort_rev(&_G.modules, &sorted_modules, &err) < 0) {
582 e_fatal("%*pM", SB_FMT_ARG(&err));
583 }
584
585 qm_for_each_pos(methods_impl, pos, &_G.methods) {
586 module_method_impl_t *m = _G.methods.values[pos];
587
588 qv_clear(&m->callbacks);
589 module_method_register_cb(m, &sorted_modules);
590 }
591 qv_wipe(&sorted_modules);
592 _G.methods_dirty = false;
593}

636__attribute__((constructor))
637#endif
638void module_register_at_fork(void)
639{
640 static bool at_fork_registered = false;
641
642 if (!at_fork_registered) {
643 void *data = NULL;
644
645 /* XXX: ensures internal ressources of the libc used to implement
646 * posix_memalign() are ready here. It looks like the glibc of some
647 * CentOs use an atfork() handler that get registered the first time
648 * you perform an aligned allocation.
649 */
650 IGNORE(posix_memalign(&data, 64, 1024));
651 free(data);
652
653 pthread_atfork(module_at_fork_prepare,
654 module_at_fork_on_parent,
655 module_at_fork_on_child);
656 at_fork_registered = true;
657 }

675 lstr_t *collision)
676{
677 t_scope;
678 qh_t(module) dependencies;
679
680 t_qh_init(module, &dependencies, len);
681 for (int pos = 0; pos < len; pos++) {
682 add_dependencies_to_qh(tab[pos], &dependencies);
683 }
684 for (int pos = 0; pos < len; pos++) {
