28#define START_LINE_CKEY(_wbuf, _key) \
29 IOPC_WRITE_START_LINE_CSTR(_wbuf, KEY(_key))
30
31#define START_LINE_KEY(_wbuf, _key) \
32 IOPC_WRITE_START_LINE(_wbuf, sb_addf, KEY("%s"), _key)
33
34#define LINE_START_OBJ_KEY(_wbuf, _key) \
35 do { \
36 const iopc_write_buf_t *_b = (_wbuf); \
37 IOPC_WRITE_START_LINE(_b, sb_addf, KEY("%s") "{", _key); \
38 iopc_write_buf_tab_inc(_b); \
39 } while (0)
40
41#define LINE_START_COLLECTION(_wbuf, _key, _sep) \
42 do { \
43 const iopc_write_buf_t *_b = (_wbuf); \
44 if (_key[0]) { \
45 IOPC_WRITE_START_LINE(_b, sb_addf, KEY("%s") _sep, _key ?: ""); \
46 } else { \
47 IOPC_WRITE_START_LINE(_b, sb_addf, _sep); \
48 } \
49 iopc_write_buf_tab_inc(_b); \
50 } while (0)
51
52#define LINE_START_OBJ_PREFIX(_wbuf, _key) \
53 LINE_START_COLLECTION(_wbuf, _key, "{")
54
55#define LINE_START_OBJ_CKEY(_wbuf, _key) \
56 LINE_START_OBJ_PREFIX(_wbuf, _key)
57
58#define LINE_START_OBJ(_wbuf) LINE_START_OBJ_PREFIX(_wbuf, "")
59
60#define LINE_START_ARRAY(_wbuf, _key) \
61 LINE_START_COLLECTION(_wbuf, _key, "[")
62
63#define LINE_END_COLLECTION(_wbuf, _end) \
64 do { \
65 const iopc_write_buf_t *_b = (_wbuf); \
66 suppress_sep(_b); \
67 iopc_write_buf_tab_dec(_b); \
68 IOPC_WRITE_START_LINE_CSTR(_b, _end SEP); \
69 } while (0)
70
71#define LINE_END_OBJ(_wbuf) LINE_END_COLLECTION(_wbuf, "}")
72
73#define LINE_END_ARRAY(_wbuf) LINE_END_COLLECTION(_wbuf, "]")

75#define LINE_CKEY_CVAL(_wbuf, _key, _val) \
76 IOPC_WRITE_START_LINE_CSTR(_wbuf, KEY(_key) _val SEP)

78#define LINE_CKEY_TRUE(_wbuf, _key) LINE_CKEY_CVAL(_wbuf, _key, "true")

80#define LINE_CKEY_VAL_FMT(_wbuf, _key, _fmt, ...) \
81 IOPC_WRITE_START_LINE(_wbuf, sb_addf, KEY(_key) _fmt SEP, ##__VA_ARGS__)

83#define LINE_KEY_VAL_FMT(_wbuf, _key, _fmt, ...) \
84 IOPC_WRITE_START_LINE(_wbuf, sb_addf, KEY("%s") _fmt SEP, \
85 _key, ##__VA_ARGS__)

87#define LINE_CKEY_VAL_BOOL(_wbuf, _key, _val) \
88 LINE_CKEY_VAL_FMT(_wbuf, _key, "%s", (_val) ? "true" : "false")

90#define LINE_CKEY_VAL_STR(_wbuf, _key, _val) \

92
93#define LINE_CKEY_VAL_CSTR(_wbuf, _key, _val) \
94 IOPC_WRITE_START_LINE_CSTR(_wbuf, KEY(_key) QUOTE(_val) SEP)

96#define LINE_CKEY_INT(_wbuf, _key, _val) \
97 do { \
98 int64_t _i = (_val); \
99 \
100 if (_i >= 1ll << 53 || _i <= -(1ll << 53)) { \
101 LINE_CKEY_VAL_FMT(_wbuf, _key, "\"%jd\"", _i); \
102 } else { \
103 LINE_CKEY_VAL_FMT(_wbuf, _key, "%jd", _i); \
104 } \
105 } while (0)
106
107#define LINE_CKEY_UINT(_wbuf, _key, _val) \
108 do { \
109 uint64_t _u = (_val); \
110 \
111 if (_u >= 1ull << 53) { \
112 LINE_CKEY_VAL_FMT(_wbuf, _key, "\"%ju\"", _u); \
113 } else { \
114 LINE_CKEY_VAL_FMT(_wbuf, _key, "%ju", _u); \
115 } \
116 } while (0)

147static void write_ps(const iopc_write_buf_t *wbuf, pstream_t ps)
148{
149 const char *prev = ps.s;
150
151 sb_addc(wbuf->buf, '"');
152 while (!ps_done(&ps)) {
153 const char *pos = ps.s;
154 int c = ps_getuc(&ps);
155
156 if (c < 0) {
157 e_fatal("unexpected invalid utf8 string");
158 }
159
160#define PUT_SPECIAL(fmt, ...) do { \
161 if (pos != prev) { \
162 sb_add(wbuf->buf, prev, pos - prev); \
163 } \
164 prev = ps.s; \
165 sb_addf(wbuf->buf, fmt, ##__VA_ARGS__); \
166 } while (0)
167 switch (c) {
168 case '"':
169 PUT_SPECIAL("\\\"");
170 break;
171
172 case '\\':
173 PUT_SPECIAL("\\\\");
174 break;
175
176 case '\b':
177 PUT_SPECIAL("\\b");
178 break;
179
180 case '\f':
181 PUT_SPECIAL("\\f");
182 break;
183
184 case '\n':
185 PUT_SPECIAL("\\n");
186 break;
187
188 case '\r':
189 PUT_SPECIAL("\\r");
190 break;
191
192 case '\t':
193 PUT_SPECIAL("\\t");
194 break;
195
196 default:
197 if (c < 0x20) {
198 PUT_SPECIAL("\\u%04x", c);
199 }
200 break;
201 }
202 }
203 if (ps.s != prev) {
204 sb_add(wbuf->buf, prev, ps.s - prev);
205 }
206#undef PUT_SPECIAL
207
208 sb_addc(wbuf->buf, '"');
209}

239static void dump_gen_attr(const iopc_write_buf_t *wbuf, iopc_attr_t *attr)
240{
241 if (!expect(attr->desc->id == IOPC_ATTR_GENERIC))
242 return;
243
244 LINE_START_OBJ_KEY(wbuf, attr->real_name.s);
245
246 switch (attr->args.tab[0].type) {
247 case ITOK_IDENT:
248 /* JSON object */
249 LINE_CKEY_VAL_CSTR(wbuf, "type", "object");
250 LINE_CKEY_VAL_FMT(wbuf, "value", "%*pM",
251 LSTR_FMT_ARG(attr->args.tab[0].v.s));
252 break;
253 case ITOK_STRING:
254 LINE_CKEY_VAL_CSTR(wbuf, "type", "string");
255 dump_key_val_string(wbuf, "value", attr->args.tab[0].v.s.s);
256 break;
257 case ITOK_INTEGER:
258 LINE_CKEY_VAL_CSTR(wbuf, "type", "int");
259 LINE_CKEY_INT(wbuf, "value", attr->args.tab[0].v.i64);
260 break;
261 case ITOK_BOOL:
262 LINE_CKEY_VAL_CSTR(wbuf, "type", "boolean");
263 LINE_CKEY_VAL_BOOL(wbuf, "value", attr->args.tab[0].v.i64);
264 break;
265 case ITOK_DOUBLE:
266 LINE_CKEY_VAL_CSTR(wbuf, "type", "double");
267 LINE_CKEY_VAL_FMT(wbuf, "value", "%e", attr->args.tab[0].v.d);
268 break;
269 default:
270 assert (false);
271 }
272
273 LINE_END_OBJ(wbuf);
274}

276static void dump_gen_attrs(const iopc_write_buf_t *wbuf,
277 const qv_t(iopc_attr) *attrs)
278{
279 bool has_gen_attr = false;
280
281 tab_for_each_entry(attr, attrs) {
282 if (attr->desc->id != IOPC_ATTR_GENERIC)
283 continue;
284
285 if (!has_gen_attr) {
286 has_gen_attr = true;
287 LINE_START_OBJ_CKEY(wbuf, "genAttrs");
288 }
289
290 dump_gen_attr(wbuf, attr);
291 }
292
293 if (has_gen_attr)
294 LINE_END_OBJ(wbuf);
295}

299static int
300iopc_dox_dump_json(const iopc_write_buf_t *wbuf,
301 const qv_t(iopc_dox) *comments, const char *key)
302{
303 if (!comments->len)
304 return -1;
305
306 LINE_START_OBJ_KEY(wbuf, key);
307 tab_for_each_ptr(dox, comments) {
308 lstr_t type = iopc_dox_type_to_lstr(dox->type);
309
310 if (!dox->desc.len) {
311 continue;
312 }
313
314 switch (dox->type) {
315 lstr_t brief;
316
317 case IOPC_DOX_TYPE_EXAMPLE:
318 LINE_KEY_VAL_FMT(wbuf, type.s, "%*pM", LSTR_FMT_ARG(dox->desc));
319 break;
320
321 case IOPC_DOX_TYPE_BRIEF:
322 brief = dox->desc;
323 if (brief.len > 0 && brief.s[brief.len - 1] == '.') {
324 brief.len--;
325 }
326 dump_key_val_lstr(wbuf, type.s, &brief);
327 break;
328
329 default:
330 dump_key_val_string(wbuf, type.s, dox->desc.s);
331 break;
332 }
333 }
334 LINE_END_OBJ(wbuf);
335 return 0;
336}

339 ((iopc_dox_dump_json)(_wbuf, &(_obj)->comments, "help"))

344static void
345iopc_enum_dump_json(const iopc_write_buf_t *wbuf, const char *pkg_name,
346 const iopc_enum_t *en)
347{
348 bool help_values = false;
349
350 dump_name(wbuf, pkg_name, en->name);
351 LINE_CKEY_VAL_CSTR(wbuf, "type", "enum");
352 iopc_dox_dump_json(wbuf, en);
353
354 if (iopc_attr_check(&en->attrs, IOPC_ATTR_STRICT, NULL) >= 0) {
355 LINE_START_OBJ_CKEY(wbuf, "constraints");
356 LINE_CKEY_TRUE(wbuf, "strict");
357 LINE_END_OBJ(wbuf);
358 }
359
360 dump_gen_attrs(wbuf, &en->attrs);
361
362 LINE_START_OBJ_CKEY(wbuf, "values");
363 tab_for_each_entry(f, &en->values) {
364 LINE_START_OBJ_KEY(wbuf, f->name);
365 LINE_KEY_VAL_FMT(wbuf, "value", "%d", f->value);
366 dump_gen_attrs(wbuf, &f->attrs);
367 if (f->comments.len) {
368 help_values = true;
369 }
370
371 tab_for_each_entry(attr, &f->attrs) {
372 if (attr->desc->id == IOPC_ATTR_ALIAS) {
373 LINE_START_ARRAY(wbuf, "aliases");
374 tab_for_each_entry(alias, &attr->args) {
375 write_string(wbuf, alias.v.s.s);
376 sep(wbuf);
377 }
378 LINE_END_ARRAY(wbuf);
379 }
380 }
381
382 LINE_END_OBJ(wbuf);
383 }
384 LINE_END_OBJ(wbuf);
385
386 if (help_values) {
387 LINE_START_OBJ_CKEY(wbuf, "valuesHelp");
388 tab_for_each_entry(f, &en->values) {
389 (iopc_dox_dump_json)(wbuf, &f->comments, f->name);
390 }
391 LINE_END_OBJ(wbuf);
392 }
393}

398static int
399write_field_scalar_attr(const iopc_write_buf_t *wbuf, const iopc_field_t *f,
400 const iopc_attr_t *attr)
401{
402 bool is_signed;
403
404 RETHROW(iopc_field_get_signed(f, &is_signed));
405 if (f->kind == IOP_T_DOUBLE) {
406 sb_addf(wbuf->buf, DOUBLE_FMT,
407 IOPC_ATTR_GET_ARG_V(double, &attr->args.tab[0]));
408 } else
409 if (is_signed) {
410 sb_addf(wbuf->buf, "%jd",
411 IOPC_ATTR_GET_ARG_V(int64_t, &attr->args.tab[0]));
412 } else {
413 sb_addf(wbuf->buf, "%ju",
414 IOPC_ATTR_GET_ARG_V(uint64_t, &attr->args.tab[0]));
415 }
416 return 0;
417}

429void dump_field_type(const iopc_write_buf_t *wbuf, const char *type_label,

443 case IOP_T_BOOL: type = "boolean"; break;
444 case IOP_T_VOID: type = "void"; break;
445 case IOP_T_DOUBLE: type = "number"; break;
446 case IOP_T_STRING: type = "string"; break;
447 case IOP_T_DATA: type = "data"; break;
448 case IOP_T_XML: type = "xml"; break;
449
450 default:
451 dump_field_type_obj(wbuf, type_label, f);
452 return;
453 }
454
455 LINE_KEY_VAL_FMT(wbuf, type_label, QUOTE("%s"), type);
456}

458static void dump_defval(const iopc_write_buf_t *wbuf, const iopc_field_t *f)
459{
460#define DEF "default"
461
462 switch (f->kind) {
463#define CASE_IOP_T_INT(N) \
464 case IOP_T_U##N: \
465 LINE_CKEY_UINT(wbuf, DEF, f->defval.u64); \
466 break; \
467 case IOP_T_I##N: \
468 LINE_CKEY_INT(wbuf, DEF, (int##N##_t)f->defval.u64); \
469 break;
470
471 CASE_IOP_T_INT(8)
472 CASE_IOP_T_INT(16)
473 CASE_IOP_T_INT(32)
474 CASE_IOP_T_INT(64)
475
476 case IOP_T_ENUM:
477 LINE_CKEY_INT(wbuf, DEF, f->defval.u64);
478 break;
479
480 case IOP_T_BOOL:
481 LINE_CKEY_VAL_BOOL(wbuf, DEF, f->defval.u64);
482 break;
483
484 case IOP_T_DOUBLE:
485 LINE_CKEY_VAL_FMT(wbuf, DEF, DOUBLE_FMT, f->defval.d);
486 break;
487
488 case IOP_T_STRING:
489 case IOP_T_DATA:
490 case IOP_T_XML:
491 dump_key_val_string(wbuf, DEF, (const char *)f->defval.ptr);
492 break;
493
494 default:
495 break;
496
497#undef CASE_IOP_T_INT
498 }
499#undef DEF

543dump_constraint(const iopc_write_buf_t *wbuf, const iopc_field_t *f,

545{

563 case IOPC_ATTR_MAX_LENGTH:

568 break;

570 case IOPC_ATTR_CDATA:
571 LINE_CKEY_TRUE(wbuf, "cdata");
572 break;
573 case IOPC_ATTR_NON_EMPTY:
574 LINE_CKEY_TRUE(wbuf, "nonEmpty");
575 break;
576 case IOPC_ATTR_NON_ZERO:
577 LINE_CKEY_TRUE(wbuf, "nonZero");
578 break;

580

591 dump_constraint_multi_token(wbuf, name, f, attr);
592 } break;
593
594 case IOPC_ATTR_PATTERN:
595 dump_key_val_string(wbuf, "pattern", attr->args.tab[0].v.s.s);
596 break;
597
598 case IOPC_ATTR_DEPRECATED:
599 /* TODO */
600 break;
601
602 case IOPC_ATTR_GENERIC:
603 /* TODO */
604 break;
605
606 case IOPC_ATTR_SNMP_PARAMS_FROM:
607 /* TODO */
608 break;
609
610 case IOPC_ATTR_SNMP_PARAM:
611 /* TODO */
612 break;
613
614 case IOPC_ATTR_SNMP_INDEX:
615 /* TODO */
616 break;
617
618 case IOPC_ATTR_FORCE_FIELD_NAME:
619 break;

621 default:

627static int
628dump_field_constraints(const iopc_write_buf_t *wbuf, const iopc_field_t *f)
629{
630 bool has_constraint = false;
631
632 tab_for_each_entry(attr, &f->attrs) {
633 if (is_dummy_constraint(attr)) {
634 continue;
635 }
636
637 if (!has_constraint) {
638 has_constraint = true;
639 LINE_START_OBJ_CKEY(wbuf, "constraints");
640 }
641
642 RETHROW(dump_constraint(wbuf, f, attr));
643 }
644
645 if (has_constraint) {
646 LINE_END_OBJ(wbuf);
647 }
648
649 return 0;
650}

652static
653int dump_fields(const iopc_write_buf_t *wbuf, const qv_t(iopc_field) *fields)
654{
655 tab_for_each_entry(f, fields) {
656 bool private = false;
657 bool deprecated = false;
658
659 LINE_START_OBJ_KEY(wbuf, f->name);
660
661 dump_field_type(wbuf, TYPE_LABEL, f);
662
663 switch (f->repeat) {
664 case IOP_R_DEFVAL:
665 dump_defval(wbuf, f);
666 break;
667 case IOP_R_OPTIONAL:
668 LINE_CKEY_TRUE(wbuf, "optional");
669 break;
670 case IOP_R_REPEATED:
671 LINE_CKEY_TRUE(wbuf, "repeated");
672 break;
673 }
674
675 iopc_dox_dump_json(wbuf, f);
676 tab_for_each_entry(attr, &f->attrs) {
677 if (attr->desc->id == IOPC_ATTR_PRIVATE) {
678 private = true;
679 } else
680 if (attr->desc->id == IOPC_ATTR_DEPRECATED) {
681 deprecated = true;
682 }
683 }
684
685 RETHROW(dump_field_constraints(wbuf, f));
686 dump_gen_attrs(wbuf, &f->attrs);
687
688 if (private) {
689 LINE_CKEY_TRUE(wbuf, "private");
690 }
691 if (deprecated) {
692 LINE_CKEY_TRUE(wbuf, "deprecated");
693 }
694
695 LINE_END_OBJ(wbuf);
696 }
697
698 return 0;
699}

714iopc_struct_dump_json(const iopc_write_buf_t *wbuf, const char *pkg_name,
715 iopc_struct_t *st)
716{
717 dump_name(wbuf, pkg_name, st->name);
718
719 switch (st->type) {
720 case STRUCT_TYPE_UNION:
721 LINE_CKEY_VAL_CSTR(wbuf, "type", "union");
722 break;
723 case STRUCT_TYPE_STRUCT:
724 LINE_CKEY_VAL_CSTR(wbuf, "type", "struct");
725 break;
726 case STRUCT_TYPE_CLASS:
727 LINE_CKEY_VAL_CSTR(wbuf, "type", "class");
728 if (st->extends.len == 1) {
729 LINE_CKEY_VAL_FMT(wbuf, "parent", XT_FMT,
730 XT_FMT_ARG(st->extends.tab[0]));
731 } else
732 if (st->extends.len) {
733 print_error("multiple class inheritance is not allowed: %s.%s",
734 pkg_name, st->name);
735 assert (false);
736 return -1;
737 }
738 LINE_CKEY_VAL_BOOL(wbuf, "isAbstract", st->is_abstract);
739 tab_for_each_entry(attr, &st->attrs) {
740 if (attr->desc->id == IOPC_ATTR_PRIVATE) {
741 LINE_CKEY_VAL_BOOL(wbuf, "private", true);
742 break;
743 }
744 }
745 break;
746 case STRUCT_TYPE_SNMP_OBJ:
747 LINE_CKEY_VAL_CSTR(wbuf, "type", "snmpObj");
748 LINE_CKEY_VAL_FMT(wbuf, "parent", XT_FMT,
749 XT_FMT_ARG(st->extends.tab[0]));
750 break;
751 case STRUCT_TYPE_SNMP_TBL:
752 LINE_CKEY_VAL_CSTR(wbuf, "type", "snmpTbl");
753 LINE_CKEY_VAL_FMT(wbuf, "parent", XT_FMT,
754 XT_FMT_ARG(st->extends.tab[0]));
755 break;
756
757 default:
758 print_error("unsupported struct type: %d", st->type);
759 assert (false);
760 return -1;
761 }
762
763 iopc_dox_dump_json(wbuf, st);
764
765 if (st->static_fields.len) {
766 LINE_START_OBJ_CKEY(wbuf, "statics");
767 qv_sort(iopc_field)(&st->static_fields,
768 ^int (iopc_field_t * const *f1,
769 iopc_field_t * const *f2) {
770 return (*f1)->field_pos - (*f2)->field_pos;
771 });
772 RETHROW(dump_fields(wbuf, &st->static_fields));
773 LINE_END_OBJ(wbuf);
774 }
775
776 dump_gen_attrs(wbuf, &st->attrs);
777
778 LINE_START_OBJ_CKEY(wbuf, "fields");
779 RETHROW(dump_fields(wbuf, &st->fields));
780 LINE_END_OBJ(wbuf);
781 return 0;
782}

856static void
857iopc_module_dump_json(const iopc_write_buf_t *wbuf, const char *pkg_name,
858 const iopc_struct_t *mod)
859{
860 /* XXX: base_tags is used in order to not put the interfaces belonging
861 * to inherited modules
862 */
863 qh_t(u32) base_tags = QH_INIT(u32, base_tags);
864
865 dump_name(wbuf, pkg_name, mod->name);
866 iopc_dox_dump_json(wbuf, mod);
867
868 if (mod->extends.len) {
869 LINE_START_ARRAY(wbuf, "parent");
870 }
871 tab_for_each_entry(xt, &mod->extends) {
872 IOPC_WRITE_START_LINE(wbuf, sb_addf, XT_FMT SEP, XT_FMT_ARG(xt));
873 tab_for_each_entry(f, &xt->st->fields) {
874 qh_add(u32, &base_tags, f->tag);
875 }
876 }
877 if (mod->extends.len)
878 LINE_END_ARRAY(wbuf);
879
880 LINE_START_OBJ_CKEY(wbuf, "interfaces");
881
882 tab_for_each_entry(f, &mod->fields) {
883 if (qh_find(u32, &base_tags, f->tag) >= 0)
884 continue;
885
886 LINE_START_OBJ_KEY(wbuf, f->name);
887 dump_field_type_obj(wbuf, TYPE_LABEL, f);
888 iopc_dox_dump_json(wbuf, f);
889 LINE_END_OBJ(wbuf);
890 }
891 LINE_END_OBJ(wbuf);
892
893 qh_wipe(u32, &base_tags);
894}

913int iopc_do_json(iopc_pkg_t *pkg, const char *outdir)
914{
915 char path[PATH_MAX];
916 SB(buf, 32 << 10);
917 SB_1k(tab);
918 iopc_write_buf_t wbuf;
919 const char *pkg_name = iopc_path_dot(pkg->name);
920
921 iopc_write_buf_init(&wbuf, &buf, &tab);
922
923 iopc_set_path(outdir, pkg, ".iop.json", sizeof(path), path, true);
924
925 sb_addc(&buf, '{');
926 iopc_write_buf_tab_inc(&wbuf);
927
928 LINE_CKEY_VAL_STR(&wbuf, "name", pkg_name);
929 newline(&wbuf);
930
931 if (iopc_dox_dump_json(&wbuf, pkg) >= 0)
932 newline(&wbuf);
933
934#define DUMP_PKG_ITEMS_TYPE(_item, _type) \
935 do { \
936 tab_for_each_entry(e, &pkg->_item##s) { \
937 LINE_START_OBJ(&wbuf); \
938 iopc_##_item##_dump_json(&wbuf, pkg_name, e); \
939 LINE_END_OBJ(&wbuf); \
940 } \
941 } while (0)
942
943#define DUMP_PKG_ITEMS(_item) DUMP_PKG_ITEMS_TYPE(_item, _item)
944
945
946 /* start objects */
947 LINE_START_ARRAY(&wbuf, "objects");
948
949 DUMP_PKG_ITEMS(enum);
950 DUMP_PKG_ITEMS(struct); /* structs, unions and classes */
951
952 tab_for_each_entry(tdef, &pkg->typedefs) {
953 LINE_START_OBJ(&wbuf);
954 RETHROW(iopc_typedef_dump_json(&wbuf, pkg_name, tdef));
955 LINE_END_OBJ(&wbuf);
956 }
957
958 tab_for_each_entry(iface, &pkg->ifaces) {
959 qv_sort(iopc_fun)(&iface->funs,
960 ^int (iopc_fun_t * const *f1,
961 iopc_fun_t * const *f2) {
962 return (*f1)->pos - (*f2)->pos;
963 });
964 tab_for_each_entry(fun, &iface->funs) {
965 t_scope;
966 const char *fbase = t_fmt("%s.%s", pkg_name, iface->name);
967
968 RETHROW(iopc_fun_struct_type_dump_json(&wbuf, &fun->arg, fbase));
969 RETHROW(iopc_fun_struct_type_dump_json(&wbuf, &fun->res, fbase));
970 RETHROW(iopc_fun_struct_type_dump_json(&wbuf, &fun->exn, fbase));
971 }
972 }
973
974 /* end objects */
975 LINE_END_ARRAY(&wbuf);
976
977 if (!pkg->ifaces.len)
978 goto modules;
979
980 /* start interfaces */
981 newline(&wbuf);
982 LINE_START_ARRAY(&wbuf, "interfaces");
983
984 DUMP_PKG_ITEMS(iface);
985
986 /* end interfaces */
987 LINE_END_ARRAY(&wbuf);
988
989 modules:
990 if (!pkg->modules.len)
991 goto end_file;
992
993 /* start modules */
994 newline(&wbuf);
995 LINE_START_ARRAY(&wbuf, "modules");
996
997 DUMP_PKG_ITEMS_TYPE(module, struct);
998
999 /* end modules */
1000 LINE_END_ARRAY(&wbuf);
1001
1002 end_file:
1003 /* end of file */
1004 suppress_sep(&wbuf);
1005 sb_adds(&buf, "\n}\n");
1006
1007 return iopc_write_file(&buf, path);
1008
1009#undef DUMP_PKG_ITEMS_TYPE
1010#undef DUMP_PKG_ITEMS
