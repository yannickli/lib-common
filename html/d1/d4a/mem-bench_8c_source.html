29mem_bench_t *mem_bench_init(mem_bench_t *sp, lstr_t type, uint32_t period)
30{
31 p_clear(sp, 1);
32 dlist_init(&sp->bench_list);
33
34 {
35 lstr_t logname = lstr_fmt("%*pM.%p", LSTR_FMT_ARG(type), sp);
36 sp->logger = logger_new(&mem_bench_logger_g, logname, LOG_INHERITS, 0);
37 lstr_wipe(&logname);
38 }
39
40 if (period) {
41 char filename[PATH_MAX];
42
43 path_extend(filename, ".", "mem.%*pM.data.%u.%p",
44 LSTR_FMT_ARG(type), getpid(), sp);
45 sp->file = fopen(filename, "w");
46 /* not fatal if sp->file is NULL : we won't log anything. */
47 }
48 sp->out_period = period;
49 sp->out_counter = period;
50
51 lstr_copy(&sp->allocator_name, type);
52
53 return sp;
54}

56void mem_bench_leak(mem_bench_t *sp)
57{
58 spin_lock(&mem_bench_leak_lock_g);
59 dlist_add_tail(&mem_bench_leak_list_g, &sp->bench_list);
60 spin_unlock(&mem_bench_leak_lock_g);
61}
62
63static void mem_bench_partial_wipe(mem_bench_t *sp)
64{
65 mem_bench_print_human(sp, 0);
66
67 spin_lock(&mem_bench_leak_lock_g);
68 dlist_remove(&sp->bench_list);
69 spin_unlock(&mem_bench_leak_lock_g);
70
71 logger_delete(&sp->logger);
72}
73
74void mem_bench_wipe(mem_bench_t *sp)
75{
76 mem_bench_print_csv(sp);
77 mem_bench_partial_wipe(sp);
78 p_fclose(&sp->file);
79 lstr_wipe(&sp->allocator_name);
80}

82static void mem_bench_print_func_csv(mem_bench_func_t *spf, FILE *file)
83{
84 assert (file);
85 fprintf(file, "%u,%u,",
86 spf->nb_calls, spf->nb_slow_path);
87 fprintf(file, "%u,%lu,%lu,%lu,",
88 spf->timer_stat.nb,
89 spf->timer_stat.hard_min,
90 spf->timer_stat.hard_max,
91 spf->timer_stat.hard_tot);
92}
93
94void mem_bench_print_csv(mem_bench_t *sp)
95{
96 if (!sp->file) {
97 return;
98 }
99
100 if (sp->logger) {
101 logger_trace(sp->logger, 1, "CSV trace");
102 }
103
104 mem_bench_print_func_csv(&sp->alloc, sp->file);
105 mem_bench_print_func_csv(&sp->realloc, sp->file);
106 mem_bench_print_func_csv(&sp->free, sp->file);
107 fprintf(sp->file, "%lu,%lu,%u,%u,%u,%u,%u,%u\n",
108 sp->total_allocated, sp->total_requested,
109 sp->max_allocated, sp->max_unused, sp->max_used,
110 sp->malloc_calls, sp->current_used, sp->current_allocated);
111}

113void mem_bench_update(mem_bench_t *sp)
114{
115 sp->max_used = MAX(sp->current_used, sp->max_used);
116 sp->max_allocated = MAX(sp->current_allocated, sp->max_allocated);
117 sp->max_unused = MAX(sp->current_allocated - sp->current_used,
118 sp->max_unused);
119
120 sp->out_counter--;
121 if (sp->file && sp->out_counter <= 0) {
122 mem_bench_print_csv(sp);
123 sp->out_counter = sp->out_period;
124 }
125
126 if (sp->logger) {
127 logger_trace(sp->logger, 2, "Update");
128 }
129}

131static void mem_bench_print_func_human(const mem_bench_t *sp,
132 const mem_bench_func_t *spf,
133 const char* prefix)
134{
135 assert (sp->logger);
136 logger_debug(sp->logger, "%s/requests : %10u",
137 prefix, spf->nb_calls);
138 logger_debug(sp->logger, "%s/slow path calls : %10u \t%u.%u %%",
139 prefix, spf->nb_slow_path,
140 100 * spf->nb_slow_path / MAX(1, spf->nb_calls),
141 (10000 * spf->nb_slow_path / MAX(1, spf->nb_calls)) % 100);
142 logger_debug(sp->logger, "%s/timer : %s",
143 prefix,
144 proctimerstat_report((proctimerstat_t *)&spf->timer_stat, "%h"));
145}
146
147void mem_bench_print_human(const mem_bench_t *sp, int flags)
148{
149 if (!sp->logger) {
150 return;
151 }
152
153 logger_debug(sp->logger, "%*pM allocator @%p stats :",
154 LSTR_FMT_ARG(sp->allocator_name), sp);
155 mem_bench_print_func_human(sp, &sp->alloc, "alloc ");
156 mem_bench_print_func_human(sp, &sp->realloc, "realloc");
157 mem_bench_print_func_human(sp, &sp->free, "free ");
158 logger_debug(sp->logger, "average request size : %10lu bytes",
159 sp->total_requested / MAX(1,sp->alloc.nb_calls));
160 logger_debug(sp->logger, "average block size : %10lu bytes",
161 sp->total_allocated / MAX(1,sp->malloc_calls));
162 logger_debug(sp->logger, "total memory allocated : %10lu K",
163 sp->total_allocated / 1024);
164 logger_debug(sp->logger, "total memory requested : %10lu K",
165 sp->total_requested / 1024);
166 logger_debug(sp->logger, "max used memory : %10d K",
167 sp->max_used / 1024);
168 logger_debug(sp->logger, "max unused memory : %10u K",
169 sp->max_unused / 1024);
170 logger_debug(sp->logger, "max memory allocated : %10u K",
171 sp->max_allocated / 1024);
172 logger_debug(sp->logger, "malloc calls : %10u",
173 sp->malloc_calls);
174 if (flags & MEM_BENCH_PRINT_CURRENT) {
175 logger_debug(sp->logger, "current used memory : %10u K",
176 sp->current_used / 1024);
177 logger_debug(sp->logger, "current allocated memory : %10u K",
178 sp->current_allocated / 1024);
179 }
180}

188static int mem_bench_shutdown(void)
189{
190 spin_lock(&mem_bench_leak_lock_g);
191 dlist_for_each_entry(mem_bench_t, sp, &mem_bench_leak_list_g, bench_list)
192 {
193 spin_unlock(&mem_bench_leak_lock_g);
194
195 mem_bench_partial_wipe(sp);
196
197 spin_lock(&mem_bench_leak_lock_g);
198 }
199 spin_unlock(&mem_bench_leak_lock_g);
200
201 return 0;
202}

212 module_require(mb_module, NULL);
213}
