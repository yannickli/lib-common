46 typeof(futex) __futex = (futex); \
47 typeof(val) __val = (val); \
48 typeof(ts) __ts = (ts); \

78void thr_ec_signal_n(thr_evc_t *ec, int count)
79{
80 atomic_fetch_add(&ec->key, 1);
81
82 if (atomic_load(&ec->waiters))
83 futex_wake_private(&ec->key, count);

92void thr_ec_timedwait(thr_evc_t *ec, uint64_t key, long timeout)
93{
94 int canceltype, res;
95
96 atomic_thread_fence(memory_order_acq_rel);
97
98 /*
99 * XXX: futex only works on integers (32bits) so we have to check if the
100 * high 32bits word changed. We can do this in a racy way because we
101 * assume it's impossible for the low 32bits to cycle between this
102 * test and the call to futex_wait_private.
103 */
104 if (unlikely(key != atomic_load(&ec->key))) {
105 pthread_testcancel();
106 return;
107 }
108
109 atomic_fetch_add(&ec->waiters, 1);
110 pthread_cleanup_push(&thr_ec_wait_cleanup, ec);
111 pthread_setcanceltype(PTHREAD_CANCEL_ASYNCHRONOUS, &canceltype);
112
113 if (timeout > 0) {
114 struct timespec spec = {
115 .tv_sec = timeout / 1000,
116 .tv_nsec = (timeout % 1000) * 1000000,
117 };
118 res = futex_wait_private(&ec->key, (uint32_t)key, &spec);
119 } else {
120 res = futex_wait_private(&ec->key, (uint32_t)key, NULL);
121 }
122 if (res == 0)
123 sched_yield();
124
125 /* XXX passing NULL to pthread_setcanceltype() breaks TSAN */
126 pthread_setcanceltype(canceltype, &res);
127 pthread_cleanup_pop(1);
128}

167void thr_ec_timedwait(thr_evc_t *ec, uint64_t key, long timeout)
168{
169 struct timespec ts;
170 int canceltype;
171
172 if (thr_ec_get(ec) != key) {
173 return;
174 }
175
176 pthread_mutex_lock(&ec->mutex);
177 if (thr_ec_get(ec) != key) {
178 pthread_mutex_unlock(&ec->mutex);
179 return;
180 }
181
182 if (timeout > 0) {
183 uint64_t usec;
184 struct timeval tv;
185
186 lp_gettv(&tv);
187 usec = tv.tv_usec + timeout * 1000;
188 ts.tv_sec = tv.tv_sec + usec / 1000000;
189 ts.tv_nsec = (usec % 1000000) * 1000;
190 }
191
192 atomic_fetch_add(&ec->waiters, 1);
193 pthread_cleanup_push(&thr_ec_wait_cleanup, ec);
194 pthread_setcanceltype(PTHREAD_CANCEL_ASYNCHRONOUS, &canceltype);
195
196 while (thr_ec_get(ec) == key) {
197 if (timeout > 0) {
198 if (pthread_cond_timedwait(&ec->cond, &ec->mutex, &ts) < 0) {
199 if (errno == ETIMEDOUT) {
200 break;
201 }
202 e_panic(E_UNIXERR("pthread_cond_timedwait"));
203 }
204 } else {
205 if (pthread_cond_wait(&ec->cond, &ec->mutex) < 0) {
206 e_panic(E_UNIXERR("pthread_cond_wait"));
207 }
208 }
209 }
210 pthread_setcanceltype(canceltype, NULL);
211 pthread_cleanup_pop(1);
212
213 pthread_mutex_unlock(&ec->mutex);
214}

216void thr_ec_signal_n(thr_evc_t *ec, int count)
217{
218 pthread_mutex_lock(&ec->mutex);
219
220 atomic_fetch_add(&ec->key, 1);
221
222 if (atomic_fetch_add(&ec->waiters, 0)) {
223 if (count == INT_MAX) {
224 pthread_cond_broadcast(&ec->cond);
225 } else {
226 while (count-- > 0) {
227 pthread_cond_signal(&ec->cond);
228 }
229 }
230 }
231 pthread_mutex_unlock(&ec->mutex);
232}
