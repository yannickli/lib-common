70static yunpack_res_t
71t_yaml_nil_to_iop_field(yunpack_env_t * nonnull env,
72 const yaml_data_t * nonnull data,
73 const iop_field_t * nonnull fdesc,
74 bool in_array, void * nonnull out)
75{
76 if (!in_array && fdesc->repeat == IOP_R_REPEATED) {
77 lstr_t *arr = out;
78
79 /* null value on an array means empty array */
80 p_clear(arr, 1);
81 return YUNPACK_OK;
82 }
83
84 switch (fdesc->type) {
85 case IOP_T_STRING:
86 case IOP_T_XML:
87 case IOP_T_DATA:
88 *(lstr_t *)out = LSTR_NULL_V;
89 return YUNPACK_OK;
90 case IOP_T_STRUCT:
91 case IOP_T_UNION:
92 if (t_yaml_data_to_typed_struct(env, data, fdesc->u1.st_desc,
93 out) < 0)
94 {
95 return YUNPACK_NULL_STRUCT_ERROR;
96 }
97 return YUNPACK_OK;
98 default:
99 return YUNPACK_TYPE_MISMATCH;
100 }
101}

103static yunpack_res_t
104t_yaml_string_to_iop_field(const lstr_t str,
105 const iop_field_t * nonnull fdesc,
106 void * nonnull out)
107{
108 switch (fdesc->type) {
109 case IOP_T_STRING:
110 case IOP_T_XML:
111 case IOP_T_DATA:
112 *(lstr_t *)out = t_lstr_dup(str);
113 return YUNPACK_OK;
114
115 case IOP_T_ENUM: {
116 bool found;
117 int i;
118
119 i = iop_enum_from_lstr_desc(fdesc->u1.en_desc, str, &found);
120 if (!found) {
121 return YUNPACK_INVALID_ENUM_VAL;
122 }
123 *(int32_t *)out = i;
124 return 0;
125 }
126
127 default:
128 return YUNPACK_TYPE_MISMATCH;
129 }

156static int
157t_yaml_uint_to_iop_field(const yaml_data_t * nonnull data,
158 uint64_t u, const iop_field_t * nonnull fdesc,
159 void * nonnull out)
160{
161#define CHECK_MAX(v, max) THROW_IF(v > max, YUNPACK_OOB)
162
163 switch (fdesc->type) {
164 case IOP_T_I8:
165 CHECK_MAX(u, INT8_MAX);
166 *(int8_t *)out = u;
167 return YUNPACK_OK;
168
169 case IOP_T_U8:
170 CHECK_MAX(u, UINT8_MAX);
171 *(uint8_t *)out = u;
172 return YUNPACK_OK;
173
174 case IOP_T_I16:
175 CHECK_MAX(u, INT16_MAX);
176 *(int16_t *)out = u;
177 return YUNPACK_OK;
178
179 case IOP_T_U16:
180 CHECK_MAX(u, UINT16_MAX);
181 *(uint16_t *)out = u;
182 return YUNPACK_OK;
183
184 case IOP_T_I32:
185 CHECK_MAX(u, INT32_MAX);
186 *(int32_t *)out = u;
187 return YUNPACK_OK;
188
189 case IOP_T_U32:
190 CHECK_MAX(u, UINT32_MAX);
191 *(uint32_t *)out = u;
192 return YUNPACK_OK;
193
194 case IOP_T_I64:
195 CHECK_MAX(u, INT64_MAX);
196 *(int64_t *)out = u;
197 return YUNPACK_OK;
198
199 case IOP_T_U64:
200 *(uint64_t *)out = u;
201 return YUNPACK_OK;
202
203 case IOP_T_DOUBLE:
204 *(double *)out = u;
205 return YUNPACK_OK;
206
207 case IOP_T_STRING:
208 t_set_string_from_stream(data, out);
209 return YUNPACK_OK;
210
211 case IOP_T_ENUM:
212 CHECK_MAX(u, INT32_MAX);
213 if (TST_BIT(&fdesc->u1.en_desc->flags, IOP_ENUM_STRICT)
214 && !iop_enum_exists_desc(fdesc->u1.en_desc, u))
215 {
216 return YUNPACK_INVALID_ENUM_VAL;
217 }
218 *(int32_t *)out = u;
219 return YUNPACK_OK;
220
221 default:
222 return YUNPACK_TYPE_MISMATCH;
223 }
224
225#undef CHECK_MAX
226}

228static int
229yaml_int_to_iop_field(int64_t i, const iop_field_t * nonnull fdesc,
230 void * nonnull out)
231{
232#define CHECK_RANGE(v, min, max) THROW_IF(v < min || v > max, YUNPACK_OOB)
233
234 /* The INT type is only used for negative values, so all unsigned types
235 * are OOB. */
236
237 switch (fdesc->type) {
238 case IOP_T_I8:
239 CHECK_RANGE(i, INT8_MIN, INT8_MAX);
240 *(int8_t *)out = i;
241 return YUNPACK_OK;
242
243 case IOP_T_U8:
244 return YUNPACK_OOB;
245
246 case IOP_T_I16:
247 CHECK_RANGE(i, INT16_MIN, INT16_MAX);
248 *(int16_t *)out = i;
249 return YUNPACK_OK;
250
251 case IOP_T_U16:
252 return YUNPACK_OOB;
253
254 case IOP_T_I32:
255 CHECK_RANGE(i, INT32_MIN, INT32_MAX);
256 *(int32_t *)out = i;
257 return YUNPACK_OK;
258
259 case IOP_T_U32:
260 return YUNPACK_OOB;
261
262 case IOP_T_I64:
263 *(int64_t *)out = i;
264 return YUNPACK_OK;
265
266 case IOP_T_U64:
267 return YUNPACK_OOB;
268
269 case IOP_T_DOUBLE:
270 *(double *)out = i;
271 return YUNPACK_OK;
272
273 case IOP_T_ENUM:
274 CHECK_RANGE(i, INT32_MIN, INT32_MAX);
275 if (TST_BIT(&fdesc->u1.en_desc->flags, IOP_ENUM_STRICT)
276 && !iop_enum_exists_desc(fdesc->u1.en_desc, i))
277 {
278 return YUNPACK_INVALID_ENUM_VAL;
279 }
280 *(int32_t *)out = i;
281 return YUNPACK_OK;
282
283 default:
284 return YUNPACK_TYPE_MISMATCH;
285 }
286}

307static yunpack_res_t
308t_yaml_scalar_to_iop_field(yunpack_env_t * nonnull env,
309 const yaml_data_t * nonnull data,
310 const iop_field_t * nonnull fdesc,
311 bool in_array, void * nonnull out)
312{
313 const yaml_scalar_t *scalar = &data->scalar;
314
315 /* backward compatibility: unpack a scalar into an array as an array
316 * of one element. */
317 if (!in_array && fdesc->repeat == IOP_R_REPEATED
318 && scalar->type != YAML_SCALAR_NULL)
319 {
320 iop_array_i8_t *arr = out;
321
322 out = ta_new_raw(byte, fdesc->size, 8);
323 *arr = IOP_TYPED_ARRAY(i8, out, 1);
324 }
325
326 switch (scalar->type) {
327 case YAML_SCALAR_NULL:
328 return t_yaml_nil_to_iop_field(env, data, fdesc, in_array, out);
329 case YAML_SCALAR_STRING:
330 case YAML_SCALAR_BYTES:
331 return t_yaml_string_to_iop_field(scalar->s, fdesc, out);
332 case YAML_SCALAR_DOUBLE:
333 return t_yaml_double_to_iop_field(data, scalar->d, fdesc, out);
334 case YAML_SCALAR_UINT:
335 return t_yaml_uint_to_iop_field(data, scalar->u, fdesc, out);
336 case YAML_SCALAR_INT:
337 return yaml_int_to_iop_field(scalar->i, fdesc, out);
338 case YAML_SCALAR_BOOL:
339 return t_yaml_bool_to_iop_field(data, scalar->b, fdesc, out);
340 }
341
342 assert (false);
343 return YUNPACK_TYPE_MISMATCH;
344}

365static int
366check_constraints(const iop_struct_t * nonnull desc,
367 const iop_field_t * nonnull fdesc, void * nonnull value)
368{
369 if (likely(!iop_field_has_constraints(desc, fdesc))) {
370 return 0;
371 }
372
373 if (fdesc->repeat == IOP_R_REPEATED) {
374 iop_array_i8_t *arr = value;
375
376 return iop_field_check_constraints(desc, fdesc, arr->tab, arr->len,
377 false);
378 } else {
379 return iop_field_check_constraints(desc, fdesc, value, 1, false);
380 }

383static int
384t_yaml_data_to_union(yunpack_env_t * nonnull env,
385 const yaml_data_t * nonnull data,
386 const iop_struct_t * nonnull st_desc, void * nonnull out)
387{
388 const iop_field_t *field_desc = NULL;
389 const yaml_key_data_t *kd;
390 const yaml_span_t *span = &data->span;
391
392 if (data->type != YAML_DATA_OBJ) {
393 sb_setf(&env->err.buf, "cannot unpack %s into a union",
394 yaml_data_get_type(data, false));
395 goto error;
396 }
397
398 if (data->obj->fields.len != 1) {
399 sb_setf(&env->err.buf, "a single key must be specified");
400 goto error;
401 }
402 kd = &data->obj->fields.tab[0];
403
404 iop_field_find_by_name(st_desc, kd->key, NULL, &field_desc);
405 if (!field_desc) {
406 sb_setf(&env->err.buf, "unknown field `%pL`", &kd->key);
407 span = &kd->key_span;
408 goto error;
409 }
410
411 iop_union_set_tag(st_desc, field_desc->tag, out);
412 out = (char *)out + field_desc->data_offs;
413 if (t_yaml_data_to_iop_field(env, &kd->data, st_desc, field_desc, false,
414 out) < 0)
415 {
416 /* keep the data causing the issue in the err. */
417 span = env->err.span;
418 goto error;
419 }
420
421 if (check_constraints(st_desc, field_desc, out) < 0) {
422 sb_setf(&env->err.buf, "field `%pL` is invalid: %s", &kd->key,
423 iop_get_err());
424 span = &kd->key_span;
425 goto error;
426 }
427
428 return 0;
429
430 error:
431 sb_prependf(&env->err.buf, "cannot unpack YAML as a `%pL` IOP union: ",
432 &st_desc->fullname);
433 env->err.span = span;
434 return -1;
435}

440static int
441check_class(yunpack_env_t * nonnull env,
442 const iop_struct_t * nonnull st)
443{
444 if (st->class_attrs->is_abstract) {
445 sb_setf(&env->err.buf, "`%pL` is abstract and cannot be unpacked",
446 &st->fullname);
447 return -1;
448 }
449
450 if (env->flags & IOP_UNPACK_FORBID_PRIVATE
451 && st->class_attrs->is_private)
452 {
453 /* TODO: the error should probably be an "unknown type" to not expose
454 * the private name. To be done for JSON unpacker too. */
455 sb_setf(&env->err.buf, "`%pL` is private and cannot be unpacked",
456 &st->fullname);
457 return -1;
458 }
459
460 return 0;
461}

463static const yaml_key_data_t * nullable
464yaml_data_get_field_value(const yaml_data_t * nonnull data,
465 const lstr_t field_name)
466{
467 if (data->type == YAML_DATA_OBJ) {
468 /* do a linear search in the obj AST. This isn't necessarily optimal,
469 * but keeps the code simple. */
470 tab_for_each_ptr(pair, &data->obj->fields) {
471 if (lstr_equal(pair->key, field_name)) {
472 return pair;
473 }
474 }
475 return NULL;
476 } else {
477 assert (data->type == YAML_DATA_SCALAR
478 && data->scalar.type == YAML_SCALAR_NULL);
479 return NULL;
480 }
481}

502static int
503t_yaml_fill_iop_field(yunpack_env_t * nonnull env,
504 const yaml_key_data_t * nonnull kd,
505 const iop_struct_t * nonnull st,
506 const iop_field_t * nonnull fdesc, void * nonnull out)
507{
508 if (env->flags & IOP_UNPACK_FORBID_PRIVATE) {
509 const iop_field_attrs_t *attrs;
510
511 attrs = iop_field_get_attrs(st, fdesc);
512 if (attrs && TST_BIT(&attrs->flags, IOP_FIELD_PRIVATE)) {
513 sb_setf(&env->err.buf, "unknown field `%pL`", &fdesc->name);
514 env->err.span = &kd->key_span;
515 return -1;
516 }
517 }
518
519 out = (char *)out + fdesc->data_offs;
520 if (t_yaml_data_to_iop_field(env, &kd->data, st, fdesc, false, out) < 0) {
521 return -1;
522 }
523
524 if (check_constraints(st, fdesc, out) < 0) {
525 sb_setf(&env->err.buf, "field `%pL` is invalid: %s", &fdesc->name,
526 iop_get_err());
527 env->err.span = &kd->data.span;
528 return -1;
529 }
530
531 return 0;
532}

550static const iop_struct_t * nullable
551get_struct_from_tag(const iop_struct_t * nonnull st, const lstr_t tag,
552 sb_t * nonnull err)
553{
554 if (iop_struct_is_class(st)) {
555 const iop_struct_t *real_st;
556
557 real_st = iop_get_class_by_fullname(st, tag);
558 if (!real_st) {
559 sb_setf(err, "unknown type `%pL` provided in tag, "
560 "or not a child of `%pL`", &tag, &st->fullname);
561 return NULL;
562 }
563 if (!iop_class_is_a(real_st, st)) {
564 sb_setf(err, "provided tag `%pL` is not a child of `%pL`",
565 &real_st->fullname, &st->fullname);
566 return NULL;
567 }
568
569 return real_st;
570 } else {
571 if (!lstr_equal(st->fullname, tag)) {
572 sb_setf(err, "wrong type `%pL` provided in tag, expected `%pL`",
573 &tag, &st->fullname);
574 return NULL;
575 }
576
577 return st;
578 }

581static int
582t_yaml_data_to_typed_struct(yunpack_env_t * nonnull env,
583 const yaml_data_t * nonnull data,
584 const iop_struct_t * nonnull st,
585 void * nonnull out)
586{
587 const iop_struct_t *real_st = st;
588 int nb_fields_matched;
589
590 switch (data->type) {
591 case YAML_DATA_SCALAR:
592 if (data->scalar.type == YAML_SCALAR_NULL) {
593 break;
594 }
595 /* FALLTHROUGH */
596 case YAML_DATA_SEQ:
597 sb_setf(&env->err.buf, "cannot unpack %s into a %s",
598 yaml_data_get_type(data, false),
599 real_st->is_union ? "union" : "struct");
600 goto error;
601 case YAML_DATA_OBJ:
602 break;
603 }
604
605 if (data->tag.s) {
606 real_st = get_struct_from_tag(st, data->tag, &env->err.buf);
607 if (!real_st) {
608 env->err.span = data->tag_span;
609 real_st = st;
610 goto error;
611 }
612 }
613
614 if (st->is_union) {
615 return t_yaml_data_to_union(env, data, st, out);
616 }
617
618 if (iop_struct_is_class(real_st)) {
619 void **out_class = out;
620
621 if (check_class(env, real_st) < 0) {
622 goto error;
623 }
624
625 *out_class = ta_new_raw(byte, real_st->size, 8);
626 *(const iop_struct_t **)(*out_class) = real_st;
627 out = *out_class;
628 }
629
630 st = real_st;
631 nb_fields_matched = 0;
632 iop_struct_for_each_field(field_desc, field_st, real_st) {
633 const yaml_key_data_t *val;
634
635 val = yaml_data_get_field_value(data, field_desc->name);
636 if (val) {
637 if (t_yaml_fill_iop_field(env, val, field_st, field_desc, out) < 0)
638 {
639 goto error;
640 }
641 nb_fields_matched++;
642 } else {
643 if (t_yaml_skip_iop_field(env, field_st, field_desc, out) < 0) {
644 goto error;
645 }
646 }
647 }
648
649 if (data->type == YAML_DATA_OBJ
650 && unlikely(nb_fields_matched != data->obj->fields.len
651 && !(env->flags & IOP_UNPACK_IGNORE_UNKNOWN)))
652 {
653 /* There are fields in the YAML object that have not been matched.
654 * The handling of this error is kept in a cold path, as it is
655 * supposed to be rare. */
656 assert (nb_fields_matched < data->obj->fields.len);
657 yaml_data_find_extra_key(env, data, real_st);
658 goto error;
659 }
660
661 return 0;
662
663 error:
664 sb_prependf(&env->err.buf, "cannot unpack YAML as a `%pL` IOP %s: ",
665 &real_st->fullname, real_st->is_union ? "union" : "struct");
666 if (!env->err.span) {
667 env->err.span = &data->span;
668 }
669 return -1;
670}

685static int
686t_yaml_seq_to_iop_field(yunpack_env_t * nonnull env,
687 const yaml_data_t * nonnull data,
688 const iop_struct_t * nonnull st_desc,
689 const iop_field_t * nonnull fdesc, void * nonnull out)
690{
691 iop_array_i8_t *arr = out;
692 int size = 0;
693
694 if (fdesc->repeat != IOP_R_REPEATED) {
695 sb_sets(&env->err.buf, "cannot set a sequence in a non-array field");
696 env->err.span = &data->span;
697 return -1;
698 }
699
700 p_clear(arr, 1);
701 assert (data->type == YAML_DATA_SEQ);
702 tab_for_each_ptr(elem, &data->seq->datas) {
703 void *elem_out;
704
705 if (arr->len >= size) {
706 size = p_alloc_nr(size);
707 ta_realloc_from(&arr->tab, arr->len * fdesc->size,
708 size * fdesc->size, 8);
709 }
710 elem_out = (void *)(arr->tab + arr->len * fdesc->size);
711 RETHROW(t_yaml_data_to_iop_field(env, elem, st_desc, fdesc, true,
712 elem_out));
713 arr->len++;
714 }
715

721
722static int
723t_yaml_data_to_iop_field(yunpack_env_t *env, const yaml_data_t * nonnull data,
724 const iop_struct_t * nonnull st_desc,
725 const iop_field_t * nonnull fdesc,
726 bool in_array, void * nonnull out)
727{
728 bool struct_or_union;
729
730 struct_or_union = fdesc->type == IOP_T_STRUCT
731 || fdesc->type == IOP_T_UNION;
732 if (struct_or_union && iop_field_is_reference(fdesc)) {
733 out = iop_field_ptr_alloc(t_pool(), fdesc, out);
734 } else
735 if (fdesc->repeat == IOP_R_OPTIONAL && !iop_field_is_class(fdesc)) {
736 out = iop_field_set_present(t_pool(), fdesc, out);
737 }
738
739 if (fdesc->type == IOP_T_VOID) {
740 logger_trace(&_G.logger, 2,
741 "discarded %s from "YAML_POS_FMT" up to "YAML_POS_FMT
742 " as field %pL of struct %pL is void",
743 yaml_data_get_type(data, false),
744 YAML_POS_ARG(data->span.start),
745 YAML_POS_ARG(data->span.end), &fdesc->name,
746 &st_desc->fullname);
747 return 0;
748 }
749
750 if (!struct_or_union && data->tag.s) {
751 sb_setf(&env->err.buf, "specifying a tag on %s is not allowed",
752 yaml_data_get_type(data, true));
753 env->err.span = data->tag_span;
754 goto err;
755 }
756
757 switch (data->type) {
758 case YAML_DATA_SCALAR:
759 switch (t_yaml_scalar_to_iop_field(env, data, fdesc, in_array, out)) {
760 case YUNPACK_OK:
761 break;
762 case YUNPACK_INVALID_B64_VAL:
763 sb_setf(&env->err.buf, "the value must be encoded in base64");
764 env->err.span = &data->span;
765 goto err;
766 case YUNPACK_INVALID_ENUM_VAL:
767 sb_setf(&env->err.buf,
768 "the value is not valid for the enum `%pL`",
769 &fdesc->u1.en_desc->name);
770 env->err.span = &data->span;
771 goto err;
772 case YUNPACK_TYPE_MISMATCH:
773 yaml_set_type_mismatch_err(env, data, fdesc);
774 goto err;
775 case YUNPACK_OOB:
776 sb_setf(&env->err.buf,
777 "the value is out of range for the field of type %s",
778 iop_type_get_string_desc(fdesc->type));
779 env->err.span = &data->span;
780 goto err;
781 case YUNPACK_NULL_STRUCT_ERROR:
782 sb_setf(&env->err.buf,
783 "cannot unpack a null value into a struct with mandatory "
784 "fields");
785 env->err.span = &data->span;
786 goto err;
787 }
788 break;
789
790 case YAML_DATA_OBJ:
791 if (struct_or_union) {
792 if (!in_array && fdesc->repeat == IOP_R_REPEATED) {
793 sb_sets(&env->err.buf,
794 "cannot unpack an object into an array");
795 env->err.span = &data->span;
796 goto err;
797 }
798 if (t_yaml_data_to_typed_struct(env, data, fdesc->u1.st_desc,
799 out) < 0)
800 {
801 goto err;
802 }
803 } else {
804 yaml_set_type_mismatch_err(env, data, fdesc);
805 goto err;
806 }
807 break;
808
809 case YAML_DATA_SEQ:
810 if (in_array) {
811 sb_setf(&env->err.buf,
812 "cannot unpack an array as an element of an array, "
813 "did you insert one too many - ?");
814 env->err.span = &data->span;
815 /* XXX: do not goto err, the "cannot set field" will be set
816 * in the previous t_yaml_seq_to_iop_field.
817 */
818 return -1;
819 }
820 if (t_yaml_seq_to_iop_field(env, data, st_desc, fdesc, out) < 0) {
821 goto err;
822 }
823 break;
824
825 default:
826 assert (false);
827 return -1;
828 }
829
830 logger_trace(&_G.logger, 2,
831 "unpack %s from "YAML_POS_FMT" up to "YAML_POS_FMT
832 " into field %pL of struct %pL",
833 yaml_data_get_type(data, false),
834 YAML_POS_ARG(data->span.start),
835 YAML_POS_ARG(data->span.end), &fdesc->name,
836 &st_desc->fullname);
837 return 0;
838
839 err:
840 if (!in_array) {
841 sb_prependf(&env->err.buf, "cannot set field `%pL`: ", &fdesc->name);
842 }
843 return -1;
844}

1010static void
1011t_iop_field_to_yaml_data(const iop_field_t * nonnull fdesc,
1012 const void * nonnull ptr, int j, int flags,
1013 yaml_data_t * nonnull data)
1014{
1015 p_clear(data, 1);
1016
1017 switch (fdesc->type) {
1018#define CASE(n) \
1019 case IOP_T_I##n: \
1020 yaml_data_set_int(data, IOP_FIELD(int##n##_t, ptr, j)); \
1021 break; \
1022 case IOP_T_U##n: \
1023 yaml_data_set_uint(data, IOP_FIELD(uint##n##_t, ptr, j)); \
1024 break;
1025 CASE(8); CASE(16); CASE(32); CASE(64);
1026#undef CASE
1027
1028 case IOP_T_ENUM: {
1029 const void *v;
1030
1031 v = iop_enum_to_str_desc(fdesc->u1.en_desc,
1032 IOP_FIELD(int, ptr, j)).s;
1033 if (likely(v)) {
1034 yaml_data_set_string(data, LSTR(v));
1035 } else {
1036 yaml_data_set_int(data, IOP_FIELD(int, ptr, j));
1037 }
1038 } break;
1039
1040 case IOP_T_BOOL:
1041 yaml_data_set_bool(data, IOP_FIELD(bool, ptr, j));
1042 break;
1043
1044 case IOP_T_DOUBLE:
1045 yaml_data_set_double(data, IOP_FIELD(double, ptr, j));
1046 break;
1047
1048 case IOP_T_UNION:
1049 case IOP_T_STRUCT: {
1050 const void *v = iop_json_get_struct_field_value(fdesc, ptr, j);
1051
1052 t_iop_struct_to_yaml_data(fdesc->u1.st_desc, v, flags, data, false);
1053 } break;
1054
1055 case IOP_T_STRING:
1056 case IOP_T_XML: {
1057 lstr_t sv = IOP_FIELD(const lstr_t, ptr, j);
1058
1059 yaml_data_set_string(data, sv);
1060 } break;
1061
1062 case IOP_T_DATA: {
1063 lstr_t sv = IOP_FIELD(const lstr_t, ptr, j);
1064
1065 yaml_data_set_bytes(data, sv);
1066 } break;
1067
1068 case IOP_T_VOID:
1069 yaml_data_set_null(data);
1070 break;
1071
1072 default:
1073 abort();
1074 }
1075}

1077static void
1078t_append_iop_struct_to_fields(const iop_struct_t * nonnull desc,
1079 const void * nonnull value, int flags,
1080 yaml_data_t *data)
1081{
1082 const iop_field_t *fstart;
1083 const iop_field_t *fend;
1084
1085 if (desc->is_union) {
1086 fstart = get_union_field(desc, value);
1087 fend = fstart + 1;
1088 } else {
1089 fstart = desc->fields;
1090 fend = desc->fields + desc->fields_len;
1091 }
1092
1093 for (const iop_field_t *fdesc = fstart; fdesc < fend; fdesc++) {
1094 bool repeated = fdesc->repeat == IOP_R_REPEATED;
1095 const void *ptr;
1096 bool is_skipped = false;
1097 yaml_data_t field_data;
1098 int n;
1099
1100 ptr = iop_json_get_n_and_ptr(desc, flags, fdesc, value, &n,
1101 &is_skipped);
1102 if (is_skipped) {
1103 continue;
1104 }
1105
1106 if (n == 1 && !repeated) {
1107 t_iop_field_to_yaml_data(fdesc, ptr, 0, flags, &field_data);
1108 } else {
1109 t_yaml_data_new_seq(&field_data, n);
1110
1111 for (int j = 0; j < n; j++) {
1112 yaml_data_t elem;
1113
1114 t_iop_field_to_yaml_data(fdesc, ptr, j, flags, &elem);
1115 yaml_seq_add_data(&field_data, elem);
1116 }
1117 }
1118
1119 yaml_obj_add_field(data, lstr_dupc(fdesc->name), field_data);
1120 }
1121}

1123static void
1124t_iop_struct_to_yaml_data(const iop_struct_t * nonnull desc,
1125 const void * nonnull value, int flags,
1126 yaml_data_t * nonnull data, bool top_level)
1127{
1128 if (iop_struct_is_class(desc)) {
1129 qv_t(iop_struct) parents;
1130 const iop_struct_t *real_desc = *(const iop_struct_t **)value;
1131 int nb_fields = 0;
1132 lstr_t tag = LSTR_NULL_V;
1133
1134 e_assert(panic, !real_desc->class_attrs->is_abstract,
1135 "packing of abstract class '%*pM' is forbidden",
1136 LSTR_FMT_ARG(real_desc->fullname));
1137
1138 /* If this assert fails, you are exporting private classes through
1139 * a public interface... this is BAD!
1140 */
1141 assert (!real_desc->class_attrs->is_private);
1142
1143 /* Write type of class */
1144 if (desc != real_desc
1145 || !(flags & IOP_JPACK_SKIP_OPTIONAL_CLASS_NAMES)
1146 || top_level)
1147 {
1148 tag = lstr_dupc(real_desc->fullname);
1149 }
1150
1151 /* We want to write the fields in the order "master -> children", and
1152 * not "children -> master", so first build a qvector of the parents.
1153 */
1154 qv_inita(&parents, 8);
1155 do {
1156 qv_append(&parents, real_desc);
1157 nb_fields += real_desc->fields_len;
1158 real_desc = real_desc->class_attrs->parent;
1159 } while (real_desc);
1160
1161 /* Write fields of different levels */
1162 t_yaml_data_new_obj(data, nb_fields);
1163 for (int pos = parents.len; pos-- > 0; ) {
1164 t_append_iop_struct_to_fields(parents.tab[pos], value, flags,
1165 data);
1166 }
1167 qv_wipe(&parents);
1168
1169 data->tag = tag;
1170 } else {
1171 t_yaml_data_new_obj(data, desc->fields_len);
1172 t_append_iop_struct_to_fields(desc, value, flags, data);
1173 }
1174}

1242/* Convert an iopPath into a YAML presentation path.
1243 *
1244 * IOPs only handle structs as root data, so they always starts with a key
1245 * name, without the '.' prefix. In YAML we always add the '.' prefix.
1246 * In addition, the IOP path refers to values, and never to the seq identifier
1247 * or obj key, so add a '!' at the end:
1248 *
1249 * iop_path => '.<iop_path>!'
1250 *
1251 * If prefix is set, it must be removed from the iop_path.
1252 */
1253static lstr_t t_iop_path_to_yaml_path(const lstr_t iop_path,
1254 const lstr_t prefix)
1255{
1256 if (prefix.s) {
1257 pstream_t ps = ps_initlstr(&iop_path);
1258
1259 assert (lstr_startswith(iop_path, prefix));
1260 ps_skiplstr(&ps, prefix);
1261
1262 return t_lstr_fmt("%*pM!", PS_FMT_ARG(&ps));
1263 } else {
1264 return t_lstr_fmt(".%pL!", &iop_path);
1265 }
1266}

1313/* Generate a node mapping for the subfile "base".
1314 *
1315 * All subfiles that starts with the same path as the base means that the were
1316 * included from the base subfile, so they must be added to the document
1317 * presentation of the base subfile.
1318 */
1319static void
1320t_gen_mapping_from_common_inc(
1321 const iop_json_subfile__t * nonnull base,
1322 const iop_json_subfile__t * nullable parent,
1323 const qv_t(iop_json_subfile) * nonnull subfiles,
1324 int * nonnull index, const iop_struct_t *nullable st,
1325 const void * nullable value,
1326 yaml__presentation_node_mapping__t * nonnull out
1327)
1328{
1329 qv_t(pres_mapping) mappings;
1330 lstr_t relpath;
1331
1332 t_qv_init(&mappings, 0);
1333 while (*index < subfiles->len) {
1334 const iop_json_subfile__t *subfile = &subfiles->tab[*index];
1335
1336 if (lstr_startswith(subfile->iop_path, base->iop_path)
1337 && subfile->iop_path.len > base->iop_path.len)
1338 {
1339 int c = subfile->iop_path.s[base->iop_path.len];
1340
1341 if (c != '[' && c != '.') {
1342 break;
1343 }
1344 *index += 1;
1345
1346 t_gen_mapping_from_common_inc(subfile, base, subfiles, index, st,
1347 value, qv_growlen(&mappings, 1));
1348 } else {
1349 break;
1350 }
1351 }
1352
1353 if (parent) {
1354 relpath = t_get_relative_path(parent->file_path, base->file_path);
1355 } else {
1356 relpath = base->file_path;
1357 }
1358
1359 iop_init(yaml__presentation_node_mapping, out);
1360 out->path = t_iop_path_to_yaml_path(base->iop_path,
1361 parent ? parent->iop_path
1362 : LSTR_NULL_V);
1363 out->node.included = t_iop_new(yaml__presentation_include);
1364 if (mappings.len > 0) {
1365 out->node.included->document_presentation.mappings
1366 = IOP_TYPED_ARRAY_TAB(yaml__presentation_node_mapping, &mappings);
1367 }
1368
1369 if (st && included_iop_field_is_raw(st, value, base->iop_path)) {
1370 /* In IOP-JSON, if the include is done for a string/data/xml field,
1371 * it is included raw. */
1372 out->node.included->path = relpath;
1373 out->node.included->raw = true;
1374 } else {
1375 out->node.included->path = t_json_file_to_yaml_file(relpath);
1376 out->node.included->raw = false;
1377 }
1378}

1425 });

1430 index += 1;
1431 t_gen_mapping_from_common_inc(subfile, NULL, &includes, &index, st,
