34static iop_dso_t *
35handle_args(int argc, char **argv)
36{
37 const char *arg0 = NEXTARG(argc, argv);
38 iop_dso_t *dso;
39 SB_1k(err);
40 popt_t options[] = {
41 OPT_FLAG('h', "help", &opts_g.help, "show help"),
42 OPT_STR('d', "dso", &opts_g.dso_path, "path to IOP dso file"),
43 OPT_STR('m', "module", &opts_g.module,
44 "fullname of the IOP module to use"),
45 OPT_STR('w', "whitelist", &opts_g.whitelist_path,
46 "path to the RPCs whitelist file"),
47 OPT_STR(0, "description", &opts_g.description,
48 "Add a description of the openapi app"),
49 OPT_END(),
50 };
51
52 argc = parseopt(argc, argv, options, 0);
53 if (argc < 3 || opts_g.help) {
54 makeusage(!opts_g.help, arg0, "<name> <version> <route>", NULL,
55 options);
56 }
57
58 opts_g.title = NEXTARG(argc, argv);
59 opts_g.version = NEXTARG(argc, argv);
60 opts_g.route = NEXTARG(argc, argv);
61
62 if (!opts_g.dso_path) {
63 e_error("A dso file must be provided");
64 return NULL;
65 }
66 if (!opts_g.module) {
67 e_error("The name of the IOP module to use must be provided");
68 return NULL;
69 }
70
71 dso = iop_dso_open(opts_g.dso_path, LM_ID_BASE, &err);
72 if (!dso) {
73 e_error("cannot open dso `%s`: %pL", opts_g.dso_path, &err);
74 return NULL;
75 }
76
77 return dso;
78}

80static const iop_mod_t * nullable
81get_iop_module(const iop_dso_t * nonnull dso)
82{
83 lstr_t wanted_module = LSTR(opts_g.module);
84 const iop_mod_t *module = NULL;
85
86 qm_for_each_value(iop_mod, mod, &dso->mod_h) {
87 if (lstr_equal(mod->fullname, wanted_module)) {
88 module = mod;
89 break;
90 }
91 }
92
93 if (!module) {
94 e_error("Could not find the IOP module `%pL` in the DSO. Here are "
95 "the available modules:", &wanted_module);
96 qm_for_each_key(iop_mod, mod_name, &dso->mod_h) {
97 e_error(" `%pL`", &mod_name);
98 }
99 }
100
101 return module;
102}

104static int
105t_whitelist_rpcs(iop_openapi_t *oa)
106{
107 FILE *file;
108 SB_1k(sb);
109 int res;
110
111 if (!opts_g.whitelist_path) {
112 return 0;
113 }
114
115 file = fopen(opts_g.whitelist_path, "r");
116 if (!file) {
117 e_error("cannot open whitelist file `%s`: %m", opts_g.whitelist_path);
118 return -1;
119 }
120
121 while ((res = sb_getline(&sb, file)) > 0) {
122 sb.len--;
123 t_iop_openapi_whitelist_rpc(oa, LSTR_SB_V(&sb));
124 sb_reset(&sb);
125 }
126 if (res < 0) {
127 e_error("error while reading whitelist file `%s`: %m",
128 opts_g.whitelist_path);
129 return -1;
130 }
131
132 return 0;
133}

142static int
143generate_openapi(const iop_mod_t * nonnull module)
144{
145 t_scope;
146 iop_openapi_t *oa;
147 yaml_data_t yaml;
148 yaml_pack_env_t *env;
149 SB_1k(err);
150
151 oa = t_new_iop_openapi(LSTR(opts_g.title), LSTR(opts_g.version),
152 module, LSTR(opts_g.route));
153 if (opts_g.description) {
154 t_iop_openapi_set_description(oa, LSTR(opts_g.description));
155 }
156
157 RETHROW(t_whitelist_rpcs(oa));
158
159 if (t_iop_openapi_to_yaml(oa, &yaml, &err) < 0) {
160 e_error("could not generate the OpenAPI application: %pL", &err);
161 return -1;
162 }
163
164 env = t_yaml_pack_env_new();
165 t_yaml_pack(env, &yaml, yaml_pack_write_stdout, NULL, &err);
166 printf("\n");
167
168 return 0;
169}

172{
