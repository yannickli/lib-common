115void sb_add_slashes(sb_t *sb, const void *_data, int len,
116 const char *toesc, const char *esc)
117{
118 uint32_t buf[BITS_TO_ARRAY_LEN(uint32_t, 256)] = { 0, };
119 uint8_t repl[256];
120 const byte *p = _data, *end = p + len;
121
122 while (*toesc) {
123 byte c = *toesc++;
124 SET_BIT(buf, c);
125 repl[c] = *esc++;
126 }
127
128 if (!TST_BIT(buf, '\\')) {
129 SET_BIT(buf, '\\');
130 repl['\\'] = '\\';
131 }
132
133 sb_grow(sb, len);
134 while (p < end) {
135 const byte *q = p;
136
137 while (p < end && !TST_BIT(buf, *p))
138 p++;
139 sb_add(sb, q, p - q);
140
141 while (p < end && TST_BIT(buf, *p)) {
142 byte c = repl[*p++];
143
144 if (c) {
145 char *s = sb_growlen(sb, 2);
146 s[0] = '\\';
147 s[1] = c;
148 }
149 }
150 }
151}

153void sb_add_unslashes(sb_t *sb, const void *_data, int len,
154 const char *tounesc, const char *unesc)
155{
156 uint32_t buf[BITS_TO_ARRAY_LEN(uint32_t, 256)] = { 0, };
157 uint8_t repl[256];
158 const byte *p = _data, *end = p + len;
159
160 while (*tounesc) {
161 byte c = *tounesc++;
162 SET_BIT(buf, c);
163 repl[c] = *unesc++;
164 }
165
166 if (!TST_BIT(buf, '\\')) {
167 SET_BIT(buf, '\\');
168 repl['\\'] = '\\';
169 }
170
171 while (p < end) {
172 const byte *q = p;
173
174 /* -1 so that we always have a char after \ */
175 p = memchr(p, '\\', end - p - 1);
176 if (!p) {
177 p = q;
178 break;
179 }
180 sb_add(sb, q, p - q);
181
182 if (TST_BIT(buf, p[1])) {
183 sb_addc(sb, repl[*++p]);
184 } else {
185 sb_addc(sb, '\\');
186 }
187 p++;
188 }
189 sb_add(sb, p, end - p);
190}

212int sb_add_expandenv(sb_t *sb, const void *data, int len)
213{
214 sb_t orig = *sb;
215 pstream_t ps = ps_init(data, len);
216 SB_1k(env_buf);
217
218 while (!ps_done(&ps)) {
219 pstream_t n;
220 pstream_t env_name;
221 const char *var;
222 bool has_env = false;
223
224 if (ps_get_ps_chr_and_skip(&ps, '$', &n) < 0) {
225 n = ps;
226 ps = ps_init(NULL, 0);
227 } else {
228 pstream_t prev = n;
229 int slashes = 0;
230
231 while (ps_endswithstr(&prev, "\\")) {
232 slashes++;
233 prev.s_end--;
234 }
235 if ((slashes & 1)) {
236 n.s_end++;
237 } else {
238 has_env = true;
239 }
240 }
241
242 sb_add_unslashes(sb, n.s, ps_len(&n), "$", "$");
243
244 if (!has_env) {
245 break;
246 }
247
248 if (ps_skipc(&ps, '{') < 0) {
249 env_name = ps_get_span(&ps, &ctype_iscvar);
250 } else
251 if (ps_get_ps_chr_and_skip(&ps, '}', &env_name) < 0) {
252 return __sb_rewind_adds(sb, &orig);
253 }
254
255 if (ps_done(&env_name)) {
256 return __sb_rewind_adds(sb, &orig);
257 }
258
259 sb_set(&env_buf, env_name.s, ps_len(&env_name));
260 var = getenv(env_buf.data);
261
262 if (!var) {
263 return __sb_rewind_adds(sb, &orig);
264 }
265
266 sb_adds(sb, var);
267 }
268 return 0;
269}

279void sb_add_unquoted(sb_t *sb, const void *data, int len)
280{
281 const char *p = data, *end = p + len;
282
283 while (p < end) {
284 const char *q = p;
285 int c;
286
287 p = memchr(q, '\\', end - q);
288 if (!p) {
289 sb_add(sb, q, end - q);
290 return;
291 }
292 sb_add(sb, q, p - q);
293 p += 1;
294
295 if (p == end) {
296 sb_addc(sb, '\\');
297 return;
298 }
299
300 switch (*p) {
301 case 'a': case 'b': case 'e': case 't': case 'n': case 'v':
302 case 'f': case 'r': case '\\': case '"': case '\'':
303 sb_addc(sb, __c_unescape[(unsigned char)*p++]);
304 continue;
305
306 case '0'...'7':
307 c = *p++ - '0';
308 if (p < end && *p >= '0' && *p <= '7')
309 c = (c << 3) + *p++ - '0';
310 if (c < '\040' && p < end && *p >= '0' && *p <= '7')
311 c = (c << 3) + *p++ - '0';
312 sb_addc(sb, c);
313 continue;
314
315 case 'x':
316 if (end - p < 3)
317 break;
318 c = hexdecode(p + 1);
319 if (c < 0)
320 break;
321 p += 3;
322 sb_addc(sb, c);
323 continue;
324
325 case 'u':
326 if (end - p < 5)
327 break;
328 c = (hexdecode(p + 1) << 8) | hexdecode(p + 3);
329 if (c < 0)
330 break;
331 p += 5;
332 sb_adduc(sb, c);
333 continue;
334 }
335 sb_addc(sb, '\\');
336 }
337}

393void sb_urldecode(sb_t *sb)
394{
395 const char *tmp, *r, *end = sb_end(sb);
396 char *w;
397
398 r = w = memchr(sb->data, '%', sb->len);
399 if (!r)
400 return;
401
402 for (;;) {
403 int c;
404
405 if (end - r < 3) {
406 *w++ = *r++;
407 continue;
408 }
409 c = hexdecode(r + 1);
410 if (c < 0) {
411 *w++ = *r++;
412 continue;
413 }
414 *w++ = c;
415 r += 3;
416
417 r = memchr(tmp = r, '%', end - r);
418 if (!r) {
419 memmove(w, tmp, end - tmp);
420 w += end - tmp;
421 __sb_fixlen(sb, w - sb->data);
422 return;
423 }
424 memmove(w, tmp, r - tmp);
425 w += r - tmp;
426 }

489int sb_add_xmlunescape(sb_t *sb, const void *data, int len)
490{
491 sb_t orig = *sb;
492 const char *p = data, *end = p + len;
493
494 while (p < end) {
495 const char *q = p;
496 const char *semi;
497
498 while (p < end && *p != '<' && *p != '&')
499 p++;
500 sb_add(sb, q, p - q);
501
502 if (p == end)
503 return 0;
504
505 if (*p++ == '<') {
506 /* strip out comments */
507 if (p + 3 <= end && !memcmp(p, "!--", 3)) {
508 p += 3;
509 for (;;) {
510 p = memchr(p, '-', end - p);
511 if (!p || p + 3 > end)
512 goto error;
513 if (!memcmp(p, "-->", 3)) {
514 p += 3;
515 break;
516 }
517 p++;
518 }
519 continue;
520 }
521
522 /* extract CDATA stuff */
523 if (p + 8 <= end && !memcmp(p, "![CDATA[", 8)) {
524 p += 8;
525 for (q = p;;) {
526 p = memchr(p, ']', end - p);
527 if (!p || p + 3 > end)
528 goto error;
529 if (!memcmp(p, "]]>", 3)) {
530 sb_add(sb, q, p - q);
531 p += 3;
532 break;
533 }
534 p++;
535 }
536 continue;
537 }
538 goto error;
539 }
540
541 /* entities: we have (p[-1] == '&') */
542 semi = memchr(p, ';', end - p);
543 if (!semi || p + 1 > semi)
544 goto error;
545
546 if (*p == '#') {
547 int c = 0;
548
549 if (semi - p > 7)
550 goto error;
551
552 if (p[1] == 'x') {
553 for (p += 2; p < semi; p++) {
554 c = (c << 4) | hexdigit(*p);
555 }
556 if (c < 0)
557 goto error;
558 } else {
559 c = memtoip(p + 1, semi - p - 1, (const byte **)&p);
560 if (p != semi)
561 goto error;
562 }
563 sb_adduc(sb, c);
564 } else {
565 /* &lt; &gt; &apos; &amp; &quot */
566 switch (semi - p) {
567 case 2:
568 if (!memcmp(p, "lt", 2)) {
569 sb_addc(sb, '<');
570 break;
571 }
572 if (!memcmp(p, "gt", 2)) {
573 sb_addc(sb, '>');
574 break;
575 }
576 goto error;
577
578 case 3:
579 if (!memcmp(p, "amp", 3)) {
580 sb_addc(sb, '&');
581 break;
582 }
583 goto error;
584
585 case 4:
586 if (!memcmp(p, "apos", 4)) {
587 sb_addc(sb, '\'');
588 break;
589 }
590 if (!memcmp(p, "quot", 4)) {
591 sb_addc(sb, '"');
592 break;
593 }
594 goto error;
595
596 default:
597 goto error;
598 }
599 }
600
601 p = semi + 1;
602 }
603 return 0;
604
605 error:
606 return __sb_rewind_adds(sb, &orig);
607}

610/* OG: should take width as a parameter */
611void sb_add_qpe(sb_t *sb, const void *data, int len)
612{
613 int c, i = 0, j = 0, col = 0;
614 const byte *src = data;
615
616 sb_grow(sb, len);
617 while (i < len) {
618 if (col + i - j >= 75) {
619 sb_add(sb, src + j, 75 - col);
620 sb_adds(sb, "=\r\n");
621 j += 75 - col;
622 col = 0;
623 }
624 if (test_quoted_printable(c = src[i++]))
625 continue;
626 /* only encode '.' if at the beginning of a line */
627 if (c == '.' && col)
628 continue;
629 /* encode spaces only at end on line */
630 if (isblank(c)) {
631 if (!(i + 2 <= len && !memcmp(src + i, "\r\n", 2)))
632 continue;
633 }
634 /* \r\n remain the same and reset col to 0 */
635 if (c == '\r' && i + 1 <= len && src[i] == '\n') {
636 i++;
637 sb_add(sb, src + j, i - j);
638 col = 0;
639 } else {
640 char *s;
641
642 sb_add(sb, src + j, i - 1 - j);
643 col += i - 1 - j ;
644 if (col > 75 - 3) {
645 sb_adds(sb, "=\r\n");
646 col = 0;
647 }
648 s = sb_growlen(sb, 3);
649 s[0] = '=';
650 s[1] = __str_digits_upper[(c >> 4) & 0xf];
651 s[2] = __str_digits_upper[(c >> 0) & 0xf];
652 col += 3;
653 }
654 j = i;
655 }
656 sb_add(sb, src + j, i - j);
657 if (sb->data[sb->len - 1] != '\n')
658 sb_adds(sb, "=\r\n");
659}

662void sb_add_unqpe(sb_t *sb, const void *data, int len)
663{
664 const byte *p = data, *end = p + len;
665 sb_grow(sb, len);
666
667 while (p < end) {
668 const byte *q = p;
669
670 while (p < end && *p != '=' && *p != '\r' && *p)
671 p++;
672 sb_add(sb, q, p - q);
673
674 if (p >= end)
675 return;
676 switch (*p++) {
677 case '\0':
678 return;
679
680 case '=':
681 if (end - p < 2) {
682 sb_addc(sb, '=');
683 } else
684 if (p[0] == '\r' && p[1] == '\n') {
685 p += 2;
686 } else {
687 int c = hexdecode((const char *)p);
688 if (c < 0) {
689 sb_addc(sb, '=');
690 } else {
691 sb_addc(sb, c);
692 p += 2;
693 }
694 }
695 break;
696
697 case '\r':
698 if (p < end && *p == '\n') {
699 sb_addc(sb, *p++);
700 } else {
701 sb_addc(sb, '\r');
702 }
703 break;
704 }
705 }

710/* computes a slightly overestimated size to write srclen bytes with `ppline`
711 * packs per line, not knowing if we start at column 0 or not.
712 *
713 * The over-estimate is of at most 4 bytes, we can live with that.
714 */
715static int b64_rough_size(int srclen, int ppline)
716{
717 int nbpacks = ((srclen + 2) / 3);
718
719 if (ppline < 0)
720 return 4 * nbpacks;
721 /*
722 * Worst case is we're at `4 * ppline` column so we have to add a \r\n
723 * straight away plus what is need for the rest.
724 */
725 return 4 * nbpacks + 2 + 2 * DIV_ROUND_UP(nbpacks, ppline);
726}

731 if (width == 0) {
732 width = 19; /* 76 characters + \r\n */
733 } else {
734 /* XXX: >>2 keeps the sign, unlike /4 */
735 width >>= 2;
736 }
737 ctx->packs_per_line = width;
738 sb_grow(dst, b64_rough_size(len, width));
739}

741static void _sb_add_b64_update(sb_t *dst, const void *src0, int len,
742 sb_b64_ctx_t *ctx, const char table[64])
743{
744 short ppline = ctx->packs_per_line;
745 short pack_num = ctx->pack_num;
746 const byte *src = src0;
747 const byte *end = src + len;
748 unsigned pack;
749 char *data;
750
751 if (ctx->trail_len + len < 3) {
752 memcpy(ctx->trail + ctx->trail_len, src, len);
753 ctx->trail_len += len;
754 return;
755 }
756
757 data = sb_grow(dst, b64_rough_size(ctx->trail_len + len, ppline));
758
759 if (ctx->trail_len) {
760 pack = ctx->trail[0] << 16;
761 pack |= (ctx->trail_len == 2 ? ctx->trail[1] : *src++) << 8;
762 pack |= *src++;
763 goto initialized;
764 }
765
766 do {
767 pack = *src++ << 16;
768 pack |= *src++ << 8;
769 pack |= *src++ << 0;
770
771 initialized:
772 *data++ = table[(pack >> (3 * 6)) & 0x3f];
773 *data++ = table[(pack >> (2 * 6)) & 0x3f];
774 *data++ = table[(pack >> (1 * 6)) & 0x3f];
775 *data++ = table[(pack >> (0 * 6)) & 0x3f];
776
777 if (ppline > 0 && ++pack_num >= ppline) {
778 pack_num = 0;
779 *data++ = '\r';
780 *data++ = '\n';
781 }
782 } while (src + 3 <= end);
783
784 memcpy(ctx->trail, src, end - src);
785 ctx->trail_len = end - src;
786 ctx->pack_num = pack_num;
787 __sb_fixlen(dst, data - dst->data);
788}

865static int _sb_add_unb64(sb_t *sb, const void *data, int len,
866 const unsigned char table[256])
867{
868 const byte *src = data, *end = src + len;
869 sb_t orig = *sb;
870
871 while (src < end) {
872 byte in[4];
873 int ilen = 0;
874 char *s;
875
876 while (ilen < 4 && src < end) {
877 int c = *src++;
878
879 if (isspace(c))
880 continue;
881
882 /*
883 * '=' must be at the end, they can only be 1 or 2 of them
884 * IOW we must have 2 or 3 fill `in` bytes already.
885 *
886 * And after them we can only have spaces. No data.
887 *
888 */
889 if (c == '=') {
890 if (ilen < 2)
891 goto error;
892 if (ilen == 2) {
893 while (src < end && isspace(*src))
894 src++;
895 if (src >= end || *src++ != '=')
896 goto error;
897 }
898 while (src < end) {
899 if (!isspace(*src++))
900 goto error;
901 }
902
903 s = sb_growlen(sb, ilen - 1);
904 if (ilen == 3) {
905 s[1] = (in[1] << 4) | (in[2] >> 2);
906 }
907 s[0] = (in[0] << 2) | (in[1] >> 4);
908 return 0;
909 }
910
911 in[ilen++] = c = table[c];
912 if (unlikely(c == 255))
913 goto error;
914 }
915
916 if (ilen == 0)
917 return 0;
918
919 if (ilen != 4)
920 goto error;
921
922 s = sb_growlen(sb, 3);
923 s[0] = (in[0] << 2) | (in[1] >> 4);
924 s[1] = (in[1] << 4) | (in[2] >> 2);
925 s[2] = (in[2] << 6) | (in[3] >> 0);
926 }
927 return 0;
928
929error:
930 return __sb_rewind_adds(sb, &orig);
931}

945void sb_add_csvescape(sb_t *sb, int sep, const void *data, int len)
946{
947 static ctype_desc_t ctype_needs_escape;
948 pstream_t ps = ps_init(data, len);
949 pstream_t cspan;
950
951 ctype_desc_build(&ctype_needs_escape, "\"\n\r");
952 SET_BIT(ctype_needs_escape.tab, sep);
953
954 cspan = ps_get_cspan(&ps, &ctype_needs_escape);
955 if (ps_done(&ps)) {
956 /* No caracter needing escaping was found, just copy the input
957 * string. */
958 sb_add(sb, data, len);
959 return;
960 }
961
962 /* There is at least one special character found in the input string, so
963 * the whole string has to be double-quoted, and the double-quotes have
964 * to be escaped by double-quotes.
965 */
966 sb_grow(sb, len + 2);
967 sb_addc(sb, '"');
968 sb_add_ps(sb, cspan);

971 if (ps_get_ps_chr_and_skip(&ps, '"', &cspan) < 0) {
972 sb_add_ps(sb, ps);
973 break;
974 }
975 cspan.s_end++;
976 sb_add_ps(sb, cspan);
977 sb_addc(sb, '"');
978 }

980 sb_addc(sb, '"');

983/*{{{ Punycode (RFC 3492) */
984
985#define PUNYCODE_DELIMITER '-'
986#define PUNYCODE_BASE 36
987#define PUNYCODE_TMIN 1
988#define PUNYCODE_TMAX 26
989#define PUNYCODE_SKEW 38
990#define PUNYCODE_DAMP 700
991#define PUNYCODE_INITIAL_BIAS 72
992#define PUNYCODE_INITIAL_N 0x80
993
994static int punycode_adapt_bias(int delta, int numpoints, bool firsttime)
995{
996 int k = 0;
997
998 if (firsttime) {
999 delta /= PUNYCODE_DAMP;
1000 } else {
1001 delta /= 2;
1002 }
1003 delta += delta / numpoints;
1004
1005 while (delta > ((PUNYCODE_BASE - PUNYCODE_TMIN) * PUNYCODE_TMAX) / 2) {
1006 delta /= PUNYCODE_BASE - PUNYCODE_TMIN;
1007 k += PUNYCODE_BASE;
1008 }
1009
1010 return k + (((PUNYCODE_BASE - PUNYCODE_TMIN + 1) * delta)
1011 / (delta + PUNYCODE_SKEW));
1012}
1013
1014static inline void punycode_output(sb_t *sb, int code_point)
1015{
1016 int c;
1017
1018 assert (code_point >= 0);
1019 assert (code_point < PUNYCODE_BASE);
1020
1021 if (code_point < PUNYCODE_TMAX) {
1022 /* 0..25 -> a..z */
1023 c = 'a' + code_point;
1024 } else {
1025 /* 26..35 -> 0..9 */
1026 c = '0' - PUNYCODE_TMAX + code_point;
1027 }
1028 sb_addc(sb, c);
1029}
1030
1031static inline void
1032punycode_output_variable_length_integer(sb_t *sb, int bias, uint32_t q)
1033{
1034 for (int k = PUNYCODE_BASE; ; k += PUNYCODE_BASE) {
1035 uint32_t t;
1036
1037 if (k <= bias) {
1038 t = PUNYCODE_TMIN;
1039 } else
1040 if (k >= bias + PUNYCODE_TMAX) {
1041 t = PUNYCODE_TMAX;
1042 } else {
1043 t = k - bias;
1044 }
1045 if (q < t) {
1046 break;
1047 }
1048 punycode_output(sb, t + ((q - t) % (PUNYCODE_BASE - t)));
1049 q = (q - t) / (PUNYCODE_BASE - t);
1050 }
1051
1052 punycode_output(sb, q);
1053}

1055int sb_add_punycode_vec(sb_t *sb, const uint32_t *code_points,
1056 int nb_code_points)
1057{
1058 /* This function implementation is a slightly enhanced version of the
1059 * algorithm described in section 6.3 (Encoding procedure) of the
1060 * RFC 3492.
1061 * The variables m, n, delta, bias and h have the same meaning as in the
1062 * RFC.
1063 */
1064 uint64_t *point_pos_pairs = p_alloca(uint64_t, nb_code_points);
1065 int point_pos_pairs_len = 0, nb_basic_code_points = 0;
1066 uint32_t n = PUNYCODE_INITIAL_N;
1067 uint64_t delta = 0;
1068 int bias = PUNYCODE_INITIAL_BIAS;
1069 int i, h;
1070
1071 /* Basic code point segregation */
1072 for (i = 0; i < nb_code_points; i++) {
1073 if (isascii(code_points[i])) {
1074 nb_basic_code_points++;
1075 sb_addc(sb, code_points[i]);
1076 } else {
1077 point_pos_pairs[point_pos_pairs_len++]
1078 = ((uint64_t)code_points[i] << 32) + i;
1079 }
1080 }
1081 if (nb_basic_code_points > 0) {
1082 sb_addc(sb, PUNYCODE_DELIMITER);
1083 }
1084
1085 /* Sort point_pos_pairs (which is the array of the non-basic code points)
1086 * in first by code point value, then by position in the input sequence */
1087 dsort64(point_pos_pairs, point_pos_pairs_len);
1088
1089 /* Insertion unsort coding */
1090 i = 0;
1091 h = nb_basic_code_points;
1092 while (i < point_pos_pairs_len) {
1093 uint32_t m = point_pos_pairs[i] >> 32;
1094 uint32_t last_pos = 0;
1095
1096 delta += (m - n) * (h + 1);
1097 n = m;
1098
1099 while (i < point_pos_pairs_len && point_pos_pairs[i] >> 32 == m) {
1100 uint32_t point_pos = point_pos_pairs[i] & 0xffffffff;
1101
1102 for (uint32_t j = last_pos; j < point_pos; j++) {
1103 if (code_points[j] < n) {
1104 delta++;
1105 }
1106 }
1107
1108 THROW_ERR_IF(delta > UINT32_MAX);
1109 punycode_output_variable_length_integer(sb, bias, delta);
1110
1111 bias = punycode_adapt_bias(delta, h + 1,
1112 h == nb_basic_code_points);
1113 last_pos = point_pos;
1114 delta = 0;
1115 i++;
1116 h++;
1117 }
1118
1119 for (int j = last_pos; j < nb_code_points; j++) {
1120 if (code_points[j] < n) {
1121 delta++;
1122 }
1123 }
1124
1125 delta++;
1126 n++;
1127 }
1128
1129 return 0;
1130}

1169static int idna_label_to_ascii(sb_t *sb, const char *label, int label_size,
1170 qv_t(u32) *code_points,
1171 bool is_ascii, unsigned flags)
1172{
1173 int initial_len = sb->len;
1174
1175 THROW_ERR_IF(label_size == 0 || code_points->len == 0);
1176
1177 if (flags & IDNA_USE_STD3_ASCII_RULES) {
1178 /* Verify the absence of leading and trailing hyphen-minus */
1179 THROW_ERR_IF(code_points->tab[0] == '-');
1180 THROW_ERR_IF(code_points->tab[code_points->len - 1] == '-');
1181 }
1182
1183 if (is_ascii) {
1184 /* All characters of this label are ASCII ones, just output the input
1185 * string. */
1186 if (flags & IDNA_ASCII_TOLOWER) {
1187 const char *end = label + label_size;
1188
1189 sb_grow(sb, label_size);
1190 while (label < end) {
1191 sb_addc(sb, tolower(*label++));
1192 }
1193 } else {
1194 sb_add(sb, label, label_size);
1195 }
1196 } else {
1197 is_ascii = true;
1198
1199 /* Perform the last NAMEPREP operations on ASCII characters (which
1200 * were not performed in idna_nameprep). */
1201 tab_for_each_ptr(c, code_points) {
1202 if (isascii(*c)) {
1203 *c = unicode_tolower(*c);
1204 } else {
1205 is_ascii = false;
1206 }
1207 }
1208
1209 if (unlikely(is_ascii)) {
1210 /* We were called in this function with is_ascii to false, but all
1211 * code points are actually ASCII ones. This can happen if
1212 * NAMEPREP filtered all the non-ASCII characters. In that case,
1213 * just output the code points without encoding them. */
1214 tab_for_each_entry(c, code_points) {
1215 sb_addc(sb, c);
1216 }
1217 } else {
1218 /* Verify that the sequence does NOT begin with the ACE prefix */
1219 if (code_points->len >= IDNA_ACE_PFX_LEN) {
1220 uint32_t ace_pfx[] = { 'x', 'n', '-', '-' };
1221
1222 THROW_ERR_IF(!memcmp(code_points->tab, ace_pfx,
1223 IDNA_ACE_PFX_LEN * sizeof(uint32_t)));
1224 }
1225
1226 /* Encode the sequence using the punycode encoding */
1227 sb_add(sb, IDNA_ACE_PFX, IDNA_ACE_PFX_LEN);
1228 RETHROW(sb_add_punycode_vec(sb, code_points->tab,
1229 code_points->len));
1230 }
1231 }
1232
1233 /* Verify that the number of code points is in the range 1 to 63
1234 * inclusive. */
1235 THROW_ERR_IF(sb->len - initial_len > 63);
1236
1237 return 0;
1238}

1240static inline int idna_nameprep(qv_t(u32) *code_points, int c, unsigned flags)

1245 * The tables referenced here are defined in RFC 3454.
1246 *
1247 * NAMEPREP is made of 5 steps:
1248 * - Mapping (tables B.1 and B.2); we are doing the exaxt checks for
1249 * table B.1, but table B.2 (which is a tolower table) is approximated
1250 * with unicode_tolower.
1251 * - Normalization; not implemented here.
1252 * - Prohibited Output; only prohibited characters of section 5.8 are
1253 * checked, because they are included in the NAMEPREP prohibited
