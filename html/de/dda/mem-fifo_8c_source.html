82static ALWAYS_INLINE mem_page_t *pageof(mem_block_t *blk)
83{
84 return (mem_page_t *)((char *)blk - blk->page_offs);

87static mem_page_t *mem_page_new(mem_fifo_pool_t *mfp, uint32_t minsize)
88{
89 mem_page_t *page = mfp->freepage;
90 uint32_t mapsize;
91
92 if (page && page->size >= minsize) {
93 mfp->freepage = NULL;
94 return page;
95 }
96
97 if (minsize < mfp->page_size - sizeof(mem_page_t)) {
98 mapsize = mfp->page_size;
99 } else {
100 mapsize = ROUND_UP(minsize + sizeof(mem_page_t), PAGE_SIZE);
101 }
102
103 page = (mem_page_t *) pa_new(byte, mapsize, 8);
104
105 page->size = mapsize - sizeof(mem_page_t);
106 mem_tool_disallow_memory(page->area, page->size);
107 mfp->nb_pages++;
108 mfp->map_size += mapsize;
109
110#ifdef MEM_BENCH
111 mfp->mem_bench.malloc_calls++;
112 mfp->mem_bench.current_allocated += mapsize;
113 mfp->mem_bench.total_allocated += mapsize;
114 mem_bench_update(&mfp->mem_bench);
115 mem_bench_print_csv(&mfp->mem_bench);
116#endif
117
118 return page;
119}

121static void mem_page_reset(mem_page_t *page)
122{
123 mem_tool_allow_memory(page->area, page->used_size, false);
124 p_clear(page->area, page->used_size);
125 mem_tool_disallow_memory(page->area, page->size);
126
127 page->used_blocks = 0;
128 page->used_size = 0;
129 page->last = NULL;
130}

132static void mem_page_delete(mem_fifo_pool_t *mfp, mem_page_t **pagep)
133{
134 mem_page_t *page = *pagep;
135
136 if (page) {
137#ifdef MEM_BENCH
138 mfp->mem_bench.current_allocated -= page->size + sizeof(mem_page_t);
139 mem_bench_update(&mfp->mem_bench);
140 mem_bench_print_csv(&mfp->mem_bench);
141#endif
142
143 mfp->nb_pages--;
144 mfp->map_size -= page->size + sizeof(mem_page_t);
145 mem_tool_allow_memory(page, page->size + sizeof(mem_page_t), true);
146 p_delete(pagep);
147 }
148}

155static void *mfp_alloc(mem_pool_t *_mfp, size_t size, size_t alignment,
156 mem_flags_t flags)
157{
158 mem_fifo_pool_t *mfp = container_of(_mfp, mem_fifo_pool_t, funcs);
159 mem_block_t *blk;
160 mem_page_t *page;
161 size_t req_size = size;
162
163#ifdef MEM_BENCH
164 proctimer_t ptimer;
165 proctimer_start(&ptimer);
166#endif
167
168 if (alignment > 8) {
169 e_panic("mem_fifo_pool does not support alignments greater than 8");
170 }
171
172 if (unlikely(size == 0)) {
173 return MEM_EMPTY_ALLOC;
174 }
175
176 page = mfp->current;
177 /* Must round size up to keep proper alignment */
178 size = ROUND_UP((unsigned)size + sizeof(mem_block_t), 8);
179 if (mem_page_size_left(page) < size) {
180#ifndef NDEBUG
181 if (unlikely(!mfp->alive)) {
182 e_panic("trying to allocate from a dead pool");
183 }
184#endif
185 if (unlikely(!page->used_blocks)) {
186 if (page->size >= size) {
187 mem_page_reset(page);
188 } else {
189 mem_page_delete(mfp, &page);
190 page = mfp->current = mem_page_new(mfp, size);
191 }
192 } else {
193 page = mfp->current = mem_page_new(mfp, size);
194 }
195#ifdef MEM_BENCH
196 mfp->mem_bench.alloc.nb_slow_path++;
197#endif
198 }
199
200 blk = acast(mem_block_t, page->area + page->used_size);
201 mem_tool_allow_memory(blk, sizeof(*blk), true);
202 mem_tool_malloclike(blk->area, req_size, 0, true);
203 blk->page_offs = (uintptr_t)blk - (uintptr_t)page;
204 blk->blk_size = size;
205 mem_tool_disallow_memory(blk, sizeof(*blk));
206
207 mfp->occupied += size;
208 page->used_size += size;
209 page->used_blocks++;
210
211#ifdef MEM_BENCH
212 proctimer_stop(&ptimer);
213 proctimerstat_addsample(&mfp->mem_bench.alloc.timer_stat, &ptimer);
214
215 mfp->mem_bench.alloc.nb_calls++;
216 mfp->mem_bench.current_used = mfp->occupied;
217 mfp->mem_bench.total_requested += req_size;
218 mem_bench_update(&mfp->mem_bench);
219 mem_bench_print_csv(&mfp->mem_bench);
220#endif
221
222 return page->last = blk->area;
223}

225static void mfp_free(mem_pool_t *_mfp, void *mem)
226{
227 mem_fifo_pool_t *mfp = container_of(_mfp, mem_fifo_pool_t, funcs);
228 mem_block_t *blk;
229 mem_page_t *page;
230
231#ifdef MEM_BENCH
232 proctimer_t ptimer;
233 proctimer_start(&ptimer);
234#endif
235
236 if (!mem || unlikely(mem == MEM_EMPTY_ALLOC)) {
237 return;
238 }
239
240 blk = container_of(mem, mem_block_t, area);
241 mem_tool_allow_memory(blk, sizeof(*blk), true);
242 page = pageof(blk);
243 mfp->occupied -= blk->blk_size;
244 mem_tool_freelike(mem, blk->blk_size - sizeof(*blk), 0);
245 mem_tool_disallow_memory(blk, sizeof(*blk));
246
247 if (--page->used_blocks > 0) {
248#ifdef MEM_BENCH
249 proctimer_stop(&ptimer);
250 proctimerstat_addsample(&mfp->mem_bench.free.timer_stat, &ptimer);
251
252 mfp->mem_bench.free.nb_calls++;
253 mfp->mem_bench.current_used = mfp->occupied;
254 mem_bench_update(&mfp->mem_bench);
255 mem_bench_print_csv(&mfp->mem_bench);
256#endif
257 return;
258 }
259
260 /* specific case for a dying pool */
261 if (unlikely(!mfp->alive)) {
262 mem_page_delete(mfp, &page);
263 if (mfp->nb_pages == 0)
264 p_delete(mfp->owner);
265 return;
266 }
267
268 /* this was the last block,
269 * reset this page unless it is the current one
270 */
271 if (page != mfp->current) {
272 /* keep the page around if we have none kept around yet */
273 if (mfp->freepage) {
274 /* if the current page is almost full, better replace the current
275 * by this one than deleting it and wasting the freepage soon
276 */
277 if (page->size >
278 8 * (mfp->current->size - mfp->current->used_size))
279 {
280 if (mfp->current->used_blocks == 0) {
281 mem_page_delete(mfp, &mfp->current);
282 }
283 mem_page_reset(page);
284 mfp->current = page;
285 } else
286 if (mfp->freepage->size >= page->size) {
287 mem_page_delete(mfp, &page);
288 } else {
289 mem_page_delete(mfp, &mfp->freepage);
290 mem_page_reset(page);
291 mfp->freepage = page;
292 }
293 } else {
294 mem_page_reset(page);
295 mfp->freepage = page;
296 }
297 }
298
299#ifdef MEM_BENCH
300 proctimer_stop(&ptimer);
301 proctimerstat_addsample(&mfp->mem_bench.free.timer_stat, &ptimer);
302
303 mfp->mem_bench.free.nb_calls++;
304 mfp->mem_bench.free.nb_slow_path++;
305 mfp->mem_bench.current_used = mfp->occupied;
306 mem_bench_update(&mfp->mem_bench);
307 mem_bench_print_csv(&mfp->mem_bench);
308#endif
309}

311static void *mfp_realloc(mem_pool_t *_mfp, void *mem, size_t oldsize,
312 size_t size, size_t alignment, mem_flags_t flags)
313{
314 mem_fifo_pool_t *mfp = container_of(_mfp, mem_fifo_pool_t, funcs);
315 mem_block_t *blk;
316 mem_page_t *page;
317 bool guessed_size = false;
318 size_t alloced_size;
319 size_t req_size = size;
320
321#ifdef MEM_BENCH
322 proctimer_t ptimer;
323 proctimer_start(&ptimer);
324#endif
325
326 if (alignment > 8) {
327 e_panic("mem_fifo_pool does not support alignments greater than 8");
328 }
329
330 if (unlikely(!mfp->alive)) {
331 e_panic("trying to reallocate from a dead pool");
332 }
333
334 if (unlikely(size == 0)) {
335 mfp_free(_mfp, mem);
336 return MEM_EMPTY_ALLOC;
337 }
338
339 if (!mem || unlikely(mem == MEM_EMPTY_ALLOC)) {
340 return mfp_alloc(_mfp, size, alignment, flags);
341 }
342
343 blk = container_of(mem, mem_block_t, area);
344 mem_tool_allow_memory(blk, sizeof(*blk), true);
345 page = pageof(blk);
346
347 alloced_size = blk->blk_size - sizeof(*blk);
348 if ((flags & MEM_RAW) && oldsize == MEM_UNKNOWN) {
349 oldsize = alloced_size;
350 guessed_size = true;
351 }
352 assert (oldsize <= alloced_size);
353 if (req_size <= alloced_size) {
354 mem_tool_freelike(mem, oldsize, 0);
355 mem_tool_malloclike(mem, req_size, 0, false);
356 mem_tool_allow_memory(mem, MIN(req_size, oldsize), true);
357 if (!(flags & MEM_RAW) && oldsize < req_size)
358 memset(blk->area + oldsize, 0, req_size - oldsize);
359 } else
360 /* optimization if it's the last block allocated */
361 if (mem == page->last
362 && req_size - alloced_size <= mem_page_size_left(page))
363 {
364 size_t diff;
365
366 size = ROUND_UP((size_t)req_size + sizeof(*blk), 8);
367 diff = size - blk->blk_size;
368 blk->blk_size = size;
369
370 mfp->occupied += diff;
371 page->used_size += diff;
372 mem_tool_freelike(mem, oldsize, 0);
373 mem_tool_malloclike(mem, req_size, 0, false);
374 mem_tool_allow_memory(mem, MIN(req_size, oldsize), true);
375 } else {
376 void *old = mem;
377
378 mem = mfp_alloc(_mfp, size, alignment, flags);
379
380 if (guessed_size) {
381 /* XXX we guessed the size from the size of the block, as a
382 * consequence, we don't know the exact amount of used memory, and
383 * only have an upper bound, this means that some trailing bytes
384 * may be part of oldsize but not actually part of the allocation,
385 * so we must ensure we are allowed to copy those bytes.
386 */
387 mem_tool_allow_memory(old, oldsize, true);
388 }
389 memcpy(mem, old, oldsize);
390 mfp_free(_mfp, old);
391 return mem;
392 }
393
394#ifdef MEM_BENCH
395 proctimer_stop(&ptimer);
396 proctimerstat_addsample(&mfp->mem_bench.realloc.timer_stat, &ptimer);
397
398 mfp->mem_bench.realloc.nb_calls++;
399 mfp->mem_bench.current_used = mfp->occupied;
400 mem_bench_update(&mfp->mem_bench);
401#endif
402
403 mem_tool_disallow_memory(blk, sizeof(*blk));
404 return mem;
405}

415mem_pool_t *mem_fifo_pool_new(const char *name, int page_size_hint)
416{
417 mem_fifo_pool_t *mfp = p_new(mem_fifo_pool_t, 1);
418
419 mfp->name = p_strdup(name);
420
421 /* bypass mem_pool if demanded */
422 if (!mem_pool_is_enabled()) {
423 mfp->funcs = mem_pool_libc;
424 return &mfp->funcs;
425 }
426
427 STATIC_ASSERT((offsetof(mem_page_t, area) % 8) == 0);
428 mfp->funcs = mem_fifo_pool_funcs;
429 mfp->page_size = MAX(16 * PAGE_SIZE,
430 ROUND_UP(page_size_hint, PAGE_SIZE));
431 mfp->alive = true;
432 mfp->current = mem_page_new(mfp, 0);
433
434#ifdef MEM_BENCH
435 mem_bench_init(&mfp->mem_bench, LSTR("fifo"), WRITE_PERIOD);
436#endif
437
438 spin_lock(&_G.all_pools_lock);
439 dlist_add_tail(&_G.all_pools, &mfp->pool_list);
440 spin_unlock(&_G.all_pools_lock);
441
442 return &mfp->funcs;
443}

445void mem_fifo_pool_delete(mem_pool_t **poolp)
446{
447 mem_fifo_pool_t *mfp;
448
449 /* bypass mem_pool if demanded */
450 if (!mem_pool_is_enabled()) {
451 p_delete(poolp);
452 return;
453 }
454
455 if (!*poolp) {
456 return;
457 }
458
459 mfp = container_of(*poolp, mem_fifo_pool_t, funcs);
460
461 spin_lock(&_G.all_pools_lock);
462 dlist_remove(&mfp->pool_list);
463 spin_unlock(&_G.all_pools_lock);
464
465#ifdef MEM_BENCH
466 mem_bench_wipe(&mfp->mem_bench);
467#endif
468
469 p_delete(&mfp->name);
470 mfp->alive = false;
471 mem_page_delete(mfp, &mfp->freepage);
472 if (mfp->current && mfp->current->used_blocks == 0) {
473 mem_page_delete(mfp, &mfp->current);
474 } else {
475 mfp->current = NULL;
476 }
477
478 if (mfp->nb_pages) {
479 e_trace(0, "keep fifo-pool alive: %d pages in use (mem: %lubytes)",
480 mfp->nb_pages, (unsigned long) mfp->occupied);
481 mfp->owner = poolp;
482 return;
483 }
484 p_delete(poolp);
485}

487void mem_fifo_pool_stats(mem_pool_t *mp, ssize_t *allocated, ssize_t *used)
488{
489 mem_fifo_pool_t *mfp = (mem_fifo_pool_t *)(mp);
490
491 /* bypass mem_pool if demanded */
492 if (!mem_pool_is_enabled()) {
493 return;
494 }
495
496 /* we don't want to account the 'spare' page as allocated, it's an
497 optimization that should not leak. */
498 *allocated = mfp->map_size - (mfp->freepage ? mfp->freepage->size : 0);
499 *used = mfp->occupied;
500}

534static void core_mem_fifo_print_state(void)
535{
536 t_scope;
537 qv_t(table_hdr) hdr;
538 qv_t(table_data) rows;
539 table_hdr_t hdr_data[] = { {
540 .title = LSTR_IMMED("FIFO POOL NAME"),
541 }, {
542 .title = LSTR_IMMED("POINTER"),
543 }, {
544 .title = LSTR_IMMED("SIZE"),
545 }, {
546 .title = LSTR_IMMED("OCCUPIED"),
547 }, {
548 .title = LSTR_IMMED("PAGE SIZE"),
549 }, {
550 .title = LSTR_IMMED("NB PAGES"),
551 }
552 };
553 uint32_t hdr_size = countof(hdr_data);
554 size_t total_size = 0;
555 size_t total_occupied = 0;
556 uint32_t total_nb_pages = 0;
557 int nb_fifo_pool = 0;
558
559 qv_init_static(&hdr, hdr_data, hdr_size);
560 t_qv_init(&rows, 200);
561
562#define ADD_NUMBER_FIELD(_what) \
563 do { \
564 t_SB(_buf, 16); \
565 \
566 sb_add_int_fmt(&_buf, _what, ','); \
567 qv_append(tab, LSTR_SB_V(&_buf)); \
568 } while (0)
569
570 spin_lock(&_G.all_pools_lock);
571
572 dlist_for_each_entry(mem_fifo_pool_t, fp, &_G.all_pools, pool_list) {
573 qv_t(lstr) *tab = qv_growlen(&rows, 1);
574
575 t_qv_init(tab, hdr_size);
576 qv_append(tab, t_lstr_fmt("%s", fp->name));
577 qv_append(tab, t_lstr_fmt("%p", fp));
578
579 ADD_NUMBER_FIELD(fp->map_size);
580 ADD_NUMBER_FIELD(fp->occupied);
581 ADD_NUMBER_FIELD(fp->page_size);
582 ADD_NUMBER_FIELD(fp->nb_pages);
583
584 nb_fifo_pool++;
585 total_size += fp->map_size;
586 total_occupied += fp->occupied;
587 total_nb_pages += fp->nb_pages;
588 }
589
590 spin_unlock(&_G.all_pools_lock);
591
592 if (nb_fifo_pool) {
593 SB_1k(buf);
594 qv_t(lstr) *tab = qv_growlen(&rows, 1);
595
596 t_qv_init(tab, hdr_size);
597 qv_append(tab, LSTR("TOTAL"));
598 qv_append(tab, LSTR("-"));
599
600 ADD_NUMBER_FIELD(total_size);
601 ADD_NUMBER_FIELD(total_occupied);
602 qv_append(tab, LSTR("-"));
603 ADD_NUMBER_FIELD(total_nb_pages);
604
605 sb_add_table(&buf, &hdr, &rows);
606 sb_shrink(&buf, 1);
607 logger_notice(&_G.logger, "fifo pools summary:\n%*pM",
608 SB_FMT_ARG(&buf));
609 }
610#undef ADD_NUMBER_FIELD
611}
