31void mp_lstr_copy_(mem_pool_t *mp, lstr_t *dst, const void *s, int len)
32{
33 mp = mp ?: &mem_pool_libc;
34 if (dst->mem_pool == (mp->mem_pool & MEM_POOL_MASK)) {
35 mp_delete(mp, &dst->v);
36 } else
37 if (dst->mem_pool == MEM_MMAP) {
38 (lstr_munmap)(dst);
39 } else {
40 ifree(dst->v, dst->mem_pool);
41 }
42 if (s == NULL) {
43 *dst = lstr_init_(NULL, 0, MEM_STATIC);
44 } else {
45 *dst = lstr_init_(s, len, mp->mem_pool & MEM_POOL_MASK);
46 }

102lstr_t mp_lstr_dup_ascii_reversed(mem_pool_t *mp, const lstr_t v)
103{
104 char *str;
105
106 if (!v.s) {
107 return v;
108 }
109
110 str = mp_new_raw(mp, char, v.len + 1);
111
112 for (int i = 0; i < v.len; i++) {
113 str[i] = v.s[v.len - i - 1];
114 }
115 str[v.len] = '\0';
116
117 return mp_lstr_init(mp, str, v.len);

120lstr_t mp_lstr_dup_utf8_reversed(mem_pool_t *mp, const lstr_t v)
121{
122 int prev_off = 0;
123 char *str;
124
125 if (!v.s) {
126 return v;
127 }
128
129 str = mp_new_raw(mp, char, v.len + 1);
130 while (prev_off < v.len) {
131 int off = prev_off;
132 int c = utf8_ngetc_at(v.s, v.len, &off);
133
134 if (unlikely(c < 0)) {
135 return LSTR_NULL_V;
136 }
137 memcpy(str + v.len - off, v.s + prev_off, off - prev_off);
138 prev_off = off;
139 }
140 return mp_lstr_init(mp, str, v.len);
141}

203int lstr_init_from_fd(lstr_t *dst, int fd, int prot, int flags)
204{
205 struct stat st;
206
207 if (unlikely(fstat(fd, &st)) < 0) {
208 return -2;
209 }
210
211 if (st.st_size <= 0) {
212 SB_8k(sb);
213
214 if (sb_read_fd(&sb, fd) < 0) {
215 return -3;
216 }
217
218 *dst = LSTR_EMPTY_V;
219 if (sb.len == 0) {
220 return 0;
221 }
222 lstr_transfer_sb(dst, &sb, false);
223 return 0;
224 }
225
226 if (st.st_size == 0) {
227 *dst = LSTR_EMPTY_V;
228 return 0;
229 }
230
231 if (st.st_size > INT_MAX) {
232 errno = ERANGE;
233 return -3;
234 }
235
236 *dst = lstr_init_(mmap(NULL, st.st_size, prot, flags, fd, 0),
237 st.st_size, MEM_MMAP);
238
239 if (dst->v == MAP_FAILED) {
240 *dst = LSTR_NULL_V;
241 return -3;
242 }
243 return 0;
244}

246int lstr_init_from_file(lstr_t *dst, const char *path, int prot, int flags)
247{
248 int fd_flags = 0;
249 int fd = -1;
250 int ret = 0;
251
252 if (flags & MAP_ANONYMOUS) {
253 assert (false);
254 errno = EINVAL;
255 return -1;
256 }
257 if (prot & PROT_READ) {
258 if (prot & PROT_WRITE) {
259 fd_flags = O_RDWR;
260 } else {
261 fd_flags = O_RDONLY;
262 }
263 } else
264 if (prot & PROT_WRITE) {
265 fd_flags = O_WRONLY;
266 } else {
267 assert (false);
268 *dst = LSTR_NULL_V;
269 errno = EINVAL;
270 return -1;
271 }
272
273 fd = RETHROW(open(path, fd_flags));
274 ret = lstr_init_from_fd(dst, fd, prot, flags);
275 PROTECT_ERRNO(p_close(&fd));
276 return ret;
277}

282void lstr_transfer_sb(lstr_t *dst, sb_t *sb, bool keep_pool)
283{
284 if (keep_pool) {
285 mem_pool_t *mp = mp_ipool(sb->mp);
286
287 if ((mp_ipool(mp)->mem_pool & MEM_BY_FRAME)) {
288 if (sb->skip) {
289 memmove(sb->data - sb->skip, sb->data, sb->len + 1);
290 }
291 }
292 mp_lstr_copy_(mp, dst, sb->data, sb->len);
293 sb_init(sb);
294 } else {
295 lstr_wipe(dst);
296 dst->v = sb_detach(sb, &dst->len);
297 dst->mem_pool = MEM_LIBC;
298 }
299}

305int lstr_ascii_icmp(const lstr_t s1, const lstr_t s2)
306{
307 int min = MIN(s1.len, s2.len);
308
309 for (int i = 0; i < min; i++) {
310 int a = tolower((unsigned char)s1.s[i]);
311 int b = tolower((unsigned char)s2.s[i]);
312
313 if (a != b) {
314 return CMP(a, b);
315 }
316 }
317
318 return CMP(s1.len, s2.len);
319}

345int lstr_dlevenshtein(const lstr_t cs1, const lstr_t cs2, int max_dist)
346{
347 t_scope;
348 lstr_t s1 = cs1;
349 lstr_t s2 = cs2;
350 int *cur, *prev, *prev2;
351
352 if (s2.len > s1.len) {
353 SWAP(lstr_t, s1, s2);
354 }
355 if (max_dist < 0) {
356 max_dist = INT32_MAX;
357 } else {
358 THROW_ERR_IF(s1.len - s2.len > max_dist);
359 }
360 if (unlikely(!s2.len)) {
361 return (s1.len <= max_dist) ? s1.len : -1;
362 }
363
364 cur = t_new_raw(int, 3 * (s2.len + 1));
365 prev = cur + 1 * (s2.len + 1);
366 prev2 = cur + 2 * (s2.len + 1);
367
368 for (int j = 0; j <= s2.len; j++) {
369 cur[j] = j;
370 }
371
372 for (int i = 0; i < s1.len; i++) {
373 int min_dist;
374 int *tmp = prev2;
375
376 prev2 = prev;
377 prev = cur;
378 cur = tmp;
379
380 cur[0] = min_dist = i + 1;
381
382 for (int j = 0; j < s2.len; j++) {
383 int cost = (s1.s[i] == s2.s[j]) ? 0 : 1;
384 int deletion_cost = prev[j + 1] + 1;
385 int insertion_cost = cur[j ] + 1;
386 int substitution_cost = prev[j ] + cost;
387
388 cur[j + 1] = MIN3(deletion_cost, insertion_cost,
389 substitution_cost);
390
391 if (i > 0 && j > 0
392 && (s1.s[i ] == s2.s[j - 1])
393 && (s1.s[i - 1] == s2.s[j ]))
394 {
395 int transposition_cost = prev2[j - 1] + cost;
396
397 cur[j + 1] = MIN(cur[j + 1], transposition_cost);
398 }
399
400 min_dist = MIN(min_dist, cur[j + 1]);
401 }
402
403 THROW_ERR_IF(min_dist > max_dist);
404 }
405
406 return cur[s2.len];
407}

635/* {{{ SQL LIKE pattern matching */

637/* XXX: This is copied from QDB, so that other daemons can use the
638 * functionality without depending on QDB.
639 *
640 * XXX The behavior *should not be modified*.
641 */
642
643/* This function need only be called if c is in [0xC2...0xF4].
644 * It completes the decoding of the current UTF-8 encoded code point
645 * from a pstream.
646 * - embedded '\0' are handled as normal characters
647 * - on correctly encoded UTF-8 stream, it returns the code point and
648 * moves the pstream to the next position
649 * - on invalid UTF-8 sequences, it leaves the pstream unchanged and
650 * returns the initial byte value.
651 */
652static inline int ps_utf8_complete(int c, pstream_t *ps)
653{
654 int c1, c2, c3;
655
656 switch (c) {
657 /* 00...7F: US-ASCII */
658 /* 80...BF: Non UTF-8 leading byte */
659 /* C0...C1: Non canonical 2 byte UTF-8 encoding */
660 case 0xC2 ... 0xDF:
661 /* 2 byte UTF-8 sequence */
662 if (ps_has(ps, 1)
663 && (unsigned)(c1 = ps->b[0] - 0x80) < 0x40)
664 {
665 c = ((c & 0x3F) << 6) + c1;
666 ps->b += 1;
667 }
668 break;
669
670 case 0xE0 ... 0xEF:
671 /* 3 byte UTF-8 sequence */
672 if (ps_has(ps, 2)
673 && (unsigned)(c1 = ps->b[0] - 0x80) < 0x40
674 && (unsigned)(c2 = ps->b[1] - 0x80) < 0x40)
675 {
676 c = ((c & 0x3F) << 12) + (c1 << 6) + c2;
677 ps->b += 2;
678 }
679 break;
680
681 case 0xF0 ... 0xF4:
682 /* 3 byte UTF-8 sequence */
683 if (ps_has(ps, 3)
684 && (unsigned)(c1 = ps->b[0] - 0x80) < 0x40
685 && (unsigned)(c2 = ps->b[1] - 0x80) < 0x40
686 && (unsigned)(c3 = ps->b[2] - 0x80) < 0x40)
687 {
688 c = ((c & 0x3F) << 18) + (c1 << 12) + (c2 << 6) + c3;
689 ps->b += 3;
690 }
691 break;
692 /* F5..F7: Start of a 4-byte sequence, Restricted by RFC 3629 */
693 /* F8..FB: Start of a 5-byte sequence, Restricted by RFC 3629 */
694 /* FC..FD: Start of a 6-byte sequence, Restricted by RFC 3629 */
695 /* FE..FF: Invalid, non UTF-8 */
696 }
697
698 return c;
699}

701#define COLLATE_MASK 0xffff
702#define COLLATE_SHIFT(c) ((unsigned)(c) >> 16)
703
704/* XXX: extracted from QDB (was named qdb_strlike), Do not change behavior! */
705static bool ps_is_like(pstream_t ps, pstream_t pattern)
706{
707 pstream_t pattern0;
708 int c1, c10, c11, c2, c20, c21;
709
710 for (;;) {
711 if (ps_done(&pattern)) {
712 return ps_done(&ps);
713 }
714 c1 = __ps_getc(&pattern);
715 if (c1 == '_') {
716 if (ps_done(&ps)) {
717 return false;
718 }
719 c2 = __ps_getc(&ps);
720 if (c2 >= 0xC2) {
721 ps_utf8_complete(c2, &ps);
722 }
723 continue;
724 }
725 if (c1 == '%') {
726 for (;;) {
727 if (ps_done(&pattern)) {
728 return true;
729 }
730
731 pattern0 = pattern;
732
733 /* Check for non canonical pattern */
734 c1 = __ps_getc(&pattern);
735 if (c1 == '_') {
736 if (ps_done(&ps)) {
737 return false;
738 }
739 ps_utf8_complete(__ps_getc(&ps), &ps);
740 continue;
741 }
742 if (c1 == '%') {
743 continue;
744 }
745 if (c1 == '\\' && !ps_done(&pattern)) {
746 c1 = __ps_getc(&pattern);
747 }
748
749 c1 = ps_utf8_complete(c1, &pattern);
750 c10 = c1;
751 c11 = 0;
752 if (c1 < countof(__str_unicode_general_ci)) {
753 int cc1 = __str_unicode_general_ci[c1];
754 c10 = cc1 & COLLATE_MASK;
755 c11 = COLLATE_SHIFT(cc1);
756 }
757
758 /* Simplistic recursive matcher */
759 for (;;) {
760 pstream_t ps0 = ps;
761
762 if (ps_done(&ps)) {
763 return false;
764 }
765 c2 = __ps_getc(&ps);
766 if (c2 >= 0xC2) {
767 c2 = ps_utf8_complete(c2, &ps);
768 }
769 c20 = c2;
770 c21 = 0;
771 if (c2 < countof(__str_unicode_general_ci)) {
772 int cc2 = __str_unicode_general_ci[c2];
773 c20 = cc2 & COLLATE_MASK;
774 c21 = COLLATE_SHIFT(cc2);
775 }
776 if (c10 != c20) {
777 continue;
778 }
779 /* Handle dual collation */
780 if (c11 != c21) {
781 /* identical leading collation chars, but
782 * different dual collation: recurse
783 * without skipping */
784 if (ps_is_like(ps0, pattern0)) {
785 return true;
786 }
787 continue;
788 }
789 /* both large, single or dual and identical */
790 if (ps_is_like(ps, pattern)) {
791 return true;
792 }
793 }
794 }
795 }
796 if (c1 == '\\' && !ps_done(&pattern)) {
797 c1 = __ps_getc(&pattern);
798 }
799 c1 = ps_utf8_complete(c1, &pattern);
800 if (ps_done(&ps)) {
801 return false;
802 }
803 c2 = __ps_getc(&ps);
804 if (c2 >= 0xC2) {
805 c2 = ps_utf8_complete(c2, &ps);
806 }
807 if (c1 == c2) {
808 continue;
809 }
810 if ((c1 | c2) >= countof(__str_unicode_general_ci)) {
811 /* large characters require exact match */
812 break;
813 }
814 c1 = __str_unicode_general_ci[c1];
815 c2 = __str_unicode_general_ci[c2];
816
817 again:
818 if ((c1 & COLLATE_MASK) != (c2 & COLLATE_MASK)) {
819 break;
820 }
821 /* Handle dual collation */
822 c1 = COLLATE_SHIFT(c1);
823 c2 = COLLATE_SHIFT(c2);
824 if ((c1 | c2) == 0) {
825 /* both collation chars are single and identical */
826 continue;
827 }
828 if (c1 == 0) {
829 /* c2 is non zero */
830 if (ps_done(&pattern)) {
831 break;
832 }
833 c1 = __ps_getc(&pattern);
834 if (c1 == '_' || c1 == '%' || c1 == '\\') {
835 /* wildcards must fall on character boundaries */
836 break;
837 }
838 c1 = ps_utf8_complete(c1, &pattern);
839 if (c1 >= countof(__str_unicode_general_ci)) {
840 /* large character cannot match c2 */
841 break;
842 }
843 c1 = __str_unicode_general_ci[c1];
844 goto again;
845 } else
846 if (c2 == 0) {
847 /* c1 is non zero */
848 if (ps_done(&ps)) {
849 break;
850 }
851 c2 = ps_utf8_complete(__ps_getc(&ps), &ps);
852 if (c2 >= countof(__str_unicode_general_ci)) {
853 /* large character cannot match c1 */
854 break;
855 }
856 c2 = __str_unicode_general_ci[c2];
857 goto again;
858 } else
859 if (c1 == c2) {
860 /* both collation chars are dual and identical */
861 continue;
862 } else {
863 /* both are dual and different */
864 break;
865 }
866 }
867 return false;
868}
