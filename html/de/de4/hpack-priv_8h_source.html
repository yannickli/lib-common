33/* Huffman decoder's state-transition table entries */
34/* Decoder works by consuming each nibble of input transiting over the Huffman
35 * tree. We have a state for every non-leaf node while leaf nodes correspond
36 * to direct transitions through the root node each emitting the corresponding
37 * decoded byte.
38 */
39typedef struct hpack_huffdec_trans_t {
40 /* new state after consuming the 4-bit chunk */
41 uint8_t state;
42 /* emitted symbol (byte) if emitter */
43 uint8_t sym;
44 /* does this transition emit a decoded symbol? */
45 uint16_t emitter : 1;
46 /* is this a final transition? */
47 uint16_t final : 1;
48 /* is this an error transition? */
49 uint16_t error : 1;
50} hpack_huffdec_trans_t;
51
52/* Huffman state-transition table based on 4-bit chunks (i.e., nibbles) */
53extern const hpack_huffdec_trans_t hpack_huffdec_trans_tab_g[256][16];
54
55/** Return the length of the huffman-coded version of \p str */
56static inline size_t hpack_get_huffman_len(lstr_t str)
57{
58 size_t bits = 0;
59
60 for (int i = 0; i != str.len; i++) {
61 uint8_t ch = str.s[i];
62
63 bits += hpack_huffcode_tab_g[ch].bitlen;
64 }
65 return DIV_ROUND_UP(bits, 8U);
66}
67
68/** Return a worst case estimate for the length of the huffman-coded version
69 * \p str
70 */
71static inline size_t hpack_get_huffman_len_estimate(lstr_t str)
72{
73 return 4 * str.len;
74}
