86/** \brief get the description of the currently unpacked RPC.
87 *
88 * This low-level function allows to retrieve the description of the currently
89 * unpacked RPC. You are allowed to call it only inside of the callback of
90 * a RPC implementation.
91 */
92static inline const iop_rpc_t * nonnull
93__ic_get_current_rpc_desc(const ichannel_t * nullable ic, uint64_t slot)
94{
95#ifndef __cplusplus
96 if (ic_slot_is_http(slot)) {
97 ichttp_query_t *iq = ichttp_slot_to_query(slot);
98
99 return iq->cbe->fun;
100 }
101#endif
102 return ic->desc;
103}
104
105/** \brief get the “command” of the currently unpacked RPC.
106 *
107 * This low-level function allows to retrieve the “command” value of the
108 * currently unpacked RPC. You are allowed to call it only inside of the
109 * callback of a RPC implementation.
110 */
111static inline int
112__ic_get_current_rpc_cmd(const ichannel_t * nullable ic, uint64_t slot)
113{
114#ifndef __cplusplus
115 if (ic_slot_is_http(slot)) {
116 ichttp_query_t *iq = ichttp_slot_to_query(slot);
117
118 return iq->cbe->cmd;
119 }
120#endif
121 return ic->cmd;
122}
