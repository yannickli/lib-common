135/* Arithmetics on timeval assume both members of timeval are signed.
136 * We keep timeval structures in normalized form:
137 * - tv_usec is always in the range [0 .. 999999]
138 * - tv_sec is negative iff timeval is negative
139 * This choice is consistent with sys/time.h macros,
140 * but it makes it a little bit harder to print timevals.
141 */
142
143/* Return reference to static buf for immediate printing */
144const char *timeval_format(struct timeval tv, bool as_duration)
145{
146 // FIXME: assuming 32 bit timeval resolution, bug in 2034
147 static char buf[32];
148 int pos, d, h, m, s, sec, usec;
149
150 usec = tv.tv_usec;
151
152 if (as_duration) {
153 pos = 0;
154 sec = tv.tv_sec;
155 if (sec < 0) {
156 buf[pos++] = '-';
157 sec = -sec;
158 if (usec != 0) {
159 usec = 1000 * 1000 - usec;
160 sec -= 1;
161 }
162 }
163 s = sec % 60;
164 sec /= 60;
165 m = sec % 60;
166 sec /= 60;
167 h = sec % 24;
168 sec /= 24;
169 d = sec;
170 snprintf(buf + pos, sizeof(buf) - pos,
171 "%d %2d:%02d:%02d.%06d", d, h, m, s, usec);
172 } else {
173 struct tm t;
174
175 localtime_r(&tv.tv_sec, &t);
176 snprintf(buf, sizeof(buf),
177 "%02d/%02d/%04d %2d:%02d:%02d.%06d",
178 t.tm_mday, t.tm_mon + 1, t.tm_year + 1900,
179 t.tm_hour, t.tm_min, t.tm_sec, usec);
180 }
181 return buf;
182}

184struct timeval timeval_mul(struct timeval tv, int k)
185{
186 struct timeval res;
187 int64_t usecs;
188
189 /* Casting to int64_t is necessary because of potential overflow */
190 /* Grouping 1000 * 1000 may yield better code */
191 usecs = (int64_t)tv.tv_sec * (1000 * 1000) + tv.tv_usec;
192 usecs *= k;
193
194 res.tv_sec = usecs / (1000 * 1000);
195 res.tv_usec = usecs - (int64_t)res.tv_sec * (1000 * 1000);
196 if (res.tv_usec < 0) {
197 res.tv_usec += 1000 * 1000;
198 res.tv_sec -= 1;
199 }
200
201 return res;
202}
203
204struct timeval timeval_div(struct timeval tv, int k)
205{
206 struct timeval res;
207 int64_t usecs;
208
209 usecs = (int64_t)tv.tv_sec * (1000 * 1000) + tv.tv_usec;
210 usecs /= k;
211
212 res.tv_sec = usecs / (1000 * 1000);
213 res.tv_usec = usecs - (int64_t)res.tv_sec * (1000 * 1000);
214 if (res.tv_usec < 0) {
215 res.tv_usec += 1000 * 1000;
216 res.tv_sec -= 1;
217 }
218
219 return res;
220}

222/* Test if timer has expired: if now is provided, test if date is past
223 * now, else test if date is past current time of day. compute
224 * available time left and return it if left is not NULL.
225 */
226bool is_expired(const struct timeval *date,
227 const struct timeval *now,
228 struct timeval *left)
229{
230 struct timeval local_now;
231
232 if (!now) {
233 now = &local_now;
234 gettimeofday(&local_now, NULL);
235 }
236 if (left) {
237 *left = timeval_sub(*date, *now);
238 return timeval_is_le0(*left);
239 }
240 return timeval_is_le0(timeval_sub(*date, *now));
241}

320time_t localtime_addday(time_t date, int n)
321{
322 struct tm t;
323
324 RETHROW(localtime_(date, &t));
325
326 /* To avoid complex calculations, we rely on mktime() to normalize
327 * the tm structure as specified in the man page:
328 *
329 * "The mktime() function converts a broken-down time structure,
330 * expressed as local time, to calendar time representation. The
331 * function ignores the specified contents of the structure members
332 * tm_wday and tm_yday and recomputes them from the other
333 * information in the broken-down time structure. If structure
334 * members are outside their legal interval, they will be
335 * normalized (so that, e.g., 40 October is changed into 9 Novem-
336 * ber). Calling mktime() also sets the external variable tzname
337 * with information about the current time zone. If the specified
338 * broken-down time cannot be represented as calendar time (seconds
339 * since the epoch), mktime() returns a value of (time_t)(-1) and
340 * does not alter the tm_wday and tm_yday members of the
341 * broken-down time structure."
342 */
343
344 t.tm_sec = 0;
345 t.tm_min = 0;
346 t.tm_hour = 0;
347 t.tm_mday += n;
348 t.tm_isdst = -1;
349
350 return mktime(&t);
351}

467bool is_mday_valid(int d, int m, int y)
468{
469 assert (0 <= m && m <= 11);
470
471 return d > 0
472 && (d <= __valid_mdays[m]
473 || (m == 1 && d == 29 && year_is_leap_year(y)));
474}

476/* We currently support only this format: DD-MMM-[YY]YY */
477/* WARNING: Only date related tm structure fields are changed, others
478 * must be initialized before this call.
479 */
480/* OG: why not use strptime ? */
481int strtotm(const char *date, struct tm *t)
482{
483 int mday, mon, year;
484 const char *p;
485 char lower_mon[4];
486 size_t len = strlen(date);
487
488 if (len < strlen("DD-MMM-YY"))
489 return -1;
490
491 /* Read day */
492 mday = cstrtol(date, &p, 10);
493 if (mday <= 0)
494 return -1;
495
496 /* Read month */
497 if (!p || *p != '-' || p - date != 2)
498 return -1;
499 p++;
500
501 lower_mon[0] = tolower((unsigned char)p[0]);
502 lower_mon[1] = tolower((unsigned char)p[1]);
503 lower_mon[2] = tolower((unsigned char)p[2]);
504 lower_mon[3] = '\0';
505
506 for (mon = 0;; mon++) {
507 if (!__abbr_months[mon])
508 return -1;
509 if (!memcmp(lower_mon, __abbr_months[mon], 4))
510 break;
511 }
512
513 /* Read year */
514 p += 3;
515 if (*p++ != '-')
516 return -1;
517
518 year = RETHROW(atoi(p));
519 if (year < 70) {
520 year += 2000;
521 } else
522 if (year < 100) {
523 year += 1900;
524 }
525 if (year < 1970 || year > 2036)
526 return -1;
527
528 /* Check mday validity */
529 if (!is_mday_valid(mday, mon, year)) {
530 return -1;
531 }
532
533 t->tm_mday = mday;
534 t->tm_mon = mon;
535 t->tm_year = year - 1900;
536 /* OG: should also update t->tm_wday and t->tm_yday */
537
538 return 0;
539}

553struct tm *time_get_localtime(const time_t *p_ts, struct tm *p_tm,
554 const char *tz)
555{
556 const char *old_tz;
557 bool tz_changed = false;
558
559 if (tz) {
560 old_tz = getenv("TZ");
561
562 if (!old_tz || !strequal(old_tz, tz)) {
563 setenv("TZ", tz, true);
564 tzset();
565 tz_changed = true;
566 }
567 }
568
569 localtime_r(p_ts, p_tm);
570
571 if (tz_changed) {
572 if (old_tz) {
573 setenv("TZ", old_tz, true);
574 } else {
575 unsetenv("TZ");
576 }
577
578 tzset();
579 }
580
581 return p_tm;
582}

600int format_timestamp(const char *fmt, time_t ts, const char *locale,
601 char out[], int out_size)
602{
603 struct tm ts_tm;
604 const char *lc_time = NULL;
605 int len;
606 int ret;
607
608 if (strequal(fmt, "%s")) {
609 len = snprintf(out, out_size, "%ld", ts);
610 return len;
611 }
612 if (locale) {
613 lc_time = setlocale(LC_TIME, NULL);
614 RETHROW_PN(setlocale(LC_TIME, locale));
615 }
616
617 if (localtime_r(&ts, &ts_tm) == NULL) {
618 ret = -1;
619 goto end;
620 }
621 len = strftime(out, out_size, fmt, &ts_tm);
622 ret = len ?: -1;
623
624 end:
625 if (locale) {
626 setlocale(LC_TIME, lc_time);
627 }
628 return ret;
629}

637const char *proctimer_report(proctimer_t *tp, const char *fmt)
638{
639 static char buf[256];
640 int pos;
641 int elapsed;
642 const char *p;
643
644 if (!fmt) {
645 fmt = "real %rms, proc %pms, user %ums, sys %sms, %h cycles";
646 }
647
648 for (p = fmt, pos = 0; *p && pos < ssizeof(buf) - 1; p++) {
649 if (*p == '%') {
650 switch (*++p) {
651 case 'r': /* real */
652 elapsed = tp->elapsed_real;
653 goto format_elapsed;
654 case 'u': /* user */
655 elapsed = tp->elapsed_user;
656 goto format_elapsed;
657 case 's': /* sys */
658 elapsed = tp->elapsed_sys;
659 goto format_elapsed;
660 case 'p': /* process */
661 elapsed = tp->elapsed_proc;
662 goto format_elapsed;
663 format_elapsed:
664 snprintf(buf + pos, sizeof(buf) - pos, "%d.%03d",
665 elapsed / 1000, elapsed % 1000);
666 pos += strlen(buf + pos);
667 continue;
668
669 case 'h': /* hardware */
670 snprintf(buf + pos, sizeof(buf) - pos, "%lu",
671 tp->elapsed_hard);
672 pos += strlen(buf + pos);
673 continue;
674
675 case '%':
676 default:
677 break;
678 }
679 }
680 buf[pos++] = *p;
681 }
682 buf[pos] = '\0';
683 return buf;
684}

686const char *proctimerstat_report(proctimerstat_t *pts, const char *fmt)
687{
688 static char buf[1024];
689 int pos;
690 unsigned long min, max, tot, mean;
691 const char *p;
692 const char *unit;
693
694 if (!fmt) {
695 fmt = "%n samples\nreal: %r\nproc: %p\nuser: %u\nsys : %s\nproc cycles: %h";
696 }
697
698 for (p = fmt, pos = 0; *p && pos < ssizeof(buf) - 1; p++) {
699 unit = "ms";
700
701 if (*p == '%') {
702 switch (*++p) {
703 case 'n': /* nb samples */
704 snprintf(buf + pos, sizeof(buf) - pos, "%d", pts->nb);
705 pos += strlen(buf + pos);
706 continue;
707 case 'r': /* real */
708 min = pts->real_min;
709 max = pts->real_max;
710 tot = pts->real_tot;
711 goto format;
712 case 'u': /* user */
713 min = pts->user_min;
714 max = pts->user_max;
715 tot = pts->user_tot;
716 goto format;
717 case 's': /* sys */
718 min = pts->sys_min;
719 max = pts->sys_max;
720 tot = pts->sys_tot;
721 goto format;
722 case 'p': /* process */
723 min = pts->proc_min;
724 max = pts->proc_max;
725 tot = pts->proc_tot;
726 goto format;
727 case 'h': /* hardware */
728 unit = "cycles";
729 min = pts->hard_min;
730 max = pts->hard_max;
731 tot = pts->hard_tot;
732 goto format;
733 format:
734 mean = tot / MAX(1, pts->nb);
735 snprintf(buf + pos, sizeof(buf) - pos,
736 "min=%ld.%03ld%s "
737 "max=%ld.%03ld%s "
738 "mean=%ld.%03ld%s",
739 min / 1000, min % 1000, unit,
740 max / 1000, max % 1000, unit,
741 mean / 1000, mean % 1000, unit
742 );
743 pos += strlen(buf + pos);
744 continue;
745 case '%':
746 default:
747 break;
748 }
749 }
750 buf[pos++] = *p;
751 }
752 buf[pos] = '\0';
753 return buf;
754}

759time_split_t split_time_interval(uint64_t seconds)
760{
761 time_split_t res;
762
763 p_clear(&res, 1);
764
765 res.years = seconds / (3600 * 24 * 365);
766 seconds %= (3600 * 24 * 365);
767
768 res.weeks = seconds / (3600 * 24 * 7);
769 seconds %= (3600 * 24 * 7);
770
771 res.days = seconds / (3600 * 24);
772 seconds %= (3600 * 24);
773
774 res.hours = seconds / 3600;
775 seconds %= 3600;
776
777 res.minutes = seconds / 60;
778 seconds %= 60;
779
780 res.seconds = seconds;
781
782 return res;
783}

914void timing_scope_finish(timing_scope_ctx_t *ctx)
915{
916 struct timeval tv_end;
917 int level;
918
919 lp_gettv(&tv_end);
920
921 if (timeval_diffmsec(&tv_end, &ctx->tv_start) >= ctx->threshold_ms) {
922 level = ctx->log_level;
923 } else {
924 level = LOG_TRACE + 1;
925 }
926
927 if (logger_has_level(ctx->logger, level)) {
928 SB_1k(log_buf);
929 struct timeval tv_diff;
930
931 tv_diff = timeval_sub(tv_end, ctx->tv_start);
932
933 sb_setf(&log_buf, "%*pM done in %ld.%06ldsec",
934 LSTR_FMT_ARG(ctx->desc), tv_diff.tv_sec, tv_diff.tv_usec);
935 if (ctx->threshold_ms > 0) {
936 sb_addf(&log_buf, " (threshold = %ld.%06ldsec)",
937 ctx->threshold_ms / 1000,
938 (ctx->threshold_ms % 1000) * 1000);
939 }
940 __logger_log(ctx->logger, level, NULL, -1,
941 ctx->file, ctx->func, ctx->line,
942 "%*pM", SB_FMT_ARG(&log_buf));
943 }
944
945 lstr_wipe(&ctx->desc);
946}
