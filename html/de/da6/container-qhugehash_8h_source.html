24/*
25 * QHugeHashes: Real Time huge hash tables.
26 *
27 * This implements a wrapper around the qh/qm that let user define tables that
28 * exceed 1GB.
29 *
30 * This works by allocating several buckets, the number being defined as a
31 * parameter of the type, each bucket will receive the data that match the
32 * hashes modulo its position.
33 */
34

83#define __QHH_FOREACH_BUCKET(it, qhh, do_it, ...) do { \
84 for (int it = 0; it < countof(qhh->buckets); it++) { \
85 do_it(&qhh->buckets[it].qm.qh, ##__VA_ARGS__); \
86 } \
87 } while (0)
88
89#define qhhash_for_each_pos(pos, hh) \
90 for (uint64_t __b_##pos = 0; \
91 __b_##pos < countof((hh)->buckets); \
92 __b_##pos++) \
93 for (uint64_t __##pos##_priv \
94 = ((hh)->buckets[__b_##pos].qm.hdr.len \
95 ? qhash_scan(&(hh)->buckets[__b_##pos].qm.qh, 0) \
96 : UINT32_MAX) | (__b_##pos << 32), \
97 pos = __##pos##_priv; \
98 __QHH_POS(hh, pos) != UINT32_MAX; \
99 __##pos##_priv = qhash_scan(&(hh)->buckets[__b_##pos].qm.qh, \
100 __QHH_POS(hh, __##pos##_priv) + 1) | \
101 (__b_##pos << 32), pos = __##pos##_priv)
102
103#define __QHH_BUCKET_ID(qhh, pos) ((pos) >> 32)
104#define __QHH_BUCKET(qhh, pos) (&(qhh)->buckets[__QHH_BUCKET_ID(qhh, pos)])
105#define __QHH_POS(qhh, pos) ((pos) & 0xffffffff)

107#define __QHH_BASE(pfx, name, hpfx, bucket_count, ckey_t, pkey_t, hf) \
108 typedef struct pfx##_bucket_t { \
109 int pos; \
110 hpfx##_t qm; \
111 } pfx##_bucket_t; \
112 typedef struct pfx##_t { \
113 qhhash_t hdr; \
114 pfx##_bucket_t buckets[bucket_count]; \
115 } pfx##_t; \
116 \
117 __unused__ \
118 static inline pfx##_t *pfx##_init(pfx##_t *qhh, bool chashes, \
119 mem_pool_t *nullable mp) \
120 { \
121 p_clear(&qhh->hdr, 1); \
122 for (int it = 0; it < countof(qhh->buckets); it++) { \
123 qhh->buckets[it].pos = it; \
124 hpfx##_init(&qhh->buckets[it].qm, chashes, mp); \
125 } \
126 return qhh; \
127 } \
128 \
129 __unused__ \
130 static inline size_t pfx##_memory_footprint(const pfx##_t *qhh) \
131 { \
132 size_t size = 0; \
133 \
134 for (int it = 0; it < countof(qhh->buckets); it++) { \
135 size += qhash_memory_footprint(&qhh->buckets[it].qm.qh); \
136 } \
137 return size; \
138 } \
139 \
140 __unused__ \
141 static inline void pfx##_wipe(pfx##_t *qhh) \
142 { \
143 __QHH_FOREACH_BUCKET(i, qhh, qhash_wipe); \
144 } \
145 \
146 __unused__ \
147 static inline void pfx##_clear(pfx##_t *qhh) \
148 { \
149 __QHH_FOREACH_BUCKET(i, qhh, qhash_clear); \
150 qhh->hdr.len = 0; \
151 } \
152 \
153 __unused__ \
154 static inline void pfx##_set_minsize(pfx##_t *qhh, uint64_t minsize) \
155 { \
156 minsize = DIV_ROUND_UP(minsize, countof(qhh->buckets)); \
157 if (minsize > INT32_MAX) { \
158 e_fatal("requested minsize too large: %ju", minsize); \
159 } \
160 for (int it = 0; it < countof(qhh->buckets); it++) { \
161 qhash_set_minsize(&qhh->buckets[it].qm.qh, minsize); \
162 } \
163 } \
164 \
165 __unused__ \
166 static inline void pfx##_del_at(pfx##_t *qhh, uint64_t pos) \
167 { \
168 hpfx##_t *bucket = &__QHH_BUCKET(qhh, pos)->qm; \
169 uint32_t old_len = bucket->qh.hdr.len; \
170 \
171 qhash_del_at(&bucket->qh, __QHH_POS(qhh, pos)); \
172 qhh->hdr.len -= old_len - bucket->qh.hdr.len; \
173 } \
174 \
175 __unused__ \
176 static inline pkey_t pfx##_key_p(const pfx##_t *qhh, uint64_t pos) \
177 { \
178 pos &= ~(uint64_t)QHASH_COLLISION; \
179 return &__QHH_BUCKET(qhh, pos)->qm.keys[__QHH_POS(qhh, pos)]; \
180 } \
181 \
182 __unused__ \
183 static inline uint32_t *pfx##_hash_p(const pfx##_t *qhh, uint64_t pos) \
184 { \
185 pos &= ~(uint64_t)QHASH_COLLISION; \
186 return &__QHH_BUCKET(qhh, pos)->qm.hashes[__QHH_POS(qhh, pos)]; \
187 } \
188 \
189 __unused__ \
190 static inline uint64_t pfx##_len(const pfx##_t *qhh) \
191 { \
192 return qhh->hdr.len; \
193 } \
194 \
195 __unused__ \
196 static inline uint32_t pfx##_hash(const pfx##_t *qhh, ckey_t key) \
197 { \
198 return hf(&qhh->hdr, key); \
199 } \
200 \
201 __unused__ \
202 static inline uint32_t pfx##__hash(const qhash_t *h, ckey_t key) \
203 { \
204 const pfx##_bucket_t *bucket = container_of(h, const pfx##_bucket_t, \
205 qm.qh); \
206 \
207 bucket -= bucket->pos; \
208 return pfx##_hash(container_of(bucket, const pfx##_t, buckets[0]), \
209 key); \
210 } \

212#define __QHH_EQUAL(pfx, name, hpfx, ckey_t, ef) \
213 __unused__ \
214 static inline bool pfx##_equal(const pfx##_t *qhh, const qhash_t *qh, \
215 ckey_t a, ckey_t b) \
216 { \
217 return ef(&qhh->hdr, qh, a, b); \
218 } \
219 __unused__ \
220 static inline bool pfx##__equal(const qhash_t *h, ckey_t a, ckey_t b) \
221 { \
222 const pfx##_bucket_t *bucket = container_of(h, const pfx##_bucket_t, \
223 qm.qh); \
224 \
225 bucket -= bucket->pos; \
226 return pfx##_equal(container_of(bucket, const pfx##_t, buckets[0]), \
227 h, a, b); \
228 }

234 uint64_t bid = h % countof(qhh->buckets); \
235 uint64_t pos; \
236 \

270 __unused__ \
271 static inline uint64_t pfx##_put_h(pfx##_t *qhh, uint32_t h, \
272 key_t key, uint32_t fl) \
273 { \
274 uint64_t bid = h % countof(qhh->buckets); \
275 uint64_t pos; \
276 \
277 pos = hpfx##_reserve_int(&qhh->buckets[bid].qm, &h, key, fl); \
278 if (!(pos & QHASH_COLLISION)) { \
279 qhh->hdr.len++; \
280 } \
281 pos |= (bid << 32); \
282 return pos; \
283 } \

293 { \
294 uint64_t bid = h % countof(qhh->buckets); \
295 int ret = hpfx##_reserve_int(&qhh->buckets[bid].qm, &h, key, 0); \
296 \
297 if (!(ret & QHASH_COLLISION)) { \
298 qhh->hdr.len++; \
299 } \

360#define __QHM_ADD(pfx, name, hpfx, key_t, val_t) \
361 __unused__ \
362 static inline val_t *pfx##_value_p(const pfx##_t *qhh, uint64_t pos) \
363 { \
364 pos &= ~(uint64_t)QHASH_COLLISION; \
365 return &__QHH_BUCKET(qhh, pos)->qm.values[__QHH_POS(qhh, pos)]; \
366 } \
367 \
368 __unused__ \
369 static inline uint64_t pfx##_put_h(pfx##_t *qhh, uint32_t h, \
370 key_t key, val_t v, uint32_t fl) \
371 { \
372 uint64_t bid = h % countof(qhh->buckets); \
373 uint64_t pos; \
374 \
375 pos = hpfx##_reserve_int(&qhh->buckets[bid].qm, &h, key, fl); \
376 if ((fl & QHASH_OVERWRITE) || !(pos & QHASH_COLLISION)) { \
377 qhh->buckets[bid].qm.values[pos & ~QHASH_COLLISION] = v; \
378 } \
379 if (!(pos & QHASH_COLLISION)) { \
380 qhh->hdr.len++; \
381 } \
382 pos |= (bid << 32); \
383 return pos; \
384 } \
385 __unused__ \
386 static inline uint64_t pfx##_put(pfx##_t *qhh, key_t key, val_t v, \
387 uint32_t fl) \

390 } \
