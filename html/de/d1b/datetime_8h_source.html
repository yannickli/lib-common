50/** Count the number of leap years since 1900.
51 *
52 * When the provided year is a leap year, it is part of the returned count.
53 * For example, the number of leap years since 1900 for 2016 is 29.
54 */
55static inline int nb_leap_years_since_1900(int year)
56{
57 static const int nb_leap_years_1900 = 460;
58
59 return (year / 4) - (year / 100) + (year / 400) - nb_leap_years_1900;
60}
61
62/** Check if a year is a leap year.
63 */
64static inline bool year_is_leap_year(int year)
65{
66 return ((year % 4) == 0 && (year % 100) != 0)
67 || (year % 400) == 0;
68}

208/** Count the number of days since January 1st, 1900.
209 */
210static inline int tm_nb_days_since_1900(struct tm *t)
211{
212 int nb_days = 365 * t->tm_year + t->tm_yday;
213
214 /* do not consider current year for leap years count since the current
215 * leap year day, if any, is already added by tm_yday.
216 */
217 return nb_days + nb_leap_years_since_1900(t->tm_year + 1900 - 1);
218}
219
220/** Count the number of days between two dates.
221 */
222static inline int tm_diff_days(struct tm *from, struct tm *to)
223{
224 return tm_nb_days_since_1900(to) - tm_nb_days_since_1900(from);
225}
226
227/** Count the number of days between two timestamps, in local time. */
228int time_diff_days(time_t from, time_t to);
229
230/** Count the number of hours between two dates.
231 *
232 * Daylight saving time and leap seconds are not considered in the count.
233 */
234static inline int tm_diff_hours(struct tm* from, struct tm *to)
235{
236 unsigned long nb_days = tm_diff_days(from, to);
237
238 return nb_days * 24 + (to->tm_hour - from->tm_hour);
239}
240
241/** Count the number of minutes between two dates.
242 *
243 * Daylight saving time and leap seconds are not considered in the count.
244 */
245static inline int tm_diff_minutes(struct tm* from, struct tm *to)
246{
247 unsigned long nb_hours = tm_diff_hours(from, to);
248
249 return nb_hours * 60 + (to->tm_min - from->tm_min);
250}

323#define ISO8601_BASE_FMT "%04d-%02d-%02dT%02d:%02d:%02d"
324#define ISO8601_BASE_FMT_ARG(tm) \
325 (tm)->tm_year + 1900, \
326 (tm)->tm_mon + 1, \
327 (tm)->tm_mday, \
328 (tm)->tm_hour, \
329 (tm)->tm_min, \
330 (tm)->tm_sec
331
332#define ISO8601_BASE_MSEC_FMT ISO8601_BASE_FMT ".%03d"
333#define ISO8601_BASE_MSEC_FMT_ARG(tm, msec) ISO8601_BASE_FMT_ARG(tm), (msec)
334
335#define ISO8601_GMT_FMT ISO8601_BASE_FMT "Z"
336#define ISO8601_GMT_FMT_ARG(tm) ISO8601_BASE_FMT_ARG(tm)
337
338#define ISO8601_GMT_MSEC_FMT \
339 ISO8601_BASE_MSEC_FMT "Z"
340#define ISO8601_GMT_MSEC_FMT_ARG(tm, msec) \
341 ISO8601_BASE_MSEC_FMT_ARG(tm, msec)
342
343#define ISO8601_TZ_FMT \
344 ISO8601_BASE_FMT "%+03d:%02d"
345#define ISO8601_TZ_FMT_ARG(tm, delta_h, delta_m) \
346 ISO8601_BASE_FMT_ARG(tm), (delta_h), (delta_m)
347
348#define ISO8601_TZ_MSEC_FMT \
349 ISO8601_BASE_MSEC_FMT "%+03d:%02d"
350#define ISO8601_TZ_MSEC_FMT_ARG(tm, msec, delta_h, delta_m) \
351 ISO8601_BASE_MSEC_FMT_ARG(tm, msec), (delta_h), (delta_m)

353static inline void
354time_get_gmt_delta(const struct tm *tm, int *delta_h, int *delta_m)
355{
356 *delta_h = tm->tm_gmtoff / 3600;
357 *delta_m = labs(tm->tm_gmtoff - *delta_h * 3600) / 60;
358}

363static inline void time_fmt_iso8601(char buf[static 21], time_t t)
364{
365 int len;
366 struct tm tm;
367
368 if (!gmtime_r(&t, &tm)) {
369 e_panic("invalid timestamp: %jd", t);
370 }
371 len = snprintf(buf, 21, ISO8601_GMT_FMT, ISO8601_GMT_FMT_ARG(&tm));
372 if (len >= 21) {
373 e_panic("invalid timestamp: %jd", t);
374 }
375}

377/* Format a timestamp into a local ISO 8601 time. This function calls the
378 * time_get_localtime function, so you should not call it concurrently to
379 * another call of it or to a call to the time_get_localtime or to the
380 * localtime_r function. See time_get_localtime for more information.
381 */
382static inline void time_fmt_localtime_iso8601(char buf[static 26], time_t t,
383 const char *tz)
384{
385 int len;
386 struct tm tm;
387 int delta_h, delta_m;
388
389 time_get_localtime(&t, &tm, tz);
390 time_get_gmt_delta(&tm, &delta_h, &delta_m);
391
392 len = snprintf(buf, 26, ISO8601_TZ_FMT,
393 ISO8601_TZ_FMT_ARG(&tm, delta_h, delta_m));
394 if (len >= 26) {
395 e_panic("invalid timestamp: %jd", t);
396 }
397}

399static inline void time_fmt_iso8601_msec(char buf[static 25], time_t t,
400 int msec)
401{
402 int len;
403 struct tm tm;
404
405 /* XXX %03d gives a minimum width but not a maximum one, so we need to be
406 * careful and not overflow the buffer of size 25 with invalid inputs.
407 */
408 if (msec < 0 || msec >= 1000) {
409 e_panic("invalid msec: %d", msec);
410 }
411 if (!gmtime_r(&t, &tm)) {
412 e_panic("invalid timestamp: %jd", t);
413 }
414 len = snprintf(buf, 25, ISO8601_GMT_MSEC_FMT,
415 ISO8601_GMT_MSEC_FMT_ARG(&tm, msec));
416 if (len >= 25) {
417 e_panic("invalid timestamp: %jd", t);
418 }
419}

516/** Parse a string as a date and builds the corresponding unix timestamp.
517 *
518 * That function parses a string from one of those three formats:
519 * - ISO8601: YYYY-MM-DD[Thh:mm:ss][TZ]
520 * - RFC822: [Day, ]D month YYYY hh:mm:ss[ TZ]
521 * - Unix timestamp in decimal; it is expected to be in seconds but values
522 * greater than 30000000000 are considered in milliseconds, and converted to
523 * seconds.
524 * For information, 30000000000 is "Fri 30 Aug 2920 07:20:00 AM CEST" in
525 * seconds, and "Mon 14 Dec 1970 06:20:00 AM CET" in milliseconds.
526 *
527 * The parsing is done case insensitively. The timezone is optional, when
528 * omitted, the local timezone is used. When present it must have one of the
529 * following format:
530 * - literal name (e.g. DST, CEST, UTC, Z, GMT, ...)
531 * - +/-hh:mm
532 * - +/-hhmm
533 * - +/-hh
534 *
535 * \param[in,out] ps the stream from which the date should be parsed.
536 * \param[out] res the timestamp (in seconds).
537 * \return a negative value in case of error.
538 */
539int time_parse(pstream_t *ps, time_t *res);

557static inline struct timeval timeval_add(struct timeval a, struct timeval b)
558{
559 struct timeval res;
560
561 res.tv_sec = a.tv_sec + b.tv_sec;
562 res.tv_usec = a.tv_usec + b.tv_usec;
563 if (res.tv_usec >= 1000 * 1000) {
564 res.tv_sec += 1;
565 res.tv_usec -= 1000 * 1000;
566 }
567 return res;
568}

570static inline struct timeval timeval_addmsec(struct timeval a, int64_t msec)
571{
572 struct timeval res;
573
574 res.tv_sec = a.tv_sec + msec / 1000;
575 res.tv_usec = a.tv_usec + ((msec % 1000) * 1000);
576 if (res.tv_usec >= 1000 * 1000) {
577 res.tv_sec += 1;
578 res.tv_usec -= 1000 * 1000;
579 }
580 return res;
581}

650/* we use gettimeofday() and getrusage() for accurate benchmark timings.
651 * - clock(); returns process time, but has a precision of only 10ms
652 * - clock_gettime() has a better precision: timespec can measure times
653 * down to the nanosecond, but only real time is supported in linux
654 * (CLOCK_REALTIME), CLOCK_PROCESS_CPUTIME_ID is not supported, and
655 * librt must be specified at link time.
656 * - times() has a precision of 10ms and units must be retrieved with a
657 * separate call to sysconf(_SC_CLK_TCK).
658 */
659typedef struct proctimer_t {
660 struct timeval tv, tv1;
661#if PROCTIMER_USE_RUSAGE
662 struct rusage ru, ru1;
663#endif
664 unsigned long hc, hc1;
665 unsigned int elapsed_real;
666 unsigned int elapsed_user;
667 unsigned int elapsed_sys;
668 unsigned int elapsed_proc;
669 unsigned long elapsed_hard;
670} proctimer_t;

672/* for each time source, the following statistics are computed :
673 * - minimal value
674 * - maximal value
675 * - cumulated sum (tot)
676 */
677#define PTIMER_STATS(T, type) \
678 T type##_min; \
679 T type##_max; \
680 T type##_tot
681
682typedef struct proctimerstat_t {
683 unsigned int nb;
684
685 PTIMER_STATS(unsigned int, real);
686 PTIMER_STATS(unsigned int, user);
687 PTIMER_STATS(unsigned int, sys);
688 PTIMER_STATS(unsigned int, proc);
689 PTIMER_STATS(unsigned long, hard);
690} proctimerstat_t;
691
692#undef PTIMER_STATS

694static inline void proctimer_start(proctimer_t *tp) {
695 gettimeofday(&tp->tv, NULL);
696#if PROCTIMER_USE_RUSAGE
697 getrusage(RUSAGE_SELF, &tp->ru);
698#endif
699 tp->hc = hardclock();
700}
701
702static inline long long proctimer_stop(proctimer_t *tp) {
703 tp->hc1 = hardclock();
704#if PROCTIMER_USE_RUSAGE
705 getrusage(RUSAGE_SELF, &tp->ru1);
706#endif
707 gettimeofday(&tp->tv1, NULL);
708 tp->elapsed_real = timeval_diff(&tp->tv1, &tp->tv);
709#if PROCTIMER_USE_RUSAGE
710 tp->elapsed_user = timeval_diff(&tp->ru1.ru_utime, &tp->ru.ru_utime);
711 tp->elapsed_sys = timeval_diff(&tp->ru1.ru_stime, &tp->ru.ru_stime);
712 tp->elapsed_proc = tp->elapsed_user + tp->elapsed_sys;
713#else
714 tp->elapsed_sys = 0;
715 tp->elapsed_proc = tp->elapsed_user = tp->elapsed_real;
716#endif
717 tp->elapsed_hard = tp->hc1 - tp->hc;
718 return tp->elapsed_proc;
719}

721static inline void proctimerstat_addsample(proctimerstat_t *pts,
722 proctimer_t *tp) {
723#define PTIMER_COUNT(type) \
724 do { \
725 if (pts->nb != 0) { \
726 pts->type##_min = MIN(tp->elapsed_##type, pts->type##_min); \
727 pts->type##_max = MAX(tp->elapsed_##type, pts->type##_max); \
728 pts->type##_tot += tp->elapsed_##type; \
729 } else { \
730 pts->type##_min = tp->elapsed_##type; \
731 pts->type##_max = tp->elapsed_##type; \
732 pts->type##_tot = tp->elapsed_##type; \
733 } \
734 } while (0)
735
736 PTIMER_COUNT(real);
737 PTIMER_COUNT(user);
738 PTIMER_COUNT(sys);
739 PTIMER_COUNT(proc);
740 PTIMER_COUNT(hard);
741#undef PTIMER_COUNT
742 pts->nb++;
743}

759/** timing_scope.
760 *
761 * timing_scope can be used to log the execution time of a scope of code.
762 * For example:
763 *
764 * {
765 * timing_scope(&_G.logger, "desc");
766 *
767 * ...
768 * }
769 *
770 * will emit a log with the execution time of the block of code (starting from
771 * the call to "timing_scope_start()").
772 *
773 * It can also avoid emitting the log if the execution time is below a given
774 * threshold, or emit a trace of level 1 otherwise.
775 *
776 * {
777 * // Emit a notice if the blocks takes more than 100ms.
778 * timing_scope_threshold(&_G.logger, 100, "desc");
779 *
780 * // Emit a warning if the blocks takes more than 500ms.
781 * timing_scope_timeout(&_G.logger, 500, "desc");
782 *
783 * ...
784 * }
785 *
786 * You can also use the low level functions \ref timing_scope_start and
787 * \ref timing_scope_finish.
788 */

790 logger_t *logger;
791 lstr_t desc;
792 const char *file;
793 const char *func;
794 int line;
795 struct timeval tv_start;
796 int64_t threshold_ms;
797 int log_level;
798} timing_scope_ctx_t;

803 * \param[in] file Source file of the code scope.
804 * \param[in] func Function where the code scope is.
805 * \param[in] line Line of the code scope in the source file.
806 *

808 * of milliseconds, a log will be emitted;
809 * otherwise, the timing will be printed in
810 * a trace of level 1.
811 *
812 * \param[in] level Log level to use if the scope execution takes more than
813 \p threshold_ms milliseconds.
814 *
815 * \param[in] fmt Description of the block of scope to profile; will
816 * be used in the log.
817 *
818 * \return a timing scope context.
819 */

829 * emits it in a log.
