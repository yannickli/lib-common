22 int level, bool do_gzip)
23{
24 int err;
25 sb_t orig = *out;
26 z_stream stream = {
27 .next_in = (Bytef *)data,
28 .avail_in = dlen,
29 };
30
31 RETHROW(deflateInit2(&stream, level, Z_DEFLATED,
32 MAX_WBITS + (do_gzip ? 16 : 0),
33 MAX_MEM_LEVEL, Z_DEFAULT_STRATEGY));
34
35 for (;;) {
36 stream.next_out = (Bytef *)sb_grow(out, MAX(stream.avail_in / 2, 128));
37 stream.avail_out = sb_avail(out);
38
39 switch ((err = deflate(&stream, Z_FINISH))) {
40 case Z_STREAM_END:
41 __sb_fixlen(out, orig.len + stream.total_out);
42 IGNORE(deflateEnd(&stream));
43 return out->len - orig.len;
44 case Z_OK:
45 /* Compression OK, but not finished, must allocate more space */
46 __sb_fixlen(out, orig.len + stream.total_out);
47 break;
48 default:
49 __sb_rewind_adds(out, &orig);
50 deflateEnd(&stream);
51 return err;
52 }
53 }
54}
