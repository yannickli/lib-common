21static void sb_add_cell(sb_t *out, const struct table_hdr_t *col,
22 int col_size, bool is_hdr, bool is_last,
23 lstr_t content)
24{
25 int len = lstr_utf8_strlen(content);
26
27 if (len > col_size && col->add_ellipsis) {
28 content = lstr_utf8_truncate(content, col_size - 1);
29 sb_add(out, content.s, content.len);
30 sb_adduc(out, 0x2026 /* â€¦ */);
31 } else
32 if (len >= col_size) {
33 content = lstr_utf8_truncate(content, col_size);
34 sb_add(out, content.s, content.len);
35 } else {
36 int padding = col_size - len;
37 int left_padding = 0;
38 int right_padding = 0;
39
40 switch (is_hdr ? ALIGN_LEFT : col->align) {
41 case ALIGN_LEFT:
42 break;
43
44 case ALIGN_CENTER:
45 left_padding = padding / 2;
46 break;
47
48 case ALIGN_RIGHT:
49 left_padding = padding;
50 break;
51 }
52 right_padding = padding - left_padding;
53 if (left_padding) {
54 sb_addnc(out, left_padding, ' ');
55 }
56 sb_add_lstr(out, content);
57 if (right_padding && !is_last) {
58 sb_addnc(out, right_padding, ' ');
59 }
60 }
61}

63/* Return 0 if something has been written. */
64static int
65sb_write_table_cell(sb_t *out, const struct table_hdr_t *col, int col_size,
66 bool is_hdr, bool is_first, bool is_last, lstr_t content,
67 int csv_sep)
68{
69 if (col_size == 0) {
70 /* Omit column. */
71 return -1;
72 }
73
74 if (!content.len) {
75 content = col->empty_value;
76 }
77
78 if (!is_first) {
79 if (csv_sep) {
80 sb_addc(out, csv_sep);
81 } else {
82 sb_adds(out, " ");
83 }
84 }
85
86 if (csv_sep) {
87 sb_add_lstr_csvescape(out, csv_sep, content);
88 } else {
89 sb_add_cell(out, col, col_size, is_hdr, is_last, content);
90 }
91
92 return 0;
93}

95/** Write table or a csv if csv_sep != 0. */
96static void
97sb_write_table(sb_t *out, const qv_t(table_hdr) *hdr,
98 const qv_t(table_data) *data, int *col_sizes, int csv_sep)
99{
100 bool first_column = true;
101
102 /* Write the header. */
103 tab_enumerate_ptr(pos, col_hdr, hdr) {
104 if (sb_write_table_cell(out, col_hdr, col_sizes[pos], true,
105 first_column, pos == hdr->len - 1,
106 col_hdr->title, csv_sep) == 0)
107 {
108 first_column = false;
109 }
110 }
111 sb_addc(out, '\n');
112
113 /* Write the content. */
114 tab_for_each_ptr(row, data) {
115 first_column = true;
116
117 tab_enumerate_ptr(pos, col_hdr, hdr) {
118 lstr_t content = LSTR_NULL;
119
120 if (pos < row->len) {
121 content = row->tab[pos];
122 }
123
124 if (sb_write_table_cell(out, col_hdr, col_sizes[pos],
125 false, first_column, pos == hdr->len - 1,
126 content, csv_sep) == 0)
127 {
128 first_column = false;
129 }
130 }
131 sb_addc(out, '\n');
132 }
133}

135void sb_add_table(sb_t *out, const qv_t(table_hdr) *hdr,
136 const qv_t(table_data) *data)
137{
138 int *col_sizes = p_alloca(int, hdr->len);
139 int row_size = 0;
140 int col_count = 0;
141
142 /* Compute the size of the columns */
143 tab_enumerate_ptr(pos, col_hdr, hdr) {
144 int *col = &col_sizes[pos];
145 bool has_value = false;
146
147 *col = MAX(col_hdr->min_width, lstr_utf8_strlen(col_hdr->title));
148
149 tab_for_each_ptr(row, data) {
150 if (row->len <= pos) {
151 *col = MAX(*col, lstr_utf8_strlen(col_hdr->empty_value));
152 } else {
153 *col = MAX(*col, lstr_utf8_strlen(row->tab[pos]));
154
155 if (row->tab[pos].len) {
156 has_value = true;
157 }
158 }
159 }
160
161 if (hdr->tab[pos].max_width) {
162 *col = MIN(*col, col_hdr->max_width);
163 }
164 if (hdr->tab[pos].omit_if_empty && !has_value) {
165 *col = 0;
166 } else {
167 col_count++;
168 }
169 row_size += *col;
170 }
171
172 row_size += 2 * (col_count - 1) + 1;
173 sb_grow(out, row_size * (data->len + 1));
174
175 sb_write_table(out, hdr, data, col_sizes, 0);
176}

178void sb_add_csv_table(sb_t *out, const qv_t(table_hdr) *hdr,
179 const qv_t(table_data) *data, int sep)
180{
181 int *populated_cols = p_alloca(int, hdr->len);
182
183 /* Check if we have empty columns. */
184 tab_enumerate_ptr(pos, col_hdr, hdr) {
185 int *populated_col = &populated_cols[pos];
186
187 *populated_col = false;
188
189 if (!col_hdr->omit_if_empty || col_hdr->empty_value.len) {
190 /* If table_hdr__t has a default empty value, the column will not
191 * be omitted.
192 */
193 *populated_col = true;
194 continue;
195 }
196
197 tab_for_each_ptr(row, data) {
198 if (row->len <= pos) {
199 /* Not enough data columns. */
200 break;
201 }
202
203 if (row->tab[pos].len) {
204 /* Column will not be omitted. */
205 *populated_col = true;
206 break;
207 }
208 }
209 }
210
211 sb_write_table(out, hdr, data, populated_cols, sep);
212}
