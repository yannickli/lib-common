58/** This macro provides assertions that remain activated in production builds.
59 *
60 * \param[in] level The syslog level (fatal, panic, error, warning, notice,
61 * info or debug).
62 * \param[in] Cond The condition to verify
63 * \param[in] fmt The message to log if the condition is not verified.
64 */
65#define __e_assert(level, Cond, StrCond, fmt, ...) do { \
66 if (unlikely(!(Cond))) { \
67 e_##level(E_PREFIX("assertion failed: \"%s\": "fmt), StrCond, \
68 ##__VA_ARGS__); \
69 } \
70 } while (0)

126#define e_name_is_traced(lvl, name) \
127 ({ \
128 int8_t e_res; \
129 \
130 if (__builtin_constant_p(lvl) && __builtin_constant_p(name)) { \
131 static int8_t e_traced; \

133 if (unlikely(e_traced == 0)) { \

183static ALWAYS_INLINE __must_check__
184bool e_expect(bool cond, bool expected, const char * nonnull expr,
185 const char * nonnull file, int line, const char * nonnull func)
186{
187 if (unlikely(cond != expected)) {
188#ifdef NDEBUG
189 ps_write_backtrace(-1, false);
190 e_error("assertion (%s) failure: %s:%d:%s", expr, file, line, func);
191#else
192 __assert_fail(expr, file, line, func);
193#endif
194 }
195 return cond;
196}

207/** Function executed when writing the .debug file.
208 *
209 * This callback is called when a crash happens in the debug stack scope where
210 * the callback has been registered.
211 *
212 * \warning It should write the user data using \p dprintf. Please notice that
213 * our custom formats (%*pM, %pL, %*pS, ...) wont work in this
214 * context.
215 *
216 * \warning This code is called in the context of a crash, some contextual
217 * data may be invalid or corrupted.
218 *
219 * \warning Do not use malloc or functions that could use malloc in the
220 * callback. It could result in deadlocks.
221 */
