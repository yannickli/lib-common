79void (obj_print_references)(const object_t *nonnull obj)
80{
81 int64_t tagged_refcnt = 0;
82 logger_notice_scope(&_G.logger);
83
84 if (obj->obj_tagged_refs_) {
85 tab_for_each_ptr(ref, &obj->obj_tagged_refs_->refs) {
86 tagged_refcnt += ref->refcnt;
87 }
88 }
89 logger_cont("object @%p, refcnt=%'zd, %'jd tagged reference(s)",
90 obj, obj->refcnt, tagged_refcnt);
91 if (obj->obj_tagged_refs_) {
92 t_scope;
93 SB_1k(table_buf);
94 qv_t(table_hdr) hdr;
95 qv_t(table_data) data;
96 table_hdr_t hdr_data[] = { {
97 /* For indentation. */
98 .title = LSTR_IMMED(" "),
99 }, {
100 .title = LSTR_IMMED("TAG"),
101 }, {
102 .title = LSTR_IMMED("FUNCTION"),
103 }, {
104 .title = LSTR_IMMED("FILE:LINE"),
105 }, {
106 .title = LSTR_IMMED("REFCNT"),
107 } };
108
109 logger_cont(":\n");
110
111 qv_init_static(&hdr, hdr_data, countof(hdr_data));
112 t_qv_init(&data, obj->obj_tagged_refs_->refs.len);
113
114 tab_for_each_ptr(ref, &obj->obj_tagged_refs_->refs) {
115 qv_t(lstr) *row;
116
117 row = t_qv_init(qv_growlen(&data, 1), 4);
118 qv_append(row, LSTR_EMPTY_V);
119 if (ref->tag) {
120 qv_append(row, LSTR(ref->tag));
121 } else {
122 qv_append(row, LSTR("<obj_retain_scope>"));
123 }
124 qv_append(row, LSTR(ref->func));
125 qv_append(row, t_lstr_fmt("%s:%d", ref->file, ref->line));
126 qv_append(row, t_lstr_fmt("%'d", ref->refcnt));
127 }
128 sb_add_table(&table_buf, &hdr, &data);
129 /* Remove the trailing '\n'. */
130 sb_shrink(&table_buf, 1);
131 logger_cont("%*pM", SB_FMT_ARG(&table_buf));
132 }
133}

135static obj_tagged_ref_t *obj_find_tagged_ref(object_t *obj, const char *tag)
136{
137 THROW_NULL_IF(!obj->obj_tagged_refs_);
138
139 tab_for_each_ptr(ref, &obj->obj_tagged_refs_->refs) {
140 if (ref->tag && strequal(ref->tag, tag)) {
141 return ref;
142 }
143 }
144
145 return NULL;
146}

148static obj_tagged_ref_t *
149obj_find_scope_ref(object_t *obj, const char *file, int line)
150{
151 THROW_NULL_IF(!obj->obj_tagged_refs_);
152
153 tab_for_each_ptr(ref, &obj->obj_tagged_refs_->refs) {
154 if (strequal(ref->file, file) && ref->line == line && !ref->tag) {
155 return ref;
156 }
157 }
158
159 return NULL;
160}

162static obj_tagged_ref_t *
163obj_add_tagged_ref(object_t *obj, const char *nullable tag,
164 const char *nonnull func,
165 const char *nonnull file, int line)
166{
167 obj_tagged_ref_t *ref;
168
169 if (!obj->obj_tagged_refs_) {
170 obj->obj_tagged_refs_ = obj_tagged_ref_list_new();
171 }
172
173 ref = qv_growlen0(&obj->obj_tagged_refs_->refs, 1);
174 ref->tag = tag;
175 ref->func = func;
176 ref->file = file;
177 ref->line = line;
178
179 return ref;
180}

182object_t *nonnull
183(obj_tagged_retain)(object_t *nonnull obj, const char *nonnull tag,
184 const char *nonnull func,
185 const char *nonnull file, int line)
186{
187 obj_tagged_ref_t *ref;
188
189 ref = obj_find_tagged_ref(obj, tag);
190 if (ref) {
191 if (!strequal(ref->file, file) || ref->line != line) {
192 obj_print_references(obj);
193 logger_panic(&_G.logger, "reference tagging collision : "
194 "the tag `%s` is used for two different retains, "
195 "in %s (%s:%d) and in %s (%s:%d)", tag,
196 ref->func, ref->file, ref->line,
197 func, file, line);
198 }
199 } else {
200 ref = obj_add_tagged_ref(obj, tag, func, file, line);
201 }
202 ref->refcnt++;
203
204 return obj_vcall(obj, retain);
205}

218void (obj_tagged_release)(object_t *nonnull *nonnull obj_p,
219 const char *nonnull tag)
220{
221 obj_tagged_ref_t *reference;
222 object_t *obj = *obj_p;
223
224 reference = obj_find_tagged_ref(obj, tag);
225 if (unlikely(!reference)) {
226 obj_print_references(obj);
227 logger_panic(&_G.logger,
228 "broken tagged release: cannot find reference for tag "
229 "`%s`", tag);
230 }
231 if (unlikely(!reference->refcnt)) {
232 obj_print_references(obj);
233 logger_panic(&_G.logger,
234 "broken tagged release: the last reference for tag `%s` "
235 "has already been released", tag);
236 }
237 reference->refcnt--;
238
239 obj_release_vcall(obj_p);
240}

242object_t *nonnull
243(obj_retain_scope)(object_t *nonnull obj,
244 const char *nonnull func,
245 const char *nonnull file, int line)
246{
247 obj_tagged_ref_t *ref;
248
249 ref = obj_find_scope_ref(obj, func, line);
250 if (!ref) {
251 ref = obj_add_tagged_ref(obj, NULL, func, file, line);
252 }
253 ref->refcnt++;
254
255 return obj_vcall(obj, retain);
256}

258void (obj_release_scope)(object_t *nonnull *nonnull obj_p,
259 const char *nonnull file, int line)
260{
261 object_t *obj = *obj_p;
262 obj_tagged_ref_t *reference;
263
264 reference = obj_find_scope_ref(obj, file, line);
265 reference->refcnt--;
266
267 obj_release_vcall(obj_p);
268}

270static void obj_check_tagged_refs_before_wipe(const object_t *obj)
271{
272 if (!obj->obj_tagged_refs_) {
273 return;
274 }
275 tab_for_each_ptr(ref, &obj->obj_tagged_refs_->refs) {
276 if (!ref->refcnt) {
277 continue;
278 }
279 obj_print_references(obj);
280 logger_panic(&_G.logger,
281 "a reference created in %s (%s:%d) with tag "
282 "`%s` wasn't released with obj_tagged_release()",
283 ref->func, ref->file, ref->line, ref->tag);
284 }

291bool cls_inherits(const void *_cls, const void *vptr)
292{
293 const object_class_t *cls = _cls;
294 while (cls) {
295 if (cls == vptr)
296 return true;
297 cls = cls->super;
298 }
299 return false;
300}

302static void obj_init_real_aux(object_t *o, const object_class_t *cls)
303{
304 object_t *(*init)(object_t *) = cls->init;
305
306 if (!init)
307 return;
308
309 while (cls->super && init == cls->super->init) {
310 cls = cls->super;
311 }
312 if (cls->super)
313 obj_init_real_aux(o, cls->super);
314 (*init)(o);
315}

317void *obj_init_real(const void *_cls, void *_o, mem_pool_t *mp)
318{
319 const object_class_t *cls = _cls;
320 object_t *o = _o;

329void obj_wipe_real(object_t *o)
330{
331 const object_class_t *cls = o->v.ptr;
332 void (*wipe)(object_t *);
333
334 /* a crash here means obj_wipe was called on a reachable object.
335 * It's likely the caller should have used obj_release() instead.
336 */
337 assert (o->refcnt == 1);
338
339 while ((wipe = cls->wipe)) {
340 (*wipe)(o);
341 do {
342 cls = cls->super;
343 if (!cls)
344 return;
345 } while (wipe == cls->wipe);
346 }
347 o->refcnt = 0;
348
349#ifndef NDEBUG
350 if (o->obj_tagged_refs_) {
351 obj_check_tagged_refs_before_wipe(o);
352 obj_tagged_ref_list_delete(&o->obj_tagged_refs_);
353 }
354#endif /* NDEBUG */
355}

357static object_t *obj_retain_(object_t *obj)
358{
359 assert (obj->mp != &mem_pool_static);
360
361 if (likely(obj->refcnt > 0)) {
362 if (likely(++obj->refcnt > 0))
363 return obj;
364 logger_panic(&_G.logger, "too many refcounts");
365 }
366
367 switch (obj->refcnt) {
368 case 0:
369 logger_panic(&_G.logger, "probably acting on a deleted object");
370 default:
371 /* WTF?! probably a memory corruption */
372 logger_panic(&_G.logger, "should not happen");
373 }
374}
375
376static void obj_release_(object_t *obj, bool *nullable destroyed)
377{
378 assert (obj->mp != &mem_pool_static);
379
380 if (obj->refcnt > 1) {
381 --obj->refcnt;
382 } else if (obj->refcnt == 1) {
383 obj_wipe_real(obj);
384 mp_delete(obj->mp, &obj);
385 } else {
386 switch (obj->refcnt) {
387 case 0:
388 logger_panic(&_G.logger, "object refcounting issue");
389 default:
390 /* Probably a memory corruption we should have hit 0 first. */
391 logger_panic(&_G.logger, "should not happen");
392 }
393 }
394
395 if (destroyed) {
396 *destroyed = !obj;
397 }
398}
