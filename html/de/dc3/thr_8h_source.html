41/** \brief declare a function to be run when a thread starts and exits.
42 *
43 * The init function is run when a thread inits, but not for the main thread,
44 * it's up to the programmer to be sure it's done or not needed for this
45 * thread.
46 *
47 * The exit function is run when a thread exit, even when it is the main.
48 *
49 * If pthreads are in use, and that the host program uses #pthread_create (or
50 * #thr_initialize or #pthread_force_use) then this system is active, else
51 * hooks are not run for threads.
52 *
53 * The exit hooks are always run for the main thread, independently from
54 * pthreads and this system beeing active or not.
55 *
56 * \param[in] init name of the function to run at thread init.
57 * \param[in] init name of the function to run at thread exit.
58 */
59#define thr_hooks(init, exit) \
60 static __attribute__((constructor)) void PT_##fn##_exit(void) { \
61 __builtin_choose_expr(__builtin_constant_p(init), (void)0, ({ \
62 static struct thr_ctor ctor = { .cb = (init) }; \
63 if (ctor.cb) { \
64 dlist_add_tail(&thr_hooks_g.init_cbs, &ctor.link); \
65 } \
66 })); \
67 __builtin_choose_expr(__builtin_constant_p(exit), (void)0, ({ \
68 static struct thr_ctor ctor = { .cb = (exit) }; \
69 if (ctor.cb) { \
70 dlist_add(&thr_hooks_g.exit_cbs, &ctor.link); \
71 } \
72 })); \
73 }
