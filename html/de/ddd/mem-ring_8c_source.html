26/*
27 * Ring memory allocator
28 * ~~~~~~~~~~~~~~~~~~~~~
29 *
30 * This allocator has mostly the same properties as the stacked allocator but
31 * works on a ring.
32 *
33 * Note: this code shares a big part with core-mem-stack.c which should be
34 * merged.
35 *
36 */
37
38/** Size tuning parameters.
39 * These are multiplicative factors over rp_alloc_mean.
40 */
41#define RESET_MIN 56 /*< minimum size in mem_ring_reset */
42#define RESET_MAX 256 /*< maximum size in mem_ring_reset */

105static size_t rp_alloc_mean(ring_pool_t *rp)
106{
107 return rp->alloc_sz / rp->alloc_nb;
108}

110static ring_blk_t *blk_entry(dlist_t *l)
111{
112 return dlist_entry(l, ring_blk_t, blist);

115static inline uint8_t align_boundary(size_t size)
116{
117 return MIN(16, 1 << bsrsz(size | 1));
118}
119
120static inline bool is_aligned_to(const void *addr, size_t boundary)
121{
122 return ((uintptr_t)addr & (boundary - 1)) == 0;
123}
124
125static byte *align_for(const void *mem, size_t size)
126{
127 size_t bmask = align_boundary(size) - 1;
128 return (byte *)(((uintptr_t)mem + bmask) & ~bmask);
129}

131static ring_blk_t *blk_create(ring_pool_t *rp, size_t size_hint)
132{
133 size_t blksize = size_hint + sizeof(ring_blk_t);
134 size_t alloc_target = MIN(100U << 20, 64 * rp_alloc_mean(rp));
135 ring_blk_t *blk;
136
137 if (blksize < rp->minsize) {
138 blksize = rp->minsize;
139 }
140 if (blksize < alloc_target) {
141 blksize = alloc_target;
142 }
143 blksize = ROUND_UP(blksize, PAGE_SIZE);
144 icheck_alloc(blksize);
145 blk = imalloc(blksize, 0, MEM_RAW | MEM_LIBC);
146 blk->start = blk->area;
147 blk->size = blksize - sizeof(*blk);
148 rp->ringsize += blk->size;
149 if (likely(rp->cblk)) {
150 dlist_add_after(&rp->cblk->blist, &blk->blist);
151 } else {
152 dlist_init(&blk->blist);
153 }
154 rp->nbpages++;
155 return blk;
156}
157
158static void blk_destroy(ring_pool_t *rp, ring_blk_t *blk)
159{
160 rp->ringsize -= blk->size;
161 rp->nbpages--;
162 dlist_remove(&blk->blist);
163 mem_tool_allow_memory(blk, blk->size + sizeof(*blk), false);
164 ifree(blk, MEM_LIBC);
165}

167static bool blk_contains(const ring_blk_t *blk, const void *ptr)
168{
169 return (const byte *)ptr >= (const byte *)blk->start
170 && (const byte *)ptr <= (const byte *)blk->start + blk->size;
171}
172
173static byte *blk_end(ring_blk_t *blk)
174{
175 return blk->area + blk->size;
176}

178static ring_blk_t *
179frame_get_next_blk(ring_pool_t *rp, size_t size)
180{
181 ring_blk_t *cur = rp->cblk;
182 frame_t *start = dlist_first_entry(&rp->fhead, frame_t, flist);
183
184 while (!dlist_is_empty(&cur->blist)) {
185 ring_blk_t *blk = dlist_next_entry(cur, blist);
186
187 /* XXX: start is the offset to a frame, IOW frame == blk->end
188 * cannot happen because a frame cannot have a zero offset
189 * from the next block, even if blocks are allocated
190 * without overhead, such as by mmapping.
191 */
192 if (blk_contains(blk, start)) {
193 break;
194 }
195
196 if (blk->size >= size && blk->size > 8 * rp_alloc_mean(rp)) {
197 return blk;
198 }
199 blk_destroy(rp, blk);
200 }
201 return blk_create(rp, size);
202}

204static void *rp_reserve(ring_pool_t *rp, size_t size, ring_blk_t **blkp)
205{
206 byte *res = align_for(rp->pos, size);
207
208 /* Note for programmers: if you abort() here, it's because you're
209 * allocating in a pool where you haven't performed a r_newframe() first
210 */
211 assert (rp->pos);
212
213 if (unlikely(res + size > blk_end(rp->cblk))) {
214 ring_blk_t *blk = frame_get_next_blk(rp, size);
215
216 *blkp = blk;
217 res = blk->area;
218 } else {
219 *blkp = rp->cblk;
220 }
221 mem_tool_allow_memory(res, size, false);
222 if (unlikely(rp->alloc_sz + size < rp->alloc_sz)
223 || unlikely(rp->alloc_nb == UINT32_MAX))
224 {
225 rp->alloc_sz /= 2;
226 rp->alloc_nb /= 2;
227 }
228 rp->alloc_sz += size;
229 rp->alloc_nb += 1;
230 return res;
231}

233__flatten
234static void *rp_alloc(mem_pool_t *_rp, size_t size, size_t alignment,
235 mem_flags_t flags)
236{
237 ring_pool_t *rp = container_of(_rp, ring_pool_t, funcs);
238 byte *res;
239
240 if (unlikely(alignment > 16)) {
241 e_panic("mem_pool_ring does not support alignments greater than 16");
242 }
243
244 if (unlikely((size == 0))) {
245 return MEM_EMPTY_ALLOC;
246 }
247
248 spin_lock(&rp->lock);
249 res = rp_reserve(rp, size, &rp->cblk);
250 rp->pos = res + size;
251 rp->last = res;
252 spin_unlock(&rp->lock);
253
254 if (!(flags & MEM_RAW)) {
255 memset(res, 0, size);
256 }
257
258 return res;
259}

265static void *rp_realloc(mem_pool_t *_rp, void *mem, size_t oldsize,
266 size_t size, size_t alignment, mem_flags_t flags)
267{
268 ring_pool_t *rp = container_of(_rp, ring_pool_t, funcs);
269 byte *res;
270
271 if (unlikely(alignment > 16)) {
272 e_panic("mem_pool_ring does not support alignments greater than 16");
273 }
274
275 if (unlikely(oldsize == MEM_UNKNOWN)) {
276 e_panic("ring pools do not support reallocs with unknown old size");
277 }
278
279 if (unlikely(mem == MEM_EMPTY_ALLOC)) {
280 mem = NULL;
281 }
282
283 if (oldsize >= size) {
284 if (mem == rp->last) {
285 rp->pos = (byte *)mem + size;
286 }
287 mem_tool_disallow_memory((byte *)mem + size, oldsize - size);
288 return size ? mem : MEM_EMPTY_ALLOC;
289 }
290
291 if (mem != NULL && mem == rp->last
292 && is_aligned_to(mem, align_boundary(size))
293 && (byte *)rp->last + size <= blk_end(rp->cblk))
294 {
295 rp->pos = (byte *)rp->last + size;
296 rp->alloc_sz += size - oldsize;
297 mem_tool_allow_memory(mem, size, true);
298 res = mem;
299 } else {
300 res = rp_alloc(_rp, size, alignment, flags | MEM_RAW);
301 if (mem != NULL) {
302 memcpy(res, mem, oldsize);
303 mem_tool_allow_memory(mem, oldsize, false);
304 }
305 }
306 if (!(flags & MEM_RAW)) {
307 memset(res + oldsize, 0, size - oldsize);
308 }
309 return res;
310}

321static void
322mem_ring_protect(const ring_pool_t *rp, const ring_blk_t *blk,
323 const void *_start, const void *_end)
324{
325 const byte *end = _end;
326 const byte *start = _start;
327
328 while (!blk_contains(blk, end)) {
329 mem_tool_disallow_memory(start, blk->area + blk->size - start);
330 blk = dlist_next_entry(blk, blist);
331 start = blk->start;
332 }
333 mem_tool_disallow_memory(start, end - start);
334}
335#else
336#define mem_ring_protect(...) ((void)0)
337#endif

339static ALWAYS_INLINE
340void ring_reset_frame(ring_pool_t *rp, frame_t *frame, bool protect)
341{
342 assert (rp->ring == frame);
343
344 if (protect && rp->pos) {
345 mem_ring_protect(rp, frame->blk, &frame[1], rp->pos);
346 }
347
348 rp->last = NULL;
349 rp->pos = NULL;
350 rp->cblk = frame->blk;
351
352 /* TODO: evaluate the possibility to reset to the _previous_ frame if:
353 * - it exists
354 * - it's free
355 */
356}

358static ALWAYS_INLINE
359void frame_unregister(frame_t *frame)
360{
361 dlist_remove(&frame->flist);
362 mem_tool_disallow_memory(frame, sizeof(frame_t));
363}

365static ALWAYS_INLINE
366void ring_reset_to_prevframe(ring_pool_t *rp, frame_t *fprev, frame_t *frame)
367{
368 assert (frame == rp->ring);
369
370 frame_unregister(frame);
371 rp->ring = fprev;
372 rp->cblk = fprev->blk;
373 fprev->rp &= ~FRAME_IS_FREE;
374}

376static ALWAYS_INLINE
377void ring_setup_frame(ring_pool_t *rp, ring_blk_t *blk, frame_t *frame)
378{
379 spin_lock(&rp->lock);
380 frame->blk = blk;
381 frame->rp = (uintptr_t)rp;
382 dlist_add_tail(&rp->fhead, &frame->flist);
383
384 rp->ring = frame;
385 ring_reset_frame(rp, frame, false);
386 spin_unlock(&rp->lock);
387}

391mem_pool_t *mem_ring_new(const char *name, int initialsize)
392{
393 ring_pool_t *rp = p_new(ring_pool_t, 1);
394 ring_blk_t *blk;
395
396 rp->name = p_strdup(name);
397
398 dlist_init(&rp->fhead);
399
400 /* 640k should be enough for everybody =) */
401 if (initialsize <= 0) {
402 initialsize = 640 << 10;
403 }
404 rp->minsize = ROUND_UP(initialsize, PAGE_SIZE);
405 rp->funcs = pool_funcs;
406 rp->alloc_nb = 1; /* avoid the division by 0 */
407 rp->frames_cnt = 0;
408 rp->alive = true;
409
410 /* Makes the first frame */
411 blk = blk_create(rp, sizeof(frame_t));
412 mem_tool_allow_memory(blk->area, sizeof(frame_t), false);
413 ring_setup_frame(rp, blk, acast(frame_t, &blk->area));
414
415 spin_lock(&_G.all_pools_lock);
416 dlist_add_tail(&_G.all_pools, &rp->pool_list);
417 spin_unlock(&_G.all_pools_lock);
418
419 return &rp->funcs;
420}

424void mem_ring_delete(mem_pool_t **rpp)
425{
426 if (*rpp) {
427 ring_pool_t *rp = container_of(*rpp, ring_pool_t, funcs);
428
429 spin_lock(&rp->lock);
430
431 if (rp->frames_cnt) {
432 assert (rp->alive);
433 rp->alive = false;
434
435 /* XXX: the log module may have been deleted already,
436 * but we cannot depend on it. See #54184 */
437 if (MODULE_IS_LOADED(log)) {
438 e_trace(0, "keep ring-pool alive: %d frames in use",
439 rp->frames_cnt);
440 } else {
441 printf("keep ring-pool alive: %d frames in use",
442 rp->frames_cnt);
443 }
444 spin_unlock(&rp->lock);
445
446 *rpp = NULL;
447 return;
448 }
449
450 spin_lock(&_G.all_pools_lock);
451 dlist_remove(&rp->pool_list);
452 spin_unlock(&_G.all_pools_lock);
453
454 dlist_for_each(e, &rp->cblk->blist) {
455 blk_destroy(rp, blk_entry(e));
456 }
457 blk_destroy(rp, rp->cblk);
458 p_delete(&rp->name);
459 p_delete(&rp);
460 *rpp = NULL;
461 }
462}

464/**
465 * Unlike the public documentation claims, this function allocates nothing.
466 * The ring implementation ensures that we always have an active frame. Here
467 * we just check that the programmer is really where he thinks he is and we
468 * enable the active frame.
469 */
470const void *mem_ring_newframe(mem_pool_t *_rp)
471{
472 ring_pool_t *rp = container_of(_rp, ring_pool_t, funcs);
473
474 e_assert_null(panic, rp->pos, "previous memory frame not released!");
475 spin_lock(&rp->lock);
476 rp->pos = &rp->ring[1];
477 rp->frames_cnt++;
478 spin_unlock(&rp->lock);
479
480 return rp->ring;
481}

488const void *mem_ring_seal(mem_pool_t *_rp)
489{
490 ring_pool_t *rp = container_of(_rp, ring_pool_t, funcs);
491 frame_t *last = rp->ring;
492 frame_t *frame;
493 ring_blk_t *blk;
494
495 /* Makes a new frame */
496 frame = rp_reserve(rp, sizeof(frame_t), &blk);
497 ring_setup_frame(rp, blk, frame);
498
499 return last;
500}

502static void __mem_ring_reset(ring_pool_t *rp)
503{
504 size_t saved_size;
505 size_t max_size;
506 ring_blk_t *saved_blk = NULL;
507 frame_t *start = dlist_first_entry(&rp->fhead, frame_t, flist);
508
509 if (!mem_pool_is_enabled()) {
510 return;
511 }
512 if (rp->frames_cnt) {
513 return;
514 }
515
516 saved_blk = NULL;
517 saved_size = RESET_MIN * rp_alloc_mean(rp);
518 max_size = RESET_MAX * rp_alloc_mean(rp);
519
520 /* Keep the current block plus the one with more adapted size
521 * regarding the mean allocation size.
522 */
523 dlist_for_each(e, &rp->cblk->blist) {
524 ring_blk_t *blk = blk_entry(e);
525
526 /* XXX: do not remove the block which contains the first frame. */
527 if (blk_contains(blk, start)) {
528 continue;
529 }
530
531 if (blk->size > saved_size && blk->size < max_size) {
532 if (saved_blk) {
533 blk_destroy(rp, saved_blk);
534 }
535 saved_blk = blk;
536 saved_size = blk->size;
537 } else {
538 blk_destroy(rp, blk);
539 }
540 }
541
542 rp->nb_frames_release = 0;
543}

554void mem_ring_release(const void *cookie)
555{
556 frame_t *frame = (frame_t *)cookie;
557 ring_pool_t *rp;
558 bool to_delete = false;
559
560 if (!cookie) {
561 return;
562 }
563
564 rp = (ring_pool_t *)frame->rp;
565 assert (!(frame->rp & FRAME_IS_FREE));
566
567 spin_lock(&rp->lock);
568 if (rp->ring == frame) {
569 if (dlist_is_empty_or_singular(&rp->fhead)) {
570 ring_reset_frame(rp, frame, true);
571 } else {
572 frame_t *fprev = dlist_prev_entry(frame, flist);
573
574 if (fprev->rp & FRAME_IS_FREE) {
575 /* TODO We could rewind on more than one frame. */
576 ring_reset_to_prevframe(rp, fprev, frame);
577 } else {
578 ring_reset_frame(rp, frame, true);
579 }
580 }
581 } else {
582 frame_t *fnext = dlist_next_entry(frame, flist);
583
584 mem_ring_protect(rp, frame->blk, &frame[1], fnext);
585
586 if (fnext->rp & FRAME_IS_FREE) {
587 frame_unregister(fnext);
588 fnext = dlist_next_entry(frame, flist);
589 }
590
591 if (dlist_first_entry(&rp->fhead, frame_t, flist) == frame) {
592 frame_unregister(frame);
593 } else {
594 frame_t *fprev = dlist_prev_entry(frame, flist);
595 ring_blk_t *blk1, *blk2;
596
597 if (fprev->rp & FRAME_IS_FREE) {
598 frame_unregister(frame);
599 frame = fprev;
600 }
601 frame->rp |= FRAME_IS_FREE;
602
603 if (fnext == rp->ring && rp->pos == NULL) {
604 ring_reset_to_prevframe(rp, frame, fnext);
605 } else {
606 blk1 = frame->blk;
607 blk2 = fnext->blk;
608 if (blk1 != blk2 && dlist_next_entry(blk1, blist) != blk2) {
609 dlist_t *first = blk1->blist.next;
610 dlist_t *last = blk2->blist.prev;
611 dlist_t *at;
612
613 /* remove elements strictly between blk1 and blk2 */
614 __dlist_remove(&blk1->blist, &blk2->blist);
615 /* and splice first->...->last between at and at->next */
616 at = &rp->cblk->blist;
617 __dlist_splice2(at, at->next, first, last);
618 }
619 }
620 }
621 }
622
623 rp->frames_cnt--;
624 rp->nb_frames_release++;
625
626 if (rp->nb_frames_release >= 256) {
627 __mem_ring_reset(rp);
628 }
629
630 to_delete = rp->frames_cnt == 0 && !rp->alive;
631
632 spin_unlock(&rp->lock);
633
634 if (to_delete) {
635 mem_pool_t *mp = &rp->funcs;
636
637 mem_ring_delete(&mp);
638 }
639}

641const void *mem_ring_checkpoint(mem_pool_t *_rp)
642{
643 ring_pool_t *rp = container_of(_rp, ring_pool_t, funcs);
644 struct mem_ring_checkpoint cp = {
645 .frame = rp->ring,
646 .cblk = rp->cblk,
647 .last = rp->last,
648 .pos = rp->pos,
649 };
650 void *res;
651
652 res = memcpy(rp_alloc(_rp, sizeof(cp), mem_bit_align(_rp, alignof(cp)),
653 MEM_RAW),
654 &cp, sizeof(cp));
655 mem_ring_seal(_rp);
656
657 return res;
658}

660void mem_ring_rewind(mem_pool_t *_rp, const void *ckpoint)

690static size_t frame_getsize(frame_t *frame, const byte *pos)
691{
692 ring_pool_t *rp = (ring_pool_t *)(frame->rp & ~FRAME_IS_FREE);
693 const byte *start, *endp;
694 size_t size = 0;
695 ring_blk_t *blk = frame->blk;
696
697 start = (const byte *)frame;
698 if (frame == rp->ring) {
699 endp = pos;
700 } else {
701 endp = (const byte *)dlist_next_entry(frame, flist);
702 }
703
704 for (;;) {
705 if (blk_contains(blk, endp)) {
706 size += endp - start;
707 break;
708 }
709 size += blk_end(blk) - start;
710 if (dlist_is_last(&rp->cblk->blist, &blk->blist)) {
711 break;
712 }
713 blk = dlist_next_entry(blk, blist);
714 start = blk->area;
715 }
716
717 return size;
718}

720void mem_ring_dump(const mem_pool_t *_rp)
721{
722 ring_pool_t *rp = container_of(_rp, ring_pool_t, funcs);
723 int num = 0;
724 size_t bytes = 0;
725 frame_t *frame;
726
727 if (rp->cblk) {
728 bytes += rp->cblk->size;
729 dlist_for_each(e, &rp->cblk->blist) {
730 ring_blk_t *blk = blk_entry(e);
731 bytes += blk->size;
732 }
733 }
734
735 printf("-- \n");
736 printf("-- mem_ring_pool at %p: {\n", rp);
737
738 printf("-- ring=%p\n", rp->ring);
739 printf("-- last=%p\n", rp->last);
740 printf("-- pos=%p\n", rp->pos);
741 printf("-- cblk=%p\n", rp->cblk);
742
743 printf("-- minsize=%zd\n", rp->minsize);
744 printf("-- ringsize=%zd\n", rp->ringsize);
745 printf("-- bytes=%zd\n", bytes);
746 printf("-- nbpages=%d\n", rp->nbpages);
747 printf("-- alloc_sz=%zd\n", rp->alloc_sz);
748 printf("-- alloc_nb=%d\n", rp->alloc_nb);
749 printf("-- mean=%zd\n", rp_alloc_mean(rp));
750 printf("-- \n");
751
752 frame = dlist_first_entry(&rp->fhead, frame_t, flist);
