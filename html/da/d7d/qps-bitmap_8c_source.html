24static
25qps_bitmap_dispatch_t *w_deref_dispatch(qps_bitmap_t *map,
26 qps_bitmap_key_t key,
27 bool create)
28{
29 qps_bitmap_node_t dispatch_node;
30 STATIC_ASSERT(sizeof(qps_bitmap_dispatch_t) == 3 * QPS_PAGE_SIZE);
31
32 qps_hptr_deref(map->qps, &map->root_cache);
33 dispatch_node = map->root->roots[key.root];
34 if (dispatch_node == 0) {
35 if (!create) {
36 return NULL;
37 }
38 qps_hptr_w_deref(map->qps, &map->root_cache);
39 dispatch_node = qps_pg_map(map->qps, 3);
40 qps_pg_zero(map->qps, dispatch_node, 3);
41 map->root->roots[key.root] = dispatch_node;
42 }
43 return qps_pg_deref(map->qps, dispatch_node);
44}

46static
47uint64_t *w_deref_leaf(qps_bitmap_t *map, qps_bitmap_dispatch_t **dispatch,
48 qps_bitmap_key_t key, bool create)
49{
50 qps_bitmap_node_t leaf_node;
51
52 if (dispatch == NULL || *dispatch == NULL) {
53 return NULL;
54 }
55
56 leaf_node = (*(*dispatch))[key.dispatch].node;
57 if (leaf_node == 0) {
58 const uint32_t pages = map->root->is_nullable ? 2 : 1;
59 if (!create) {
60 return NULL;
61 }
62 *dispatch = w_deref_dispatch(map, key, false);
63 assert (*dispatch);
64 leaf_node = qps_pg_map(map->qps, pages);
65 qps_pg_zero(map->qps, leaf_node, pages);
66 (*(*dispatch))[key.dispatch].node = leaf_node;
67 (*(*dispatch))[key.dispatch].active_bits = 0;
68 }
69 return qps_pg_deref(map->qps, leaf_node);
70}

72static
73void delete_leaf(qps_bitmap_t *map, qps_bitmap_key_t key)
74{
75 qps_bitmap_dispatch_t *dispatch = w_deref_dispatch(map, key, false);
76 qps_bitmap_node_t leaf_node;
77 if (dispatch == NULL) {
78 return;
79 }
80
81 leaf_node = (*dispatch)[key.dispatch].node;
82 if (leaf_node == 0) {
83 return;
84 }
85
86 qps_pg_unmap(map->qps, leaf_node);
87 (*dispatch)[key.dispatch].node = 0;
88 for (int i = 0; i < QPS_BITMAP_DISPATCH; i++) {
89 if ((*dispatch)[i].node != 0) {
90 return;
91 }
92 }
93 qps_hptr_w_deref(map->qps, &map->root_cache);
94 qps_pg_unmap(map->qps, map->root->roots[key.root]);
95 map->root->roots[key.root] = 0;
96}

98static
99void delete_nodes(qps_bitmap_t *map)
100{
101 for (int i = 0; i < QPS_BITMAP_ROOTS; i++) {
102 const qps_bitmap_dispatch_t *dispatch;
103 const void *buf;
104 int pos = 0;
105
106 if (map->root->roots[i] == 0) {
107 continue;
108 }
109 dispatch = qps_pg_deref(map->qps, map->root->roots[i]);
110 buf = *dispatch;
111
112 /* scan_non_zero16 return the first non nul u16 position in range
113 * [pos, len] in given u16 buffer buf.
114 * We're looking for a nul node in qps_bitmap_dispatch_t which is
115 * (u32int_t) node + (uint16_t) active_bits
116 * we assume active_bits == 0 if node == 0
117 */
118 STATIC_ASSERT(sizeof(qps_bitmap_dispatch_t)
119 == QPS_BITMAP_DISPATCH * 3 * sizeof(uint16_t));
120 while ((pos = scan_non_zero16(buf, pos, 3 * QPS_BITMAP_DISPATCH)) >= 0) {
121 qps_bitmap_node_t node;
122 int p = pos / 3;
123 int r = pos % 3;
124
125 node = (*dispatch)[p].node;
126 if (!expect(node > 0)) {
127 /* XXX: "node" is not supposed to be 0 here, but it was
128 * observed on some production platforms; when it happens,
129 * following values of "node" are garbage and thus
130 * qps_pg_unmap crashes, so just avoid the crash.
131 */
132 break;
133 }
134 qps_pg_unmap(map->qps, node);
135 pos += 3 - r;
136 }
137 qps_pg_unmap(map->qps, map->root->roots[i]);
138 }
139}

141static
142void unload_nodes(qps_bitmap_t *map)
143{
144 for (int i = 0; i < QPS_BITMAP_ROOTS; i++) {
145 const qps_bitmap_dispatch_t *dispatch;
146 if (map->root->roots[i] == 0) {
147 continue;
148 }
149 dispatch = qps_pg_deref(map->qps, map->root->roots[i]);
150 for (int j = 0; j < QPS_BITMAP_DISPATCH; j++) {
151 if ((*dispatch)[j].node == 0) {
152 continue;
153 }
154 qps_pg_unload(map->qps, (*dispatch)[j].node);
155 }
156 qps_pg_unload(map->qps, map->root->roots[i]);
157 }
158}

164{

166 qps_hptr_t cache;

197qps_bitmap_state_t qps_bitmap_get(qps_bitmap_t *map, uint32_t row)
198{
199 qps_bitmap_key_t key = { .key = row };
200 const qps_bitmap_dispatch_t *dispatch;
201 const uint64_t *leaf;
202 qps_bitmap_node_t dispatch_node;
203 qps_bitmap_node_t leaf_node;
204
205 qps_hptr_deref(map->qps, &map->root_cache);
206 dispatch_node = map->root->roots[key.root];
207 if (dispatch_node == QPS_HANDLE_NULL) {
208 return map->root->is_nullable ? QPS_BITMAP_NULL : QPS_BITMAP_0;
209 }
210
211 dispatch = qps_pg_deref(map->qps, dispatch_node);
212 leaf_node = (*dispatch)[key.dispatch].node;
213 if (leaf_node == 0) {
214 return map->root->is_nullable ? QPS_BITMAP_NULL : QPS_BITMAP_0;
215 }
216
217 leaf = qps_pg_deref(map->qps, leaf_node);
218 if (map->root->is_nullable) {
219 uint64_t word = leaf[key.word_null];
220 word >>= (key.bit_null * 2);
221 if (!(word & 0x2)) {
222 return QPS_BITMAP_NULL;
223 }
224 return (qps_bitmap_state_t)(word & 0x1);
225 } else {
226 uint64_t word = leaf[key.word];
227 word >>= key.bit;
228 return (qps_bitmap_state_t)(word & 0x1);
229 }
230}

232qps_bitmap_state_t qps_bitmap_set(qps_bitmap_t *map, uint32_t row)
233{
234 qps_bitmap_key_t key = { .key = row };
235 qps_bitmap_dispatch_t *dispatch;
236 uint64_t *leaf;
237
238 map->bitmap_gen++;
239 dispatch = w_deref_dispatch(map, key, true);
240 leaf = w_deref_leaf(map, &dispatch, key, true);
241
242 if (map->root->is_nullable) {
243 uint64_t word = leaf[key.word_null];
244 word >>= (key.bit_null * 2);
245 if (!(word & 0x2)) {
246 word = (UINT64_C(0x3) << (key.bit_null * 2));
247 leaf[key.word_null] |= word;
248 (*dispatch)[key.dispatch].active_bits++;
249 return QPS_BITMAP_NULL;
250 } else
251 if (!(word & 0x1)) {
252 word = (UINT64_C(0x3) << (key.bit_null * 2));
253 leaf[key.word_null] |= word;
254 return QPS_BITMAP_0;
255 }
256 return QPS_BITMAP_1;
257 } else {
258 uint64_t word = UINT64_C(1) << key.bit;
259 if (!(leaf[key.word] & word)) {
260 leaf[key.word] |= word;
261 (*dispatch)[key.dispatch].active_bits++;
262 return QPS_BITMAP_0;
263 }
264 return QPS_BITMAP_1;
265 }
266}

268qps_bitmap_state_t qps_bitmap_reset(qps_bitmap_t *map, uint32_t row)
269{
270 qps_bitmap_key_t key = { .key = row };
271 qps_bitmap_dispatch_t *dispatch;
272 uint64_t *leaf;
273
274 map->bitmap_gen++;
275 qps_hptr_deref(map->qps, &map->root_cache);
276 dispatch = w_deref_dispatch(map, key, map->root->is_nullable);
277 leaf = w_deref_leaf(map, &dispatch, key, map->root->is_nullable);
278 if (leaf == NULL) {
279 return QPS_BITMAP_0;
280 }
281
282 if (map->root->is_nullable) {
283 uint64_t word = leaf[key.word_null];
284 uint64_t mask;
285 word >>= (key.bit_null * 2);
286 if (!(word & 0x2)) {
287 mask = (UINT64_C(0x3) << (key.bit_null * 2));
288 word = (UINT64_C(0x2) << (key.bit_null * 2));
289 leaf[key.word_null] &= ~mask;
290 leaf[key.word_null] |= word;
291 (*dispatch)[key.dispatch].active_bits++;
292 return QPS_BITMAP_NULL;
293 } else
294 if ((word & 0x1)) {
295 mask = (UINT64_C(0x3) << (key.bit_null * 2));
296 word = (UINT64_C(0x2) << (key.bit_null * 2));
297 leaf[key.word_null] &= ~mask;
298 leaf[key.word_null] |= word;
299 return QPS_BITMAP_1;
300 }
301 return QPS_BITMAP_0;
302 } else {
303 uint64_t word = UINT64_C(1) << key.bit;
304 if ((leaf[key.word] & word)) {
305 leaf[key.word] &= ~word;
306 (*dispatch)[key.dispatch].active_bits--;
307 if ((*dispatch)[key.dispatch].active_bits == 0) {
308 delete_leaf(map, key);
309 }
310 return QPS_BITMAP_1;
311 }
312 return QPS_BITMAP_0;
313 }
314}

316qps_bitmap_state_t qps_bitmap_remove(qps_bitmap_t *map, uint32_t row)
317{
318 qps_bitmap_key_t key = { .key = row };
319 qps_bitmap_dispatch_t *dispatch;
320 uint64_t *leaf;
321
322 map->bitmap_gen++;
323 dispatch = w_deref_dispatch(map, key, false);
324 leaf = w_deref_leaf(map, &dispatch, key, false);
325 if (leaf == NULL) {
326 return map->root->is_nullable ? QPS_BITMAP_NULL : QPS_BITMAP_0;
327 }
328
329 if (map->root->is_nullable) {
330 qps_bitmap_state_t previous;
331 uint64_t word = leaf[key.word_null];
332 uint64_t mask;
333 word >>= (key.bit_null * 2);
334 if (!(word & 0x2)) {
335 return QPS_BITMAP_NULL;
336 } else {
337 mask = (UINT64_C(0x3) << (key.bit_null * 2));
338 leaf[key.word_null] &= ~mask;
339 (*dispatch)[key.dispatch].active_bits--;
340 previous = (qps_bitmap_state_t)(word & 0x1);
341 if ((*dispatch)[key.dispatch].active_bits == 0) {
342 delete_leaf(map, key);
343 }
344 return previous;
345 }
346 } else {
347 uint64_t word = UINT64_C(1) << key.bit;
348 if ((leaf[key.word] & word)) {
349 leaf[key.word] &= ~word;
350 (*dispatch)[key.dispatch].active_bits--;
351 if ((*dispatch)[key.dispatch].active_bits == 0) {
352 delete_leaf(map, key);
353 }
354 return QPS_BITMAP_1;
355 }
356 return QPS_BITMAP_0;
357 }

361void qps_bitmap_compute_stats(qps_bitmap_t *map, size_t *_memory,
362 uint32_t *_entries, uint32_t *_slots)
363{
364 size_t memory = 0;
365 uint32_t entries = 0;
366 uint32_t slots = 0;
367
368 qps_hptr_deref(map->qps, &map->root_cache);
369 for (int i = 0; i < QPS_BITMAP_ROOTS; i++) {
370 if (map->root->roots[i]) {
371 const qps_bitmap_dispatch_t *dispatch;
372
373 memory += 3 * QPS_PAGE_SIZE;
374 dispatch = qps_pg_deref(map->qps, map->root->roots[i]);
375
376 for (int j = 0; j < QPS_BITMAP_DISPATCH; j++) {
377 if ((*dispatch)[j].node) {
378 if (map->root->is_nullable) {
379 memory += 2 * QPS_PAGE_SIZE;
380 } else {
381 memory += QPS_PAGE_SIZE;
382 }
383 entries += (*dispatch)[j].active_bits;
384 slots += QPS_BITMAP_LEAF;
385 }
386 }
387 }
388 }
389
390 *_memory = memory;
391 *_entries = entries;
392 *_slots = slots;
393}

398void qps_bitmap_get_qps_roots(qps_bitmap_t *map, qps_roots_t *roots)
399{
400 qps_hptr_deref(map->qps, &map->root_cache);
401 for (int i = 0; i < QPS_BITMAP_ROOTS; i++) {
402 const qps_bitmap_dispatch_t *dispatch;
403

405 continue;
406 }
407 qv_append(&roots->pages, map->root->roots[i]);
408 dispatch = qps_pg_deref(map->qps, map->root->roots[i]);
409 for (int j = 0; j < QPS_BITMAP_DISPATCH; j++) {
410 if ((*dispatch)[j].node == 0) {
411 continue;
412 }
413 qv_append(&roots->pages, (*dispatch)[j].node);
414 }
415 }
416 qv_append(&roots->handles, map->root_cache.handle);
417}
