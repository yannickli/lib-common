19/** This file is used to be loaded as LD_PRELOAD when running doxygen.
20 *
21 * This is required because doxygen, which uses libclang, uses
22 * clang_parseTranslationUnit() with an empty `source_filename` and the files
23 * are passed in `unsaved_files`.
24 *
25 * Due to a bug in libclang, using `clang_parseTranslationUnit()` on a `.blk`
26 * or `.blkk` with the file as `unsaved_files` returns an error
27 * (CXError_ASTReadError).
28 *
29 * The solution is to use `source_filename` in that case instead.
30 */
31
32#include <lib-common/core.h>
33
34#include <clang-c/Index.h>

37CXTranslationUnit clang_parseTranslationUnit(
38 CXIndex CIdx, const char *source_filename,
39 const char * const *command_line_args,
40 int num_command_line_args, struct CXUnsavedFile *unsaved_files,
41 unsigned num_unsaved_files, unsigned options)
42{
43 CXTranslationUnit out;
44 enum CXErrorCode err_code;
45
46 p_clear(&out, 1);
47
48 /* First, try with the given arguments with clang_parseTranslationUnit2()
49 * to retrieve the error code. */
50 err_code = clang_parseTranslationUnit2(
51 CIdx, source_filename, command_line_args,
52 num_command_line_args, unsaved_files, num_unsaved_files,
53 options, &out);
54
55 if (err_code == CXError_Success) {
56 /* If this is ok, return the translation unit directly. */
57 return out;
58 }
59
60 /* If we have only one file and no source_filname, we can try with the
61 * file as a source_filename instead. */
62 if (num_unsaved_files >= 1 && source_filename == NULL) {
63 const char *filename = unsaved_files[0].Filename;
64 struct stat s;
65
66 /* File exists, try with the explicit filename instead. */
67 if (stat(filename, &s) == 0) {
68 source_filename = filename;
69 num_unsaved_files--;
70 unsaved_files++;
71
72 p_clear(&out, 1);
73 err_code = clang_parseTranslationUnit2(
74 CIdx, source_filename, command_line_args,
75 num_command_line_args, unsaved_files, num_unsaved_files,
76 options, &out);
77 }
78 }
79
80 if (err_code != CXError_Success) {
81 const char *filename;
82
83 if (source_filename == NULL && num_unsaved_files >= 1) {
84 /* If no source filename, use the first unsaved file instead. */
85 filename = unsaved_files[0].Filename;
86 } else {
87 /* Otherwise, use the source filename. */
88 filename = source_filename;
89 }
90
91 fprintf(stderr, "clang - failed to parse translation unit for file "
92 "`%s` (error code %d)\n", filename, err_code);
93 }
94
95 return out;
96}
