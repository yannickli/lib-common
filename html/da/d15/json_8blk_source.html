235/* Format an error of type: "expected `foo', got `bar'. The len of the error
236 * string is:
237 * len(the bad token)
238 * + len(expected expression)
239 * + len("expected ?, got `?'") <=> 17
240 * + 1 // for the \0
241 */
242static int
243rjerror_exp(iop_json_lex_t *ll, int err, const char *expected, int len)
244{
245 char *dst;
246 int buflen = 0, pos = 0, pslen;
247 bool overflow = false;
248
249 __ps_clip_at(&ll->s_ps, PS->p);
250 pslen = ps_len(&ll->s_ps);
251
252 if (pslen > JERROR_MAXLEN) {
253 /* len = MAXLEN + sizeof("...") + sizeof("\"") */
254 pslen = JERROR_MAXLEN;
255 buflen = 3;
256 if (ll->s_ps.s[0] == '"')
257 buflen++;
258 overflow = true;
259 }
260
261 if (len < 0)
262 len = strlen(expected);
263
264 buflen += pslen + len + 17 + 1;
265 dst = ll->err_str = mp_new_raw(ll->mp, char, buflen);
266
267 pos += pstrcpy(dst, buflen, "expected ");
268 pos += pstrcpy(dst + pos, buflen - pos, expected);
269 pos += pstrcpy(dst + pos, buflen - pos, ", got `");
270 pos += pstrcpymem(dst + pos, buflen - pos, ll->s_ps.s, pslen);
271 if (overflow && ll->s_ps.s[0] == '"') {
272 pos += pstrcpy(dst + pos, buflen - pos, "...\"");
273 } else
274 if (overflow) {
275 pos += pstrcpy(dst + pos, buflen - pos, "...");
276 }
277 pos += pstrcpy(dst + pos, buflen - pos, "'");
278 assert((buflen - pos) == 1);
279
280 return RJERROR(err);
281}

283static void jerror_strncpy(iop_json_lex_t *ll, const char *str, int len)
284{
285 if (len > JERROR_MAXLEN) {
286 /* len = MAXLEN + strlen("...") + strlen("\"") + 1 */
287 ll->err_str = mp_new_raw(ll->mp, char, JERROR_MAXLEN + 5);
288 memcpy(ll->err_str, str, JERROR_MAXLEN);
289 ll->err_str[JERROR_MAXLEN] = '.';
290 ll->err_str[JERROR_MAXLEN + 1] = '.';
291 ll->err_str[JERROR_MAXLEN + 2] = '.';
292 /* If str is a string we close it */
293 if (ll->err_str[0] == '"') {
294 ll->err_str[JERROR_MAXLEN + 3] = '"';
295 ll->err_str[JERROR_MAXLEN + 4] = '\0';
296 } else {
297 ll->err_str[JERROR_MAXLEN + 3] = '\0';
298 }
299 } else {
300 ll->err_str = mp_dupz(ll->mp, str, len);
301 }
302}

350static int iop_json_lex_token(iop_json_lex_t *ll)
351{
352 const char *start = PS->s;
353
354 while (!ps_done(PS) && (isalnum(READC()) || READC() == '_'))
355 SKIP(1);
356
357 if (HAS(1)) {
358 /* Check for the authorised tokens after an identifier */
359 switch (READC()) {
360 case ':': case '=': case ',': case ';': case '.':
361 case '[': case ']': case '{': case '}':
362 break;
363 default:
364 if (!isspace(READC())) {
365 SKIP(1);
366 return RJERROR_WARG(IOP_JERR_BAD_IDENT);
367 }
368 }
369 }
370
371 sb_set(&ll->ctx->b, start, PS->s - start);
372 return IOP_JSON_IDENT;
373}

375static int iop_json_lex_number(iop_json_lex_t *ll)
376{
377 unsigned int pos;
378 bool is_d_err;
379 int i_res;
380 uint64_t i;
381 double d;
382 const char *p, *pd, *pi;
383 unsigned int d_count, i_count;
384
385 /* we try to detect where the integer ends (i.e. where strto* functions
386 * will stop reading). it does not matter if we go too far (e.g. with
387 * '2+3+4' it will read the whole expression and not just '2' */
388 for (pos = 0; pos < ps_len(PS); pos++) {
389 if (isalnum(READAT(pos)))
390 continue;
391 switch (READAT(pos)) {
392 case '.': case '+': case '-':
393 continue;
394 }
395 break;
396 }
397
398 /* in the unlikely case that the integer reaches the end of the PS,
399 * we need a null-terminated buffer to pass to the strto* functions */
400 if (unlikely(pos == ps_len(PS))) {
401 sb_set(&ll->ctx->b, PS->b, ps_len(PS));
402 p = ll->ctx->b.data;
403 } else {
404 p = PS->s;
405 }
406
407 errno = 0;
408 d = strtod_allow_subnormal(p, (char **)&pd);
409 is_d_err = errno;
410 d_count = pd - p;
411
412 if (*p == '-') {
413 ll->ctx->is_signed = true;
414 i_res = strtoll_ext(p, (int64_t *)&i, &pi, 0);
415 } else {
416 ll->ctx->is_signed = false;
417 i_res = strtoull_ext(p, &i, &pi, 0);
418 }
419
420 i_count = pi - p;
421
422 /* rationale: if we parse to the same point, it was an integer
423 * and double _may_ lose precision on 64bits ints
424 * and if i_count is more important than d_count in case of error
425 * we return an integer error */
426 if (d_count <= i_count) {
427 is_d_err = true;
428 }
429
430 switch (is_d_err << 1 | (i_res < 0)) {
431 case 0:
432 /* FALLTHROUGH */
433
434 case 1:
435 if (unlikely(d_count > pos)) {
436 e_trace(0, "strtod read further than us: %*pM vs %*pM",
437 d_count, PS->s, pos, PS->s);
438 }
439 ll->ctx->u.d = d;
440 SKIP(d_count);
441 return IOP_JSON_DOUBLE;
442
443 case 2:
444 if (unlikely(i_count > pos)) {
445 e_trace(0, "strtoxll_ext read further than us: %*pM vs %*pM",
446 i_count , PS->s, pos, PS->s);
447 }
448 ll->ctx->u.i = i;
449 SKIP(i_count);
450 return IOP_JSON_INTEGER;
451
452 default:
453 if (errno == ERANGE)
454 return JERROR_WARG(IOP_JERR_TOO_BIG_INT, pos);
455 if (errno == EDOM)
456 return JERROR_WARG(IOP_JERR_BAD_INT_EXT, pos);
457 }
458 return JERROR_WARG(IOP_JERR_PARSE_NUM, pos);
459}

461static int iop_json_lex_enum(iop_json_lex_t *ll, int terminator,
462 const iop_field_t *fdesc)
463{
464 bool found;
465 const char *end;
466 int len;
467 lstr_t s;
468
469 end = (const char *)memchr(PS->p, terminator, ps_len(PS));
470 if (!end)
471 return JERROR(IOP_JERR_UNCLOSED_STRING);
472 len = end - PS->s;
473 s = LSTR_INIT_V(PS->s, len);
474
475 ll->ctx->u.i = iop_enum_from_lstr_desc(fdesc->u1.en_desc, s, &found);
476 if (!found) {
477 const char *q;
478
479 errno = 0;
480 /* FIXME: for strict enums, only integer values that matches enum
481 * values should be allowed. */
482 ll->ctx->u.i = strtoull(s.s, &q, 0);
483
484 if (errno || q != end) {
485 return JERROR_WARG(IOP_JERR_ENUM_VALUE, len);
486 }
487 }
488
489 SKIP(len + 1);
490 return IOP_JSON_INTEGER;
491}

493static int iop_json_lex_expr(iop_json_lex_t *ll, const iop_field_t *fdesc)
494{
495 uint64_t num = 0;
496 int type = IOP_JSON_INTEGER;
497
498 if (READC() == '+')
499 SKIP(1);
500
501 for(;;) {
502 int c;
503
504 while (!ps_done(PS) && isspace(READC())) {
505 if (EATC() == '\n')
506 NEWLINE();
507 }
508 switch (c = READC()) {
509
510 case '<': case '>':
511 SKIP(1);
512 if (HAS(1) && READC() == c) {
513 switch(c) {
514 case '<': c = CF_OP_LSHIFT; break;
515 case '>': c = CF_OP_RSHIFT; break;
516 default : break;
517 }
518 } else {
519 return RJERROR_EXP("a bitwise shift operator");
520 }
521 /* FALLTHROUGH */
522 case '-': case '+': case '/': case '~':
523 case '&': case '|': case '%': case '^':
524 case '(': case ')': case '*':
525 if (c == '*' && HAS(2) && READAT(1) == '*') {
526 SKIP(1);
527 c = CF_OP_EXP;
528 }
529 if (c == '-' && iop_cfolder_empty(ll->cfolder)) {
530 goto feed_number;
531 }
532 e_trace(1, "feed operator %c", c);
533 if (iop_cfolder_feed_operator(ll->cfolder, c) < 0)
534 return RJERROR_WARG(IOP_JERR_PARSE_NUM);
535 SKIP(1);
536 break;
537
538 case '.': case '0' ... '9':
539 feed_number:
540 type = RETHROW(iop_json_lex_number(ll));
541 if (type == IOP_JSON_DOUBLE) {
542 if (!iop_cfolder_empty(ll->cfolder)) {
543 e_warning("double value in an expression");
544 return RJERROR_EXP("a valid integer expression");
545 }
546 e_trace(1, "single double value %f", ll->ctx->u.d);
547 return IOP_JSON_DOUBLE;
548 }
549 assert (type == IOP_JSON_INTEGER);
550 e_trace(1, "feed number %jd", ll->ctx->u.i);
551 if (iop_cfolder_feed_number(ll->cfolder, ll->ctx->u.i,
552 ll->ctx->is_signed) < 0)
553 {
554 return RJERROR_WARG(IOP_JERR_PARSE_NUM);
555 }
556 break;
557
558 case '\'': case '"':
559 if (!fdesc || fdesc->type != IOP_T_ENUM) {
560 return RJERROR_EXP("an integer");
561 }
562 RETHROW(iop_json_lex_enum(ll, EATC(), fdesc));
563 e_trace(1, "feed number %jd", ll->ctx->u.i);
564 if (iop_cfolder_feed_number(ll->cfolder, ll->ctx->u.i,
565 (ll->ctx->u.i < 0)) < 0)
566 {
567 return RJERROR_WARG(IOP_JERR_PARSE_NUM);
568 }
569 break;
570
571 default:
572 goto result;
573 }
574 }
575
576 result:
577 /* Let's try to get a result */
578 if (iop_cfolder_get_result(ll->cfolder, &num, NULL) < 0)
579 return RJERROR_WARG(IOP_JERR_PARSE_NUM);
580
581 ll->ctx->u.i = num;
582 e_trace(1, "-> result %jd", ll->ctx->u.i);
583
584 return IOP_JSON_INTEGER;
585}

587static int iop_json_lex_char(iop_json_lex_t *ll, int terminator)
588{
589 if (!HAS(2)) {
590 return RJERROR_WARG(IOP_JERR_EXP_SMTH);
591 }
592 if (READAT(1) == terminator) {
593 /* single character */
594 ll->ctx->u.i = READAT(0);
595 SKIP(2);
596 return IOP_JSON_INTEGER;
597 } else
598 /* escape sequence */
599 if (READAT(0) == '\\') {
600#define OCTAL -1
601#define UNICODE -2
602#define HEXADECIMAL -3
603 static int escaped[256] = {
604 ['a'] = '\a',
605 ['b'] = '\b',
606 ['e'] = '\e',
607 ['t'] = '\t',
608 ['n'] = '\n',
609 ['v'] = '\v',
610 ['f'] = '\f',
611 ['r'] = '\r',
612 ['\\'] = '\\',
613 ['"'] = '\"',
614 ['\''] = '\'',
615 ['0'] = OCTAL,
616 ['1'] = OCTAL,
617 ['2'] = OCTAL,
618 ['u'] = UNICODE,
619 ['x'] = HEXADECIMAL,
620 };
621 int c = escaped[READAT(1)];
622
623 if (!HAS(2))
624 return RJERROR_WARG(IOP_JERR_EXP_SMTH);
625 switch (c) {
626 int a, b;
627 case OCTAL:
628 if (HAS(4)
629 && READAT(2) >= '0' && READAT(2) <= '7'
630 && READAT(3) >= '0' && READAT(3) <= '7')
631 {
632 ll->ctx->u.i = ((READAT(1) - '0') << 6) |
633 ((READAT(2) - '0') << 3) | (READAT(3) - '0');
634 SKIP(2);
635 } else
636 /* null */
637 if (READAT(1) == '0') {
638 ll->ctx->u.i = '\0';
639 }
640 break;
641 case HEXADECIMAL:
642 if (HAS(4) && (a = hexdecode(PS->s + 2)) >= 0) {
643 ll->ctx->u.i = a;
644 SKIP(2);
645 }
646 break;
647 case UNICODE:
648 if (HAS(6)
649 && (a = hexdecode(PS->s + 2)) >= 0
650 && (b = hexdecode(PS->s + 4)) >= 0)
651 {
652 ll->ctx->u.i = a << 8 | b;
653 SKIP(4);
654 }
655 break;
656 case 0:
657 return RJERROR_WARG(IOP_JERR_EXP_SMTH);
658 default:
659 ll->ctx->u.i = c;
660 break;
661 }
662 SKIP(2);
663 if (READC() != terminator) {
664 return RJERROR_WARG(IOP_JERR_EXP_SMTH);
665 }
666 SKIP(1);
667 return IOP_JSON_INTEGER;
668 } else {
669 return RJERROR_WARG(IOP_JERR_EXP_SMTH);
670 }
671}

673static int iop_json_parse_str(pstream_t *ps, sb_t *buf, int *line, int *col,
674 int terminator)
675{
676 sb_reset(buf);
677
678 for (;;) {
679 for (unsigned i = 0; i < ps_len(ps); i++) {
680 if (ps->b[i] == '\n') {
681 return IOP_JERR_UNCLOSED_STRING;
682 } else
683 if (ps->b[i] == '\\') {
684 sb_add(buf, ps->p, i);
685 PS_SKIP(ps, *col, i);
686 goto parse_bslash;
687 } else
688 if (ps->b[i] == terminator) {
689 sb_add(buf, ps->p, i);
690 PS_SKIP(ps, *col, i + 1);
691 return IOP_JSON_STRING;
692 }
693 }
694 return IOP_JERR_UNCLOSED_STRING;
695
696 parse_bslash:
697 if (parse_backslash(ps, buf, line, col) < 0) {
698 return IOP_JERR_EXP_SMTH;
699 }
700 }
701}

703static int iop_json_lex_str(iop_json_lex_t *ll, int terminator)
704{
705 int res;
706
707 res = iop_json_parse_str(PS, &ll->ctx->b, &ll->ctx->line, &ll->ctx->col,
708 terminator);
709 if (res == IOP_JERR_EXP_SMTH) {
710 return RJERROR_WARG(IOP_JERR_EXP_SMTH);
711 } else
712 if (res < 0) {
713 return JERROR(res);
714 }
715
716 return res;
717}

719static int iop_json_lex(iop_json_lex_t *ll, const iop_field_t *fdesc)
720{
721 if (ll->peek >= 0) {
722 int res = ll->peek;
723 ll->peek = -1;
724
725 /* copy the peeked context */
726 sb_setsb(&ll->cur_ctx.b, &ll->peeked_ctx.b);
727 ll->cur_ctx.line = ll->peeked_ctx.line;
728 ll->cur_ctx.col = ll->peeked_ctx.col;
729 ll->cur_ctx.u = ll->peeked_ctx.u;
730 ll->cur_ctx.is_signed = ll->peeked_ctx.is_signed;
731
732 return res;
733 }
734
735 trackback:
736 while (!ps_done(PS) && isspace(READC())) {
737 if (EATC() == '\n')
738 NEWLINE();
739 }
740
741 if (ps_done(PS))
742 return 0;
743
744 STORECTX();
745 switch (READC()) {
746 int c;
747 case '=': SKIP(1); return ':';
748 case ';': SKIP(1); return ',';
749 case ':': case ',':
750 case '{': case '}':
751 case '[': case ']':
752 case '@':
753 return EATC();
754
755 case 'c': /* string character prefix */
756 c = READAT(1);
757 if (c == '\'' || c == '"') {
758 SKIP(2);
759 return iop_json_lex_char(ll, c);
760 }
761 /* FALLTHROUGH */
762 case 'a' ... 'b': case 'd' ... 'z': case 'A' ... 'Z':
763 return iop_json_lex_token(ll);
764
765 case '.':
766 if (!(HAS(2) && READAT(1) >= '0' && READAT(1) <= '9'))
767 return EATC();
768 /* FALLTHROUGH */
769 case '-': case '+': case '~':
770 case '0' ... '9':
771 case '(': case ')':
772 return iop_json_lex_expr(ll, fdesc);
773
774 case '\'': case '"':
775 if (fdesc && fdesc->type == IOP_T_ENUM) {
776 return iop_json_lex_expr(ll, fdesc);
777 }
778 return iop_json_lex_str(ll, EATC());
779
780 case '/':
781 if (!HAS(2))
782 return JERROR_WARG(IOP_JERR_EXP_SMTH, 1);
783 SKIP(1);
784 c = EATC();
785
786 if (c == '*') {
787 PS_CHECK(iop_json_skip_comment(ll));
788 goto trackback;
789 } else
790 if (c != '/') {
791 return RJERROR_EXP("/");
792 }
793 /* FALLTHROUGH */
794 case '#':
795 if (ps_skip_afterchr(PS, '\n') < 0) {
796 ll->ctx->col += ps_len(PS);
797 ps_skip(PS, ps_len(PS));
798 return 0;
799 }
800 NEWLINE();
801 goto trackback;
802 default:
803 return JERROR_WARG(IOP_JERR_BAD_TOKEN, 1);
804 }
805}

807static int iop_json_lex_peek(iop_json_lex_t *ll, const iop_field_t *fdesc)
808{
809 if (ll->peek < 0) {
810 /* Change of parser context */
811 ll->peeked_ctx.line = ll->cur_ctx.line;
812 ll->peeked_ctx.col = ll->cur_ctx.col;
813 ll->ctx = &ll->peeked_ctx;
814
815 /* Peek the next token */
816 ll->peek = iop_json_lex(ll, fdesc);
817
818 /* Restore parser context */
819 ll->ctx = &ll->cur_ctx;
820 }
821 return ll->peek;
822}

979static int skip_val(iop_json_lex_t *ll, bool in_arr)
980{
981 switch (PS_CHECK(iop_json_lex(ll, NULL))) {
982 case IOP_JSON_IDENT:
983 if (IS_TRUE() || IS_FALSE() || IS_NULL())
984 return 0;
985 return RJERROR_WARG(IOP_JERR_BAD_VALUE);
986
987 case IOP_JSON_STRING:
988 case IOP_JSON_DOUBLE:
989 case IOP_JSON_INTEGER:
990 return 0;
991
992 case '[':
993 return in_arr ? -1 : unpack_arr(ll, NULL, NULL);
994
995 case '{':
996 return unpack_struct(ll, NULL, NULL, false);
997
998 case '@':
999 if (ps_startswithlstr(PS, LSTR("include("))) {
1000 return unpack_val_file_inclusion(ll, NULL, NULL);
1001 } else {
1002 return unpack_struct(ll, NULL, NULL, true);
1003 }
1004
1005 case '.':
1006 return unpack_union(ll, NULL, NULL, true);
1007
1008 case IOP_JSON_EOF:
1009 return RJERROR_WARG(IOP_JERR_EXP_SMTH);
1010
1011 default:
1012 return RJERROR_WARG(IOP_JERR_BAD_TOKEN);
1013 }
1014}

1016static int unpack_val_file_inclusion(iop_json_lex_t *ll,
1017 const iop_field_t *fdesc, void *value)
1018{
1019 char prefix[PATH_MAX];
1020 char path[PATH_MAX];
1021 char relative_path[PATH_MAX];
1022 lstr_t path_lstr;
1023
1024 if (!filenames_g.len) {
1025 return JERROR_VARIOUS("file inclusion only supported when parsing a "
1026 "file");
1027 }
1028
1029 SKIP(strlen("include("));
1030
1031 switch (READC()) {
1032 case '\'': case '"':
1033 STORECTX();
1034 RETHROW(iop_json_lex_str(ll, EATC()));
1035 break;
1036 default:
1037 return JERROR_WARG(IOP_JERR_BAD_TOKEN, 1);
1038 }
1039
1040 if (!HAS(1)) {
1041 RESTORECTX();
1042 return JERROR_WARG(IOP_JERR_EXP_SMTH, 1);
1043 }
1044 if (READC() != ')') {
1045 STORECTX();
1046 IGNORE(EATC());
1047 return RJERROR_EXP(")");
1048 }
1049
1050 if (!fdesc) {
1051 IGNORE(EATC());
1052 return 0;
1053 }
1054
1055 path_dirname(prefix, sizeof(prefix), tab_last(&filenames_g)->s);
1056 if (path_extend(path, prefix, "%*pM", SB_FMT_ARG(&ll->ctx->b)) < 0) {
1057 return JERROR_VARIOUS("cannot get fullpath of `%*pM`",
1058 SB_FMT_ARG(&ll->ctx->b));
1059 }
1060
1061 path_lstr = LSTR(path);
1062 tab_for_each_entry(s, &filenames_g) {
1063 if (lstr_equal(s, path_lstr)) {
1064 RESTORECTX();
1065 return JERROR_VARIOUS("infinite recursion detected in includes");
1066 }
1067 }
1068
1069 if (subfiles_g) {
1070 lstr_t basedir;
1071 iop_json_subfile__t subfile;
1072 int len;
1073
1074 if (relative_dirs_g.len) {
1075 basedir = *tab_last(&relative_dirs_g);
1076 } else {
1077 basedir = LSTR_EMPTY_V;
1078 }
1079
1080 snprintf(relative_path, PATH_MAX, "%*pM%*pM",
1081 LSTR_FMT_ARG(basedir), SB_FMT_ARG(&ll->ctx->b));
1082 len = path_simplify2(relative_path, false);
1083
1084 if (len < 0) {
1085 return JERROR_VARIOUS("cannot get relative path of `%*pM`",
1086 SB_FMT_ARG(&ll->ctx->b));
1087 }
1088
1089 p_clear(&subfile, 1);
1090 subfile.file_path = t_lstr_dups(relative_path, len);
1091 subfile.iop_path = t_get_current_field_path(&field_paths_g);
1092 qv_append(subfiles_g, subfile);
1093 }
1094
1095 switch (fdesc->type) {
1096 case IOP_T_STRING: case IOP_T_DATA: case IOP_T_XML: {
1097 sb_t content;
1098
1099 mp_sb_init(ll->mp, &content, 1024);
1100 if (sb_read_file(&content, path) < 0) {
1101 RESTORECTX();
1102 return JERROR_VARIOUS("cannot read file `%s`: %m", path);
1103 }
1104
1105 *(lstr_t *)value = LSTR_SB_V(&content);
1106 } break;
1107
1108 case IOP_T_UNION: case IOP_T_STRUCT: {
1109 SB_1k(err);
1110 char relative_dir[PATH_MAX];
1111 int res;
1112
1113 if (subfiles_g) {
1114 path_dirname(relative_dir, sizeof(relative_dir), relative_path);
1115 if (*relative_dir) {
1116 pstrcat(relative_dir, sizeof(relative_dir), "/");
1117 }
1118 qv_append(&relative_dirs_g, LSTR(relative_dir));
1119 }
1120
1121 if (iop_field_is_class(fdesc)) {
1122 *(void **)value = NULL;
1123 res = t_iop_junpack_ptr_file(path, fdesc->u1.st_desc, value,
1124 ll->flags, NULL, &err);
1125 } else {
1126 res = t_iop_junpack_file(path, fdesc->u1.st_desc, value,
1127 ll->flags, NULL, &err);
1128 }
1129
1130 if (subfiles_g) {
1131 qv_remove_last(&relative_dirs_g);
1132 }
1133
1134 if (res < 0) {
1135 RESTORECTX();
1136 return JERROR_VARIOUS("cannot unpack file `%s`: %*pM",
1137 path, SB_FMT_ARG(&err));
1138 }
1139 } break;
1140
1141 default:
1142 RESTORECTX();
1143 return JERROR_VARIOUS("file inclusion not supported for %s fields",
1144 iop_type_get_string_desc(fdesc->type));
1145 }
1146
1147 IGNORE(EATC());
1148 return 0;
1149}

1151static int unpack_val(iop_json_lex_t *ll, const iop_field_t *fdesc,
1152 void *value, bool in_arr)
1153{
1154 if (fdesc->type == IOP_T_VOID) {
1155 e_named_trace(3, "iop/json/unpacker", "dropped value into void field "
1156 "`%*pM`", LSTR_FMT_ARG(fdesc->name));
1157 return skip_val(ll, in_arr);
1158 }
1159
1160 if (!in_arr && fdesc->repeat == IOP_R_REPEATED) {
1161 if (PS_CHECK(iop_json_lex_peek(ll, NULL)) == '[') {
1162 iop_json_lex(ll, NULL);
1163 return unpack_arr(ll, fdesc, value);
1164 } else {
1165 /* We support single value in repeated fields without brackets. */
1166 lstr_t *arr = value;
1167
1168 arr->len = 1;
1169 arr->data = mp_imalloc(ll->mp, fdesc->size, 8, 0);
1170
1171 return unpack_val(ll, fdesc, arr->data, true);
1172 }
1173 }
1174
1175 if (iop_field_is_reference(fdesc)) {
1176 /* reference fields must be dereferenced */
1177 value = iop_field_ptr_alloc(ll->mp, fdesc, value);
1178 }
1179
1180 switch (PS_CHECK(iop_json_lex(ll, fdesc))) {
1181 case IOP_JSON_IDENT:
1182 if (IS_TRUE()) {
1183 ll->ctx->u.i = 1;
1184 goto integer;
1185 } else
1186 if (IS_FALSE()) {
1187 ll->ctx->u.i = 0;
1188 goto integer;
1189 } else
1190 if (IS_NULL()) {
1191 switch (fdesc->type) {
1192 case IOP_T_STRING:
1193 case IOP_T_XML:
1194 case IOP_T_DATA:
1195 *(lstr_t *)value = LSTR_NULL_V;
1196 return 0;
1197 default:
1198 return RJERROR_EXP_TYPE(fdesc->type);
1199 }
1200 }
1201 return RJERROR_WARG(IOP_JERR_BAD_VALUE);
1202
1203 case IOP_JSON_STRING:
1204 switch (fdesc->type) {
1205 const char *q;
1206 lstr_t *data;
1207
1208 case IOP_T_I8: case IOP_T_U8:
1209 case IOP_T_I16: case IOP_T_U16:
1210 case IOP_T_I32: case IOP_T_U32:
1211 case IOP_T_I64: case IOP_T_U64:
1212 case IOP_T_BOOL:
1213 if (IS_TRUE()) {
1214 ll->ctx->u.i = 1;
1215 goto integer;
1216 } else
1217 if (IS_FALSE()) {
1218 ll->ctx->u.i = 0;
1219 goto integer;
1220 }
1221 errno = 0;
1222 switch (fdesc->type) {
1223 case IOP_T_U8:
1224 case IOP_T_U16:
1225 case IOP_T_U32:
1226 case IOP_T_U64:
1227 ll->ctx->u.i = strtoull(ll->ctx->b.data, &q, 0);
1228 break;
1229
1230 default:
1231 ll->ctx->u.i = strtoll(ll->ctx->b.data, &q, 0);
1232 break;
1233 }
1234 if (errno || q != (const char *)ll->ctx->b.data + ll->ctx->b.len)
1235 return RJERROR_WARG(IOP_JERR_PARSE_NUM);
1236 goto integer;
1237
1238 case IOP_T_DOUBLE:
1239 errno = 0;
1240 ll->ctx->u.d = strtod_allow_subnormal(ll->ctx->b.data,
1241 (char **)&q);
1242 if (errno || q != (const char *)ll->ctx->b.data + ll->ctx->b.len)
1243 return RJERROR_WARG(IOP_JERR_PARSE_NUM);
1244 goto do_double;
1245
1246 case IOP_T_DATA:
1247 data = (lstr_t *)value;
1248 if (ll->ctx->b.len == 0) {
1249 data->data = mp_new_raw(ll->mp, char, 1);
1250 data->len = 0;
1251 } else {
1252 sb_t sb;
1253 int blen = DIV_ROUND_UP(ll->ctx->b.len * 3, 4);
1254 char *buf = mp_new_raw(ll->mp, char, blen + 1);
1255
1256 sb_init_full(&sb, buf, 0, blen + 1, &mem_pool_static);
1257 if (sb_add_unb64(&sb, ll->ctx->b.data, ll->ctx->b.len)) {
1258 mp_delete(ll->mp, &buf);
1259 return RJERROR_WARG(IOP_JERR_BAD_VALUE);
1260 }
1261 data->data = buf;
1262 data->len = sb.len;
1263 }
1264 return 0;
1265
1266 case IOP_T_STRING:
1267 case IOP_T_XML:
1268 data = (lstr_t *)value;
1269 data->data = mp_dupz(ll->mp, ll->ctx->b.data, ll->ctx->b.len);
1270 data->len = ll->ctx->b.len;
1271 return 0;
1272
1273 default:
1274 return RJERROR_EXP_TYPE(fdesc->type);
1275 }
1276
1277 case IOP_JSON_DOUBLE:
1278do_double:
1279 switch (fdesc->type) {
1280 case IOP_T_DOUBLE:
1281 *(double *)value = ll->ctx->u.d;
1282 return 0;
1283 default:
1284 return RJERROR_EXP_TYPE(fdesc->type);
1285 }
1286
1287 case IOP_JSON_INTEGER:
1288 integer:
1289 switch (fdesc->type) {
1290#define CHECK_RANGE(_min, _max) \
1291 do { \
1292 if (ll->ctx->u.i < _min || ll->ctx->u.i > _max) { \
1293 return RJERROR_WARG(IOP_JERR_OUT_OF_RANGE); \
1294 } \
1295 } while (0)
1296
1297 case IOP_T_DOUBLE:
1298 *(double *)value = ll->ctx->u.i;
1299 return 0;
1300 case IOP_T_I8:
1301 CHECK_RANGE(INT8_MIN, INT8_MAX);
1302 *(int8_t *)value = ll->ctx->u.i;
1303 break;
1304 case IOP_T_U8:
1305 CHECK_RANGE(0, UINT8_MAX);
1306 *(uint8_t *)value = ll->ctx->u.i;
1307 break;
1308 case IOP_T_I16:
1309 CHECK_RANGE(INT16_MIN, INT16_MAX);
1310 *(int16_t *)value = ll->ctx->u.i;
1311 break;
1312 case IOP_T_U16:
1313 CHECK_RANGE(0, UINT16_MAX);
1314 *(uint16_t *)value = ll->ctx->u.i;
1315 break;
1316 case IOP_T_ENUM:
1317 case IOP_T_I32:
1318 CHECK_RANGE(INT32_MIN, INT32_MAX);
1319 *(int32_t *)value = ll->ctx->u.i;
1320 break;
1321 case IOP_T_U32:
1322 CHECK_RANGE(0, UINT32_MAX);
1323 *(uint32_t *)value = ll->ctx->u.i;
1324 break;
1325 case IOP_T_I64: case IOP_T_U64:
1326 *(uint64_t *)value = ll->ctx->u.i;
1327 break;
1328 case IOP_T_BOOL:
1329 CHECK_RANGE(0, 1);
1330 *(bool *)value = ll->ctx->u.i;
1331 return 0;
1332#undef CHECK_RANGE
1333 default:
1334 return RJERROR_EXP_TYPE(fdesc->type);
1335 }
1336 return 0;
1337
1338 case '{':
1339 if (fdesc->type == IOP_T_STRUCT) {
1340 if (iop_field_is_class(fdesc)) {
1341 *(void **)value = NULL;
1342 }
1343 return unpack_struct(ll, fdesc->u1.st_desc, value, false);
1344 } else
1345 if (fdesc->type == IOP_T_UNION) {
1346 return unpack_union(ll, fdesc->u1.st_desc, value, false);
1347 }
1348 return RJERROR_EXP_TYPE(fdesc->type);
1349
1350 case '.':
1351 /* Extended syntax of union */
1352 if (fdesc->type != IOP_T_UNION) {
1353 return RJERROR(IOP_JERR_UNION_RESERVED);
1354 }
1355 return unpack_union(ll, fdesc->u1.st_desc, value, true);
1356
1357 case '@':
1358 if (ps_startswithlstr(PS, LSTR("include("))) {
1359 /* File inclusion */
1360 return unpack_val_file_inclusion(ll, fdesc, value);
1361 } else {
1362 /* Prefix extended syntax */
1363 if (fdesc->type != IOP_T_STRUCT) {
1364 return RJERROR_EXP_TYPE(fdesc->type);
1365 }
1366 if (iop_field_is_class(fdesc)) {
1367 *(void **)value = NULL;
1368 }
1369 return unpack_struct(ll, fdesc->u1.st_desc, value, true);
1370 }
1371
1372 case IOP_JSON_EOF:
1373 return RJERROR_WARG(IOP_JERR_EXP_SMTH);
1374
1375 default:
1376 return RJERROR_EXP_TYPE(fdesc->type);
1377 }
1378}

1380/** Unpack a JSON array into a C array.
1381 *
1382 * \param[in] ll the lexer.
1383 * \param[in] fdesc a descriptor of the repeated element. Must be set to
1384 * NULL if value is NULL, in which case the array will be skipped.
1385 * \param[out] value memory to write to. Can be set to NULL iff fdesc is
1386 * NULL, in order to skip the array content.
1387 */
1388static int unpack_arr(iop_json_lex_t *ll, const iop_field_t *fdesc,
1389 void *value)
1390{
1391 lstr_t *arr = value;
1392 int size = 0;
1393 void *ptr;
1394
1395 if (arr) {
1396 p_clear(arr, 1);
1397 } else {
1398 assert (!fdesc);
1399 }
1400
1401 for (;;) {
1402 if (PS_CHECK(iop_json_lex_peek(ll, fdesc)) == ']') {
1403 iop_json_lex(ll, NULL);
1404 return 0;
1405 }
1406 if (fdesc) {
1407 if (arr->len >= size) {
1408 size = p_alloc_nr(size);
1409 arr->data = mp_irealloc(ll->mp, arr->data,
1410 arr->len * fdesc->size,
1411 size * fdesc->size, 8, 0);
1412 }
1413 ptr = (void *)((char *)arr->data + arr->len * fdesc->size);
1414 iop_junpack_add_field_path_index(arr->len);
1415 PS_CHECK(unpack_val(ll, fdesc, ptr, true));
1416 iop_junpack_remove_last_field_path();
1417 arr->len++;
1418 } else {
1419 PS_CHECK(skip_val(ll, true));
1420 }
1421 switch(PS_CHECK(iop_json_lex(ll, NULL))) {
1422 case ',': break;
1423 case ']': return 0;
1424 default: return RJERROR_EXP("`,', `;' or `]'");
1425 }
1426 }
1427}

1429/* unions have two syntaxes, depending where they are:
1430 * outside of a struct: { selected_field: value }
1431 * inside a struct, there is two alternatives:
1432 * { ...
1433 * umember: { selected_field: value };
1434 * umember_arr: [ { sfield: val }, { sfield: val }, ... ];
1435 * ...
1436 * }
1437 * or
1438 * { ...
1439 * umember.selected_field [:=] value;
1440 * umember_arr: [ .sfield: val, .sfield: val, ... ];
1441 * ...
1442 * }
1443 *
1444 * XXX Note: skip_val works with unions only because without the extended
1445 * syntax a union can be handled like two overlapping structures.
1446 * skip_val calls this function when the extended syntax is detected.
1447 */
1448static int unpack_union(iop_json_lex_t *ll, const iop_struct_t *desc,
1449 void* value, bool ext_syntax)
1450{
1451 const iop_field_t *fdesc = NULL;
1452
1453 switch (PS_CHECK(iop_json_lex(ll, NULL))) {
1454 case IOP_JSON_IDENT:
1455 case IOP_JSON_STRING:
1456 if (desc) {
1457 SB_1k(camelcase_name);
1458 lstr_t name = LSTR_SB_V(&ll->ctx->b);
1459
1460 if (ll->flags & IOP_UNPACK_USE_C_CASE) {
1461 if (c_to_camelcase(name, false, &camelcase_name) >= 0) {
1462 name = LSTR_SB_V(&camelcase_name);
1463 }
1464 }
1465 if (iop_field_find_by_name(desc, name, NULL, &fdesc) < 0) {
1466 return RJERROR_EXP("a valid union member name");
1467 }
1468 }
1469 break;
1470 default:
1471 return RJERROR_EXP("a valid member name");
1472 }
1473
1474 switch (PS_CHECK(iop_json_lex(ll, NULL))) {
1475 case ':':
1476 if (fdesc) {
1477 /* Write the selected field */
1478 iop_union_set_tag(desc, fdesc->tag, value);
1479 value = (char *)value + fdesc->data_offs;
1480
1481 iop_junpack_add_field_path_field(fdesc->name);
1482 PS_CHECK(unpack_val(ll, fdesc, value, false));
1483 iop_junpack_remove_last_field_path();
1484
1485 if (unlikely(iop_field_has_constraints(desc, fdesc))) {
1486 int ret = iop_field_check_constraints(desc, fdesc, value, 1,
1487 false);
1488 if (ret < 0) {
1489 return RJERROR_WARG(IOP_JERR_CONSTRAINT);
1490 }
1491 }
1492
1493 } else {
1494 PS_CHECK(skip_val(ll, false));
1495 }
1496 break;
1497
1498 /* Extended syntax of union */
1499 case '.':
1500 if (fdesc) {
1501 if (fdesc->type != IOP_T_UNION)
1502 return RJERROR(IOP_JERR_UNION_RESERVED);
1503
1504 /* Write the selected field */
1505 iop_union_set_tag(desc, fdesc->tag, value);
1506 value = (char *)value + fdesc->data_offs;
1507 if (iop_field_is_reference(fdesc)) {
1508 /* reference fields must be dereferenced */
1509 value = iop_field_ptr_alloc(ll->mp, fdesc, value);
1510 }
1511
1512 iop_junpack_add_field_path_field(fdesc->name);
1513 PS_CHECK(unpack_union(ll, fdesc->u1.st_desc, value, true));
1514 iop_junpack_remove_last_field_path();
1515 } else {
1516 PS_CHECK(unpack_union(ll, NULL, NULL, true));
1517 }
1518 break;
1519
1520 default:
1521 return RJERROR_EXP("`:' or `='");
1522 }
1523
1524 /* With the json compliant syntax we must check for a `}' */
1525 if (!ext_syntax && PS_CHECK(iop_json_lex(ll, NULL)) != '}')
1526 return RJERROR_EXP("`}'");
1527 return 0;
1528}

1530static int
1531unpack_struct_prepare_class(iop_json_lex_t *ll, const iop_struct_t *desc,
1532 const iop_struct_t **real_desc, void **value)
1533{
1534 const iop_struct_t *desc_it = desc;
1535 int nb_fields = desc->fields_len;
1536
1537 /* Count the number of fields contained in the classes from the
1538 * wanted one to the master. */
1539 while (desc_it->class_attrs->parent) {
1540 desc_it = desc_it->class_attrs->parent;
1541 nb_fields += desc_it->fields_len;
1542 }
1543
1544 if (*real_desc) {
1545 /* Real class type is already known (because "_class" field was not
1546 * found). */
1547 goto check;
1548 }
1549
1550 /* Get the value of the "_class" field */
1551 if (PS_CHECK(iop_json_lex(ll, NULL)) != IOP_JSON_STRING) {
1552 RJERROR_EXP_TYPE(IOP_T_STRING);
1553 return -1;
1554 }
1555
1556 /* Get the iop_struct_t of the instanciated class from its fullname */
1557 *real_desc = iop_get_class_by_fullname(desc_it, LSTR_SB_V(&ll->ctx->b));
1558 if (!*real_desc) {
1559 e_trace(0, "cannot find child class `%*pM' in the children of "
1560 "`%*pM'; missing IOP_REGISTER_PACKAGES in the binary "
1561 "or IOP_EXPORT_PACKAGES in the DSO?",
1562 SB_FMT_ARG(&ll->ctx->b), LSTR_FMT_ARG(desc->fullname));
1563 return RJERROR_EXP_FMT("a child of `%*pM'",
1564 LSTR_FMT_ARG(desc->fullname));
1565 }
1566
1567 /* We are trying to unpack a class of type "desc", and the packed
1568 * class is of type "real_desc". Check that this is authorized.
1569 *
1570 * Also count the fields contained in the classes from the instantiated
1571 * one to the wanted one (excluded). The total sum is the number of fields
1572 * of the instantiated type.
1573 */
1574 check:
1575 if ((*real_desc)->class_attrs->is_abstract) {
1576 return RJERROR_EXP("a non-abstract class");
1577 }
1578
1579 if (ll->flags & IOP_UNPACK_FORBID_PRIVATE
1580 && (*real_desc)->class_attrs->is_private)
1581 {
1582 /* TODO: the error should probably be an "unknown type" to not expose
1583 * the private name. To be done for YAML unpacker too. */
1584 return RJERROR_EXP_FMT("a non-private child of `%*pM`",
1585 LSTR_FMT_ARG(desc->fullname));
1586 }
1587
1588 desc_it = *real_desc;
1589 while (desc_it && desc_it != desc) {
1590 nb_fields += desc_it->fields_len;
1591 desc_it = desc_it->class_attrs->parent;
1592 }
1593 if (!desc_it) {
1594 return RJERROR_EXP_FMT("a child of `%*pM'",
1595 LSTR_FMT_ARG(desc->fullname));
1596 }
1597
1598 /* Allocate output value */
1599 *value = mp_irealloc(ll->mp, *value, 0, (*real_desc)->size, 8, MEM_RAW);
1600
1601 /* Set the _vprt pointer */
1602 *(const iop_struct_t **)(*value) = *real_desc;
1603
1604 return nb_fields;
1605}

1613/** Unpack a json struct into a C struct.
1614 *
1615 * \param[in] ll the lexer.
1616 * \param[in] desc a descriptor of the output structure. Must be NULL if
1617 * value is NULL, which will skip all the fields (as we don't know the output
1618 * structure layout).
1619 * \param[out] value a pointer to the structure to fill. Can be NULL iff
1620 * desc is NULL.
1621 * \param[in] prefixed if set to true, the function assumes that it has
1622 * to parse a value with the prefix syntax:
1623 * @member value {
1624 * ...
1625 * }
1626 */
1627static int unpack_struct(iop_json_lex_t *ll, const iop_struct_t *desc,
1628 void *value, bool prefixed)
1629{
1630 const iop_field_t *fdesc;
1631 uint32_t seen_buf[BITS_TO_ARRAY_LEN(uint32_t, 256)];
1632 uint32_t *seen_alloc __attribute__((cleanup(seen_free))) = NULL;
1633 uint32_t *seen = seen_buf;
1634 const iop_struct_t *real_desc = desc;
1635 bool is_class = false;
1636 SB_1k(camelcase_name);
1637
1638 pstream_t ctx_ps;
1639 int ctx_line = 0, ctx_col = 0;
1640 bool ctx_prefixed = prefixed;
1641
1642#define INIT_SEEN(_len) \
1643 do { \
1644 size_t _count = BITS_TO_ARRAY_LEN(uint32_t, _len); \
1645 \
1646 if (unlikely(_len > bitsizeof(seen_buf))) { \
1647 seen = seen_alloc = p_new(uint32_t, _count); \
1648 } else { \
1649 p_clear(seen, _count); \
1650 } \
1651 } while (0)
1652
1653 if (desc) {
1654 if ((is_class = iop_struct_is_class(desc))) {
1655 /* We are unpacking a class; first of all, we have to know its
1656 * real type, that is finding the "_class" field.
1657 * This will be done in the loop below. Store context so that we
1658 * can loop again of fields once the "_class" will be found.
1659 */
1660 real_desc = NULL;
1661 ctx_ps = *PS;
1662 ctx_line = ll->ctx->line;
1663 ctx_col = ll->ctx->col;
1664 } else {
1665 /* We are unpacking a struct or a union, so we already know the
1666 * list of expected fields. */
1667 INIT_SEEN(desc->fields_len);
1668 }
1669 }
1670
1671 for (;;) {
1672 bool found_class_field = false;
1673 bool skip_field = false;
1674
1675 fdesc = NULL;
1676
1677 if (PS_CHECK(iop_json_lex_peek(ll, NULL)) == '}') {
1678 iop_json_lex(ll, NULL);
1679 break;
1680 }
1681
1682 switch (PS_CHECK(iop_json_lex(ll, NULL))) {
1683 case IOP_JSON_IDENT:
1684 case IOP_JSON_STRING:
1685 if (!desc) {
1686 /* skip the field (fdesc will be null in next step) */
1687 break;
1688 }
1689 if (real_desc) {
1690 int ifield;
1691 lstr_t name = LSTR_SB_V(&ll->ctx->b);
1692
1693 if (is_class) {
1694 /* We are looping on the fields of a class and we
1695 * already found the "_class" field. We have to skip
1696 * it when reading it again. */
1697 if (lstr_equal(name, LSTR("_class"))) {
1698 skip_field = true;
1699 break;
1700 }
1701 }
1702 if (ll->flags & IOP_UNPACK_USE_C_CASE) {
1703 sb_reset(&camelcase_name);
1704
1705 if (c_to_camelcase(name, false, &camelcase_name) >= 0) {
1706 name = LSTR_SB_V(&camelcase_name);
1707 }
1708 }
1709 ifield = iop_field_find_by_name(real_desc, name, &desc,
1710 &fdesc);
1711 if (fdesc) {
1712 if (TST_BIT(seen, ifield)) {
1713 return RJERROR_SARG(IOP_JERR_DUPLICATED_MEMBER,
1714 fdesc->name.s);
1715 }
1716 SET_BIT(seen, ifield);
1717 } else {
1718 if (!(ll->flags & IOP_UNPACK_IGNORE_UNKNOWN)) {
1719 return RJERROR_EXP_FMT("field of struct %s",
1720 real_desc->fullname.s);
1721 }
1722 }
1723 } else {
1724 /* We are looping on the fields of a class, looking for
1725 * the "_class" one, which determines the real class type.
1726 */
1727 if (lstr_equal(LSTR_SB_V(&ll->ctx->b), LSTR("_class"))) {
1728 found_class_field = true;
1729 }
1730 skip_field = true;
1731 }
1732 break;
1733
1734 default:
1735 return RJERROR_EXP("a valid member name");
1736 }
1737
1738 if (fdesc && ll->flags & IOP_UNPACK_FORBID_PRIVATE) {
1739 const iop_field_attrs_t *attrs = iop_field_get_attrs(desc, fdesc);
1740 if (attrs && TST_BIT(&attrs->flags, IOP_FIELD_PRIVATE)) {
1741 return RJERROR_EXP("a valid member name");
1742 }
1743 }
1744
1745 /* XXX `.' must be kept (using lex_peek) for the unpack_val
1746 * function */
1747 if (!prefixed && PS_CHECK(iop_json_lex_peek(ll, NULL)) != '.'
1748 && PS_CHECK(iop_json_lex(ll, NULL)) != ':')
1749 {
1750 return RJERROR_EXP("`:' or `='");
1751 }
1752 if (found_class_field) {
1753 int count;
1754
1755 prepare_class:
1756 count = unpack_struct_prepare_class(ll, desc, &real_desc,
1757 (void **)value);
1758 PS_CHECK(count);
1759 INIT_SEEN((size_t)count);
1760 value = *(void **)value;
1761 /* The real class is now known; re-loop on the fields */
1762 *PS = ctx_ps;
1763 ll->ctx->line = ctx_line;
1764 ll->ctx->col = ctx_col;
1765 prefixed = ctx_prefixed;
1766 continue;
1767 } else
1768 if (fdesc && !skip_field) {
1769 void *ptr = (char *)value + fdesc->data_offs;
1770
1771 if (fdesc->repeat == IOP_R_OPTIONAL) {
1772 /* check if value is different of null */
1773 if (PS_CHECK(iop_json_lex_peek(ll, fdesc)) == IOP_JSON_IDENT)
1774 {
1775 /* In case of optional field, the member value could be
1776 * null, which is equivalent to an absent member, except
1777 * for a void type, where the "set" value is null */
1778 if (DO_IN_CTX(peeked, IS_NULL())
1779 && fdesc->type != IOP_T_VOID)
1780 {
1781 iop_field_set_absent(fdesc, ptr);
1782 /* consume the null token */
1783 iop_json_lex(ll, NULL);
1784 goto nextfield;
1785 }
1786 }
1787 if (!iop_field_is_class(fdesc)) {
1788 ptr = iop_field_set_present(ll->mp, fdesc, ptr);
1789 }
1790 }
1791
1792 if (prefixed && (((1 << fdesc->type) & IOP_STRUCTS_OK)
1793 || fdesc->repeat == IOP_R_REPEATED))
1794 {
1795 return RJERROR_EXP("a member with a scalar type");
1796 }
1797 iop_junpack_add_field_path_field(fdesc->name);
1798 PS_CHECK(unpack_val(ll, fdesc, ptr, false));
1799 iop_junpack_remove_last_field_path();
1800
1801 if (unlikely(iop_field_has_constraints(desc, fdesc))) {
1802 int ret;
1803
1804 if (fdesc->repeat == IOP_R_REPEATED) {
1805 lstr_t *arr = ptr;
1806
1807 ret = iop_field_check_constraints(desc, fdesc, arr->data,
1808 arr->len, false);
1809 } else {
1810 ret = iop_field_check_constraints(desc, fdesc, ptr, 1,
1811 false);
1812 }
1813 if (ret < 0) {
1814 return RJERROR_WARG(IOP_JERR_CONSTRAINT);
1815 }
1816 }
1817 } else {
1818 PS_CHECK(skip_val(ll, false));
1819 }
1820
1821 nextfield:
1822 if (prefixed) {
1823 /* Special handling of prefixed value */
1824 if (PS_CHECK(iop_json_lex(ll, NULL)) != '{')
1825 return RJERROR_EXP("`{'");
1826
1827 prefixed = false;
1828 continue;
1829 }
1830
1831 switch (PS_CHECK(iop_json_lex(ll, NULL))) {
1832 case ',':
1833 break;
1834 case '}':
1835 goto endcheck;
1836 default:
1837 return RJERROR_EXP("`,', `;' or `}'");
1838 }
1839 }
1840 endcheck:
1841 if (!desc) {
1842 return 0;
1843 }
1844 if (!real_desc) {
1845 /* "_class" field not found */
1846 if (desc->class_attrs->is_abstract) {
1847 /* This is mandatory for abstract classes */
1848 return RJERROR_EXP("`_class' field");
1849 } else {
1850 /* Consider it's of the expected type */
1851 real_desc = desc;
1852 goto prepare_class;
1853 }
1854 }
1855
1856 /* Scan remaining fields */
1857 for (int i = 0;;) {
1858 int acc = i;
1859
1860 fdesc = real_desc->fields;
1861 for (; i < acc + real_desc->fields_len; i++, fdesc++) {
1862 if (TST_BIT(seen, i)) {
1863 continue;
1864 }
1865 if (iop_skip_absent_field_desc(ll->mp, value, real_desc, fdesc)
1866 < 0)
1867 {
1868 return RJERROR_SFIELD(IOP_JERR_MISSING_MEMBER, real_desc,
1869 fdesc);
1870 }
1871 }
1872 if (!is_class) {
1873 break;
1874 }
1875 if (!(real_desc = real_desc->class_attrs->parent)) {
1876 break;
1877 }
1878 }
1879
1880#undef INIT_SEEN
1881 return 0;
1882}

1884/* If "desc" is a structure or a union, "value" is a pointer on the structure
1885 * to fill.
1886 * If "desc" is a class, "value" is a double-pointer on the structure to fill.
1887 * It will be (re)allocated when the size of the real class to unpack will be
1888 * known.
1889 */
1890int iop_junpack(iop_json_lex_t *ll, const iop_struct_t *desc, void *value,
1891 bool single_value)
1892{
1893 const char *src = PS->s;
1894 iop_jlex_reset(ll);
1895 switch (PS_CHECK(iop_json_lex(ll, NULL))) {
1896 case '{':
1897 if (desc->is_union) {
1898 PS_CHECK(unpack_union(ll, desc, value, false));
1899 } else {
1900 PS_CHECK(unpack_struct(ll, desc, value, false));
1901 }
1902 goto endcheck;
1903 case '@':
1904 if (desc->is_union) {
1905 return RJERROR_EXP("`{'");
1906 } else {
1907 PS_CHECK(unpack_struct(ll, desc, value, true));
1908 }
1909
1910 endcheck:
1911 /* Skip trailing characters (insignificant characters) */
1912 for (;;) {
1913 switch (PS_CHECK(iop_json_lex(ll, NULL))) {
1914 case ',':
1915 continue;
1916 case IOP_JSON_EOF:
1917 break;
1918 default:
1919 /* The parser isn't allowed to consume this character, then
1920 * we restore the parser context to its previous position */
1921 RESTORECTX();
1922 }
1923 break;
1924 }
1925
1926 if (single_value) {
1927 if (iop_json_lex(ll, NULL) == IOP_JSON_EOF)
1928 return 0;
1929 return RJERROR_EXP("nothing at this point but");
1930 } else {
1931 return PS->s - src;
1932 }
1933 return 0;
1934
1935 case IOP_JSON_EOF:
1936 if (!single_value)
1937 return 0;
1938 return JERROR(IOP_JERR_NOTHING_TO_READ);
1939 default:
1940 return RJERROR_EXP("`{'");
1941 }
1942}

1944int iop_junpack_ptr(iop_json_lex_t *ll, const iop_struct_t *st, void **out,
1945 bool single_value)
1946{
1947 if (iop_struct_is_class(st)) {
1948 /* "out" will be (re)allocated after, when the real packed class type
1949 * will be known. */
1950 return iop_junpack(ll, st, out, single_value);
1951 }
1952
1953 *out = mp_irealloc(ll->mp, *out, 0, st->size, 8, MEM_RAW);
1954 return iop_junpack(ll, st, *out, single_value);
1955}

2797static int
2798pack_txt(const iop_struct_t *desc, const void *value, int lvl,
2799 iop_jpack_writecb_f *writecb, void *priv, unsigned flags,
2800 jpack_file_ctx_t *file_ctx)
2801{
2802 int res = 0;
2803 bool first = true;
2804 const bool with_indent = !(flags & IOP_JPACK_NO_WHITESPACES);
2805
2806 if (desc->is_union) {
2807 IPUTS("{ ", "{");
2808 } else {
2809 IPUTS("{\n", "{");
2810 lvl++;
2811 }
2812
2813 if (iop_struct_is_class(desc)) {
2814 qv_t(iop_struct) parents;
2815 const iop_struct_t *real_desc = *(const iop_struct_t **)value;
2816
2817 e_assert(panic, !real_desc->class_attrs->is_abstract,
2818 "packing of abstract class '%*pM' is forbidden",
2819 LSTR_FMT_ARG(real_desc->fullname));
2820
2821 /* If this assert fails, you are exporting private classes through
2822 * a public interface... this is BAD!
2823 */
2824 assert (!real_desc->class_attrs->is_private
2825 || !(flags & IOP_JPACK_SKIP_PRIVATE));
2826
2827 /* Write type of class */
2828 if (!(flags & IOP_JPACK_SKIP_CLASS_NAMES)
2829 && !(flags & IOP_JPACK_SKIP_OPTIONAL_CLASS_NAMES
2830 && desc == real_desc))
2831 {
2832 INDENT();
2833 IPUTS("\"_class\": ", "\"_class\":");
2834 PUTS("\"");
2835 PUTLSTR(real_desc->fullname);
2836 PUTS("\"");
2837 first = false;
2838 }
2839
2840 /* We want to write the fields in the order "master -> children", and
2841 * not "children -> master", so first build a qvector of the parents.
2842 */
2843 qv_inita(&parents, 8);
2844 do {
2845 qv_append(&parents, real_desc);
2846 } while ((real_desc = real_desc->class_attrs->parent));
2847
2848 /* Write fields of different levels */
2849 for (int pos = parents.len; pos-- > 0; ) {
2850 res += RETHROW(__pack_txt(parents.tab[pos], value, lvl, writecb,
2851 priv, flags, file_ctx, &first));
2852 }
2853 qv_wipe(&parents);
2854
2855 } else {
2856 res += RETHROW(__pack_txt(desc, value, lvl, writecb, priv, flags,
2857 file_ctx, &first));
2858 }
2859
2860 if (desc->is_union) {
2861 IPUTS(" }", "}");
2862 } else {
2863 if (!first && with_indent) {
2864 PUTS("\n");
2865 }
2866 lvl--;
2867 INDENT();
2868 PUTS("}");
2869 }
2870
2871 if (lvl == 0 && !(flags & IOP_JPACK_NO_TRAILING_EOL)) {
2872 PUTS("\n");
2873 }
2874
2875 return res;
2876}
