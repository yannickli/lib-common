37/** Size tuning parameters.
38 * These are multiplicative factors over sp_alloc_mean.
39 */
40#define ALLOC_MIN 64 /*< minimum block allocation */
41#define RESET_MIN 56 /*< minimum size in mem_stack_pool_reset */
42#define RESET_MAX 256 /*< maximum size in mem_stack_pool_reset */

65__cold
66static mem_stack_blk_t *blk_create(mem_stack_pool_t *sp,
67 mem_stack_blk_t *cur, size_t size_hint)
68{
69 size_t blksize = size_hint + sizeof(mem_stack_blk_t);
70 size_t alloc_target = MIN(100U << 20, ALLOC_MIN * sp_alloc_mean(sp));
71 mem_stack_blk_t *blk;
72
73 if (blksize < sp->minsize)
74 blksize = sp->minsize;
75 if (blksize < alloc_target)
76 blksize = alloc_target;
77 blksize = ROUND_UP(blksize, PAGE_SIZE);
78 blk = imalloc(blksize, 0, MEM_RAW | MEM_LIBC);
79 blk->size = blksize - sizeof(*blk);
80 dlist_add_after(&cur->blk_list, &blk->blk_list);
81
82 sp->stacksize += blk->size;
83 sp->nb_blocks++;
84
85#ifdef MEM_BENCH
86 sp->mem_bench->malloc_calls++;
87 sp->mem_bench->current_allocated += blk->size;
88 sp->mem_bench->total_allocated += blksize;
89 mem_bench_update(sp->mem_bench);
90 mem_bench_print_csv(sp->mem_bench);
91#endif
92
93 return blk;
94}

96__cold
97static void blk_destroy(mem_stack_pool_t *sp, mem_stack_blk_t *blk)
98{
99#ifdef MEM_BENCH
100 /* if called by mem_stack_pool_wipe,
101 * the mem_bench might be deleted
102 */
103 if (sp->mem_bench) {
104 sp->mem_bench->current_allocated -= blk->size;
105 mem_bench_update(sp->mem_bench);
106 mem_bench_print_csv(sp->mem_bench);
107 }
108#endif
109
110 sp->stacksize -= blk->size;
111 sp->nb_blocks--;
112
113 dlist_remove(&blk->blk_list);
114 mem_tool_allow_memory(blk, blk->size + sizeof(*blk), false);
115 ifree(blk, MEM_LIBC);
116}

118static ALWAYS_INLINE mem_stack_blk_t *
119frame_get_next_blk(mem_stack_pool_t *sp, mem_stack_blk_t *cur, size_t alignment,
120 size_t size)
121{
122 size_t deleted_size = 0;
123
124#ifdef MEM_BENCH
125 sp->mem_bench->alloc.nb_slow_path++;
126#endif
127
128 dlist_for_each_entry_after(cur, blk, &sp->blk_list, blk_list) {
129 size_t needed_size = size;
130 uint8_t *aligned_area;
131
132 aligned_area = (uint8_t *)mem_align_ptr((uintptr_t)blk->area, alignment);
133 needed_size += aligned_area - blk->area;
134
135 if (blk->size >= needed_size) {
136 return blk;
137 }
138
139 /* bound deleted size by created size */
140 if (deleted_size >= needed_size) {
141 break;
142 }
143
144 deleted_size += blk->size;
145 blk_destroy(sp, blk);
146 }
147
148 if ((offsetof(mem_stack_blk_t, area) & (alignment - 1)) != 0) {
149 /* require enough free space so we're sure we can allocate the size
150 * bytes properly aligned.
151 */
152 size += alignment;
153 }
154 return blk_create(sp, cur, size);
155}

159 return blk->area + blk->size;

177static void *sp_reserve(mem_stack_pool_t *sp, size_t asked, size_t alignment,
178 uint8_t **end)
179{
180 uint8_t *res;
181 mem_stack_frame_t *frame = sp->stack;
182
183 res = (uint8_t *)mem_align_ptr((uintptr_t)frame->pos, alignment);
184
185 if (unlikely(res + asked > frame_end(frame))) {
186 mem_stack_blk_t *blk = frame_get_next_blk(sp, frame->blk, alignment,
187 asked);
188 frame_set_blk(frame, blk);
189
190 res = blk->area;
191 res = (uint8_t *)mem_align_ptr((uintptr_t)res, alignment);
192 }
193
194 mem_tool_disallow_memory(frame->pos, res - frame->pos);
195 mem_tool_allow_memory(res, asked, false);
196
197 *end = res + asked;
198
199 /* compute a progressively forgotten mean of the allocation size.
200 *
201 * Every 64k allocations, we divide the sum of allocations by four so that
202 * the distant past has less and less consequences on the mean in the hope
203 * that it will converge.
204 *
205 * There is no risk of overflow on alloc_sz,
206 * since mp_imalloc checks that asked < MEM_ALLOC_MAX = (1 << 30),
207 * and MEM_ALLOC_MAX * UINT16_MAX < SIZE_MAX.
208 */
209 if (unlikely(sp->alloc_nb >= UINT16_MAX)) {
210 STATIC_ASSERT (MEM_ALLOC_MAX * UINT16_MAX < SIZE_MAX);
211
212 sp->alloc_sz /= 4;
213 sp->alloc_nb /= 4;
214 }
215 sp->alloc_sz += asked;
216 sp->alloc_nb += 1;
217
218 return res;
219}

221__flatten
222static void *sp_alloc(mem_pool_t *_sp, size_t size, size_t alignment,
223 mem_flags_t flags)
224{
225 mem_stack_pool_t *sp = mem_stack_get_pool(_sp);
226 mem_stack_frame_t *frame = sp->stack;
227 uint8_t *res;
228#ifdef MEM_BENCH
229 proctimer_t ptimer;
230 proctimer_start(&ptimer);
231#endif
232
233 if (unlikely((size == 0))) {
234 return MEM_EMPTY_ALLOC;
235 }
236
237#ifndef NDEBUG
238 if (unlikely(frame == &sp->base))
239 e_panic("allocation performed without a t_scope");
240 if (frame->prev & 1)
241 e_panic("allocation performed on a sealed stack");
242 size += alignment;
243#endif
244 res = sp_reserve(sp, size, alignment, &frame->pos);
245 if (!(flags & MEM_RAW)) {
246#ifdef MEM_BENCH
247 /* since sp_free is no-op, we use the fields to measure memset */
248 proctimer_t free_timer;
249 proctimer_start(&free_timer);
250#endif
251 memset(res, 0, size);
252
253#ifdef MEM_BENCH
254 proctimer_stop(&free_timer);
255 proctimerstat_addsample(&sp->mem_bench->free.timer_stat, &free_timer);
256
257 sp->mem_bench->free.nb_calls++;
258 mem_bench_update(sp->mem_bench);
259#endif
260 }
261
262#ifndef NDEBUG
263 res += alignment;
264 ((void **)res)[-1] = sp->stack;
265 mem_tool_disallow_memory(res - alignment, alignment);
266#endif
267
268#ifdef MEM_BENCH
269 proctimer_stop(&ptimer);
270 proctimerstat_addsample(&sp->mem_bench->alloc.timer_stat, &ptimer);
271
272 sp->mem_bench->alloc.nb_calls++;
273 sp->mem_bench->current_used += size;
274 sp->mem_bench->total_requested += size;
275 mem_bench_update(sp->mem_bench);
276#endif
277 return frame->last = res;
278}

284static void *sp_realloc(mem_pool_t *_sp, void *mem, size_t oldsize,
285 size_t asked, size_t alignment, mem_flags_t flags)
286{
287 mem_stack_pool_t *sp = mem_stack_get_pool(_sp);
288 mem_stack_frame_t *frame = sp->stack;
289 ssize_t sizediff = asked - oldsize;
290 uint8_t *res = mem;
291
292#ifdef MEM_BENCH
293 proctimer_t ptimer;
294 proctimer_start(&ptimer);
295#endif
296
297 if (unlikely(mem == MEM_EMPTY_ALLOC)) {
298 mem = NULL;
299 }
300
301#ifndef NDEBUG
302 if (frame->prev & 1)
303 e_panic("allocation performed on a sealed stack");
304 if (mem != NULL) {
305 mem_tool_allow_memory((byte *)mem - sizeof(void *), sizeof(void *), true);
306 if (unlikely(((void **)mem)[-1] != sp->stack))
307 e_panic("%p wasn't allocated in that frame, realloc is forbidden", mem);
308 mem_tool_disallow_memory((byte *)mem - sizeof(void *), sizeof(void *));
309 }
310 if (unlikely(oldsize == MEM_UNKNOWN))
311 e_panic("stack pools do not support reallocs with unknown old size");
312#endif
313
314 if (likely(res == frame->last)
315 && likely(res + asked <= frame_end(frame)))
316 {
317 assert (res);
318
319 frame->pos = res + asked;
320 sp->alloc_sz += sizediff;
321
322 if (likely(sizediff >= 0)) {
323 mem_tool_allow_memory(res + oldsize, sizediff, false);
324
325 if (!(flags & MEM_RAW)) {
326 p_clear(res + oldsize, sizediff);
327 }
328 } else {
329 mem_tool_disallow_memory(res + asked, -sizediff);
330 if (!asked) {
331 res = MEM_EMPTY_ALLOC;
332 }
333 }
334
335#ifdef MEM_BENCH
336 proctimer_stop(&ptimer);
337 proctimerstat_addsample(&sp->mem_bench->realloc.timer_stat, &ptimer);
338
339 sp->mem_bench->realloc.nb_calls++;
340 sp->mem_bench->total_requested += sizediff;
341 sp->mem_bench->current_used += sizediff;
342 mem_bench_update(sp->mem_bench);
343#endif
344
345 return res;
346 }
347
348 if (likely(sizediff <= 0)) {
349 mem_tool_disallow_memory(res + asked, -sizediff);
350
351#ifdef MEM_BENCH
352 proctimer_stop(&ptimer);
353 proctimerstat_addsample(&sp->mem_bench->realloc.timer_stat, &ptimer);
354
355 sp->mem_bench->realloc.nb_calls++;
356 sp->mem_bench->current_used -= sizediff;
357 mem_bench_update(sp->mem_bench);
358#endif
359
360 return asked ? res : MEM_EMPTY_ALLOC;
361 }
362
363 res = sp_alloc(_sp, asked, alignment, flags | MEM_RAW);
364 if (mem) {
365 memcpy(res, mem, oldsize);
366 mem_tool_disallow_memory(mem, oldsize);
367 }
368 if (!(flags & MEM_RAW)) {
369 p_clear(res + oldsize, sizediff);
370 }
371
372 return res;
373}

383#ifndef NDEBUG
384/** Special code to bypass the allocator.
385 * The frames keep their usual behaviour.
386 * The blk objects are now the prefixes of all the allocations performed.
387 */
388static void *sp_alloc_libc(mem_pool_t *_sp, size_t asked, size_t alignment,
389 mem_flags_t flags)
390{
391 mem_stack_pool_t *sp = mem_stack_get_pool(_sp);
392 size_t oversize = mem_align_ptr(sizeof(mem_stack_blk_t), alignment);
393 mem_stack_blk_t *blk;
394 uint8_t *ptr;
395
396 assert (alignment >= 8);
397
398 ptr = mp_imalloc(&mem_pool_libc, asked + oversize, alignment, flags);
399 ptr += oversize;
400
401 blk = (mem_stack_blk_t *)ptr - 1;
402 blk->size = oversize;
403 dlist_add_tail(&sp->blk_list, &blk->blk_list);
404
405 return ptr;
406}

408static void *sp_realloc_libc(mem_pool_t *_sp, void *mem, size_t oldsize,
409 size_t asked, size_t alignment,
410 mem_flags_t flags)
411{
412 mem_stack_blk_t *blk = (mem_stack_blk_t *)mem - 1;
413 size_t oversize = blk->size;
414 uint8_t *ptr = (uint8_t *)mem;
415
416 assert (oversize >= sizeof(mem_stack_blk_t));
417
418 ptr -= oversize;
419 ptr = mp_irealloc(&mem_pool_libc, ptr, oldsize + oversize,
420 asked + oversize, alignment, flags);
421 ptr += oversize;
422
423 blk = (mem_stack_blk_t *)ptr - 1;
424 __dlist_repair(&blk->blk_list);
425
426 return ptr;
427}

443const void *mem_stack_pool_pop_libc(mem_stack_pool_t *sp)
444{
445 mem_stack_frame_t *frame = sp->stack;
446
447 sp->stack = mem_stack_pool_prev(frame);
448
449 dlist_for_each_entry_after(frame->blk, blk, &sp->blk_list, blk_list) {
450 uint8_t *ptr = (uint8_t *)(blk + 1) - blk->size;
451
452 dlist_remove(&blk->blk_list);
453 mp_ifree(&mem_pool_libc, ptr);
454 }
455
456 {
457 mem_stack_frame_t *f = frame;
458
459 p_delete(&f);
460 }
461
462 return frame;
463}

474mem_stack_pool_t *mem_stack_pool_init(mem_stack_pool_t *sp, const char *name,
475 int initialsize)
476{
477 /* no p_clear is made for two reasons :
478 * - there is few objects that shall be zero-initialized
479 * - it is very poorly optimized by gcc
480 * Therefore, is is needed to explicitly initialize
481 * all the added fields. Thus, it is advised to keep initializations
482 * of the fields in the order of declaration.
483 */
484 sp->stack = &sp->base;
485
486 sp->alloc_sz = 0;
487 sp->alloc_nb = 1; /* avoid the division by 0 */
488 sp->last_reset = lp_getsec();
489
490 sp->funcs = pool_funcs;
491
492 /* root block */
493 sp->size = 0;
494 dlist_init(&sp->blk_list);
495
496 /* root frame */
497 frame_set_blk(&sp->base, blk_entry(&sp->blk_list));
498 sp->base.prev = 0;
499
500 /* 640k should be enough for everybody =) */
501 if (initialsize <= 0)
502 initialsize = 640 << 10;
503 sp->minsize = ROUND_UP(initialsize, PAGE_SIZE);
504
505 sp->stacksize = 0;
506 sp->nb_blocks = 0;
507
508#ifndef NDEBUG
509 /* bypass mem_pool if demanded
510 * XXX this code is intentionnally
511 * placed at the end of the init function,
512 * to avoid problems with seal/unseal macros */
513 if (!mem_pool_is_enabled()) {
514 sp->funcs = pool_funcs_libc;
515 return sp;
516 }
517#endif
518
519#ifdef MEM_BENCH
520 sp->mem_bench = mem_bench_new(LSTR("stack"), WRITE_PERIOD);
521 mem_bench_leak(sp->mem_bench);
522#endif
523
524 sp->name = p_strdup(name);
525 sp->pthread_id = pthread_self();
526
527 spin_lock(&_G.all_pools_lock);
528 dlist_add_tail(&_G.all_pools, &sp->pool_list);
529 spin_unlock(&_G.all_pools_lock);
530
531 return sp;
532}

534void mem_stack_pool_reset(mem_stack_pool_t *sp)
535{
536 mem_stack_blk_t *saved_blk;
537 size_t saved_size;
538 size_t max_size;
539
540 /* bypass mem_pool if demanded */
541 if (!mem_pool_is_enabled()) {
542 return;
543 }
544
545 /* we do not want to wipe everything :
546 * we will keep one block, iff
547 * its size is more than 56*alloc_mean
548 * (blk_create has minimum 64*alloc_mean)
549 * and less than 256*alloc_mean.
550 * we keep the biggest in this range.
551 */
552 sp->last_reset = lp_getsec();
553 saved_blk = NULL;
554 saved_size = RESET_MIN * sp_alloc_mean(sp);
555 max_size = RESET_MAX * sp_alloc_mean(sp);
556
557 dlist_for_each(e, &sp->blk_list) {
558 mem_stack_blk_t *blk = blk_entry(e);
559
560 if (blk->size > saved_size && blk->size < max_size) {
561 if (saved_blk) {
562 blk_destroy(sp, saved_blk);
563 }
564 saved_blk = blk;
565 saved_size = blk->size;
566 } else {
567 blk_destroy(sp, blk);
568 }
569 }
570
571 if (saved_blk) {
572 frame_set_blk(&sp->base, saved_blk);
573 } else {
574 frame_set_blk(&sp->base, blk_entry(&sp->blk_list));
575 }
576}

578void mem_stack_pool_try_reset(mem_stack_pool_t *sp)
579{
580 size_t size_limit = 1 << 20; /* 1 MiB */
581
582 /* Reset only at top stacks. */
583 if (!mem_stack_pool_is_at_top(sp)) {
584 return;
585 }
586
587 /* Do not reset small stacks (10 MiB on the main thread, 1 MiB on the
588 * others). */
589 if (thr_is_on_queue(thr_queue_main_g)) {
590 size_limit *= 10;
591 }
592 if (sp->stacksize < size_limit) {
593 return;
594 }
595
596 /* Do not reset more than once per minute. */
597 if (sp->last_reset + 60 > lp_getsec()) {
598 return;
599 }
600
601 mem_stack_pool_reset(sp);
602}

604void mem_stack_pool_wipe(mem_stack_pool_t *sp)
605{
606 /* bypass mem_pool if demanded */
607 if (!mem_pool_is_enabled()) {
608 return;
609 }
610
611 spin_lock(&_G.all_pools_lock);
612 dlist_remove(&sp->pool_list);
613 spin_unlock(&_G.all_pools_lock);
614
615 p_delete(&sp->name);
616
617#ifdef MEM_BENCH
618 mem_bench_delete(&sp->mem_bench);
619#endif
620
621 frame_set_blk(&sp->base, blk_entry(&sp->blk_list));
622
623 dlist_for_each(e, &sp->blk_list) {
624 blk_destroy(sp, blk_entry(e));
625 }
626 assert (sp->stacksize == 0);
627}

629const void *mem_stack_pool_push(mem_stack_pool_t *sp)
630{
631 uint8_t *end;
632 uint8_t *res;
633 mem_stack_frame_t *frame;
634 mem_stack_frame_t *oldframe = sp->stack;
635
636#ifndef NDEBUG
637 /* bypass mem_pool if demanded */
638 if (!mem_pool_is_enabled()) {
639 return sp_push_libc(sp);
640 }
641#endif
642
643 res = sp_reserve(sp, sizeof(mem_stack_frame_t),
644 __BIGGEST_ALIGNMENT__, &end);
645
646#ifdef MEM_BENCH
647 /* if the assert fires,
648 * it means the stack pool has been wiped by mem_stack_pool_wipe.
649 * t_push'ing again is then an incorrect behaviour.
650 */
651 assert (sp->mem_bench);
652
653 mem_bench_print_csv(sp->mem_bench);
654#endif
655 frame = (mem_stack_frame_t *)res;
656 frame->blk = oldframe->blk;
657 frame->pos = end;
658 frame->end = oldframe->end;
659 frame->last = end;
660 frame->prev = (uintptr_t)oldframe;
661 return sp->stack = frame;
662}

664#ifdef MEM_BENCH
665void mem_stack_bench_pop(mem_stack_pool_t *sp, mem_stack_frame_t * frame)
666{
667 mem_stack_blk_t *last_block = frame->blk;
668 int32_t cused = sp->mem_bench->current_used;
669
670 mem_bench_print_csv(sp->mem_bench);
671 if (sp->stack->blk == last_block) {
672 cused -= (frame->pos - sp->stack->pos
673 - sizeof(mem_stack_frame_t));
674 } else {
675 cused -= frame->pos - last_block->area;
676 last_block = container_of(last_block->blk_list.prev,
677 mem_stack_blk_t, blk_list);
678 while (sp->stack->blk != last_block) {
679 cused -= last_block->size;
680 /* Note: this is inaccurate, because we don't know the size of the
681 unused space at the end of the block
682 */
683 last_block = container_of(last_block->blk_list.prev,
684 mem_stack_blk_t, blk_list);
685 }
686 cused -= (last_block->area + last_block->size
687 - sp->stack->pos + sizeof(mem_stack_frame_t));
688 }
689 if (cused <= 0 || mem_stack_is_at_top(sp)) {
690 cused = 0;
691 }
692 sp->mem_bench->current_used = cused;
693 mem_bench_update(sp->mem_bench);
694}

726#ifndef NDEBUG
727void mem_stack_pool_protect(mem_stack_pool_t *sp, const mem_stack_frame_t *up_to)
728{
729 if (up_to->blk == sp->stack->blk) {
730 mem_tool_disallow_memory(sp->stack->pos, up_to->pos - sp->stack->pos);
731 } else {
732 const byte *end = up_to->pos;

744 }

792static void mem_stack_fix_all_pools_at_fork(void)
793{
794 /* When a process forks, the stack pools of the main thread remain (like
795 * t_pool_g or log_thr_g.mp_stack), but the stack pools of other threads
796 * are no longer valid. This is only a problem when the forked process
797 * does not call exec. */
798 dlist_for_each_entry(mem_stack_pool_t, sp, &_G.all_pools, pool_list) {
799 if (!pthread_equal(sp->pthread_id, pthread_self())) {
800 dlist_remove(&sp->pool_list);
801 }
802 }
803}
