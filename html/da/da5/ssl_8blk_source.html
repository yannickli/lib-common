76void ssl_ctx_wipe(ssl_ctx_t *ctx)
77{
78#ifndef NDEBUG
79 if (unlikely(ctx->encrypt_state == SSL_CTX_UPDATE))
80 e_trace(0, "SSL context closed with inconsistent encrypt state");
81 if (unlikely(ctx->decrypt_state == SSL_CTX_UPDATE))
82 e_trace(0, "SSL context closed with inconsistent decrypt state");
83#endif
84 if (ctx->pkey_encrypt) {
85 EVP_PKEY_CTX_free(ctx->pkey_encrypt);
86 ctx->pkey_encrypt = NULL;
87 }
88 if (ctx->pkey_decrypt) {
89 EVP_PKEY_CTX_free(ctx->pkey_decrypt);
90 ctx->pkey_decrypt = NULL;
91 }
92 if (ctx->pkey) {
93 EVP_PKEY_free(ctx->pkey);
94 ctx->pkey = NULL;
95 }
96
97 EVP_CIPHER_CTX_free(ctx->encrypt);
98 EVP_CIPHER_CTX_free(ctx->decrypt);
99}

102static void
103t_ssl_get_aes_key(ssl_ctx_t *ctx, const EVP_CIPHER *type, const EVP_MD *md,
104 lstr_t password, uint64_t salt, int nb_rounds, lstr_t *key,
105 lstr_t *iv)
106{
107 int keylen = EVP_CIPHER_key_length(type);
108 int ivlen = EVP_CIPHER_iv_length(type);
109 unsigned char *keybuf = t_new_raw(unsigned char, keylen);
110 unsigned char *ivbuf = t_new_raw(unsigned char, ivlen);
111
112 EVP_BytesToKey(type, md, (const unsigned char *)&salt,
113 (const unsigned char *)password.s, password.len, nb_rounds,
114 keybuf, ivbuf);
115 *key = LSTR_INIT_V((char *)keybuf, keylen);
116 *iv = LSTR_INIT_V((char *)ivbuf, ivlen);
117}

187ssl_ctx_t *ssl_ctx_init_pkey(ssl_ctx_t *ctx,
188 lstr_t priv_key, lstr_t pub_key, lstr_t pass)
189{
190 t_scope;
191 lstr_t key = priv_key.len ? priv_key : pub_key;
192 void *u = pass.len ? t_dupz(pass.data, pass.len) : NULL;
193 BIO *bio = NULL;
194
195 ssl_ctx_init(ctx);
196
197 if (!key.len) {
198 __ssl_set_error("invalid key");
199 goto error;
200 }
201
202 bio = BIO_new_mem_buf(key.v, key.len);
203 if (!bio) {
204 ssl_set_error("allocation failure");
205 goto error;
206 }
207
208 if (priv_key.len) {
209 ctx->pkey = PEM_read_bio_PrivateKey(bio, NULL, NULL, u);
210 } else {
211 ctx->pkey = PEM_read_bio_PUBKEY(bio, NULL, NULL, u);
212 }
213 if (!ctx->pkey) {
214 ssl_set_error("read key failure");
215 goto error;
216 }
217 BIO_free(bio);
218 bio = NULL;
219
220 /* whatever key we have, we can alyas encrypt data */
221 ctx->pkey_encrypt = EVP_PKEY_CTX_new(ctx->pkey, NULL);
222 if (!ctx->pkey_encrypt) {
223 ssl_set_error("allocation failure");
224 goto error;
225 }
226
227 if (priv_key.len) {
228 /* private key allows us to decrypt data */
229 ctx->pkey_decrypt = EVP_PKEY_CTX_dup(ctx->pkey_encrypt);
230 if (!ctx->pkey_decrypt) {
231 ssl_set_error("allocation failure");
232 goto error;
233 }
234 if (EVP_PKEY_decrypt_init(ctx->pkey_decrypt) <= 0) {
235 ssl_set_error("decrypt context initialization failure");
236 goto error;
237 }
238 ctx->decrypt_state = SSL_CTX_INIT;
239 }
240
241 if (EVP_PKEY_encrypt_init(ctx->pkey_encrypt) <= 0) {
242 ssl_set_error("encrypt context initialization failure");
243 goto error;
244 }
245 ctx->encrypt_state = SSL_CTX_INIT;
246
247 return ctx;
248
249 error:
250 if (bio) {
251 BIO_free(bio);
252 }
253 ssl_ctx_wipe(ctx);
254 return NULL;
255}

257int ssl_ctx_reset(ssl_ctx_t *ctx, lstr_t password, uint64_t salt,
258 int nb_rounds)
259{
260 t_scope;
261 lstr_t key, iv;
262
263 assert (ctx->encrypt_state == SSL_CTX_NONE
264 || ctx->encrypt_state == SSL_CTX_INIT
265 || ctx->encrypt_state == SSL_CTX_FINISH);
266 assert (ctx->decrypt_state == SSL_CTX_NONE
267 || ctx->decrypt_state == SSL_CTX_INIT
268 || ctx->decrypt_state == SSL_CTX_FINISH);
269
270 t_ssl_get_aes_key(ctx, EVP_aes_256_cbc(), EVP_sha256(), password, salt,
271 nb_rounds, &key, &iv);
272 return ssl_ctx_configure(ctx, EVP_aes_256_cbc(), key, iv);
273}

284int ssl_encrypt_update(ssl_ctx_t *ctx, lstr_t data, sb_t *out)
285{
286 int clen = data.len + EVP_CIPHER_CTX_block_size(ctx->encrypt);
287 unsigned char *cbuf;
288
289 assert (ctx->encrypt_state == SSL_CTX_INIT
290 || ctx->encrypt_state == SSL_CTX_UPDATE);
291
292 cbuf = (unsigned char *)sb_grow(out, clen);
293
294 if (unlikely(data.len <= 0)) {
295 /* XXX with openssl < 1.0, EVP_EncryptUpdate will abort() with empty
296 * data. */
297 ctx->encrypt_state = SSL_CTX_UPDATE;
298 return 0;
299 }
300
301 if (unlikely(!EVP_EncryptUpdate(ctx->encrypt, cbuf, &clen,
302 (unsigned char *)data.s, data.len)))
303 {
304 ctx->encrypt_state = SSL_CTX_NONE;
305 return ssl_set_error("encrypt error");
306 }
307
308 assert (sb_avail(out) >= clen);
309 sb_growlen(out, clen);
310 ctx->encrypt_state = SSL_CTX_UPDATE;
311
312 return 0;
313}
314
315int ssl_encrypt_finish(ssl_ctx_t *ctx, sb_t *out)
316{
317 int outlen = 0;
318
319 assert (ctx->encrypt_state == SSL_CTX_UPDATE);
320 assert (sb_avail(out) >= EVP_CIPHER_CTX_block_size(ctx->encrypt));
321 if (unlikely(!EVP_EncryptFinal_ex(ctx->encrypt,
322 (unsigned char *)sb_end(out), &outlen)))
323 {
324 ctx->encrypt_state = SSL_CTX_NONE;
325 return ssl_set_error("encrypt finalization error");
326 }
327
328 assert (sb_avail(out) >= outlen);
329 sb_growlen(out, outlen);
330 ctx->encrypt_state = SSL_CTX_FINISH;
331
332 return 0;
333}
334
335int ssl_encrypt_reset(ssl_ctx_t *ctx, sb_t *out)
336{
337 assert (ctx->encrypt_state != SSL_CTX_NONE);
338
339 if (ctx->encrypt_state == SSL_CTX_UPDATE) {
340 assert (out);
341 RETHROW(ssl_encrypt_finish(ctx, out));
342 }
343
344 if (ctx->encrypt_state == SSL_CTX_FINISH) {
345 if (unlikely(!EVP_EncryptInit_ex(ctx->encrypt, NULL, NULL, NULL,
346 NULL)))
347 {
348 ctx->encrypt_state = SSL_CTX_NONE;
349 return ssl_set_error("encrypt reset error");
350 }
351 ctx->encrypt_state = SSL_CTX_INIT;
352 }
353 assert (ctx->encrypt_state == SSL_CTX_INIT);
354
355 return 0;
356}

431/** Return the size of the key.
432 *
433 * The length of the key determine both the maximum amount of encryptable data
434 * and the length of the resulted encrypted string. For example, if you have a
435 * 2048 bits length RSA key, then you can only encrypt up to 256 bytes and the
436 * resulting string will (always) be 256 bytes long.
437 */
438static int ssl_pkey_size(ssl_ctx_t *ctx) {
439#if OPENSSL_VERSION_IS(<,1,1,0)
440 RSA *tmp = EVP_PKEY_get1_RSA(ctx->pkey);
441 int size;
442
443 size = RSA_size(tmp);
444 RSA_free(tmp);
445 return size;
446#elif OPENSSL_VERSION_IS(<,3,0,0)
447 return RSA_size(EVP_PKEY_get0_RSA(ctx->pkey));
448#else /* OPENSSL_VERSION_IS(>=,3,0,0) */
449 return EVP_PKEY_get_size(ctx->pkey);
450#endif
451}

455static int ssl_encrypt_pkey(ssl_ctx_t *ctx, lstr_t data, sb_t *out)
456{
457 size_t outlen;
458 size_t inlen = data.len;
459 const unsigned char *in = data.data;
460
461 if (!ctx->pkey
462 || !ctx->pkey_encrypt
463 || ctx->encrypt_state != SSL_CTX_INIT)
464 {
465 __ssl_set_error("invalid encrypt context state");
466 return -1;
467 }
468
469 if (EVP_PKEY_encrypt(ctx->pkey_encrypt, NULL, &outlen, in, inlen) <= 0)
470 {
471 ssl_set_error("encryption failure (size computation)");
472 return -1;
473 }
474 if (EVP_PKEY_encrypt(ctx->pkey_encrypt,
475 (unsigned char *)sb_grow(out, outlen),
476 &outlen, in, inlen) <= 0)
477 {
478 ssl_set_error("encryption failure");
479 return -1;
480 }
481 sb_growlen(out, outlen);
482 return 0;
483}
484
485/**
486 * RSA keys can only encrypt data as long as the size of the key. A 2048 bit
487 * key can only encrypt 256 bytes. Plus, the encryption should (to increase
488 * security) add a random padding. While the most common padding weight 11
489 * bytes, the new recommanded one is about 41 bytes, which reduces again the
490 * available space for encryption.
491 *
492 * Plus, it is NOT RECOMMANDED to split a string and encode each part with the
493 * RSA key (especially for performance purposes). Instead, we generate a
494 * symmetric AES key and a random initialisation vector, and we put them
495 * together with all data that can fit in the RSA header. The rest of the data
496 * are encrypted with the AES key.
497 *
498 * [ -------- RSA encrypted data --------- ][ AES encrypted data ]
499 * [ AES key | AES IV | Data | RSA padding || rest of data ]
500 *
501 * However, for small strings (ligher than rsa_size - rsa_padding bytes), all
502 * data can fit in one RSA header. Thus we continue to use only this header.
503 *
504 * [ -------- RSA encrypted data --------- ]
505 * [ Data | RSA padding ]
506 */
507int ssl_encrypt_pkey_sb(ssl_ctx_t *ctx, lstr_t data, sb_t *out)
508{
509 ssl_ctx_t actx;
510 char key_buf[32];
511 char iv_buf[16];
512 lstr_t key = LSTR_INIT(key_buf, 32);
513 lstr_t iv = LSTR_INIT(iv_buf, 16);
514 int key_size = ssl_pkey_size(ctx);
515 int rsa_buf_size = key_size - RSA_OAEP_PADDING_SIZE;
516 int free_space = rsa_buf_size - key.len - iv.len;
517 SB_1k(rsa_buf);
518
519 /* If the data are small enough, let's encrypt them directly. */
520 if (data.len <= rsa_buf_size) {
521 return ssl_encrypt_pkey(ctx, data, out);
522 }
523
524 /* Otherwise, generate AES-256 key and initialisation vector. */
525 THROW_IF(RAND_bytes(key.data, 32) != 1, -1);
526 THROW_IF(RAND_bytes(iv.data, 16) != 1, -1);
527
528 /* Append the encrypted AES key, its IV, and the first data. */
529 sb_add_lstr(&rsa_buf, key);
530 sb_add_lstr(&rsa_buf, iv);
531 sb_add(&rsa_buf, data.s, free_space);
532 data = LSTR_INIT_V(data.s + free_space, data.len - free_space);
533
534 RETHROW(ssl_encrypt_pkey(ctx, LSTR_SB_V(&rsa_buf), out));
535
536 /* Encrypt the rest of the data with the AES algorithm. */
537 RETHROW_PN(ssl_ctx_init_aes256_by_key(&actx, key, iv));
538 if (ssl_encrypt_update(&actx, data, out) < 0) {
539 goto error;
540 }
541 if (ssl_encrypt_reset(&actx, out) < 0) {
542 goto error;
543 }
544
545 ssl_ctx_wipe(&actx);
546 return 0;
547
548 error:
549 ssl_ctx_wipe(&actx);
550 return -1;
551}

586/* See t_ssl_encrypt_pkey_lstr for encoding informations. */
587int ssl_decrypt_pkey_sb(ssl_ctx_t *ctx, lstr_t data, sb_t *out)
588{
589 ssl_ctx_t actx;
590 lstr_t rsa_enc;
591 lstr_t key;
592 lstr_t iv;
593 lstr_t aes_enc;
594 int key_size = ssl_pkey_size(ctx);
595 SB_1k(rsa_buf);
596
597 /* If the data are small enough, decrypt them directly. */
598 if (data.len == key_size) {
599 return ssl_decrypt_pkey(ctx, data, out);
600 }
601
602 if (data.len < key_size) {
603 __ssl_set_error("invalid encrypted content (invalid size)");
604 return -1;
605 }
606
607 /* Separate the RSA header and the AES encrypted data. */
608 rsa_enc = LSTR_INIT_V(data.s, key_size);
609 aes_enc = LSTR_INIT_V(data.s + key_size, data.len - key_size);
610
611 /* Extract RSA header (get the AES key and IV, and the first data). */
612 RETHROW(ssl_decrypt_pkey(ctx, rsa_enc, &rsa_buf));
613 if (rsa_buf.len < 32 + 16) {
614 __ssl_set_error("invalid AES key or IV in the rsa header");
615 return -1;
616 }
617 key = LSTR_INIT_V(rsa_buf.data, 32);
618 iv = LSTR_INIT_V(rsa_buf.data + 32, 16);
619 sb_add(out, rsa_buf.data + 32 + 16, rsa_buf.len - 32 - 16);
620
621 /* Decrypt the data with the AES algorithm. */
622 RETHROW_PN(ssl_ctx_init_aes256_by_key(&actx, key, iv));
623 if (ssl_decrypt_update(&actx, aes_enc, out) < 0) {
624 goto error;
625 }
626 if (ssl_decrypt_reset(&actx, out) < 0) {
627 goto error;
628 }
629
630 ssl_ctx_wipe(&actx);
631 return 0;
632
633 error:
634 ssl_ctx_wipe(&actx);
635 return -1;
636}

696

892SSL_CTX *ssl_ctx_new_tls(const SSL_METHOD *method, lstr_t key, lstr_t cert,
893 int verify_mode, SSL_verify_cb verify_callback,
894 sb_t *err)
895{
896 SSL_CTX *ctx;
897 long mode;
898
899 assert ((cert.s && key.s) || (!cert.s && !key.s));
900
901 /* Create TLS context -- the ssl module must be loaded. */
902 ctx = SSL_CTX_new(method);
903 if (ctx == NULL) {
904 sb_adds(err, "failed to create SSL context");
905 return NULL;
906 }
907
908 /* Configure TLS context. */
909 SSL_CTX_set_ecdh_auto(ctx, 1);
910
911 /* Ignore unexpected EOF for OpenSSL >= 3.
912 * See https://www.openssl.org/docs/man3.0/man3/SSL_CTX_set_options.html#SSL_OP_IGNORE_UNEXPECTED_EOF
913 */
914#if OPENSSL_VERSION_IS(>=,3,0,0)
915 SSL_CTX_set_options(ctx, SSL_OP_IGNORE_UNEXPECTED_EOF);
916#endif /* OpenSSL >= 3 */
917
918 if (cert.s && key.s) {
919 char path[PATH_MAX] = "/tmp/tls-cert-XXXXXX";
920 int ret;
921
922 /* write certificate in a file in order to use
923 * SSL_CTX_use_certificate_chain_file as we can have chained cert */
924 ret = write_in_tmp_file(path, cert.s, cert.len, err);
925
926 if (ret < 0) {
927 sb_setf(err, "tls: failed to dump certificate: %*pM",
928 SB_FMT_ARG(err));
929 goto error;
930 }
931
932 ret = ssl_ctx_use_certificate_chain_file(ctx, path);
933 unlink(path);
934
935 if (ret < 0) {
936 sb_adds(err, "failed to load certificate");
937 goto error;
938 }
939
940 if (ssl_ctx_use_privatekey_lstr(ctx, key) < 0) {
941 sb_adds(err, "failed to load private key");
942 goto error;
943 }
944
945 if (SSL_CTX_check_private_key(ctx) != 1) {
946 sb_adds(err, "consistency of certificate / key is wrong");
947 goto error;
948 }
949 }
950
951 if (verify_mode != SSL_VERIFY_NONE) {
952 SSL_CTX_set_verify(ctx, verify_mode, verify_callback);
953 }
954
955 mode = SSL_MODE_ENABLE_PARTIAL_WRITE
956 | SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER;
957 if (!(SSL_CTX_set_mode(ctx, mode) & mode)) {
958 sb_adds(err, "failed to set partial write mode");
959 goto error;
960 }
961
962 return ctx;
963
964 error:
965 SSL_CTX_free(ctx);
966 return NULL;
967}

969static inline int ssl_error_conv(int ssl_error)
970{
971 switch (ssl_error) {
972 case SSL_ERROR_ZERO_RETURN:
973 /* Graceful exit. */
974 return 0;
975 case SSL_ERROR_WANT_READ:
976 case SSL_ERROR_WANT_WRITE:
977 errno = EAGAIN;
978 return -1;
979 case SSL_ERROR_SSL:
980 break;
981 case SSL_ERROR_SYSCALL:
982 return -1;
983 default:
984 assert(false);
985 break;
986 }
987
988 e_error("unexpected ssl error: %s",
989 ERR_error_string(ERR_peek_error(), NULL));
990 errno = EIO;
991 return -1;
992}

994/** Helper to convert openssl errors to classical system errors.
995 *
996 * \param[in] ssl The ssl context on which the operation has been called.
997 * \param[in] ret The return code of the operation.
998 * \return ret if ret > 0, 0 on graceful exit, and -1 on error, with errno set
999 * to an appropriate value.
1000 */
1001static int ssl_get_error_conv(SSL *ssl, int ret)
1002{
1003 return ret > 0 ? ret : ssl_error_conv(SSL_get_error(ssl, ret));
1004}

1006static void ssl_change_rbio(SSL *ssl, int fd, sb_t *rbuf)
1007{
1008 BIO *rbio_null = SSL_get_rbio(ssl);
1009 BIO *rbio_next = BIO_next(rbio_null);
1010
1011 /* The previous read was done in our read buffer and trailing
1012 * data were the beginning of the TLS handshake. Now that SSL
1013 * has read these data, fall back on directly reading from the
1014 * socket. */
1015 if (BIO_method_type(rbio_null) == BIO_TYPE_NULL_FILTER
1016 && BIO_method_type(rbio_next) == BIO_TYPE_MEM)
1017 {
1018 BIO *rbio_fd = BIO_new_fd(fd, BIO_NOCLOSE);
1019
1020 BIO_free(BIO_pop(rbio_null));
1021 BIO_push(rbio_null, rbio_fd);
1022 sb_reset(rbuf);
1023 }
1024}

1026ssl_handshake_status_t
1027ssl_do_handshake(SSL *ssl, el_t ev, int fd, sb_t *rbuf)
1028{
1029 int ret;
1030 int ssl_error;
1031
1032 ret = SSL_do_handshake(ssl);
1033 if (ret > 0) {
1034 /* Handshake completed. */
1035 return SSL_HANDSHAKE_SUCCESS;
1036 }
1037 if (ret == 0) {
1038 /* Cleanly closed with respect to the TLS protocol. */
1039 return SSL_HANDSHAKE_CLOSED;
1040 }
1041
1042 errno = 0;
1043 ssl_error = SSL_get_error(ssl, ret);
1044 switch (ssl_error) {
1045 case SSL_ERROR_WANT_READ:
1046 if (unlikely(!ev)) {
1047 /* This may happen if we add a timeout on the socket. */
1048 errno = ETIMEDOUT;
1049 return SSL_HANDSHAKE_ERROR;
1050 }
1051 if (rbuf) {
1052 ssl_change_rbio(ssl, fd, rbuf);
1053 }
1054 el_fd_set_mask(ev, POLLIN);
1055 return SSL_HANDSHAKE_PENDING;
1056 case SSL_ERROR_WANT_WRITE:
1057 el_fd_set_mask(ev, POLLOUT);
1058 return SSL_HANDSHAKE_PENDING;
1059 default:
1060 ssl_error_conv(ssl_error);
1061 break;
1062 }
1063
1064 return SSL_HANDSHAKE_ERROR;
1065}

1095/* This function sends `len` data from `buf` through the `ssl` connection. It
1096 * returns 0 if all data have been sent, and -1 in all other cases. If all or
1097 * some (partial write) data are sent, `total_wr` is increased by the number
1098 * of bytes sent.
1099 *
1100 * This function assumes that `ssl` is configured to allow partial write.
1101 */
1102static int ssl_write_chunk(SSL *ssl, const char *buf, int len, int *total_wr)
1103{
1104 int wr;
1105
1106 wr = ssl_write(ssl, buf, len);
1107 if (wr <= 0) {
1108 return -1;
1109 }
1110 *total_wr += wr;
1111 if (wr != len) {
1112 /* Its unlikely that the system could absorb more data for now. */
1113 errno = EAGAIN;
1114 return -1;
1115 }
1116 return 0;
1117}
1118
1119ssize_t ssl_writev(int fd, const struct iovec *iov, int iovcnt, void *priv)
1120{
1121 SSL *ssl = priv;
1122 char buf[BUFSIZ];
1123 int buflen = 0;
1124 int bufrest = BUFSIZ;
1125 int total_wr = 0;
1126
1127 for (int i = 0; i < iovcnt; i++) {
1128 const char *base = iov[i].iov_base;
1129 int len = iov[i].iov_len;
1130
1131 while (len > 0) {
1132 if (buflen == 0 && len > BUFSIZ) {
1133 /* The iov buffer is large enougth to be directly written. */
1134 if (ssl_write_chunk(ssl, base, len, &total_wr) < 0) {
1135 goto end;
1136 }
1137 break;
1138 }
1139 if (len <= bufrest) {
1140 /* The iov fits entirely in buf. */
1141 memcpy(buf + buflen, base, len);
1142 buflen += len;
1143 bufrest -= len;
1144 break;
1145 }
1146 /* Append as many data as we can in buf and send everything. */
1147 memcpy(buf + buflen, base, bufrest);
1148 base += bufrest;
1149 len -= bufrest;
1150
1151 if (ssl_write_chunk(ssl, buf, BUFSIZ, &total_wr) < 0) {
1152 goto end;
1153 }
1154 buflen = 0;
1155 bufrest = BUFSIZ;
1156 }
1157 }
1158
1159 /* Send the rest. */
1160 if (buflen > 0) {
1161 ssl_write_chunk(ssl, buf, buflen, &total_wr);
1162 }
1163 end:
1164 return total_wr > 0 ? total_wr : -1;
1165}

1167static ssize_t __ssl_sb_read(sb_t *sb, const sb_t *orig, SSL *ssl, int len)
1168{
1169 char *buf;
1170 ssize_t res;
1171
1172 buf = sb_grow(sb, len);
1173 res = ssl_read(ssl, buf, len);
1174
1175 if (res < 0) {
1176 return __sb_rewind_adds(sb, orig);
1177 }
1178
1179 __sb_fixlen(sb, sb->len + res);
1180 return res;
1181}
1182
1183ssize_t ssl_sb_read(sb_t *sb, SSL *ssl, int hint)
1184{
1185 sb_t orig = *sb;
1186 int nb_pending;
1187
1188 /* First read the wanted size. */
1189 RETHROW(__ssl_sb_read(sb, &orig, ssl, hint <= 0 ? BUFSIZ : hint));
1190
1191 /* Then check if there are bufferized data in the SSL context; if so then
1192 * we MUST read them now because the socket might have been entirely read
1193 * by openssl, and otherwise we could end-up in the deadlock described in
1194 * the documentation of ssl_read.
1195 */
1196 while ((nb_pending = SSL_pending(ssl)) > 0) {
1197 ssize_t res = RETHROW(__ssl_sb_read(sb, &orig, ssl, nb_pending));
1198
1199 if (res == 0) {
1200 break;
1201 }
1202 }
1203
1204 return sb->len - orig.len;
1205}
