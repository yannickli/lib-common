333/** RETHROW macros.
334 *
335 * P = Pointer
336 * N = Negative
337 *
338 * These macros can be used on the following pattern:
339 * \code
340 * int toto(...) {
341 * object_t *obj; int toto(...) {
342 * object *obj;
343 * obj = get_object(...); =>
344 * if (!obj) { obj = RETHROW_PN(get_object(...));
345 * return -1; ...
346 * } }
347 * ...
348 * }
349 * \endcode
350 */
351#define RETHROW(e) \
352 ({ typeof(e) __res = (e); \
353 if (unlikely(__res < 0)) { \
354 return __res; \
355 } \
356 __res; \
357 })
358
359#define RETHROW_P(e) \
360 ({ typeof(e) __res = (e); \
361 if (unlikely(__res == NULL)) { \
362 return NULL; \
363 } \
364 __res; \
365 })
366
367#define RETHROW_PN(e) \
368 ({ typeof(e) __res = (e); \
369 if (unlikely(__res == NULL)) { \
370 return -1; \
371 } \
372 __res; \
373 })
374
375#define RETHROW_NP(e) \
376 ({ typeof(e) __res = (e); \
377 if (unlikely(__res < 0)) { \
378 return NULL; \
379 } \
380 __res; \
381 })

383/** THROW macros.
384 *
385 * These macros can be used on the following pattern:
386 * \code
387 * if (time < lunch_time) {
388 * return false; => THROW_FALSE_IF(time < lunch_time );
389 * }
390 * \endcode
391 */
392#define THROW_IF(e, val) \
393 do { \
394 if (unlikely(e)) { \
395 return (val); \
396 } \
397 } while (0)
398#define THROW_UNLESS(e, val) THROW_IF(!(e), (val))

402#define THROW_ERR_IF(e) THROW_IF((e), -1)
403#define THROW_ERR_UNLESS(e) THROW_UNLESS((e), -1)
404#define THROW_FALSE_IF(e) THROW_IF((e), false)

559 * when building for_each-like macros. The `ctx` argument is used to avoid
560 * conflicts with multiple invocations but must be the same for the same
561 * instruction block.
562 *
563 * Note that ';' is not allowed because we're inside a for loop.
564 *
565 * You will find a typical example with __iop_array_for_each.
566 */
567#define FOR_INSTR_INIT(ctx) \
568 for (bool ctx##stop = false; !ctx##stop; ctx##stop = true)
