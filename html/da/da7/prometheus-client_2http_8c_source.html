38static void metrics_query_on_done(httpd_query_t *q)
39{
40 SB_8k(buf);
41 outbuf_t *ob;
42
43 /* Send request headers */
44 ob = httpd_reply_hdrs_start(q, HTTP_CODE_OK, true);
45 ob_adds(ob, "Content-Type: text/plain; version=0.0.4\n");
46 httpd_reply_hdrs_done(q, -1, false);
47
48 /* Reply with metrics data */
49 prom_collector_bridge(&prom_collector_g, &buf);
50 ob_addsb(ob, &buf);

65int prom_http_start_server(const core__httpd_cfg__t *cfg,
66 sb_t * nullable err)
67{
68 t_scope;
69 pstream_t host;
70 sockunion_t su;
71 lstr_t addr = cfg->bind_addr;
72 httpd_trigger_t *trigger;
73
74 if (_G.httpd) {
75 if (err) {
76 sb_addf(err, "server is already running");
77 return -1;
78 }
79 }
80
81 /* Resolve address */
82 RETHROW(addr_resolve2("prometheus HTTP server", addr, 0, 0, &su,
83 &host, &_G.listen_port, err));
84
85 /* Start HTTP server */
86 _G.httpd_cfg = httpd_cfg_new();
87 httpd_cfg_from_iop(_G.httpd_cfg, cfg);
88 if (!(_G.httpd = httpd_listen(&su, _G.httpd_cfg))) {
89 httpd_cfg_delete(&_G.httpd_cfg);
90 if (err) {
91 sb_addf(err, "cannot bind HTTP server on %*pM",
92 LSTR_FMT_ARG(addr));
93 return -1;
94 }
95 }
96
97 if (_G.listen_port == 0) {
98 _G.listen_port = getsockport(el_fd_get_fd(_G.httpd), su.family);
99 addr = t_lstr_fmt("%*pM:%d", PS_FMT_ARG(&host), _G.listen_port);
100 }
101 _G.listen_host = lstr_dup(LSTR_PS_V(&host));
102
103 /* Register "metrics/" trigger */
104 trigger = httpd_trigger_new();
105 trigger->cb = metrics_query_hook;
106 httpd_trigger_register(_G.httpd_cfg, GET, "/metrics", trigger);
107
108 logger_notice(&_G.logger, "listening for prometheus scraping on %*pM",
109 LSTR_FMT_ARG(addr));
110 return 0;
111}
