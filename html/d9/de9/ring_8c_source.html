21void generic_ring_ensure(generic_ring *r, int newlen, int el_siz)
22{
23 int cursize = r->size;
24
25 if (unlikely(newlen < 0 || newlen * el_siz < 0))
26 e_panic("trying to allocate insane amount of RAM");
27
28 if (newlen <= r->size)
29 return;
30
31 r->size = p_alloc_nr(r->size);
32 if (r->size < newlen)
33 r->size = newlen;
34 r->tab = irealloc(r->tab, r->len * el_siz, r->size * el_siz, 0,
35 MEM_RAW | MEM_LIBC);
36
37 /* if elements are split in two parts. Move the shortest one */
38 if (r->first + r->len > cursize) {
39 char *base = r->tab;
40 int right_len = cursize - r->first;
41 int left_len = r->len - right_len;
42
43 if (left_len > right_len || left_len > r->size - cursize) {
44 memmove(base + el_siz * (r->size - right_len),
45 base + el_siz * r->first, el_siz * right_len);
46 r->first = r->size - right_len;
47 } else {
48 memcpy(base + el_siz * cursize, base, el_siz * left_len);
49 }
50 }
51}
