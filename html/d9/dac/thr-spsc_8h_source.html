28/*
29 * This file provides an implementation of:
30 * - unbounded: means that the queue allocates as many nodes as its high
31 * watermark, and never releases them, so ensure you control its maximum
32 * size externally.
33 * - wait free: there is absolutely *no* atomic instruction involved.
34 * - non blocking: dequeue returns "false" if it seems there is nothing in the
35 * queue instead of blocking. It can be a false negative.
36 * - SPSC queue: single producer, single consumer. It means that while enqueue
37 * and dequeue can run concurrently, only one consumer and one producer can
38 * work the queue at the same time.
39 *
40 * The code is adapted from
41 * http://www.1024cores.net/home/lock-free-algorithms/queues/unbounded-spsc-queue
42 */

97static inline spsc_node_t *spsc_queue_alloc_node(spsc_queue_t *q)
98{
99 if (likely(q->first != q->head_copy)) {
100 spsc_node_t *n = q->first;
101
102 q->first = atomic_load(&n->next);
103 return n;
104 }
105 q->head_copy = atomic_load(&q->head);
106 if (likely(q->first != q->head_copy)) {
107 spsc_node_t *n = q->first;
108
109 q->first = atomic_load(&n->next);
110 return n;
111 }
112
113 return p_new_raw(spsc_node_t, 1);
114}

116static ALWAYS_INLINE void spsc_queue_push(spsc_queue_t *q, void *v)
117{
118 spsc_node_t *n = spsc_queue_alloc_node(q);
119
120 atomic_init(&n->next, NULL);
121 n->value = v;
122 atomic_store(&q->tail->next, n);
123 q->tail = n;
124}
125
126static ALWAYS_INLINE bool spsc_queue_pop(spsc_queue_t *q, void *v, size_t v_size)
127{
128 spsc_node_t *head = atomic_load_explicit(&q->head, memory_order_relaxed);
129 spsc_node_t *n = atomic_load(&head->next);
130
131 if (n) {
132#if __BYTE_ORDER == __LITTLE_ENDIAN
133 memcpy(v, &n->value, v_size);
134#else
135 memcpy(v, (char *)&n->value[1] - v_size, v_size);
136#endif
137 atomic_store(&q->head, n);
138 return true;
139 }
140 return false;
141}

143static inline void *spsc_queue_pop_ptr(spsc_queue_t *q)
144{
145 spsc_node_t *head = atomic_load_explicit(&q->head, memory_order_relaxed);
146 spsc_node_t *n = atomic_load(&head->next);
147
148 if (n) {
149 void *v = n->value;
150 atomic_store(&q->head, n);
151 return v;
152 }
153 return NULL;
154}
