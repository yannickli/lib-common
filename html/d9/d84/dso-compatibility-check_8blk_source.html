65int main(int argc, char *argv[])
66{
67 t_scope;
68 iop_dso_t *dso;
69 iop_dso_t *dso_old;
70 iop_compat_ctx_t *ctx;
71 unsigned flags;
72 SB_1k(err);
73 lstr_t mode;
74 const char *arg0 = NEXTARG(argc, argv);
75 qv_t(lstr) sorted_pkg_names;
76
77 if (!argc) {
78 makeusage(EX_USAGE, arg0, "", usage, popt);
79 }
80
81 argc = parseopt(argc, argv, popt, 0);
82
83 if (argc != 0 || _G.help || !_G.old_dso || !_G.new_dso) {
84 makeusage(_G.help ? EX_OK : EX_USAGE, arg0, "", usage, popt);
85 }
86
87 mode = _G.mode ? LSTR(_G.mode) : LSTR_EMPTY_V;
88
89 if (_G.mode) {
90 if (lstr_ascii_iequal(mode, LSTR("JSON"))) {
91 flags = IOP_COMPAT_JSON;
92 }
93 else
94 if (lstr_ascii_iequal(mode, LSTR("BINARY"))) {
95 flags = IOP_COMPAT_BIN;
96 }
97 else
98 if (lstr_ascii_iequal(mode, LSTR("ALL"))) {
99 flags = IOP_COMPAT_ALL;
100 }
101 else {
102 e_fatal("unkown mode `%*pM`", LSTR_FMT_ARG(mode));
103 }
104 } else {
105 flags = IOP_COMPAT_ALL;
106 }
107
108 dso = open_dso(_G.new_dso);
109 dso_old = open_dso(_G.old_dso);
110 ctx = iop_compat_ctx_new();
111
112 t_qv_init(&sorted_pkg_names, qm_len(iop_pkg, &dso_old->pkg_h));
113 qm_for_each_pos(iop_pkg, pos, &dso_old->pkg_h) {
114 qv_append(&sorted_pkg_names, dso_old->pkg_h.values[pos]->name);
115 }
116 qv_sort(lstr)(&sorted_pkg_names, core_lstr_cmp);
117
118 tab_for_each_entry(name, &sorted_pkg_names) {
119 const iop_pkg_t *pkg_old;
120 const iop_pkg_t *pkg;
121
122 pkg_old = qm_get(iop_pkg, &dso_old->pkg_h, &name);
123
124 pkg = qm_get_def(iop_pkg, &dso->pkg_h, &name, NULL);
125 if (!pkg) {
126 fprintf(stderr, "package `%*pM` is not in the new dso\n",
127 LSTR_FMT_ARG(name));
128 continue;
129 }
130
131 if (iop_pkg_check_backward_compat_ctx(pkg_old, pkg, ctx,
132 flags, &err) < 0)
133 {
134 fprintf(stderr, "%*pM\n", SB_FMT_ARG(&err));
135 sb_reset(&err);
136 }
137 }
138
139 iop_compat_ctx_delete(&ctx);
140 iop_dso_close(&dso);
141 iop_dso_close(&dso_old);
142
143 return 0;
144}
