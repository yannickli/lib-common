22static void mime_put_http_ctype(outbuf_t *ob, const char *path)
23{
24 t_scope;
25 const char *fpart, *ext;
26 bool has_enc = false;
27 static const struct {
28 int elen;
29 const char *ext;
30 const char *ct;
31 const char *ce;
32 } map[] = {
33#define E2(ext, ct, ce) { sizeof(ext) - 1, ext, ct, ce }
34#define E(ext, ct) E2(ext, ct, NULL)
35 E("dbg", "text/plain"),
36 E("cfg", "text/plain"),
37 E("err", "text/plain"),
38 E("log", "text/plain"),
39 E("lst", "text/plain"),
40 E("txt", "text/plain"),
41
42 E("wsdl", "text/xml"),
43 E("xml", "text/xml"),
44 E("xsd", "text/xml"),
45 E("xsl", "text/xml"),
46
47 E("htm", "text/html"),
48 E("html", "text/html"),
49
50 E("pcap", "application/x-pcap"),
51
52 E("pdf", "application/pdf"),
53 E("csv", "application/csv"),
54
55 E("tar", "application/x-tar"),
56 E2("tgz", "application/x-tar", "gzip"),
57 E2("tbz2", "application/x-tar", "bzip2"),
58
59 E("rar", "application/rar"),
60 E("zip", "application/zip"),
61#undef E2
62#undef E
63 };
64
65 fpart = path_filepart(path);
66 fpart = t_dupz(fpart, strlen(fpart));
67 ext = path_extnul(fpart);
68 if (strequal(ext, ".gz")) {
69 ob_adds(ob, "Content-Encoding: gzip\r\n");
70 has_enc = true;
71 *(char *)ext = '\0';
72 ext = path_extnul(fpart);
73 } else
74 if (strequal(ext, ".Z")) {
75 ob_adds(ob, "Content-Encoding: compress\r\n");
76 has_enc = true;
77 *(char *)ext = '\0';
78 ext = path_extnul(fpart);
79 } else
80 if (strequal(ext, ".bz2")) {
81 ob_adds(ob, "Content-Encoding: bzip2\r\n");
82 has_enc = true;
83 *(char *)ext = '\0';
84 ext = path_extnul(fpart);
85 }
86
87 if (*ext++ == '.') {
88 int extlen = strlen(ext);
89
90 for (int i = 0; i < countof(map); i++) {
91 if (map[i].elen == extlen && !strcasecmp(map[i].ext, ext)) {
92 ob_addf(ob, "Content-Type: %s\r\n", map[i].ct);
93 if (!has_enc && map[i].ce)
94 ob_addf(ob, "Content-Encoding: %s\r\n", map[i].ce);
95 return;
96 }
97 }
98 }
99 ob_adds(ob, "Content-Type: application/octet-stream\r\n");
100}
101static void httpd_query_reply_make_index_(httpd_query_t *q, int dfd,
102 const struct stat *st, bool head)
103{
104 DIR *dir = fdopendir(dfd);
105 outbuf_t *ob;
106
107 if (!dir) {
108 httpd_reject(q, NOT_FOUND, "");
109 return;
110 }
111
112 ob = httpd_reply_hdrs_start(q, HTTP_CODE_OK, false);
113 httpd_put_date_hdr(ob, "Last-Modified", st->st_mtime);
114 ob_adds(ob, "Content-Type: text/html\r\n");
115 httpd_reply_hdrs_done(q, -1, false);
116 if (!head) {
117 struct dirent *de;
118
119 ob_adds(ob, "<html><body><h1>Index</h1>");
120
121 rewinddir(dir);
122 while ((de = readdir(dir))) {
123 struct stat tmp;
124
125 if (de->d_name[0] == '.')
126 continue;
127 if (fstatat(dfd, de->d_name, &tmp, AT_SYMLINK_NOFOLLOW))
128 continue;
129 if (S_ISDIR(tmp.st_mode)) {
130 ob_addf(ob, "<a href=\"%s/\">%s/</a><br>", de->d_name, de->d_name);
131 } else
132 if (S_ISREG(tmp.st_mode)) {
133 ob_addf(ob, "<a href=\"%s\">%s</a><br>", de->d_name, de->d_name);
134 }
135 }
136 closedir(dir);
137
138 ob_adds(ob, "</body></html>\r\n");
139 }
140 httpd_reply_done(q);
141}

154void httpd_reply_file(httpd_query_t *q, int dfd, const char *file, bool head)
155{
156 int fd = openat(dfd, file, O_RDONLY);
157 struct stat st;
158 outbuf_t *ob;
159 void *map = NULL;
160
161 if (fd < 0)
162 goto ret404;
163
164 if (fstat(fd, &st))
165 goto ret404;
166 if (S_ISDIR(st.st_mode)) {
167 if (file[strlen(file) - 1] != '/')
168 goto ret404;
169 httpd_reply_make_index(q, fd, head);
170 return;
171 }
172 if (!S_ISREG(st.st_mode))
173 goto ret404;
174 if (!head && st.st_size > (16 << 10)) {
175 map = mmap(NULL, st.st_size, PROT_READ, MAP_SHARED, fd, 0);
176 if (map == MAP_FAILED) {
177 httpd_reject(q, INTERNAL_SERVER_ERROR, "mmap failed: %m");
178 close(fd);
179 return;
180 }
181 madvise(map, st.st_size, MADV_SEQUENTIAL);
182 }
183
184 ob = httpd_reply_hdrs_start(q, HTTP_CODE_OK, false);
185 httpd_put_date_hdr(ob, "Last-Modified", st.st_mtime);
186 if (st.st_mtime >= lp_getsec() - 10) {
187 ob_addf(ob, "ETag: W/\"%jx-%jxx-%lx\"\r\n",
188 (int64_t)st.st_ino, st.st_size, st.st_mtime);
189 } else {
190 ob_addf(ob, "ETag: \"%jx-%jxx-%lx\"\r\n",
191 (int64_t)st.st_ino, st.st_size, st.st_mtime);
192 }
193 mime_put_http_ctype(ob, file);
194 httpd_reply_hdrs_done(q, st.st_size, false);
195 if (!head) {
196 if (map) {
197 ob_add_memmap(ob, map, st.st_size);
198 } else {
199 ob_xread(ob, fd, st.st_size);
200 }
201 }
202 httpd_reply_done(q);
203 close(fd);
204 return;
205
206 ret404:
207 if (fd >= 0)
208 close(fd);
209 httpd_reject(q, NOT_FOUND, "");
210}

219static void httpd_trigger__dir_cb(httpd_trigger_t *cb, httpd_query_t *q,
220 const httpd_qinfo_t *req)

222 t_scope;

229 httpd_reply_file(q, AT_FDCWD, buf, req->method == HTTP_METHOD_HEAD);

233{

235 httpd_trigger__dir_t *cbdir = p_new_extra(httpd_trigger__dir_t, len + 1);
236

238 len--;

240 cbdir->dirlen = len;
