69/** Format a table.
70 *
71 * This function appends a table formatted from the given columns, whose
72 * descriptions are provided in \p hdr, and rows whose content is provided in
73 * \p data in the buffer \p out. The content is guaranteed to end with a
74 * newline character.
75 *
76 * The output contains a first row with the column title, followed by one line
77 * per entry of \p data. The width of the columns is adjusted to their content
78 * as well as the dimensioning parameters provided in the column description.
79 * Columns are separated by two spaces. A row may contain less columns than
80 * the header, in which case the missing cells are filled with the default
81 * values for those columns.
82 *
83 * The header of the columns is always left-aligned. The last column may
84 * contain extra data that does not fit on a single line.
85 *
86 * \param[in] out The output buffer.
87 * \param[in] hdr The description of the columns.
88 * \param[in] data The content of the table.
89 */
90void sb_add_table(sb_t *out, const qv_t(table_hdr) *hdr,
91 const qv_t(table_data) *data);
92
93/** Format a table in CSV.
94 *
95 * This function appends a CSV table formatted from the given columns, whose
96 * descriptions are provided in \p hdr, and rows whose content is provided in
97 * \p data in the buffer \p out. The content is guaranteed to end with a
98 * newline character. This function is meant to offer the possibility to print
99 * content as CSV or as a table. If you only want to print CSV and not tables,
100 * a more straightforward and better option is to used directly CSV helpers
101 * for string buffers 'sb_t'.
102 *
103 * The output contains a first row with the column names, followed by one line
104 * per entry of \p data. The columns are separated by \ref sep.
105 *
106 * Only `omit_if_empty` and `empty_value` column description variables are
107 * used to format the CSV output.
108 *
109 * \param[in,out] out The output buffer.
110 * \param[in] sep CSV separator.
111 * \param[in] hdr The description of the columns.
112 * \param[in] data The content of the table.
113 */
114void sb_add_csv_table(sb_t *out, const qv_t(table_hdr) *hdr,
115 const qv_t(table_data) *data, int sep);
