60static int
61parse_class_id_range(const char *class_id_range, int *min, int *max)
62{
63 pstream_t ps = ps_initstr(class_id_range);
64
65 if (!ps_len(&ps)) {
66 return -1;
67 }
68
69 errno = 0;
70 *min = ps_geti(&ps);
71 THROW_ERR_IF(errno != 0 || *min < 0);
72
73 RETHROW(ps_skipc(&ps, '-'));
74
75 *max = ps_geti(&ps);
76 THROW_ERR_IF(errno != 0 || *max < *min || *max > UINT16_MAX);
77
78 return 0;
79}

81int main(int argc, char *argv[])
82{
83 iop_dso_t *dso;
84 const char *arg0 = NEXTARG(argc, argv);
85 int class_id_min;
86 int class_id_max;
87 int first_available_id = -1;
88 const char *fullname;
89 const iop_obj_t *obj;
90 bool used_ids_hdr_printed = false;
91
92 argc = parseopt(argc, argv, popt_g, 0);
93 if (argc != 3 || _G.help) {
94 makeusage(_G.help ? EX_OK : EX_USAGE, arg0,
95 "<dso_path> <class_id_range> <class_name>",
96 usage_g, popt_g);
97 }
98
99 dso = open_dso(NEXTARG(argc, argv));
100 if (!dso) {
101 goto end;
102 }
103
104 if (parse_class_id_range(NEXTARG(argc, argv),
105 &class_id_min, &class_id_max) < 0)
106 {
107 printf("invalid class id range\n");
108 goto end;
109 }
110
111 fullname = NEXTARG(argc, argv);
112 if (!(obj = iop_get_obj(LSTR(fullname)))) {
113 printf("cannot find IOP object `%s`\n", fullname);
114 goto end;
115 }
116 if (obj->type != IOP_OBJ_TYPE_ST || !iop_struct_is_class(obj->desc.st)) {
117 printf("IOP object `%s` is not a class\n", fullname);
118 goto end;
119 }
120
121 for (int i = class_id_min; i <= class_id_max; i++) {
122 const iop_struct_t *st = iop_get_class_by_id(obj->desc.st, i);
123
124 if (st) {
125 if (!used_ids_hdr_printed) {
126 printf("Used class ids in the family of `%s`:\n", fullname);
127 used_ids_hdr_printed = true;
128 }
129 printf(" %d (`%*pM`)\n", i, LSTR_FMT_ARG(st->fullname));
130 } else
131 if (first_available_id < 0) {
132 first_available_id = i;
133 }
134 }
135
136 if (used_ids_hdr_printed) {
137 printf("\n\n");
138 }
139
140 if (first_available_id >= 0) {
141 printf("First available class id in the family of `%s` is %d\n",
142 fullname, first_available_id);
143 } else {
144 printf("No available class id found in the family of `%s`\n",
145 fullname);
146 }
147
148 end:
149 iop_dso_close(&dso);
150 return first_available_id >= 0 ? 0 : EX_DATAERR;
151}
