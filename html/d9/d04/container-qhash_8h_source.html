32/*
33 * QHashes: Real Time hash tables
34 *
35 * The trick is during resize. When we resize, we keep the old qhash_hdr_t
36 * around. the keys/values arrays are shared during a resize between the old
37 * and new values, in place.
38 *
39 * When we look for a given value, there are two functions:
40 *
41 * qhash_get_safe::
42 * which doesn't modify the hash table, but has to look up in both the
43 * old and new view of the values to be sure the value doesn't exist.
44 *
45 * qhash_get::
46 * this function preemptively moves all the collision chain that
47 * correspond to the searched key to the new hashtable, plus makes the
48 * move progress if one is in progress.
49 *
50 * This function must not be used in a qhash enumeration.
51 *
52 *
53 * To reserve a new slot, one must use __qhash_put or a wrapper. __qhash_put
54 * returns the position where the key lives in the 31 least significant bits
55 * of the result. The most significant bit is used to notify that there is a
56 * value in that slot already.
57 *
58 *
59 * When we insert an element, it always goes in the "new" view of the
60 * hashtable. But the slot it has to be put may be occupied by the "old"
61 * view of the hashtable. Should that happen, we reinsert the value that
62 * lives at the offending place (note that the value may actually never
63 * move, but it doesn't really matter). Such a move may trigger more moves,
64 * but we assume that collision chains are usually short due to our double
65 * hashing.
66 *
67 */
68
69#define QHASH_COLLISION (1U << 31)
70#define QHASH_OVERWRITE (1U << 0)
71
72/*
73 * len holds:
74 * - the number of elements in the hash when accessed through qh->hdr.len
75 * - the maximum position at which the old view still has elements through
76 * qh->old->len.

128static inline void qhash_slot_inv_flags(size_t * nonnull bits, uint32_t pos)
129{
130 size_t off = (2 * pos) % bitsizeof(size_t);
131
132 bits[2 * pos / bitsizeof(size_t)] ^= (size_t)3 << off;
133}
134static inline size_t qhash_slot_get_flags(const size_t * nonnull bits,
135 uint32_t pos)
136{
137 size_t off = (2 * pos) % bitsizeof(size_t);
138 return (bits[2 * pos / bitsizeof(size_t)] >> off) & (size_t)3;
139}
140static inline size_t qhash_slot_is_set(const qhash_hdr_t * nonnull hdr,
141 uint32_t pos)
142{
143 if (unlikely(pos >= hdr->size)) {
144 return 0;
145 }
146 return TST_BIT(hdr->bits, 2 * pos);
147}

149static inline void qhash_del_at(qhash_t * nonnull qh, uint32_t pos)
150{
151 qhash_hdr_t *hdr = &qh->hdr;
152 qhash_hdr_t *old = qh->old;
153
154#ifndef NDEBUG
155 e_assert(panic, qh->ghosts != UINT32_MAX,
156 "delete operation performed on a sealed hash table");
157#endif
158
159 if (likely(qhash_slot_is_set(hdr, pos))) {
160 qhash_slot_inv_flags(hdr->bits, pos);
161 hdr->len--;
162 qh->ghosts++;
163 } else
164 if (unlikely(old != NULL) && qhash_slot_is_set(old, pos))
165 {
166 qhash_slot_inv_flags(old->bits, pos);
167 hdr->len--;
168 }
169}

605 * may even try using \ref qh_set_minsize before activating the caching if you

1039/** Find-reserve slot to insert {key,v} pair.
1040 *
1041 * These functions finds the slot where to insert {\a key, \a v} in the qmap.
1042 *
1043 * qm_reserve[_h]():
1044 *
1045 * If there is no collision, the slot is reserved and the key slot is filled,
1046 * the user still have to fill the value slot.
1047 * If there is a collision, the key slot is overwritten or not, depending on
1048 * the \a fl argument.
1049 * In both cases, the value slot is left unchanged and its update is up to the
1050 * caller.
1051 *
1052 * qm_put[_h]():
1053 *
1054 * If there is no collision, the slot is reserved and filled with
1055 * {\a key, \a v}, else the behavior depends upon the \a fl argument.
1056 *
1057 * These functions are useful to write efficient code (spare one lookup) in
1058 * code that could naively be written:
1059 * <code>
1060 * if (qm_find(..., qh, key) < 0) {
1061 * // prepare 'v'
1062 * qm_add(..., qh, key, v);
1063 * }
1064 * </code>
1065 * and can instead be written:
1066 * <code>
1067 * pos = qm_put(..., qh, key, v, 0);
1068 * if (pos & QHASH_COLLISION) {
1069 * // fixup qh->{keys,values}[pos ^ QHASH_COLLISION];
1070 * }
1071 * </code>
1072 * or:
1073 * <code>
1074 * pos = qm_reserve(..., qh, key, 0);
1075 * if (pos & QHASH_COLLISION) {
1076 * // fixup qh->keys[pos ^ QHASH_COLLISION];
1077 * }
1078 * qh->values[pos & ~QHASH_COLLISION] = v;
1079 * </code>
1080 *
1081 * @param name the base name of the qmap
1082 * @param qh pointer to the qmap in wich the value shall be inserted
1083 * @param key the value of the key
1084 * @param v the value associated to the key
1085 * @param fl
1086 * 0 or QHASH_OVERWRITE. If QHASH_OVERWRITE is set, \a key and \a v are used
1087 * to overwrite the {key,value} stored in the qmap when there is a
1088 * collision.
1089 * @return
1090 * the position of the slot is returned in the 31 least significant bits.
1091 * The most significant bit is used to tell if there was a collision.
1092 * The constant #QHASH_COLLISION is provided to extract and mask this most
1093 * significant bit.
1094 */
