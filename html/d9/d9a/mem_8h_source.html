40static inline void * nullable memrchr(const void * nonnull s, int c, size_t n)
41{
42 const uint8_t *start = (const uint8_t *)s;
43 const uint8_t *end = start + n;
44
45 while (end > start) {
46 --end;
47 if (*end == (uint8_t)c) {
48 return (void *)end;
49 }
50 }
51 return NULL;
52}

114/** Copies at <code>n</code> characters from the string pointed to by
115 * <code>src</code> to the buffer <code>dest</code> of <code>size</code>
116 * bytes.
117 * If <code>size</code> is greater than 0, a terminating '\0' character will
118 * be put at the end of the copied string.
119 *
120 * The source and destination strings should not overlap. No assumption
121 * should be made on the values of the characters after the first '\0'
122 * character in the destination buffer.
123 *
124 * @return <code>n</code>
125 */
126size_t pstrcpymem(char * nonnull restrict dest, ssize_t size,
127 const void * nonnull restrict src, size_t n)
128 __leaf;
129
130/** Copies the string pointed to by <code>src</code> to the buffer
131 * <code>dest</code> of <code>size</code> bytes.
132 * If <code>size</code> is greater than 0, a terminating '\0' character will
133 * be put at the end of the copied string.
134 *
135 * The source and destination strings should not overlap. No assumption
136 * should be made on the values of the characters after the first '\0'
137 * character in the destination buffer.
138 *
139 * @return the length of the source string.
140 * @see pstrcpylen
141 *
142 * RFE/RFC: In a lot of cases, we do not care that much about the length of
143 * the source string. What we want is "has the string been truncated
144 * and we should stop here, or is everything fine ?". Therefore, may be
145 * we need a function like :
146 * int pstrcpy_ok(char *dest, ssize_t size, const char *src)
147 * {
148 * if (pstrcpy(dest, size, src) < size) {
149 * return 0;
150 * } else {
151 * return 1;
152 * }
153 * }
154 *
155 */
156__attr_nonnull__((1, 3))
157static ALWAYS_INLINE
158size_t pstrcpy(char * nonnull restrict dest, ssize_t size,
159 const char * nonnull restrict src)
160{
161 return pstrcpymem(dest, size, src, strlen(src));
162}
163
164/** Copies at most <code>n</code> characters from the string pointed
165 * to by <code>src</code> to the buffer <code>dest</code> of
166 * <code>size</code> bytes.
167 * If <code>size</code> is greater than 0, a terminating '\0' character will
168 * be put at the end of the copied string.
169 *
170 * The source and destination strings should not overlap. No assumption
171 * should be made on the values of the characters after the first '\0'
172 * character in the destination buffer.
173 *
174 * @return the length of the source string or <code>n</code> if smaller.
175 */
176__attr_nonnull__((1, 3))
177static ALWAYS_INLINE
178size_t pstrcpylen(char * nonnull restrict dest, ssize_t size,
179 const char * nonnull restrict src, size_t n)
180{
181 return pstrcpymem(dest, size, src, strnlen(src, n));
182}
183
184/** Appends the string pointed to by <code>src</code> at the end of
185 * the string pointed to by <code>dest</code> not overflowing
186 * <code>size</code> bytes.
187 *
188 * The source and destination strings should not overlap.
189 * No assumption should be made on the values of the characters
190 * after the first '\0' character in the destination buffer.
191 *
192 * If the destination buffer doesn't contain a properly '\0' terminated
193 * string, dest is unchanged and the value returned is size+strlen(src).
194 *
195 * @return the length of the source string plus the length of the
196 * destination string.
197 */
198__attr_nonnull__((1, 3))
199static inline size_t
200pstrcat(char * nonnull restrict dest, ssize_t size,
201 const char * nonnull restrict src)
202{
203 size_t dlen = size > 0 ? strnlen(dest, size) : 0;
204 return dlen + pstrcpy(dest + dlen, size - dlen, src);
205}

234/*
235 * mem_flags_t modify the ialloc functions behaviours.
236 *
237 * %MEM_RAW::
238 * - imalloc does not zero returned memory if set.
239 * - irealloc does not zero [oldsize .. size[ if set.
240 * Note that irealloc implementations should not trust oldsize if
241 * MEM_RAW is set and oldsize is MEM_UNKNOWN.
242 *
243 * %MEM_ERRORS_OK::
244 * caller can deal with allocation errors from malloc or realloc.
245 *
246 * pool origins:
247 *
248 * %MEM_STATIC::
249 * memory comes from alloca() or a static buffer.
250 *
251 * %MEM_LIBC::
252 * This flag makes no sense for pools. irealloc and ifree shall assume
253 * that this memory is allocated using the libc allocator (malloc or
254 * calloc).
255 *
256 * %MEM_STACK::
257 * This flags asks the allocations to be performed on the system default
258 * stack pool.
259 *
260 */
261enum mem_pools_t {
262 MEM_STATIC = 0,
263 MEM_OTHER = 1,
264 MEM_LIBC = 2,
265 MEM_STACK = 3,
266 MEM_MMAP = 4,
267};
268#define MEM_POOL_MASK force_cast(mem_flags_t, 0x00ff)
269#define MEM_FLAGS_MASK force_cast(mem_flags_t, 0xff00)
270#define MEM_RAW force_cast(mem_flags_t, 1 << 8)
271#define MEM_ERRORS_OK force_cast(mem_flags_t, 1 << 9)
272#define MEM_UNALIGN_OK force_cast(mem_flags_t, 1 << 10)
273#define MEM_BY_FRAME force_cast(mem_flags_t, 1 << 11)
274#define MEM_EFFICIENT_REALLOC force_cast(mem_flags_t, 1 << 12)

703/** Create a new memory ring-pool.
704 *
705 * The memory ring-pool is an allocator working with frames of memory which
706 * rotates on a extending ring.
707 *
708 * This is quite a fifo-pool with the oldest frames always re-appended at the
709 * head of the pool.
710 *
711 * \param[name] Name of the ring pool, used for debug.
712 * \param[initialsize] First memory block size.
713 */

718
719/** Force clean the ring pool from empty memory blocks.
720 *
721 * Adapted from mem_stack_pool_reset().
722 *
723 * Only keeps the current frame so as the one that fits the best the current
724 * needs regarding the mean allocation size (see code for details).
725 *
726 * Should be called when the pool is idle: won't do anything if called while
727 * there are still some frames not released.
728 *
729 * XXX The clean-up is supposed to be automatically triggered every 64K frame
730 * releases, using this function to force it means that you have a clear
731 * understanding of what you are doing.
732 */
733void mem_ring_reset(mem_pool_t * nonnull) __leaf;

735/** Create a new frame of memory in the ring.
736 *
737 * This function push a new extensible page of memory in the ring. You need to
738 * call it before allocating memory. But be careful, you cannot push a new
739 * frame if you haven't sealed the previous one.
740 *
741 * \return Frame cookie (needed to release the frame later).
742 */
743const void * nonnull mem_ring_newframe(mem_pool_t * nonnull) __leaf;

745/** Get the active frame cookie. */
746const void * nonnull mem_ring_getframe(mem_pool_t * nonnull) __leaf;
747
748/** Seal the active frame.
749 *
750 * When you seal the active frame, you cannot perform new allocations in it
751 * but the allocated data is still accessible. Do not forget to release it
752 * later!
753 *
754 * \return Frame cookie (needed to release the frame later).
755 */
756const void * nonnull mem_ring_seal(mem_pool_t * nonnull) __leaf;

758/** Release the frame identified by the given cookie.
759 *
760 * This function will free an existing frame using the given cookie. Be
761 * careful that the cookie will be invalidated after the frame release.
762 */

773/** Rewind the ring-pool at a given checkpoint.

776 * current active frame and the checkpoint. Then it will unseal the frame
