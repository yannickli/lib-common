22/* {{{ Huffman encoding & decoding */
23
24int hpack_encode_huffman(lstr_t str, void *out_, int len)
25{
26#define OUTPUT_BYTE(val) \
27 do { \
28 *out++ = (val); \
29 if (out == out_end) { \
30 return len; \
31 } \
32 } while (0)
33
34 uint8_t *out = (uint8_t *)out_;
35 uint8_t *out_end = out + len;
36
37 /* acts a bit buffer aligned on the left (i.e., MSB) */
38 struct {
39 uint64_t word;
40 unsigned bits;
41 } bitbuf = {0, 0};
42
43 if (unlikely(len == 0)) {
44 return 0;
45 }
46 for (int i = 0; i != str.len; i++) {
47 uint8_t ch = str.s[i];
48 uint64_t codeword64 = hpack_huffcode_tab_g[ch].codeword;
49 unsigned bitlen = hpack_huffcode_tab_g[ch].bitlen;
50
51 bitbuf.word |= codeword64 << (64 - bitlen - bitbuf.bits);
52 bitbuf.bits += bitlen;
53 if (bitbuf.bits > 32) {
54 /* output full bytes from the left of our 64-bit bitbuf */
55 for (; bitbuf.bits >= 8; bitbuf.word <<= 8, bitbuf.bits -= 8) {
56 OUTPUT_BYTE(bitbuf.word >> (64 - 8));
57 }
58 }
59 }
60 /* pad to a byte boundary with one-valued bits */
61 if (likely(bitbuf.bits % 8)) {
62 uint64_t codeword64 = 0xFF;
63 unsigned bitlen = 8 - (bitbuf.bits % 8);
64
65 bitbuf.word |= codeword64 << (64 - 8 - bitbuf.bits);
66 bitbuf.bits += bitlen;
67 }
68 /* output the remaining full bytes */
69 for (; bitbuf.bits >= 8; bitbuf.word <<= 8, bitbuf.bits -= 8) {
70 OUTPUT_BYTE(bitbuf.word >> (64 - 8));
71 }
72 return out - (uint8_t *)out_;
73#undef OUTPUT_BYTE
74}

76int hpack_decode_huffman(lstr_t str, void *out_)
77{
78 uint8_t *out = (uint8_t *)out_;
79 uint8_t state;
80 const hpack_huffdec_trans_t *trans = NULL;
81
82 if (unlikely(str.len == 0)) {
83 return 0;
84 }
85 state = 0;
86 for (int i = 0; i != str.len; i++) {
87 uint8_t b = str.s[i];
88 uint8_t nibbles[2];
89
90 nibbles[0] = b >> 4;
91 nibbles[1] = b & 0xF;
92
93 for (int n = 0; n < 2; n++) {
94 trans = &hpack_huffdec_trans_tab_g[state][nibbles[n]];
95 THROW_ERR_IF(trans->error);
96 if (trans->emitter) {
97 *(out)++ = trans->sym;
98 }
99 state = trans->state;
100 }
101 }
102 THROW_ERR_UNLESS(trans->final);
103 return out - (uint8_t *)out_;
104}

109int hpack_encode_int(uint32_t val, uint8_t prefix_bits, byte out[8])
110{
111 byte *out_ = out;
112 uint32_t max_prefix_num;
113
114 /* RFC 7541 §5.1: Integer Representation */
115 assert(prefix_bits >= 1 && prefix_bits <= 8);
116 max_prefix_num = (1u << prefix_bits) - 1;
117 if (val < max_prefix_num) {
118 *out++ = val;
119 } else {
120 *out++ = max_prefix_num;
121 val -= max_prefix_num;
122 for (; val >> 7; val >>= 7) {
123 *out++ = 0x80u | val;
124 }
125 *out++ = val;
126 }
127 return out - out_;
128}

130int hpack_decode_int(pstream_t *in, uint8_t prefix_bits, uint32_t *val)
131{
132 byte b;
133 unsigned m;
134 uint64_t res;
135 uint32_t max_prefix_num;
136
137 /* RFC 7541 §5.1: Integer Representation */
138 /* Implementation limits: value <= UINT32_MAX && coded size <= 8 bytes */
139 assert(prefix_bits >= 1 && prefix_bits <= 8);
140 max_prefix_num = (1u << prefix_bits) - 1;
141 THROW_ERR_IF(ps_done(in));
142 b = max_prefix_num & __ps_getc(in);
143 if (b < max_prefix_num) {
144 *val = b;
145 return 0;
146 }
147 res = 0;
148 m = 0;
149 for (int i = 1; i < 8; i++) {
150 THROW_ERR_IF(ps_done(in));
151 b = __ps_getc(in);
152 res |= ((uint64_t)(0x7Fu & b)) << m;
153 m += 7;
154 if (b < 128) {
155 res += max_prefix_num;
156 /* check for overflow */
157 THROW_ERR_IF(res > UINT32_MAX);
158 *val = res;
159 return 0;
160 }
161 }
162 /* coded integer overruns the 8-byte size limit */
163 return -1;
164}

417{

526{
527 hpack_enc_dtbl_entry_t e;
528
529 /* XXX: placate compilers which complain that e may be uninitialized */

538 assert(e.key_id);

572 .sz = HPACK_HDR_SIZE(key.len, val.len),
573 .key_id = key_id,
574 .val_id = val_id,

576 .mre_val = val_id,

580 * dtbl */

582 hpack_enc_dtbl_evict_last_entry(dtbl);
583 }

585 return;
586 }

588 dtbl->tbl_size += e.sz;
589 dtbl->ins_cnt++;
590 if (!key_id) {
591 return;
592 }
593 pos = qm_reserve(hpack_insertion_idx, &dtbl->ins_idx, key_id, 0);
594 if (pos & QHASH_COLLISION) {
595 pos &= ~QHASH_COLLISION;
596 ins_idx = dtbl->ins_idx.values[pos];
597 hpack_enc_dtbl_get_ent_at_ins(dtbl, ins_idx)->mre_key = false;
598 }
599 dtbl->ins_idx.values[pos] = dtbl->ins_cnt;
600 if (!val_id) {
601 return;
602 }
603 pos = qm_reserve(hpack_insertion_idx, &dtbl->ins_idx,
604 key_id | val_id << 16u, 0);
605 if (pos & QHASH_COLLISION) {
606 pos &= ~QHASH_COLLISION;
607 ins_idx = dtbl->ins_idx.values[pos];
608 hpack_enc_dtbl_get_ent_at_ins(dtbl, ins_idx)->mre_val = false;
609 }
610 dtbl->ins_idx.values[pos] = dtbl->ins_cnt;
611}

617 int idx;
618
619 pos = qm_find(hpack_insertion_idx, &dtbl->ins_idx, key_id | val_id << 16);

632 return -(1 + idx);
633 }
634 return 0;

755int hpack_encoder_write_hdr(hpack_enc_dtbl_t *nonnull dtbl, lstr_t key,
756 lstr_t val, uint16_t key_id, uint16_t val_id,
757 unsigned flags, byte *out_)
758{
759 t_scope;
760 byte *out = out_;
761 int idx_stbl = 0;
762 int idx_dtbl = 0;
763 int idx;
764 uint8_t prefix_bits;
765 unsigned start_byte_mask;
766 int zlen;
767 bool add_dtbl;
768
769 assert(dtbl->tbl_size_limit <= dtbl->tbl_size_max);
770 /* XXX: negative indices signal partial match (key-part only) */
771 if (!(flags & HPACK_FLG_SKIP_STBL)) {
772 if (flags & HPACK_FLG_SKIP_VAL) {
773 idx_stbl = -hpack_stbl_find_hdr_by_key(&key);
774 if (idx_stbl) {
775 /* found a partial match in the static table and a perfect
776 * match is not required. So, we can start the encoding. */
777 goto encode;
778 }
779 } else {
780 idx_stbl = hpack_stbl_find_hdr(key, val);
781 if (idx_stbl > 0) {
782 /* found a perfect match in the static table. No need to look
783 * in the dynamic one. So, we can start the encoding. */
784 goto encode;
785 }
786 }
787 }
788 /* Static table is skipped or a match was not found or partial. Look for a
789 * better match in the dynamic table unless it is skipped. */
790 if (key_id && !(flags & HPACK_FLG_SKIP_DTBL)) {
791 if (flags & HPACK_FLG_SKIP_VAL || !val_id) {
792 idx_dtbl = -hpack_enc_dtbl_find_hdr(dtbl, key_id, 0);
793 if (idx_dtbl) {
794 /* found a partial match in the dynamic table and a perfect
795 * match is not required. */
796 goto encode;
797 }
798 } else {
799 idx_dtbl = hpack_enc_dtbl_find_hdr(dtbl, key_id, val_id);
800 }
801 }
802encode:
803 if (idx_stbl > 0 || idx_dtbl > 0) {
804 /* full match: i.e., value is part of the match */
805 idx = idx_stbl > 0 ? idx_stbl : HPACK_STBL_IDX_MAX + idx_dtbl;
806 /* Indexed Header Representation */
807 out += hpack_write_int(idx, 7, 0x80u, out);
808 goto done;
809 }
810 /* value not matched: Non-Indexed Header Representation */
811 /* Q: shall we add to the DTBL? */
812 /* A: add only if forced by flags, or, it is a token header and not
813 * forbidden by flags. */
814 add_dtbl =
815 (flags & HPACK_FLG_ADD_DTBL) ||
816 (key_id &&
817 !(flags & (HPACK_FLG_NVRADD_DTBL | HPACK_FLG_NOADD_DTBL)));
818 if (!add_dtbl) {
819 /* Literal Header Field Never Indexed or Without Indexing */
820 prefix_bits = 4;
821 start_byte_mask = flags & HPACK_FLG_NVRADD_DTBL ? 0x10u : 0x00u;
822 } else {
823 /* Literal Header Field with Incremental Indexing */
824 prefix_bits = 6;
825 start_byte_mask = 0x40u;
826 hpack_enc_dtbl_add_hdr(dtbl, key, val, key_id, val_id);
827 }
828 assert(idx_stbl <= 0 && idx_dtbl <= 0);
829 if (!idx_stbl && !idx_dtbl) {
830 /* New Name */
831 idx = 0;
832 } else {
833 /* Indexed Name */
834 idx = idx_stbl ? -idx_stbl : HPACK_STBL_IDX_MAX - idx_dtbl;
835 }
836 /* encode key : write index */
837 out += hpack_write_int(idx, prefix_bits, start_byte_mask, out);
838 if (idx) {
839 goto encode_value;
840 }
841 /* encode key: New Name */
842 assert(!idx_stbl && !idx_dtbl);
843 if (flags & HPACK_FLG_LWR_KEY) {
844 key = t_lstr_dup(key);
845 lstr_ascii_tolower(&key);
846 }
847 /* Q: shall we zip? */
848 /* A: Don't zip if prevented by flags, if not, ...*/
849 if (flags & HPACK_FLG_NOZIP_KEY) {
850 zlen = 0;
851 } else {
852 /* zip the key, ... */
853 zlen = hpack_get_huffman_len(key);
854 /* but only if efficient or forced by flags */
855 zlen = zlen < key.len ? zlen : flags & HPACK_FLG_ZIP_KEY ? zlen : 0;
856 }
857 /* zip key if zlen > 0 */
858 out += hpack_write_str(key, zlen, out);
859encode_value:
860 /* encode value: Q: shall we zip? */
861 /* A: Don't zip if prevented by flags, if not, ...*/
862 if (flags & HPACK_FLG_NOZIP_VAL) {
863 zlen = 0;
864 } else {
865 /* zip the value, ... */
866 zlen = hpack_get_huffman_len(val);
867 /* but only if efficient or forced by flags */
868 zlen = zlen < val.len ? zlen : flags & HPACK_FLG_ZIP_VAL ? zlen : 0;
869 }
870 /* zip val if zlen > 0 */
871 out += hpack_write_str(val, zlen, out);
872done:
873 return out - out_;
874}

905/* flags for hpack_xhdr_t */
906enum {
907 XHDR_ADD_DTBL = 1 << 0,
908 XHDR_NEW_KEY = 1 << 1,
909 XHDR_NEW_VAL = 1 << 2,
910 XHDR_RAW_KEY = 1 << 3,

914int hpack_decoder_extract_hdr(hpack_dec_dtbl_t *dtbl, pstream_t *in,
915 hpack_xhdr_t *xhdr)
916{
917 uint32_t idx;
918 bool new_val;
919 bool add_dtbl;
920 uint8_t prefix_bits;
921 unsigned flags;
922 pstream_t key;
923 pstream_t val;
924 uint32_t len;
925 uint32_t keylen;
926 uint32_t vallen;
927
928 assert(dtbl->tbl_size_limit <= dtbl->tbl_size_max);
929 assert(ps_has(in, 1));
930 if (in->b[0] >= 0x80u) {
931 /* Indexed Header Representation */
932 prefix_bits = 7;
933 add_dtbl = new_val = false;
934 } else if (in->b[0] >= 0x40u) {
935 /* Literal Header Field Representation with Incremental Indexing */
936 prefix_bits = 6;
937 add_dtbl = new_val = true;
938 } else if (in->b[0] >= 0x20u) {
939 return LOGE("unexpected dts update while reading a header");
940 } else {
941 /* Literal Header Representation without Indexing / Never Indexed */
942 prefix_bits = 4;
943 new_val = true;
944 add_dtbl = false;
945 }
946 if (hpack_decode_int(in, prefix_bits, &idx)) {
947 return LOGE("unable to decode a header's index");
948 }
949 if (!idx && !new_val) {
950 /* Indexed Header Representation with 0 index */
951 return LOGE("unexpected index: 0");
952 }
953 flags = 0;
954 if (!idx) {
955 /* New Name (key) */
956 flags |= XHDR_NEW_KEY;
957 flags = in->b[0] < 0x80 ? flags | XHDR_RAW_KEY : flags;
958 if (hpack_decode_int(in, 7, &len)) {
959 return LOGE("unable to decode the key length");
960 }
961 if (!ps_has(in, len)) {
962 return LOGE("key length is bigger than the containing header");
963 }
964 key = __ps_get_ps(in, len);
965 /* XXX: huffman encoding maximum compression ratio: 5/8 */
966 keylen = (flags & XHDR_RAW_KEY) ? len : 2u * len;
967 } else if (idx <= HPACK_STBL_IDX_MAX) {
968 keylen = hpack_stbl_g[idx].key.len;
969 vallen = hpack_stbl_g[idx].val.len;
970 } else if (idx <= HPACK_STBL_IDX_MAX + dtbl->entries.len) {
971 int idx_ = idx - HPACK_STBL_IDX_MAX - 1;
972 hpack_dec_dtbl_entry_t *ent = hpack_dec_dtbl_get_ent(dtbl, idx_);
973
974 keylen = ent->key.len;
975 vallen = ent->val.len;
976 } else {
977 return LOGE("unexpected idx: %u", idx);
978 }
979 if (new_val) {
980 flags |= XHDR_NEW_VAL;
981 flags = in->b[0] < 0x80 ? flags | XHDR_RAW_VAL : flags;
982 if (hpack_decode_int(in, 7, &len)) {
983 return LOGE("unable to decode the value length");
984 }
985 if (!ps_has(in, len)) {
986 return LOGE("value length is bigger than the containing header");
987 }
988 val = __ps_get_ps(in, len);
989 /* XXX: huffman encoding maximum compression ratio: 5/8 */
990 vallen = (flags & XHDR_RAW_KEY) ? len : 2u * len;
991 }
992 if (add_dtbl) {
993 flags |= XHDR_ADD_DTBL;
994 }
995 xhdr->flags = flags;
996 if (idx) {
997 xhdr->idx = idx;
998 } else {
999 xhdr->key = key;
1000 }
1001 if (new_val) {
1002 xhdr->val = val;
1003 }
1004 return keylen + vallen;
1005}
