170/** Convert IOP-JSon to an IOP C structure.
171 *
172 * This function unpacks an IOP structure encoded in JSon format. You have to
173 * initialize an iop_json_lex_t structure and iop_jlex_attach() it on the data
174 * you want to unpack before calling this function.
175 *
176 * If you want to set some special flags, you have to set it using
177 * iop_jlex_set_flags() before calling this this function.
178 *
179 * This function cannot be used to unpack a class; use `iop_junpack_ptr`
180 * instead.
181 *
182 * The provided memory pool *MUST* be a frame-based pool. The unpacker does
183 * not provide any guarantee about the individual allocations made.
184 * Use iop_dup() if you want to transfer the unpacked object to a pool with a
185 * single allocation.
186 *
187 * \param[in] ll The JSon parser.
188 * \param[in] st The IOP structure description.
189 * \param[out] out Pointer on the IOP structure to write.
190 * \param[in] single_value Whether there is one or more structures to unpack
191 * in the attached pstream_t.
192 * \return
193 * If `single_value` is true, the function returns 0 upon success and
194 * something < 0 in case of errors (see iop_json_error). Note that it
195 * returns IOP_JERR_NOTHING_TO_READ if EOF is reached without finding
196 * a structure to unpack.
197 *
198 * If `single_value` is false, the function returns the number of bytes
199 * read successfully, or 0 if it reaches EOF. An empty buffer will not
200 * raise an error.
201 */
202__must_check__
203int iop_junpack(iop_json_lex_t * nonnull ll, const iop_struct_t * nonnull st,
204 void * nonnull out, bool single_value);

344/** JSon packer custom flags */
345enum iop_jpack_flags {
346 /** Obsolete, kept for backward compatibility.
347 */
348 IOP_JPACK_STRICT = (1U << 0),
349
350 /** Generate compact JSON.
351 *
352 * Omit cosmetic whitespaces such as indentations and spaces after colons.
353 */
354 IOP_JPACK_NO_WHITESPACES = (1U << 1),
355
356 /** Do not append '\n' when done.
357 */
358 IOP_JPACK_NO_TRAILING_EOL = (1U << 2),
359
360 /** Don't pack private fields (lossy).
361 *
362 * Use this flag when packing objects on a public interface. This will
363 * omit private fields from the generated JSON.
364 */
365 IOP_JPACK_SKIP_PRIVATE = (1U << 3),
366
367 /** Write big integers as integers.
368 *
369 * By default, the packer writes integers that cannot be safely unpacked
370 * in Javascript as strings instead of integers. With this flag set, the
371 * packer will always pack integers as integers.
372 */
373 IOP_JPACK_UNSAFE_INTEGERS = (1U << 4),
374
375 /** Skip fields having their default value.
376 *
377 * This is good to make the JSon more compact, but is dangerous if a
378 * default value changes.
379 */
380 IOP_JPACK_SKIP_DEFAULT = (1U << 5),
381
382 /** Skip empty repeated fields. */
383 IOP_JPACK_SKIP_EMPTY_ARRAYS = (1U << 6),
384
385 /** Skip empty sub-structures. */
386 IOP_JPACK_SKIP_EMPTY_STRUCTS = (1U << 7),
387
388 /** Shorten long data strings when not writing a file (lossy).
389 *
390 * Data longer than 25 characters will be replaced by
391 * "XXXXXXXXXXX …(skip x bytes)… YYYYYYYYYYY" where only the first and
392 * last 11 characters are kept.
393 */
394 IOP_JPACK_SHORTEN_DATA = (1U << 8),
395
396 /** Skip class names (lossy). */
397 IOP_JPACK_SKIP_CLASS_NAMES = (1U << 9),
398
399 /** Skip class names when not needed.
400 *
401 * If set, the class names won't be written if they are equal to the
402 * actual type of the field (missing class names are supported by the
403 * unpacker in that case).
404 */
405 IOP_JPACK_SKIP_OPTIONAL_CLASS_NAMES = (1U << 10),
406
407 /** Produce the smallest non-lossy possible json. */
408 IOP_JPACK_MINIMAL = IOP_JPACK_NO_WHITESPACES
409 | IOP_JPACK_NO_TRAILING_EOL
410 | IOP_JPACK_SKIP_DEFAULT
411 | IOP_JPACK_SKIP_EMPTY_ARRAYS
412 | IOP_JPACK_SKIP_EMPTY_STRUCTS
413 | IOP_JPACK_SKIP_OPTIONAL_CLASS_NAMES,
414};

491#define iop_jtrace(lvl, st, v) \

493 if (e_is_traced(lvl)) { \

526/** Provide the appropriate arguments to the %*pS modifier.
527 *
528 * '%*pS' can be used in format string in order to print the JSON-encoded
529 * content of an IOP stuct or union. You can provide the flags to be used to
530 * the JSON packer in the format arguments (\ref iop_jpack_flags).
531 *
532 * \param[in] pfx IOP struct descriptor prefix.
533 * \param[in] _val The IOP struct or union to print.
534 * \param[in] _flags The JSON packing flags
535 *
536 * See \ref IOP_ST_FMT_ARG() for a convenience helper to print compact JSON.
537 */
538#define IOP_ST_FMT_ARG_FLAGS(pfx, _val, _flags) \
539 (_flags), &(struct iop_struct_value){ \
540 .st = &pfx##__s, \
541 .val = ({ const pfx##__t *__val = (_val); __val; }) }
542
543/** Provide the appropriate argument to print compact JSON with the %*pS
544 * format.
545 *
546 * This macro is a convenience helper for \ref IOP_ST_FMT_ARG_FLAGS to
547 * cover the usual use case where compact JSON is needed.
548 *
549 * \param[in] pfx IOP struct descriptor prefix.
550 * \param[in] _val The IOP struct or union to print.
551 *
552 * \note If the struct is actually a class, use \ref IOP_OBJ_FMT_ARG
553 * formatting helper instead to get the code smaller.
554 */
555#define IOP_ST_FMT_ARG(pfx, _val) \
556 IOP_ST_FMT_ARG_FLAGS(pfx, _val, \
557 IOP_JPACK_NO_WHITESPACES | IOP_JPACK_NO_TRAILING_EOL)
558
559/** Same as \ref IOP_ST_FMT_ARG_FLAGS but with explicit description pointer.
560 */
561#define IOP_ST_DESC_FMT_ARG_FLAGS(desc, _val, _flags) \
562 (_flags), &(struct iop_struct_value){ \
563 .st = desc, \
564 .val = (_val) }
565
566/** Provide the appropriate arguments to print an IOP class instance in JSON
567 * with the %*pS modifier.
568 *
569 * '%*pS' can be used in format string in order to print the JSON-encoded
570 * content of an IOP object (instance of an IOP class). You can provide the
571 * flags to be used to the JSON packer in the format arguments
572 * (\ref iop_jpack_flags).
573 *
574 * \param[in] _obj The object to print.
575 * \param[in] _flags The JSON packing flags
576 *
577 * See \ref IOP_OBJ_FMT_ARG() for a convenience helper to print compact JSON.
578 */
579#define IOP_OBJ_FMT_ARG_FLAGS(_obj, _flags) \
580 (_flags), \
581 &(struct iop_struct_value){ \
582 .st = NULL, \
583 .val = ({ \
584 typeof(*_obj) *_fmt_obj = (_obj); \
585 __unused__ const iop_struct_t *__ofa_st = _fmt_obj->__vptr; \
586 _fmt_obj; \
587 }) \
588 }
589
590/** Provide the appropriate argument to print an IOP class instance in compact
591 * JSON with the %*pS modifier.
592 *
593 * This macro is a convenience helper for \ref IOP_OBJ_FMT_ARG_FLAGS to
594 * cover the usual use case where compact JSON is needed.
595 *
596 * \param[in] _obj The object to print.
597 */
598#define IOP_OBJ_FMT_ARG(_obj) \
599 IOP_OBJ_FMT_ARG_FLAGS((_obj), \
600 IOP_JPACK_NO_WHITESPACES | IOP_JPACK_NO_TRAILING_EOL)

628/** Check the validity of the JSON examples contained in an IOP package.
629 *
630 * IOP language supports providing examples of objects instanciation, using
631 * the "example" doxygen tag. Each provided example must be a valid json
632 * representation of the documented object.
633 *
634 * However, iopc only checks that each provided example is valid json, but not
635 * that it actually corresponds to the documented object (this would require
636 * having the iop_struct_t and the json unpacker, which is not possible in
637 * iopc).
638 *
639 * The purpose of this function is to check the consistency of the examples
640 * contained in a package, at runtime.
641 *
642 * \param[in] pkg the package to check.
643 * \param[out] err buffer to fill in case of error.
644 *
