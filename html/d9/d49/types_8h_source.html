38#define REFCNT_NEW(type, pfx) \
39 __unused__ static inline __attribute__((malloc)) \
40 type *nonnull pfx##_new(void) \
41 { \
42 type *res = pfx##_init(p_new_raw(type, 1)); \
43 res->refcnt = 1; \
44 return res; \
45 }
46
47#define REFCNT_RETAIN(type, pfx) \
48 __unused__ static inline __attr_nonnull__((1)) \
49 type *nonnull pfx##_retain(type *nonnull t) \
50 { \
51 if (unlikely(t->refcnt < 1)) { \
52 e_panic("memory corruption: dead object revival detected"); \
53 } \
54 t->refcnt++; \
55 return t; \
56 }
57
58#define REFCNT_RELEASE(type, pfx) \
59 __unused__ static inline void __attr_nonnull__((1)) \
60 pfx##_release(type *nullable *nonnull tp) \
61 { \
62 type * const t = *tp; \
63 \
64 if (t) { \
65 if (unlikely(t->refcnt <= 0)) { \
66 e_panic("memory corruption: double free detected"); \
67 } else \
68 if (!--t->refcnt) { \
69 pfx##_wipe(t); \
70 assert (likely(t == *tp) && "pointer corruption detected"); \
71 p_delete(tp); \
72 } \
73 } \
74 }
75
76#define REFCNT_DELETE(type, pfx) \
77 __unused__ static inline void __attr_nonnull__((1)) \
78 pfx##_delete(type *nullable *nonnull tp) \
79 { \
80 if (*tp) { \
81 pfx##_release(tp); \
82 *tp = NULL; \
83 } \
84 }

109/** Initialize an optional field. */
110#define OPT(val) { .v = (val), .has_field = true }
111/** Initialize an optional field to “absent”. */
112#define OPT_NONE { .has_field = false }
113/** Initialize an optional field if `cond` is fulfilled. */
114#define OPT_IF(cond, val) { .has_field = (cond), .v = (cond) ? (val) : 0 }
115
116/** Tell whether the optional field is set or not. */
117#define OPT_ISSET(_v) ((_v).has_field == true)
118
119/** Tell whether the optional field is set and checks a given condition. */
120#define OPT_TEST(_opt, _op, _val) \
121 ({ \
122 typeof(_opt) opt_is_val_ = (_opt); \
123 OPT_ISSET(opt_is_val_) && OPT_VAL(opt_is_val_) _op (_val); \
124 })
125
126/** Thell whether the optional field is set with a given value. */
127#define OPT_IS(_op, _val) OPT_TEST((_op), ==, (_val))
128
129/** Get the optional field value. */
130#define OPT_VAL_P(_v) \
131 ({ \
132 typeof(_v) __p_opt = (_v); \
133 assert (OPT_ISSET(*__p_opt)); \
134 &__p_opt->v; \
135 })
136#define OPT_VAL(_v) ({ typeof(_v) _opt = (_v); *OPT_VAL_P(&(_opt)); })
137
138/** Get the value of an optional field, use a default value if not set. */
139#define OPT_DEFVAL(_v, _defval) \
140 ({ \
141 typeof(_v) opt_defval_val_ = (_v); \
142 opt_defval_val_.has_field ? opt_defval_val_.v : (_defval); \
143 })
144
145/** Same as \p OPT_DEFVAL but the default value is zero. */
146#define OPT_DEF0(_v) OPT_DEFVAL(_v, 0)
147
148#define OPT_GET(_v) \
149 ({ typeof(_v) __v = (_v); __v->has_field ? &__v->v : NULL; })
150
151/** Set the optional field value. */
152#define OPT_SET(dst, val) \
153 ({ typeof(dst) *_dst = &(dst); _dst->v = (val); _dst->has_field = true; })
154/** Clear the optional field value. */
155#define OPT_CLR(dst) (void)((dst).has_field = false)
156/** Set the optional field value if `cond` is fulfilled. */
157#define OPT_SET_IF(dst, cond, val) \
158 ({ if (cond) { \
159 OPT_SET(dst, val); \
160 } else { \
161 OPT_CLR(dst); \
162 } \
163 })
164/** Clear the optional field value if `cond` is fulfilled. */
165#define OPT_CLR_IF(dst, cond) \
166 do { \
167 if (cond) { \
168 OPT_CLR(dst); \
169 } \
170 } while (0)
171
172/** Copy `src` in `dst`. */
173#define OPT_COPY(dst, src) \
174 do { \
175 typeof(src) *_src = &(src); \
176 \
177 OPT_SET_IF((dst), OPT_ISSET(*_src), OPT_VAL(*_src)); \
178 } while (0)
179
180/** Get whether 2 optional fields are equal are not.
181 *
182 * Optional fields are equal if:
183 * - they are both unset;
184 * - or, they are both set and their values are equal.
185 * Otherwise optional fields are different.
186 */
187#define OPT_EQUAL(v, w) \
188 ({ \
189 typeof(v) _v = (v); \
190 typeof(w) _w = (w); \
191 \
192 (OPT_ISSET(_v) == OPT_ISSET(_w) \
193 && (!OPT_ISSET(_v) || OPT_VAL(_v) == OPT_VAL(_w))); \
194 })
