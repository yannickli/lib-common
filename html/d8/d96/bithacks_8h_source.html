92/** Reverses the bit order of any unsigned integer:
93 * 16 bits example:
94 * 0x3445 -> 0xa22c
95 * 0011010001000101 -> 1010001000101100
96 */
97
98extern uint8_t const __bit_reverse8[1 << 8];
99
100static ALWAYS_INLINE uint8_t bit_reverse8(uint8_t u8) {
101 return __bit_reverse8[u8];
102}
103
104static ALWAYS_INLINE uint16_t bit_reverse16(uint16_t u16) {
105 return ((uint16_t)bit_reverse8(u16) << 8) | bit_reverse8(u16 >> 8);
106}
107
108static ALWAYS_INLINE uint32_t bit_reverse32(uint32_t u32) {
109 return ((uint32_t)bit_reverse16(u32) << 16) | bit_reverse16(u32 >> 16);
110}
111
112static ALWAYS_INLINE uint64_t bit_reverse64(uint64_t u64) {
113 return ((uint64_t)bit_reverse32(u64) << 32) | bit_reverse32(u64 >> 32);
114}
