23static ALWAYS_INLINE int64_t
24memtoip_impl(const byte *s, int _len, const byte **endp,
25 const int64_t min, const int64_t max, bool ll, bool use_len)
26{
27 int64_t value = 0;
28
29#define len ((use_len) ? _len : INT_MAX)
30#define declen ((use_len) ? --_len : INT_MAX)
31
32 if (!s || len < 0) {
33 errno = EINVAL;
34 goto done;
35 }
36 while (len && isspace((unsigned char)*s)) {
37 s++;
38 (void)declen;
39 }
40 if (!len) {
41 errno = EINVAL;
42 goto done;
43 }
44 if (*s == '-') {
45 s++;
46 (void)declen;
47 if (!len || !isdigit((unsigned char)*s)) {
48 errno = EINVAL;
49 goto done;
50 }
51 value = '0' - *s++;
52 while (declen && isdigit((unsigned char)*s)) {
53 int digit = '0' - *s++;
54 if ((value <= min / 10)
55 && (value < min / 10 || digit < min % 10)) {
56 errno = ERANGE;
57 value = min;
58 /* keep looping inefficiently in case of overflow */
59 } else {
60 value = value * 10 + digit;
61 }
62 }
63 } else {
64 if (*s == '+') {
65 s++;
66 (void)declen;
67 }
68 if (!len || !isdigit((unsigned char)*s)) {
69 errno = EINVAL;
70 goto done;
71 }
72 value = *s++ - '0';
73 while (declen && isdigit((unsigned char)*s)) {
74 int digit = *s++ - '0';
75 if ((value >= max / 10)
76 && (value > max / 10 || digit > max % 10)) {
77 errno = ERANGE;
78 value = max;
79 /* keep looping inefficiently in case of overflow */
80 } else {
81 value = value * 10 + digit;
82 }
83 }
84 }
85done:
86 if (endp) {
87 *endp = s;
88 }
89 return value;
90#undef len
91#undef declen
92}

110#define INVALID_NUMBER INT64_MIN
111int64_t parse_number(const char *str)
112{
113 int64_t value;
114 int64_t mult = 1;
115 int frac = 0;
116 int denom = 1;
117 int exponent;
118
119 value = strtoll(str, &str, 0);
120 if (*str == '.') {
121 for (str++; isdigit((unsigned char)*str); str++) {
122 if (denom <= (INT_MAX / 10)) {
123 frac = frac * 10 + *str - '0';
124 denom *= 10;
125 }
126 }
127 }
128 switch (*str) {
129 case 'P':
130 mult <<= 10;
131 /* FALL THRU */
132 case 'T':
133 mult <<= 10;
134 /* FALL THRU */
135 case 'G':
136 mult <<= 10;
137 /* FALL THRU */
138 case 'M':
139 mult <<= 10;
140 /* FALL THRU */
141 case 'K':
142 mult <<= 10;
143 str++;
144 break;
145 case 'p':
146 mult *= 1000;
147 /* FALL THRU */
148 case 't':
149 mult *= 1000;
150 /* FALL THRU */
151 case 'g':
152 mult *= 1000;
153 /* FALL THRU */
154 case 'm':
155 mult *= 1000;
156 /* FALL THRU */
157 case 'k':
158 mult *= 1000;
159 str++;
160 break;
161 case 'E':
162 case 'e':
163 exponent = strtol(str + 1, &str, 10);
164 for (; exponent > 0; exponent--) {
165 if (mult > (INT64_MAX / 10))
166 return INVALID_NUMBER;
167 mult *= 10;
168 }
169 break;
170 }
171 if (*str != '\0') {
172 return INVALID_NUMBER;
173 }
174 /* Catch most overflow cases */
175 if ((value | mult) > INT32_MAX && INT64_MAX / mult > value) {
176 return INVALID_NUMBER;
177 }
178 return value * mult + frac * mult / denom;
179}

189uint64_t memtoullp(const void *s, int len, const byte **endp)
190{
191 t_scope;
192 const char *str = t_dupz(s, len);
193 const char *tail;
194 uint64_t res;
195
196 if (!len) {
197 errno = EINVAL;
198 return 0;
199 }
200
201 res = strtoull(str, &tail, 10);
202 if ((int64_t)res < 0 && mem_startswith_minus(str, tail - str)) {
203 errno = ERANGE;
204 return 0;
205 }
206 if (endp) {
207 *endp = (const byte *)s + (tail - str);
208 }
209
210 return res;
211}

213/** Parses a string to extract a long, checking some constraints.
214 * <code>res</code> points to the destination of the long value
215 * <code>p</code> points to the string to parse
216 * <code>endp</code> points to the end of the parse (the next char to
217 * parse, after the value. spaces after the value are skipped if
218 * STRTOLP_IGNORE_SPACES is set)
219 * <code>min</code> and <code>max</code> are extrema values (only checked
220 * if STRTOLP_CHECK_RANGE is set.
221 * <code>dest</code> of <code>size</code> bytes.
222 *
223 * If STRTOLP_IGNORE_SPACES is set, leading and trailing spaces are
224 * considered normal and skipped. If not set, then even leading spaces
225 * lead to a failure.
226 *
227 * If STRTOLP_CHECK_END is set, the end of the value is supposed to
228 * be the end of the string.
229 *
230 * @returns 0 if all constraints are met. Otherwise returns a negative
231 * value corresponding to the error
232 */
233int strtolp(const char *p, const char **endp, int base, long *res,
234 int flags, long min, long max)
235{
236 const char *end;
237 long value;
238 bool clamped;
239
240 if (!endp) {
241 endp = &end;
242 }
243 if (!res) {
244 res = &value;
245 }
246 if (flags & STRTOLP_IGNORE_SPACES) {
247 p = skipspaces(p);
248 } else {
249 if (isspace((unsigned char)*p))
250 return -EINVAL;
251 }
252 errno = 0;
253 *res = strtol(p, endp, base);
254 if (flags & STRTOLP_IGNORE_SPACES) {
255 *endp = skipspaces(*endp);
256 }
257 if ((flags & STRTOLP_CHECK_END) && **endp != '\0') {
258 return -EINVAL;
259 }
260 if (*endp == p && !(flags & STRTOLP_EMPTY_OK)) {
261 return -EINVAL;
262 }
263 clamped = false;
264 if (flags & STRTOLP_CLAMP_RANGE) {
265 if (*res < min) {
266 *res = min;
267 clamped = true;
268 } else
269 if (*res > max) {
270 *res = max;
271 clamped = true;
272 }
273 if (errno == ERANGE)
274 errno = 0;
275 }
276 if (errno) {
277 return -errno; /* -ERANGE or maybe EINVAL as checked by strtol() */
278 }
279 if (flags & STRTOLP_CHECK_RANGE) {
280 if (clamped || *res < min || *res > max) {
281 return -ERANGE;
282 }
283 }
284 return 0;
285}

289static
290int str_read_number_extension(const void **p, int len, uint64_t *out)
291{
292 uint64_t mult = 1;
293 const char *s = *(const char **)p;
294 int res = 0;
295
296 if (!len)
297 goto end;
298
299 switch (*s) {
300 /* times */
301 case 'w':
302 mult *= 7;
303 /* FALLTHROUGH */
304 case 'd':
305 mult *= 24;
306 /* FALLTHROUGH */
307 case 'h':
308 mult *= 60;
309 /* FALLTHROUGH */
310 case 'm':
311 mult *= 60;
312 /* FALLTHROUGH */
313 case 's':
314 mult *= 1;
315 res = 1;
316 break;
317
318 /* sizes */
319 case 'T':
320 mult *= 1024;
321 /* FALLTHROUGH */
322 case 'G':
323 mult *= 1024;
324 /* FALLTHROUGH */
325 case 'M':
326 mult *= 1024;
327 /* FALLTHROUGH */
328 case 'K':
329 mult *= 1024;
330 res = 1;
331 break;
332
333 default:
334 if (isalpha(*s))
335 return -1;
336 goto end;
337 }
338
339 *p = ++s;
340
341 if (--len && isalnum(*s))
342 return -1;
343
344 end:
345 *out = mult;
346 return res;
347}

349static int
350str_apply_number_extension(uint64_t mult, bool is_signed, uint64_t *number)
351{
352 if (is_signed) {
353 int64_t signed_number = *(int64_t *)number;
354
355 /* XXX: here mult <= INT64_MAX by implementation */
356
357 if (signed_number > INT64_MAX / (int64_t)mult) {
358 *number = INT64_MAX;
359 return -1;
360 }
361
362 if (signed_number < INT64_MIN / (int64_t)mult) {
363 *number = INT64_MIN;
364 return -1;
365 }
366 } else
367 if (*number > UINT64_MAX / mult) {
368 *number = UINT64_MAX;
369 return -1;
370 }
371
372 *number *= mult;
373
374 return 0;
375}

377static int
378memtoxll_ext(const void *p, int len, bool is_signed, uint64_t *out,
379 const void **endp, int base, bool use_len)
380{
381 t_scope;
382 uint64_t mult = 1;
383 int ext_count;
384 const char *s = p;
385 const char *tail;
386 const void *local_endp;
387 int res;
388
389 if (!endp)
390 endp = &local_endp;
391
392 *endp = p;
393 errno = 0;
394
395 if (use_len) {
396 char last_char;
397
398 if (len < 0) {
399 errno = EINVAL;
400 return -1;
401 }
402 if (!len) {
403 *out = 0;
404 return 0;
405 }
406
407 last_char = s[len - 1];
408 if (isalnum(last_char) || last_char == '+' || last_char == '-') {
409 s = t_dupz(p, len);
410 }
411 } else {
412 len = INT_MAX;
413 }
414
415 if (is_signed) {
416 *out = strtoll(s, &tail, base);
417 } else {
418 uint64_t val = strtoull(s, &tail, base);
419
420 if ((int64_t)val < 0 && mem_startswith_minus(s, tail - s)) {
421 errno = ERANGE;
422 return -1;
423 }
424 *out = val;
425 }
426
427 res = tail - s;
428 *endp = (const char *)p + res;
429
430 if (res == 0) {
431 errno = EINVAL;
432 }
433 if (errno)
434 return -1;
435
436 len -= res;
437
438 if ((ext_count = str_read_number_extension(endp, len, &mult)) < 0) {
439 errno = EDOM;
440 return -1;
441 }
442 res += ext_count;
443
444 if (str_apply_number_extension(mult, is_signed, out) < 0) {
445 errno = ERANGE;
446 return -1;
447 }
448
449 return res;
450}
