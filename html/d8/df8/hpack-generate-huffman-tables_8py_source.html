20""" This script generates the static Huffman encoding/decoding tables for the
21HPACK algorithm (of RFC-7541) as c arrays in autogenerated c files. The script
22accepts as parameter the text of RFC-7541 which can be shortened to the
23appendix section. For encoding, it generates the char-to-huffman-string
24mapping. For decoding, it generates the decoder's state-transition table
25based on input chunks of 1 or 2 or 4 bits at a time. This script is meant to
26generate the '.c' files during the build step and accepts as parameter
27a preamble text that '#include's a common '.h' file. For example:
28
29file: hpack-priv.h
30=========================================================================
31
32// Type definition of the encoder's table:
33// The name of this type is passed as parameter to the script.
34// Respect the definition order, and use a compatible type.
35// For example:
36typedef struct hpack_huffcode_t {
37 uint32_t codeword;
38 unsigned bitlen;
39} hpack_huffcode_t;
40
41
42// Type definition of the decoder's state-transition table.
43// The name of this type is passed as parameter to the script.
44// Respect definition order, and use a compatible type.
45// For example:
46typedef struct hpack_huffdec_trans_t {
47 uint8_t state;
48 uint8_t sym;
49 unsigned emitter : 1;
50 unsigned final : 1;
51 unsigned error : 1;
52} hpack_huffdec_trans_t;
53
54// The declaration of tables
55extern const hpack_huffcode_t hpack_huffcode_tab_g[256 + 1];
56
57// 16 is compatible with input chunks of 4 bits.
58extern const hpack_huffdec_trans_t hpack_huffdec_trans_tab_g[256][16];
59========================================================================
60
61See 'options' for further details on command line paramters.
62
63Note: default paramters are the ones used in building lib-common.
64"""

93def get_huffman_code_table_from(rfc_fn):
94 def extract_huffman_code_table_text(rfc_fn):
95 with open(rfc_fn, mode="r", encoding='utf-8') as f:
96 lines = ''.join(f.readlines())
97
98 regex = r"\n(Appendix B\. Huffman Code.+)\nAppendix"
99 match = next(re.finditer(regex, lines, re.IGNORECASE | re.DOTALL))
100
101 return match.group(1)
102
103 huff_code_txt = extract_huffman_code_table_text(rfc_fn)
104 regex = (r"\s*.*\‍(\s*(\d+)\s*\‍)\s*(\|[01\|]+)\s*([\da-fA-f]+)"
105 r"\s*\[\s*(\d+)\s*\]")
106 matches = re.finditer(regex, huff_code_txt)
107
108 code_table = {}
109
110 for match in matches:
111 sym = int(match.group(1))
112 code_str = str(match.group(2)).replace("|", "")
113 code = int(match.group(3), 16)
114 bitlen = int(match.group(4))
115 code_table[sym] = [sym, code, bitlen, code_str]
116
117 for i in range(257):
118 assert(len(code_table[i]) == 4)
119
120 return [code_table[i] for i in range(257)]

150class Tree(dict):
151 nodes, sealed = [], False
152
153 def __missing__(self, key):
154 assert(not self.sealed)
155 if key not in ('*', 0, 1):
156 raise KeyError
157 value = self[key] = type(self)()
158 value['sym'] = None
159 value['prefix'] = '' if key == '*' else self['prefix'] + str(key)
160 value['final'] = ('1' * 7).startswith(value['prefix'])
161 self.nodes.append(value)
162 return value

164 @classmethod
165 def seal(cls):
166 cls.sealed = True

169def get_huffman_tree(code_table):
170 root = Tree()['*']
171 for sym in range(257):
172 code_str = code_table[sym][3]
173
174 node = root
175 for c in (int(ch) for ch in code_str):
176 node = node[c]
177 node['sym'] = sym
178 assert(code_str == node['prefix'])
179
180 root.seal()
181 return root

184def walk_from(root, node, code_str):
185 sym, error, final, current = [], False, False, node
186 for bit in (int(ch) for ch in code_str):
187 # non leaf node that has two children [0] [1]
188 assert(current['sym'] is None and bit in current)
189 current = current[bit]
190 if current['sym'] is not None:
191 # reached a leaf node : emit sym and continue from the root
192 if current['sym'] != HUFFMAN_EOS: # EOS
193 sym.append(current['sym'])
194 else:
195 error = True
196 current = root
197
198 final = not error and current['final']
199 n_next = current
200 return (sym, final, error, n_next)

204 def new_state(node):
205 node['IDX'] = len(st_tab)
206 st_tab.append(
207 [{'next': None, 'sym': [], 'final': False, 'error': False}
208 for _ in range(inputs)])

214 total = errors = finals = 0
215

234 st_tab[idx][chunk_code]['next'] = n_next

240def compact_prefix(prefix):
241 if prefix == '':
242 return 'START'
243
244 le = len(prefix)
245 if le < 4:
246 label = prefix
247 else:
248 nibbles = le // 4
249 val = int(prefix[0:nibbles * 4], 2)
250 prefix_nibbles = prefix[nibbles * 4:]
251 label = "{val:0{nibbles}X}:{prefix_nibbles:s}".format(
252 val=val, nibbles=nibbles, prefix_nibbles=prefix_nibbles,
253 )
254
255 return label

258def get_huffman_st_table_as_c_array(rfc_fn, tbl_elem_t, tbl_name, chunk_bits):
259 def get_state_transitions_row(idx, state):
260 cols = []
261 for chunk_vec in product('01', repeat=chunk_bits):
262 chunk_str = ''.join(chunk_vec)
263 chunk_code = int(chunk_str, 2)
264
265 trans = state[chunk_code]
266 assert(len(trans['sym']) < 2)
267
268 next_idx, emitter, = trans['next']['IDX'], len(trans['sym']) > 0
269 final, error = trans['final'], trans['error']
270
271 sym = trans['sym'][0] if emitter else 0
272 prefix = compact_prefix(trans['next']['prefix'])
273
274 cols.append("/* %s */ {/* %-10s */ %3d, %3d, %d, %d, %d}" %
275 (chunk_str, prefix, next_idx, sym, emitter,
276 final, error))
277
278 return " {\n " + ",\n ".join(cols) + "\n }"
