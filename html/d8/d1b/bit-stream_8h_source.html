25/*
26 * bit_stream_t's are basically the two bit-wise bounds in a memory chunk.
27 *
28 * They are very similar to pstreams.
29 */
30
31/*
32 * In order to avoid useless arithmetics, bitstreams use chunks of 64bits and
33 * always keep aligned pointers. Properly used this will never trigger crashes
34 * since the memory atom for the kernel is a page aligned on page size that is
35 * itself a multiple of 64bits.
36 */
37
38struct bit_ptroff {
39 union {
40 const byte *b;
41 const uint64_t *p;
42 };
43 size_t offset;
44};
45
46typedef struct bit_stream_t {
47 struct bit_ptroff s;
48 struct bit_ptroff e;
49} bit_stream_t;
50
51/* Helpers {{{ */
52
53static inline void bit_ptroff_add(struct bit_ptroff *s, size_t offset)
54{
55 s->offset += offset;
56 if (s->offset >= 64) {
57 s->p += s->offset / 64;
58 s->offset %= 64;
59 }
60}
61
62static inline void bit_ptroff_sub(struct bit_ptroff *s, size_t offset)
63{
64 ssize_t off = s->offset;
65
66 off -= offset;
67 if (off < 0) {
68 s->p -= DIV_ROUND_UP(-off, 64);
69 if (off % 64) {
70 s->offset = 64 + (off % 64);
71 } else {
72 s->offset = 0;
73 }
74 } else {
75 s->offset = off;
76 }
77}
78
79static inline void bit_ptroff_normalize(struct bit_ptroff *s)
80{
81 const byte *p = (const byte *)s->p;
82 const byte *a = (const byte *)(((uintptr_t)p) & ~7ul);
83
84 if (a != p) {
85 s->offset += (p - a) * 8;
86 }
87 s->p = (const uint64_t *)a;
88
89 bit_ptroff_add(s, 0);
90}
91
92#define BIT_PTROFF_INIT(Ptr, Offset) { { (const byte *)(Ptr) }, (Offset) }
93#define BIT_PTROFF_NORMALIZED(Ptr, Offset) ({ \
94 struct bit_ptroff __poff = BIT_PTROFF_INIT(Ptr, Offset); \
95 bit_ptroff_normalize(&__poff); \
96 __poff; \
97 })
98
99#define BIT_PTROFF_CMP(P1, P2) \
100 (CMP((P1)->p, (P2)->p) ?: CMP((P1)->offset, (P2)->offset))
101#define BIT_PTROFF_LEN(P1, P2) \
102 (((P2)->p - (P1)->p) * 64 - (P1)->offset + (P2)->offset)
103

147static inline size_t bs_len(const bit_stream_t *bs)
148{
149 return BIT_PTROFF_LEN(&bs->s, &bs->e);
150}
151
152static inline bool bs_has(const bit_stream_t *bs, size_t blen)
153{
154 return blen <= bs_len(bs);
155}
156
157static inline bool bs_has_bytes(const bit_stream_t *bs, size_t olen)
158{
159 return olen * 8 <= bs_len(bs);

162static inline bool bs_done(const bit_stream_t *bs)
163{
164 return BIT_PTROFF_CMP(&bs->s, &bs->e) >= 0;
165}

167static inline bool __bs_contains(const bit_stream_t *bs,
168 const struct bit_ptroff *p)
169{
170 return BIT_PTROFF_CMP(&bs->s, p) <= 0
171 && BIT_PTROFF_CMP(p, &bs->e) <= 0;
172}

366static inline uint64_t __bs_peek_bits(const bit_stream_t *bs, size_t blen)
367{
368 uint64_t res;
369
370 assert (blen <= 64);
371 if (unlikely(!blen))
372 return 0;
373
374 assert (bs_has(bs, blen));
375
376 if (bs->e.p == bs->s.p) {
377 mem_tool_allow_memory(bs->s.p, 8, true);
378 }
379 res = *bs->s.p >> bs->s.offset;
380 if (bs->s.offset + blen > 64) {
381 if (bs->e.p == &bs->s.p[1]) {
382 mem_tool_allow_memory(&bs->s.p[1], 8, true);
383 }
384 res |= bs->s.p[1] << (64 - bs->s.offset);
385 }
386 if (blen != 64) {
387 res &= BITMASK_LT(uint64_t, blen);
388 }
389 return res;
390}

392static inline uint64_t __bs_get_bits(bit_stream_t *bs, size_t blen)

394 uint64_t res = __bs_peek_bits(bs, blen);
395 __bs_skip(bs, blen);
396 return res;
397}

472static inline uint64_t __bs_be_peek_bits(const bit_stream_t *bs, size_t blen)
473{
474 const byte *b = &((const byte *)bs->s.p)[bs->s.offset / 8];
475 size_t offset = bs->s.offset % 8;
476 size_t remain = blen;
477 uint64_t res = 0;
478
479 if (offset + blen <= 8) {
480 res = (*b >> (8 - (offset + blen))) & BITMASK_LT(uint64_t, blen);
481 return res;
482 }
483 if (offset) {
484 remain -= 8 - offset;
485 res |= ((uint64_t)*b << remain);
486 if (blen != 64) {
487 res &= BITMASK_LT(uint64_t, blen);
488 }
489 b++;
490 }
491 while (remain >= 8) {
492 remain -= 8;
493 res |= ((uint64_t)*b << remain);
494 b++;
495 }
496 if (remain) {
497 res |= (*b >> (8 - remain)) & BITMASK_LT(uint64_t, remain);
498 }
499 return res;
500}

504 uint64_t res = __bs_be_peek_bits(bs, blen);

506 return res;
507}

564static inline int __bs_scan_forward(const bit_stream_t *bs, bool b,
565 struct bit_ptroff *poff,
566 ssize_t max_len)
567{
568 size_t pos;
569
570 max_len = (max_len < 0) ? bs_len(bs) : MIN(bs_len(bs), (size_t)max_len);
571 pos = RETHROW(bsf(bs->s.p, bs->s.offset, max_len, !b));
572
573 *poff = BIT_PTROFF_NORMALIZED(bs->s.p, bs->s.offset + pos);
574 return 0;
575}

609static inline int __bs_scan_reverse(const bit_stream_t *bs, bool b,
610 struct bit_ptroff *poffp,
611 ssize_t max_len)
612{
613 size_t pos, len = bs_len(bs);
614 struct bit_ptroff poff = bs->s;
615

617 bit_ptroff_add(&poff, MAX(0, (ssize_t)len - max_len));
618 len = MIN(len, (size_t)max_len);

620 pos = BS_CHECK(bsr(poff.p, poff.offset, len, !b));
621
622 *poffp = BIT_PTROFF_NORMALIZED(poff.p, poff.offset + pos);
623 return 0;
