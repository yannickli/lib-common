172/** Extend a relative path from a prefix.
173 *
174 * prefix + fmt = prefix + / + fmt
175 *
176 * A '/' will be added between /p prefix and /p fmt if necessary.
177 *
178 * If /p fmt begins with a '/', the prefix will be ignored.
179 * If /p fmt begins with '~/', the prefix will be ignored, and ~ replaced by
180 * the HOME prefix of the current user.
181 *
182 * \param[out] buf buffer where the path will be written.
183 * \param[in] prefix
184 * \param[in] fmt format of the suffix
185 * \param[in] args va_list containing the arguments for the \p fmt
186 * formatting
187 *
188 * \return -1 if the path overflows the buffer,
189 * length of the resulting path otherwise.
190 */
191__attribute__((format(printf, 3, 0)))
192int path_va_extend(char buf[static PATH_MAX], const char * nonnull prefix,
193 const char * nonnull fmt, va_list args);
194__attribute__((format(printf, 3, 4)))
195int path_extend(char buf[static PATH_MAX], const char * nonnull prefix,
196 const char * nonnull fmt, ...);

198/** Create a relative path from one path to another path.
199 *
200 * \p from and \p to can be absolute or relative paths from the current
201 * working directory.
202 * If \p from ends with a '/', it is treated as a directory and the created
203 * relative path startpoint is within the directory.
204 *
205 * Examples:
206 * - The relative path from "/a/b/c/d" to "/a/b/e/f" is "../e/f".
207 * - The relative path from "/a/b/c/d/" to "/a/b/e/f" is "../../e/f".
208 * - The relative path from "a/b/c" to "d/e/" is "../../d/e".
209 * - The relative path from "a/b/c/" to "a/b/c" is "c".
210 * - The relative path from "/a/b/c" to "d/e" is "../d/e" with the current
211 * working directory set to "/a".
212 *
213 * \param[out] buf buffer where the path will be written.
214 * \param[in] len length of the buffer.
215 * \param[in] from the path that defines the startpoint of the relative path.
216 * \param[in] to the path that defines the endpoint of the relative path.
217 *
218 * \return -1 if the length of the absolute path of \p from or \p to, or the
219 * length of the resulting path exceed PATH_MAX.
220 * Length of the resulting path otherwise.
221 */
222int path_relative_to(char buf[static PATH_MAX], const char * nonnull from,
223 const char * nonnull to);
