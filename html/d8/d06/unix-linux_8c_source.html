34static unsigned int my_jiffies_to_msecs(const unsigned long j)
35{
36#if HZ <= 1000 && !(1000 % HZ)
37 return (1000 / HZ) * j;
38#elif HZ > 1000 && !(HZ % 1000)
39 return DIV_ROUND_UP(j, HZ / 1000);
40#else
41 return (j * 1000) / HZ;
42#endif
43}

75static int psinfo_read_proc(pid_t pid, sb_t *output, sb_t *buf)
76{
77 char path[PATH_MAX];
78 const char *p;
79 int pos;
80
81 snprintf(path, sizeof(path), "/proc/%d/stat", pid);
82 if (sb_read_file(buf, path) < 0) {
83 return -1;
84 }
85
86 sb_addf(output, "/proc/%d/stat information:\n", pid);
87
88 p = buf->data;
89 pos = 0;
90 while (p && *p != '\0') {
91 int i;
92 unsigned long lu;
93 long l;
94
95 switch (pos) {
96 case 0: /*pid %d The process id.*/
97 case 1: /*comm %s
98 The filename of the executable, in parentheses. This is visible whether or not the executable is swapped out.*/
99 break;
100 case 2: /*state %c
101 One character from the string "RSDZTW" where R is running, S is sleeping in an interruptible wait, D is waiting in uninterruptible disk
102 sleep, Z is zombie, T is traced or stopped (on a signal), and W is paging.*/
103 switch (*p) {
104 case 'R':
105 sb_adds(output, "State: running\n");
106 break;
107 case 'S':
108 sb_adds(output, "State: sleeping in an interruptible wait\n");
109 break;
110 case 'D':
111 sb_adds(output, "State: waiting in uninterruptible disk\n");
112 break;
113 case 'Z':
114 sb_adds(output, "State: zombie\n");
115 break;
116 case 'T':
117 sb_adds(output, "State: traced or stopped (on a signal)\n");
118 break;
119 case 'W':
120 sb_adds(output, "State: paging\n");
121 break;
122 default:
123 sb_addf(output, "State: unknown state: %c\n", *p);
124 break;
125 }
126 break;
127 case 3: /*ppid %d
128 The PID of the parent.*/
129 i = atoi(p);
130 if (i) {
131 sb_addf(output, "Parent process pid: %d\n", i);
132 }
133 break;
134 case 4: /*pgrp %d
135 The process group ID of the process.*/
136 i = atoi(p);
137 if (i) {
138 sb_addf(output, "Process group id: %d\n", i);
139 }
140 break;
141 case 5: /*session %d
142 The session ID of the process.*/
143 i = atoi(p);
144 if (i) {
145 sb_addf(output, "Session id: %d\n", i);
146 }
147 break;
148 case 6: /*tty_nr %d
149 The tty the process uses.*/
150 i = atoi(p);
151 if (i) {
152 sb_addf(output, "tty number: %d\n", i);
153 }
154 break;
155 case 7: /*tpgid %d
156 The process group ID of the process which currently owns the tty that the process is connected to.*/
157 break;
158 case 8: /*flags %lu
159 The kernel flags word of the process. For bit meanings, see the PF_* defines in <linux/sched.h>. Details depend on the kernel version.*/
160 lu = atol(p);
161#define CHECK_KERNEL_FLAG(flag, txt) \
162 if (lu & flag) { \
163 sb_adds(output, "kernel flag: " #flag "(" txt ")\n"); \
164 }
165 CHECK_KERNEL_FLAG(PF_ALIGNWARN, "Print alignment warning msgs");
166 CHECK_KERNEL_FLAG(PF_STARTING, "being created");
167 CHECK_KERNEL_FLAG(PF_EXITING, "getting shut down");
168 CHECK_KERNEL_FLAG(PF_DEAD, "Dead");
169 CHECK_KERNEL_FLAG(PF_FORKNOEXEC, "forked but did not exec");
170 CHECK_KERNEL_FLAG(PF_SUPERPRIV, "used super-user privileges");
171 CHECK_KERNEL_FLAG(PF_DUMPCORE, "dumped core");
172 CHECK_KERNEL_FLAG(PF_SIGNALED, "killed by a signal");
173 CHECK_KERNEL_FLAG(PF_MEMALLOC, "Allocating memory");
174 CHECK_KERNEL_FLAG(PF_FLUSHER, "responsible for disk writeback");
175 CHECK_KERNEL_FLAG(PF_USED_MATH, "if unset the fpu must be initialized before use");
176 CHECK_KERNEL_FLAG(PF_FREEZE, "this task is being frozen for suspend now");
177 CHECK_KERNEL_FLAG(PF_NOFREEZE, "this thread should not be frozen");
178 CHECK_KERNEL_FLAG(PF_FROZEN, "frozen for system suspend");
179 CHECK_KERNEL_FLAG(PF_FSTRANS, "inside a filesystem transaction");
180 CHECK_KERNEL_FLAG(PF_KSWAPD, "I am kswapd");
181 CHECK_KERNEL_FLAG(PF_SWAPOFF, "I am in swapoff");
182 CHECK_KERNEL_FLAG(PF_LESS_THROTTLE, "Throttle me less: I clean memory");
183 CHECK_KERNEL_FLAG(PF_BORROWED_MM, "I am a kthread doing use_mm");
184 CHECK_KERNEL_FLAG(PF_RANDOMIZE, "randomize virtual address space");
185 CHECK_KERNEL_FLAG(PF_SWAPWRITE, "Allowed to write to swap");
186 CHECK_KERNEL_FLAG(PF_SPREAD_PAGE, "Spread page cache over cpuset");
187 CHECK_KERNEL_FLAG(PF_SPREAD_SLAB, "Spread some slab caches over cpuset");
188 CHECK_KERNEL_FLAG(PF_MEMPOLICY, "Non-default NUMA mempolicy");
189 CHECK_KERNEL_FLAG(PF_MUTEX_TESTER, "Thread belongs to the rt mutex tester");
190#undef CHECK_KERNEL_FLAG
191 break;
192
193 case 9: /*minflt %lu
194 The number of minor faults the process has made which have not required loading a memory page from disk.*/
195 lu = atol(p);
196 if (lu) {
197 sb_addf(output, "minor faults the process has made: %lu\n", lu);
198 }
199 break;
200 case 10: /*cminflt %lu
201 The number of minor faults that the process's waited-for children have made.*/
202 lu = atol(p);
203 if (lu) {
204 sb_addf(output, "minor faults that the process's waited-for children have made: %lu\n", lu);
205 }
206 break;
207 case 11: /*majflt %lu
208 The number of major faults the process has made which have required loading a memory page from disk.*/
209 lu = atol(p);
210 if (lu) {
211 sb_addf(output, "major faults the process has made: %lu\n", lu);
212 }
213 break;
214 case 12: /*cmajflt %lu
215 The number of major faults that the process's waited-for children have made.*/
216 lu = atol(p);
217 if (lu) {
218 sb_addf(output, "major faults that the process's waited-for children have made: %lu\n", lu);
219 }
220 break;
221 case 13: /*utime %lu
222 The number of jiffies that this process has been scheduled in user mode.*/
223 lu = atol(p);
224 if (lu) {
225 int ms = my_jiffies_to_msecs(lu);
226 sb_addf(output, "user mode: %dms\n", ms);
227 }
228 break;
229 case 14: /*stime %lu
230 The number of jiffies that this process has been scheduled in kernel mode.*/
231 lu = atol(p);
232 if (lu) {
233 int ms = my_jiffies_to_msecs(lu);
234 sb_addf(output, "kernel mode: %dms\n", ms);
235 }
236 break;
237 case 15: /*cutime %ld
238 The number of jiffies that this process's waited-for children have been scheduled in user mode. (See also times(2).)*/
239 l = atol(p);
240 if (l) {
241 int ms = my_jiffies_to_msecs(l);
242 sb_addf(output, "this process's waited-for children user mode: %dms\n", ms);
243 }
244 break;
245 case 16: /*cstime %ld
246 The number of jiffies that this process's waited-for children have been scheduled in kernel mode.*/
247 l = atol(p);
248 if (l) {
249 int ms = my_jiffies_to_msecs(l);
250 sb_addf(output, "this process's waited-for children kernel mode: %dms\n", ms);
251 }
252 break;
253 case 17: /*priority %ld
254 The standard nice value, plus fifteen. The value is never negative in the kernel.*/
255 l = atol(p);
256 sb_addf(output, "standard nice: %ld\n", l - 15);
257 break;
258 case 18: /*nice %ld
259 The nice value ranges from 19 (nicest) to -19 (not nice to others).*/
260 l = atol(p);
261 sb_addf(output, "nice: %ld (from 19 (nicest) to -19 (not nice to others))\n", l);
262 break;
263
264 default:
265 /* ... */
266 goto end;
267 }
268
269 p = strchr(p, ' ');
270 p++; /* Skip space */
271 pos++;
272 }
273
274end:
275 return 0;
276}

278static int psinfo_read_maps(pid_t pid, sb_t *output, sb_t *buf)
279{
280 char path[PATH_MAX];
281 const char *p;
282 size_t total = 0;
283
284 snprintf(path, sizeof(path), "/proc/%d/maps", pid);
285 RETHROW(sb_read_file(buf, path));
286
287 sb_addf(output, "\n/proc/%d/maps information:\n", pid);
288 p = buf->data;
289
290 while (p && *p) {
291 const char *end, *q;
292 size_t start, stop;
293
294 end = strchr(p, '\n');
295 if (!end) {
296 break;
297 }
298
299 q = p;
300 start = strtoll(q, &q, 16);
301 if (*q != '-') {
302 sb_addf(output, "Could not parse start: %*pM\n",
303 (int)(end - p), p);
304 break;
305 }
306 q++;
307 stop = strtoll(q, &q, 16);
308 if (*q != ' ') {
309 sb_addf(output, "Could not parse stop: %*pM\n",
310 (int)(end - p), p);
311 break;
312 }
313
314 sb_addf(output, "% 12lld %*pM\n", (long long)(stop - start),
315 (int)(end - p), p);
316 total += stop - start;
317
318 p = end + 1;
319 }
320
321 if (p && *p == '\0') {
322 sb_addf(output, "total mapped size: %lld bytes\n", (long long)total);
323 }
324
325 return 0;
326}

356static void jiffies_to_tv(unsigned long long jiff, struct timeval *tv)
357{
358 if (!hertz)
359 hertz = sysconf(_SC_CLK_TCK);
360
361 tv->tv_sec = jiff / hertz;
362 tv->tv_usec = (jiff % hertz) * (1000000UL / hertz);
363}

365int pid_get_starttime(pid_t pid, struct timeval *tv)
366{
367 unsigned long long starttime;
368 FILE *pidstat = pidproc_open(pid ? pid : getpid(), "stat");
369 static int boot_time;
370 int res;
371
372 if (!pidstat)
373 return -1;
374
375 if (!hertz)
376 hertz = sysconf(_SC_CLK_TCK);
377
378 /* see proc(3), here we want starttime */
379 res = fscanf(pidstat,
380 "%*d %*s %*c %*d %*d %*d %*d %*d %*u %*u %*u %*u %*u %*u "
381 "%*u %*d %*d %*d %*d %*d 0 %llu ", &starttime);
382 p_fclose(&pidstat);
383
384 if (res < 1) {
385 errno = EINVAL;
386 return -1;
387 }
388
389 if (!boot_time) {
390 char buf[BUFSIZ];
391 const char *p;
392 FILE *f = fopen("/proc/stat", "r");
393
394 if (!f)
395 e_panic("cannot open /proc/stat");
396
397 while (fgets(buf, sizeof(buf), f)) {
398 if (strstart(buf, "btime", &p)) {
399 boot_time = strtoip(p, NULL);
400 break;
401 }
402 }
403 p_fclose(&f);
404 if (boot_time == 0)
405 e_panic("could not parse boot time");
406 }
407
408 jiffies_to_tv(starttime, tv);
409 tv->tv_sec += boot_time;
410
411 return 0;
412}

417void close_fds(int fd_min, qv_t(u32) * nullable to_keep)
418{
419 DIR *proc_fds = opendir("/proc/self/fd");
420 struct dirent *entry;
421 const char *p;
422 int opendir_fd;
423
424 if (!proc_fds) {
425 /* Fall back to unix.c implementation */
426 close_fds_unix(fd_min, to_keep);
427 return;
428 }
429
430 if (to_keep) {
431 dsort32(to_keep->tab, to_keep->len);
432 to_keep->len = uniq32(to_keep->tab, to_keep->len);
433 }
434
435 /* XXX: opendir opens a fd. Do not close it while using it */
436 opendir_fd = dirfd(proc_fds);
437 while ((entry = readdir(proc_fds)) != NULL) {
438 int fd = strtol(entry->d_name, &p, 10);
439
440 if (*p || fd == opendir_fd) {
441 continue;
442 }
443 if (fd > fd_min
444 && (!to_keep
445 || !contains32(fd, to_keep->tab, to_keep->len)))
446 {
447 p_close(&fd);
448 }
449 }
450 closedir(proc_fds);
451}
