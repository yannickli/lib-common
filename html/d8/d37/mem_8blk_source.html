54static void *libc_malloc(mem_pool_t *m, size_t size, size_t alignment,
55 mem_flags_t flags)
56{
57 void *res;
58
59 if (unlikely(size == 0)) {
60 return MEM_EMPTY_ALLOC;
61 }
62
63 if (alignment <= 8) {
64 if (flags & MEM_RAW) {
65 res = malloc(size);
66 } else {
67 res = calloc(1, size);
68 }
69 if (unlikely(res == NULL)) {
70 if (flags & MEM_ERRORS_OK) {
71 return NULL;
72 }
73 logger_panic(&libc_g.logger, "out of memory");
74 }
75 } else {
76 int ret = posix_memalign(&res, alignment, size);
77
78 if (unlikely(ret != 0)) {
79 errno = ret;
80 if (flags & MEM_ERRORS_OK) {
81 return NULL;
82 }
83 logger_panic(&libc_g.logger, "cannot allocate memory: %m");
84 }
85 if (!(flags & MEM_RAW)) {
86 memset(res, 0, size);
87 }
88 }
89 return res;
90}

92static void *libc_realloc(mem_pool_t *m, void *mem, size_t oldsize,
93 size_t size, size_t alignment, mem_flags_t flags)
94{
95 byte *res = NULL;
96
97 if (unlikely(mem == MEM_EMPTY_ALLOC)) {
98 mem = NULL;
99 }
100
101 if (unlikely(size == 0)) {
102 if (mem) {
103 free(mem);
104 }
105 return MEM_EMPTY_ALLOC;
106 }
107
108 if (alignment > 8 && mem == NULL) {
109 return libc_malloc(m, size, alignment, flags);
110 }
111
112 res = realloc(mem, size);
113
114 if (unlikely(res == NULL)) {
115 if (flags & MEM_ERRORS_OK) {
116 return NULL;
117 }
118 logger_panic(&libc_g.logger, "out of memory");
119 }
120
121 if (alignment > 8 && ((uintptr_t)res & (alignment - 1))) {
122 byte *cpy = libc_malloc(m, size, alignment, flags | MEM_RAW);
123
124 p_copy(cpy, res, oldsize == MEM_UNKNOWN ? size : oldsize);
125 free(res);
126 res = cpy;
127 }
128
129 if (!(flags & MEM_RAW) && oldsize < size)
130 memset(res + oldsize, 0, size - oldsize);
131 return res;
132}

214void core_mem_set_malloc_trim_period(int64_t period)
215{
216 int next = rand_range(0, period);
217
218 /* Call malloc_trim every "period" milliseconds in order to try releasing
219 * some memory to the kernel.
220 * Randomly pick the first call to avoid that every daemon of a product
221 * call it at the same time, causing a potential slowdown. */
222 el_unregister(&libc_g.malloc_trim_cron);
223 libc_g.malloc_trim_cron = el_timer_register(next, period, EL_TIMER_LOWRES,
224 &malloc_trim_cron, NULL);
225 el_unref(libc_g.malloc_trim_cron);
226}

298void *__mp_imalloc(mem_pool_t *mp, size_t size, size_t alignment,
299 mem_flags_t flags)
300{
301 void *res;
302
303 icheck_alloc(size);
304 mp = mp ?: &mem_pool_libc;
305 alignment = mem_bit_align(mp, alignment);
306
307 res = (*mp->malloc)(mp, size, alignment, flags);
308
309 if (unlikely(size == 0)) {
310 assert (res == MEM_EMPTY_ALLOC);
311 }
312
313 return res;
314}
315
316void *__mp_irealloc(mem_pool_t *mp, void *mem, size_t oldsize, size_t size,
317 size_t alignment, mem_flags_t flags)
318{
319 void *res;
320
321 icheck_alloc(size);
322 mp = mp ?: &mem_pool_libc;
323
324 if (!mem) {
325 return mp_imalloc(mp, size, alignment, flags);
326 }
327
328 alignment = mem_bit_align(mp, alignment);
329 if (!(flags & MEM_UNALIGN_OK)) {
330 assert ((uintptr_t)mem == mem_align_ptr((uintptr_t)mem, alignment)
331 && "reallocation must have the same alignment as allocation");
332 }
333
334 res = (*mp->realloc)(mp, mem, oldsize, size, alignment, flags);
335
336 if (unlikely(size == 0)) {
337 assert (res == MEM_EMPTY_ALLOC);
338 }
339
340 return res;
341}

349void *__mp_irealloc_fallback(mem_pool_t **pmp, void *mem, size_t oldsize,
350 size_t size, size_t alignment,
351 mem_flags_t flags)
352{
353 mem_pool_t *mp = *pmp;
354
355 assert (oldsize != MEM_UNKNOWN);
356 icheck_alloc(size);
357 mp = mp ?: &mem_pool_libc;
358
359 if (mp->realloc_fallback && size > oldsize) {
360 void *out;
361
362 mp = mp->realloc_fallback;
363 out = mp_imalloc(mp, size, alignment, flags);
364 if (oldsize != MEM_UNKNOWN) {
365 memcpy(out, mem, oldsize);
366 }
367 mp_ifree(*pmp, mem);
368 *pmp = mp;
369 return out;
370 } else {
371 return mp_irealloc(mp, mem, oldsize, size, alignment, flags);
372 }
373}

394char *mp_vfmt(mem_pool_t *mp, int *lenp, const char *fmt, va_list va)
395{
396#define MP_FMT_LEN 1024
397 char *res;
398 int len;
399 va_list cpy;
400
401 res = mp_new_raw(mp, char, MP_FMT_LEN);
402 va_copy(cpy, va);
403 len = vsnprintf(res, MP_FMT_LEN, fmt, cpy);
404 va_end(cpy);
405 if (likely(len < MP_FMT_LEN)) {
406 res = mp_irealloc(mp, res, MP_FMT_LEN, len + 1, 1, MEM_RAW);
407 } else {
408 res = mp_irealloc(mp, res, 0, len + 1, 1, MEM_RAW);
409 len = vsnprintf(res, len + 1, fmt, va);
410 }
411 if (lenp) {
412 *lenp = len;
413 }
414 return res;
415#undef MP_FMT_LEN
416}

519void mem_tool_malloclike(const void *mem, size_t len, size_t rz, bool zeroed)
520{
521 if (!mem) {
522 return;
523 }
524
525 VALGRIND_MALLOCLIKE_BLOCK(mem, len, rz, zeroed);
526 if (!len) {
527 return;
528 }
529
530 __asan_unpoison_memory_region(mem, len);

533 __asan_poison_memory_region((const byte *)mem + len, rz);
534 }
