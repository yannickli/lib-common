37/**
38 * \brief Callback when an outgoing connection succeed or failed.
39 *
40 * This callback is used on the client side only (on connections contexts
41 * returned by \fn sctp_connect()).
42 *
43 * If the connection succeed and the callback returns an error (<0) then the
44 * connection will be closed.
45 */
46typedef int (*sctp_on_connect_f)(sctp_conn_t * nonnull conn, bool success,
47 int err);
48
49/**
50 * \brief Callback when the connection is closed, to clean context.
51 *
52 * This callback is used on the active connections. This includes contexts
53 * returned by the function \fn sctp_connect() and contexts provided by the \p
54 * sctp_on_accept_f callback. This excludes the listening contexts that are
55 * returned by the \fn sctp_listen() function.
56 *
57 * The connection can be closed when an error is encountered so this might not
58 * be trigerred by the \fn sctp_conn_close() function nor by the remote
59 * endpoint disconnecting. In this case the connection will be cleaned and the
60 * \fn sctp_conn_close() function does not need to be called.
61 */
62typedef void (*sctp_on_disconnect_f)(sctp_conn_t * nonnull conn);
63
64/**
65 * \brief Callback called when a message is received.
66 *
67 * This callback is used on the active connections. This includes contexts
68 * returned by the function \fn sctp_connect() and contexts provided by the \p
69 * sctp_on_accept_f callback. This excludes the listening contexts that are
70 * returned by the \fn sctp_listen() function.
71 *
72 * This function will be called one extra time after the last message of an
73 * SCTP packet with the flag \p no_more_msgs set to \p true to allow user
74 * application to batch multiple messages received at the same time. This last
75 * call will not have data.
76 */
77typedef int (*sctp_on_data_f)(sctp_conn_t * nonnull conn,
78 const sb_t * nullable data,
79 bool no_more_msgs);
80
81/**
82 * \brief Callback when a remote entity connects to the listening socket.
83 *
84 * This callback is used on listening contexts. This includes the contexts
85 * returned by the \fn sctp_listen() function and this excludes contexts
86 * returned by the \fn sctp_connect() function and the \p sctp_on_accept_f
87 * callbacks.
88 *
89 * The provided connection's context must be cleaned eventually. If the remote
90 * end disconnect first or upon errors, the connection will be cleaned by the
91 * library and it's disconnection callback will be called. Otherwise the user
92 * must call \fn sctp_conn_close().
93 */
94typedef int (*sctp_on_accept_f)(sctp_conn_t * nonnull conn);

128/**
129 * \brief Listen to incoming SCTP connections.
130 *
131 * This function will return a listening context.
132 *
133 * The returned context needs to be deallocated eventually.
134 *
135 * The purpose of the returrned context is to listen for new connections. It
136 * doesn't allow the user to send messages and the disconnection callback will
137 * not be called for this context.
138 *
139 * \param[in] addrs addresses to listen to, multiple IPs means multi-homing
140 * \param[in] port the port to listen to
141 * \param[in] sndbuf maximum size of sending buffer (man 7 socket, SO_SNDBUF)
142 * \param[in] entity_id the name of the remote endpoint (see sctp_conn_t)
143 * \param[in] priv user context that will be accessible from callbacks
144 * \param[in] on_accept_cb function called when a new connection is received
145 * \param[in] on_data_cb callback that will be provided to new connections
146 * \param[in] on_disconnect_cb callback provided to new connections
147 *
148 * \retval NULL in case of failure
149 * \retval a pointer to a sctp_conn_t
150 */
151sctp_conn_t * nullable
152sctp_listen(const lstr__array_t * nonnull addrs,
153 int port, size_t sndbuf, lstr_t entity_id, void * nullable priv,
154 sctp_on_accept_f nonnull on_accept_cb,
155 sctp_on_data_f nonnull on_data_cb,
156 sctp_on_disconnect_f nullable on_disconnect_cb);
157
158/**
159 * \brief Connect to an SCTP endpoint.
160 *
161 * This function begins to establish a connection with the remote. The
162 * connection is not established at the end of the function. When it will be
163 * established, the \p on_connect_cb function will be called with its status
164 * (successfully connected or failed to connect).
165 *
166 * The returned pointer will be necessary to send messages and/or close the
167 * connection.
168 *
169 * \param[in] source_addrs the IPs used as source, can be empty
170 * \param[in] dest_addrs the destination IPs, multiple IPs means multi-homing
171 * \param[in] port the port to connect to on the remote end
172 * \param[in] sndbuf maximum size of sending buffer (man 7 socket, SO_SNDBUF)
173 * \param[in] entity_id the name of the remote endpoint (see sctp_conn_t)
174 * \param[in] priv user context that will be accessible from callbacks
175 * \param[in] on_connect_cb function called on connection success/failure
176 * \param[in] on_data_cb function called when data is received
177 * \param[in] on_disconnect_cb function called when connection ends
178 *
179 * \retval NULL in case of failure
180 * \retval a pointer to a sctp_conn_t
181 */
182sctp_conn_t * nullable
183sctp_connect(qv_t(lstr) source_addrs,
184 qv_t(lstr) dest_addrs,
185 uint16_t port, size_t sndbuf, lstr_t entity_id,
186 void * nullable priv,
187 sctp_on_connect_f nonnull on_connect_cb,
188 sctp_on_data_f nonnull on_data_cb,
189 sctp_on_disconnect_f nullable on_disconnect_cb);
190
191/**
192 * \brief Enqueue a message to be sent.
193 *
194 * Note that sending is asynchronous. The message will be sent when the socket
195 * allows it.
196 *
197 * \param[in] conn the connection context
198 * \param[in] payload the message's content
199 * \param[in] payload_protocol_id defined in the protocol specification
200 */
201void sctp_send_msg(sctp_conn_t * nonnull conn, lstr_t payload,
202 uint32_t payload_protocol_id);
203
204/**
205 * \brief Close the connection.
206 *
207 * This is required when the connection is not automatically closed:
208 *
209 * - If the remote end did not close the connection first;
210 *
211 * - If no error were previously detected, in which case the connection
212 * would have closed itself;
213 *
214 * \param[in] conn the connection context
215 */
216void sctp_conn_close(sctp_conn_t * nullable * nullable pconn);
