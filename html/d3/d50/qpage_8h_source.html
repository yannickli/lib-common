24/* ACHTUNG MINEN: QPAGE_SIZE must be an unsigned literal to force unsigned
25 * arithmetics in expressions, and an uintptr_t to allow dirty masking against
26 * pointers.
27 */
28#define QPAGE_SHIFT 12U
29#define QPAGE_SIZE ((uintptr_t)1 << QPAGE_SHIFT)
30#define QPAGE_MASK (QPAGE_SIZE - 1)
