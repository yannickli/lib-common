47static inline int opt_state_end(popt_state_t *st)
48{
49 memmove(st->left_argv + st->left_argc, st->pending_argv,

59static const char *opt_arg(popt_state_t *st)
60{
61 if (st->p) {

70static int opterror(popt_t *opt, const char *reason, int flags)
71{
72 if (flags & FLAG_SHORT) {
73 e_error("option `%c' %s", opt->shrt, reason);
74 } else
75 if (flags & FLAG_UNSET) {
76 e_error("option `no-%s' %s", opt->lng, reason);
77 } else {
78 e_error("option `%s' %s", opt->lng, reason);
79 }
80 return -1;
81}

83static int put_int_value(popt_t *opt, uint64_t v)
84{
85 switch (opt->int_vsize) {
86#define CASE(_sc) case _sc / 8: \
87 if (opt->kind == OPTION_UINT) { \
88 if (v <= UINT##_sc##_MAX) { \
89 *(uint##_sc##_t *)opt->value = v; \
90 } else { \
91 return -1; \
92 } \
93 } else { \
94 int64_t w = (int64_t)v; \
95 if (w >= INT##_sc##_MIN && w <= INT##_sc##_MAX) { \
96 *(int##_sc##_t *)opt->value = v; \
97 } else { \
98 return -1; \
99 } \
100 } \
101 break;
102
103 CASE(8);
104 CASE(16);
105 CASE(32);
106 CASE(64);
107
108#undef CASE
109
110 default: e_panic("should not happen");
111 }
112
113 return 0;
114}

116static int get_value(popt_state_t *st, popt_t *opt, int flags)
117{
118 if (st->p && (flags & FLAG_UNSET)) {
119 return opterror(opt, "takes no value", flags);
120 }
121
122 switch (opt->kind) {
123 const char *s;
124
125 case OPTION_FLAG:
126 if (!(flags & FLAG_SHORT) && st->p) {
127 return opterror(opt, "takes no value", flags);
128 }
129 put_int_value(opt, !(flags & FLAG_UNSET));
130 return 0;
131
132 case OPTION_STR:
133 if (flags & FLAG_UNSET) {
134 *(const char **)opt->value = (const char *)opt->init;
135 } else {
136 if (!st->p && st->pending_argc < 2) {
137 return opterror(opt, "requires a value", flags);
138 }
139 *(const char **)opt->value = opt_arg(st);
140 }
141 return 0;
142
143 case OPTION_CHAR:
144 if (flags & FLAG_UNSET) {
145 *(char *)opt->value = (char)opt->init;
146 } else {
147 const char *value;
148 if (!st->p && st->pending_argc < 2) {
149 return opterror(opt, "requires a value", flags);
150 }
151 value = opt_arg(st);
152 if (strlen(value) != 1) {
153 return opterror(opt, "expects a single character", flags);
154 }
155 *(char *)opt->value = value[0];
156 }
157 return 0;
158
159 case OPTION_INT:
160 case OPTION_UINT:
161 {
162 uint64_t v;
163
164 if (flags & FLAG_UNSET) {
165 v = opt->init;
166 } else {
167 if (!st->p && st->pending_argc < 2) {
168 return opterror(opt, "requires a value", flags);
169 }
170
171 errno = 0;
172 if (opt->kind == OPTION_UINT) {
173 lstr_t value = lstr_ltrim(LSTR(opt_arg(st)));
174
175 if (lstr_startswithc(value, '-')) {
176 /* -0 will return an error. */
177 return opterror(opt, "expects a positive value", flags);
178 }
179 v = strtoull(value.s, &s, 10);
180 } else {
181 v = strtoll(opt_arg(st), &s, 10);
182 }
183 if (*s || (errno && errno != ERANGE)) {
184 return opterror(opt, "expects a numerical value", flags);
185 }
186 }
187
188 if (errno == ERANGE || put_int_value(opt, v) < 0) {
189 return opterror(opt, "integer overflow", flags);
190 }
191 }
192 return 0;
193
194 case OPTION_VERSION:
195 if (flags & FLAG_UNSET) {
196 return opterror(opt, "takes no value", flags);
197 }
198 makeversion(EX_OK, opt->value, (void *)opt->init);
199
200 default:
201 e_panic("should not happen, programmer is a moron");
202 }
203}

215static int parse_short_opt(popt_state_t *st, char *arg, popt_t *opts)
216{
217 bool ignore_unknown = st->flags & POPT_IGNORE_UNKNOWN_OPTS;
218
219 st->p = arg + 1;
220 do {
221 popt_t *p_opts = find_opts_short_opt(st, opts);
222
223 if (!p_opts) {
224 if (ignore_unknown) {
225 st->p = NULL;
226 opt_add_left_arg(st, arg);
227 return 0;
228 } else {
229 return e_error("unknown option `%c'", *st->p);
230 }
231 }
232 ignore_unknown = false;
233
234 st->p = st->p[1] ? st->p + 1 : NULL;
235 RETHROW(get_value(st, p_opts, FLAG_SHORT));
236 } while (st->p);
237
238 return 0;
239}

241static int parse_long_opt(popt_state_t *st, char *arg, popt_t *opts)
242{
243 const char *arg_opt = arg + 2;
244
245 for (; opts->kind; opts++) {
246 const char *p;
247 int flags = 0;
248
249 if (!opts->lng) {
250 continue;
251 }
252
253 if (!strstart(arg_opt, opts->lng, &p)) {
254 if (!strstart(arg_opt, "no-", &p) || !strstart(p, opts->lng, &p))
255 {
256 continue;
257 }
258 flags = FLAG_UNSET;
259 }
260 if (*p) {
261 if (*p != '=') {
262 continue;
263 }
264 st->p = p + 1;
265 }
266 return get_value(st, opts, flags);
267 }
268 if (st->flags & POPT_IGNORE_UNKNOWN_OPTS) {
269 opt_add_left_arg(st, arg);
270 return 0;
271 } else {
272 return e_error("unknown option `%s'", arg_opt);
273 }
274}

328static void copyinits(popt_t *opts)
329{
330 for (;;) {
331 switch (opts->kind) {
332 case OPTION_INT:
333 case OPTION_UINT:
334 opts->init = get_int_init(opts);
335 break;
336 case OPTION_STR:
337 opts->init = (intptr_t)*(const char **)opts->value;
338 break;
339 case OPTION_CHAR:
340 opts->init = *(char *)opts->value;
341 break;
342 case OPTION_FLAG:
343 case OPTION_GROUP:
344 case OPTION_VERSION:
345 break;
346 case OPTION_END:
347 return;
348 }
349 opts++;
350 }
351}

353int parseopt(int argc, char **argv, popt_t *opts, int flags)
354{
355 popt_state_t st;
356
357 opt_state_init(&st, argc, argv, flags);
358 copyinits(opts);
359
360 for (; st.pending_argc > 0; st.pending_argc--, st.pending_argv++) {
361 char *arg = st.pending_argv[0];
362
363 if (*arg != '-' || !arg[1]) {
364 if (flags & POPT_STOP_AT_NONARG) {
365 break;
366 }
367 opt_add_left_arg(&st, arg);
368 continue;
369 }
370
371 if (arg[1] != '-') {
372 RETHROW(parse_short_opt(&st, arg, opts));
373 continue;
374 }
375
376 if (!arg[2]) { /* "--" */
377 st.pending_argc--;
378 st.pending_argv++;
379 break;
380 }
381
382 RETHROW(parse_long_opt(&st, arg, opts));
383 }
384
385 return opt_state_end(&st);
386}

391void makeusage(int ret, const char *arg0, const char *usage,
392 const char * const text[], popt_t *opts)
393{
394 const char *p = strrchr(arg0, '/');
395
396 printf("Usage: %s [options] %s\n", p ? p + 1 : arg0, usage);
397 if (text) {
398 putchar('\n');
399 while (*text) {
400 printf(" %s\n", *text++);
401 }
402 }
403 if (opts->kind != OPTION_GROUP)
404 putchar('\n');
405 for (; opts->kind; opts++) {
406 int pos = 4;
407 pstream_t help;
408
409 if (opts->kind == OPTION_GROUP) {
410 putchar('\n');
411 if (*opts->help)
412 printf("%s\n", opts->help);
413 continue;
414 }
415 printf(" ");
416 if (opts->shrt) {
417 pos += printf("-%c", opts->shrt);
418 }
419 if (opts->lng) {
420 pos += printf(opts->shrt ? ", --%s" : "--%s", opts->lng);
421 }
422 if (opts->kind != OPTION_FLAG) {
423 pos += printf(" ...");
424 }
425
426 help = ps_initstr(opts->help);
427 while (!ps_done(&help)) {
428 pstream_t line;
429
430 if (ps_get_ps_chr_and_skip(&help, '\n', &line) < 0) {
431 line = help;
432 help = ps_init(NULL, 0);
433 }
434 if (pos <= OPTS_WIDTH) {
435 printf("%*s%*pM", OPTS_WIDTH + OPTS_GAP - pos, "",
436 PS_FMT_ARG(&line));
437 pos = OPTS_WIDTH + 1;
438 } else {
439 printf("\n%*s%*pM", OPTS_WIDTH + OPTS_GAP, "",
440 PS_FMT_ARG(&line));
441 }
442 }
443 printf("\n");
444 }
445 exit(ret);
446}

448void makeversion(int ret, const char *name, const char *(*get_version)(void))
