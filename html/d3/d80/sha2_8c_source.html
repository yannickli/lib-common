56uint64_t sha2_hash_64(const void *data, ssize_t len)
57{
58 union {
59 byte b[32];
60 uint64_t u[4];
61 } res;
62
63 sha2(data, len, res.b, false);
64 return res.u[0] ^ res.u[1] ^ res.u[2] ^ res.u[3];
65}

81/*
82 * output = SHA-256( file contents )
83 */
84int sha2_file(const char *path, byte output[32], int is224 )
85{
86 FILE *f;
87 size_t n;
88 sha2_ctx ctx;
89 byte buf[1024];
90
91 if( ( f = fopen( path, "rb" ) ) == NULL )
92 return( 1 );
93
94 sha2_starts( &ctx, is224 );
95
96 while( ( n = fread( buf, 1, sizeof( buf ), f ) ) > 0 )
97 sha2_update( &ctx, buf, (int) n );
98
99 sha2_finish( &ctx, output );
100
101 memset( &ctx, 0, sizeof( sha2_ctx ) );
102
103 if( ferror( f ) != 0 )
104 {
105 fclose( f );
106 return( 2 );
107 }
108
109 fclose( f );
110 return( 0 );
111}

132 * released by Ulrich Drepper in public domain */

134/* Table used for base64 transformation */
135static const char sha2_crypt_base64char_g[64] =
136 "./0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz";

138void sha2_crypt(const void *input, size_t ilen, const void *salt,
139 size_t slen, uint32_t rounds, sb_t *output)
140{
141 uint32_t i;
142 unsigned char alt_result[32];
143 unsigned char tmp_result[32];
144 sha2_ctx ctx;
145 sha2_ctx alt_ctx;
146 SB_1k(buf);
147 SB_1k(alt_buf);
148
149 slen = MIN(slen, SHA256_CRYPT_SALT_LEN_MAX);
150 rounds = (rounds > 0) ? rounds : SHA256_CRYPT_DEFAULT_ROUNDS;
151 rounds = MAX(rounds, SHA256_CRYPT_MIN_ROUNDS);
152 rounds = MIN(rounds, SHA256_CRYPT_MAX_ROUNDS);
153
154 /* Let's begin */
155 sha2_starts(&ctx, false);
156
157 /* Add the input string */
158 sha2_update(&ctx, input, ilen);
159
160 /* Add the salt */
161 sha2_update(&ctx, salt, slen);
162
163 /* Alternate hash with INPUT-SALT-INPUT */
164 sha2_starts(&alt_ctx, false);
165 sha2_update(&alt_ctx, input, ilen);
166 sha2_update(&alt_ctx, salt, slen);
167 sha2_update(&alt_ctx, input, ilen);
168 sha2_finish(&alt_ctx, alt_result);
169
170 for (i = ilen; i > 32; i -= 32) {
171 sha2_update(&ctx, alt_result, 32);
172 }
173 sha2_update(&ctx, alt_result, i);
174
175 /* For every 1 in the binary representation of ilen add alt,
176 * for every 0 add the input
177 */
178 for (i = ilen; i > 0; i >>= 1) {
179 if ((i & 1) != 0) {
180 sha2_update(&ctx, alt_result, 32);
181 } else {
182 sha2_update(&ctx, input, ilen);
183 }
184 }
185
186 /* Intermediate result */
187 sha2_finish(&ctx, alt_result);
188
189 /* New alternate hash */
190 sha2_starts(&alt_ctx, false);
191
192 /* For every character in the input add the entire input */
193 for (i = 0; i < ilen; i++) {
194 sha2_update(&alt_ctx, input, ilen);
195 }
196
197 /* Get temp result */
198 sha2_finish(&alt_ctx, tmp_result);
199
200 for (i = ilen; i >= 32; i -= 32) {
201 sb_add(&buf, tmp_result, 32);
202 }
203 sb_add(&buf, tmp_result, i);
204
205 sha2_starts(&alt_ctx, false);
206
207 /* For each character in the input, add input */
208 for (i = 0; i < 16U + alt_result[0]; i++) {
209 sha2_update(&alt_ctx, salt, slen);
210 }
211
212 sha2_finish(&alt_ctx, tmp_result);
213
214 for (i = slen; i >= 32; i -= 32) {
215 sb_add(&alt_buf, tmp_result, 32);
216 }
217 sb_add(&alt_buf, tmp_result, i);
218
219 /* The loop */
220 for (i = 0; i < rounds; i++) {
221 sha2_starts(&ctx, false);
222
223 /* Add input or last result */
224 if ((i & 1) != 0) {
225 sha2_update(&ctx, buf.data, buf.len);
226 } else {
227 sha2_update(&ctx, alt_result, 32);
228 }
229
230 /* Add salt for numbers not divisible by 3 */
231 if (i % 3 != 0) {
232 sha2_update(&ctx, alt_buf.data, alt_buf.len);
233 }
234
235 /* Add input for numbers not divisible by 7 */
236 if (i % 7 != 0) {
237 sha2_update(&ctx, buf.data, buf.len);
238 }
239
240 /* Add intput or last result */
241 if ((i & 1) != 0) {
242 sha2_update(&ctx, alt_result, 32);
243 } else {
244 sha2_update(&ctx, buf.data, buf.len);
245 }
246
247 /* Create intermediate result */
248 sha2_finish(&ctx, alt_result);
249 }
250
251 /* Construction of the result */
252 sb_reset(output);
253 sb_addf(output, SHA256_PREFIX "$");
254
255 /* Here, we ALWAYS put the round prefix and round number in the result
256 * string, not only if round number != default number.
257 * This seems very less dangerous for compatibility between
258 * encryption version used in our products. It is still compliant with
259 * the specifications, even if the implementation given as example by
260 * Drepper does not follow this precept.
261 */
262 sb_addf(output, SHA256_ROUNDS_PREFIX);
263 sb_addf(output, "%d$", rounds);
264 sb_add(output, salt, slen);
265 sb_addf(output, "$");
266
267#define b64_from_24bits(b2_, b1_, b0_, n_) \
268 do { \
269 unsigned int w = cpu_to_le32(((b2_) << 16) | ((b1_) << 8) | (b0_)); \
270 for (int i_ = n_; i_-- > 0; w >>= 6) { \
271 sb_add(output, &sha2_crypt_base64char_g[w & 0x3f], 1); \
272 } \
273 } while (0)
274
275 b64_from_24bits(alt_result[0], alt_result[10], alt_result[20], 4);
276 b64_from_24bits(alt_result[21], alt_result[1], alt_result[11], 4);
277 b64_from_24bits(alt_result[12], alt_result[22], alt_result[2], 4);
278 b64_from_24bits(alt_result[3], alt_result[13], alt_result[23], 4);
279 b64_from_24bits(alt_result[24], alt_result[4], alt_result[14], 4);
280 b64_from_24bits(alt_result[15], alt_result[25], alt_result[5], 4);
281 b64_from_24bits(alt_result[6], alt_result[16], alt_result[26], 4);
282 b64_from_24bits(alt_result[27], alt_result[7], alt_result[17], 4);
283 b64_from_24bits(alt_result[18], alt_result[28], alt_result[8], 4);
284 b64_from_24bits(alt_result[9], alt_result[19], alt_result[29], 4);
285 b64_from_24bits(0, alt_result[31], alt_result[30], 3);
286
287#undef b64_from_24bits
288
289 /* Clearing datas to avoid core dump attacks */
290 sha2_starts(&ctx, false);
291 sha2_finish(&ctx, alt_result);
292
293 p_clear(&tmp_result, 1);
294 p_clear(&ctx, 1);
295 p_clear(&alt_ctx, 1);
296}

298int sha2_crypt_parse(lstr_t input, int *out_rounds, pstream_t *out_salt,
299 pstream_t *out_hash)
300{
301 /* Correct format for SHA256-crypt is
302 * $5$rounds=2000$salt$hash */
303
304 pstream_t ps = ps_initlstr(&input);
305 pstream_t salt;
306 pstream_t hash;
307 int rounds;
308 static bool first_call = true;
309 static ctype_desc_t sha2_crypt_ctype;
310
311 out_salt = out_salt ?: &salt;
312 out_hash = out_hash ?: &hash;
313 out_rounds = out_rounds ?: &rounds;
314
315 /* Check prefix */
316 RETHROW(ps_skipstr(&ps, "$5$rounds="));
317
318 /* Check rounds */
319 errno = 0;
320 *out_rounds = ps_geti(&ps);
321 THROW_ERR_IF(errno != 0);
322 THROW_ERR_IF(*out_rounds < SHA256_CRYPT_MIN_ROUNDS);
323 THROW_ERR_IF(*out_rounds > SHA256_CRYPT_MAX_ROUNDS);
324
325 /* Check salt */
326 RETHROW(ps_skipc(&ps, '$'));
327 RETHROW(ps_get_ps_chr_and_skip(&ps, '$', out_salt));
328 THROW_ERR_IF(ps_done(out_salt));
329 THROW_ERR_IF(ps_len(out_salt) > SHA256_CRYPT_SALT_LEN_MAX);
330
331 /* Check hash-part */
332 if (unlikely(first_call)) {
333 first_call = false;
334 ctype_desc_build2(&sha2_crypt_ctype, sha2_crypt_base64char_g,
335 sizeof(sha2_crypt_base64char_g));
336 }
337 THROW_ERR_IF(ps_len(&ps) != SHA256_CRYPT_DIGEST_SIZE);
338 *out_hash = ps;
339 ps_skip_span(&ps, &sha2_crypt_ctype);
340 THROW_ERR_IF(!ps_done(&ps));
341
342 return 0;
343}
