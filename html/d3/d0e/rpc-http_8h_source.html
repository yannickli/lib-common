27/*
28 * RPC-HTTP: a library for wrapping and exposing IoP RPCs as HTTP-based API
29 * end-points.
30 *
31 * At the server side, this api enables the calling code to wrap a set of RPCs
32 * under a common server url such as http[s]://api.example.com/v1/.
33 * This is done as follows:
34 * - create a httpd_trigger__ic_t (subclassed from httpd_trigger) using
35 * httpd_trigger__ic_new.
36 * - register the trigger for the server url using httpd_trigger_register.
37 * - register the IoP RPCs.
38 * <code>
39 * ...
40 * hrpc_trigger = httpd_trigger__ic_new(&iop_mod, api_schema, MAX_REQ_SZ);
41 * httpd_trigger_register(httpd_cfg, POST, server_url, &hrpc_trigger->cb);
42 * ichttp_register(hrpc_trigger, iop_mod_name, iop_interface, rpc1);
43 * ichttp_register(hrpc_trigger, iop_mod_name, iop_interface, rpc2);
44 * ...
45 * </code>
46 * The trigger then handles the unpacking of http requests into the
47 * corresponding rpc in two modes: XML mode or JSON mode. The XML mode is
48 * compatible with SOAP. This step involves finding the rpc to invoke and
49 * deserializing input parameters for the rpc. For the response,
50 * the corresponding packing process is done automatically by ic_reply.
51 *
52 *
53 * At the client side, this api enables the calling code to call HTTP RPCs and
54 * consume their responses (in JSON format).
55 */

195/** \brief register a local callback for an rpc on the given http iop trigger.
196 * \param[in] tcb
197 * the #httpd_trigger__ic_t to register the callback implementation into.
198 * \param[in] _mod name of the package+module of the RPC
199 * \param[in] _i name of the interface of the RPC
200 * \param[in] _r name of the rpc
201 * \param[in] _cb
202 * the implementation callback. Its type should be:
203 * <tt>void (*)(IOP_RPC_IMPL_ARGS(_mod, _if, _rpc))</tt>.
204 * it can be the same implementation callback as the one used for an
205 * #ic_register call.
206 * \param[in] _pre_cb
207 * the pre_hook callback. Its type should be:
208 * <tt>void (*)(ichannel_t *, uint64_t, const ic__hdr__t *, void *)</tt>
209 * it can be the same implementation callback as the one used for an
210 * #ic_register call.
211 * \param[in] _post_cb
212 * the post_hook callback. Its type should be:
213 * <tt>void (*)(ichannel_t *, ic_status_t, ic_hook_ctx_t *, void *)</tt>
214 * it can be the same implementation callback as the one used for an
215 * #ic_register call.
216 * \param[in] _pre_arg argument we want to pass to pre_hook
217 * \param[in] _post_arg argument we want to pass to post_hook
218 */
219#define ichttp_register_pre_post_hook_(tcb, _mod, _if, _rpc, _cb, _pre_cb, \
220 _post_cb, _pre_arg, _post_arg) \
221 do { \
222 void (*__cb)(IOP_RPC_IMPL_ARGS(_mod, _if, _rpc)) = _cb; \
223 ic_cb_entry_t __cb_e = { \
224 .cb_type = IC_CB_NORMAL, \
225 .rpc = IOP_RPC(_mod, _if, _rpc), \
226 .pre_hook = _pre_cb, \
227 .post_hook = _post_cb, \
228 .pre_hook_args = _pre_arg, \
229 .post_hook_args = _post_arg, \
230 .u = { .cb = { \
231 .cb = (void *)__cb, \
232 } }, \
233 }; \
234 \
235 ___ichttp_register(tcb, _mod, _if, _rpc, &__cb_e); \
236 } while (0)
