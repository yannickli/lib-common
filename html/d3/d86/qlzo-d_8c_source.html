27static ALWAYS_INLINE
28int lzo_get_varlen(pstream_t *in, unsigned u, unsigned mask)
29{
30 unsigned sz = u & mask;
31 const uint8_t *b = in->b;
32
33 if (!sz) {
34 do {
35 if (unlikely(++b >= in->b_end))
36 return LZO_ERR_INPUT_OVERRUN;
37 } while (!*b);
38 sz = *b + mask + 255 * (b - in->b - 1);
39 }
40 in->b = b + 1;
41 return sz;
42}

44static ALWAYS_INLINE
45int lzo_copy_input(pstream_t *in, ostream_t *os, unsigned sz)
46{
47 if (unlikely(!ps_has(in, sz)))
48 return LZO_ERR_INPUT_OVERRUN;
49 if (unlikely(os->b + sz > os->b_end))
50 return LZO_ERR_OUTPUT_OVERRUN;
51 os->b = mempcpy(os->b, in->b, sz);
52 return __ps_skip(in, sz);
53}

56static ALWAYS_INLINE
57int lzo_copy_input_small(pstream_t *in, ostream_t *os, unsigned sz)
58{
59 const uint8_t *src = in->b;
60 uint8_t *dst = os->b;
61
62 if (unlikely(!ps_has(in, sz)))
63 return LZO_ERR_INPUT_OVERRUN;
64 if (likely(dst + 4 <= os->b_end)) {
65 put_unaligned_cpu32(dst, get_unaligned_cpu32(src));
66 } else {
67 if (unlikely(dst + sz > os->b_end))
68 return LZO_ERR_OUTPUT_OVERRUN;
69 dst[0] = src[0];
70 if (sz > 1) {
71 dst[1] = src[1];
72 if (sz > 2)
73 dst[2] = src[2];
74 }
75 }
76 os->b += sz;
77 return __ps_skip(in, sz);
78}

80static ALWAYS_INLINE
81int lzo_copy_backptr(ostream_t *os, const uint8_t *out_orig,
82 unsigned back, unsigned sz)
83{
84 const uint8_t *src = os->b - back;
85 uint8_t *dst = os->b;
86
87 if (unlikely(src < out_orig)) {
88 return LZO_ERR_BACKPTR_OVERRUN;
89 }
90 if (unlikely(dst + sz > os->b_end)) {
91 return LZO_ERR_OUTPUT_OVERRUN;
92 }
93
94 if (back == 1) {
95 memset(dst, *src, sz);
96 os->b += sz;
97 return 0;
98 }
99
100 while (sz >= back) {
101 memcpy(dst, src, back);
102 dst += back;
103 src += back;
104 sz -= back;
105 }
106 while (sz-- > 0) {
107 *dst++ = *src++;
108 }
109 os->b = dst;
110 return 0;
111}

148static ssize_t decompress(void *_out, size_t outlen, pstream_t in, bool safe)
149{
150 uint8_t *out_orig = _out;
151 ostream_t os = {
152 .b = out_orig,
153 .b_end = out_orig + outlen,
154 };
155#define SAFE_CHECK_LEN(len) \
156 if (safe && unlikely(!ps_has(&in, len))) \
157 return LZO_ERR_INPUT_OVERRUN
158
159 SAFE_CHECK_LEN(1);
160 if (in.b[0] > 17)
161 RETHROW(lzo_copy_input(&in, &os, __ps_getc(&in) - 17));
162 for (;;) {
163 unsigned u, sz, back;
164
165 SAFE_CHECK_LEN(1);
166 if ((u = in.b[0]) >= LZO_M4_MARKER) {
167 match_2to4:
168 if (u >= LZO_M2_MARKER) {
169 SAFE_CHECK_LEN(2);
170 sz = (u >> 5) + 1;
171 back = (in.b[1] << 3) + ((u >> 2) & 7) + 1;
172 __ps_skip(&in, 2);
173 } else if (u >= LZO_M3_MARKER) {
174 sz = RETHROW(lzo_get_varlen(&in, u, 31)) + 2;
175 SAFE_CHECK_LEN(2);
176 u = __ps_get_le16(&in);
177 back = (u >> 2) + 1;
178 } else {
179 sz = RETHROW(lzo_get_varlen(&in, u, 7)) + 2;
180 back = ((u & 8) << 11);
181 SAFE_CHECK_LEN(2);
182 u = __ps_get_le16(&in);
183 back += (u >> 2);
184 if (back == 0)
185 break;
186 back += LZO_M3_MAX_OFFSET;
187 }
188 RETHROW(lzo_copy_backptr(&os, out_orig, back, sz));
189 } else {
190 sz = RETHROW(lzo_get_varlen(&in, u, 15)) + 3;
191 RETHROW(lzo_copy_input(&in, &os, sz));
192 SAFE_CHECK_LEN(1);
193 if ((u = in.b[0]) >= LZO_M4_MARKER)
194 goto match_2to4;
195 SAFE_CHECK_LEN(2);
196 back = (1 << 11) + (in.b[1] << 2) + (u >> 2) + 1;
197 __ps_skip(&in, 2);
198 RETHROW(lzo_copy_backptr3(&os, out_orig, back));
199 }
200
201 while ((sz = (u & 3))) {
202 RETHROW(lzo_copy_input_small(&in, &os, sz));
203 SAFE_CHECK_LEN(1);
204 if ((u = in.b[0]) >= LZO_M4_MARKER)
205 goto match_2to4;
206 SAFE_CHECK_LEN(2);
207 back = (in.b[1] << 2) + (u >> 2) + 1;
208 __ps_skip(&in, 2);
209 RETHROW(lzo_copy_backptr2(&os, out_orig, back));
210 }
211 }
212
213 if (likely(in.b == in.b_end))
214 return os.b - out_orig;
215 if (in.b < in.b_end)
216 return LZO_ERR_INPUT_NOT_CONSUMED;
217 return LZO_ERR_INPUT_OVERRUN;
218#undef SAFE_CHECK_LEN
219}
