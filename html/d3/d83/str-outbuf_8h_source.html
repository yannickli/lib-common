105static inline void ob_add_chunk(outbuf_t * nonnull ob,
106 outbuf_chunk_t * nonnull obc)
107{
108 htlist_add_tail(&ob->chunks_list, &obc->chunks_link);
109 ob->length += (obc->length - obc->offset);
110 obc->sb_leading = ob->sb_trailing;
111 ob->sb_trailing = 0;

157/** adds a memory block (\p ptr, \p len) to the chunks referenced by \p ob.
158 *
159 * \param is_const: if false memory ownership is transfered to \p ob.
160 *
161 * XXX: if is_const is true, the pointed memory must live long enough until it
162 * is consumed, in more practical terms, it must outlive the \p ob object. */
163static inline
164void ob_add_memchunk(outbuf_t * nonnull ob, const void * nonnull ptr,
165 int len, bool is_const)
166{
167 if (len <= OUTBUF_CHUNK_MIN_SIZE) {
168 ob_add(ob, ptr, len);
169 if (!is_const)
170 ifree((void *)ptr, MEM_LIBC);
171 } else {
172 outbuf_chunk_t *obc = p_new(outbuf_chunk_t, 1);
173
174 obc->u.p = ptr;
175 obc->length = len;
176 if (!is_const)
177 obc->on_wipe = OUTBUF_DO_FREE;
178 ob_add_chunk(ob, obc);
179 }
180}
181
182static inline void ob_add_memmap(outbuf_t * nonnull ob, void * nonnull map,
183 int len)
184{
185 if (len <= OUTBUF_CHUNK_MIN_SIZE) {
186 ob_add(ob, map, len);
187 munmap(map, len);
188 } else {
189 outbuf_chunk_t *obc = p_new(outbuf_chunk_t, 1);
190
191 obc->u.p = map;
192 obc->length = len;
193 obc->on_wipe = OUTBUF_DO_MUNMAP;
194 ob_add_chunk(ob, obc);
195 }
196}
