75static int dump_and_obfuscate(const char *data, int len, FILE *out)
76{
77 int nb_chunk = 0;
78
79 while (len > 0) {
80 char buffer[FARCH_MAX_SYMBOL_SIZE];
81 lstr_t obfuscated_chunk;
82 uint32_t chunk_size;
83
84 chunk_size = rand_range(FARCH_MAX_SYMBOL_SIZE / 2,
85 FARCH_MAX_SYMBOL_SIZE);
86 chunk_size = MIN((uint32_t)len, chunk_size);
87 obfuscated_chunk = LSTR_INIT_V(buffer, chunk_size);
88 lstr_obfuscate(LSTR_INIT_V(data, chunk_size), chunk_size,
89 obfuscated_chunk);
90 put_chunk(obfuscated_chunk, out);
91 data += chunk_size;
92 len -= chunk_size;
93 nb_chunk++;
94 }
95
96 assert (len == 0);
97
98 return nb_chunk;
99}

101static void dump_file(const char *path, farch_entry_t *entry, FILE *out)
102{
103 lstr_t file;
104
105 DIE_IF(lstr_init_from_file(&file, path, PROT_READ, MAP_SHARED) < 0,
106 "unable to open `%s` for reading: %m", path);
107 entry->size = file.len;
108
109 if (opts_g.compress_lzo) {
110 t_scope;
111 byte lzo_buf[LZO_BUF_MEM_SIZE];
112 char *cbuf;
113 int clen;
114
115 clen = lzo_cbuf_size(file.len);
116 cbuf = t_new_raw(char, clen);
117 clen = qlzo1x_compress(cbuf, clen, ps_initlstr(&file), lzo_buf);
118
119 if (clen < file.len) {
120 entry->nb_chunks = dump_and_obfuscate(cbuf, clen, out);
121 entry->compressed_size = clen;
122 } else {
123 entry->nb_chunks = dump_and_obfuscate(file.s, file.len, out);
124 entry->compressed_size = file.len;
125 }
126 } else {
127 entry->nb_chunks = dump_and_obfuscate(file.s, file.len, out);
128 entry->compressed_size = file.len;
129 }
130
131 lstr_wipe(&file);
132}

136static void dump_entries(const char *archname,
137 const qv_t(farch_entry) *entries, FILE *out)
138{
139 int chunk = 0;
140
141 tab_for_each_ptr(entry, entries) {
142 char buffer[2 * PATH_MAX];
143 lstr_t obfuscated_name = LSTR_INIT(buffer, entry->name.len);
144
145 lstr_obfuscate(entry->name, entry->nb_chunks, obfuscated_name);
146
147 fprintf(out, "/* {""{{ %*pM */\n", LSTR_FMT_ARG(entry->name));
148 fprintf(out, "{\n"
149 " .name = LSTR_IMMED(\"");
150 put_as_str(obfuscated_name, out);
151 fprintf(out, "\"),\n"
152 " .chunks = &%s_data[%d],\n"
153 " .size = %d,\n"
154 " .compressed_size = %d,\n"
155 " .nb_chunks = %d,\n"
156 "},\n"
157 "/* }""}} */\n",
158 archname, chunk, entry->size, entry->compressed_size,
159 entry->nb_chunks);
160 chunk += entry->nb_chunks;
161 }
162}

164static int do_work(const char *reldir, FILE *in, FILE *out, FILE *deps)
165{
166 t_scope;
167 SB_1k(dep);
168 char srcdir[PATH_MAX];
169 char buf[PATH_MAX];
170 char name[PATH_MAX];
171 int srcdirlen = 0;
172 qv_t(farch_entry) entries;
173
174 t_qv_init(&entries, 10);
175
176 do {
177 DIE_IF(!fgets(name, sizeof(name), in), "no variable name specified: %m");
178 DIE_IF(name[strlen(name) - 1] != '\n', "line 1 is too long: %m");
179 } while (*skipspaces(name) == '#' || *skipspaces(name) == '\0');
180 strrtrim(name);
181 TRACE("creating `%s`", name);
182
183 fprintf(out, "/* This file is generated by farchc. */\n");
184 fprintf(out, "\n");
185 fprintf(out, "#include <lib-common/farch.h>\n");
186 fprintf(out, "\n");
187 fprintf(out, "static const farch_data_t %s_data[] = {\n", name);
188
189 if (opts_g.target)
190 sb_addf(&dep, "%s%s ", reldir, opts_g.target);
191 if (opts_g.out)
192 sb_addf(&dep, "%s ", opts_g.out);
193 sb_addf(&dep, "%s: ", opts_g.deps);
194
195 snprintf(srcdir, sizeof(srcdir), "%s", reldir);
196 path_simplify(srcdir);
197 path_join(srcdir, sizeof(srcdir), "/");
198 srcdirlen = strlen(srcdir);
199
200 for (int lineno = 2; fgets(buf, sizeof(buf), in); lineno++) {
201 const char *s = skipspaces(buf);
202 char path[PATH_MAX];
203 farch_entry_t entry;
204 char *fullname;
205
206 DIE_IF(buf[strlen(buf) - 1] != '\n', "line %d is too long", lineno);
207 strrtrim(buf);
208 if (!*s) {
209 continue;
210 }
211 if (*s == '#') {
212 TRACE("%s", s);
213 continue;
214 }
215
216 snprintf(path, sizeof(path), "%s%s", srcdir, s);
217 fullname = path + srcdirlen;
218
219 if (deps) {
220 fprintf(deps, "%s%s\n", dep.data, path);
221 fprintf(deps, "%s:\n", path);
222 }
223
224 TRACE("adding `%s` as `%s`", path, fullname);
225 entry.name = t_lstr_dups(fullname, -1);
226 fprintf(out, "/* {""{{ %s */\n", fullname);
227 dump_file(path, &entry, out);
228 fprintf(out, "/* }""}} */\n");
229 qv_append(&entries, entry);
230 }
231
232 fprintf(out, "};\n\n"
233 "static const farch_entry_t %s[] = {\n", name);
234 dump_entries(name, &entries, out);
235
236 fprintf(out,
237 "{ .name = LSTR_NULL },\n"
238 "};\n");
239 return 0;
240}

242int main(int argc, char *argv[])
243{
244 t_scope;
245 const char *arg0 = NEXTARG(argc, argv);
246 char reldir[PATH_MAX];
247 FILE *in = stdin, *out = stdout, *deps = NULL;
248
249 argc = parseopt(argc, argv, popt, 0);
250 if (argc < 0 || argc > 1 || opts_g.help)
251 makeusage(EXIT_FAILURE, arg0, "<farch-script>", NULL, popt);
252
253 if (opts_g.out) {
254 out = fopen(opts_g.out, "w");
255 DIE_IF(!out, "unable to open `%s` for writing: %m", opts_g.out);
256 }
257
258 if (argc > 0) {
259 in = fopen(argv[0], "r");
260 path_dirname(reldir, sizeof(reldir), argv[0]);
261 path_join(reldir, sizeof(reldir), "/");
262 DIE_IF(!in, "unable to open `%s` for reading: %m", argv[0]);
263 } else {
264 reldir[0] = '\0';
265 }
266
267 if (opts_g.deps) {
268 deps = fopen(opts_g.deps, "w");
269 DIE_IF(!deps, "unable to open `%s` for writing: %m", opts_g.deps);
270 }
271
272 do_work(reldir, in, out, deps);
273 p_fclose(&in);
274 p_fclose(&out);
275 p_fclose(&deps);
276 if (opts_g.out) {
277 DIE_IF(chmod(opts_g.out, 0440), "unable to chmod `%s`: %m",
278 opts_g.out);
279 }
280 TRACE("OK !");
281 return 0;
282}
