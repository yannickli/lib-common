63/** \brief SCTP connection private context.
64 *
65 * This structure is only known by this file. Any reference to this structure
66 * in the user context is managed by the \p user_ctx structure.
67 */
68typedef struct sctp_conn_priv_t {
69 el_t nonnull evh;
70
71 /* Allows to differentiate a context for active connection and a context
72 * listening for connections. */
73 bool is_listening;
74
75 /* Indicates that a full message has been read in the read buffer, and so
76 * that the read buffer should be reset before the next read.
77 */
78 bool reset_rbuf;
79
80 /* Read buffer. */
81 sb_t rbuf;
82
83 /* Queue of messages to be sent on this connection. */
84 dlist_t msgs;
85
86 sctp_on_connect_f on_connect_cb;
87
88 sctp_on_disconnect_f on_disconnect_cb;
89
90 sctp_on_data_f on_data_cb;
91
92 sctp_on_accept_f on_accept_cb;
93
94 /* Public context, can be modified by the user. */
95 sctp_conn_t user_ctx;
96} sctp_conn_priv_t;
97
98static sctp_conn_priv_t * nonnull
99sctp_conn_priv_init(sctp_conn_priv_t * nonnull conn)
100{
101 p_clear(conn, 1);
102
103 sb_init(&conn->rbuf);
104 dlist_init(&conn->msgs);
105 return conn;
106}
107GENERIC_NEW(sctp_conn_priv_t, sctp_conn_priv);
108
109static void sctp_conn_priv_wipe(sctp_conn_priv_t * nonnull conn)
110{
111 sb_wipe(&conn->rbuf);
112 sctp_msg_list_wipe(&conn->msgs);
113 lstr_wipe(&(conn->user_ctx.entity_id));
114 lstr_wipe(&(conn->user_ctx.host));
115 if (!conn->is_listening) {
116 /* XXX: The context created by sctp_listen() doesn't use a specific
117 * logger but use the global logger instead, do not remove it! */
118 logger_delete(&conn->user_ctx.logger);
119 }
120}
121GENERIC_DELETE(sctp_conn_priv_t, sctp_conn_priv);

153static int
154addr_list_to_su(qv_t(lstr) addrs, uint16_t defport, qv_t(sctp_su) *out)
155{
156 tab_for_each_entry(name, &addrs) {
157 pstream_t addr = ps_initlstr(&name);
158 pstream_t host;
159 in_port_t port;
160 sockunion_t su;
161
162 ps_trim(&addr);
163 RETHROW(addr_parse(addr, &host, &port, defport));
164 RETHROW(addr_info(&su, AF_UNSPEC, host, port));
165 qv_append(out, su);
166 }
167
168 return 0;
169}

171static sockunion_t *t_sctp_sus_array(const qv_t(sctp_su) *sus,
172 int * nullable olen)
173{
174 sockunion_t *res = t_new_raw(sockunion_t, sus->len);
175 size_t len = 0;
176 size_t sl;
177
178 tab_for_each_ptr(su, sus) {
179 sl = sockunion_len(su);
180 memcpy((byte *)res + len, su, sl);
181 len += sl;
182 }
183 if (olen) {
184 *olen = len;
185 }
186 return res;
187}

192static int
193sctp_ep_init_sock(int fd, size_t sndbuf)
194{
195 int nodelay = 1;
196
197 assert(fd >= 0);
198
199 RETHROW(sctp_enable_events(fd, SCTP_DATA_IO_EV | SCTP_ASSOCIATION_EV |
200 SCTP_SEND_FAILURE_EV | SCTP_PEER_ERROR_EV |
201 SCTP_ADDRESS_EV | SCTP_SHUTDOWN_EV));
202
203 RETHROW(setsockopt(fd, IPPROTO_SCTP, SCTP_NODELAY, &nodelay,
204 sizeof(nodelay)));
205 RETHROW(setsockopt(fd, SOL_SOCKET, SO_SNDBUF, &sndbuf, sizeof(sndbuf)));
206 return 0;
207}

209static int sctp_send_queued_msgs(sctp_conn_priv_t **pconn)
210{
211 sctp_conn_priv_t *conn = *pconn;
212 int fd = el_fd_get_fd(conn->evh);
213 uint16_t stream_no = 0;
214 int msg_flags = 0;
215 int len;
216
217 dlist_for_each_entry(sctp_msg_t, msg, &conn->msgs, msg_list) {
218 errno = 0;
219 len = sctp_sendmsg(fd, msg->msg.data, msg->msg.len, NULL, 0,
220 htonl(msg->ppid), msg_flags, stream_no, 0, 0);
221 if (len < 0) {
222 if (!ERR_RW_RETRIABLE(errno)) {
223 logger_error(conn->user_ctx.logger,
224 "SCTP send message to fd: %d failed: %m", fd);
225 sctp_conn_close_priv(pconn);
226 return -1;
227 }
228 return 0;
229 }
230
231 logger_trace(conn->user_ctx.logger, 2,
232 "send message to %d succeed (%d bytes)",
233 fd, len);
234 sctp_msg_delete(&msg);
235 }
236 el_fd_set_mask(conn->evh, POLLIN);
237 return 0;
238}

240/* return -1 if we closed the connection, 0 otherwise */
241static int sctp_handle_notification(sctp_conn_priv_t **pconn)
242{
243 sctp_conn_priv_t *conn = *pconn;
244 const union sctp_notification *sn =
245 (union sctp_notification *)conn->rbuf.data;
246
247 if (conn->rbuf.len < ssizeof(sn->sn_header)) {
248 logger_error(conn->user_ctx.logger, "invalid NOTIF: len = %d < %zd",
249 conn->rbuf.len, sizeof(sn->sn_header));
250 return 0;
251 }
252
253 switch (sn->sn_header.sn_type) {
254 case SCTP_ASSOC_CHANGE: {
255 const struct sctp_assoc_change *sac = &sn->sn_assoc_change;
256
257 logger_trace(conn->user_ctx.logger, 2, "got notif ASSOC_CHANGE with "
258 "state:%d error:%d sac:%d os:%d is:%d",
259 sac->sac_state, sac->sac_error, sac->sac_assoc_id,
260 sac->sac_outbound_streams,
261 sac->sac_inbound_streams);
262
263 switch (sac->sac_state) {
264 case SCTP_COMM_UP:
265 case SCTP_RESTART:
266 break;
267
268 case SCTP_COMM_LOST:
269 case SCTP_SHUTDOWN_COMP:
270 case SCTP_CANT_STR_ASSOC:
271 logger_trace(conn->user_ctx.logger, 2,
272 "SCTP notification (shutdown or lost)");
273 sctp_conn_close_priv(pconn);
274 return -1;
275 }
276 } break;
277
278 case SCTP_SEND_FAILED: {
279 const struct sctp_send_failed *ssf = &sn->sn_send_failed;
280
281 logger_error(conn->user_ctx.logger,
282 "got send failed error %d with peer `%pL'",
283 ssf->ssf_error, &conn->user_ctx.entity_id);
284 } break;
285
286 case SCTP_REMOTE_ERROR: {
287 const struct sctp_remote_error *sre = &sn->sn_remote_error;
288
289 logger_error(conn->user_ctx.logger,
290 "got remote error %d with peer `%pL'",
291 sre->sre_error, &conn->user_ctx.entity_id);
292 } break;
293
294 case SCTP_PEER_ADDR_CHANGE: {
295 const struct sctp_paddr_change *spc = &sn->sn_paddr_change;
296
297 logger_trace(conn->user_ctx.logger, 2,
298 "got notif SCTP_PEER_ADDR_CHANGE with "
299 "state:%d error:%d assoc_id:%d",
300 spc->spc_state, spc->spc_error, spc->spc_assoc_id);
301 } break;
302
303 default:
304 logger_trace(conn->user_ctx.logger, 2,
305 "got notif %d", sn->sn_header.sn_type);
306 break;
307 }
308
309 return 0;
310}

312/* return:
313 * -1 on error (connection closed)
314 * 0 for RETRIABLE error
315 * sctp_msg_type_t on a valid message
316 */
317static int sctp_receive_full_msg(sctp_conn_priv_t **pconn)
318{
319 struct sctp_sndrcvinfo sinfo;
320 sctp_conn_priv_t *conn = *pconn;
321 int len;
322 int fd = el_fd_get_fd(conn->evh);
323 sb_t *ibuf = &conn->rbuf;
324 int msg_flags = 0;
325
326 if (conn->reset_rbuf) {
327 sb_reset(ibuf);
328 conn->reset_rbuf = false;
329 }
330 p_clear(&sinfo, 1);
331
332 /* support partial delivery : loop until we get the flag MSG_EOR which
333 * indicates the end of record */
334 for (;;) {
335 errno = 0;
336 len = sctp_recvmsg(fd, sb_grow(ibuf, BUFSIZ), BUFSIZ, NULL, NULL,
337 &sinfo, &msg_flags);
338 if (len < 0) {
339 __sb_fixlen(ibuf, ibuf->len);
340 if (ERR_RW_RETRIABLE(errno)) {
341 return 0;
342 }
343 logger_error(conn->user_ctx.logger, "recvmsg error %m");
344 sctp_conn_close_priv(pconn);
345 return -1;
346 }
347 __sb_fixlen(ibuf, ibuf->len + len);
348
349 if (msg_flags & MSG_EOR) {
350 break;
351 }
352 }
353
354 if (msg_flags & MSG_NOTIFICATION) {
355 conn->reset_rbuf = true;
356 return SCTP_MSG_TYPE_NOTIF;
357 }
358
359 if (unlikely(msg_flags & MSG_CTRUNC)) {
360 logger_error(conn->user_ctx.logger, "ancillary data truncated, "
361 "increase /proc/sys/net/core/optmem_max");
362 } else if (sinfo.sinfo_flags & (SCTP_EOF | SCTP_ABORT)) {
363 logger_error(conn->user_ctx.logger, "SCTP close request fd %d", fd);
364 sctp_conn_close_priv(pconn);
365 return -1;
366 }
367
368 logger_trace(conn->user_ctx.logger, 2,
369 "SCTP received %d bytes from fd: %d",
370 ibuf->len, fd);
371 conn->reset_rbuf = true;
372
373 return SCTP_MSG_TYPE_DATA;
374}

376static int sctp_conn_on_event_in(sctp_conn_priv_t **pconn)
377{
378 sctp_conn_priv_t *conn = *pconn;
379 bool data_received = false;
380
381 for (;;) {
382 int res;
383
384 res = RETHROW(sctp_receive_full_msg(pconn));
385 if (res == 0) {
386 break;
387 }
388
389 switch ((sctp_msg_type_t)res) {
390 case SCTP_MSG_TYPE_NOTIF:
391 RETHROW(sctp_handle_notification(pconn));
392 break;
393 case SCTP_MSG_TYPE_DATA:
394 data_received = true;
395 if (conn->on_data_cb(&conn->user_ctx, &conn->rbuf, false) < 0) {
396 logger_warning(conn->user_ctx.logger,
397 "user on_data function returned an error");
398 goto error;
399 }
400 break;
401 }
402 }
403
404 if (data_received) {
405 if (conn->on_data_cb(&conn->user_ctx, NULL, true) < 0) {
406 logger_warning(conn->user_ctx.logger,
407 "user on_data fn returned an error");
408 goto error;
409 }
410 }
411
412 return 0;
413error:
414 sctp_conn_close_priv(pconn);
415 return -1;
416}

418static int sctp_conn_on_event(el_t evh, int fd, short ev, el_data_t priv)
419{
420 sctp_conn_priv_t *conn = priv.ptr;
421
422 if (ev & POLLIN) {
423 RETHROW(sctp_conn_on_event_in(&conn));
424 }
425
426 if (ev & POLLOUT) {
427 RETHROW(sctp_send_queued_msgs(&conn));
428 }
429
430 if (ev & POLLHUP) {
431 logger_info(conn->user_ctx.logger, "got POLLHUP, closing connection");
432 sctp_conn_close_priv(&conn);
433 return -1;
434 }
435
436 return 0;
437}

439static int sctp_connecting(el_t evh, int fd, short ev, el_data_t priv)
440{
441 sctp_conn_priv_t *conn = priv.ptr;
442 int ret;
443 int err = 0;
444 socklen_t len = sizeof(err);
445
446 ret = getsockopt(fd, SOL_SOCKET, SO_ERROR, (void *)&err, &len);
447 if (ret < 0 || err) {
448 logger_trace(conn->user_ctx.logger, 4,
449 "error returned while connecting. Using error callback");
450 if (len != sizeof(err)) {
451 logger_warning(conn->user_ctx.logger,
452 "unexpected error code length (%d != %ld)",
453 len, sizeof(err));
454 }
455 conn->on_connect_cb(&conn->user_ctx, false, err);
456 sctp_conn_close_priv(&conn);
457 return -1;
458 }
459
460 if (conn->on_connect_cb(&conn->user_ctx, true, 0) < 0) {
461 logger_error(conn->user_ctx.logger,
462 "user on_connect_cb returned an error");
463 sctp_conn_close_priv(&conn);
464 return -1;
465 }
466
467 el_fd_set_hook(conn->evh, sctp_conn_on_event);
468
469 return 0;
470}

472static int sctp_srv_on_event(el_t evh, int fd, short ev, el_data_t priv)
473{
474 int sock;
475 sctp_conn_priv_t *listen_ctx = priv.ptr;
476 sctp_conn_t *user_ctx = &listen_ctx->user_ctx;
477
478 while ((sock = acceptx(fd, O_NONBLOCK)) >= 0) {
479 t_scope;
480 sctp_conn_priv_t *conn;
481 int ret;
482 sockunion_t local = {.family = AF_INET};
483 socklen_t size = sockunion_len(&local);
484 char host[NI_MAXHOST];
485 char serv[NI_MAXSERV];
486 int port;
487 logger_t *logger;
488 lstr_t logger_name;
489
490 logger_name = t_lstr_cat3(user_ctx->entity_id,
491 LSTR("-"),
492 t_lstr_fmt("%d", sock));
493
494 logger = logger_new(&sctp_logger_g, logger_name,
495 LOG_INHERITS, 0);
496
497 if (getpeername(sock, &local.sa, &size) < 0) {
498 logger_error(logger, "cannot get peer name for `%*pM': %m",
499 LSTR_FMT_ARG(user_ctx->entity_id));
500 close(sock);
501 logger_delete(&logger);
502 return -1;
503 }
504
505 if ((ret = getnameinfo(&local.sa, size, host, NI_MAXHOST,
506 serv, NI_MAXSERV,
507 NI_NUMERICHOST | NI_NUMERICSERV)) != 0)
508 {
509 logger_error(logger,
510 "cannot get peer informations for `%*pM': %s",
511 LSTR_FMT_ARG(user_ctx->entity_id),
512 gai_strerror(ret));
513 close(sock);
514 logger_delete(&logger);
515 return -1;
516 }
517
518 port = atoi(serv);
519 logger_info(logger, "connection for `%*pM' from %s:%d",
520 LSTR_FMT_ARG(user_ctx->entity_id),
521 host, port);
522
523
524 conn = sctp_conn_priv_new();
525 conn->evh = el_fd_register(sock, true, POLLIN,
526 sctp_conn_on_event, conn);
527 conn->is_listening = false;
528 conn->on_disconnect_cb = listen_ctx->on_disconnect_cb;
529 conn->on_data_cb = listen_ctx->on_data_cb;
530 conn->user_ctx.entity_id = lstr_dup(user_ctx->entity_id);
531 conn->user_ctx.priv = user_ctx->priv;
532 conn->user_ctx.logger = logger;
533 conn->user_ctx.host = lstr_dup(LSTR(host));
534 conn->user_ctx.port = port;
535
536 listen_ctx->on_accept_cb(&conn->user_ctx);
537 }
538
539 return 0;
540}

542static void sctp_conn_close_priv(sctp_conn_priv_t **pconn)
543{
544 sctp_conn_priv_t *conn = *pconn;
545
546 logger_info(conn->user_ctx.logger, "closing connection with %pL",
547 &conn->user_ctx.entity_id);
548
549 el_unregister(&conn->evh);

551 logger_trace(conn->user_ctx.logger, 4, "connection closed");

562sctp_conn_t *sctp_connect(qv_t(lstr) source_addrs, qv_t(lstr) dest_addrs,
563 uint16_t port, size_t sndbuf, lstr_t entity_id,
564 void *priv,
565 sctp_on_connect_f on_connect_cb,
566 sctp_on_data_f on_data_cb,
567 sctp_on_disconnect_f on_disconnect_cb)
568{
569 t_scope;
570 sctp_conn_priv_t *conn;
571 sockunion_t *sus_array;
572 logger_t *logger = &sctp_logger_g;
573 int fd;
574 int id = -1;
575 qv_t(sctp_su) sus;
576 SB_1k(errbuf);
577
578 assert(on_connect_cb);
579 assert(on_data_cb);
580
581 if ((fd = socket(AF_INET, SOCK_STREAM, IPPROTO_SCTP)) < 0) {
582 logger_error(logger, "unable to create socket %m");
583 return NULL;
584 }
585
586 t_qv_init(&sus, 8);
587 if (source_addrs.len) {
588 lstr_t s;
589
590 tab_for_each_entry(addr, &source_addrs) {
591 sockunion_t su;
592
593 if (addr_source_resolve("sourceAddrs", addr, &su, &errbuf) < 0) {
594 logger_error(logger, "%*pM", SB_FMT_ARG(&errbuf));
595 close(fd);
596 return NULL;
597 }
598 qv_append(&sus, su);
599 }
600
601 s = t_sus_to_str(&sus);
602 sus_array = t_sctp_sus_array(&sus, NULL);
603 if (bindx(fd, sus_array, sus.len, SOCK_SEQPACKET, IPPROTO_SCTP,
604 O_NONBLOCK) < 0)
605 {
606 logger_error(logger, "SCTP bind on address(es) %pL failed: %m",
607 &s);
608 close(fd);
609 return NULL;
610 }
611 logger_info(logger, "outgoing connection: bind on `%pL` succeed "
612 "for %pL", &s, &entity_id);
613 qv_clear(&sus);
614 }
615
616 if (addr_list_to_su(dest_addrs, port, &sus) < 0) {
617 logger_error(logger, "failed to get sockunions");
618 goto close_sock;
619 }
620
621 if (sctp_ep_init_sock(fd, sndbuf) < 0) {
622 logger_error(logger, "set SCTP init message failed %m");
623 goto close_sock;
624 }
625 fd_set_features(fd, O_NONBLOCK);
626
627 sus_array = t_sctp_sus_array(&sus, NULL);
628 if (sctp_connectx_ng(fd, (void *)sus_array, sus.len, &id) < 0 &&
629 !ERR_CONNECT_RETRIABLE(errno))
630 {
631 logger_error(logger, "SCTP connectx error: %m");
632 goto close_sock;
633 }
634
635 conn = sctp_conn_priv_new();
636 conn->evh = el_fd_register(fd, true, POLLIN, sctp_connecting, conn);
637 conn->is_listening = false;
638 conn->on_connect_cb = on_connect_cb;
639 conn->on_disconnect_cb = on_disconnect_cb;
640 conn->on_data_cb = on_data_cb;
641 conn->user_ctx.entity_id = lstr_dup(entity_id);
642 conn->user_ctx.priv = priv;
643 conn->user_ctx.logger = logger_new(&sctp_logger_g, entity_id,
644 LOG_INHERITS, 0);
645
646 el_unref(conn->evh);
647
648 return &conn->user_ctx;
649
650close_sock:
651 p_close(&fd);
652 return NULL;
653}
654
655sctp_conn_t *sctp_listen(const lstr__array_t *addrs, int port, size_t sndbuf,
656 lstr_t entity_id, void *priv,
657 sctp_on_accept_f on_accept_cb,
658 sctp_on_data_f on_data_cb,
659 sctp_on_disconnect_f on_disconnect_cb)
660{
661 t_scope;
662 int fd;
663 lstr_t sus_str;
664 qv_t(sctp_su) sus;
665 logger_t *logger = &sctp_logger_g;
666 sctp_conn_priv_t *conn;
667 SB_1k(err);
668
669 t_qv_init(&sus, 8);
670 tab_for_each_entry(addr, addrs) {
671 sockunion_t su;
672
673 sb_reset(&err);
674 if (addr_resolve2("addrs", addr, 0, port, &su, NULL, NULL, &err) < 0)
675 {
676 logger_error(logger, "invalid address: %*pM", SB_FMT_ARG(&err));
677 return NULL;
678 }
679 qv_append(&sus, su);
680 }
681
682 if ((fd = socket(AF_INET, SOCK_STREAM, IPPROTO_SCTP)) < 0) {
683 logger_error(logger, "unable to create socket %m");
684 return NULL;
685 }
686
687 sus_str = t_sus_to_str(&sus);
688
689 if (bindx(fd, t_sctp_sus_array(&sus, NULL), sus.len, SOCK_SEQPACKET,

693 LSTR_FMT_ARG(sus_str));

696 }
697

699 logger_error(logger, "SCTP listen on address %*pM failed: %m",
700 LSTR_FMT_ARG(sus_str));
701 close(fd);
702 return NULL;
703 }
704
705 if (sctp_ep_init_sock(fd, sndbuf) < 0) {
706 logger_error(logger, "set SCTP init message failed %m");
707 close(fd);
708 return NULL;
709 }
710
711 logger_info(logger, "listening on SCTP address(es): %*pM",
712 LSTR_FMT_ARG(sus_str));
713

715 conn->evh = el_fd_register(fd, true, POLLIN,
716 sctp_srv_on_event, conn);;
717 conn->is_listening = true;
718 conn->user_ctx.entity_id = lstr_dup(entity_id);
719 conn->user_ctx.priv = priv;
720 conn->user_ctx.logger = logger;
721 conn->on_accept_cb = on_accept_cb;
722 conn->on_data_cb = on_data_cb;
723 conn->on_disconnect_cb = on_disconnect_cb;
724

726}

728void sctp_send_msg(sctp_conn_t *pub_ctx, lstr_t payload,
729 uint32_t payload_protocol_id)
730{
731 sctp_conn_priv_t *conn = NULL;
732 sctp_msg_t *msg;
733
734 if (!pub_ctx) {
735 logger_error(&sctp_logger_g, "invalid argument pub_ctx");
736 assert(false);
737 return;
738 }
739
740 conn = GET_PRIV_CONN(pub_ctx);
741
742 if (conn->is_listening) {
743 logger_error(conn->user_ctx.logger,
744 "this context is only for accepting new connections, "
745 "it cannot send messages!");
746 assert(false);
747 return;
748 }
749
750 msg = sctp_msg_new();
751 msg->ppid = payload_protocol_id;
752 msg->msg = lstr_dup(payload);
753
754 dlist_add_tail(&conn->msgs, &msg->msg_list);
755
756 el_fd_set_mask(conn->evh, POLLINOUT);
757}
