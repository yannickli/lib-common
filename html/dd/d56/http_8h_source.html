247/** type for HTTPD triggers authentication callbacks.
248 * The authentication callback is always called as soon as there is one on a
249 * given trigger descriptor.
250 *
251 * Though so that the authentication callback can allow non authenticated
252 * content to be returned, it is ALSO called if there was no Authorization:
253 * header in the HTTP query. In that case, \a user and \a pw are set to the
254 * "NULL" pstream (meaning <code>{ NULL, NULL }</code>). In the other case
255 * both \a user and \a pw point to valid NUL-terminated strings.
256 *
257 * If the Authorization field isn't valid, then of course the query is
258 * rejected by the http library and the callback isn't called.
259 *
260 * This callback is fired as soon as the HTTP headers are received, meaning
261 * that for many reasons the actual query may never happen (connection lost,
262 * invalid formatting of the rest of the query, etc…). So be very careful if
263 * you store allocated information in the query descriptor to either:
264 * - use the httpd_trigger_t#query_cls trick, and deallocate this information
265 * in the class destructor;
266 * - make good use of httpd_trigger_t#on_query_wipe to finalize the query.
267 * so that this data gets properly deallocated.
268 *
269 * \param[in] cb the callback that was matched.
270 * \param[in] q the descriptor of the incoming query.
271 * \param[in] user "user" part of the Authorization field.
272 * \param[in] pw "password" part of the Authorization field.
273 */
274typedef void (httpd_trigger_auth_f)(httpd_trigger_t * nonnull cb,
275 struct httpd_query_t * nonnull q,
276 pstream_t user, pstream_t pw);

278/** an HTTP trigger that can be fired on given path fragments.
279 *
280 * An httpd trigger captures everything under the path fragment it was
281 * registered under with #httpd_trigger_register. Unless there is a better
282 * match for the query path.
283 *
284 * Note that an httpd_trigger are meant to be allocated then registered into
285 * one or more httpd trigger trees. Once registered at least once, the trigger
286 * is owned by all those trees.
287 *
288 * If a trigger should survive the http tree it is registered into, then
289 * #httpd_trigger_persist() must be called. #httpd_trigger_loose() is its
290 * contrary, and if the trigger wasn't registered anywhere a call to this
291 * function will destroy the trigger. Hence if you're not sure whether a
292 * trigger you create will be registered or not, the proper sequence to avoid
293 * leaks is:
294 *
295 * <code>
296 * httpd_trigger_t *cb = httpd_trigger_new();
297 *
298 * cb->... = ...; // configure callback
299 * httpd_trigger_persist(cb);
300 * call_some_function_that_may_register_cb_somewhere_or_maybe_not(cb);
301 * httpd_trigger_loose(cb);
302 * </code>
303 */

305 unsigned refcnt;
306 lstr_t auth_realm;
307
308 /* Called before main callback with authentication information. */
309 httpd_trigger_auth_f * nullable auth;
310 const object_class_t * nullable query_cls;
311
312 /* Main callback, called each time the path fragment is queried. */
313 void (* nonnull cb)(httpd_trigger_t * nonnull,
314 struct httpd_query_t * nonnull,
315 const httpd_qinfo_t * nonnull);
316
317 /* Called when trigger is destroyed for cleaning purpose. */
318 void (* nullable destroy)(httpd_trigger_t * nonnull);
319
320 /* Called after each query on the path fragment for cleaning purpose. */
321 void (* nullable on_query_wipe)(struct httpd_query_t * nonnull q);

323 /* Callback on http query exception, it can modify the HTTP error code. */

436/** \typedef http_query_t.
437 * \brief HTTP Query base class.
438 *
439 * An http_query_t is the base class for queries received on an #httpd_t.
440 *
441 * It is refcounted, and is valid until it's answered (no matter if the
442 * underlying #httpd_t available as http_query_t#owner is still valid or not).
443 *
444 * As a consequence it means that nobody should ever suppose that a query is
445 * still valid once #httpd_reply_done() #httpd_reject() or any function using
446 * them internaly (#httpd_reply_202accepted() e.g.) is called.
447 *
448 * If this is required, then use #obj_retain() and #obj_release() accordingly
449 * to ensure the liveness of the #httpd_query_t.
450 *
451 * <h1>How to use an #http_query_t</h1>
452 *
453 * When a the Headers of an HTTP query is received, the matching
454 * #httpd_trigger_t is looked up, the #httpd_query_t (or a subclass if
455 * httpd_trigger_t#query_cls is set) is created. Then if there is a
456 * httpd_trigger_t#auth callback, it is called (possibly with empty
457 * #pstream_t's if there is no Authentication header).
458 *
459 * If the authentication callback hasn't rejected the query, then the
460 * httpd_trigger_t#cb callback is called with the created query (no body
461 * still has been received at this point). This is the moment to setup the
462 * #httpd_query_t on_data/on_done/on_ready e.g. using #httpd_bufferize().
463 *
464 * Note that the httpd_query_t#on_done hook may never ever be called if the
465 * HTTP query was invalid or the connection lost. That's why it's important
466 * to properly use the httpd_trigger_t#on_query_wipe hook (for when no
467 * subclassing of the #httpd_query_t has been done and its
468 * httpd_query_t#priv pointer is used) or properly augment the
469 * httpd_query_t#wipe implementation to wipe all memory clean.
470 *
471 * <h1>Important considerations</h1>
472 *
473 * #httpd_query_t can be answered to asynchronously, but that does not mean
474 * that the underlying connection is still here. One can know at each time
475 * if there is still a connection looking at httpd_query_t#owner pointer. If
476 * it's NULL, the #httpd_t is dead. In that case it is correct to
477 * #obj_release() the query and go away since anything that would else be
478 * answered would be discarded anyway.
479 */
