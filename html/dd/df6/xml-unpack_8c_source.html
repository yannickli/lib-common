36static int parse_int(xml_reader_t xr, const char *s, int64_t *i64p)
37{
38 errno = 0;
39 *i64p = strtoll(s, &s, 0);
40 if (unlikely(skipspaces(s)[0] || errno))
41 return xmlr_fail(xr, "node value is not a valid integer");
42 return 0;
43}

45/* parse the href attribute (attr) which contains a Content-Id and return the
46 * associated message part found in parts_g qm.
47 * Example: href="cid:12345"
48 */
49static int get_part_from_href(xml_reader_t xr, xmlAttrPtr *attr, lstr_t *part)
50{
51 t_scope;
52
53 lstr_t href, cid;
54 pstream_t ps;
55 int pos;
56
57 if (!parts_g)
58 return xmlr_fail(xr, "found href attribute with no message parts");
59
60 if (t_xmlr_getattr_str(xr, *attr, false, &href) < 0)
61 return xmlr_fail(xr, "failed to read href");
62
63 ps = ps_initlstr(&href);
64 if (ps_skipstr(&ps, "cid:") < 0)
65 return xmlr_fail(xr, "failed to parse href");
66
67 cid = LSTR_INIT_V(ps.s, ps_len(&ps));
68 pos = qm_find(part, parts_g, &cid);
69 if (pos < 0)
70 return xmlr_fail(xr, "unknown cid in href");
71
72 *part = parts_g->values[pos];
73 return 0;
74}

76/* unpack a string value, supporting references to message parts */
77static int get_text(xml_reader_t xr, mem_pool_t *mp, bool b64, lstr_t *str)
78{
79 xmlAttrPtr attr;
80 lstr_t part;
81
82 if (RETHROW(xmlr_node_is_empty(xr))) {
83 /* empty element -> check for a href attribute (SOAP package) */
84 if ((attr = xmlr_find_attr_s(xr, "href", false))) {
85 RETHROW(get_part_from_href(xr, &attr, &part));
86 *str = lstr_dupc(part);
87 } else {
88 *str = LSTR_EMPTY_V;
89 }
90 RETHROW(xmlr_next_node(xr));
91 return 0;
92 }
93
94 RETHROW(xmlr_get_cstr_start(xr, true, str));
95 if (str->len == 0) {
96 /* no text -> check for an empty Include element with a href
97 * attribute (XOP package) */
98 if (!RETHROW(xmlr_node_is_closing(xr))) {
99 RETHROW(xmlr_node_want_s(xr, "Include"));
100 if (!RETHROW(xmlr_node_is_empty(xr))) {
101 return xmlr_fail(xr, "Include element must be empty");
102 }
103 RETHROW_PN((attr = xmlr_find_attr_s(xr, "href", true)));
104 RETHROW(get_part_from_href(xr, &attr, &part));
105 *str = lstr_dupc(part);
106 RETHROW(xmlr_node_close(xr));
107 } else {
108 *str = LSTR_EMPTY_V;
109 }
110 } else {
111 /* common case: text not empty */
112 if (!b64) {
113 *str = mp_lstr_dup(mp, *str);
114 } else {
115 sb_t sb;
116 int blen = DIV_ROUND_UP(str->len * 3, 4);
117 char *buf = mp_new_raw(mp, char, blen + 1);
118
119 sb_init_full(&sb, buf, 0, blen + 1, &mem_pool_static);
120 if (sb_add_unb64(&sb, str->s, str->len)) {
121 mp_delete(mp, &buf);
122 return xmlr_fail(xr, "value isn't valid base64");
123 }
124 str->s = buf;
125 str->len = sb.len;
126 }
127 }
128 RETHROW(xmlr_get_cstr_done(xr));
129 return 0;
130}

132static int get_enum_value(xml_reader_t xr, const iop_enum_t *en_desc,
133 int64_t *intval)
134{

149static int xunpack_value(xml_reader_t xr, mem_pool_t *mp,
150 const iop_field_t *fdesc, void *v, int flags)
151{
152 int64_t intval = 0;
153 uint64_t uintval = 0;
154 lstr_t *str = NULL;
155
156 switch (fdesc->type) {
157#define CHECK_RANGE(_min, _max) \
158 THROW_ERR_IF(intval < _min || intval > _max)
159
160 case IOP_T_I8:
161 RETHROW(xmlr_get_i64_base(xr, 0, &intval));
162 CHECK_RANGE(INT8_MIN, INT8_MAX);
163 *(int8_t *)v = intval;
164 break;
165 case IOP_T_U8:
166 RETHROW(xmlr_get_i64_base(xr, 0, &intval));
167 CHECK_RANGE(0, UINT8_MAX);
168 *(uint8_t *)v = intval;
169 break;
170 case IOP_T_I16:
171 RETHROW(xmlr_get_i64_base(xr, 0, &intval));
172 CHECK_RANGE(INT16_MIN, INT16_MAX);
173 *(int16_t *)v = intval;
174 break;
175 case IOP_T_U16:
176 RETHROW(xmlr_get_i64_base(xr, 0, &intval));
177 CHECK_RANGE(0, UINT16_MAX);
178 *(uint16_t *)v = intval;
179 break;
180 case IOP_T_I32:
181 RETHROW(xmlr_get_i64_base(xr, 0, &intval));
182 CHECK_RANGE(INT32_MIN, INT32_MAX);
183 *(int32_t *)v = intval;
184 break;
185 case IOP_T_U32:
186 RETHROW(xmlr_get_i64_base(xr, 0, &intval));
187 CHECK_RANGE(0, UINT32_MAX);
188 *(uint32_t *)v = intval;
189 break;
190 case IOP_T_I64:
191 RETHROW(xmlr_get_i64_base(xr, 0, &intval));
192 *(int64_t *)v = intval;
193 break;
194 case IOP_T_U64:
195 RETHROW(xmlr_get_u64_base(xr, 0, &uintval));
196 *(uint64_t *)v = uintval;
197 break;
198 case IOP_T_ENUM:
199 RETHROW(get_enum_value(xr, fdesc->u1.en_desc, &intval));
200 CHECK_RANGE(INT32_MIN, INT32_MAX);
201 *(int32_t *)v = intval;
202 break;
203 case IOP_T_BOOL:
204 RETHROW(xmlr_get_bool(xr, (bool *)v));
205 break;
206 case IOP_T_DOUBLE:
207 RETHROW(xmlr_get_dbl(xr, (double *)v));
208 break;
209 case IOP_T_STRING:
210 str = v;
211 RETHROW(get_text(xr, mp, false, str));
212 break;
213 case IOP_T_DATA:
214 str = v;
215 RETHROW(get_text(xr, mp, true, str));
216 break;
217 case IOP_T_XML:
218 str = v;
219 RETHROW(mp_xmlr_get_inner_xml(mp, xr, str));
220 break;
221 case IOP_T_UNION:
222 return xunpack_union(xr, mp, fdesc->u1.st_desc, v, flags);
223 case IOP_T_STRUCT:
224 if (iop_field_is_class(fdesc)) {
225 *(void **)v = NULL;
226 return xunpack_class(xr, mp, fdesc->u1.st_desc, v, flags);
227 } else {
228 return xunpack_struct(xr, mp, fdesc->u1.st_desc, v, flags);
229 }
230 case IOP_T_VOID: {
231 int i = 0;
232
233 do {
234 i++;
235 xmlr_next_sibling(xr);
236 } while (xmlr_node_is(xr, fdesc->name.s, fdesc->name.len));
237
238 e_named_trace(3, "iop/xml/unpacker", "dropped %d value(s) into void "
239 "field `%*pM`", i, LSTR_FMT_ARG(fdesc->name));
240 break;
241 }
242
243#undef CHECK_RANGE
244 default:
245 e_panic("should not happen");
246 }
247
248 return 0;
249}

251/* Unpack a vector of scalar values. Because a scalar value do not recurse in
252 * this function we can safely use realloc. */
253static int xunpack_scalar_vec(xml_reader_t xr, mem_pool_t *mp,
254 const iop_field_t *fdesc, void *v)
255{
256 lstr_t *data = v;
257 int bufsize = 0, datasize = fdesc->size;
258
259 p_clear(data, 1);
260
261 do {
262 int64_t intval = 0;
263 uint64_t uintval = 0;
264
265 if (datasize >= bufsize) {
266 int size = p_alloc_nr(bufsize);
267 data->data = mp_irealloc(mp, data->data, bufsize, size, 8, MEM_RAW);
268 bufsize = size;
269 }
270
271 switch (fdesc->type) {
272#define CHECK_RANGE(_min, _max) \
273 THROW_ERR_IF(intval < _min || intval > _max)
274
275 case IOP_T_I8:
276 RETHROW(xmlr_get_i64_base(xr, 0, &intval));
277 CHECK_RANGE(INT8_MIN, INT8_MAX);
278 ((int8_t *)data->data)[data->len] = intval;
279 break;
280 case IOP_T_U8:
281 RETHROW(xmlr_get_i64_base(xr, 0, &intval));
282 CHECK_RANGE(0, UINT8_MAX);
283 ((uint8_t *)data->data)[data->len] = intval;
284 break;
285 case IOP_T_I16:
286 RETHROW(xmlr_get_i64_base(xr, 0, &intval));
287 CHECK_RANGE(INT16_MIN, INT16_MAX);
288 ((int16_t *)data->data)[data->len] = intval;
289 break;
290 case IOP_T_U16:
291 RETHROW(xmlr_get_i64_base(xr, 0, &intval));
292 CHECK_RANGE(0, UINT16_MAX);
293 ((uint16_t *)data->data)[data->len] = intval;
294 break;
295 case IOP_T_I32:
296 RETHROW(xmlr_get_i64_base(xr, 0, &intval));
297 CHECK_RANGE(INT32_MIN, INT32_MAX);
298 ((int32_t *)data->data)[data->len] = intval;
299 break;
300 case IOP_T_U32:
301 RETHROW(xmlr_get_i64_base(xr, 0, &intval));
302 CHECK_RANGE(0, UINT32_MAX);
303 ((uint32_t *)data->data)[data->len] = intval;
304 break;
305 case IOP_T_I64:
306 RETHROW(xmlr_get_i64_base(xr, 0, &intval));
307 ((int64_t *)data->data)[data->len] = intval;
308 break;
309 case IOP_T_U64:
310 RETHROW(xmlr_get_u64_base(xr, 0, &uintval));
311 ((uint64_t *)data->data)[data->len] = uintval;
312 break;
313 case IOP_T_ENUM:
314 RETHROW(get_enum_value(xr, fdesc->u1.en_desc, &intval));
315 CHECK_RANGE(INT32_MIN, INT32_MAX);
316 ((int32_t *)data->data)[data->len] = intval;
317 break;
318 case IOP_T_BOOL:
319 RETHROW(xmlr_get_bool(xr, (bool *)data->data + data->len));
320 break;
321 case IOP_T_DOUBLE:
322 RETHROW(xmlr_get_dbl(xr, (double *)data->data + data->len));
323 break;
324
325#undef CHECK_RANGE
326 default:
327 assert(false);
328 }
329
330 data->len++;
331 datasize += fdesc->size;
332
333 /* Check for another repeated element */
334 } while (RETHROW(xmlr_node_is(xr, fdesc->name.s, fdesc->name.len)));
335 return 0;
336}

338/* Unpack a vector of "block" values (structure | union | data | string).
339 * We can't safely use realloc because a block has an unkown length.
340 * We use a hack to chain each unpacked value and rebuild a continuous array.
341 *
342 * n = 0
343 * foreach value
344 * the value is unpacked on the stack
345 * two pointers are added on the stack (the chain)
346 * [0] point on the last chain
347 * [1] point on the data we've just added
348 * n++
349 * next
350 *
351 * then allocates a new buffer of (fdesc->size * n) and read back the chain
352 * to fill the new continous array.
353 */
354static int xunpack_block_vec(xml_reader_t xr, mem_pool_t *mp,
355 const iop_field_t *fdesc, void *v, int flags)
356{
357 lstr_t *data = v;
358 void **prev = NULL, **chain, *ptr;
359 int n = 0;
360
361 do {
362 ptr = mp_new_raw(mp, char, fdesc->size);
363
364 RETHROW(xunpack_value(xr, mp, fdesc, ptr, flags));
365 n++;
366
367 chain = mp_new_raw(mp, void *, 2);
368 chain[0] = prev;
369 chain[1] = ptr;
370 prev = chain;
371 } while (RETHROW(xmlr_node_is(xr, fdesc->name.s, fdesc->name.len)));
372
373 /* Now we can rebuild the array of value */
374 data->len = n;
375 data->data = mp_imalloc(mp, fdesc->size * n, 8, MEM_RAW);
376 ptr = (char *)data->data + (n - 1) * fdesc->size;
377 while (n--) {
378 memcpy(ptr, chain[1], fdesc->size);
379 ptr = (char *)ptr - fdesc->size;
380 chain = chain[0];
381 }
382
383 return 0;
384}

407static int
408__xunpack_struct(xml_reader_t xr, mem_pool_t *mp, void *value, int flags,
409 qv_t(iop_xfield) *fields)
410{
411 const iop_xfield_t *fdesc = fields->tab;
412 const iop_xfield_t *end = fields->tab + fields->len;
413 int res = xmlr_next_child(xr);
414
415 /* No children */
416 if (res == XMLR_NOCHILD)
417 goto end;
418 if (res < 0)
419 return res;
420 do {
421 const iop_xfield_t *xfdesc;
422 lstr_t name;
423 void *v;
424 int n = 1;
425
426 if (unlikely(fdesc == end)) {
427 if (!(flags & IOP_UNPACK_IGNORE_UNKNOWN))
428 return xmlr_fail(xr, "expecting closing tag");
429 return xmlr_next_uncle(xr);
430 }
431
432 /* Find the field description by the tag name */
433 RETHROW(xmlr_node_get_local_name(xr, &name));
434 xfdesc = get_xfield_by_name(fdesc, end, name);
435 if (unlikely(!xfdesc)) {
436 if (!(flags & IOP_UNPACK_IGNORE_UNKNOWN))
437 return xmlr_fail(xr, "unknown tag <%*pM>", LSTR_FMT_ARG(name));
438 do {
439 RETHROW(xmlr_next_sibling(xr));
440 if (RETHROW(xmlr_node_is_closing(xr)))
441 goto end;
442 RETHROW(xmlr_node_get_local_name(xr, &name));
443 xfdesc = get_xfield_by_name(fdesc, end, name);
444 } while (!xfdesc);
445 }
446
447 if (flags & IOP_UNPACK_FORBID_PRIVATE) {
448 const iop_field_attrs_t *attrs;
449
450 attrs = iop_field_get_attrs(xfdesc->desc, xfdesc->fdesc);
451 if (attrs && TST_BIT(&attrs->flags, IOP_FIELD_PRIVATE)) {
452 return xmlr_fail(xr, "private tag <%*pM>", LSTR_FMT_ARG(name));
453 }
454 }
455
456 /* Handle optional fields */
457 while (unlikely(xfdesc != fdesc)) {
458 if (iop_skip_absent_field_desc(mp, value, fdesc->desc,
459 fdesc->fdesc) < 0)
460 {
461 return xmlr_fail(xr, "missing mandatory tag <%*pM>",
462 LSTR_FMT_ARG(fdesc->fdesc->name));
463 }
464 fdesc++;
465 }
466
467 /* Read field value */
468 v = (char *)value + fdesc->fdesc->data_offs;
469 if (fdesc->fdesc->repeat == IOP_R_REPEATED) {
470 lstr_t *data = v;
471
472 if ((1 << fdesc->fdesc->type) & IOP_BLK_OK) {
473 RETHROW(xunpack_block_vec(xr, mp, fdesc->fdesc, v, flags));
474 } else {
475 RETHROW(xunpack_scalar_vec(xr, mp, fdesc->fdesc, v));
476 }
477 v = data->data;
478 n = data->len;
479 goto next;
480 } else
481 if (iop_field_is_reference(fdesc->fdesc)) {
482 v = iop_field_ptr_alloc(mp, fdesc->fdesc, v);
483 } else
484 if (fdesc->fdesc->repeat == IOP_R_OPTIONAL
485 && !iop_field_is_class(fdesc->fdesc))
486 {
487 v = iop_field_set_present(mp, fdesc->fdesc, v);
488 }
489
490 RETHROW(xunpack_value(xr, mp, fdesc->fdesc, v, flags));
491
492 next:
493 if (unlikely(iop_field_has_constraints(fdesc->desc, fdesc->fdesc))) {
494 if (iop_field_check_constraints(fdesc->desc, fdesc->fdesc, v, n,
495 false) < 0)
496 {
497 return xmlr_fail(xr, "%*pM", LSTR_FMT_ARG(iop_get_err_lstr()));
498 }
499 }
500 fdesc++;
501 } while (!xmlr_node_is_closing(xr));
502
503 /* Check for absent fields */
504 end:
505 for (; fdesc < end; fdesc++) {
506 if (iop_skip_absent_field_desc(mp, value, fdesc->desc, fdesc->fdesc)
507 < 0)
508 {
509 return xmlr_fail(xr, "missing mandatory tag <%*pM>",
510 LSTR_FMT_ARG(fdesc->fdesc->name));
511 }
512 }
513 return xmlr_node_close(xr);
514}

541static int
542xunpack_class(xml_reader_t xr, mem_pool_t *mp, const iop_struct_t *desc,
543 void **value, int flags)
544{
545 const iop_struct_t *real_desc, *desc_it;
546 qv_t(iop_struct) parents;
547 qv_t(iop_xfield) fields;
548 bool found_desc = false;
549 int res;
550
551 /* Get the real class type. Create a t_scope here because mp could be (and
552 * is most of time) t_pool(). */
553 {
554 t_scope;
555 xmlAttrPtr attr = xmlr_find_attr_s(xr, "type", false);
556 lstr_t real_type_str;
557 pstream_t ps;
558
559 if (!attr) {
560 if (desc->class_attrs->is_abstract) {
561 return xmlr_fail(xr, "type attribute not found (mandatory "
562 "for abstract classes)");
563 }
564
565 /* If type attribute is not present, consider we are unpacking a
566 * class of the expected type. */
567 real_desc = desc;
568 goto build_parents;
569 }
570
571 RETHROW(t_xmlr_getattr_str(xr, attr, false, &real_type_str));
572 ps = ps_initlstr(&real_type_str);
573 /* Skip mandatory namespace */
574 ps_skip_afterchr(&ps, ':');
575 if (!(real_desc = iop_get_class_by_fullname(desc, LSTR_PS_V(&ps)))) {
576 return xmlr_fail(xr, "class `%*pM' not found",
577 PS_FMT_ARG(&ps));
578 }
579 }
580
581
582 if (real_desc->class_attrs->is_abstract) {
583 return xmlr_fail(xr, "class `%*pM' is an abstract class",
584 LSTR_FMT_ARG(real_desc->fullname));
585 }
586
587 build_parents:
588 if (flags & IOP_UNPACK_FORBID_PRIVATE
589 && real_desc->class_attrs->is_private)
590 {
591 return xmlr_fail(xr, "class `%*pM` is private",
592 LSTR_FMT_ARG(real_desc->fullname));
593 }
594
595 /* The fields will be present in the order "master -> children", not
596 * "children -> master", so build a qvector of the parents.
597 * Also check this the types are compatible. */
598 qv_inita(&parents, 8);
599 desc_it = real_desc;
600 do {
601 qv_append(&parents, desc_it);
602 if (desc_it == desc) {
603 found_desc = true;
604 }
605 } while ((desc_it = desc_it->class_attrs->parent));
606 if (!found_desc) {
607 xmlr_fail(xr, "class `%*pM' is not a child of `%*pM'",
608 LSTR_FMT_ARG(real_desc->fullname),
609 LSTR_FMT_ARG(desc->fullname));
610 qv_wipe(&parents);
611 return -1;
612 }
613
614 /* Allocate output value */
615 *value = mp_irealloc(mp, *value, 0, real_desc->size, 8, MEM_RAW);
616
617 /* Set the _vprt pointer */
618 *(const iop_struct_t **)(*value) = real_desc;
619
620 /* Build fields vector, and unpack fields */
621 qv_inita(&fields, 32);
622 for (int pos = parents.len; pos-- > 0; ) {
623 qv_append_struct_xfields(&fields, parents.tab[pos]);
624 }
625 qv_wipe(&parents);
626
627 res = __xunpack_struct(xr, mp, *value, flags, &fields);
628
629 qv_wipe(&fields);
630 return res;
631}

633static int
634xunpack_union(xml_reader_t xr, mem_pool_t *mp, const iop_struct_t *desc,
635 void *value, int flags)
636{
637 const iop_field_t *fdesc;
638 lstr_t name;
639
640 RETHROW(xmlr_next_child(xr));
641 RETHROW(xmlr_node_get_local_name(xr, &name));
642 fdesc = get_field_by_name(desc, desc->fields, name.s, name.len);
643 if (unlikely(!fdesc))
644 return xmlr_fail(xr, "unknown tag <%*pM>", LSTR_FMT_ARG(name));
645
646 /* Write the selected tag */
647 iop_union_set_tag(desc, fdesc->tag, value);
648 value = (char *)value + fdesc->data_offs;
649

667 * It will be (re)allocated when the size of the real class to unpack will be
668 * known.

673{
