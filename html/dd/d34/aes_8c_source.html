307void aes_setkey_enc(aes_ctx *ctx, const byte *key, int keysize)
308{
309 int i;
310 uint32_t *RK;
311
312 switch (keysize) {
313 case 128: ctx->nr = 10; break;
314 case 192: ctx->nr = 12; break;

321#else
322 ctx->rk = RK = ctx->buf;
323#endif

326#if __BYTE_ORDER == __BIG_ENDIAN

328 RK[i] = swap32(RK[i]);
