25double double_round(double val, uint8_t precision)
26{
27 double val_floor;
28
29 if (isinf(val) || isnan(val)) {
30 return val;
31 }
32
33 val_floor = floor(val);
34 if (!expect(precision < countof(powerof10))) {
35 return val;
36 }
37
38 val -= val_floor;
39 val *= powerof10[precision];
40 val = round(val);
41 val /= powerof10[precision];
42
43 return val + val_floor;
44}

46double double_round_significant(double d, uint8_t precision)
47{
48 double base = round(d);
49 uint64_t ubase = (uint64_t)(base < 0 ? -base : base);
50 size_t base_nb_digits = 0;
51
52 if (!expect(precision < countof(powerof10))) {
53 return d;
54 }
55 assert (precision != 0);
56
57 /* powerof10[X-1] is the smallest number with precision X */
58 if (ubase >= powerof10[precision - 1]) {
59 return base;
60 }
61
62 while (ubase >= powerof10[base_nb_digits]) {
63 base_nb_digits++;
64 }
65
66 return double_round(d, precision - base_nb_digits);
67}
