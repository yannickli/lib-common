42static int t_parse_json(ichttp_query_t *iq, ichttp_cb_t *cbe, void **vout)
43{
44 httpd_trigger__ic_t *tcb = container_of(iq->trig_cb, httpd_trigger__ic_t, cb);
45 const iop_struct_t *st = cbe->fun->args;
46 pstream_t ps;
47 iop_json_lex_t jll;
48 int res = 0;
49 SB_8k(buf);
50
51 *vout = NULL;
52 iop_jlex_init(t_pool(), &jll);
53 ps = ps_initsb(&iq->payload);
54 iop_jlex_attach(&jll, &ps);
55
56 jll.flags = tcb->unpack_flags;
57
58 if (iop_junpack_ptr(&jll, st, vout, true)) {
59 sb_reset(&buf);
60 iop_jlex_write_error(&jll, &buf);
61
62 __ichttp_err_ctx_set(LSTR_SB_V(&buf));
63 httpd_reject(obj_vcast(httpd_query, iq), BAD_REQUEST, "%s", buf.data);
64 __ichttp_err_ctx_clear();
65 res = -1;
66 goto end;
67 }
68 iop_jlex_detach(&jll);
69
70 end:
71 iop_jlex_wipe(&jll);
72 return res;
73}

75static int t_parse_soap(ichttp_query_t *iq,
76 ichttp_cb_t **cbout, void **vout)
77{
78#define xr xmlr_g
79#define XCHECK(expr) XMLR_CHECK(expr, goto xmlerror)
80
81 const char *buf = iq->payload.data;
82 int len = iq->payload.len;
83
84 httpd_trigger__ic_t *tcb = container_of(iq->trig_cb, httpd_trigger__ic_t, cb);
85 ichttp_cb_t *cbe;
86 lstr_t s;
87 int pos;
88
89 /* Initialize the xmlReader object */
90 XCHECK(xmlr_setup(&xr, buf, len));
91 XCHECK(xmlr_node_open_s(xr, "Envelope"));
92 if (XCHECK(xmlr_node_is_s(xr, "Header")))
93 XCHECK(xmlr_next_sibling(xr));
94 XCHECK(xmlr_node_open_s(xr, "Body"));
95 XCHECK(xmlr_node_get_local_name(xr, &s));
96 pos = qm_find(ichttp_cbs, &tcb->impl, &s);
97 if (pos < 0) {
98 __ichttp_reply_soap_err_cst(ichttp_query_to_slot(iq), false,
99 "unknown rpc");
100 goto error;
101 }
102 iq->cbe = *cbout = cbe = ichttp_cb_retain(tcb->impl.values[pos]);
103
104 XCHECK(iop_xunpack_ptr_flags(xr, t_pool(), cbe->fun->args, vout,
105 tcb->unpack_flags));
106 /* Close opened elements */
107
108 XCHECK(xmlr_node_close(xr)); /* </Body> */
109 XCHECK(xmlr_node_close(xr)); /* </Envelope> */
110 xmlr_close(&xr);
111 return 0;
112
113 xmlerror:
114 s = LSTR(xmlr_get_err() ?: "parsing error");
115 __ichttp_reply_soap_err(ichttp_query_to_slot(iq), false, &s);
116 error:
117 xmlr_close(&xr);
118 return -1;
119
120#undef XCHECK
121#undef xr
122}

124static int is_ctype_json(const httpd_qinfo_t *info)
125{
126 const http_qhdr_t *ctype;
127
128 ctype = http_qhdr_find(info->hdrs, info->hdrs_len,
129 HTTP_WKHDR_CONTENT_TYPE);
130 if (ctype) {
131 pstream_t v = ctype->val;
132
133 ps_skipspaces(&v);
134 return ps_skipcasestr(&v, "application/json") == 0;
135 }
136 return false;
137}

139int __t_ichttp_query_on_done_stage1(httpd_query_t *q, ichttp_cb_t **cbe,
140 void **value, bool *soap)
141{
142 ichttp_query_t *iq = obj_vcast(ichttp_query, q);
143 httpd_trigger__ic_t *tcb = container_of(iq->trig_cb, httpd_trigger__ic_t, cb);
144 pstream_t url = iq->qinfo->query;
145 int res;
146
147 *soap = false;
148 ps_skipstr(&url, "/");
149
150 if (ps_len(&url)) {
151 const char *url_s = url.s;
152 lstr_t s;
153
154 if (ps_skip_uptochr(&url, '/') < 0) {
155 not_found:
156 httpd_reject(obj_vcast(httpd_query, iq), NOT_FOUND, "");
157 return -1;
158 }
159 __ps_skip(&url, 1);
160 if (ps_skip_uptochr(&url, '/') < 0) {
161 s = LSTR_INIT_V(url_s, url.s_end - url_s);
162 } else {
163 s = LSTR_INIT_V(url_s, url.s - url_s);
164 }
165 res = qm_find(ichttp_cbs, &tcb->impl, &s);
166 if (res < 0)
167 goto not_found;
168 iq->cbe = *cbe = ichttp_cb_retain(tcb->impl.values[res]);
169
170 if (is_ctype_json(q->qinfo)) {
171 iq->json = true;
172 res = t_parse_json(iq, *cbe, value);
173 } else {
174 httpd_reject(obj_vcast(httpd_query, iq), NOT_ACCEPTABLE,
175 "Content-Type must be application/json");
176 return -1;
177 }
178 } else {
179 *soap = true;
180 return 0;
181 }
182 if (unlikely(res < 0))
183 return -1;
184 return 0;
185}

187void __t_ichttp_query_on_done_stage2(httpd_query_t *q, ichttp_cb_t *cbe,
188 void *value)
189{
190 ichttp_query_t *iq = obj_vcast(ichttp_query, q);
191 httpd_trigger__ic_t *tcb = container_of(iq->trig_cb, httpd_trigger__ic_t,
192 cb);
193 ic__hdr__t default_hdr = IOP_UNION_VA(ic__hdr, simple,
194 .kind = LSTR_OPT(tcb->auth_kind),
195 .payload = q->received_body_length,
196 .source = LSTR("webservice"),
197 .workspace_id = OPT_NONE,
198 );
199 ic_cb_entry_t *e;
200
201 pstream_t login, pw;
202 uint64_t slot = ichttp_query_to_slot(iq);
203 ichannel_t *pxy;
204 ic__hdr__t *pxy_hdr = NULL;
205 bool force_pxy_hdr = false;
206 bool hdr_modified = false;
207 ic__hdr__t *hdr;
208 ichttp_query_t *ic_q = obj_vcast(ichttp_query, q);
209 ic_msg_t *msg;
210
211 if (ic_q && ic_q->ic_hdr) {
212 hdr = ic_q->ic_hdr;
213 } else {
214 hdr = &default_hdr;
215
216 if (t_httpd_qinfo_get_basic_auth(q->qinfo, &login, &pw) == 0) {
217 hdr->simple.login = LSTR_PS_V(&login);
218 hdr->simple.password = LSTR_PS_V(&pw);
219 }
220 hdr->simple.host = httpd_get_peer_address(q->owner);
221 }
222
223 e = &cbe->e;
224 if (ic_query_do_pre_hook(NULL, slot, e, hdr, &hdr_modified) < 0) {
225 return;
226 }
227
228 switch (e->cb_type) {
229 case IC_CB_NORMAL:
230 case IC_CB_WS_SHARED:
231 t_seal();
232
233 if (e->cb_type == IC_CB_NORMAL_BLK) {
234 assert (false);
235 } else {
236 (*e->u.iws_cb.cb)(NULL, slot, value, hdr);
237 }
238 if (cbe->fun->async)
239 httpd_reply_202accepted(q);
240
241 t_unseal();
242 return;
243
244 case IC_CB_PROXY_P:
245 pxy = e->u.proxy_p.ic_p;
246 pxy_hdr = e->u.proxy_p.hdr_p;
247 break;
248 case IC_CB_PROXY_PP:
249 pxy = *e->u.proxy_pp.ic_pp;
250 if (e->u.proxy_pp.hdr_pp)
251 pxy_hdr = *e->u.proxy_pp.hdr_pp;
252 break;
253 case IC_CB_DYNAMIC_PROXY:
254 {
255 ic_dynproxy_t dynproxy;
256
257 /* XXX dynproxy are allowed to return memory allocated on the
258 * t_pool() and thus mustn't be wrapped in a local t_scope */
259 dynproxy = (*e->u.dynproxy.get_ic)(hdr, e->u.dynproxy.priv);
260 pxy = dynproxy.ic;
261 pxy_hdr = dynproxy.hdr;
262 force_pxy_hdr = pxy_hdr != NULL;
263 }
264 break;
265 default:
266 e_panic("should not happen");
267 break;
268 }
269
270 if (unlikely(!pxy)) {
271 __ichttp_reply_err(slot, IC_MSG_PROXY_ERROR, NULL);
272 return;
273 }
274
275 if (hdr_modified && expect(hdr)) {
276 pxy_hdr = force_pxy_hdr ? pxy_hdr : hdr;
277 force_pxy_hdr = true;
278 }
279
280 msg = ic_msg_new(sizeof(uint64_t));
281
282 if ((!ps_len(&login) || force_pxy_hdr) && pxy_hdr) {
283 /* XXX on simple header we write the payload size of the HTTP
284 * query */
285 if (unlikely(IOP_UNION_IS(ic__hdr, pxy_hdr, simple))) {
286 ic__simple_hdr__t *shdr = &pxy_hdr->simple;
287 shdr->payload = iq->payload.len;
288 }
289
290 msg->hdr = pxy_hdr;
291 } else {
292 assert(!pxy_hdr);
293 /* XXX We do not support header replacement with proxyfication */
294 msg->hdr = hdr;
295 }
296 msg->cmd = cbe->cmd;
297 msg->rpc = cbe->fun;
298 msg->async = cbe->fun->async;
299
300 if (!msg->async) {
301 msg->cb = IC_PROXY_MAGIC_CB;
302 put_unaligned_cpu64(msg->priv, slot);
303 }
304 __ic_bpack(msg, cbe->fun->args, value);
305 __ic_query(pxy, msg);
306 if (msg->async) {
307 httpd_reply_202accepted(q);
308 }
309}

311static void ichttp_query_on_done(httpd_query_t *q)
312{
313 t_scope;
314
315 ichttp_query_t *iq = obj_vcast(ichttp_query, q);
316
317 int res;
318 bool soap = false;
319 ichttp_cb_t *cbe = NULL;
320 void *value = NULL;
321
322 res = __t_ichttp_query_on_done_stage1(q, &cbe, &value, &soap);
323 if (unlikely(res < 0))
324 return;
325 if (soap) {
326 res = t_parse_soap(iq, &cbe, &value);
327 if (unlikely(res < 0))
328 return;
329 }
330 __t_ichttp_query_on_done_stage2(q, cbe, value);
331}

370static ichttp_cb_t *
371ichttp_register_function(httpd_trigger__ic_t *tcb,
372 const iop_iface_alias_t *alias, const iop_rpc_t *fun,
373 int32_t cmd, const ic_cb_entry_t *_cb)
374{
375 ichttp_cb_t *cb = ichttp_cb_new();
376
377 cb->cmd = cmd;
378 cb->fun = fun;
379 cb->name = lstr_fmt("%s.%sReq", alias->name.s, fun->name.s);
380 cb->name_uri = lstr_fmt("%s/%s", alias->name.s, fun->name.s);
381 cb->name_res = lstr_fmt("%s.%sRes", alias->name.s, fun->name.s);
382 cb->name_exn = lstr_fmt("%s.%s.Fault", alias->name.s, fun->name.s);
383 cb->e = *_cb;
384
385 /* Register RPC name (takes ownership of "cb") */
386 e_assert_n(panic, qm_add(ichttp_cbs, &tcb->impl, &cb->name, cb),
387 "RPC %s.%s", alias->name.s, fun->name.s);
388 /* Register RPC URI (duplicates "cb") */
389 e_assert_n(panic, qm_add(ichttp_cbs, &tcb->impl, &cb->name_uri,
390 ichttp_cb_retain(cb)),
391 "RPC %s.%s", alias->name.s, fun->name.s);
392
393 return cb;
394}

396ichttp_cb_t *
397__ichttp_register(httpd_trigger__ic_t *tcb, const iop_iface_alias_t *alias,
398 const iop_rpc_t *fun, int32_t cmd, const ic_cb_entry_t *cb)
399{
400 const unsigned fun_flags = fun->flags;
401
402 if (TST_BIT(&fun_flags, IOP_RPC_HAS_ALIAS)) {
403 const iop_rpc_attrs_t *attrs = iop_rpc_get_attrs(alias->iface, fun);
404
405 for (int i = 0; i < attrs->attrs_len; i++) {
406 const iop_rpc_attr_t attr = attrs->attrs[i];
407
408 if (attr.type == IOP_RPC_ALIAS) {
409 const iop_rpc_attr_arg_t arg = attr.args[0];
410 const iop_rpc_t *fun_alias = (const iop_rpc_t *)arg.v.p;
411
412 /* The alias callback will never leave the trigger hashtable
413 * and will be destroyed with it. */
414 ichttp_register_function(tcb, alias, fun_alias, cmd, cb);
415 }
416 }
417 }
418
419 return ichttp_register_function(tcb, alias, fun, cmd, cb);
420}

443static void ichttp_serialize_soap(sb_t *sb, ichttp_query_t *iq, int cmd,
444 const iop_struct_t *st, const void *v)
445{
446 xmlpp_t pp;
447 httpd_trigger__ic_t *tcb;
448
449 tcb = container_of(iq->trig_cb, httpd_trigger__ic_t, cb);
450
451 xmlpp_open_banner(&pp, sb);
452 pp.nospace = true;
453 xmlpp_opentag(&pp, "s:Envelope");
454 xmlpp_putattr(&pp, "xmlns:s", "http://schemas.xmlsoap.org/soap/envelope/");
455 xmlpp_putattr(&pp, "xmlns:n", tcb->schema);
456 xmlpp_putattr(&pp, "xmlns:xsi",
457 "http://www.w3.org/2001/XMLSchema-instance");
458
459 xmlpp_opentag(&pp, "s:Body");
460 if (cmd == IC_MSG_OK) {
461 ichttp_cb_t *cbe = iq->cbe;
462
463 if (v) {
464 if (iop_struct_is_class(st)) {
465 const iop_struct_t *real_st = *(const iop_struct_t **)v;
466
467 sb_addf(sb, "<n:%*pM xsi:type=\"n:%*pM\">",
468 LSTR_FMT_ARG(cbe->name_res),
469 LSTR_FMT_ARG(real_st->fullname));
470 } else {
471 sb_addf(sb, "<n:%*pM>", LSTR_FMT_ARG(cbe->name_res));
472 }
473 iop_xpack_flags(sb, st, v, tcb->xpack_flags);
474 sb_addf(sb, "</n:%*pM>", LSTR_FMT_ARG(cbe->name_res));
475 } else {
476 sb_addf(sb, "<n:%*pM />", LSTR_FMT_ARG(cbe->name_res));
477 }
478 } else {
479 ichttp_cb_t *cbe = iq->cbe;
480
481 xmlpp_opentag(&pp, "s:Fault");
482 xmlpp_opentag(&pp, "faultcode");
483 xmlpp_puts(&pp, "s:Server");
484 xmlpp_opensib(&pp, "faultstring");
485 xmlpp_opensib(&pp, "detail");
486
487 /* FIXME handle union of exceptions which are an array of exceptions */
488 if (v) {
489 if (iop_struct_is_class(st)) {
490 const iop_struct_t *real_st = *(const iop_struct_t **)v;
491
492 sb_addf(sb, "<n:%*pM xsi:type=\"n:%*pM\">",
493 LSTR_FMT_ARG(cbe->name_exn),
494 LSTR_FMT_ARG(real_st->fullname));
495 } else {
496 sb_addf(sb, "<n:%*pM>", LSTR_FMT_ARG(cbe->name_exn));
497 }
498 iop_xpack_flags(sb, st, v, tcb->xpack_flags);
499 sb_addf(sb, "</n:%*pM>", LSTR_FMT_ARG(cbe->name_exn));
500 } else {
501 sb_addf(sb, "<n:%*pM />", LSTR_FMT_ARG(cbe->name_exn));
502 }
503 }
504 pp.can_do_attr = false;
505 xmlpp_close(&pp);
506 iq->iop_answered = true;
507}

509void
510__ichttp_reply(uint64_t slot, int cmd, const iop_struct_t *st, const void *v)
511{
512 ichttp_query_t *iq = ichttp_slot_to_query(slot);
513 httpd_query_t *q = obj_vcast(httpd_query, iq);
514 httpd_trigger__ic_t *tcb;
515 outbuf_t *ob;
516 sb_t *out;
517 int oldlen, gzenc, is_gzip;
518 http_code_t code;
519 size_t oblen;
520
521 ic_query_do_post_hook(NULL, cmd, slot, st, v);
522 gzenc = httpd_qinfo_accept_enc_get(q->qinfo);
523 tcb = container_of(iq->trig_cb, httpd_trigger__ic_t, cb);
524
525 switch (cmd) {
526 case IC_MSG_OK:
527 code = HTTP_CODE_OK;
528 break;
529
530 default:
531 assert (false);
532 /* FALLTHROUGH */
533
534 case IC_MSG_EXN:
535 /* Use INTERNAL_SERVER_ERROR for exceptions:
536 * - in SOAP, this is the error code that must always be used.
537 * - in REST, use it to allow clients to distinguish exceptions
538 * (other error cases must not return INTERNAL_SERVER_ERROR).
539 */
540 code = HTTP_CODE_INTERNAL_SERVER_ERROR;
541 if (tcb->cb.on_query_exn) {
542 tcb->cb.on_query_exn(q, st, v, &code);
543 }
544 break;
545 }
546
547 ob = httpd_reply_hdrs_start(q, code, true);
548
549
550 if (iq->json) {
551 ob_adds(ob, "Content-Type: application/json; charset=utf-8\r\n");
552 } else {
553 ob_adds(ob, "Content-Type: text/xml; charset=utf-8\r\n");
554 }
555 if (gzenc & HTTPD_ACCEPT_ENC_GZIP) {
556 ob_adds(ob, "Content-Encoding: gzip\r\n");
557 is_gzip = true;
558 } else
559 if (gzenc & HTTPD_ACCEPT_ENC_DEFLATE) {
560 ob_adds(ob, "Content-Encoding: deflate\r\n");
561 is_gzip = false;
562 } else {
563 /* Ignore compress we don't support it */
564 gzenc = 0;
565 }
566 httpd_reply_hdrs_done(q, -1, false);
567 oblen = ob->length;
568
569 out = outbuf_sb_start(ob, &oldlen);
570
571 if (gzenc) {
572 t_scope;
573 sb_t buf;
574
575 t_sb_init(&buf, BUFSIZ);
576 if (iq->json) {
577 iop_jpack(st, v, iop_sb_write, &buf, tcb->jpack_flags);
578 iq->iop_answered = true;
579 } else {
580 ichttp_serialize_soap(&buf, iq, cmd, st, v);
581 }
582 sb_add_compressed(out, buf.data, buf.len, Z_BEST_COMPRESSION, is_gzip);
583 } else
584 if (iq->json) {
585 iop_jpack(st, v, iop_sb_write, out, tcb->jpack_flags);
586 iq->iop_answered = true;
587 } else {
588 ichttp_serialize_soap(out, iq, cmd, st, v);
589 }
590 outbuf_sb_end(ob, oldlen);
591
592 oblen = ob->length - oblen;
593 if (tcb->on_reply)
594 (*tcb->on_reply)(tcb, iq, oblen, code);
595 httpd_reply_done(q);
596}

598void __ichttp_reply_soap_err(uint64_t slot, bool serverfault, const lstr_t *err)
599{
600 ichttp_query_t *iq = ichttp_slot_to_query(slot);
601 httpd_query_t *q = obj_vcast(httpd_query, iq);
602 httpd_trigger__ic_t *tcb;
603 outbuf_t *ob;
604 sb_t *out;
605 int oldlen;
606 size_t oblen;
607 xmlpp_t pp;
608
609 assert (!iq->json);
610
611 /* SOAP specifies that failing queries must return error code
612 * INTERNAL_SERVER_ERROR. */
613 __ichttp_err_ctx_set(*err);
614 ob = httpd_reply_hdrs_start(q, HTTP_CODE_INTERNAL_SERVER_ERROR, true);
615 ob_adds(ob, "Content-Type: text/xml; charset=utf-8\r\n");
616 httpd_reply_hdrs_done(q, -1, false);
617 oblen = ob->length;
618
619 out = outbuf_sb_start(ob, &oldlen);
620 tcb = container_of(iq->trig_cb, ichttp_trigger_cb_t, cb);
621
622 xmlpp_open_banner(&pp, out);
623 pp.nospace = true;
624 xmlpp_opentag(&pp, "s:Envelope");
625 xmlpp_putattr(&pp, "xmlns:s",
626 "http://schemas.xmlsoap.org/soap/envelope/");
627
628 xmlpp_opentag(&pp, "s:Body");
629 xmlpp_opentag(&pp, "s:Fault");
630 xmlpp_opentag(&pp, "s:faultcode");
631 if (serverfault) {
632 xmlpp_puts(&pp,"s:Server");
633 } else {
634 xmlpp_puts(&pp,"s:Client");
635 }

673void __ichttp_reply_err(uint64_t slot, int err, const lstr_t *err_str)
674{
675 ichttp_query_t *iq = ichttp_slot_to_query(slot);
676
677 ic_query_do_post_hook(NULL, err, slot, NULL, NULL);
678
679 switch (err) {
680 case IC_MSG_OK:
681 case IC_MSG_EXN:
682 e_panic("should not happen");
683 case IC_MSG_RETRY:
684 case IC_MSG_ABORT:
685 case IC_MSG_PROXY_ERROR:
686 __ichttp_reject(slot, iq, HTTP_CODE_BAD_REQUEST,
687 "query temporary refused");
688 break;
689 case IC_MSG_INVALID:
690 case IC_MSG_SERVER_ERROR:
691 if (err_str && err_str->len) {
692 __ichttp_reject(slot, iq, HTTP_CODE_BAD_REQUEST,
693 "%*pM", LSTR_FMT_ARG(*err_str));
694 } else {
695 __ichttp_reject(slot, iq, HTTP_CODE_BAD_REQUEST,
696 "query refused by server");
697 }
698 break;
699 case IC_MSG_UNIMPLEMENTED:
700 __ichttp_reject(slot, iq, HTTP_CODE_NOT_FOUND,
701 "query not implemented by server");
702 break;
703 }
704}

706void __ichttp_proxify(uint64_t slot, int cmd, const void *data, int dlen)
707{
708 ichttp_query_t *iq = ichttp_slot_to_query(slot);
709 const iop_rpc_t *rpc = iq->cbe->fun;
710 const iop_struct_t *st;
711 pstream_t ps;
712 void *v;
713
714 iq->iop_res_size = IC_MSG_HDR_LEN + dlen;
715 switch (cmd) {
716 case IC_MSG_OK:
717 st = rpc->result;
718 break;
719 case IC_MSG_EXN:
720 st = rpc->exn;
721 break;
722 default:
723 __ichttp_reply_err(slot, cmd, &LSTR_INIT_V(data, dlen));
724 return;
725 }
726
727 {
728 t_scope;
729
730 v = t_new_raw(char, st->size);
731 ps = ps_init(data, dlen);
732 if (unlikely(iop_bunpack(t_pool(), st, v, ps, false) < 0)) {
733 lstr_t err_str = iop_get_err_lstr();
734#ifndef NDEBUG
735 if (!err_str.s)
736 e_trace(0, "%s: answer with invalid encoding", rpc->name.s);
737#endif
738 __ichttp_reply_err(slot, IC_MSG_INVALID, &err_str);
739 } else {
740 __ichttp_reply(slot, cmd, st, v);
741 }
742 }
743}

749 const iop_rpc_t *rpc = iq->cbe->fun;
750 const iop_struct_t *st;
751 const void *v = (cmd == IC_MSG_OK) ? res : exn;
752 sb_t *buf = &pxy_ic->rbuf;
753 int dlen = get_unaligned_cpu32(buf->data + IC_MSG_DLEN_OFFSET);
754
755 iq->iop_res_size = IC_MSG_HDR_LEN;
756 switch (cmd) {
757 case IC_MSG_OK:

764 break;
765 default:
766 __ichttp_reply_err(slot, cmd, exn);
767 return;
768 }
769
770 __ichttp_reply(slot, cmd, st, v);
771}
