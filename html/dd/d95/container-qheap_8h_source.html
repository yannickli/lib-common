24/* Min/Max Heap Container
25 * ----------
26 *
27 * This container can be used as a max heap or as a min heap.
28 * Its content is stored into a qvector.
29 *
30 * The nodes are defined by the user. A macro or function that compares
31 * two nodes.
32 * ----------
33 *
34 * This code is a re-write of lib-common event loop code min heap.
35 * ----------
36 *
37 * Scalar heaps are defined at the end of this file.
38 * ----------
39 *
40 * Example:
41 * -----
42 *
43 * typedef struct test_node_t {
44 * int heap_pos;
45 * int val;
46 * } test_node_t;
47 *
48 * #define CMP(a, op, b) ((a)->val op (b)->val)
49 * #define SET_POS(node, pos) ((node)->heap_pos = pos)
50 *
51 * qhp_min_t(min_heap, test_node_t *, CMP, SET_POS);
52 * qhp_max_t(max_heap, test_node_t *, CMP, SET_POS);
53 *
54 * ----------
55 * More code examples into: lib-common/zchk-container.blk
56 */
57

71#define QHP_FANOUT 4
72#define QHP_CHILD(pos, i) (QHP_FANOUT * (pos) + 1 + (i))
73#define QHP_PARENT(pos) (((pos) - 1) / QHP_FANOUT)

75#define QHP_CMP(is_min_heap, cmp, n1, op, n2) \
76 (is_min_heap ? cmp(n1, op, n2) : cmp(n2, op, n1))

80#define qhp_funcs_t(n, type_t, is_min_heap, cmp, set_pos) \
81 \
82 __unused__ \
83 static ALWAYS_INLINE int \
84 __qhp_##n##_set(qhp_t(n) *heap, int pos, type_t node) \
85 { \
86 set_pos(node, pos); \
87 heap->tab[pos] = node; \
88 \
89 return pos; \
90 } \
91 \
92 __unused__ \
93 static inline int __qhp_##n##_down(qhp_t(n) *heap, int _pos) \
94 { \
95 type_t node = heap->tab[_pos]; \
96 type_t *E = heap->tab + heap->len; \
97 int pos = _pos, child_pos; \
98 \
99 while ((child_pos = QHP_CHILD(pos, 0)) + QHP_FANOUT <= heap->len) { \
100 type_t *c = heap->tab + child_pos; \
101 int n = 0; /* pos of the minimum in c[0 .. QHP_FANOUT] */ \
102 \
103 if (QHP_CMP(is_min_heap, cmp, c[1], <, c[n])) n = 1; \
104 if (QHP_CMP(is_min_heap, cmp, c[2], <, c[n])) n = 2; \
105 if (QHP_CMP(is_min_heap, cmp, c[3], <, c[n])) n = 3; \
106 STATIC_ASSERT(QHP_FANOUT == 4); \
107 \
108 if (QHP_CMP(is_min_heap, cmp, c[n], >=, node)) \
109 break; \
110 \
111 __qhp_##n##_set(heap, pos, c[n]); \
112 pos = child_pos + n; \
113 } \
114 if (child_pos < heap->len) { \
115 type_t *c = heap->tab + child_pos; \
116 int n = 0; /* pos of the minimum in c[0 .. QHP_FANOUT] */ \
117 \
118 if (c + 1 < E && QHP_CMP(is_min_heap, cmp, c[1], <, c[n])) n = 1; \
119 if (c + 2 < E && QHP_CMP(is_min_heap, cmp, c[2], <, c[n])) n = 2; \
120 if (c + 3 < E && QHP_CMP(is_min_heap, cmp, c[3], <, c[n])) n = 3; \
121 STATIC_ASSERT(QHP_FANOUT == 4); \
122 \
123 if (QHP_CMP(is_min_heap, cmp, c[n], <, node)) { \
124 __qhp_##n##_set(heap, pos, c[n]); \
125 pos = child_pos + n; \
126 } \
127 } \
128 \
129 return _pos != pos ? __qhp_##n##_set(heap, pos, node) : _pos; \
130 } \
131 \
132 __unused__ \
133 static inline int __qhp_##n##_up(qhp_t(n) *heap, int _pos) \
134 { \
135 type_t node = heap->tab[_pos]; \
136 int pos = _pos; \
137 \
138 while (pos > 0) { \
139 int parent_pos = QHP_PARENT(pos); \
140 type_t *parent = &heap->tab[parent_pos]; \
141 \
142 if (QHP_CMP(is_min_heap, cmp, node, >=, *parent)) \
143 break; \
144 \
145 __qhp_##n##_set(heap, pos, *parent); \
146 pos = parent_pos; \
147 } \
148 \
149 return _pos != pos ? __qhp_##n##_set(heap, pos, node) : _pos; \
150 } \
151 \
152 __unused__ \
153 static inline int qhp_##n##_insert(qhp_t(n) *heap, type_t node) \
154 { \
155 set_pos(node, heap->len); \
156 qv_append(heap, node); \
157 \
158 return __qhp_up(n, heap, heap->len - 1); \
159 } \
160 \
161 __unused__ \
162 static inline int qhp_##n##_fixup(qhp_t(n) *heap, int pos) \
163 { \
164 type_t *node = &heap->tab[pos]; \
165 type_t *parent = &heap->tab[QHP_PARENT(pos)]; \
166 \
167 if (pos > 0 && QHP_CMP(is_min_heap, cmp, *parent, >=, *node)) { \
168 return __qhp_up(n, heap, pos); \
169 } else { \
170 return __qhp_down(n, heap, pos); \
171 } \
172 } \
173 \
174 __unused__ \
175 static inline type_t qhp_##n##_remove(qhp_t(n) *heap, int pos) \
176 { \
177 type_t node; \
178 \
179 if (unlikely(pos < 0)) { \
180 p_clear(&node, 1); \
181 return node; \
182 } \
183 \
184 node = heap->tab[pos]; \
185 if (pos == heap->len - 1) { \
186 qv_shrink(heap, 1); \
187 } else { \
188 type_t last = *tab_last(heap); \
189 \
190 qv_shrink(heap, 1); \
191 __qhp_##n##_set(heap, pos, last); \
192 qhp_fixup(n, heap, pos); \
193 } \
194 \
195 set_pos(node, -1); \
196 return node; \
197 }

201/* \macro qhp_t
202 * \brief Declares a heap and its functions
203 *
204 * \param name Name of the heap
205 *
206 * \param type_t Node type
207 *
208 * \param is_min_heap Boolean. Defines whether the heap is a min heap or a
209 * max heap.
210 *
211 * \param cmp Comparison macro. Take two nodes and a logical
212 * operator. Examples: QHP_SCALAR_CMP and QHP_LSTR_CMP.
213 *
214 * \param set_pos Position setter function or macro. This setter is
215 * optional: if user does not need to maintain the
216 * position of his nodes into the heap, he just have to
217 * pass QHP_IGNORE for this parameter. This setter is
218 * useful for users who want to remove or fix up an
219 * object which is not the first of the heap.
220 */
221#define qhp_full_t(name, type_t, is_min_heap, cmp, set_pos) \
222 qhp_type_t(name, type_t); \
223 qhp_funcs_t(name, type_t, is_min_heap, cmp, set_pos);
224
225#define qhp_min_t(name, type_t, cmp, set_pos) \
226 qhp_full_t(name, type_t, true, cmp, set_pos)
227
228#define qhp_max_t(name, type_t, cmp, set_pos) \
229 qhp_full_t(name, type_t, false, cmp, set_pos)
