21#define rb_parent(n) ((rb_node_t *)((n)->__parent & ~1))
22#define rb_color(n) (n->__parent & 1)
23#define rb_is_red(n) ({ rb_node_t *__n = (n); __n && !rb_color(__n); })
24#define rb_is_black(n) ({ rb_node_t *__n = (n); !__n || rb_color(__n); })
25#define rb_set_red(n) ((n)->__parent &= ~1)
26#define rb_set_black(n) ((n)->__parent |= 1)

94static void rb_rotate_left(rb_node_t **root, rb_node_t *x)
95{
96 rb_node_t *p = rb_parent(x);
97 rb_node_t *y = x->right;
98

100 rb_set_parent(y->left, x);

120static ALWAYS_INLINE void rb_add_fix_color(rb_node_t **root, rb_node_t *z)
121{
122 rb_node_t *p_z, *y;
123
124 while (rb_is_red(p_z = rb_parent(z))) {
125 rb_node_t *gp_z = rb_parent(p_z);
126
127 if (p_z == gp_z->left) {
128 if (rb_is_red(y = gp_z->right)) {
129 rb_set_black(p_z);
130 rb_set_black(y);
131 rb_set_red(gp_z);
132 z = gp_z;
133 continue;
134 }
135
136 if (p_z->right == z) {
137 rb_rotate_left(root, p_z);
138 SWAP(rb_node_t *, z, p_z);
139 }
140
141 rb_set_black(p_z);
142 rb_set_red(gp_z);
143 rb_rotate_right(root, gp_z);
144 } else {
145 if (rb_is_red(y = gp_z->left)) {
146 rb_set_black(y);
147 rb_set_black(p_z);
148 rb_set_red(gp_z);
149 z = gp_z;
150 continue;
151 }
152
153 if (p_z->left == z) {
154 rb_rotate_right(root, p_z);
155 SWAP(rb_node_t *, z, p_z);
156 }
157
158 rb_set_black(p_z);
159 rb_set_red(gp_z);
160 rb_rotate_left(root, gp_z);
161 }
162 }
163
164 rb_set_black(*root);
165}

177static ALWAYS_INLINE void
178rb_del_fix_color(rb_node_t **root, rb_node_t *p, rb_node_t *z)
179{
180 rb_node_t *w;
181
182 while (rb_is_black(z) && z != *root) {
183 if (p->left == z) {
184 if (rb_is_red(w = p->right)) {
185 rb_set_black(w);
186 rb_set_red(p);
187 rb_rotate_left(root, p);
188 w = p->right;
189 }
190 assert (w);
191 if (rb_is_black(w->left) && rb_is_black(w->right)) {
192 rb_set_red(w);
193 z = p;
194 p = rb_parent(z);
195 } else {
196 if (rb_is_black(w->right)) {
197 rb_set_black2(w->left);
198 rb_set_red(w);
199 rb_rotate_right(root, w);
200 w = p->right;
201 }
202 rb_copy_color(w, p);
203 rb_set_black(p);
204 rb_set_black2(w->right);
205 rb_rotate_left(root, p);
206 z = *root;
207 break;
208 }
209 } else {
210 if (rb_is_red(w = p->left)) {
211 rb_set_black(w);
212 rb_set_red(p);
213 rb_rotate_right(root, p);
214 w = p->left;
215 }
216 assert (w);
217 if (rb_is_black(w->left) && rb_is_black(w->right)) {
218 rb_set_red(w);

240void rb_del_node(rb_node_t **root, rb_node_t *z)
241{
242 struct rb_node_t *child, *p;
243 int color;
244
245 if (z->left && z->right) {
246 rb_node_t *old = z;
247
248 z = __rb_next(z);
249 child = z->right;
250 p = rb_parent(z);
251 color = rb_color(z);
252
253 if (child)
254 rb_set_parent(child, p);
255 if (p == old) {
256 p->right = child;
257 p = z;
258 } else {
259 p->left = child;
260 }
261 *z = *old;
262
263 rb_reparent(root, rb_parent(old), old, z);
264 rb_set_parent(old->left, z);
265 if (old->right)
266 rb_set_parent(old->right, z);
267 } else {
268 child = z->right ?: z->left;
269 p = rb_parent(z);
270 color = rb_color(z);
271 if (child)
272 rb_set_parent(child, p);
273 rb_reparent(root, p, z, child);
274 }
275 check_rb(root, false);
276
277 if (color) /* it's black */
278 rb_del_fix_color(root, p, child);
279 check_rb(root, true);
280}
