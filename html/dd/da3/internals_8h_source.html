226/*
227 * .ranges helps finding tags into .fields.
228 * ----------------------------------------
229 *
230 * it's a suite of [i_1, tag_1, ... i_k, tag_k, ... t_n, i_{n+1}]
231 *
232 * it says that at .fields[i_k] starts a contiguous range of tags from
233 * value tag_k to tag_k + i_{k+1} - i_k.
234 * Of course n == .ranges_len and i_{n+1} == .fields_len.
235 *
236 *
237 * Example:
238 * -------
239 *
240 * If you have the tags [1, 2, 3, 7, 9, 10, 11], the ranges will read:
241 * --v---- v ----v----
242 * [0, 1, 3, 7, 4, 9, 7]
243 *
244 * Looking for a given tag needs a binary search the odd places of the
245 * ranges array. Then, if you're tag T on the kth range, the index of your
246 * description in .fields is: (i_k + T - t_k)
247 *
248 */
249struct iop_enum_t {
250 const lstr_t name;
251 const lstr_t fullname;
252 const lstr_t * nonnull names;
253 const int * nonnull values;
254 const int * nonnull ranges;
255 uint16_t enum_len;
256 uint16_t flags; /**< bitfield of iop_enum_flags_t */
257 int ranges_len;
258 /* XXX do not dereference the following 2 members without checking
259 * TST_BIT(this->flags, IOP_ENUM_EXTENDED) first */
260 const iop_enum_attrs_t * nullable en_attrs;
261 const iop_enum_value_attrs_t * nullable values_attrs;
262 /* XXX do not dereference the following member without checking
263 * TST_BIT(this->flags, IOP_ENUM_ALIASES) first */
264 const iop_enum_aliases_t * nullable aliases;
265};
