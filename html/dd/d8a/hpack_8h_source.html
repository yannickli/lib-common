139/* flag values to control the header encoding function */
140enum {
141 HPACK_FLG_NOZIP_KEY = 1u << 0, /* Encode key as-is (raw) */
142 HPACK_FLG_ZIP_KEY = 1u << 1, /* Encode key as Huffman string */
143 HPACK_FLG_NOZIP_VAL = 1u << 2, /* Encode value as-is */
144 HPACK_FLG_ZIP_VAL = 1u << 3, /* Encode value as Huffman string */
145 HPACK_FLG_SKIP_STBL = 1u << 4, /* Don't look up in the static table */
146 HPACK_FLG_SKIP_DTBL = 1u << 5, /* Don't loop up in the dynamic table */
147 HPACK_FLG_LWR_KEY = 1u << 6, /* Normalize key (lowercase) */
148 HPACK_FLG_SKIP_VAL = 1u << 7, /* Don't try to match the value */
149 HPACK_FLG_NOADD_DTBL = 1u << 8, /* Don't add hdr to the dynamic table */
150 HPACK_FLG_NVRADD_DTBL = 1u << 9, /* Don't add hdr to the dynamic table
151 * and instruct proxies to reuse the
152 * same encoding for this hdr if
153 * they are to re-encode headers
154 * (used for sensitive headers) */
155 HPACK_FLG_ADD_DTBL = 1u << 10, /* Force adding hdr to the dynamic
156 * table */
157
158 HPACK_FLG_NOZIP_STR = HPACK_FLG_NOZIP_KEY | HPACK_FLG_NOZIP_VAL,
159 HPACK_FLG_ZIP_STR = HPACK_FLG_ZIP_KEY | HPACK_FLG_ZIP_VAL,
160 HPACK_FLG_SKIP_TBLS = HPACK_FLG_SKIP_STBL | HPACK_FLG_SKIP_DTBL,
161};

206/** Read a dtbl size update, if any, from \p in
207 *
208 * \return 1 if an update is read, 0 if there is not, or -1 on error
209 */
210static inline int
211hpack_decoder_read_dts_update(hpack_dec_dtbl_t *dtbl, pstream_t *in)
212{
213 if (likely(ps_done(in) || (0xE0u & in->b[0]) != 0x20u)) {
214 return 0;
215 }
216 return hpack_decoder_read_dts_update_(dtbl,in);
217}
