55static void iop_xwsdl_put_attr_value(wsdlpp_t *wpp, iop_type_t type,
56 const iop_field_attr_t *attr)
57{
58 switch (type) {
59 case IOP_T_I8: case IOP_T_I16: case IOP_T_I32: case IOP_T_I64:
60 xmlpp_putattrfmt(&wpp->pp, "value", "%jd", attr->args[0].v.i64);
61 break;
62
63 case IOP_T_U8: case IOP_T_U16: case IOP_T_U32: case IOP_T_U64:
64 xmlpp_putattrfmt(&wpp->pp, "value", "%ju",
65 (uint64_t)attr->args[0].v.i64);
66 break;
67
68 case IOP_T_DOUBLE:
69 xmlpp_putattrfmt(&wpp->pp, "value", "%.17e", attr->args[0].v.d);
70 break;
71
72 case IOP_T_BOOL:
73 case IOP_T_ENUM:
74 case IOP_T_STRING:
75 case IOP_T_DATA:
76 case IOP_T_XML:
77 case IOP_T_STRUCT:
78 case IOP_T_UNION:
79 case IOP_T_VOID:
80 e_panic("should not happen");
81 }
82}

84static void put_type_constraints(wsdlpp_t *wpp, iop_type_t type,
85 const iop_field_attrs_t *attrs)
86{
87 for (int i = 0; i < attrs->attrs_len; i++) {
88 const iop_field_attr_t *attr = &attrs->attrs[i];
89
90 switch (attr->type) {
91 case IOP_FIELD_MIN:
92 xmlpp_opentag(&wpp->pp, "minInclusive");
93 iop_xwsdl_put_attr_value(wpp, type, attr);
94 break;
95
96 case IOP_FIELD_MAX:
97 xmlpp_opentag(&wpp->pp, "maxInclusive");
98 iop_xwsdl_put_attr_value(wpp, type, attr);
99 break;
100
101 case IOP_FIELD_NON_EMPTY:
102 xmlpp_opentag(&wpp->pp, "minLength");
103 xmlpp_putattr(&wpp->pp, "value", "1");
104 break;
105
106 case IOP_FIELD_MIN_LENGTH:
107 xmlpp_opentag(&wpp->pp, "minLength");
108 xmlpp_putattrfmt(&wpp->pp, "value", "%jd",
109 attr->args[0].v.i64);
110 break;
111
112 case IOP_FIELD_MAX_LENGTH:
113 xmlpp_opentag(&wpp->pp, "maxLength");
114 xmlpp_putattrfmt(&wpp->pp, "value", "%jd",
115 attr->args[0].v.i64);
116 break;
117
118 case IOP_FIELD_PATTERN:
119 xmlpp_opentag(&wpp->pp, "pattern");
120 xmlpp_putattrfmt(&wpp->pp, "value", "(%*pM)*",
121 LSTR_FMT_ARG(attr->args[0].v.s));
122 break;
123
124 default:
125 continue;
126 }
127 xmlpp_closetag(&wpp->pp);
128
129 /* TODO non-zero isn't yet supported in WSDL */
130 }

133static void put_int_type(wsdlpp_t *wpp, iop_type_t type, const char *name,
134 const iop_field_attrs_t *attrs)
135{
136 static char const * const xs_types[] = {
137 [IOP_T_I8] = "byte",
138 [IOP_T_U8] = "unsignedByte",
139 [IOP_T_I16] = "short",
140 [IOP_T_U16] = "unsignedShort",
141 [IOP_T_I32] = "int",
142 [IOP_T_ENUM] = "int",
143 [IOP_T_U32] = "unsignedInt",
144 [IOP_T_I64] = "long",
145 [IOP_T_U64] = "unsignedLong",
146 };
147
148#define mk_pattern(sz) "0x0*[0-9a-fA-F]{1,"#sz"}"
149 static char const * const hex_patterns[] = {
150 [IOP_T_I8] = mk_pattern(2),
151 [IOP_T_U8] = mk_pattern(2),
152 [IOP_T_I16] = mk_pattern(4),
153 [IOP_T_U16] = mk_pattern(4),
154 [IOP_T_I32] = mk_pattern(8),
155 [IOP_T_ENUM] = mk_pattern(8),
156 [IOP_T_U32] = mk_pattern(8),
157 [IOP_T_I64] = mk_pattern(16),
158 [IOP_T_U64] = mk_pattern(16),
159 };
160#undef mk_pattern
161
162 xmlpp_t *pp = &wpp->pp;
163
164 xmlpp_opentag(pp, "simpleType");
165 if (name) {
166 xmlpp_putattr(pp, "name", name);
167 }
168 xmlpp_opentag(pp, "union");
169 xmlpp_opentag(pp, "simpleType");
170 xmlpp_opentag(pp, "restriction");
171 xmlpp_putattr(pp, "base", xs_types[type]);
172
173 if (attrs) {
174 /* If given, constraints are applied only on the “integer” part of the
175 * type, not on the hexadecimal pattern. */
176 put_type_constraints(wpp, type, attrs);
177 }
178
179 xmlpp_closentag(pp, 2);
180 xmlpp_opentag(pp, "simpleType");
181 xmlpp_opentag(pp, "restriction");
182 xmlpp_putattr(pp, "base", "string");
183 xmlpp_opentag(pp, "pattern");
184 xmlpp_putattr(pp, "value", hex_patterns[type]);
185 /* TODO support constraints on hexadecimal pattern */
186
187 xmlpp_closentag(pp, 5);
188}

190static void iop_xwsdl_put_enum(wsdlpp_t *wpp, const iop_enum_t *e)
191{
192 xmlpp_opentag(&wpp->pp, "simpleType");
193 xmlpp_putattr(&wpp->pp, "name", e->fullname.s);
194 xmlpp_opentag(&wpp->pp, "union");
195
196 xmlpp_opentag(&wpp->pp, "simpleType");
197 xmlpp_opentag(&wpp->pp, "restriction");
198 xmlpp_putattr(&wpp->pp, "base", "string");
199
200 for (int i = 0; i < e->enum_len; i++) {
201 xmlpp_opentag(&wpp->pp, "enumeration");
202 xmlpp_putattr(&wpp->pp, "value", e->names[i].s);
203 xmlpp_closetag(&wpp->pp);
204 }

209 xmlpp_putattr(&wpp->pp, "base", "tns:intersec.int");
210
211 xmlpp_closentag(&wpp->pp, 4);
212}

214static void iop_xwsdl_put_occurs(wsdlpp_t *wpp, const iop_struct_t *st,
215 const iop_field_t *f)
216{
217 const iop_field_attrs_t *attrs = iop_field_get_attrs(st, f);
218 unsigned min = 0, max = 0;
219
220 if (attrs
221 && (attrs->flags & ( (1 << IOP_FIELD_MIN_OCCURS)
222 | (1 << IOP_FIELD_MAX_OCCURS))))
223 {
224 for (int i = 0; i < attrs->attrs_len; i++) {
225 const iop_field_attr_t *attr = &attrs->attrs[i];
226
227 if (attr->type == IOP_FIELD_MIN_OCCURS) {
228 assert (attr->args[0].v.i64 >= 0);
229 min = (unsigned)attr->args[0].v.i64;
230 } else
231 if (attr->type == IOP_FIELD_MAX_OCCURS) {
232 assert (attr->args[0].v.i64 > 0);
233 max = (unsigned)attr->args[0].v.i64;
234 }
235 }
236 }
237
238 xmlpp_putattrfmt(&wpp->pp, "minOccurs", "%u", min);
239 if (max > 0) {
240 xmlpp_putattrfmt(&wpp->pp, "maxOccurs", "%u", max);
241 } else {
242 xmlpp_putattr(&wpp->pp, "maxOccurs", "unbounded");
243 }
244}

246static void iop_xwsdl_put_constraints(wsdlpp_t *wpp, const iop_struct_t *st,
247 const iop_field_t *f)
248{
249 static char const * const types[] = {
250 [IOP_T_I8] = "tns:intersec.byte",
251 [IOP_T_U8] = "tns:intersec.unsignedByte",
252 [IOP_T_I16] = "tns:intersec.short",
253 [IOP_T_U16] = "tns:intersec.unsignedShort",
254 [IOP_T_I32] = "tns:intersec.int",
255 [IOP_T_ENUM] = "tns:intersec.int",
256 [IOP_T_U32] = "tns:intersec.unsignedInt",
257 [IOP_T_I64] = "tns:intersec.long",
258 [IOP_T_U64] = "tns:intersec.unsignedLong",
259 [IOP_T_BOOL] = "boolean",
260 [IOP_T_DOUBLE] = "double",
261 [IOP_T_STRING] = "string",
262 [IOP_T_DATA] = "base64Binary",
263 [IOP_T_VOID] = "anyType",
264 };
265
266 const iop_field_attrs_t *attrs = iop_field_get_attrs(st, f);
267 unsigned restriction_flags = 0;
268
269 SET_BIT(&restriction_flags, IOP_FIELD_MIN);
270 SET_BIT(&restriction_flags, IOP_FIELD_MAX);
271 SET_BIT(&restriction_flags, IOP_FIELD_NON_EMPTY);
272 SET_BIT(&restriction_flags, IOP_FIELD_MIN_LENGTH);
273 SET_BIT(&restriction_flags, IOP_FIELD_MAX_LENGTH);
274 SET_BIT(&restriction_flags, IOP_FIELD_PATTERN);
275
276 if (!attrs || !(attrs->flags & restriction_flags)) {
277 /* No specific restrictions, so we just use the generic types */
278 xmlpp_putattr(&wpp->pp, "type", types[f->type]);
279 return;
280 } else {
281 /* When we have have constraints, we need to dump a custom type */
282 switch (f->type) {
283 case IOP_T_I8: case IOP_T_U8:
284 case IOP_T_I16: case IOP_T_U16:
285 case IOP_T_I32: case IOP_T_U32:
286 case IOP_T_ENUM:
287 case IOP_T_I64: case IOP_T_U64:
288 /* Integer values are handled separately */
289 put_int_type(wpp, f->type, NULL, attrs);
290 return;
291
292 case IOP_T_DOUBLE:
293 case IOP_T_STRING:
294 case IOP_T_DATA:
295 xmlpp_opentag(&wpp->pp, "simpleType");
296 xmlpp_opentag(&wpp->pp, "restriction");
297 xmlpp_putattr(&wpp->pp, "base", types[f->type]);
298
299 put_type_constraints(wpp, f->type, attrs);
300
301 xmlpp_closentag(&wpp->pp, 2);
302 break;
303
304 default:
305 e_panic("unsupported type with constraints: %d", f->type);
306 }
307 }
308
309#undef ATTR_TOUCH_TYPES
310}

312static void iop_xwsdl_put_type(wsdlpp_t *wpp, const iop_struct_t *st)
313{
314 bool is_class = iop_struct_is_class(st);
315 bool is_extended = is_class && st->class_attrs->parent;
316
317 xmlpp_opentag(&wpp->pp, "complexType");
318 xmlpp_putattr(&wpp->pp, "name", st->fullname.s);
319 if (st == &iop__void__s) {
320 xmlpp_closetag(&wpp->pp);
321 return;
322 }
323 if (st->is_union) {
324 xmlpp_opentag(&wpp->pp, "choice");
325 } else {
326 if (is_class && st->class_attrs->is_abstract) {
327 xmlpp_putattr(&wpp->pp, "abstract", "true");
328 }
329 if (is_extended) {
330 xmlpp_opentag(&wpp->pp, "complexContent");
331 xmlpp_opentag(&wpp->pp, "extension");
332 xmlpp_putattrfmt(&wpp->pp, "base", "tns:%s",
333 st->class_attrs->parent->fullname.s);
334 }
335 xmlpp_opentag(&wpp->pp, "sequence");
336 }
337 for (int i = 0; i < st->fields_len; i++) {
338 const iop_field_t *f = &st->fields[i];
339 const iop_field_attrs_t *attrs;
340
341 attrs = iop_field_get_attrs(st, f);
342 if (attrs && TST_BIT(&attrs->flags, IOP_FIELD_PRIVATE))
343 continue;
344
345 switch (f->type) {
346 case IOP_T_XML:
347 /* xsd:any with the attribute processContents="skip" means "any
348 * valid xml element" */
349 xmlpp_opentag(&wpp->pp, "any");
350 xmlpp_putattr(&wpp->pp, "processContents", "skip");
351 break;
352 default:
353 xmlpp_opentag(&wpp->pp, "element");
354 }
355
356 xmlpp_putattr(&wpp->pp, "name", f->name.s);
357 switch (f->repeat) {
358 case IOP_R_DEFVAL:
359 case IOP_R_OPTIONAL:
360 xmlpp_putattr(&wpp->pp, "minOccurs", "0");
361 break;
362 case IOP_R_REPEATED:
363 iop_xwsdl_put_occurs(wpp, st, f);
364 break;
365 case IOP_R_REQUIRED:
366 if (f->type == IOP_T_VOID && !st->is_union) {
367 xmlpp_putattr(&wpp->pp, "minOccurs", "0");
368 }
369 }
370
371 if (f->type == IOP_T_VOID) {
372 xmlpp_putattr(&wpp->pp, "nillable", "true");
373 }
374
375 if ((1 << f->type) & IOP_STRUCTS_OK) {
376 const iop_struct_t *desc = f->u1.st_desc;
377 xmlpp_putattrfmt(&wpp->pp, "type", "tns:%s", desc->fullname.s);
378 } else
379 if (wpp->wenums && f->type == IOP_T_ENUM) {
380 const iop_enum_t *desc = f->u1.en_desc;
381 xmlpp_putattrfmt(&wpp->pp, "type", "tns:%s", desc->fullname.s);
382 } else
383 if (f->type != IOP_T_XML) {
384 iop_xwsdl_put_constraints(wpp, st, f);
385
386 }
387 xmlpp_closetag(&wpp->pp);
388 if (f->repeat == IOP_R_DEFVAL) {
389 switch (f->type) {
390 case IOP_T_BOOL:
391 wpp_add_comment(wpp, "\n", "default: %s=%s",
392 f->name.s, f->u1.defval_u64 ? "true" : "false");
393 break;
394 case IOP_T_I8: case IOP_T_I16: case IOP_T_I32: case IOP_T_I64:
395 wpp_add_comment(wpp, "\n", "default: %s=%jd",
396 f->name.s, f->u1.defval_u64);
397 break;
398 case IOP_T_ENUM:
399 wpp_add_comment(wpp, "\n", "default: %s=%d",
400 f->name.s, f->u0.defval_enum);
401 break;
402 case IOP_T_U8: case IOP_T_U16: case IOP_T_U32: case IOP_T_U64:
403 wpp_add_comment(wpp, "\n", "default: %s=%ju",
404 f->name.s, f->u1.defval_u64);
405 break;
406 case IOP_T_DOUBLE:
407 wpp_add_comment(wpp, "\n", "default: %s=%.17e",
408 f->name.s, f->u1.defval_d);
409 break;
410 case IOP_T_STRING:
411 wpp_add_comment(wpp, "\n", "default: %s=%s",
412 f->name.s, (const char *)f->u1.defval_data);
413 break;
414 case IOP_T_DATA:
415 case IOP_T_XML:
416 case IOP_T_STRUCT:
417 case IOP_T_UNION:
418 case IOP_T_VOID:
419 break;
420 }
421 }
422 }
423 xmlpp_closentag(&wpp->pp, is_extended ? 4 : 2);
424}

426static void
427iop_xwsdl_scan_types(wsdlpp_t *wpp, const iop_struct_t *st,
428 qh_t(iop_type) *type_idx,
429 qv_t(iop_type) *types, qv_t(iop_enum) *enums)
430{
431 if (qh_add(iop_type, type_idx, st))
432 return;
433
434 for (int i = 0; i < st->fields_len; i++) {
435 const iop_field_t *field = &st->fields[i];
436 const iop_field_attrs_t *attrs;
437
438 attrs = iop_field_get_attrs(st, field);
439 if (attrs && TST_BIT(&attrs->flags, IOP_FIELD_PRIVATE))
440 continue;
441
442 if ((1 << field->type) & IOP_STRUCTS_OK) {
443 iop_xwsdl_scan_types(wpp, field->u1.st_desc, type_idx, types, enums);
444 } else
445 if (field->type == IOP_T_ENUM && wpp->wenums) {
446 if (qh_add(iop_type, type_idx, field->u1.st_desc))
447 continue;
448
449 qv_append(enums, field->u1.en_desc);
450 }
451 }
452
453 if (iop_struct_is_class(st) && st->class_attrs->parent) {
454 iop_xwsdl_scan_types(wpp, st->class_attrs->parent, type_idx,
455 types, enums);
456 }
457
458 qv_append(types, st);
459}

461typedef void (^iop_rpc_loop_body_b)(const iop_iface_alias_t *, const iop_rpc_t *);
462static void iop_xwsdl_rpc_loop(wsdlpp_t *wpp, const iop_mod_t *mod,
463 iop_rpc_loop_body_b loop_body)
464{
465 for (int i = 0; i < mod->ifaces_len; i++) {
466 const iop_iface_alias_t *alias = &mod->ifaces[i];
467
468 for (int j = 0; j < alias->iface->funs_len; j++) {
469 const iop_rpc_t *fun = &alias->iface->funs[j];
470 const unsigned fun_flags = fun->flags;
471 const iop_rpc_attrs_t *attrs;
472
473 /* Check implementation table */
474 WSDL_CHECK_SKIP(wpp, alias, fun);
475 if (TST_BIT(&fun_flags, IOP_RPC_IS_ALIAS)) {
476 continue;
477 }
478
479 loop_body(alias, fun);
480
481 /* loop on rpc aliases */
482 if (TST_BIT(&fun_flags, IOP_RPC_HAS_ALIAS)) {
483 attrs = iop_rpc_get_attrs(alias->iface, fun);
484 for (int k = 0; k < attrs->attrs_len; k++) {
485 const iop_rpc_attr_t attr = attrs->attrs[k];
486
487 if (attr.type == IOP_RPC_ALIAS) {
488 const iop_rpc_attr_arg_t arg = attr.args[0];
489 const iop_rpc_t *falias = (const iop_rpc_t *)arg.v.p;
490
491 loop_body(alias, falias);
492 }
493 }
494 }
495 }
496 }

500typedef struct iop_xwsdl_put_types_t {

506static void
507iop_xwsdl_put_types(wsdlpp_t *wpp, const iop_mod_t *mod, const char *ns)
508{
509 qh_t(iop_type) h;
510 qv_t(iop_type) a;
511 qv_t(iop_enum) b;
512 __block iop_xwsdl_put_types_t p = {.h = &h, .a = &a, .b = &b};
513 iop_rpc_loop_body_b scan_types, dump_alias;
514
515 qh_init(iop_type, &h);
516 qv_inita(&a, 1024);
517 qv_inita(&b, 1024);
518
519 scan_types = ^(const iop_iface_alias_t *alias, const iop_rpc_t *rpc) {
520 const iop_struct_t *st;
521
522 iop_xwsdl_scan_types(wpp, rpc->args, p.h, p.a, p.b);
523 if (!rpc->async) {
524 iop_xwsdl_scan_types(wpp, rpc->result, p.h, p.a, p.b);
525 }
526
527 if (rpc->exn == &iop__void__s)
528 return;
529 st = rpc->exn;
530 if (st->is_union) {
531 /* XXX An union is understood as a sequence of exceptions */
532 for (int k = 0; k < st->fields_len; k++) {
533 const iop_field_t *f = st->fields + k;
534 assert(f->type == IOP_T_STRUCT);
535 iop_xwsdl_scan_types(wpp, f->u1.st_desc, p.h, p.a, p.b);
536 }
537 } else {
538 iop_xwsdl_scan_types(wpp, st, p.h, p.a, p.b);
539 }
540 };
541 iop_xwsdl_rpc_loop(wpp, mod, scan_types);

543 iop_for_each_registered_classes(^(const iop_struct_t *st) {
544 /* A class should appear in the WSDL if one of its parents appears in
545 * the WSDL. */
546 const iop_struct_t *parent = st;
547 qh_t(iop_type) *types = p.h;
548
549 if (qh_find(iop_type, types, st) >= 0) {
550 return;
551 }
552 while ((parent = parent->class_attrs->parent)) {
553 if (qh_find(iop_type, types, parent) >= 0) {
554 iop_xwsdl_scan_types(wpp, st, p.h, p.a, p.b);
555 return;
556 }
557 }
558 });
559
560 wpp_add_comment(wpp, "\n\n", "WSDL types");
561 xmlpp_opentag(&wpp->pp, "types");
562 xmlpp_opentag(&wpp->pp, "schema");
563 xmlpp_putattr(&wpp->pp, "targetNamespace", ns);
564 xmlpp_putattr(&wpp->pp, "xmlns", "http://www.w3.org/2001/XMLSchema");
565
566 /* Dump the generic Intersec integer types */
567 put_int_type(wpp, IOP_T_I8, "intersec.byte", NULL);
568 put_int_type(wpp, IOP_T_U8, "intersec.unsignedByte", NULL);
569 put_int_type(wpp, IOP_T_I16, "intersec.short", NULL);
570 put_int_type(wpp, IOP_T_U16, "intersec.unsignedShort", NULL);
571 put_int_type(wpp, IOP_T_I32, "intersec.int", NULL);
572 put_int_type(wpp, IOP_T_U32, "intersec.unsignedInt", NULL);
573 put_int_type(wpp, IOP_T_I64, "intersec.long", NULL);
574 put_int_type(wpp, IOP_T_U64, "intersec.unsignedLong", NULL);
575
576 /* Dump the authentification header type */
577 if (wpp->wauth) {
578 xmlpp_opentag(&wpp->pp, "element");
579 xmlpp_putattr(&wpp->pp, "name", "callerIdentity");
580 xmlpp_opentag(&wpp->pp, "complexType");
581 xmlpp_opentag(&wpp->pp, "sequence");
582 xmlpp_opentag(&wpp->pp, "element");
583 xmlpp_putattr(&wpp->pp, "name", "login");
584 xmlpp_putattr(&wpp->pp, "type", "string");
585 xmlpp_closetag(&wpp->pp);
586 xmlpp_opentag(&wpp->pp, "element");
