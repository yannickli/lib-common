24bool sockunion_equal(const sockunion_t *a1, const sockunion_t *a2)
25{
26 if (a1->family != a2->family)
27 return false;
28
29 switch (a1->family) {
30 case AF_INET:
31 return a1->sin.sin_port == a2->sin.sin_port
32 && a1->sin.sin_addr.s_addr == a2->sin.sin_addr.s_addr;
33
34 case AF_INET6:
35 if (a1->sin6.sin6_port != a2->sin6.sin6_port)
36 return false;
37 return !memcmp(a1->sin6.sin6_addr.s6_addr, a2->sin6.sin6_addr.s6_addr,
38 sizeof(a2->sin6.sin6_addr.s6_addr));
39
40 case AF_UNIX:
41 return !strcmp(a1->sunix.sun_path, a2->sunix.sun_path);
42
43 default:
44 e_panic("unknown kind of sockaddr: %d", a1->family);
45 }
46}

48uint32_t sockunion_hash(const sockunion_t *su)
49{
50 uint64_t u64;
51 uint32_t u32;
52
53 switch (su->family) {
54 case AF_INET:
55 u64 = su->sin.sin_family | (su->sin.sin_port << 16)
56 | ((uint64_t)su->sin.sin_addr.s_addr << 32);
57 return u64_hash32(u64);
58
59 case AF_INET6:
60 u32 = su->sin6.sin6_family | (su->sin6.sin6_port << 16);
61 return u32 ^ mem_hash32(su->sin6.sin6_addr.s6_addr,
62 sizeof(su->sin6.sin6_addr.s6_addr));
63
64 case AF_UNIX:
65 return mem_hash32(&su->sunix, sockunion_len(su));
66
67 default:
68 e_panic("unknown kind of sockaddr: %d", su->family);
69 }
70}

72int sockunion_gethost(const sockunion_t *su, char *buf, int size)
73{
74 switch (su->family) {
75 case AF_INET:
76 RETHROW_PN(inet_ntop(AF_INET, &su->sin.sin_addr.s_addr, buf, size));
77 break;
78
79 case AF_INET6:
80 RETHROW_PN(inet_ntop(AF_INET6, &su->sin6.sin6_addr.s6_addr, buf,
81 size));
82 break;
83
84 default:
85 return -1;
86 }
87 return strlen(buf);
88}

105int addr_parse_minport(pstream_t ps, pstream_t *host, in_port_t *port,
106 int minport, int defport)
107{
108 int i;
109
110 PS_WANT(ps_has(&ps, 1));
111 if (ps.b[0] == '[') {
112 __ps_skip(&ps, 1);
113 PS_CHECK(ps_get_ps_chr(&ps, ']', host));
114 __ps_skip(&ps, 1);
115 } else {
116 if (ps_get_ps_chr(&ps, ':', host) < 0) {
117 *host = ps;
118 ps.b = ps.b_end;
119 goto no_port;
120 }
121 }
122 if (!ps_has(&ps, 1)) {
123 no_port:
124 *port = defport;
125 return defport < 0 ? -1 : 0;
126 }
127 PS_WANT(__ps_getc(&ps) == ':');
128 PS_WANT(ps_has(&ps, 1));
129 i = ps_geti(&ps);
130 if (i < minport || i > UINT16_MAX)
131 return -1;
132 *port = i;
133 return ps_done(&ps) ? 0 : -1;
134}

136int addr_resolve2(const char * nonnull what, const lstr_t s,
137 int minport, int defport,
138 sockunion_t * nonnull out_su,
139 pstream_t * nullable out_host,
140 in_port_t * nullable out_port,
141 sb_t * nullable err)
142{
143 pstream_t host;
144 in_port_t port;
145
146 if (addr_parse_minport(ps_init(s.s, s.len), &host, &port,
147 minport, defport) < 0)
148 {
149 if (err) {
150 sb_addf(err, "unable to parse %s address `%*pM`",
151 what, LSTR_FMT_ARG(s));
152 }
153 return -1;
154 }
155 if (addr_info(out_su, AF_UNSPEC, host, port) < 0) {
156 if (err) {
157 sb_addf(err, "unable to resolve %s address `%*pM`",
158 what, LSTR_FMT_ARG(s));
159 }
160 return -1;
161 }
162
163 if (out_host) {
164 *out_host = host;
165 }
166 if (out_port) {
167 *out_port = port;
168 }
169
170 return 0;
171}

173const char *t_addr_fmt(const sockunion_t *su, int *slen)
174{
175 char buf[BUFSIZ];
176 size_t pos;
177
178 switch (su->family) {
179 case AF_INET:
180 inet_ntop(AF_INET, &su->sin.sin_addr.s_addr, buf, sizeof(buf));
181 pos = strlen(buf);
182 break;
183
184 case AF_INET6:
185 buf[0] = '[';
186 inet_ntop(AF_INET6, &su->sin6.sin6_addr.s6_addr,
187 buf + 1, sizeof(buf) - 1);
188 pos = strlen(buf);
189 buf[pos++] = ']';
190 break;
191
192 case AF_UNIX: {
193 lstr_t res;
194
195 if (su->sunix.sun_path[0] == '\0') {
196 if (su->sunix.sun_path[1] == '\0') {
197 res = LSTR("unknown unix socket");
198 } else {
199 res = t_lstr_fmt("@%s", su->sunix.sun_path + 1);
200 }
201 } else {
202 res = LSTR(su->sunix.sun_path);
203 }
204 if (slen) {
205 *slen = res.len;
206 }
207 return res.s;
208 }
209
210 default:
211 if (slen) {
212 *slen = 0;
213 }
214 return "";
215 }
216
217 /* Add port for AF_INET and AF_INET6. */
218 assert (pos < sizeof(buf));
219 pos += snprintf(buf + pos, sizeof(buf) - pos, ":%d",
220 sockunion_getport(su));
221 if (slen) {
222 *slen = pos;
223 }
224 return t_dupz(buf, pos);
225}

227int addr_info(sockunion_t *su, sa_family_t af, pstream_t host, in_port_t port)
228{
229 t_scope;
230 struct addrinfo *ai = NULL;
231 struct addrinfo *cur;
232 struct addrinfo hint = { .ai_family = af };
233
234 RETHROW(getaddrinfo(t_dupz(host.p, ps_len(&host)), NULL, &hint, &ai));
235 for (cur = ai; cur != NULL; cur = cur->ai_next) {
236 switch (cur->ai_family) {
237 case AF_INET:
238 case AF_INET6:
239 case AF_UNIX:
240 if (cur->ai_addrlen > ssizeof(*su)) {
241 continue;
242 }
243 break;
244
245 default:
246
247 continue;
248 }
249 break;
250 }
251
252 if (cur) {
253 memcpy(su, cur->ai_addr, cur->ai_addrlen);
254 sockunion_setport(su, port);
255 }
256 freeaddrinfo(ai);
257 return cur ? 0 : -1;
258}

260int addr_filter_build(lstr_t subnet, addr_filter_t *filter)
261{
262 t_scope;
263 pstream_t ps = ps_initlstr(&subnet);
264 pstream_t ip = ps_init(NULL, 0);
265 int maxmask = 32;
266 struct addrinfo *ai = NULL;
267 struct addrinfo hint = {
268 .ai_flags = AI_NUMERICHOST,
269 .ai_family = AF_UNSPEC,
270 };
271
272 if (ps_get_ps_chr_and_skip(&ps, '/', &ip) < 0) {
273 SWAP(pstream_t, ip, ps);
274 }
275
276 RETHROW(getaddrinfo(t_dupz(ip.s, ps_len(&ip)), NULL, &hint, &ai));
277
278 p_clear(filter, 1);
279 filter->family = ai->ai_family;
280 if (filter->family == AF_INET) {
281 filter->u.v4.addr =
282 ((struct sockaddr_in*)(ai->ai_addr))->sin_addr.s_addr;
283 } else
284 if (filter->family == AF_INET6) {
285 filter->u.v6.addr = ((struct sockaddr_in6*)ai->ai_addr)->sin6_addr;
286 maxmask = 128;
287 } else {
288 freeaddrinfo(ai);
289 return -1;
290 }
291
292 freeaddrinfo(ai);
293
294 if (!ps_done(&ps)) {
295 int mask;
296
297 errno = 0;
298 mask = ps_geti(&ps);
299
300 THROW_ERR_IF(errno != 0 || !ps_done(&ps) || mask < 0
301 || mask > maxmask);
302
303#define NET_U32_MASK(_mask) htonl(BITMASK_GE(uint64_t, 32 - (_mask)))
304#define NET_U32_MASK_BOUNDED(_mask) NET_U32_MASK(MAX(MIN((_mask), 32), 0))
305
306 if (filter->family == AF_INET) {
307 filter->u.v4.mask = NET_U32_MASK(mask);
308 filter->u.v4.addr &= filter->u.v4.mask;
309 } else {
310 filter->u.v6.mask.s6_addr32[0] = NET_U32_MASK_BOUNDED(mask);
311 filter->u.v6.mask.s6_addr32[1] = NET_U32_MASK_BOUNDED(mask - 32);
312 filter->u.v6.mask.s6_addr32[2] = NET_U32_MASK_BOUNDED(mask - 64);
313 filter->u.v6.mask.s6_addr32[3] = NET_U32_MASK_BOUNDED(mask - 96);
314
315 filter->u.v6.addr.s6_addr32[0] &= filter->u.v6.mask.s6_addr32[0];
316 filter->u.v6.addr.s6_addr32[1] &= filter->u.v6.mask.s6_addr32[1];
317 filter->u.v6.addr.s6_addr32[2] &= filter->u.v6.mask.s6_addr32[2];
318 filter->u.v6.addr.s6_addr32[3] &= filter->u.v6.mask.s6_addr32[3];
319 }
320
321#undef NET_U32_MASK_BOUNDED
322#undef NET_U32_MASK
323 } else {
324 if (filter->family == AF_INET) {
325 filter->u.v4.mask = UINT32_MAX;
326 } else {
327 filter->u.v6.mask.s6_addr32[0] = UINT32_MAX;
328 filter->u.v6.mask.s6_addr32[1] = UINT32_MAX;
329 filter->u.v6.mask.s6_addr32[2] = UINT32_MAX;
330 filter->u.v6.mask.s6_addr32[3] = UINT32_MAX;
331 }
332 }
333
334 return 0;
335}

337int addr_filter_matches(const addr_filter_t *filter, const sockunion_t *peer)
338{
339 if (peer->family != filter->family)
340 return -1;
341
342 if (filter->port && filter->port != sockunion_getport(peer))
343 return -1;
344
345 if (filter->family == AF_INET) {
346 if (filter->u.v4.addr
347 != (peer->sin.sin_addr.s_addr & filter->u.v4.mask))
348 {
349 return -1;
350 }
351 } else {
352 /* filter->family == AF_INET6 */
353 for (int i = 3; i >= 0; i--) {
354 if (filter->u.v6.addr.s6_addr32[i]
355 != (peer->sin6.sin6_addr.s6_addr32[i]
356 & filter->u.v6.mask.s6_addr32[i]))
357 {
358 return -1;
359 }
360 }
361 }
362
363 return 0;
364}

376int parse_http_url(const char *url_path, bool allow_https, http_url_t *url)
377{
378 static bool first_call = true;
379 static ctype_desc_t ctype_isurlseparator;
380 pstream_t ps;
381 pstream_t host_ps, user_ps, pass_ps, path_ps;
382
383 RETHROW_PN(url);
384 p_clear(url, 1);
385 url->port = 80;
386
387 ps = ps_initstr(url_path);
388
389 /* A URL should always start with "http://" or "https://" with proxy */
390 PS_WANT(!ps_skipstr(&ps, "http"));
391 if (allow_https) {
392 if (ps_startswith(&ps, "s", 1)) {
393 url->port = 443;
394 }
395 ps_skipstr(&ps, "s");
396 }
397 PS_WANT(!ps_skipstr(&ps, "://"));
398 PS_WANT(!ps_done(&ps));
399
400 if (unlikely(first_call)) {
401 ctype_desc_build(&ctype_isurlseparator, ":/@");
402 first_call = false;
403 }
404
405
406 host_ps = ps;
407 user_ps = ps_get_cspan(&ps, &ctype_isurlseparator);
408
409#define PS_COPY(_field, _ps) \
410 pstrcpymem(url->_field, sizeof(url->_field), _ps.s, ps_len(&_ps))
411
412 if (!ps_done(&ps) && *ps.s == ':') {
413 /* Colon is either the separator of user:passwd
414 * or the separator of host:port */
415 __ps_skip(&ps, 1);
416 pass_ps = ps_get_cspan(&ps, &ctype_isurlseparator);
417 if (!ps_done(&ps) && *ps.s == '@') {
418 /* @ is the separator of user:passwd@host so assume
419 * the colon was a user:passwd separator */
420 __ps_skip(&ps, 1);
421 PS_WANT(!ps_done(&ps));
422 PS_WANT(!ps_done(&user_ps));
423 PS_WANT(!ps_done(&pass_ps));
424 PS_COPY(user, user_ps);
425 PS_COPY(pass, pass_ps);
426
427 host_ps = ps_get_cspan(&ps, &ctype_isurlseparator);
428 if (!ps_done(&ps) && *ps.s == ':') {
429 __ps_skip(&ps, 1);
430 url->port = getport(ps.s, &ps.s);
431 }
432 } else {
433 /* The colon was a port separator so the preceding
434 * part was the host (not the user) */
435 host_ps = user_ps;
436 url->port = getport(pass_ps.s, &pass_ps.s);
437 }
438 } else {
439 host_ps = user_ps;
440 }
441
442 PS_WANT(url->port > 0);
443
444 /* hostname cannot be empty */
445 PS_WANT(!ps_done(&host_ps));
446
447 /* After parsing the host part, we must have reach a path separator */
448 PS_WANT(ps_done(&ps) || *ps.s == '/');
449 PS_COPY(host, host_ps);
450
451 /* Parse path */
452 if (ps_done(&ps)) {
453 pstrcpy(url->path, sizeof(url->path), "/");
454 pstrcpy(url->path_without_args, sizeof(url->path_without_args), "/");
455 } else {
456 PS_COPY(path, ps);
457 if (ps_get_ps_chr(&ps, '?', &path_ps) < 0) {
458 path_ps = ps;
459 ps = ps_init(NULL, 0);
460 }
461 PS_COPY(path_without_args, path_ps);
462 PS_COPY(args, ps);
463 }
464#undef PS_COPY
465
466 return 0;
467}
