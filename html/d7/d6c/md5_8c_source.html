176/*
177 * MD5 process buffer
178 */
179void md5_update( md5_ctx *ctx, const void *_input, ssize_t ilen )
180{
181 const byte *input = _input;
182 int fill;
183 uint32_t left;
184
185 if( ilen <= 0 )
186 return;
187
188 left = ctx->total[0] & 0x3F;
189 fill = 64 - left;
190
191 ctx->total[0] += ilen;
192 ctx->total[0] &= 0xFFFFFFFF;
193
194 if( ctx->total[0] < (uint32_t) ilen )
195 ctx->total[1]++;
196
197 if( left && ilen >= fill )
198 {
199 memcpy( (void *) (ctx->buffer + left),
200 (void *) input, fill );
201 md5_process( ctx, ctx->buffer );
202 input += fill;
203 ilen -= fill;
204 left = 0;
205 }
206
207 while( ilen >= 64 )
208 {
209 md5_process( ctx, input );
210 input += 64;
211 ilen -= 64;
212 }
213
214 if( ilen > 0 )
215 {
216 memcpy( (void *) (ctx->buffer + left),
217 (void *) input, ilen );
218 }
219}

229/*
230 * MD5 final digest
231 */
232void md5_finish( md5_ctx *ctx, byte output[16] )
233{
234 uint32_t last, padn;
235 uint32_t high, low;
236 byte msglen[8];
237
238 high = ( ctx->total[0] >> 29 )
239 | ( ctx->total[1] << 3 );
240 low = ( ctx->total[0] << 3 );
241
242 PUT_U32_LE( low, msglen, 0 );
243 PUT_U32_LE( high, msglen, 4 );
244
245 last = ctx->total[0] & 0x3F;
246 padn = ( last < 56 ) ? ( 56 - last ) : ( 120 - last );
247
248 md5_update( ctx, (byte *) md5_padding, padn );
249 md5_update( ctx, msglen, 8 );
250
251 PUT_U32_LE( ctx->state[0], output, 0 );
252 PUT_U32_LE( ctx->state[1], output, 4 );
253 PUT_U32_LE( ctx->state[2], output, 8 );
254 PUT_U32_LE( ctx->state[3], output, 12 );
255}

303/*
304 * output = MD5( file contents )
305 */
306int md5_file( char *path, byte output[16] )
307{
308 FILE *f;
309 size_t n;
310 md5_ctx ctx;
311 byte buf[1024];
312
313 if( ( f = fopen( path, "rb" ) ) == NULL )
314 return( 1 );
315
316 md5_starts( &ctx );
317
318 while( ( n = fread( buf, 1, sizeof( buf ), f ) ) > 0 )
319 md5_update( &ctx, buf, (int) n );
320
321 md5_finish( &ctx, output );
322
323 memset( &ctx, 0, sizeof( md5_ctx ) );
324
325 if( ferror( f ) != 0 )
326 {
327 fclose( f );
328 return( 2 );
329 }
330
331 fclose( f );
332 return( 0 );
333}

335/*
336 * MD5 HMAC context setup
337 */
338void md5_hmac_starts( md5_ctx *ctx, const void *_key, int keylen )
339{
340 const byte *key = _key;
341 int i;
342 byte sum[16];
343
344 if( keylen > 64 )
345 {
346 md5( key, keylen, sum );
347 keylen = 16;
348 key = sum;
349 }
350
351 memset( ctx->ipad, 0x36, 64 );
352 memset( ctx->opad, 0x5C, 64 );
353
354 for( i = 0; i < keylen; i++ )
355 {
356 ctx->ipad[i] = (byte)( ctx->ipad[i] ^ key[i] );
357 ctx->opad[i] = (byte)( ctx->opad[i] ^ key[i] );
358 }
359
360 md5_starts( ctx );
361 md5_update( ctx, ctx->ipad, 64 );
362
363 memset( sum, 0, sizeof( sum ) );
364}
