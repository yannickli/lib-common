34void xmlpp_close(xmlpp_t *pp)
35{
36 while (pp->stack.len) {
37 xmlpp_closetag(pp);
38 }
39 if (!pp->nospace && pp->buf->data[pp->buf->len - 1] != '\n') {
40 sb_addc(pp->buf, '\n');
41 }
42 qv_deep_wipe(&pp->stack, lstr_wipe);
43}

45void xmlpp_opentag(xmlpp_t *pp, const char *tag_)
46{
47 lstr_t tag = lstr_dups(tag_, strlen(tag_));
48
49 if (pp->nospace) {
50 sb_addf(pp->buf, "<%*pM>", tag.len, tag.s);
51 } else {
52 sb_addf(pp->buf, "%-*c<%*pM>", pp->stack.len * 2 + 1, '\n',
53 tag.len, tag.s);
54 }
55 qv_append(&pp->stack, tag);
56 pp->can_do_attr = true;
57 pp->was_a_tag = true;
58}

60void xmlpp_closetag(xmlpp_t *pp)
61{
62 lstr_t tag;
63
64 if (!pp->stack.len) {
65 return;
66 }
67
68 tag = *tab_last(&pp->stack);
69 qv_shrink(&pp->stack, 1);
70 if (pp->can_do_attr) {
71 sb_shrink(pp->buf, 1);
72 sb_adds(pp->buf, " />");
73 } else {
74 if (pp->was_a_tag && !pp->nospace) {
75 sb_addc(pp->buf, '\n');
76 sb_addnc(pp->buf, pp->stack.len * 2, ' ');
77 }
78 sb_addf(pp->buf, "</%*pM>", tag.len, tag.s);
79 }
80 pp->can_do_attr = false;
81 pp->was_a_tag = true;
82 lstr_wipe(&tag);
83}

85void xmlpp_nl(xmlpp_t *pp)

110void xmlpp_putattrfmt(xmlpp_t *pp, const char *key, const char *fmt, ...)
111{

121 va_end(ap);

123 sb_shrink(pp->buf, 1);
124 sb_addf(pp->buf, " %s=\"", key);
125 sb_add_xmlescape(pp->buf, tmp.data, tmp.len);

129void xmlpp_put_cdata(xmlpp_t *pp, const char *s, size_t len)
130{
131 const char *p;
132
133 pp->can_do_attr = false;
134 pp->was_a_tag = false;
135 sb_adds(pp->buf, "<![CDATA[");
136 while (unlikely((p = memmem(s, len, "]]>", 3)) != NULL)) {
137 sb_add(pp->buf, s, p - s);
138 sb_adds(pp->buf, "]]>]]<![CDATA[>");
139 len -= (p - s) + 3;
140 s = p + 3;
141 }
142 sb_add(pp->buf, s, len);
143 sb_adds(pp->buf, "]]>");
144}
