53static lstr_t t_get_short_name(const lstr_t fullname, bool down)
54{
55 lstr_t name = t_lstr_dup(fullname);
56 pstream_t obj_name = ps_initlstr(&name);
57 lstr_t out;
58
59 if (ps_skip_afterlastchr(&obj_name, '.') < 0) {
60 logger_fatal(&_G.logger, "fullname `%*pM` should be at least "
61 "composed by `pkg.name`", LSTR_FMT_ARG(fullname));
62 }
63 out = t_lstr_fmt("%s", obj_name.s);
64 /* First letter must be down */
65 if (down) {
66 out.v[0] = tolower((unsigned char)out.v[0]);
67 }
68
69 return out;
70}

72static lstr_t t_split_on_str(lstr_t name, const char *letter, bool enums)
73{
74 t_SB(buf, 64);
75 qv_t(lstr) parts;
76 lstr_t word = t_lstr_dup(name);
77 pstream_t ps = ps_initlstr(&word);
78 ctype_desc_t dot;
79
80 t_qv_init(&parts, 80);
81 ctype_desc_build(&dot, letter);
82 ps_split(ps, &dot, 0, &parts);
83 if (!parts.len) {
84 return LSTR_EMPTY_V;
85 }
86
87 tab_enumerate_ptr(i, part, &parts) {
88 if (enums) {
89 for (int j = i != 0; j < part->len; j++) {
90 parts.tab[i].v[j] = tolower(part->v[j]);
91 }
92 } else
93 if (i < parts.len -1) {
94 *part = t_lstr_cat(*part, LSTR("\'"));
95 }
96 sb_add_lstr(&buf, parts.tab[i]);
97 }
98
99 return t_lstr_fmt("%*pM", SB_FMT_ARG(&buf));
100}

102static lstr_t t_mib_put_enum(const iop_enum_t *en)
103{
104 SB_1k(names);

106 sb_add_lstr(&names, LSTR("INTEGER {"));
107 for (int i = 0; i < en->enum_len; i++) {

114 return t_lstr_fmt("%*pM", SB_FMT_ARG(&names));

117static lstr_t t_get_type_to_lstr(const iop_field_t *field, bool seq,
118 bool is_index)
119{
120 switch (field->type) {
121 case IOP_T_STRING:
122 return is_index ? LSTR("OCTET STRING (SIZE(0..100))") :
123 LSTR("OCTET STRING");
124 case IOP_T_I8:
125 case IOP_T_I16:
126 case IOP_T_I32:
127 return LSTR("Integer32");
128 case IOP_T_U32:
129 return is_index ? LSTR("Integer32 (1..2147483647)") :
130 LSTR("Integer32");
131 case IOP_T_BOOL:
132 return LSTR("BOOLEAN");
133
134 case IOP_T_ENUM:
135 return seq ? LSTR("INTEGER") : t_mib_put_enum(field->u1.en_desc);
136
137 default:
138 logger_panic(&_G.logger, "type not handled");
139 }
140}

207static void mib_get_head(const qv_t(pkg) *pkgs)
208{
209 bool resolved = false;
210
211 if (pkgs->len <= 0) {
212 logger_fatal(&_G.logger,
213 "a package must be provided to build the MIB");
214 }
215
216 tab_for_each_entry(pkg, pkgs) {
217 for (const iop_struct_t *const *it = pkg->structs; *it; it++) {
218 const iop_struct_t *desc = *it;
219
220 if (!iop_struct_is_snmp_obj(desc)) {
221 continue;
222 }
223 if (qh_add(lstr, &_G.objects_identifier, &desc->fullname) < 0) {
224 logger_fatal(&_G.logger, "name `%*pM` already exists",
225 LSTR_FMT_ARG(desc->fullname));
226 }
227 if (desc->snmp_attrs->parent) {
228 qv_append(&_G.objects_identifier_parent,
229 desc->snmp_attrs->parent->fullname);
230 }
231 }
232 for (const iop_iface_t *const *it = pkg->ifaces; *it; it++) {
233 const iop_iface_t *iface = *it;
234
235 if (!iop_iface_is_snmp_iface(iface)) {
236 continue;
237 }
238 if (qh_add(lstr, &_G.objects_identifier, &iface->fullname) < 0) {
239 logger_fatal(&_G.logger, "name `%*pM` already exists",
240 LSTR_FMT_ARG(iface->fullname));
241 }
242 qv_append(&_G.objects_identifier_parent,
243 iface->snmp_iface_attrs->parent->fullname);
244 }
245 }
246
247 tab_for_each_entry(name, &_G.objects_identifier_parent) {
248 if (qh_find(lstr, &_G.objects_identifier, &name) < 0) {
249 t_scope;
250 lstr_t short_name = t_get_short_name(name, true);
251
252 if (!lstr_equal(short_name, _G.head)) {
253 if (resolved) {
254 logger_fatal(&_G.logger, "only one snmpObj parent should "
255 "be imported");
256 }
257 lstr_copy(&_G.head, short_name);
258 }
259 resolved = true;
260 }
261 }
262
263 if (!resolved) {
264 _G.head = LSTR("intersec");
265 _G.head_is_intersec = true;
266 }
267}

333{
334 t_scope;
335 const iop_snmp_attrs_t *snmp_attrs = snmp_obj->snmp_attrs;

341 t_get_short_name(snmp_attrs->parent->fullname, true) :
342 LSTR("intersec");
343
344 sb_addf(buf,
345 "%*pM" LVL1 "OBJECT IDENTIFIER ::= { %*pM %d }\n",
346 LSTR_FMT_ARG(t_get_short_name(snmp_obj->fullname, true)),
347 LSTR_FMT_ARG(parent), snmp_attrs->oid);
348}

367{

399static void mib_put_tbl_entries(const iop_struct_t *st, lstr_t name_down,
400 sb_t *buf)
401{
402 t_scope;
403
404 sb_addf(buf, "\n%*pMEntry ::= SEQUENCE {\n",
405 LSTR_FMT_ARG(t_get_short_name(st->fullname, false)));
406
407 for (int i = 0; i < st->fields_len; i++) {
408 const iop_field_t field = st->fields[i];
409
410 /* In the sequence, the limits should not be given so the
411 * is_index argument of t_get_type_to_lstr is set at false
412 */
413 sb_addf(buf, LVL1 "%*pM %*pM",
414 LSTR_FMT_ARG(field.name),
415 LSTR_FMT_ARG(t_get_type_to_lstr(&field, true, false)));
416
417 if (i == st->fields_len - 1) {
418 sb_addc(buf, '\n');
419 } else {
420 sb_adds(buf, ",\n");
421 }
422 }
423 sb_adds(buf, "}\n");
424}

426static void mib_put_snmp_tbl(const iop_struct_t *st, bool has_index,
427 sb_t *buf)
428{
429 t_scope;
430 const iop_snmp_attrs_t *snmp_attrs;
431 lstr_t name_up = t_get_short_name(st->fullname, false);
432 lstr_t name_down = t_get_short_name(st->fullname, true);
433 lstr_t help;
434 int nb = 0;
435
436 assert (iop_struct_is_snmp_tbl(st));
437
438 snmp_attrs = st->snmp_attrs;
439 help = t_mib_tbl_get_help(st->st_attrs);
440
441 help = t_split_on_str(help, "\'", false);
442
443 sb_addf(buf,
444 "\n%*pMTable OBJECT-TYPE\n"
445 LVL1 "SYNTAX SEQUENCE OF %*pMEntry\n"
446 LVL1 "MAX-ACCESS not-accessible\n"
447 LVL1 "STATUS current\n"
448 LVL1 "DESCRIPTION\n"
449 LVL2 "\"%*pM\"\n"
450 LVL1 "::= { %*pM %d }\n",
451 LSTR_FMT_ARG(name_down), LSTR_FMT_ARG(name_up), LSTR_FMT_ARG(help),
452 LSTR_FMT_ARG(t_get_short_name(snmp_attrs->parent->fullname, true)),
453 snmp_attrs->oid);
454
455 /* Define the table entry that gives global information about table
456 * entries */
457 sb_addf(buf,
458 "\n%*pMEntry OBJECT-TYPE\n"
459 LVL1 "SYNTAX %*pMEntry\n"
460 LVL1 "MAX-ACCESS not-accessible\n"
461 LVL1 "STATUS current\n"
462 LVL1 "DESCRIPTION\n"
463 LVL2 "\"An entry in the table of %*pM\"\n",
464 LSTR_FMT_ARG(name_down), LSTR_FMT_ARG(name_up),
465 LSTR_FMT_ARG(name_down));
466
467 sb_adds(buf, LVL1 "INDEX { ");
468 for (int i = 0; i < st->fields_len; i++) {
469 if (iop_field_is_snmp_index(&st->fields[i])) {
470 sb_addf(buf, "%*pM", LSTR_FMT_ARG(st->fields[i].name));
471 nb++;
472
473 if (nb < iop_struct_get_nb_snmp_indexes(st)) {
474 sb_adds(buf, ", ");
475 }
476 }
477 }
478 sb_adds(buf, " }\n");
479
480 sb_addf(buf, LVL1 "::= { %*pMTable 1 }\n", LSTR_FMT_ARG(name_down));
481
482 /* Define the table entries (corresponding to the columns) */
483 mib_put_tbl_entries(st, name_down, buf);
484}

489static void mib_put_field(sb_t *buf, lstr_t name, int pos,
490 const iop_struct_t *st, bool from_tbl)
491{
492 t_scope;
493 const iop_field_attrs_t *field_attrs = &st->fields_attrs[pos];
494 const iop_field_t *field = &st->fields[pos];
495 const iop_snmp_attrs_t *snmp_attrs;
496 bool is_index = iop_field_is_snmp_index(field);
497 const char *access_str;
498
499 snmp_attrs = iop_get_snmp_attrs(field_attrs);
500
501 access_str = iop_struct_is_snmp_param(snmp_attrs->parent) ?
502 "accessible-for-notify" : "read-only";
503
504 sb_addf(buf,
505 "\n%*pM OBJECT-TYPE\n"
506 LVL1 "SYNTAX %*pM\n"
507 LVL1 "MAX-ACCESS %s\n"
508 LVL1 "STATUS current\n"
509 LVL1 "DESCRIPTION\n"
510 LVL2 "\"%*pM\"\n"
511 LVL1 "::= { %*pM%s %d }\n",
512 LSTR_FMT_ARG(name),
513 LSTR_FMT_ARG(t_get_type_to_lstr(field, false, is_index)),
514 is_index ? "not-accessible" : access_str,
515 LSTR_FMT_ARG(t_mib_field_get_help(field_attrs)),
516 LSTR_FMT_ARG(t_get_short_name(snmp_attrs->parent->fullname, true)),
517 from_tbl ? "Entry" : "",
518 snmp_attrs->oid);
519
520 if (qh_add(lstr, &_G.unicity_conformance_objects, &name) < 0) {
521 logger_fatal(&_G.logger,
522 "conflicting field name `%*pM`", LSTR_FMT_ARG(name));
523 }
524 if (!is_index) {
525 qv_append(&_G.conformance_objects, name);
526 }
527}

541static void mib_put_fields_and_tbl(sb_t *buf, const iop_pkg_t *pkg)
542{
543 for (const iop_struct_t *const *it = pkg->structs; *it; it++) {
544 const iop_struct_t *desc = *it;
545 bool has_fields = desc->fields_len > 0;
546
547 if (!iop_struct_is_snmp_st(desc)) {
548 continue;
549 }
550
551 if (iop_struct_is_snmp_tbl(desc)) {
552 t_scope;
553
554 sb_addf(buf, "-- {{{ %*pMTable\n",
555 LSTR_FMT_ARG(t_get_short_name(desc->fullname, false)));
556
557
558 mib_put_snmp_tbl(desc, iop_struct_get_nb_snmp_indexes(desc) > 0,
559 buf);
560 mib_put_tbl_fields(buf, desc);
561 sb_addf(buf, "\n-- }}}\n");
562 continue;
563 }
564
565 if (has_fields) {
566 t_scope;
567
568 sb_addf(buf,
569 "-- {{{ %*pM\n",
570 LSTR_FMT_ARG(t_get_short_name(desc->fullname, false)));
571 }
572
573 /* deal with snmp fields */
574 for (int i = 0; i < desc->fields_len; i++) {
575 const iop_field_t field = desc->fields[i];
576
577 if (!iop_field_has_snmp_info(&field)) {
578 continue;
579 }
580 mib_put_field(buf, field.name, i, desc, false);
581 }
582
583 if (has_fields) {
584 sb_addf(buf, "\n-- }}}\n");
585 }
586 }
587}

624static void mib_put_rpcs(sb_t *buf, const iop_pkg_t *pkg)
625{
626 for (const iop_iface_t *const *it = pkg->ifaces; *it; it++) {
627 const iop_iface_t *iface = *it;
628 bool has_rpcs = iface->funs_len > 0;

631 continue;
632 }
633 if (has_rpcs) {
634 t_scope;
635

637 LSTR_FMT_ARG(t_get_short_name(iface->fullname, false)));
638 }
639 /* deal with snmp rpcs */

641 const iop_rpc_t rpc = iface->funs[i];
642 const iop_rpc_attrs_t attrs = iface->rpc_attrs[i];
643
644 mib_put_rpc(buf, i, &rpc, attrs, iface->fullname);
645 }
646

648 sb_addf(buf, "\n-- }}}\n");
649 }
650 }
651}

778void iop_write_mib(sb_t *sb, const qv_t(pkg) *pkgs,
779 const qv_t(mib_rev) *revisions)
780{
781 SB_8k(buffer);
782
783 MODULE_REQUIRE(iop_mib);
784
785 mib_get_head(pkgs);
786
787 mib_put_object_identifier(&buffer, pkgs);
788 tab_for_each_entry(pkg, pkgs) {
789 mib_put_fields_and_tbl(&buffer, pkg);
790 mib_put_rpcs(&buffer, pkg);
791 }
792
793 mib_open_banner(sb);
794 mib_put_imports(sb);
795 mib_put_identity(sb, revisions);
796 mib_put_compliance_fold(sb);
797
798 /* Concat both sb */
799 sb_addsb(sb, &buffer);
800 mib_close_banner(sb);
801
802 MODULE_RELEASE(iop_mib);
803}

817 return -1;
818 }
819 } else {
820 fprintf(stdout, "%*pM", SB_FMT_ARG(&sb));
821 }
822
823 return 0;
824}
