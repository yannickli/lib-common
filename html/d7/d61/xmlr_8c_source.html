28/*
29 * This libxml wrapper supposes the following:
30 *
31 * - all is utf8
32 * - there is no meaningful interleaved text and nodes like in html:
33 * <p>text 1<b>some b node</b>text2</p>
34 *
35 *
36 * The following invariant is maintained: at any given time, the xml_reader_t
37 * points onto an opening or closing element, except when the parsing failed
38 * for some reason.
39 *
40 */
41
42__thread xml_reader_t xmlr_g;
43
44#define XMLR_OPTS \
45 (XML_PARSE_NOERROR | XML_PARSE_NOWARNING | \
46 XML_PARSE_NONET | XML_PARSE_NOCDATA | XML_PARSE_NOBLANKS)
47
48#define XTYPE(type) XML_READER_TYPE_##type

54static __attribute__((format(printf, 2, 3)))
55void xmlr_debug_error(void *ctx, const char *fmt, ...)
56{
57 va_list ap;
58
59 va_start(ap, fmt);
60 sb_addvf(ctx, fmt, ap);
61 va_end(ap);
62
63#ifndef NDEBUG
64 va_start(ap, fmt);
65 vfprintf(stderr, fmt, ap);
66 va_end(ap);
67#endif
68}

70static void xmlr_initialize(void)
71{
72 if (unlikely(xmlr_err_g.size == 0))
73 sb_init(&xmlr_err_g);
74 sb_reset(&xmlr_err_g);
75 xmlGenericError = (xmlGenericErrorFunc)xmlr_debug_error;
76 xmlGenericErrorContext = &xmlr_err_g;
77}

100static void xmlr_fmt_loc(xml_reader_t xr, sb_t *sb)
101{
102 xmlNodePtr n[32];
103 int nlen = 0;
104
105 n[0] = xmlTextReaderCurrentNode(xr);
106 for (; n[nlen] && nlen + 1 < countof(n); nlen++) {
107 n[nlen + 1] = n[nlen]->parent;
108 }
109 sb_addf(sb, "%ld", xmlGetLineNo(n[0]));
110 if (nlen) {
111 if (nlen == countof(n) && n[nlen - 1]->parent) {
112 sb_adds(sb, ": near ...");
113 } else {
114 sb_adds(sb, ": near ");
115 }
116 for (int i = nlen; i-- > 0; ) {
117 if (n[i]->type == XML_ELEMENT_NODE) {
118 sb_addf(sb, "/%s", (const char *)n[i]->name);
119 } else
120 if (n[i]->type == XML_TEXT_NODE) {
121 assert (i == 0);
122 sb_adds(sb, "/text()");
123 } else
124 if (n[i]->type == XML_DOCUMENT_NODE) {
125 assert (i == nlen - 1);
126 } else {
127 assert (false);
128 }
129 }
130 }
131}
132
133int xmlr_fail(xml_reader_t xr, const char *fmt, ...)
134{
135 va_list ap;
136
137 sb_reset(&xmlr_err_g);
138 xmlr_fmt_loc(xr, &xmlr_err_g);
139
140 sb_adds(&xmlr_err_g, ": ");
141 va_start(ap, fmt);
142 sb_addvf(&xmlr_err_g, fmt, ap);
143 va_end(ap);
144 e_named_trace(1, "xml/reader", "%s", xmlr_err_g.data);
145
146 return XMLR_ERROR;
147}

160static ALWAYS_INLINE bool xmlr_on_element(xml_reader_t xr, bool allow_closing)
161{
162 switch (xmlTextReaderNodeType(xr)) {
163 case XTYPE(ELEMENT):
164 return true;
165 case XTYPE(END_ELEMENT):
166 return allow_closing;
167 default:
168 return false;
169 }
170}
171
172static ALWAYS_INLINE int xmlr_scan_node(xml_reader_t xr, bool stop_on_text)
173{
174 for (;;) {
175 switch (xmlTextReaderNodeType(xr)) {
176 case -1:
177 return xmlr_fail(xr, "unable to get node type");
178 case XTYPE(ELEMENT):
179 case XTYPE(END_ELEMENT):
180 case XTYPE(NONE):
181 return 0;
182 case XTYPE(TEXT):
183 if (stop_on_text)
184 return 0;
185 break;
186 default:
187 break;
188 }
189 RETHROW(xmlTextReaderRead(xr));
190 }
191}

193int xmlr_node_close(xml_reader_t xr)
194{
195 if (xmlr_node_is_empty(xr) == 1 || xmlr_node_is_closing(xr) == 1) {
196 return xmlr_next_node(xr);
197 }
198 /* XXX: consider <toto></toto> as autoclosing because libxml doesn't */
199 if (!xmlr_on_element(xr, false) || xmlTextReaderRead(xr) < 0) {
200 return xmlr_fail(xr, "closing tag expected");
201 }
202 RETHROW(xmlr_scan_node(xr, true));
203 if (xmlr_node_is_empty(xr) == 1 || xmlr_node_is_closing(xr) == 1) {
204 return xmlr_next_node(xr);
205 }
206 return xmlr_fail(xr, "closing tag expected");
207}

209int xmlr_setup(xml_reader_t *xrp, const void *buf, int len)
210{
211 xml_reader_t xr = *xrp;
212
213 xmlr_initialize();
214
215 if (xr) {
216 if (xmlReaderNewMemory(xr, buf, len, NULL, NULL, XMLR_OPTS) < 0) {
217 xmlFreeTextReader(xr);
218 *xrp = xr = NULL;
219 return xmlr_fail(xr, "unable to allocate parser");
220 }
221 } else {
222 *xrp = xr = xmlReaderForMemory(buf, len, NULL, NULL, XMLR_OPTS);
223 if (!xr)
224 return xmlr_fail(xr, "unable to allocate parser");
225 }
226
227 xmlTextReaderSetErrorHandler(xr, xmlr_err_handler, xr);
228 if (RETHROW(xmlTextReaderRead(xr)) != 1)
229 return xmlr_fail(xr, "unable to load root node");
230 return xmlr_scan_node(xr, false);
231}
232
233void xmlr_close(xml_reader_t *xrp)
234{
235 xml_reader_t xr = *xrp;
236
237 /* when the parsing was interrupted, namespace stack isn't reset properly
238 * leading to bugs, so workaround it when we're not sure the parse ended
239 * well
240 */
241 if (xmlTextReaderRead(xr) == 0 &&
242 xmlTextReaderReadState(xr) == XML_TEXTREADER_MODE_EOF)
243 {
244 xmlTextReaderClose(xr);
245 } else {
246 xmlFreeTextReader(xr);
247 *xrp = NULL;
248 }
249}

284int xmlr_next_node(xml_reader_t xr)
285{
286 assert (xmlr_on_element(xr, true));
287
288 RETHROW(xmlTextReaderRead(xr));
289 return xmlr_scan_node(xr, false);
290}
291
292int xmlr_next_child(xml_reader_t xr)
293{
294 assert (xmlr_on_element(xr, false));
295
296 if (RETHROW(xmlr_node_is_empty(xr))) {
297 xmlr_fail(xr, "node has no children");
298 return XMLR_NOCHILD;
299 }
300 RETHROW(xmlr_next_node(xr));
301 if (RETHROW(xmlr_node_is_closing(xr))) {
302 xmlr_fail(xr, "node has no children");
303 return XMLR_NOCHILD;
304 }
305 return 0;
306}
307
308int xmlr_next_sibling(xml_reader_t xr)
309{
310 assert (xmlr_on_element(xr, true));
311
312 if (RETHROW(xmlTextReaderNext(xr)) == 1)
313 return xmlr_scan_node(xr, false);
314 return xmlr_fail(xr, "node has no sibling");
315}
316
317int xmlr_next_uncle(xml_reader_t xr)
318{
319 while (!RETHROW(xmlr_node_is_closing(xr))) {
320 RETHROW(xmlr_next_sibling(xr));
321 }
322 return xmlr_next_node(xr);
323}
324
325int xmlr_node_enter(xml_reader_t xr, const char *s, size_t len, int flags)
326{
327 int res = XMLR_NOCHILD;
328
329 if (!RETHROW(xmlr_node_is(xr, s, len))) {
330 if (flags & XMLR_ENTER_MISSING_OK)
331 return 0;
332 return xmlr_fail(xr, "expecting tag <%s>", s);
333 }
334
335 if (!RETHROW(xmlr_node_is_empty(xr)))
336 res = xmlr_next_child(xr);
337
338 if (res == XMLR_NOCHILD) {
339 if (flags & XMLR_ENTER_EMPTY_OK)
340 return xmlr_next_node(xr);
341 return xmlr_fail(xr, "no child in node <%s>", s);
342 }
343 return res ?: 1;
344}
345
346int xmlr_node_skip_until(xml_reader_t xr, const char *s, int len)
347{
348 while (!RETHROW(xmlr_node_is_closing(xr))) {
349 if (RETHROW(xmlr_node_is(xr, s, len)))
350 return 0;
351 RETHROW(xmlr_next_sibling(xr));
352 }
353 return xmlr_fail(xr, "missing <%s> tag", s);
354}

359__flatten
360int xmlr_get_cstr_start(xml_reader_t xr, bool emptyok, lstr_t *out)
361{
362 const char *s = NULL;
363
364 assert (xmlr_on_element(xr, false));
365
366 if (!RETHROW(xmlr_node_is_empty(xr))) {
367 if (RETHROW(xmlTextReaderRead(xr)) != 1)
368 return xmlr_fail(xr, "expecting text or closing element");
369 RETHROW(xmlr_scan_node(xr, true));
370 if (RETHROW(xmlTextReaderNodeType(xr)) == XTYPE(TEXT))
371 s = (const char *)xmlTextReaderConstValue(xr);
372 }
373 if (s) {
374 *out = LSTR(s);
375 } else {
376 if (!emptyok)
377 return xmlr_fail(xr, "node value is missing");
378 *out = LSTR_EMPTY_V;
379 }
380 return 0;
381}
382
383int xmlr_get_cstr_done(xml_reader_t xr)
384{
385 if (!RETHROW(xmlr_node_is_empty(xr))) {
386 if (RETHROW(xmlTextReaderNodeType(xr)) == XTYPE(TEXT)) {
387 RETHROW(xmlr_scan_node(xr, false));
388 }
389 if (RETHROW(xmlTextReaderNodeType(xr)) != XTYPE(END_ELEMENT))
390 return xmlr_fail(xr, "expecting closing tag");
391 }
392 return xmlr_next_node(xr);
393}

572xmlAttrPtr
573xmlr_find_attr(xml_reader_t xr, const char *name, size_t len, bool needed)
574{
575 xmlr_for_each_attr(attr, xr) {
576 if (strlen((char *)attr->name) != len)
577 continue;
578 if (memcmp(attr->name, name, len))
579 continue;
580 return attr;
581 }
582 if (needed)
583 xmlr_fail(xr, "missing [%s] attribute", name);
584 return NULL;
585}
