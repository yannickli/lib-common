26{
27 /* To generate such test data, you can:
28 * $ openssl genpkey -algorithm RSA -out priv.pem -pkeyopt
29 * rsa_keygen_bits:2048
30 * $ openssl pkey -in priv.pem -pubout -out pub.pem
31 * $ echo 'secret pioupiou23' > a
32 * $ openssl pkeyutl -encrypt -pubin -inkey pub.pem -in a -hexdump
33 */
34 t_scope;
35
36 if (sb_read_file(priv, t_fmt("%*pM/tests/test-data/keys/priv.pem",
37 LSTR_FMT_ARG(libcommon_path))) < 0)

39 e_fatal("cannot read private key");
40 }
41 if (sb_read_file(priv_encrypted,
42 t_fmt("%*pM/tests/test-data/keys/priv.encrypted.pem",

46 }

48 LSTR_FMT_ARG(libcommon_path))) < 0)
49 {
50 e_fatal("cannot read public key");
51 }
52}

54int z_run_command(const char * nonnull file,
55 const char * nullable argv[],
56 const char * nullable envp[],
57 int timeout, int expected_status,
58 sb_t *output)
59{

63

65 ^void (el_t ev, pid_t pid, int st, lstr_t out) {
66 block_status = st;
67 sb_add_lstr(output, out);
68 finished = true;
69 }, NULL);
70
71 do {
72 el_loop_timeout(100);

76 Z_ASSERT_EQ(status, expected_status);
