35static lstr_t iop_pkgname_from_fullname(lstr_t fullname)
36{
37 const void *p;
38 pstream_t pkgname;
39 pstream_t ps = ps_initlstr(&fullname);
40
41 p = memrchr(ps.s, '.', ps_len(&ps));
42 if (!p) {
43 /* XXX: This case case may happen with the special 'Void' package */
44 return LSTR_NULL_V;
45 }
46 pkgname = __ps_get_ps_upto(&ps, p);
47 return LSTR_PS_V(&pkgname);
48}

50static const iop_struct_t *
51iop_get_struct(const iop_pkg_t *pkg, lstr_t fullname)
52{
53 for (const iop_struct_t * const *st = pkg->structs; *st; st++) {
54 if (lstr_equal(fullname, (*st)->fullname)) {
55 return *st;
56 }
57 }
58 for (const iop_iface_t * const *iface = pkg->ifaces; *iface; iface++) {
59 for (int i = 0; i < (*iface)->funs_len; i++) {
60 const iop_rpc_t *rpc = &(*iface)->funs[i];
61
62 if (lstr_equal(fullname, rpc->args->fullname)) {
63 return rpc->args;
64 }
65 if (lstr_equal(fullname, rpc->result->fullname)) {
66 return rpc->result;
67 }
68 if (lstr_equal(fullname, rpc->exn->fullname)) {
69 return rpc->exn;
70 }
71 }
72 }
73
74 return NULL;
75}

77static int iopdso_fix_struct_ref(iop_dso_t *dso, const iop_struct_t **st,
78 const iop_pkg_t *own_pkg, sb_t *err)
79{
80 const iop_struct_t *fix;
81 lstr_t pkgname = iop_pkgname_from_fullname((*st)->fullname);
82 const iop_pkg_t *pkg;
83 iop_dso_t *dep;
84
85 pkg = iop_get_pkg(pkgname);
86 if (!pkg) {
87 lstr_t pkgname2 = iop_pkgname_from_fullname(pkgname);
88
89 pkg = iop_get_pkg(pkgname2);
90 if (!pkg) {
91 e_trace(4, "cannot find package `%*pM` in current environment",
92 LSTR_FMT_ARG(pkgname));
93 return 0;
94 }
95 }
96
97 if (lstr_equal(pkg->name, own_pkg->name)) {
98 return 0;
99 }
100
101 fix = iop_get_struct(pkg, (*st)->fullname);
102 if (!fix) {
103 e_error("IOP DSO: did not find struct %s in memory",
104 (*st)->fullname.s);
105 return 0;
106 }
107 if (fix != *st) {
108 if (dso->dont_replace_fix_pkg) {
109 sb_setf(err, "package `%*pM` is already defined when loading IOP "
110 "DSO `%*pM`", LSTR_FMT_ARG((*st)->fullname),
111 LSTR_FMT_ARG(dso->path));
112 return -1;
113 }
114
115 /* XXX: Kept for backward compatibility.
116 * To be deleted once all clients will have a product >= 2017.1. */
117 e_trace(3, "fixup `%*pM`, %p => %p", LSTR_FMT_ARG((*st)->fullname),
118 *st, fix);
119 mprotect((void *)(((uintptr_t)st >> PAGE_SIZE_SHIFT) << PAGE_SIZE_SHIFT),
120 PAGE_SIZE, PROT_READ | PROT_WRITE);
121 *st = fix;
122 }
123
124 dep = iop_dso_get_from_pkg(pkg);
125 if (dep && dep != dso) {
126 qh_add(ptr, &dso->depends_on, dep);
127 qh_add(ptr, &dep->needed_by, dso);
128 }
129
130 return 0;
131}

133static int iopdso_fix_class_parent(iop_dso_t *dso, const iop_struct_t *desc,
134 const iop_pkg_t *own_pkg, sb_t *err)
135{
136 iop_class_attrs_t *class_attrs;
137
138 if (!iop_struct_is_class(desc)) {
139 return 0;
140 }
141
142 class_attrs = (iop_class_attrs_t *)desc->class_attrs;
143 if (class_attrs->parent) {
144 RETHROW(iopdso_fix_struct_ref(dso, &class_attrs->parent, own_pkg,
145 err));
146 }
147 return 0;
148}

150static int iopdso_fix_pkg(iop_dso_t *dso, const iop_pkg_t *pkg, sb_t *err)
151{
152 for (const iop_struct_t *const *it = pkg->structs; *it; it++) {
153 const iop_struct_t *desc = *it;
154
155 RETHROW(iopdso_fix_struct_ref(dso, &desc, pkg, err));
156 RETHROW(iopdso_fix_class_parent(dso, desc, pkg, err));
157
158 for (int i = 0; i < desc->fields_len; i++) {
159 iop_field_t *f = (iop_field_t *)&desc->fields[i];
160
161 if (f->type == IOP_T_STRUCT || f->type == IOP_T_UNION) {
162 RETHROW(iopdso_fix_struct_ref(dso, &f->u1.st_desc, pkg, err));
163 }
164 }
165 }
166 for (const iop_iface_t *const *it = pkg->ifaces; *it; it++) {
167 for (int i = 0; i < (*it)->funs_len; i++) {
168 iop_rpc_t *rpc = (iop_rpc_t *)&(*it)->funs[i];
169
170 RETHROW(iopdso_fix_struct_ref(dso, &rpc->args, pkg, err));
171 RETHROW(iopdso_fix_struct_ref(dso, &rpc->result, pkg, err));
172 RETHROW(iopdso_fix_struct_ref(dso, &rpc->exn, pkg, err));
173 }
174 }
175 return 0;
176}

178static int iopdso_register_pkg(iop_dso_t *dso, iop_pkg_t const *pkg,
179 iop_env_t *env, sb_t *err)
180{
181 if (qm_add(iop_pkg, &dso->pkg_h, &pkg->name, pkg) < 0) {
182 return 0;
183 }
184 if (dso->use_external_packages) {
185 e_trace(1, "fixup package `%*pM` (%p)", LSTR_FMT_ARG(pkg->name), pkg);
186 RETHROW(iopdso_fix_pkg(dso, pkg, err));
187 }
188 RETHROW(iop_register_packages_env(&pkg, 1, dso, env, IOP_REGPKG_FROM_DSO,
189 err));
190 for (const iop_enum_t *const *it = pkg->enums; *it; it++) {
191 qm_add(iop_enum, &dso->enum_h, &(*it)->fullname, *it);
192 }
193 for (const iop_struct_t *const *it = pkg->structs; *it; it++) {
194 iopdso_register_struct(dso, *it);
195 }
196 for (const iop_iface_t *const *it = pkg->ifaces; *it; it++) {
197 qm_add(iop_iface, &dso->iface_h, &(*it)->fullname, *it);
198 for (int i = 0; i < (*it)->funs_len; i++) {
199 const iop_rpc_t *rpc = &(*it)->funs[i];
200
201 iopdso_register_struct(dso, rpc->args);
202 iopdso_register_struct(dso, rpc->result);
203 iopdso_register_struct(dso, rpc->exn);
204 }
205 }
206 for (const iop_mod_t *const *it = pkg->mods; *it; it++) {
207 qm_add(iop_mod, &dso->mod_h, &(*it)->fullname, *it);
208 }
209 for (const iop_pkg_t *const *it = pkg->deps; *it; it++) {
210 if (dso->use_external_packages && iop_get_pkg_env((*it)->name, env)) {
211 continue;
212 }
213 RETHROW(iopdso_register_pkg(dso, *it, env, err));
214 }
215 return 0;
216}

234static void iop_dso_unload(iop_dso_t *dso)
235{
236 SB_1k(err);
237
238 e_trace(1, "close dso %p (%*pM)", dso, LSTR_FMT_ARG(dso->path));
239
240 /* Delete references of this DSO in depends_on. */
241 qh_for_each_pos(ptr, pos, &dso->depends_on) {
242 iop_dso_t *depends_on = dso->depends_on.keys[pos];
243
244 qh_del_key(ptr, &depends_on->needed_by, dso);
245 }
246
247 /* Unregister needed_by (otherwise they'll have orphan classes when
248 * unregistering this one). */
249 qh_for_each_pos(ptr, pos, &dso->needed_by) {
250 iop_dso_t *needed_by = dso->needed_by.keys[pos];
251
252 iop_dso_unregister(needed_by);
253 }
254
255 /* Unregister DSO. */
256 iop_dso_unregister(dso);
257
258 /* Reload needed_by. */
259 qh_for_each_pos(ptr, pos, &dso->needed_by) {
260 iop_dso_t *needed_by = dso->needed_by.keys[pos];
261
262 if (iop_dso_reopen(needed_by, &err) < 0) {
263 e_panic("IOP DSO: unable to reload plugin `%*pM` when unloading "
264 "plugin `%*pM`: %*pM",
265 LSTR_FMT_ARG(needed_by->path), LSTR_FMT_ARG(dso->path),
266 SB_FMT_ARG(&err));
267 }
268 }
269}

297iop_dso_t *iop_dso_open(const char *path, Lmid_t lmid, sb_t *err)
298{
299 int flags = RTLD_LAZY | RTLD_DEEPBIND;
300 void *handle;
301 iop_dso_t *dso;
302
303 if (lmid == LM_ID_BASE) {
304 flags |= RTLD_GLOBAL;
305 }
306
307 handle = dlmopen(lmid, path, flags);
308 if (handle == NULL) {
309 sb_setf(err, "unable to dlopen `%s`: %s", path, dlerror());
310 return NULL;
311 }
312
313 dso = iop_dso_load_handle(handle, path, lmid, err);
314 if (!dso) {
315 dlclose(handle);
316 return NULL;
317 }
318
319 return dso;
320}

322iop_dso_t *iop_dso_load_handle(void *handle, const char *path,
323 Lmid_t lmid, sb_t *err)
324{
325 iop_dso_t *dso;
326 iop_dso_vt_t *dso_vt;
327 iop_pkg_t **pkgp;
328
329 dso = qm_get_def(iop_dso_by_handle, &_G.dsos_by_handle, handle, NULL);
330 if (dso) {
331 e_trace(1, "reuse dso %p (%*pM)", dso, LSTR_FMT_ARG(dso->path));
332 dlclose(handle);
333 return iop_dso_dup(dso);
334 }
335
336 if (lmid == LM_ID_NEWLM && dlinfo(handle, RTLD_DI_LMID, &lmid) < 0) {
337 sb_setf(err, "unable to get lmid of plugin `%s`: %s", path,
338 dlerror());
339 return NULL;
340 }
341
342 dso_vt = dlsym(handle, "iop_vtable");
343 if (dso_vt == NULL || dso_vt->vt_size == 0) {
344 e_warning("IOP DSO: unable to find valid IOP vtable in plugin "
345 "`%s`, no error management allowed: %s", path, dlerror());
346 } else {
347 dso_vt->iop_set_verr = &iop_set_verr;
348 }
349
350 pkgp = dlsym(handle, "iop_packages");
351 if (pkgp == NULL) {
352 sb_setf(err, "unable to find IOP packages in plugin `%s`: %s",
353 path, dlerror());
354 return NULL;
355 }
356
357 dso = iop_dso_new();
358 dso->path = lstr_dups(path, -1);
359 dso->handle = handle;
360 dso->lmid = lmid;
361 dso->use_external_packages = !!dlsym(handle, "iop_use_external_packages");
362 dso->dont_replace_fix_pkg = !!dlsym(handle, "iop_dont_replace_fix_pkg");
363
364 e_trace(1, "open new dso %p (%*pM)", dso, LSTR_FMT_ARG(dso->path));
365
366 if (iop_dso_register_(dso, err) < 0) {
367 dso->handle = NULL;
368 iop_dso_delete(&dso);
369 return NULL;
370 }
371
372 qm_add(iop_dso_by_handle, &_G.dsos_by_handle, handle, dso);
373
374 return dso;
375}

400static int iop_dso_register_(iop_dso_t *dso, sb_t *err)
401{
402 if (!dso->is_registered) {
403 iop_env_t env;
404 iop_pkg_t **pkgp = dlsym(dso->handle, "iop_packages");
405
406 if (!pkgp) {
407 /* This should not happen because this was checked before. */
408 e_panic("IOP DSO: iop_packages not found when registering DSO");
409 }
410
411 qm_clear(iop_pkg, &dso->pkg_h);
412 iop_env_get(&env);
413 while (*pkgp) {
414 if (iopdso_register_pkg(dso, *pkgp++, &env, err) < 0) {
415 iop_env_wipe(&env);
416 return -1;
417 }
418 }
419 RETHROW(iop_check_registered_classes(&env, err));
420 iop_env_set(&env);
421 dso->is_registered = true;
422 }
423 return 0;
424}

476iop_struct_t const *iop_dso_find_type(iop_dso_t const *dso, lstr_t name)
477{
478 const char *s;
479 int what, pos;
480 lstr_t fname;
481 iop_rpc_t const *rpc;
482
483 pos = qm_find_safe(iop_struct, &dso->struct_h, &name);
484 if (pos >= 0) {
485 return dso->struct_h.values[pos];
486 }
487
488 if (lstr_endswith(name, LSTR("Args"))) {
489 name.len -= strlen("Args");
490 what = 0;
491 } else
492 if (lstr_endswith(name, LSTR("Res"))) {
493 name.len -= strlen("Res");
494 what = 1;
495 } else
496 if (lstr_endswith(name, LSTR("Exn"))) {
497 name.len -= strlen("Exn");
498 what = 2;
499 } else {
500 return NULL;
501 }
502
503 /* name is some.path.to.pkg.{Module.iface.rpc|Iface.rpc} */
504 s = RETHROW_P(memrchr(name.s, '.', name.len));
505 fname = LSTR_INIT_V(s + 1, name.s + name.len - s - 1);
506 name.len = s - name.s;
507
508 s = RETHROW_P(memrchr(name.s, '.', name.len));
509 if ('A' <= s[1] && s[1] <= 'Z') {
510 /* name is an Iface */
511 pos = RETHROW_NP(qm_find_safe(iop_iface, &dso->iface_h, &name));
512 rpc = RETHROW_P(find_rpc_in_iface(dso->iface_h.values[pos], fname));
513 } else {
514 /* name is a Module.iface */
515 lstr_t mod = LSTR_INIT(name.s, s - name.s);
516 lstr_t ifn = LSTR_INIT(s + 1, name.s + name.len - (s + 1));
517
518 pos = RETHROW_NP(qm_find_safe(iop_mod, &dso->mod_h, &mod));
519 rpc = RETHROW_P(find_rpc_in_mod(dso->mod_h.values[pos], ifn, fname));
520 }
521 switch (what) {
522 case 0: return rpc->args;
523 case 1: return rpc->result;
524 default: return rpc->exn;
525 }
526}
