32/* An implementation of anchor-based htlists.
33 *
34 * Htlists are basically single-linked lists with head and tail pointers.
35 *
36 * You can add prepend and append elements in O(1), or concatenate such
37 * lists in O(1).
38 *
39 * Htlists point to a sublist of elements and can even overlap.
40 *
41 * Here is an example with 3 htlists and 7 htnodes:
42 *
43 * l3-----------------.
44 * l1-----------|-----. l2-----|-----.
45 * | | | | | |
46 * v v v v v v
47 * x --> x --> x --> x --> x --> x --> x --> NULL
48 *
49 * You can add elements to l1 at the end, it won't break l3, and will
50 * make it "longer". If you insert elements at the beginning of l2,
51 * they will not be shared by l3.
52 *
53 * IOW, for an htlist, the tail pointer may point to an element whose
54 * "next" pointer is _not_ NULL without breaking anything.
55 *
56 * The programmer is responsible for potential side effects when lists are
57 * modified: in the example above, doing htlist_add_tail(l1, l3) will
58 * result in a big mess:
59 *
60 * .---------.
61 * l1-------------------' l2-----|-----.
62 * | | | |
63 * v v v v
64 * x --> x --> x --> x x --> x --> x --> NULL
65 * ^ |
66 * `-----'
67 *
68 * Note that an empty htlist looks like this:
69 *
70 * l --> NULL
71 * |
72 * v
73 * ?? (may not be NULL, should not be dereferenced)
74 *
75 * For now, no function is provided to remove elements inside an
76 * htlist, only pop at the start. Such an operation would be
77 * inefficient and rather ill defined anyway.
78 */

98static inline void htlist_add(htlist_t * nonnull l, htnode_t * nonnull n)
99{
100 n->next = l->head;
101 l->head = n;
102 if (htlist_is_empty(l)) {
103 l->tail = n;
104 }
105}

107static inline void htlist_add_tail(htlist_t * nonnull l, htnode_t * nonnull n)
108{
109 if (htlist_is_empty(l)) {
110 htlist_add(l, n);
111 } else {
112 n->next = l->tail->next;
113 l->tail->next = n;
114 l->tail = n;
115 }
116}

118/* Adding a node after another one. If prev is NULL, the new_node will be
119 * added to the head */
120static inline void
121htlist_add_after(htlist_t * nonnull l, htnode_t * nullable prev,
122 htnode_t * nonnull new_node)
123{
124 if (!prev) {
125 return htlist_add(l, new_node);
126 }
127
128 new_node->next = prev->next;
129 prev->next = new_node;
130
131 if (l->tail == prev) {
132 l->tail = new_node;
133 }
134}

136static inline htnode_t * nonnull htlist_pop(htlist_t * nonnull l)
137{
138 htnode_t *res = l->head;
139
140 assert (!htlist_is_empty(l));
141
142 l->head = res->next;
143 if (l->tail == res) {
144 l->tail = NULL;
145 }
146 return res;
147}

149static inline void htlist_splice(htlist_t * nonnull dst,
150 htlist_t * nonnull src)
151{
152 if (!htlist_is_empty(src)) {
153 src->tail->next = dst->head;
154 dst->head = src->head;
155
156 if (htlist_is_empty(dst)) {
157 dst->tail = src->tail;
158 }
159 }
160}

162static inline void htlist_move(htlist_t * nonnull dst,
163 htlist_t * nonnull src)
164{
165 htlist_init(dst);
166 htlist_splice(dst, src);
167 htlist_init(src);
168}

170static inline void htlist_splice_tail(htlist_t * nonnull dst,
171 htlist_t * nonnull src)
172{
173 if (!htlist_is_empty(src)) {
174 src->tail->next = dst->tail->next;
175 dst->tail->next = src->head;
176 /* XXX: src->tail points to an actual element because src is
177 * not empty.
178 */
179 dst->tail = src->tail;
180 }
181}
