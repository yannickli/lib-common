33/** \defgroup log Logging facility.
34 * \ingroup log
35 * \brief Logging Facility.
36 *
37 * \{
38 *
39 * \section log_intro Loggers
40 *
41 * This module provides a hierarchical logging facility. This is done through
42 * a tree of \ref logger_t structures. Each logger comes with it's own logging
43 * level (no log with a level greater to that one will be printed), that can
44 * be inherited from its parent.
45 *
46 * Each \ref logger_t also comes with a name, that name can be printed in the
47 * logs, allowing more contextual informations to be provided.
48 *
49 * When a log is rejected it costs near to 0 since only an integer comparison
50 * is performed. This let you put logging even in critical paths with low
51 * impact on global performances. Arguments of the logging functions are
52 * evaluated only in case the log is emitted. As a consequence, you must take
53 * care only passing arguments that have no side effects.
54 *
55 *
56 * Every logger is initialized with a default logging level. That level can be
57 * either an inheritance mark, in which case the logger will always inherits
58 * its level from its parent (unless explicitely set for that particular
59 * logger).
60 *
61 * You can manage the logging of every single logger independently from each
62 * other or choose to update the level of the whole tree of loggers at once.
63 * This gives you (and the user) a fine-grained control on what should be
64 * logged.
65 *
66 *
67 * \section log_hander Handlers
68 *
69 * When a log is accepted, it is pushed to a handler that performs the output.
70 * The handler can be changed by a call to \ref log_set_handler. A default
71 * handler that prints on stderr is provided.
72 *
73 *
74 * \section log_scope Logging scope
75 *
76 * The logging facility provides two main APIs:
77 * - the classic API that formats a whole line.
78 * - a scoped API that let you build one or more line chunk-by-chunk.
79 *
80 * The scoped API works by delimiting a logging scope in which you will build
81 * the logs using calls to \ref logger_cont ("continue logging"). The scope
82 * itself is delimited by a starting call to \ref logger_*_start and an
83 * ending call to \ref logger_end. However, a more convenient solution
84 * consists in using \ref logger_*_scope that automatically start and stop the
85 * scope according to the C syntaxic scope.
86 *
87 * <pre>
88 * {
89 * logger_trace_scope(&logger, 1);
90 *
91 * logger_cont("start ");
92 * logger_cont("writing %d ", 2);
93 * logger_cont("logs");
94 * }
95 * </pre>
96 */

236static ALWAYS_INLINE
237int logger_get_level(logger_t * nonnull logger)
238{
239 if (atomic_load_explicit(&logger->conf_gen, memory_order_acquire)
240 != log_conf_gen_g)
241 {
242 __logger_refresh(logger);
243 }
244
245 /* We always want to catch fatal/panic issues which are associated to the
246 * LOG_CRIT level.
247 */
248 return MAX(logger->level, LOG_CRIT);
249}

251static ALWAYS_INLINE
252bool logger_has_level(logger_t * nonnull logger, int level)
253{
254 return logger_get_level(logger) >= level;
255}

293#define logger_is_traced(Logger, Level) ({ \
294 static int8_t __logger_traced; \
295 static const logger_t *__last_logger = NULL; \
296 const logger_t *__i_clogger = (Logger); \
297 logger_t *__i_logger = (logger_t *)__i_clogger; \
298 const int __logger_i_level = (Level); \
299 bool __logger_h_level = logger_has_level(__i_logger, \
300 LOG_TRACE + __logger_i_level); \
301 \
302 if (!__logger_h_level) { \
303 if (unlikely(!__builtin_constant_p(Level) \
304 || __i_clogger != __last_logger)) \
305 { \
306 __logger_traced = __logger_is_traced(__i_logger, \
307 __logger_i_level,__FILE__, __func__, \
308 __i_logger->full_name.s); \
309 __last_logger = __i_clogger; \
310 } \
311 } \
312 __logger_h_level || __logger_traced > 0; \
313 })

315#define __LOGGER_HAS_LEVEL(__logger, __level) \
316 ((__level >= LOG_TRACE) ? \
317 logger_is_traced(__logger, __level - LOG_TRACE) : \
318 logger_has_level(__logger, __level))

326 logger_has_level(__i_logger, (Level) + LOG_TRACE); \

411#define __LOGGER_LOG(Logger, Level, Mark, Fmt, ...) ({ \
412 const logger_t *__clogger = (Logger); \
413 logger_t *__logger = (logger_t *)__clogger; \
414 int __logger_res; \
415 \
416 Mark; \
417 /* XXX We need two different cases: the first one (Level non-const) \
418 * to protect Level against multi-evaluation; the second one to \
419 * preserve the constness of Level until we reach logger_is_traced() \
420 * and allow its __builtin_constant_p to work even in O0. \
421 */ \
422 if (unlikely(!__builtin_constant_p(Level))) { \
423 const int __logger_level = (Level); \
424 \
425 if (__LOGGER_HAS_LEVEL(__logger, __logger_level)) { \
426 __logger_log(__logger, __logger_level, NULL, -1, __FILE__, \
427 __func__, __LINE__, Fmt, ##__VA_ARGS__); \
428 } \
429 __logger_res = __logger_level <= LOG_WARNING ? -1 : 0; \
430 } else { \
431 if (__LOGGER_HAS_LEVEL(__logger, (Level))) { \
432 __logger_log(__logger, (Level), NULL, -1, __FILE__, \
433 __func__, __LINE__, Fmt, ##__VA_ARGS__); \
434 } \
435 __logger_res = (Level) <= LOG_WARNING ? -1 : 0; \
436 } \
437 __logger_res; \
438 })
