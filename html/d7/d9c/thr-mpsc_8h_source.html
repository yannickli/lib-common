28/*
29 * This file provides an implementation of lock-free intrusive MPSC queue.
30 *
31 * MPSC stands for Multiple-Producer, Single-Consumer.
32 *
33 * It means that it's possible to concurently add elements to these queues
34 * without taking any locks, but that only one single consumer is allowed at
35 * any time.
36 */
37
38typedef struct mpsc_node_t mpsc_node_t;
39typedef struct mpsc_queue_t mpsc_queue_t;
40typedef _Atomic(mpsc_node_t *) atomic_mpsc_node_t;
41
42/** \brief node to embed into structures to be put in an mpsc queue.
43 */
44struct mpsc_node_t {
45 atomic_mpsc_node_t next;
46};
47
48/** \brief head of an mpsc queue.
49 */
50struct mpsc_queue_t {
51 mpsc_node_t head;
52 atomic_mpsc_node_t tail;
53};

63 p_clear(q, 1);
64 atomic_init(&q->tail, &q->head);

68/** \brief tells whether an mpsc queue looks empty or not.
69 *
70 * This should be only used from the only mpsc consumer thread.
71 *
72 * The function may return false positives but no false negatives: the
73 * consumer thread is the sole consumer, hence if it sees a non empty queue
74 * the queue can't be freed without it knowing about it.
75 *
76 * If the function returns true though, some other thread may enqueue
77 * something at the same time and the function will return a false positive,
78 * hence the name "the queue looks empty but maybe it isn't".
79 *
80 * \param[in] q the queue
81 * \returns true if the queue looks empty, false else.
82 */
83static inline bool mpsc_queue_looks_empty(mpsc_queue_t *q)
84{
85 return atomic_load_explicit(&q->head.next, memory_order_relaxed) == NULL;
86}

88/** \brief enqueue a task in the queue.
89 *
90 * This function has a vital role as it does empty -> non empty transition
91 * detections which make it suitable to use like:
92 *
93 * <code>
94 * if (mpsc_queue_push(some_q, some_node))
95 * schedule_processing_of(some_q);
96 * </code>
97 *
98 * \param[in] q the queue
99 * \param[in] n the node to push
100 * \returns true if the queue was empty before the push, false else.
101 */
102static inline bool mpsc_queue_push(mpsc_queue_t *q, mpsc_node_t *n)
103{
104 mpsc_node_t *prev;
105
106 atomic_store_explicit(&n->next, NULL, memory_order_release);
107 prev = atomic_exchange_explicit(&q->tail, n, memory_order_seq_cst);
108 atomic_store_explicit(&prev->next, n, memory_order_seq_cst);
109 return prev == &q->head;
110}

112/** \brief type used to enumerate through a queue during a drain.
113 */
114typedef struct mpsc_it_t {
115 mpsc_queue_t *q;
116 mpsc_node_t *h;
117} mpsc_it_t;
118
119/** \brief initiates the iterator to start a drain.
120 *
121 * \warning
122 * it's forbidden not to drain fully after this function has been called.
123 *
124 * a typical drain looks like that:
125 *
126 * <code>
127 * static void doit(mpsc_node_t *node, data_t data)
128 * {
129 * process_node(node);
130 * freenode(node);
131 * }
132 *
133 * {
134 * mpsc_it_t it;
135 *
136 * mpsc_queue_drain_start(&it, &path->to.your_queue);
137 * do {
138 * mpsc_node_t *n = mpsc_queue_drain_fast(&it, doit,
139 * (data_t){ .ptr = NULL });
140 * process_node(n);
141 * // XXX do NOT freenode(n) here mpsc_queue_drain_end will use it
142 * } while (!mpsc_queue_drain_end(&it, freenode));
143 * }
144 * </code>
145 *
146 * It is important to note that the queue may go through the draining loop
147 * multiple times if jobs are enqueued while the queue is drained.
148 *
149 * \param[out] it the iterator to initialize
150 * \param[in] q the queue to drain
151 */
152static inline void mpsc_queue_drain_start(mpsc_it_t *it, mpsc_queue_t *q)
153{
154 it->q = q;
155 it->h = atomic_load_explicit(&q->head.next, memory_order_acquire);
156 atomic_store_explicit(&q->head.next, NULL, memory_order_relaxed);
157 /* breaks if someone called mpsc_queue_drain_start with the queue empty */
158 assert (it->h);
159}

161/** \brief fast path of the drain.
162 *
163 * \param[in] it the iterator to use
164 * \param[in] doit
165 * a function or macro that takes one mpsc_node_t as an argument: the node
166 * to process.
167 * \param[in] data additionnal data to pass to \p doit.
168 *
169 * \returns
170 * a non NULL mpsc_node_t that may be the last one in the queue for
171 * processing purposes. mpsc_queue_drain_fast doesn't process this node, its
172 * up to the caller to do it.
173 * This node should NOT be freed, as mpsc_queue_drain_end will need it.
174 */
175static inline
176mpsc_node_t *mpsc_queue_drain_fast(mpsc_it_t *it,
177 void (*doit)(mpsc_node_t *, data_t data),
178 data_t data)
179{
180 mpsc_node_t *h = it->h;
181 mpsc_node_t *n;
182
183 while (likely(n = atomic_load_explicit(&h->next, memory_order_acquire))) {
184 (*doit)(h, data);
185 h = n;
186 }
187 it->h = h;
188 return h;
189}

191/** \brief implementation of mpsc_queue_drain_end.
192 *
193 * Do not use directly unless you want to override relax and know what you are
194 * doing.
195 */
196static inline
197bool __mpsc_queue_drain_end(mpsc_it_t *it, void (*freenode)(mpsc_node_t *),
198 void (*relax)(void))
199{
200 mpsc_queue_t *q = it->q;
201 mpsc_node_t *h = it->h;
202 mpsc_node_t *hq = h;
203
204 if (h == atomic_load_explicit(&q->tail, memory_order_acquire)
205 && atomic_compare_exchange_strong(&q->tail, &hq, &q->head))
206 {
207 it->h = NULL;
208 } else {
209 while (!(it->h = atomic_load_explicit(&h->next, memory_order_acquire))) {
210 if (relax) {
211 (*relax)();
212 } else {
213 cpu_relax();
214 }
215 }
216 }
217 if (freenode && h) {
218 (*freenode)(h);
219 }
220 return it->h == NULL;
221}

223/** \brief test for the drain completion.
224 *
225 * \param[in] it the iterator to use
226 * \param[in] freenode
227 * a function or macro that will be called on the last node (actually the
228 * one #mpsc_queue_drain_fast returned) when the test for the mpsc queue
229 * emptyness has been done. If you don't need to free your nodes, use
230 * IGNORE.
231 *
232 * \returns
233 * true if the queue is really empty, in which case the drain is done, false
234 * else in which case the caller has to restart the drain.
235 */
236#define mpsc_queue_drain_end(it, freenode) \
237 __mpsc_queue_drain_end(it, freenode, NULL)

242static inline __cold
243bool mpsc_queue_pop_slow(mpsc_queue_t *q, mpsc_node_t *head, bool block)
244{
245 mpsc_node_t *tail = atomic_load_explicit(&q->tail, memory_order_relaxed);
246 mpsc_node_t *next;
247
248 if (head == tail) {
249 atomic_store_explicit(&q->head.next, NULL, memory_order_relaxed);
250 if (atomic_compare_exchange_strong(&q->tail, &tail, &q->head)) {
251 return true;
252 }
253 atomic_store_explicit(&q->head.next, head, memory_order_relaxed);
254 }
255
256 next = atomic_load_explicit(&head->next, memory_order_relaxed);
257 if (next == NULL) {
258 if (!block) {
259 return false;
260 }
261 while ((next = atomic_load_explicit(&head->next, memory_order_relaxed)) == NULL) {
262 cpu_relax();
263 }
264 }
265 atomic_store_explicit(&q->head.next, next, memory_order_relaxed);
266 return true;
267}

269/** \brief pop one entry from the mpsc queue.
270 *
271 * \param[in] q The queue.
272 * \param[in] block If the pop emptied the queue while an insertion is in
273 * progress, wait for this insertion to be finished.
274 *
275 * \returns
276 * The node extracted from the queue or NULL if the queue is empty.
277 *
278 * \warning
279 * This API is unsafe and should be used only in specific use-cases where
280 * the enumeration of a queue must support imbrication (that is a call that
281 * enumerate the queue within an enumeration).
282 */
283static inline mpsc_node_t *mpsc_queue_pop(mpsc_queue_t *q, bool block)
284{
285 mpsc_node_t *head = atomic_load_explicit(&q->head.next, memory_order_relaxed);
286 mpsc_node_t *next;
287
288 if (head == NULL)
289 return NULL;
290
291 if (likely(next = atomic_load_explicit(&head->next, memory_order_relaxed))) {
292 atomic_store_explicit(&q->head.next, next, memory_order_relaxed);
293 return head;
294 }
295 if (block) {
296 mpsc_queue_pop_slow(q, head, block);
297 return head;
298 }
299 return mpsc_queue_pop_slow(q, head, block) ? head : NULL;
300}
