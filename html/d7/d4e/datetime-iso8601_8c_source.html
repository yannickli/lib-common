21static int time_parse_timezone(pstream_t *ps, int *tz_h, int *tz_m)
22{
23 *tz_m = 0;
24 *tz_h = 0;
25
26 if (ps_strcaseequal(ps, "ut") || ps_strcaseequal(ps, "gmt")
27 || ps_strcaseequal(ps, "z"))
28 {
29 return 0;
30 } else
31 if (ps_strcaseequal(ps, "edt")) {
32 *tz_h = -4;
33 return 0;
34 } else
35 if (ps_strcaseequal(ps, "est") || ps_strcaseequal(ps, "cdt")) {
36 *tz_h = -5;
37 return 0;
38 } else
39 if (ps_strcaseequal(ps, "cst") || ps_strcaseequal(ps, "mdt")) {
40 *tz_h = -6;
41 return 0;
42 } else
43 if (ps_strcaseequal(ps, "mst") || ps_strcaseequal(ps, "pdt")) {
44 *tz_h = -7;
45 return 0;
46 } else
47 if (ps_strcaseequal(ps, "pst")) {
48 *tz_h = -8;
49 return 0;
50 } else
51 if (ps_strcaseequal(ps, "a")) {
52 *tz_h = -1;
53 return 0;
54 } else
55 if (ps_strcaseequal(ps, "m")) {
56 *tz_h = -12;
57 return 0;
58 } else
59 if (ps_strcaseequal(ps, "n")) {
60 *tz_h = +1;
61 return 0;
62 } else
63 if (ps_strcaseequal(ps, "y")) {
64 *tz_h = +12;
65 return 0;
66 } else {
67 int sgn = ps_getc(ps);
68
69 PS_WANT(sgn == '+' || sgn == '-');
70
71 errno = 0;
72 PS_WANT(isdigit(ps->b[0]));
73 if (ps_len(ps) == 5) {
74 *tz_h = ps_geti(ps);
75 PS_WANT(errno == 0 && ps_len(ps) == 3);
76 PS_CHECK(ps_skipc(ps, ':'));
77 PS_WANT(isdigit(ps->b[0]));
78 *tz_m = ps_geti(ps);
79 PS_WANT(errno == 0 && ps_done(ps));
80 } else
81 if (ps_len(ps) == 4) {
82 uint32_t raw = ps_geti(ps);
83
84 PS_WANT(errno == 0 && ps_done(ps));
85 *tz_h = raw / 100;
86 *tz_m = raw % 100;
87 } else
88 if (ps_len(ps) == 2) {
89 *tz_h = ps_geti(ps);
90 PS_WANT(errno == 0 && ps_done(ps));
91 } else {
92 return -1;
93 }
94
95 if (sgn == '-') {
96 *tz_h = -1 * *tz_h;
97 *tz_m = -1 * *tz_m;
98 }
99 return 0;
100 }
101 return -1;
102}

104static int time_parse_iso8601_tok(pstream_t *ps, int *nb, int *type)
105{
106 *nb = ps_geti(ps);
107 if (!ps_has(ps, 1)) {
108 e_debug("invalid date tok");

110 }

112 return 0;

115int time_parse_iso8601_flags(pstream_t *ps, time_t *res, unsigned flags)
116{
117 struct tm t;
118 bool local = false, syslog_format = false;
119 int tz_h, tz_m;
120 char c;
121
122 if (!ps_has(ps, 1)) {
123 e_debug("invalid date: empty");
124 return -1;
125 }
126
127 if (*ps->s == 'P') {
128 /* Relative date */
129 int nb, tok;
130 time_t now = lp_getsec();
131
132 localtime_r(&now, &t);
133
134 __ps_skip(ps, 1);
135 if (ps_done(ps))
136 goto end_rel;
137 RETHROW(time_parse_iso8601_tok(ps, &nb, &tok));
138 if (tok == 'Y') {
139 t.tm_year += nb;
140 if (ps_done(ps))
141 goto end_rel;
142 RETHROW(time_parse_iso8601_tok(ps, &nb, &tok));
143 }
144 if (tok == 'M') {
145 t.tm_mon += nb;
146 if (ps_done(ps))
147 goto end_rel;
148 RETHROW(time_parse_iso8601_tok(ps, &nb, &tok));
149 }
150 if (tok == 'D') {
151 t.tm_mday += nb;
152 if (ps_done(ps))
153 goto end_rel;
154 RETHROW(time_parse_iso8601_tok(ps, &nb, &tok));
155 }
156 if (tok != 'T') {
157 e_debug("missing 'T' time mark");
158 return -1;
159 }
160 RETHROW(time_parse_iso8601_tok(ps, &nb, &tok));
161 if (tok == 'H') {
162 t.tm_hour += nb;
163 if (ps_done(ps))
164 goto end_rel;
165 RETHROW(time_parse_iso8601_tok(ps, &nb, &tok));
166 }
167 if (tok == 'M') {
168 t.tm_min += nb;
169 if (ps_done(ps))
170 goto end_rel;
171 RETHROW(time_parse_iso8601_tok(ps, &nb, &tok));
172 }
173 if (tok == 'S') {
174 t.tm_sec += nb;
175 }
176
177 end_rel:
178 t.tm_isdst = -1;
179 *res = mktime(&t);
180 return 0;
181 }
182
183 p_clear(&t, 1);
184
185 t.tm_year = ps_geti(ps) - 1900;
186 if (t.tm_year <= 0 || t.tm_year > 200) {
187 e_debug("invalid year in date");
188 return -1;
189 }
190 if (ps_getc(ps) != '-') {
191 e_debug("invalid date format, missing '-' after year");
192 return -1;
193 }
194
195 t.tm_mon = ps_geti(ps) - 1;
196 if (t.tm_mon < 0 || t.tm_mon > 11) {
197 e_debug("invalid month in date");
198 return -1;
199 }
200 if (ps_getc(ps) != '-') {
201 e_debug("invalid date format, missing '-' after month");
202 return -1;
203 }
204
205 t.tm_mday = ps_geti(ps);
206 if (!is_mday_valid(t.tm_mday, t.tm_mon, t.tm_year + 1900)) {
207 e_debug("invalid day in date");
208 return -1;
209 }
210
211 if (ps_done(ps)) {
212 if ((flags & ISO8601_RESTRICT_DAY_DATE_FORMAT)
213 || (flags & ISO8601_ALLOW_DAY_DATE_FORMAT))
214 {
215 t.tm_isdst = -1;
216 *res = mktime(&t);
217 return 0;
218 } else {
219 e_debug("day date format `YYYY-MM-DD` is not allowed");
220 return -1;
221 }
222 }
223
224 if (flags & ISO8601_RESTRICT_DAY_DATE_FORMAT) {
225 e_debug("input is not a day");
226 return -1;
227 }
228
229 c = ps_getc(ps);
230 if (toupper(c) != 'T') {
231 if ((flags & ISO8601_ALLOW_SYSLOG_FORMAT) && c == ' ') {
232 syslog_format = true;
233 } else {
234 e_debug("invalid date format, missing 'T' after day");
235 return -1;
236 }
237 }
238
239 t.tm_hour = ps_geti(ps);
240 switch (ps_getc(ps)) {
241 case 'L':
242 case 'l':
243 local = true;
244 break;
245
246 case ':':
247 break;
248
249 default:
250 e_debug("invalid date format, invalid char after hour");
251 return -1;
252 }
253
254 t.tm_min = ps_geti(ps);
255 if (ps_getc(ps) != ':') {
256 e_debug("invalid date format, missing ':' after minutes");
257 return -1;
258 }
259
260 t.tm_sec = ps_geti(ps);
261 if (ps_startswith(ps, ".", 1)) { /* Ignore it */
262 ps_getc(ps);
263 ps_geti(ps);
264 }
265
266 if (local) {
267 t.tm_isdst = -1;
268 *res = mktime(&t);
269 return 0;
270 }
271
272 if (ps_done(ps)) {
273 t.tm_isdst = -1;
274 *res = mktime(&t);
275 return 0;
276 }
277
278 if (syslog_format) {
279 RETHROW(ps_skipc(ps, ' '));
280 }
281 RETHROW(time_parse_timezone(ps, &tz_h, &tz_m));
282
283 /* subtract the offset from the local time to get UTC time */
284 t.tm_hour -= tz_h;
285 t.tm_min -= tz_m;
286 t.tm_isdst = 0;
287 *res = timegm(&t);
288 return 0;
289}

291int time_parse(pstream_t *ps, time_t *d)
292{
293 struct tm date;
294 int len = ps_len(ps);
295
296 p_clear(&date, 1);
297 date.tm_isdst = -1;
298
299#define PARSE_FORMAT(format, skip_space) \
300 do { \
301 t_scope; \
302 char *end = strptime(t_dupz(ps->s, len), format, &date); \
303 \
304 if (end == NULL) { \
305 return -1; \
306 } else \
307 if (*end != '\0') { \
308 pstream_t ts = ps_initstr(end); \
309 int tz_h, tz_m; \
310 \
311 if (skip_space) { \
312 ps_ltrim(&ts); \
313 } \
314 RETHROW(time_parse_timezone(&ts, &tz_h, &tz_m)); \
315 date.tm_hour -= tz_h; \
316 date.tm_min -= tz_m; \
317 *d = timegm(&date); \
318 return 0; \
319 } \
320 *d = mktime(&date); \
321 return 0; \
322 } while (0);
323
324 if (len > 4 && (ps->s[0] == 'P' || ps->s[4] == '-')) {
325 unsigned flags;
326
327 /* ISO8601: YYYY-MM-DD[Thh:mm:ss] */
328 flags = ISO8601_ALLOW_SYSLOG_FORMAT | ISO8601_ALLOW_DAY_DATE_FORMAT;
329 return time_parse_iso8601_flags(ps, d, flags);
330 } else
331 if (len > 3 && (ps->s[1] == ' ' || ps->s[2] == ' ')) {
332 /* RFC822: D month YYYY hh:mm:ss tz */
333 PARSE_FORMAT("%d%n%h%n%Y%n%T", true);
334 } else
335 if (len > 4 && ps->s[3] == ',') {
336 /* RFC822: Day, D month YYYY hh:mm:ss tz */
337 PARSE_FORMAT("%a,%n%d%n%h%n%Y%n%T", true);
338 } else {
339 /* Unix timestamp */
340 errno = 0;
341 *d = ps_getlli(ps);
342 if (errno || !ps_done(ps)) {
343 return -1;
344 }
345 /* Support millisecond timestamps (convert them to seconds). */
346 if (unlikely(*d >= 30000000000)) {
347 *d /= 1000;
348 }
349 return 0;
350 }
351#undef PARSE_FORMAT
352}
