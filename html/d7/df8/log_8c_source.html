114void log_spin_lock(void)
115{
116 if (unlikely(log_thr_g.in_spinlock)) {
117 /* Deadlock detected. This can happen in tricky situations, for
118 * example if malloc fails we calling logger_panic, which calls
119 * logger_refresh, which takes a lock and tries to allocate memory.
120 * If it fails again, the same stack is called again and the deadlock
121 * occurs.
122 * Catch it and do the best we can to avoid blocking a thread: commit
123 * suicide.
124 */
125 syslog(LOG_USER | LOG_CRIT, "deadlock detected in log library");
126 printf("deadlock detected in log library\n");
127 abort();
128 }
129 log_thr_g.in_spinlock = true;
130 spin_lock(&_G.update_lock);
131}
132
133void log_spin_unlock(void)
134{
135 spin_unlock(&_G.update_lock);
136 log_thr_g.in_spinlock = false;
137}

139lstr_t t_logger_sanitize_name(const lstr_t name)
140{
141 t_SB_1k(normalized);
142 uint32_t used_characters;
143
144 sb_normalize_utf8(&normalized, name.s, name.len, false);
145 used_characters = 0;
146
147 for (int i = 0; i < normalized.len; i++) {
148 int c = normalized.data[i];
149
150 if (isprint(c) && c != '!' && c != '/' && !isspace(c)) {
151 normalized.data[used_characters] = c;
152 used_characters++;
153 }
154
155 }
156
157 sb_shrink(&normalized, normalized.len - used_characters);
158
159 if (!normalized.len || !isalnum(normalized.data[0])) {
160 return LSTR_NULL_V;
161 }
162
163 return LSTR_SB_V(&normalized);
164}

227static void logger_compute_fullname(logger_t *logger)
228{
229 /* The name of a logger must be a non-empty printable string
230 * without any '/' or '!'
231 */
232 assert (memchr(logger->name.s, '/', logger->name.len) == NULL);
233 assert (memchr(logger->name.s, '!', logger->name.len) == NULL);
234 assert (logger->name.len);
235 for (int i = 0; i < logger->name.len; i++) {
236 assert (isprint((unsigned char)logger->name.s[i]));
237 }
238
239 if (logger->parent->full_name.len) {
240 lstr_t name = logger->name;
241 lstr_t full_name;
242
243 full_name = lstr_fmt("%*pM/%*pM",
244 LSTR_FMT_ARG(logger->parent->full_name),
245 LSTR_FMT_ARG(logger->name));
246
247 logger->full_name = full_name;
248 logger->name = LSTR_INIT_V(full_name.s + full_name.len - name.len,
249 name.len);
250 lstr_wipe(&name);
251 } else
252 if (logger->name.len) {
253 logger->full_name = lstr_dupc(logger->name);
254 }
255}

257void __logger_do_refresh(logger_t *logger)
258{
259 if (atomic_load_explicit(&logger->conf_gen, memory_order_acquire)
260 == log_conf_gen_g)
261 {
262 return;
263 }
264
265 logger->level_flags &= ~LOG_FORCED;
266
267 if (logger->parent == NULL && logger != &_G.root_logger) {
268 logger->parent = &_G.root_logger;
269 }
270 if (logger->parent) {
271 __logger_do_refresh(logger->parent);
272 }
273
274 if (!logger->full_name.s) {
275 int pos = 0;
276
277 logger_compute_fullname(logger);
278
279 assert (logger->level >= LOG_UNDEFINED);
280 assert (logger->default_level >= LOG_INHERITS);
281 assert (logger->defined_level >= LOG_UNDEFINED);
282
283 dlist_for_each_entry(logger_t, sibling, &logger->parent->children,
284 siblings)
285 {
286 assert (!lstr_equal(sibling->name, logger->name));
287 }
288 dlist_add(&logger->parent->children, &logger->siblings);
289 dlist_init(&logger->children);
290
291 pos = qm_del_key(level, &_G.pending_levels, &logger->full_name);
292 if (pos >= 0) {
293 logger->defined_level = _G.pending_levels.values[pos].level;
294 logger->level_flags = _G.pending_levels.values[pos].flags;
295 lstr_wipe(&_G.pending_levels.keys[pos]);
296 }
297 }
298
299 logger->level = logger->default_level;
300
301 if (logger->defined_level >= 0) {
302 logger->level = logger->defined_level;
303 } else
304 if (logger->parent) {
305 if (logger->parent->level_flags & (LOG_FORCED | LOG_RECURSIVE)) {
306 logger->level = logger->parent->level;
307 logger->level_flags |= LOG_FORCED;
308 } else
309 if (logger->level == LOG_INHERITS) {
310 logger->level = logger->parent->level;
311 }
312 }
313
314 assert (logger->level >= 0);
315 atomic_store_explicit(&logger->conf_gen, log_conf_gen_g,
316 memory_order_release);
317}

329 log_spin_unlock();

337logger_t *logger_get_by_name(lstr_t name)
338{
339 pstream_t ps = ps_initlstr(&name);
340 logger_t *logger = &log_g.root_logger;
341
342 while (!ps_done(&ps)) {
343 logger_t *next = NULL;
344 pstream_t n;
345
346 if (ps_get_ps_chr_and_skip(&ps, '/', &n) < 0) {
347 n = ps;
348 ps = ps_init(NULL, 0);
349 }
350
351 dlist_for_each_entry(logger_t, child, &logger->children, siblings) {
352 if (lstr_equal(child->name, LSTR_PS_V(&n))) {
353 next = child;
354 break;
355 }
356 }
357
358 RETHROW_P(next);
359 logger = next;
360 }
361
362 return logger;
363}

365int logger_set_level(lstr_t name, int level, unsigned flags)
366{
367 logger_t *logger;
368
369 log_spin_lock();
370 logger = logger_get_by_name(name);
371
372 assert (level >= LOG_UNDEFINED);
373 assert ((flags & (LOG_RECURSIVE | LOG_SILENT)) == flags);
374 assert (!(flags & LOG_RECURSIVE) || level >= 0);
375
376 /* -2 == LOG_LEVEL_DEFAULT, which cannot be used here because it is
377 * defined in core.iop, and we are in a source file of libcommon-minimal
378 * that cannot use content from iop files. */
379 if (level == -2) {
380 level = LOG_DEFAULT;
381 }
382
383 if (!logger) {
384 if (level == LOG_UNDEFINED) {
385 int pos = qm_del_key(level, &_G.pending_levels, &name);
386
387 if (pos >= 0) {
388 lstr_wipe(&_G.pending_levels.keys[pos]);
389 }
390 } else {
391 struct level l = { .level = level, .flags = flags };
392 uint32_t pos;
393
394 pos = qm_put(level, &_G.pending_levels, &name, l, QHASH_OVERWRITE);
395 if (!(pos & QHASH_COLLISION)) {
396 _G.pending_levels.keys[pos] = lstr_dup(name);
397 }
398 }
399 log_spin_unlock();
400 return LOG_UNDEFINED;
401 }
402
403 if (level == LOG_UNDEFINED) {
404 logger->level_flags = logger->default_level_flags;
405 } else {
406 logger->level_flags = flags;
407 }
408 SWAP(int, logger->level, level);
409 logger->defined_level = logger->level;
410 log_conf_gen_g += 2;
411
412 log_spin_unlock();
413 return level;
414}

502static __attr_printf__(3, 0)
503void logger_putv(const log_ctx_t *ctx, bool do_log,
504 const char *fmt, va_list va)
505{
506 buffer_instance_t *buff;
507
508 if (ctx->level <= LOG_CRIT) {
509 va_list cpy;
510
511 syslog_is_critical = true;
512 va_copy(cpy, va);
513 logger_vsyslog(ctx->level, fmt, cpy);
514 va_end(cpy);
515 }
516
517 if (!do_log) {
518 return;
519 }
520
521 if (!log_thr_g.nb_buffer_started) {
522 (*_G.handler)(ctx, fmt, va);
523 return;
524 }
525
526 buff = &log_thr_g.vec_buff_stack.tab[log_thr_g.nb_buffer_started - 1];
527
528 if (ctx->level <= buff->buffer_log_level) {
529 int size_fmt;
530 log_buffer_t *log_save;
531 va_list cpy;
532 char *buffer;
533 qv_t(log_buffer) *vec_buffer = &buff->vec_buffer;
534
535 if (buff->use_handler) {
536 va_copy(cpy, va);
537 (*_G.handler)(ctx, fmt, cpy);
538 va_end(cpy);
539 }
540
541 va_copy(cpy, va);
542 size_fmt = vsnprintf(NULL, 0, fmt, cpy) + 1;
543 va_end(cpy);
544
545 free_last_buffer();
546 buffer = mp_new_raw(&log_thr_g.mp_stack.funcs, char, size_fmt);
547 vsnprintf(buffer, size_fmt, fmt, va);
548
549 log_save = qv_growlen(vec_buffer, 1);
550 log_save->ctx = *ctx;
551 log_save->msg = LSTR_INIT_V(buffer, size_fmt - 1);
552 }
553}

710 }
711 level = spec->level;
712 }

1122/** Parse the content of the IS_DEBUG environment variable.
1123 *
1124 * It is composed of a series of blank separated <specs>:
1125 *
1126 * <specs> ::= [<path-pattern>][@<funcname>][+<featurename>][:<level>]
1127 */
1128void log_parse_specs(char *p, qv_t(spec) *out)
1129{
1130 pstream_t ps = ps_initstr(p);
1131 ctype_desc_t ctype;
1132
1133 ps_trim(&ps);
1134
1135 ctype_desc_build(&ctype, "@+:");
1136
1137 while (!ps_done(&ps)) {
1138 log_trace_spec_t spec = {
1139 .path = NULL,
1140 .func = NULL,
1141 .name = NULL,
1142 .level = INT_MAX,
1143 };
1144 pstream_t spec_ps;
1145 int c = '\0';
1146
1147 spec_ps = ps_get_cspan(&ps, &ctype_isspace);
1148 ps_skip_span(&ps, &ctype_isspace);
1149
1150#define GET_ELEM(_dst) \
1151 do { \
1152 pstream_t elem = ps_get_cspan(&spec_ps, &ctype); \
1153 \
1154 if (ps_len(&elem)) { \
1155 _dst = elem.s; \
1156 } \
1157 c = *spec_ps.s; \
1158 *(char *)spec_ps.s = '\0'; \
1159 ps_skip(&spec_ps, 1); \
1160 } while (0)
1161
1162 GET_ELEM(spec.path);
1163 if (c == '@') {
1164 GET_ELEM(spec.func);
1165 }
1166 if (c == '+') {
1167 GET_ELEM(spec.name);
1168 }

1171
1172 GET_ELEM(level);
1173 if (level) {
1174 spec.level = LOG_TRACE + atoi(level);
1175 }
1176 }

1180 }
